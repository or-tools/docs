14// Tool to run MathOpt on the given problems.
15//
16// Examples:
17// * Solve a model stored as a proto (infer the file/proto type):
18// mathopt_solve --input_file model.pb
19// * Solve a gzipped mps file, pick your solver:
20// mathopt_solve --input_file model.mps.gz --solver_type=glop
21// * Set a time limit:
22// mathopt_solve --input_file model.pb --time_limit 10s
23// * Set solve parameters in proto text format (see parameters.proto):
24// mathopt_solve --input_file model.pb --solve_parameters 'threads: 4'
25// * Specify the file format:
26// mathopt_solve --input_file model --format=mathopt

136// Returns the ModelUpdateProto read from the given file. The format must be
137// kMathOptBinary or kMathOptText; other values will generate an error.
138absl::StatusOr<ModelUpdateProto> ReadModelUpdate(
139 const absl::string_view file_path, const FileFormat format) {
140 switch (format) {
141 case FileFormat::kMathOptBinary:
142 return file::GetBinaryProto<ModelUpdateProto>(file_path,
143 file::Defaults());
144 case FileFormat::kMathOptText:
145 return file::GetTextProto<ModelUpdateProto>(file_path, file::Defaults());
146 default:
147 return util::InternalErrorBuilder() << "invalid format " << format;
148 }
149}

151// Prints the summary of the solve result.
152//
153// If feasibility_check_tolerances is not nullopt then a check of feasibility of
154// solution is done with the provided tolerances.
155absl::Status PrintSummary(const Model& model, const SolveResult& result,
156 const std::optional<FeasibilityCheckerOptions>
157 feasibility_check_tolerances) {
158 std::cout << "Solve finished:\n"
159 << " termination: " << result.termination << "\n"
160 << " solve time: " << result.solve_stats.solve_time
161 << "\n best primal bound: " << result.solve_stats.best_primal_bound
162 << "\n best dual bound: " << result.solve_stats.best_dual_bound
163 << std::endl;
164 if (result.solutions.empty()) {
165 std::cout << " no solution" << std::endl;
166 }
167 for (int i = 0; i < result.solutions.size(); ++i) {
168 const Solution& solution = result.solutions[i];
169 std::cout << " solution #" << (i + 1) << " objective: ";
170 if (solution.primal_solution.has_value()) {
171 std::cout << solution.primal_solution->objective_value;
172 if (feasibility_check_tolerances.has_value()) {
173 OR_ASSIGN_OR_RETURN3(
174 const ModelSubset broken_constraints,
175 CheckPrimalSolutionFeasibility(
176 model, solution.primal_solution->variable_values,
177 *feasibility_check_tolerances),
178 _ << "failed to check the primal solution feasibility of solution #"
179 << (i + 1));
180 if (!broken_constraints.empty()) {
181 std::cout << " (numerically infeasible: " << broken_constraints
182 << ')';
183 } else {
184 std::cout << " (numerically feasible)";
185 }
186 }
187 } else {
188 std::cout << "n/a";
189 }
190 std::cout << std::endl;
191 }
192
193 return absl::OkStatus();
194}

196absl::Status RunSolver() {
197 const std::string input_file_path = absl::GetFlag(FLAGS_input_file);
198 if (input_file_path.empty()) {
199 LOG(QFATAL) << "The flag --input_file is mandatory.";
200 }
201
202 // Parses --format.
203 const FileFormat format = [&]() {
204 const std::optional<FileFormat> format =
205 FormatFromFlagOrFilePath(absl::GetFlag(FLAGS_format), input_file_path);
206 if (format.has_value()) {
207 return *format;
208 }
209 LOG(QFATAL) << "Can't guess the format from the file extension, please "
210 "use --format to specify the file format explicitly.";
211 }();
212 // We deal with input validation in the ReadModel() function.
213
214 // Read the model and the optional updates.
215 const std::vector<std::string> update_file_paths =
216 absl::GetFlag(FLAGS_update_files);
217 if (!update_file_paths.empty() && format != FileFormat::kMathOptBinary &&
218 format != FileFormat::kMathOptText) {
219 LOG(QFATAL) << "Can't use --update_files with a input of format " << format
220 << ".";
221 }
222
223 OR_ASSIGN_OR_RETURN3((auto [model_proto, optional_hint]),
224 ReadModel(input_file_path, format),
225 _ << "failed to read " << input_file_path);
226
227 std::vector<ModelUpdateProto> model_updates;
228 for (const std::string& update_file_path : update_file_paths) {
229 ASSIGN_OR_RETURN(ModelUpdateProto update,
230 ReadModelUpdate(update_file_path, format));
231 model_updates.emplace_back(std::move(update));
232 }
233
234 if (!absl::GetFlag(FLAGS_names)) {
235 RemoveNames(model_proto);
236 for (ModelUpdateProto& update : model_updates) {
237 RemoveNames(update);
238 }
239 }
240
241 // Parse the problem and the updates.
242 ASSIGN_OR_RETURN(const std::unique_ptr<Model> model,
243 Model::FromModelProto(model_proto));
244 for (int u = 0; u < model_updates.size(); ++u) {
245 const ModelUpdateProto& update = model_updates[u];
246 RETURN_IF_ERROR(model->ApplyUpdateProto(update))
247 << "failed to apply the update file: " << update_file_paths[u];
248 }
249 if (absl::GetFlag(FLAGS_lp_relaxation)) {
250 for (const Variable v : model->Variables()) {
251 model->set_continuous(v);
252 }
253 }
254
255 if (absl::GetFlag(FLAGS_ranges)) {
256 std::cout << "Ranges of finite non-zero values in the model:\n"
257 << ComputeModelRanges(*model) << std::endl;
258 }
259
260 // Optionally prints the problem.
261 if (absl::GetFlag(FLAGS_print_model)) {
262 std::cout << *model;
263 std::cout.flush();
264 }
265
266 // Solve the problem.
267 SolveParameters solve_parameters = absl::GetFlag(FLAGS_solve_parameters);
268 solve_parameters.time_limit = absl::GetFlag(FLAGS_time_limit);
269 SolveArguments solve_args = {.parameters = solve_parameters};
270 if (optional_hint.has_value()) {
271 OR_ASSIGN_OR_RETURN3(ModelSolveParameters::SolutionHint hint,
272 ModelSolveParameters::SolutionHint::FromProto(
273 *model, optional_hint.value()),
274 _ << "invalid solution hint");
275 solve_args.model_parameters.solution_hints.push_back(std::move(hint));
276 std::cout << "Using the solution hint from the MPModelProto." << std::endl;
277 }
278 if (absl::GetFlag(FLAGS_solver_logs)) {
279 solve_args.message_callback = PrinterMessageCallback(std::cout, "logs| ");
280 }
281 OR_ASSIGN_OR_RETURN3(
282 const SolveResult result,
283 Solve(*model, absl::GetFlag(FLAGS_solver_type), solve_args),
284 _ << "the solver failed");
285
286 const FeasibilityCheckerOptions feasiblity_checker_options = {
287 .absolute_constraint_tolerance =
288 absl::GetFlag(FLAGS_absolute_constraint_tolerance),
289 .integrality_tolerance = absl::GetFlag(FLAGS_integrality_tolerance),
290 .nonzero_tolerance = absl::GetFlag(FLAGS_nonzero_tolerance),
291 };
292 RETURN_IF_ERROR(
293 PrintSummary(*model, result,
294 absl::GetFlag(FLAGS_check_solutions)
295 ? std::make_optional(feasiblity_checker_options)
296 : std::nullopt));
297
298 return absl::OkStatus();
299}
