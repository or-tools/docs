37// A solver for a given model and solver implementation.
38//
39// Use the New() function to build a new solver instance; then call Solve() to
40// solve the model. You can then update the model using Update() and resolve.
41//
42// Thread-safety: methods Solve() and Update() must not be called concurrently;
43// they will immediately return with an error status if this happens. Some
44// solvers may add more restriction regarding threading. Please see
45// SOLVER_TYPE_XXX documentation for details.
46//
47// Usage:
48// const ModelProto model = ...;
49// const auto solver = Solver::New(SOLVER_TYPE_GSCIP,
50// model,
51// /*arguments=*/{});
52// CHECK_OK(solver.status());
53// Solver::SolveArgs solve_arguments;
54// ...
55//
56// // First solve of the initial Model.
57// const auto first_solution = (*solver)->Solve(solve_arguments);
58// CHECK_OK(first_solution.status());
59// // Use the first_solution here.
60//
61// // Update the Model with a ModelUpdate.
62// const ModelUpdate update = ...;
63// CHECK_OK((*solver)->Update(update));
64// const auto second_solution = (*solver)->Solve(solve_arguments);
65// CHECK_OK(second_solution.status());
66// // Use the second_solution of the updated problem here.
67//
68class Solver {
69 public:
70 using InitArgs = SolverInterface::InitArgs;
71
72 // Callback function for messages callback sent by the solver.
73 //
74 // Each message represents a single output line from the solver, and each
75 // message does not contain any '\n' character in it.
76 //
77 // Thread-safety: a callback may be called concurrently from multiple
78 // threads. The users is expected to use proper synchronization primitives to
79 // deal with that.
80 using MessageCallback = SolverInterface::MessageCallback;
81
82 // Callback function type for MIP/LP callbacks.
83 using Callback = std::function<CallbackResultProto(const CallbackDataProto&)>;
84
85 // Arguments used when calling Solve() to solve the problem.
86 struct SolveArgs {
87 SolveParametersProto parameters;
88 ModelSolveParametersProto model_parameters;
89
90 // An optional callback for messages emitted by the solver.
91 //
92 // When set it enables the solver messages and ignores the `enable_output`
93 // in solve parameters; messages are redirected to the callback and not
94 // printed on stdout/stderr/logs anymore.
95 MessageCallback message_callback = nullptr;
96
97 CallbackRegistrationProto callback_registration;
98 Callback user_cb = nullptr;
99
100 // An optional interrupter that the solver can use to interrupt the solve
101 // early.
102 SolveInterrupter* interrupter = nullptr;
103 };
104
105 // Arguments used when calling ComputeInfeasibleSubsystem().

112 // in solve parameters; messages are redirected to the callback and not
113 // printed on stdout/stderr/logs anymore.
114 MessageCallback message_callback = nullptr;
115
116 // An optional interrupter that the solver can use to interrupt the solve

118 SolveInterrupter* interrupter = nullptr;
119 };

137 // Solves the current model (included all updates).
138 absl::StatusOr<SolveResultProto> Solve(const SolveArgs& arguments);
139
140 // Updates the model to solve and returns true, or returns false if this
141 // update is not supported by the underlying solver.
142 //
143 // A status error will be returned if the model_update is invalid or the
144 // underlying solver has an internal error.
145 //
146 // When this function returns false, the Solver object is in a failed
147 // state. In that case the underlying SolverInterface implementation has been
148 // destroyed (this enables the caller to instantiate a new Solver without
149 // destroying the previous one first even if they use Gurobi with a single-use
150 // license).
151 absl::StatusOr<bool> Update(const ModelUpdateProto& model_update);
152
153 // Computes an infeasible subsystem of `model`.
154 absl::StatusOr<ComputeInfeasibleSubsystemResultProto>
155 ComputeInfeasibleSubsystem(const ComputeInfeasibleSubsystemArgs& arguments);
156
157 // A shortcut for calling Solver::New() and then
158 // Solver()::ComputeInfeasibleSubsystem()
159 static absl::StatusOr<ComputeInfeasibleSubsystemResultProto>
160 NonIncrementalComputeInfeasibleSubsystem(
161 const ModelProto& model, SolverTypeProto solver_type,
162 const InitArgs& init_args,
163 const ComputeInfeasibleSubsystemArgs& compute_infeasible_subsystem_args);
164
165 private:
166 Solver(std::unique_ptr<SolverInterface> underlying_solver,
167 ModelSummary model_summary);
168
169 // Tracker used to ensure that Solve() and Update() are not called
170 // concurrently.
171 ConcurrentCallsGuard::Tracker concurrent_calls_tracker_;
172
173 // Can be nullptr only if fatal_failure_occurred_ is true (but the contrary is
174 // not true). This happens when Update() returns false.
175 std::unique_ptr<SolverInterface> underlying_solver_;
176
177 ModelSummary model_summary_;
178
179 // Set to true if a previous call to Solve() or Update() returned a failing
180 // status (or if Update() returned false).
181 //
182 // This is guarded by concurrent_calls_tracker_.
183 bool fatal_failure_occurred_ = false;
184};
