23std::string ToString(MPCallbackEvent event) {
24 switch (event) {
25 case MPCallbackEvent::kMipSolution:
26 return "MIP_SOLUTION";
27 case MPCallbackEvent::kMip:
28 return "MIP";
29 case MPCallbackEvent::kMipNode:
30 return "MIP_NODE";
31 case MPCallbackEvent::kBarrier:
32 return "BARRIER";
33 case MPCallbackEvent::kMessage:
34 return "MESSAGE";
35 case MPCallbackEvent::kPresolve:
36 return "PRESOLVE";
37 case MPCallbackEvent::kPolling:
38 return "POLLING";
39 case MPCallbackEvent::kMultiObj:
40 return "MULTI_OBJ";
41 case MPCallbackEvent::kSimplex:
42 return "SIMPLEX";
43 case MPCallbackEvent::kUnknown:
44 return "UNKNOWN";
45 default:
46 LOG(FATAL) << "Unrecognized callback event: " << static_cast<int>(event);
47 }

52// Returns true if any of the callbacks in a list might add cuts.
53bool CallbacksMightAddCuts(const std::vector<MPCallback*>& callbacks) {
54 for (MPCallback* callback : callbacks) {
55 if (callback->might_add_cuts()) {
56 return true;
57 }
58 }
59 return false;
60}
