246TEST(CarpParserTest, LoadInstanceFile) {
247 std::string file_name =
248 file::JoinPath(absl::GetFlag(FLAGS_test_srcdir),
249 ROOT_DIR "ortools/routing/testdata/carp_gdb19.dat");
250 CarpParser parser;
251 EXPECT_TRUE(parser.LoadFile(file_name));
252 EXPECT_EQ(parser.name(), "gdb19");
253 EXPECT_EQ(parser.comment(), "10000 (cota superior)");
254 EXPECT_EQ(parser.NumberOfNodes(), 8);
255 EXPECT_EQ(parser.NumberOfEdgesWithServicing(), 11);
256 EXPECT_EQ(parser.NumberOfEdgesWithoutServicing(), 0);
257 EXPECT_EQ(parser.NumberOfEdges(), 11);
258 EXPECT_EQ(parser.NumberOfVehicles(), 3);
259 EXPECT_EQ(parser.capacity(), 27);
260 EXPECT_EQ(parser.TotalServicingCost(), 45);
261 EXPECT_EQ(parser.depot(), 0);
262
263 EXPECT_EQ(parser.traversing_costs().size(), 11);
264 EXPECT_EQ(parser.GetTraversingCost(0, 1), 4);
265 EXPECT_EQ(parser.GetTraversingCost(1, 0), 4);
266 EXPECT_EQ(parser.servicing_demands().size(), 11);
267 EXPECT_EQ(parser.GetServicing(0, 1), 8);
268 EXPECT_EQ(parser.GetServicing(1, 0), 8);
269}
