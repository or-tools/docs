51// Describes completely a solution to a routing problem in preparation of its
52// serialization as a string.
53class RoutingSolution {
54 public:
55 // Describes a state transition performed by a vehicle: starting from/ending
56 // at a given depot, serving a given customer, etc.
57 // When need be, each event can have a specific demand ID (this is mostly
58 // useful when servicing arcs and edges). An event always stores an arc:
59 // this is simply the edge when servicing the edge (it should correspond to
60 // the direction in which the edge is traversed); when the event is about
61 // a node (either a depot or a demand), both ends of the arc should be the
62 // node the event is about.
63 struct Event {
64 // Describes the type of events that occur along a route.
65 enum class Type {
66 // The vehicle starts its route at a depot.
67 kStart,
68 // The vehicle ends its route at a depot (not necessarily the same as the
69 // starting one).
70 kEnd,
71 // The vehicle traverses the arc while servicing it.
72 kServeArc,
73 // The vehicle traverses the edge while servicing it.
74 kServeEdge,
75 // The vehicle serves the demand of the node.
76 kServeNode,
77 // The vehicle simply goes through an edge or an arc without servicing.
78 kTransit
79 };
80
81 Type type;
82 int64_t demand_id;
83 Arc arc;
84 std::string arc_name;
85
86 Event(Type type, int64_t demand_id, Arc arc)
87 : type(type), demand_id(demand_id), arc(arc) {}
88 Event(Type type, int64_t demand_id, Arc arc, std::string_view arc_name)
89 : type(type), demand_id(demand_id), arc(arc), arc_name(arc_name) {}
90
91 bool operator==(const Event& other) const {
92 return type == other.type && demand_id == other.demand_id &&
93 arc == other.arc && arc_name == other.arc_name;
94 }
95 bool operator!=(const Event& other) const { return !(*this == other); }
96 };
97
98 using Route = std::vector<Event>;
99
100 RoutingSolution(std::vector<Route> routes, std::vector<int64_t> total_demands,
101 std::vector<int64_t> total_distances, int64_t total_cost = -1,
102 int64_t total_distance = -1, double total_time = -1.0,
103 std::string_view name = "")
104 : routes_(std::move(routes)),
105 total_demands_(std::move(total_demands)),
106 total_distances_(std::move(total_distances)),
107 total_cost_(total_cost),
108 total_distance_(total_distance),
109 total_time_(total_time),
110 name_(name) {
111 CHECK_EQ(routes_.size(), total_demands_.size());
112 CHECK_EQ(routes_.size(), total_distances_.size());
113 }
114
115 bool operator==(const RoutingSolution& other) const {
116 return routes_ == other.routes_ && total_demands_ == other.total_demands_ &&
117 total_distances_ == other.total_distances_ &&
118 total_cost_ == other.total_cost_ && total_time_ == other.total_time_;
119 }
120 bool operator!=(const RoutingSolution& other) const {
121 return !(*this == other);
122 }
123
124 // Setters for solution metadata.
125 void SetTotalTime(double total_time) { total_time_ = total_time; }
126 void SetTotalCost(int64_t total_cost) { total_cost_ = total_cost; }
127 void SetTotalDistance(int64_t total_distance) {
128 total_distance_ = total_distance;
129 }
130 void SetName(std::string_view name) { name_ = name; }
131 void SetAuthors(std::string_view authors) { authors_ = authors; }

148 // Serializes the bare solution to a string, i.e. only the routes for the
149 // vehicles, without other metadata that is typically present in solution
150 // files.
151 std::string SerializeToString(RoutingOutputFormat format) const {
152 switch (format) {
153 case RoutingOutputFormat::kNone:
154 return "";
155 case RoutingOutputFormat::kTSPLIB:
156 return SerializeToTSPLIBString();
157 case RoutingOutputFormat::kCVRPLIB:
158 return SerializeToCVRPLIBString();
159 case RoutingOutputFormat::kCARPLIB:
160 return SerializeToCARPLIBString();
161 case RoutingOutputFormat::kNEARPLIB:
162 return SerializeToNEARPLIBString();
163 }
164 }

166 // Serializes the full solution to the given file, including metadata like
167 // instance name or total cost, depending on the format.
168 // For TSPLIB, solution files are typically called "tours".
169 std::string SerializeToSolutionFile(RoutingOutputFormat format) const {
170 switch (format) {
171 case RoutingOutputFormat::kNone:
172 return "";
173 case RoutingOutputFormat::kTSPLIB:
174 return SerializeToTSPLIBSolutionFile();
175 case RoutingOutputFormat::kCVRPLIB:
176 return SerializeToCVRPLIBSolutionFile();
177 case RoutingOutputFormat::kCARPLIB:
178 return SerializeToCARPLIBSolutionFile();
179 case RoutingOutputFormat::kNEARPLIB:
180 return SerializeToNEARPLIBSolutionFile();
181 }
182 }

184 // Serializes the full solution to the given file, including metadata like
185 // instance name or total cost, depending on the format.
186 void WriteToSolutionFile(RoutingOutputFormat format,
187 const std::string& file_name) const;
188
189 private:
190 // Description of the solution. Typically, one element per route (e.g., one
191 // vector of visited nodes per route). These elements are supposed to be
192 // returned by a solver.
193 // Depots are not explicitly stored as a route-level attribute, but rather by
194 // specific transitions (starting or ending at a depot).
195 std::vector<std::vector<Event>> routes_;
196 std::vector<int64_t> total_demands_;
197 std::vector<int64_t> total_distances_;
198
199 // Solution metadata. These elements could be set either by the solver or by
200 // the caller.
201 int64_t total_cost_;
202 int64_t total_distance_;
203 double total_time_;
204 std::string name_;
205 std::string authors_;
206
207 int64_t NumberOfNonemptyRoutes() const;

209 // The various implementations of SerializeToString depending on the format.
210
211 // Generates a string representation of a solution in the TSPLIB format.
212 // TSPLIB explicitly outputs the depot in its tours.
213 // It has been defined in
214 // http://comopt.ifi.uni-heidelberg.de/software/TSPLIB95/ where solutions are
215 // referred to as "tours".
216 std::string SerializeToTSPLIBString() const;
217 // Generates a string representation of a solution in the CVRPLIB format.
218 // CVRPLIB doesn't explicitly output the depot in its tours.
219 // Format used in http://vrp.atd-lab.inf.puc-rio.br/
220 // Better description of the format:
221 // http://dimacs.rutgers.edu/programs/challenge/vrp/cvrp/
222 std::string SerializeToCVRPLIBString() const;
223 // Generates a string representation of a solution in the CARPLIB format.
224 // Format used in https://www.uv.es/belengue/carp.html
225 // Formal description of the format: https://www.uv.es/~belengue/carp/READ_ME
226 // Another description of the format:
227 // http://dimacs.rutgers.edu/programs/challenge/vrp/carp/
228 std::string SerializeToCARPLIBString() const;
229 // Generates a string representation of a solution in the NEARPLIB format.
230 // Format used in https://www.sintef.no/projectweb/top/nearp/
231 // Formal description of the format:
232 // https://www.sintef.no/projectweb/top/nearp/documentation/
233 // Example:
234 // https://www.sintef.no/globalassets/project/top/nearp/solutionformat.txt
235 std::string SerializeToNEARPLIBString() const;
236
237 // The various implementations of SerializeToSolutionFile depending on the
238 // format. These methods are highly similar to the previous ones.
239 std::string SerializeToTSPLIBSolutionFile() const;
240 std::string SerializeToCVRPLIBSolutionFile() const;
241 std::string SerializeToCARPLIBSolutionFile() const;
242 std::string SerializeToNEARPLIBSolutionFile() const;
243};

245// Formats a solution or solver statistic according to the given format.
246template <typename T>
247std::string FormatStatistic(absl::string_view name, T value,
248 RoutingOutputFormat format) {
249 // TODO(user): think about using an enum instead of names (or even a
250 // full-fledged struct/class) for the various types of fields.
251 switch (format) {
252 case RoutingOutputFormat::kNone:
253 ABSL_FALLTHROUGH_INTENDED;
254 case RoutingOutputFormat::kTSPLIB:
255 return absl::StrCat(name, " = ", value);
256 case RoutingOutputFormat::kCVRPLIB:
257 return absl::StrCat(name, " ", value);
258 case RoutingOutputFormat::kCARPLIB:
259 // For CARPLIB, the statistics do not have names, it's up to the user to
260 // memorize their order.
261 return absl::StrCat(value);
262 case RoutingOutputFormat::kNEARPLIB:
263 return absl::StrCat(name, " : ", value);
264 }
265}
266
267// Specialization for doubles to show a higher precision: without this
268// specialization, 591.556557 is displayed as 591.557.
269template <>
270inline std::string FormatStatistic(absl::string_view name, double value,
271 RoutingOutputFormat format) {
272 switch (format) {
273 case RoutingOutputFormat::kNone:
274 ABSL_FALLTHROUGH_INTENDED;
275 case RoutingOutputFormat::kTSPLIB:
276 return absl::StrFormat("%s = %f", name, value);
277 case RoutingOutputFormat::kCVRPLIB:
278 return absl::StrFormat("%s %f", name, value);
279 case RoutingOutputFormat::kCARPLIB:
280 return absl::StrFormat("%f", value);
281 case RoutingOutputFormat::kNEARPLIB:
282 return absl::StrFormat("%s : %f", name, value);
283 }
284}

286// Prints a formatted solution or solver statistic according to the given
287// format.
288template <typename T>
289void PrintStatistic(absl::string_view name, T value,
290 RoutingOutputFormat format) {
291 absl::PrintF("%s\n", FormatStatistic(name, value, format));
292}
