32// Useful constants: word and double word will all bits set.
33static const uint64_t kAllBits64 = uint64_t{0xFFFFFFFFFFFFFFFF};
34static const uint64_t kAllBitsButLsb64 = uint64_t{0xFFFFFFFFFFFFFFFE};
35static const uint32_t kAllBits32 = 0xFFFFFFFFU;

37// Returns a word with only bit pos set.
38inline uint64_t OneBit64(int pos) { return uint64_t{1} << pos; }
39inline uint32_t OneBit32(int pos) { return 1U << pos; }

41// Returns the number of bits set in n.
42inline uint64_t BitCount64(uint64_t n) {
43 const uint64_t m1 = uint64_t{0x5555555555555555};
44 const uint64_t m2 = uint64_t{0x3333333333333333};
45 const uint64_t m4 = uint64_t{0x0F0F0F0F0F0F0F0F};
46 const uint64_t h01 = uint64_t{0x0101010101010101};
47 n -= (n >> 1) & m1;
48 n = (n & m2) + ((n >> 2) & m2);
49 n = (n + (n >> 4)) & m4;
50 n = (n * h01) >> 56;
51 return n;
52}
53inline uint32_t BitCount32(uint32_t n) {
54 n -= (n >> 1) & 0x55555555UL;
55 n = (n & 0x33333333) + ((n >> 2) & 0x33333333UL);
56 n = (n + (n >> 4)) & 0x0F0F0F0FUL;
57 n = n + (n >> 8);
58 n = n + (n >> 16);
59 return n & 0x0000003FUL;
60}

66// Returns the least significant bit position in n.
67// Discussion around lsb computation:
68// De Bruijn is almost as fast as the bsr/bsf-instruction-based intrinsics.

81inline int LeastSignificantBitPosition64DeBruijn(uint64_t n) {
82 static const uint64_t kSeq = uint64_t{0x0218a392dd5fb34f};
83 static const int kTab[64] = {
84 // initialized by 'kTab[(kSeq << i) >> 58] = i
85 0, 1, 2, 7, 3, 13, 8, 19, 4, 25, 14, 28, 9, 52, 20, 58,
86 5, 17, 26, 56, 15, 38, 29, 40, 10, 49, 53, 31, 21, 34, 59, 42,
87 63, 6, 12, 18, 24, 27, 51, 57, 16, 55, 37, 39, 48, 30, 33, 41,
88 62, 11, 23, 50, 54, 36, 47, 32, 61, 22, 35, 46, 60, 45, 44, 43,
89 };
90 return kTab[((n & (~n + 1)) * kSeq) >> 58];
91}

93inline int LeastSignificantBitPosition64Default(uint64_t n) {
94 DCHECK_NE(n, 0);
95 int pos = 63;
96 if (n & 0x00000000FFFFFFFFLL) {
97 pos -= 32;
98 } else {
99 n >>= 32;
100 }
101 if (n & 0x000000000000FFFFLL) {
102 pos -= 16;
103 } else {
104 n >>= 16;
105 }
106 if (n & 0x00000000000000FFLL) {
107 pos -= 8;
108 } else {
109 n >>= 8;
110 }
111 if (n & 0x000000000000000FLL) {
112 pos -= 4;
113 } else {
114 n >>= 4;
115 }
116 if (n & 0x0000000000000003LL) {
117 pos -= 2;
118 } else {
119 n >>= 2;
120 }
121 if (n & 0x0000000000000001LL) {
122 pos -= 1;
123 }
124 return pos;
125}

193// Returns the most significant bit position in n.
194#if USE_FAST_LEAST_SIGNIFICANT_BIT
195inline int MostSignificantBitPosition64Fast(uint64_t n) {
196 // __builtin_clzll(1) should always return 63. There is no penalty in
197 // using offset, and the code looks more like its uint32_t counterpart.
198 const int offset = __builtin_clzll(1);
199 return n == 0 ? 0 : (offset - __builtin_clzll(n));
200}
201#endif
202
203inline int MostSignificantBitPosition64Default(uint64_t n) {
204 int b = 0;
205 if (0 != (n & (kAllBits64 << (1 << 5)))) {
206 b |= (1 << 5);
207 n >>= (1 << 5);
208 }
209 if (0 != (n & (kAllBits64 << (1 << 4)))) {
210 b |= (1 << 4);
211 n >>= (1 << 4);
212 }
213 if (0 != (n & (kAllBits64 << (1 << 3)))) {
214 b |= (1 << 3);
215 n >>= (1 << 3);
216 }
217 if (0 != (n & (kAllBits64 << (1 << 2)))) {
218 b |= (1 << 2);
219 n >>= (1 << 2);
220 }
221 if (0 != (n & (kAllBits64 << (1 << 1)))) {
222 b |= (1 << 1);
223 n >>= (1 << 1);
224 }
225 if (0 != (n & (kAllBits64 << (1 << 0)))) {
226 b |= (1 << 0);
227 }
228 return b;
229}

284// Returns a word with bits from s to e set.
285inline uint64_t OneRange64(uint64_t s, uint64_t e) {
286 DCHECK_LE(s, 63);
287 DCHECK_LE(e, 63);
288 DCHECK_LE(s, e);
289 return (kAllBits64 << s) ^ ((kAllBits64 - 1) << e);
290}
291
292inline uint32_t OneRange32(uint32_t s, uint32_t e) {
293 DCHECK_LE(s, 31);
294 DCHECK_LE(e, 31);
295 DCHECK_LE(s, e);
296 return (kAllBits32 << s) ^ ((kAllBits32 - 1) << e);
297}

299// Returns a word with s least significant bits unset.
300inline uint64_t IntervalUp64(uint64_t s) {
301 DCHECK_LE(s, 63);
302 return kAllBits64 << s;
303}

305inline uint32_t IntervalUp32(uint32_t s) {
306 DCHECK_LE(s, 31);
307 return kAllBits32 << s;

310// Returns a word with the s most significant bits unset.
311inline uint64_t IntervalDown64(uint64_t s) {

316inline uint32_t IntervalDown32(uint32_t s) {
317 DCHECK_LE(s, 31);

324// Bit operators used to manipulates bitsets.
325
326// Returns the bit number in the word computed by BitOffsetXX,
327// corresponding to the bit at position pos in the bitset.
328// Note: '& 63' is faster than '% 64'
329// TODO(user): rename BitPos and BitOffset to something more understandable.
330inline uint32_t BitPos64(uint64_t pos) { return (pos & 63); }
331inline uint32_t BitPos32(uint32_t pos) { return (pos & 31); }
332
333// Returns the word number corresponding to bit number pos.
334inline uint64_t BitOffset64(uint64_t pos) { return (pos >> 6); }
335inline uint32_t BitOffset32(uint32_t pos) { return (pos >> 5); }
336
337// Returns the number of words needed to store size bits.
338inline uint64_t BitLength64(uint64_t size) { return ((size + 63) >> 6); }
339inline uint32_t BitLength32(uint32_t size) { return ((size + 31) >> 5); }
340
341// Returns the bit number in the bitset of the first bit of word number v.
342inline uint64_t BitShift64(uint64_t v) { return v << 6; }
343inline uint32_t BitShift32(uint32_t v) { return v << 5; }
344
345// Returns true if the bit pos is set in bitset.
346inline bool IsBitSet64(const uint64_t* const bitset, uint64_t pos) {
347 return (bitset[BitOffset64(pos)] & OneBit64(BitPos64(pos)));
348}
349inline bool IsBitSet32(const uint32_t* const bitset, uint32_t pos) {
350 return (bitset[BitOffset32(pos)] & OneBit32(BitPos32(pos)));
351}
352
353// Sets the bit pos to true in bitset.
354inline void SetBit64(uint64_t* const bitset, uint64_t pos) {
355 bitset[BitOffset64(pos)] |= OneBit64(BitPos64(pos));
356}
357inline void SetBit32(uint32_t* const bitset, uint32_t pos) {
358 bitset[BitOffset32(pos)] |= OneBit32(BitPos32(pos));
359}
360
361// Sets the bit pos to false in bitset.
362inline void ClearBit64(uint64_t* const bitset, uint64_t pos) {
363 bitset[BitOffset64(pos)] &= ~OneBit64(BitPos64(pos));
364}
365inline void ClearBit32(uint32_t* const bitset, uint32_t pos) {
366 bitset[BitOffset32(pos)] &= ~OneBit32(BitPos32(pos));
367}

401// Returns a mask with the bits pos % 64 and (pos ^ 1) % 64 sets.
402inline uint64_t TwoBitsFromPos64(uint64_t pos) {
403 return uint64_t{3} << (pos & 62);

406// This class is like an ITIVector<IndexType, bool> except that it provides a
407// more efficient way to iterate over the positions set to true. It achieves
408// this by caching the current uint64_t bucket in the Iterator and using
409// LeastSignificantBitPosition64() to iterate over the positions at 1 in this
410// bucket.
411template <typename IndexType = int64_t>
412class Bitset64 {
413 public:
414 // When speed matter, caching the base pointer like this to access this class
415 // in a read only mode help.
416 class ConstView {
417 public:
418 explicit ConstView(const Bitset64* bitset) : data_(bitset->data_.data()) {}
419
420 bool operator[](IndexType i) const {
421 return data_[BitOffset64(Value(i))] & OneBit64(BitPos64(Value(i)));
422 }
423
424 const uint64_t* data() const { return data_; }
425
426 private:
427 const uint64_t* const data_;
428 };
429
430 Bitset64() : size_(), data_() {}
431 explicit Bitset64(IndexType size)
432 : size_(Value(size) > 0 ? size : IndexType(0)),
433 data_(BitLength64(Value(size_))) {}
434
435 // This type is neither copyable nor movable.
436 Bitset64(const Bitset64&) = delete;
437 Bitset64& operator=(const Bitset64&) = delete;
438
439 ConstView const_view() const { return ConstView(this); }
440
441 // Returns how many bits this Bitset64 can hold.
442 IndexType size() const { return size_; }
443
444 // Appends value at the end of the bitset.
445 void PushBack(bool value) {
446 ++size_;
447 data_.resize(BitLength64(Value(size_)), 0);
448 Set(size_ - 1, value);
449 }
450
451 // Resizes the Bitset64 to the given number of bits. New bits are sets to 0.
452 void resize(int size) { Resize(IndexType(size)); }
453 void Resize(IndexType size) {
454 DCHECK_GE(Value(size), 0);
455 size_ = Value(size) > 0 ? size : IndexType(0);
456 data_.resize(BitLength64(Value(size_)), 0);
457 }
458
459 // Changes the number of bits the Bitset64 can hold and set all of them to 0.
460 void ClearAndResize(IndexType size) {
461 DCHECK_GE(Value(size), 0);
462 size_ = Value(size) > 0 ? size : IndexType(0);
463
464 // Memset is 4x faster than data_.assign() as of 19/03/2014.
465 // TODO(user): Ideally if a realloc happens, we don't need to copy the old
466 // data...
467 const size_t bit_length = static_cast<size_t>(BitLength64(Value(size_)));
468 const size_t to_clear = std::min(data_.size(), bit_length);
469 data_.resize(bit_length, 0);
470 memset(data_.data(), 0, to_clear * sizeof(int64_t));
471 }
472
473 // Sets all bits to 0.
474 void ClearAll() { memset(data_.data(), 0, data_.size() * sizeof(int64_t)); }
475
476 // Sets the bit at position i to 0.
477 void Clear(IndexType i) {
478 DCHECK_GE(Value(i), 0);
479 DCHECK_LT(Value(i), Value(size_));
480 data_[BitOffset64(Value(i))] &= ~OneBit64(BitPos64(Value(i)));
481 }
482
483 // Sets bucket containing bit i to 0.
484 void ClearBucket(IndexType i) {
485 DCHECK_GE(Value(i), 0);
486 DCHECK_LT(Value(i), Value(size_));
487 data_[BitOffset64(Value(i))] = 0;
488 }
489
490 // Clears the bits at position i and i ^ 1.
491 void ClearTwoBits(IndexType i) {
492 DCHECK_GE(Value(i), 0);
493 DCHECK_LT(Value(i), Value(size_));
494 data_[BitOffset64(Value(i))] &= ~TwoBitsFromPos64(Value(i));
495 }
496
497 // Returns true if the bit at position i or the one at position i ^ 1 is set.
498 bool AreOneOfTwoBitsSet(IndexType i) const {
499 DCHECK_GE(Value(i), 0);
500 DCHECK_LT(Value(i), Value(size_));
501 return data_[BitOffset64(Value(i))] & TwoBitsFromPos64(Value(i));
502 }
503
504 // Returns true if the bit at position i is set.
505 bool IsSet(IndexType i) const {
506 DCHECK_GE(Value(i), 0);
507 DCHECK_LT(Value(i), Value(size_));
508 return data_[BitOffset64(Value(i))] & OneBit64(BitPos64(Value(i)));
509 }
510
511 // Same as IsSet().
512 bool operator[](IndexType i) const { return IsSet(i); }
513
514 // Sets the bit at position i to 1.
515 void Set(IndexType i) {
516 DCHECK_GE(Value(i), 0);
517 DCHECK_LT(Value(i), size_);
518 data_[BitOffset64(Value(i))] |= OneBit64(BitPos64(Value(i)));
519 }
520
521 // If value is true, sets the bit at position i to 1, sets it to 0 otherwise.
522 void Set(IndexType i, bool value) {
523 if (value) {
524 Set(i);
525 } else {
526 Clear(i);
527 }
528 }

530 // Copies bucket containing bit i from "other" to "this".
531 void CopyBucket(const Bitset64<IndexType>& other, IndexType i) {
532 const uint64_t offset = BitOffset64(Value(i));
533 data_[offset] = other.data_[offset];
534 }
535
536 // Copies "other" to "this". The bitsets do not have to be of the same size.
537 // If "other" is smaller, high order bits are not changed. If "other" is
538 // larger, its high order bits are ignored. In any case "this" is not resized.
539 template <typename OtherIndexType>
540 void SetContentFromBitset(const Bitset64<OtherIndexType>& other) {
541 const int64_t min_size = std::min(data_.size(), other.data_.size());
542 if (min_size == 0) return;
543 const uint64_t last_common_bucket = data_[min_size - 1];
544 memcpy(data_.data(), other.data_.data(), min_size * sizeof(uint64_t));
545 if (data_.size() >= other.data_.size()) {
546 const uint64_t bitmask = kAllBitsButLsb64
547 << BitPos64(other.Value(other.size() - 1));
548 data_[min_size - 1] &= ~bitmask;
549 data_[min_size - 1] |= (bitmask & last_common_bucket);
550 }
551 }
552
553 // Same as SetContentFromBitset where "this" and "other" have the same size.
554 template <typename OtherIndexType>
555 void SetContentFromBitsetOfSameSize(const Bitset64<OtherIndexType>& other) {
556 DCHECK_EQ(Value(size()), other.Value(other.size()));
557 memcpy(data_.data(), other.data_.data(), data_.size() * sizeof(uint64_t));
558 }
559
560 // Sets "this" to be the intersection of "this" and "other". The
561 // bitsets do not have to be the same size. If other is smaller, all
562 // the higher order bits are assumed to be 0.
563 void Intersection(const Bitset64<IndexType>& other) {
564 const int min_size = std::min(data_.size(), other.data_.size());
565 for (int i = 0; i < min_size; ++i) {
566 data_[i] &= other.data_[i];
567 }
568 for (int i = min_size; i < data_.size(); ++i) {
569 data_[i] = 0;
570 }
571 }
572
573 // Sets "this" to be the union of "this" and "other". The
574 // bitsets do not have to be the same size. If other is smaller, all
575 // the higher order bits are assumed to be 0.
576 void Union(const Bitset64<IndexType>& other) {
577 const int min_size = std::min(data_.size(), other.data_.size());
578 for (int i = 0; i < min_size; ++i) {
579 data_[i] |= other.data_[i];
580 }
581 }

583 // Class to iterate over the bit positions at 1 of a Bitset64.
584 //
585 // IMPORTANT: Because the iterator "caches" the current uint64_t bucket, this
586 // will probably not do what you want if Bitset64 is modified while iterating.
587 class EndIterator {};
588 class Iterator {
589 public:
590 explicit Iterator(const Bitset64& bitset)
591 : data_(bitset.data_.data()), size_(bitset.data_.size()) {
592 if (!bitset.data_.empty()) {
593 current_ = data_[0];
594 this->operator++();
595 }
596 }
597
598 bool operator!=(const EndIterator&) const { return size_ != 0; }
599
600 IndexType operator*() const { return IndexType(index_); }
601
602 void operator++() {
603 int bucket = BitOffset64(index_);
604 while (current_ == 0) {
605 bucket++;
606 if (bucket == size_) {
607 size_ = 0;
608 return;
609 }
610 current_ = data_[bucket];
611 }
612
613 // Computes the index and clear the least significant bit of current_.
614 index_ = BitShift64(bucket) | LeastSignificantBitPosition64(current_);
615 current_ &= current_ - 1;
616 }
617
618 private:
619 const uint64_t* const data_;
620 int size_;
621 int index_ = 0;
622 uint64_t current_ = 0;
623 };
624
625 // Allows range-based "for" loop on the non-zero positions:
626 // for (const IndexType index : bitset) {}
627 Iterator begin() const { return Iterator(*this); }
628 EndIterator end() const { return EndIterator(); }

630 // Cryptic function! This is just an optimized version of a given piece of
631 // code and has probably little general use.
632 static uint64_t ConditionalXorOfTwoBits(IndexType i, uint64_t use1,
633 Bitset64<IndexType>::ConstView set1,
634 uint64_t use2,
635 Bitset64<IndexType>::ConstView set2) {
636 DCHECK(use1 == 0 || use1 == 1);
637 DCHECK(use2 == 0 || use2 == 1);
638 const int bucket = BitOffset64(Value(i));
639 const int pos = BitPos64(Value(i));
640 return ((use1 << pos) & set1.data()[bucket]) ^
641 ((use2 << pos) & set2.data()[bucket]);
642 }

645 std::string DebugString() const {

647 for (IndexType i(0); i < size(); ++i) {
648 output += IsSet(i) ? "1" : "0";
649 }
650 return output;
651 }

656 static int Value(IndexType input);

658 IndexType size_;

665// Specialized version of Bitset64 that allows to query the last bit set more
666// efficiently.
667class BitQueue64 {
668 public:
669 BitQueue64() : size_(), top_(-1), data_() {}
670 explicit BitQueue64(int size)
671 : size_(size), top_(-1), data_(BitLength64(size), 0) {}
672
673 // This type is neither copyable nor movable.
674 BitQueue64(const BitQueue64&) = delete;
675 BitQueue64& operator=(const BitQueue64&) = delete;
676
677 void IncreaseSize(int size) {
678 CHECK_GE(size, size_);
679 size_ = size;
680 data_.resize(BitLength64(size), 0);
681 }
682
683 void ClearAndResize(int size) {
684 top_ = -1;
685 size_ = size;
686 data_.assign(BitLength64(size), 0);
687 }
688
689 void Set(int i) {
690 DCHECK_GE(i, 0);
691 DCHECK_LT(i, size_);
692 top_ = std::max(top_, i);
693 data_[BitOffset64(i)] |= OneBit64(BitPos64(i));
694 }
695
696 // Sets all the bits from 0 up to i-1 to 1.
697 void SetAllBefore(int i) {
698 DCHECK_GE(i, 0);
699 DCHECK_LT(i, size_);
700 top_ = std::max(top_, i - 1);
701 int bucket_index = static_cast<int>(BitOffset64(i));
702 data_[bucket_index] |= OneBit64(BitPos64(i)) - 1;
703 for (--bucket_index; bucket_index >= 0; --bucket_index) {
704 data_[bucket_index] = kAllBits64;
705 }
706 }
707
708 // Returns the position of the highest bit set in O(1) or -1 if no bit is set.
709 int Top() const { return top_; }
710
711 // Clears the Top() bit and recomputes the position of the next Top().
712 void ClearTop() {
713 DCHECK_NE(top_, -1);
714 int bucket_index = static_cast<int>(BitOffset64(top_));
715 uint64_t bucket = data_[bucket_index] &= ~OneBit64(BitPos64(top_));
716 while (!bucket) {
717 if (bucket_index == 0) {
718 top_ = -1;
719 return;
720 }
721 bucket = data_[--bucket_index];
722 }
723
724 // Note(user): I experimented with reversing the bit order in a bucket to
725 // use LeastSignificantBitPosition64() and it is only slightly faster at the
726 // cost of a lower Set() speed. So I preferred this version.
727 top_ = static_cast<int>(BitShift64(bucket_index) +
728 MostSignificantBitPosition64(bucket));
729 }
730
731 private:
732 int size_;
733 int top_;
734 std::vector<uint64_t> data_;
735};

768 to_clear_.clear();
769 }

776 const int kSparseThreshold = 300;
777 if (to_clear_.size() * kSparseThreshold < size) {
778 SparseClearAll();

780 } else {

789 if (index < size) {
790 to_clear_[new_index] = index;
791 ++new_index;
792 }
793 }
794 to_clear_.resize(new_index);
795 }
796 bitset_.Resize(size);
797 }
798 bool operator[](IntegerType index) const { return bitset_[index]; }
799 void Set(IntegerType index) {
800 if (!bitset_[index]) {
801 bitset_.Set(index);
802 to_clear_.push_back(index);
803 }
804 }
805 void SetUnsafe(IntegerType index) {
806 bitset_.Set(index);
807 to_clear_.push_back(index);
808 }
809 void Clear(IntegerType index) { bitset_.Clear(index); }
810 int NumberOfSetCallsWithDifferentArguments() const {
811 return to_clear_.size();
812 }
813 const std::vector<IntegerType>& PositionsSetAtLeastOnce() const {
814 return to_clear_;
815 }

817 // Tells the class that all its bits are cleared, so it can reset to_clear_

819 // the class is actually all cleared is only checked in debug mode.
820 //

822 // instance. This way, after the loop, a client can call this for efficiency.
823 void NotifyAllClear() {
824 if (DEBUG_MODE) {

826 }
827 to_clear_.clear();
828 }

832 }
