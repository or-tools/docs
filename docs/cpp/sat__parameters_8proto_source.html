33 // Branching and polarity
34 // ==========================================================================
35
36 // Variables without activity (i.e. at the beginning of the search) will be
37 // tried in this preferred order.
38 enum VariableOrder {
39 IN_ORDER = 0; // As specified by the problem.
40 IN_REVERSE_ORDER = 1;
41 IN_RANDOM_ORDER = 2;
42 }
43 optional VariableOrder preferred_variable_order = 1 [default = IN_ORDER];
44
45 // Specifies the initial polarity (true/false) when the solver branches on a
46 // variable. This can be modified later by the user, or the phase saving
47 // heuristic.
48 //
49 // Note(user): POLARITY_FALSE is usually a good choice because of the
50 // "natural" way to express a linear boolean problem.
51 enum Polarity {
52 POLARITY_TRUE = 0;
53 POLARITY_FALSE = 1;
54 POLARITY_RANDOM = 2;
55
56 // Choose the sign that tends to satisfy the most constraints. This is
57 // computed using a weighted sum: if a literal l appears in a constraint of
58 // the form: ... + coeff * l +... <= rhs with positive coefficients and
59 // rhs, then -sign(l) * coeff / rhs is added to the weight of l.variable().
60 POLARITY_WEIGHTED_SIGN = 3;
61
62 // The opposite choice of POLARITY_WEIGHTED_SIGN.
63 POLARITY_REVERSE_WEIGHTED_SIGN = 4;
64 }
65 optional Polarity initial_polarity = 2 [default = POLARITY_FALSE];
66
67 // If this is true, then the polarity of a variable will be the last value it
68 // was assigned to, or its default polarity if it was never assigned since the
69 // call to ResetDecisionHeuristic().
70 //
71 // Actually, we use a newer version where we follow the last value in the
72 // longest non-conflicting partial assignment in the current phase.
73 //
74 // This is called 'literal phase saving'. For details see 'A Lightweight
75 // Component Caching Scheme for Satisfiability Solvers' K. Pipatsrisawat and
76 // A.Darwiche, In 10th International Conference on Theory and Applications of
77 // Satisfiability Testing, 2007.
78 optional bool use_phase_saving = 44 [default = true];
79
80 // If non-zero, then we change the polarity heuristic after that many number
81 // of conflicts in an arithmetically increasing fashion. So x the first time,
82 // 2 * x the second time, etc...
83 optional int32 polarity_rephase_increment = 168 [default = 1000];
84
85 // The proportion of polarity chosen at random. Note that this take
86 // precedence over the phase saving heuristic. This is different from
87 // initial_polarity:POLARITY_RANDOM because it will select a new random
88 // polarity each time the variable is branched upon instead of selecting one
89 // initially and then always taking this choice.
90 optional double random_polarity_ratio = 45 [default = 0.0];
91
92 // A number between 0 and 1 that indicates the proportion of branching
93 // variables that are selected randomly instead of choosing the first variable
94 // from the given variable_ordering strategy.
95 optional double random_branches_ratio = 32 [default = 0.0];
96
97 // Whether we use the ERWA (Exponential Recency Weighted Average) heuristic as
98 // described in "Learning Rate Based Branching Heuristic for SAT solvers",
99 // J.H.Liang, V. Ganesh, P. Poupart, K.Czarnecki, SAT 2016.
100 optional bool use_erwa_heuristic = 75 [default = false];
101
102 // The initial value of the variables activity. A non-zero value only make
103 // sense when use_erwa_heuristic is true. Experiments with a value of 1e-2
104 // together with the ERWA heuristic showed slighthly better result than simply
105 // using zero. The idea is that when the "learning rate" of a variable becomes
106 // lower than this value, then we prefer to branch on never explored before
107 // variables. This is not in the ERWA paper.
108 optional double initial_variables_activity = 76 [default = 0.0];
109
110 // When this is true, then the variables that appear in any of the reason of
111 // the variables in a conflict have their activity bumped. This is addition to
112 // the variables in the conflict, and the one that were used during conflict
113 // resolution.
114 optional bool also_bump_variables_in_conflict_reasons = 77 [default = false];

240
241 // Restart algorithms.
242 //
243 // A reference for the more advanced ones is:
244 // Gilles Audemard, Laurent Simon, "Refining Restarts Strategies for SAT
245 // and UNSAT", Principles and Practice of Constraint Programming Lecture
246 // Notes in Computer Science 2012, pp 118-126
247 enum RestartAlgorithm {
248 NO_RESTART = 0;
249
250 // Just follow a Luby sequence times restart_period.
251 LUBY_RESTART = 1;
252
253 // Moving average restart based on the decision level of conflicts.
254 DL_MOVING_AVERAGE_RESTART = 2;
255
256 // Moving average restart based on the LBD of conflicts.
257 LBD_MOVING_AVERAGE_RESTART = 3;
258
259 // Fixed period restart every restart period.
260 FIXED_RESTART = 4;
261 }
262
263 // The restart strategies will change each time the strategy_counter is
264 // increased. The current strategy will simply be the one at index
265 // strategy_counter modulo the number of strategy. Note that if this list
266 // includes a NO_RESTART, nothing will change when it is reached because the
267 // strategy_counter will only increment after a restart.
268 //
269 // The idea of switching of search strategy tailored for SAT/UNSAT comes from
270 // Chanseok Oh with his COMiniSatPS solver, see http://cs.nyu.edu/~chanseok/.
271 // But more generally, it seems REALLY beneficial to try different strategy.
272 repeated RestartAlgorithm restart_algorithms = 61;
273 optional string default_restart_algorithms = 70
274 [default =
275 "LUBY_RESTART,LBD_MOVING_AVERAGE_RESTART,DL_MOVING_AVERAGE_RESTART"];
276
277 // Restart period for the FIXED_RESTART strategy. This is also the multiplier
278 // used by the LUBY_RESTART strategy.
279 optional int32 restart_period = 30 [default = 50];
280
281 // Size of the window for the moving average restarts.
282 optional int32 restart_running_window_size = 62 [default = 50];
283
284 // In the moving average restart algorithms, a restart is triggered if the
285 // window average times this ratio is greater that the global average.
286 optional double restart_dl_average_ratio = 63 [default = 1.0];
287 optional double restart_lbd_average_ratio = 71 [default = 1.0];
288
289 // Block a moving restart algorithm if the trail size of the current conflict
290 // is greater than the multiplier times the moving average of the trail size
291 // at the previous conflicts.
292 optional bool use_blocking_restart = 64 [default = false];
293 optional int32 blocking_restart_window_size = 65 [default = 5000];
294 optional double blocking_restart_multiplier = 66 [default = 1.4];
295
296 // After each restart, if the number of conflict since the last strategy
297 // change is greater that this, then we increment a "strategy_counter" that
298 // can be use to change the search strategy used by the following restarts.
299 optional int32 num_conflicts_before_strategy_changes = 68 [default = 0];
300
301 // The parameter num_conflicts_before_strategy_changes is increased by that
302 // much after each strategy change.
303 optional double strategy_change_increase_ratio = 69 [default = 0.0];

329 // solver will abort as soon as it detects that this limit is crossed. As a
330 // result, this limit is approximative, but usually the solver will not go too
331 // much over.
332 //
333 // TODO(user): This is only used by the pure SAT solver, generalize to CP-SAT.

336 // Stop the search when the gap between the best feasible objective (O) and
337 // our best objective bound (B) is smaller than a limit.
338 // The exact definition is:
339 // - Absolute: abs(O - B)
340 // - Relative: abs(O - B) / max(1, abs(O)).
341 //
342 // Important: The relative gap depends on the objective offset! If you
343 // artificially shift the objective, you will get widely different value of
344 // the relative gap.
345 //
346 // Note that if the gap is reached, the search status will be OPTIMAL. But
347 // one can check the best objective bound to see the actual gap.
348 //
349 // If the objective is integer, then any absolute gap < 1 will lead to a true
350 // optimal. If the objective is floating point, a gap of zero make little
351 // sense so is is why we use a non-zero default value. At the end of the
352 // search, we will display a warning if OPTIMAL is reported yet the gap is
353 // greater than this absolute gap.
354 optional double absolute_gap_limit = 159 [default = 1e-4];
355 optional double relative_gap_limit = 160 [default = 0.0];
