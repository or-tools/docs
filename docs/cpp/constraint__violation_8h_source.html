129 int64_t ObjectiveDelta(int var, int64_t delta) const {
130 if (var >= var_entries_.size()) return 0;
131 const std::vector<Entry>& data = var_entries_[var];
132 if (data.empty()) return 0;
133 if (data[0].ct_index != 0) return 0;
134 return data[0].coefficient * delta;
135 }

140 data.num_pos_literal + data.num_neg_literal + data.num_linear_entries;
141 if (size == 0) return {};
142 return absl::MakeSpan(&row_var_buffer_[data.start], size);
143 }

493// The violation of an int_mod constraint is defined as follow:
494//
495// if target and expr0 have the same sign:
496// min(
497// abs(value(target) - (value(expr0) % value(expr1))),
498// abs(value(target)) + abs((value(expr0) % value(expr1)) - value(expr1)),
499// abs(value(expr0) % value(expr1)) + abs(value(target) - value(expr1)),
500// )

502// if target and expr0 have different sign:
503// abs(target) + abs(expr0)
504// Note: the modulo (expr1) is always fixed.
