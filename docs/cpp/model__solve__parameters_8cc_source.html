54absl::Status ModelSolveParameters::CheckModelStorage(
55 const ModelStorage* const expected_storage) const {
56 for (const SolutionHint& hint : solution_hints) {
57 RETURN_IF_ERROR(hint.CheckModelStorage(expected_storage))
58 << "invalid hint in solution_hints";
59 }
60 if (initial_basis.has_value()) {
61 RETURN_IF_ERROR(initial_basis->CheckModelStorage(expected_storage))
62 << "invalid initial_basis";
63 }
64 RETURN_IF_ERROR(variable_values_filter.CheckModelStorage(expected_storage))
65 << "invalid variable_values_filter";
66 RETURN_IF_ERROR(dual_values_filter.CheckModelStorage(expected_storage))
67 << "invalid dual_values_filter";
68 RETURN_IF_ERROR(reduced_costs_filter.CheckModelStorage(expected_storage))
69 << "invalid reduced_costs_filter";
70 for (const auto [var, unused] : branching_priorities) {
71 RETURN_IF_ERROR(internal::CheckModelStorage(
72 /*storage=*/var.storage(),
73 /*expected_storage=*/expected_storage))
74 << "invalid variable " << var << " in branching_priorities";
75 }
76 for (const auto& [objective, params] : objective_parameters) {
77 RETURN_IF_ERROR(internal::CheckModelStorage(
78 /*storage=*/objective.storage(),
79 /*expected_storage=*/expected_storage))
80 << "invalid objective " << objective << " in objective_parameters";
81 }
82 for (const LinearConstraint lazy_linear_constraint :
83 lazy_linear_constraints) {
84 RETURN_IF_ERROR(internal::CheckModelStorage(
85 /*storage=*/lazy_linear_constraint.storage(),
86 /*expected_storage=*/expected_storage))
87 << "invalid LinearConstraint " << lazy_linear_constraint
88 << " in lazy_linear_constraints";
89 }
90 return absl::OkStatus();
91}

93absl::Status ModelSolveParameters::SolutionHint::CheckModelStorage(
94 const ModelStorage* expected_storage) const {
95 for (const auto& [v, _] : variable_values) {
96 RETURN_IF_ERROR(internal::CheckModelStorage(
97 /*storage=*/v.storage(),
98 /*expected_storage=*/expected_storage))
99 << "invalid variable " << v << " in variable_values";
100 }
101 for (const auto& [c, _] : dual_values) {
102 RETURN_IF_ERROR(internal::CheckModelStorage(
103 /*storage=*/c.storage(),
104 /*expected_storage=*/expected_storage))
105 << "invalid constraint " << c << " in dual_values";
106 }
107 return absl::OkStatus();
108}

117absl::StatusOr<ModelSolveParameters::SolutionHint>
118ModelSolveParameters::SolutionHint::FromProto(
119 const Model& model, const SolutionHintProto& hint_proto) {
120 OR_ASSIGN_OR_RETURN3(
121 VariableMap<double> variable_values,
122 VariableValuesFromProto(model.storage(), hint_proto.variable_values()),
123 _ << "failed to parse SolutionHintProto.variable_values");
124 OR_ASSIGN_OR_RETURN3(LinearConstraintMap<double> dual_values,
125 LinearConstraintValuesFromProto(
126 model.storage(), hint_proto.dual_values()),
127 _ << "failed to parse SolutionHintProto.dual_values");
128 return SolutionHint{
129 .variable_values = std::move(variable_values),
130 .dual_values = std::move(dual_values),
131 };
132}

149ModelSolveParametersProto ModelSolveParameters::Proto() const {
150 ModelSolveParametersProto ret;
151 *ret.mutable_variable_values_filter() = variable_values_filter.Proto();
152 *ret.mutable_dual_values_filter() = dual_values_filter.Proto();
153 *ret.mutable_reduced_costs_filter() = reduced_costs_filter.Proto();
154
155 // TODO(b/183616124): consolidate code. Probably best to add an
156 // export_to_proto to IdMap
157 if (initial_basis) {
158 RepeatedField<int64_t>& constraint_status_ids =
159 *ret.mutable_initial_basis()
160 ->mutable_constraint_status()
161 ->mutable_ids();
162 RepeatedField<int>& constraint_status_values =
163 *ret.mutable_initial_basis()
164 ->mutable_constraint_status()
165 ->mutable_values();
166 constraint_status_ids.Reserve(
167 static_cast<int>(initial_basis->constraint_status.size()));
168 constraint_status_values.Reserve(
169 static_cast<int>(initial_basis->constraint_status.size()));
170 for (const LinearConstraint& key :
171 SortedKeys(initial_basis->constraint_status)) {
172 constraint_status_ids.Add(key.id());
173 constraint_status_values.Add(
174 EnumToProto(initial_basis->constraint_status.at(key)));
175 }
176 RepeatedField<int64_t>& variable_status_ids =
177 *ret.mutable_initial_basis()->mutable_variable_status()->mutable_ids();
178 RepeatedField<int>& variable_status_values =
179 *ret.mutable_initial_basis()
180 ->mutable_variable_status()
181 ->mutable_values();
182 variable_status_ids.Reserve(
183 static_cast<int>(initial_basis->variable_status.size()));
184 variable_status_values.Reserve(
185 static_cast<int>(initial_basis->variable_status.size()));
186 for (const Variable& key : SortedKeys(initial_basis->variable_status)) {
187 variable_status_ids.Add(key.id());
188 variable_status_values.Add(
189 EnumToProto(initial_basis->variable_status.at(key)));
190 }
191 }
192 for (const SolutionHint& solution_hint : solution_hints) {
193 *ret.add_solution_hints() = solution_hint.Proto();
194 }
195 if (!branching_priorities.empty()) {
196 RepeatedField<int64_t>& variable_ids =
197 *ret.mutable_branching_priorities()->mutable_ids();
198 RepeatedField<int32_t>& variable_values =
199 *ret.mutable_branching_priorities()->mutable_values();
200 variable_ids.Reserve(static_cast<int>(branching_priorities.size()));
201 variable_values.Reserve(static_cast<int>(branching_priorities.size()));
202 for (const Variable& key : SortedKeys(branching_priorities)) {
203 variable_ids.Add(key.id());
204 variable_values.Add(branching_priorities.at(key));
205 }
206 }
207 for (const auto& [objective, params] : objective_parameters) {
208 if (objective.id()) {
209 (*ret.mutable_auxiliary_objective_parameters())[*objective.id()] =
210 params.Proto();
211 } else {
212 *ret.mutable_primary_objective_parameters() = params.Proto();
213 }
214 }
215 if (!lazy_linear_constraints.empty()) {
216 RepeatedField<int64_t>& lazy_linear_constraint_ids =
217 *ret.mutable_lazy_linear_constraint_ids();
218 lazy_linear_constraint_ids.Reserve(
219 static_cast<int>(lazy_linear_constraints.size()));
220 for (const LinearConstraint lazy_linear_constraint :
221 lazy_linear_constraints) {
222 lazy_linear_constraint_ids.Add(lazy_linear_constraint.id());
223 }
224 absl::c_sort(lazy_linear_constraint_ids);
225 }
226 return ret;
227}
