51absl::Status ModelSolveParameters::CheckModelStorage(
52 const ModelStorage* const expected_storage) const {
53 for (const SolutionHint& hint : solution_hints) {
54 RETURN_IF_ERROR(hint.CheckModelStorage(expected_storage))
55 << "invalid hint in solution_hints";
56 }
57 if (initial_basis.has_value()) {
58 RETURN_IF_ERROR(initial_basis->CheckModelStorage(expected_storage))
59 << "invalid initial_basis";
60 }
61 RETURN_IF_ERROR(variable_values_filter.CheckModelStorage(expected_storage))
62 << "invalid variable_values_filter";
63 RETURN_IF_ERROR(dual_values_filter.CheckModelStorage(expected_storage))
64 << "invalid dual_values_filter";
65 RETURN_IF_ERROR(reduced_costs_filter.CheckModelStorage(expected_storage))
66 << "invalid reduced_costs_filter";
67 return absl::OkStatus();
68}

70absl::Status ModelSolveParameters::SolutionHint::CheckModelStorage(
71 const ModelStorage* expected_storage) const {
72 for (const auto& [v, _] : variable_values) {
73 RETURN_IF_ERROR(internal::CheckModelStorage(
74 /*storage=*/v.storage(),
75 /*expected_storage=*/expected_storage))
76 << "invalid variable " << v << " in variable_values";
77 }
78 for (const auto& [c, _] : dual_values) {
79 RETURN_IF_ERROR(internal::CheckModelStorage(
80 /*storage=*/c.storage(),
81 /*expected_storage=*/expected_storage))
82 << "invalid constraint " << c << " in dual_values";
83 }
84 return absl::OkStatus();
85}

94absl::StatusOr<ModelSolveParameters::SolutionHint>
95ModelSolveParameters::SolutionHint::FromProto(
96 const Model& model, const SolutionHintProto& hint_proto) {
97 OR_ASSIGN_OR_RETURN3(
98 VariableMap<double> variable_values,
99 VariableValuesFromProto(model.storage(), hint_proto.variable_values()),
100 _ << "failed to parse SolutionHintProto.variable_values");
101 OR_ASSIGN_OR_RETURN3(LinearConstraintMap<double> dual_values,
102 LinearConstraintValuesFromProto(
103 model.storage(), hint_proto.dual_values()),
104 _ << "failed to parse SolutionHintProto.dual_values");
105 return SolutionHint{
106 .variable_values = std::move(variable_values),
107 .dual_values = std::move(dual_values),
108 };
109}

111ModelSolveParametersProto ModelSolveParameters::Proto() const {
112 ModelSolveParametersProto ret;
113 *ret.mutable_variable_values_filter() = variable_values_filter.Proto();
114 *ret.mutable_dual_values_filter() = dual_values_filter.Proto();
115 *ret.mutable_reduced_costs_filter() = reduced_costs_filter.Proto();
116
117 // TODO(b/183616124): consolidate code. Probably best to add an
118 // export_to_proto to IdMap
119 if (initial_basis) {
120 RepeatedField<int64_t>* const constraint_status_ids =
121 ret.mutable_initial_basis()->mutable_constraint_status()->mutable_ids();
122 RepeatedField<int>* const constraint_status_values =
123 ret.mutable_initial_basis()
124 ->mutable_constraint_status()
125 ->mutable_values();
126 constraint_status_ids->Reserve(initial_basis->constraint_status.size());
127 constraint_status_values->Reserve(initial_basis->constraint_status.size());
128 for (const LinearConstraint& key :
129 SortedKeys(initial_basis->constraint_status)) {
130 constraint_status_ids->Add(key.id());
131 constraint_status_values->Add(
132 EnumToProto(initial_basis->constraint_status.at(key)));
133 }
134 RepeatedField<int64_t>* const variable_status_ids =
135 ret.mutable_initial_basis()->mutable_variable_status()->mutable_ids();
136 RepeatedField<int>* const variable_status_values =
137 ret.mutable_initial_basis()
138 ->mutable_variable_status()
139 ->mutable_values();
140 variable_status_ids->Reserve(initial_basis->variable_status.size());
141 variable_status_values->Reserve(initial_basis->variable_status.size());
142 for (const Variable& key : SortedKeys(initial_basis->variable_status)) {
143 variable_status_ids->Add(key.id());
144 variable_status_values->Add(
145 EnumToProto(initial_basis->variable_status.at(key)));
146 }
147 }
148 for (const SolutionHint& solution_hint : solution_hints) {
149 *ret.add_solution_hints() = solution_hint.Proto();
150 }
151 if (!branching_priorities.empty()) {
152 RepeatedField<int64_t>* const variable_ids =
153 ret.mutable_branching_priorities()->mutable_ids();
154 RepeatedField<int32_t>* const variable_values =
155 ret.mutable_branching_priorities()->mutable_values();
156 variable_ids->Reserve(branching_priorities.size());
157 variable_values->Reserve(branching_priorities.size());
158 for (const Variable& key : SortedKeys(branching_priorities)) {
159 variable_ids->Add(key.id());
160 variable_values->Add(branching_priorities.at(key));
161 }
162 }
163 return ret;
164}
