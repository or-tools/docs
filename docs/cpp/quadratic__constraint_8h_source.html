39// A value type that references a quadratic constraint from ModelStorage.
40// Usually this type is passed by copy.
41class QuadraticConstraint {
42 public:
43 // The typed integer used for ids.
44 using IdType = QuadraticConstraintId;
45
46 inline QuadraticConstraint(const ModelStorage* storage,
47 QuadraticConstraintId id);
48
49 inline int64_t id() const;
50
51 inline QuadraticConstraintId typed_id() const;
52 inline const ModelStorage* storage() const;
53
54 inline double lower_bound() const;
55 inline double upper_bound() const;
56 inline absl::string_view name() const;
57
58 inline bool is_linear_coefficient_nonzero(Variable variable) const;
59 inline bool is_quadratic_coefficient_nonzero(Variable first_variable,
60 Variable second_variable) const;
61
62 // Returns 0.0 if the variable does not appear in the linear part of the
63 // constraint.
64 inline double linear_coefficient(Variable variable) const;
65 // Returns 0.0 if the variable does not appear in the quadratic part of the
66 // constraint.
67 inline double quadratic_coefficient(Variable first_variable,
68 Variable second_variable) const;
69
70 // All variables that appear in the quadratic constraint with a nonzero
71 // coefficient: in the linear terms, the quadratic terms, or both. Order is
72 // not defined.
73 inline std::vector<Variable> NonzeroVariables() const;
74
75 // Returns the constraints as a bounded quadratic expression.
76 //
77 // The quadratic expression will have a zero offset, even if the constraint
78 // was created with a non-zero one. For example:
79 //
80 // const LinearConstraint c =
81 // model.AddQuadraticConstraint(3.2 <= x*x + 1.0 <= 4.2);
82 //
83 // // Here `e` will contain 3.2 - 1.0 <= x*x <= 4.2 - 1.0.
84 // const BoundedQuadraticExpression e = c.AsBoundedQuadraticExpression();
85 BoundedQuadraticExpression AsBoundedQuadraticExpression() const;
86
87 // Returns a detailed string description of the contents of the constraint
88 // (not its name, use `<<` for that instead).
89 inline std::string ToString() const;
90
91 friend inline bool operator==(const QuadraticConstraint& lhs,
92 const QuadraticConstraint& rhs);
93 friend inline bool operator!=(const QuadraticConstraint& lhs,
94 const QuadraticConstraint& rhs);
95 template <typename H>
96 friend H AbslHashValue(H h, const QuadraticConstraint& quadratic_constraint);
97 friend std::ostream& operator<<(
98 std::ostream& ostr, const QuadraticConstraint& quadratic_constraint);
99
100 private:
101 const ModelStorage* storage_;
102 QuadraticConstraintId id_;
103};

135bool QuadraticConstraint::is_linear_coefficient_nonzero(
136 const Variable variable) const {
137 return linear_coefficient(variable) != 0.0;
138}
139
140bool QuadraticConstraint::is_quadratic_coefficient_nonzero(
141 const Variable first_variable, const Variable second_variable) const {
142 return quadratic_coefficient(first_variable, second_variable) != 0.0;

145double QuadraticConstraint::linear_coefficient(const Variable variable) const {
146 CHECK_EQ(variable.storage(), storage_)
147 << internal::kObjectsFromOtherModelStorage;
148 return storage_->constraint_data(id_).linear_terms.get(variable.typed_id());
149}
150
151double QuadraticConstraint::quadratic_coefficient(
152 const Variable first_variable, const Variable second_variable) const {
153 CHECK_EQ(first_variable.storage(), storage_)
154 << internal::kObjectsFromOtherModelStorage;
155 CHECK_EQ(second_variable.storage(), storage_)
156 << internal::kObjectsFromOtherModelStorage;
157 return storage_->constraint_data(id_).quadratic_terms.get(
158 first_variable.typed_id(), second_variable.typed_id());
159}
160
161std::vector<Variable> QuadraticConstraint::NonzeroVariables() const {
162 return AtomicConstraintNonzeroVariables(*storage_, id_);
163}
164
165std::string QuadraticConstraint::ToString() const {
166 if (!storage()->has_constraint(id_)) {
167 return std::string(kDeletedConstraintDefaultDescription);
168 }
169 std::stringstream str;
170 str << AsBoundedQuadraticExpression();
171 return str.str();
172}
173
174bool operator==(const QuadraticConstraint& lhs,
175 const QuadraticConstraint& rhs) {
176 return lhs.id_ == rhs.id_ && lhs.storage_ == rhs.storage_;
177}
178
179bool operator!=(const QuadraticConstraint& lhs,
180 const QuadraticConstraint& rhs) {
181 return !(lhs == rhs);
182}

184template <typename H>
185H AbslHashValue(H h, const QuadraticConstraint& quadratic_constraint) {
186 return H::combine(std::move(h), quadratic_constraint.id_.value(),
187 quadratic_constraint.storage_);
188}
