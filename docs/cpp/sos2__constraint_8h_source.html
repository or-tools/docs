113bool operator==(const Sos2Constraint& lhs, const Sos2Constraint& rhs) {
114 return lhs.id_ == rhs.id_ && lhs.storage_ == rhs.storage_;
115}

117bool operator!=(const Sos2Constraint& lhs, const Sos2Constraint& rhs) {

121template <typename H>
