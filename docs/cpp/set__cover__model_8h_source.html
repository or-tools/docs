24// Representation class for the weighted set-covering problem.
25//
26// Let S be a set, let (T_j) be a family (j in J) of subsets of S, and c_j costs
27// associated to each T_j.
28//
29// The minimum-cost set-covering problem consists in finding K, a subset of J
30// such that the union of all the T_j for k in K is equal to S (the subsets
31// indexed by K "cover" S), with the minimal total cost sum c_k (k in K).
32//
33// In Mixed-Integer Programming and matrix terms, the goal is to find values
34// of binary variables x_j, where x_j is 1 when subset T_j is in K, 0
35// otherwise, that minimize the sum of c_j * x_j subject to M.x >= 1. Each row
36// corresponds to an element in S.
37//
38// The matrix M for linear constraints is defined as follows:
39// - it has as many rows as there are elements in S.
40// - its columns are such that M(i, j) = 1 iff the i-th element of S is present
41// in T_j.
42

72// Main class for describing a weighted set-covering problem.
73class SetCoverModel {
74 public:
75 // Constructs an empty weighted set-covering problem.
76 SetCoverModel()
77 : num_elements_(0),
78 row_view_is_valid_(false),
79 subset_costs_(),
80 columns_(),
81 rows_(),
82 all_subsets_() {}
83
84 // Current number of elements to be covered in the model, i.e. the number of
85 // elements in S. In matrix terms, this is the number of rows.
86 ElementIndex num_elements() const { return num_elements_; }
87
88 // Current number of subsets in the model. In matrix terms, this is the
89 // number of columns.
90 SubsetIndex num_subsets() const { return columns_.size(); }
91
92 // Vector of costs for each subset.
93 const SubsetCostVector& subset_costs() const { return subset_costs_; }
94
95 // Column view of the set covering problem.
96 const SparseColumnView& columns() const { return columns_; }
97
98 // Row view of the set covering problem.
99 const SparseRowView& rows() const {
100 DCHECK(row_view_is_valid_);
101 return rows_;
102 }
103
104 // Returns true if rows_ and columns_ represent the same problem.
105 bool row_view_is_valid() const { return row_view_is_valid_; }
106
107 // Returns the list of indices for all the subsets in the model.
108 std::vector<SubsetIndex> all_subsets() const { return all_subsets_; }
109
110 // Adds an empty subset with a cost to the problem. In matrix terms, this
111 // adds a column to the matrix.
112 void AddEmptySubset(Cost cost);
113
114 // Adds an element to the last subset created. In matrix terms, this adds a
115 // 1 on row 'element' of the current last column of the matrix.
116 void AddElementToLastSubset(int element);
117 void AddElementToLastSubset(ElementIndex element);
118
119 // Sets 'cost' to an already existing 'subset'.
120 // This will CHECK-fail if cost is infinite or a NaN.
121 void SetSubsetCost(int subset, Cost cost);
122
123 // Adds 'element' to and already existing 'subset'.
124 void AddElementToSubset(int element, int subset);
125
126 // Creates the sparse ("dual") representation of the problem.
127 void CreateSparseRowView();
128
129 // Returns true if the problem is feasible, i.e. if the subsets cover all
130 // the elements.
131 bool ComputeFeasibility() const;
132
133 // Reserves num_subsets columns in the model.
134 void ReserveNumSubsets(int num_subsets);
135
136 // Reserves num_elements rows in the column indexed by subset.
137 void ReserveNumElementsInSubset(int num_elements, int subset);
138
139 // Returns the model as a SetCoverProto. The function is not const because
140 // the element indices in the columns need to be sorted for the representation
141 // as a protobuf to be canonical.
142 SetCoverProto ExportModelAsProto();
143
144 // Imports the model from a SetCoverProto.
145 void ImportModelFromProto(const SetCoverProto& message);
146
147 private:
148 // Updates the all_subsets_ vector so that it always contains 0 to
149 // columns.size() - 1
150 void UpdateAllSubsetsList();
151
152 // Number of elements.
153 ElementIndex num_elements_;
154
155 // True when the SparseRowView is up-to-date.
156 bool row_view_is_valid_;
157
158 // Costs for each subset.
159 SubsetCostVector subset_costs_;
160
161 // Vector of columns. Each column corresponds to a subset and contains the
162 // elements of the given subset.
163 SparseColumnView columns_;
164
165 // Vector of rows. Each row corresponds to an element and contains the
166 // subsets containing the element.
167 SparseRowView rows_;
168
169 // Vector of indices from 0 to columns.size() - 1. (Like std::iota, but built
170 // incrementally.) Used to (un)focus optimization algorithms on the complete
171 // problem.
172 // TODO(user): use this to enable deletion and recycling of columns/subsets.
173 std::vector<SubsetIndex> all_subsets_;
174};
