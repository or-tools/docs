33// Representation class for the weighted set-covering problem.
34//
35// Let S be a set, let (T_j) be a family (j in J) of subsets of S, and c_j costs
36// associated to each T_j.
37//
38// The minimum-cost set-covering problem consists in finding K, a subset of J
39// such that the union of all the T_j for k in K is equal to S (the subsets
40// indexed by K "cover" S), with the minimal total cost sum c_k (k in K).
41//
42// In Mixed-Integer Programming and matrix terms, the goal is to find values
43// of binary variables x_j, where x_j is 1 when subset T_j is in K, 0
44// otherwise, that minimize the sum of c_j * x_j subject to M.x >= 1. Each row
45// corresponds to an element in S.
46//
47// The matrix M for linear constraints is defined as follows:
48// - it has as many rows as there are elements in S.
49// - its columns are such that M(i, j) = 1 iff the i-th element of S is present
50// in T_j.
51

81// Main class for describing a weighted set-covering problem.
82class SetCoverModel {
83 public:
84 // Constructs an empty weighted set-covering problem.
85 SetCoverModel()
86 : num_elements_(0),
87 num_nonzeros_(0),
88 row_view_is_valid_(false),
89 subset_costs_(),
90 columns_(),
91 rows_(),
92 all_subsets_() {}
93
94 // Current number of elements to be covered in the model, i.e. the number of
95 // elements in S. In matrix terms, this is the number of rows.
96 ElementIndex num_elements() const { return num_elements_; }
97
98 // Current number of subsets in the model. In matrix terms, this is the
99 // number of columns.
100 SubsetIndex num_subsets() const { return columns_.size(); }
101
102 // Current number of nonzeros in the matrix.
103 ssize_t num_nonzeros() const { return num_nonzeros_; }
104
105 double FillRate() const {
106 return static_cast<double>(num_nonzeros_) /
107 (static_cast<double>(num_elements().value()) *
108 static_cast<double>(num_subsets().value()));
109 }
110
111 // Vector of costs for each subset.
112 const SubsetCostVector& subset_costs() const { return subset_costs_; }
113
114 // Column view of the set covering problem.
115 const SparseColumnView& columns() const { return columns_; }
116
117 // Row view of the set covering problem.
118 const SparseRowView& rows() const {
119 DCHECK(row_view_is_valid_);
120 return rows_;
121 }
122
123 // Returns true if rows_ and columns_ represent the same problem.
124 bool row_view_is_valid() const { return row_view_is_valid_; }
125
126 // Returns the list of indices for all the subsets in the model.
127 std::vector<SubsetIndex> all_subsets() const { return all_subsets_; }
128
129 // Adds an empty subset with a cost to the problem. In matrix terms, this
130 // adds a column to the matrix.
131 void AddEmptySubset(Cost cost);
132
133 // Adds an element to the last subset created. In matrix terms, this adds a
134 // 1 on row 'element' of the current last column of the matrix.
135 void AddElementToLastSubset(int element);
136 void AddElementToLastSubset(ElementIndex element);
137
138 // Sets 'cost' to an already existing 'subset'.
139 // This will CHECK-fail if cost is infinite or a NaN.
140 void SetSubsetCost(int subset, Cost cost);
141
142 // Adds 'element' to an already existing 'subset'.
143 // No check is done if element is already in the subset.
144 void AddElementToSubset(int element, int subset);
145
146 // Creates the sparse ("dual") representation of the problem.
147 void CreateSparseRowView();
148
149 // Returns true if the problem is feasible, i.e. if the subsets cover all
150 // the elements.
151 bool ComputeFeasibility() const;
152
153 // Reserves num_subsets columns in the model.
154 void ReserveNumSubsets(int num_subsets);
155
156 // Reserves num_elements rows in the column indexed by subset.
157 void ReserveNumElementsInSubset(int num_elements, int subset);
158
159 // Returns the model as a SetCoverProto. The function is not const because
160 // the element indices in the columns need to be sorted for the representation
161 // as a protobuf to be canonical.
162 SetCoverProto ExportModelAsProto();
163
164 // Imports the model from a SetCoverProto.
165 void ImportModelFromProto(const SetCoverProto& message);
166
167 // A struct enabling to show basic statistics on rows and columns.
168 // The meaning of the fields is obvious.
169 struct Stats {
170 double min;
171 double max;
172 double median;
173 double mean;
174 double stddev;
175
176 std::string DebugString() const {
177 return absl::StrCat("min = ", min, ", max = ", max, ", mean = ", mean,
178 ", median = ", median, ", stddev = ", stddev, ", ");
179 }
180 };
181
182 // Computes basic statistics on rows and returns a Stats structure.
183 Stats ComputeRowStats();
184
185 // Computes basic statistics on columns and returns a Stats structure.
186 Stats ComputeColumnStats();
187
188 // Computes deciles on rows and returns a vector of deciles.
189 std::vector<ssize_t> ComputeRowDeciles() const;
190
191 // Computes deciles on columns and returns a vector of deciles.
192 std::vector<ssize_t> ComputeColumnDeciles() const;
193
194 private:
195 // Updates the all_subsets_ vector so that it always contains 0 to
196 // columns.size() - 1
197 void UpdateAllSubsetsList();
198
199 // Number of elements.
200 ElementIndex num_elements_;
201
202 // Number of nonzeros in the matrix.
203 ssize_t num_nonzeros_;
204
205 // True when the SparseRowView is up-to-date.
206 bool row_view_is_valid_;
207
208 // Costs for each subset.
209 SubsetCostVector subset_costs_;
210
211 // Vector of columns. Each column corresponds to a subset and contains the
212 // elements of the given subset.
213 SparseColumnView columns_;
214
215 // Vector of rows. Each row corresponds to an element and contains the
216 // subsets containing the element.
217 SparseRowView rows_;
218
219 // Vector of indices from 0 to columns.size() - 1. (Like std::iota, but built
220 // incrementally.) Used to (un)focus optimization algorithms on the complete
221 // problem.
222 // TODO(user): use this to enable deletion and recycling of columns/subsets.
223 std::vector<SubsetIndex> all_subsets_;
224};
