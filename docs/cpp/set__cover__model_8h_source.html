19// Representation class for the weighted set-covering problem.
20//
21// Let S be a set, let (T_j) be a family (j in J) of subsets of S, and c_j costs
22// associated to each T_j.
23//
24// The minimum-cost set-covering problem consists in finding K, a subset of J
25// such that the union of all the T_j for k in K is equal to S (the subsets
26// indexed by K "cover" S), with the minimal total cost sum c_k (k in K).
27//
28// In Mixed-Integer Programming and matrix terms, the goal is to find values
29// of binary variables x_j, where x_j is 1 when subset T_j is in K, 0
30// otherwise, that minimize the sum of c_j * x_j subject to M.x >= 1. Each row
31// corresponds to an element in S.
32//
33// The matrix M for linear constraints is defined as follows:
34// - it has as many rows as there are elements in S.
35// - its columns are such that M(i, j) = 1 iff the i-th element of S is present
36// in T_j.
37

66// Main class for describing a weighted set-covering problem.
67class SetCoverModel {
68 public:
69 // Constructs an empty weighted set-covering problem.
70 SetCoverModel()
71 : num_elements_(0),
72 row_view_is_valid_(false),
73 subset_costs_(),
74 columns_(),
75 rows_() {}
76
77 // Current number of elements to be covered in the model, i.e. the number of
78 // elements in S. In matrix terms, this is the number of rows.
79 ElementIndex num_elements() const { return num_elements_; }
80
81 // Current number of subsets in the model. In matrix terms, this is the
82 // number of columns.
83 SubsetIndex num_subsets() const { return columns_.size(); }
84
85 const SubsetCostVector& subset_costs() { return subset_costs_; }
86
87 const SparseColumnView& columns() { return columns_; }
88
89 const SparseRowView& rows() {
90 DCHECK(row_view_is_valid_);
91 return rows_;
92 }
93
94 // Returns true if rows_ and columns_ represent the same problem.
95 bool row_view_is_valid() const { return row_view_is_valid_; }
96
97 // Adds an empty subset with a cost to the problem. In matrix terms, this
98 // adds a column to the matrix.
99 void AddEmptySubset(Cost cost);
100
101 // Adds an element to the last subset created. In matrix terms, this adds a
102 // 1 on row 'element' of the current last column of the matrix.
103 void AddElementToLastSubset(int element);
104 void AddElementToLastSubset(ElementIndex element);
105
106 // Sets 'cost' to an already existing 'subset'.
107 void SetSubsetCost(int subset, Cost cost);
108
109 // Adds 'element' to and already existing 'subset'.
110 void AddElementToSubset(int element, int subset);
111
112 // Creates the sparse ("dual") representation of the problem.
113 void CreateSparseRowView();
114
115 // Returns true if the problem is feasible, i.e. if the subsets cover all
116 // the elements.
117 bool ComputeFeasibility() const;
118
119 // Reserves num_subsets columns in the model.
120 void ReserveNumSubsets(int num_subsets);
121
122 // Reserves num_elements rows in the column indexed by subset.
123 void ReserveNumElementsInSubset(int num_elements, int subset);
124
125 private:
126 // Number of elements.
127 ElementIndex num_elements_;
128
129 // True when the SparseRowView is up-to-date.
130 bool row_view_is_valid_;
131
132 // Costs for each subset.
133 SubsetCostVector subset_costs_;
134
135 // Vector of columns. Each column corresponds to a subset and contains the
136 // elements of the given subset.
137 SparseColumnView columns_;
138
139 // Vector of rows. Each row corresponds to an element and contains the
140 // subsets containing the element.
141 SparseRowView rows_;
142};
