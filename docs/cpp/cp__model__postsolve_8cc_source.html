30// This postsolve is "special". If the clause is not satisfied, we fix the
31// first literal in the clause to true (even if it was fixed to false). This
32// allows to handle more complex presolve operations used by the SAT presolver.
33//
34// Also, any "free" Boolean should be fixed to some value for the subsequent
35// postsolve steps.
36void PostsolveClause(const ConstraintProto& ct, std::vector<Domain>* domains) {
37 const int size = ct.bool_or().literals_size();
38 CHECK_NE(size, 0);
39 bool satisfied = false;
40 for (int i = 0; i < size; ++i) {
41 const int ref = ct.bool_or().literals(i);
42 const int var = PositiveRef(ref);
43 if ((*domains)[var].IsFixed()) {
44 if ((*domains)[var].FixedValue() == (RefIsPositive(ref) ? 1 : 0)) {
45 satisfied = true;
46 }
47 } else {
48 // We still need to assign free variable. Any value should work.
49 (*domains)[PositiveRef(ref)] = Domain(0);
50 }
51 }
52 if (satisfied) return;
53
54 // Change the value of the first variable (which was chosen at presolve).
55 const int first_ref = ct.bool_or().literals(0);
56 (*domains)[PositiveRef(first_ref)] = Domain(RefIsPositive(first_ref) ? 1 : 0);
57}

59void PostsolveExactlyOne(const ConstraintProto& ct,
60 std::vector<Domain>* domains) {
61 bool satisfied = false;
62 std::vector<int> free_variables;
63 for (const int ref : ct.exactly_one().literals()) {
64 const int var = PositiveRef(ref);
65 if ((*domains)[var].IsFixed()) {
66 if ((*domains)[var].FixedValue() == (RefIsPositive(ref) ? 1 : 0)) {
67 CHECK(!satisfied) << "Two variables at one in exactly one.";
68 satisfied = true;
69 }
70 } else {
71 free_variables.push_back(ref);
72 }
73 }
74 if (!satisfied) {
75 // Fix one at true.
76 CHECK(!free_variables.empty()) << "All zero in exactly one";
77 const int ref = free_variables.back();
78 (*domains)[PositiveRef(ref)] = Domain(RefIsPositive(ref) ? 1 : 0);
79 free_variables.pop_back();
80 }
81
82 // Fix any free variable left at false.
83 for (const int ref : free_variables) {
84 (*domains)[PositiveRef(ref)] = Domain(RefIsPositive(ref) ? 0 : 1);
85 }
86}

88// For now we set the first unset enforcement literal to false.
89// There must be one.
90void SetEnforcementLiteralToFalse(const ConstraintProto& ct,
91 std::vector<Domain>* domains) {
92 CHECK(!ct.enforcement_literal().empty());
93 bool has_free_enforcement_literal = false;
94 for (const int enf : ct.enforcement_literal()) {
95 if ((*domains)[PositiveRef(enf)].IsFixed()) continue;
96 has_free_enforcement_literal = true;
97 if (RefIsPositive(enf)) {
98 (*domains)[enf] = Domain(0);
99 } else {
100 (*domains)[PositiveRef(enf)] = Domain(1);
101 }
102 break;
103 }
104 if (!has_free_enforcement_literal) {
105 LOG(FATAL)
106 << "Unsatisfied linear constraint with no free enforcement literal: "
107 << ct.ShortDebugString();
108 }
109}

111// Here we simply assign all non-fixed variable to a feasible value. Which
112// should always exists by construction.
113void PostsolveLinear(const ConstraintProto& ct, std::vector<Domain>* domains) {
114 int64_t fixed_activity = 0;
115 const int size = ct.linear().vars().size();
116 std::vector<int> free_vars;
117 std::vector<int64_t> free_coeffs;
118 for (int i = 0; i < size; ++i) {
119 const int var = ct.linear().vars(i);
120 const int64_t coeff = ct.linear().coeffs(i);
121 CHECK_LT(var, domains->size());
122 if (coeff == 0) continue;
123 if ((*domains)[var].IsFixed()) {
124 fixed_activity += (*domains)[var].FixedValue() * coeff;
125 } else {
126 free_vars.push_back(var);
127 free_coeffs.push_back(coeff);
128 }
129 }
130 if (free_vars.empty()) {
131 const Domain rhs = ReadDomainFromProto(ct.linear());
132 if (!rhs.Contains(fixed_activity)) {
133 SetEnforcementLiteralToFalse(ct, domains);
134 }
135 return;
136 }
137
138 // Fast track for the most common case.
139 const Domain initial_rhs = ReadDomainFromProto(ct.linear());
140 if (free_vars.size() == 1) {
141 const int var = free_vars[0];
142 const Domain domain = initial_rhs.AdditionWith(Domain(-fixed_activity))
143 .InverseMultiplicationBy(free_coeffs[0])
144 .IntersectionWith((*domains)[var]);
145 if (domain.IsEmpty()) {
146 SetEnforcementLiteralToFalse(ct, domains);
147 return;
148 }
149 (*domains)[var] = Domain(domain.SmallestValue());
150 return;
151 }
152
153 // The postsolve code is a bit involved if there is more than one free
154 // variable, we have to postsolve them one by one.
155 //
156 // Here we recompute the same domains as during the presolve. Everything is
157 // like if we where substiting the variable one by one:
158 // terms[i] + fixed_activity \in rhs_domains[i]
159 // In the reverse order.
160 std::vector<Domain> rhs_domains;
161 rhs_domains.push_back(initial_rhs);
162 for (int i = 0; i + 1 < free_vars.size(); ++i) {
163 // Note that these should be exactly the same computation as the one done
164 // during presolve and should be exact. However, we have some tests that do
165 // not comply, so we don't check exactness here. Also, as long as we don't
166 // get empty domain below, and the complexity of the domain do not explode
167 // here, we should be fine.
168 Domain term = (*domains)[free_vars[i]].MultiplicationBy(-free_coeffs[i]);
169 rhs_domains.push_back(term.AdditionWith(rhs_domains.back()));
170 }
171 for (int i = free_vars.size() - 1; i >= 0; --i) {
172 // Choose a value for free_vars[i] that fall into rhs_domains[i] -
173 // fixed_activity. This will crash if the intersection is empty, but it
174 // shouldn't be.
175 const int var = free_vars[i];
176 const int64_t coeff = free_coeffs[i];
177 const Domain domain = rhs_domains[i]
178 .AdditionWith(Domain(-fixed_activity))
179 .InverseMultiplicationBy(coeff)
180 .IntersectionWith((*domains)[var]);
181
182 // TODO(user): I am not 100% that the algo here might cover all the presolve
183 // case, so if this fail, it might indicate an issue here and not in the
184 // presolve/solver code.
185 CHECK(!domain.IsEmpty()) << ct.ShortDebugString();
186 const int64_t value = domain.SmallestValue();
187 (*domains)[var] = Domain(value);
188
189 fixed_activity += coeff * value;
190 }
191 DCHECK(initial_rhs.Contains(fixed_activity));
192}

196// Note that if a domain is not fixed, we just take its Min() value.
197int64_t EvaluateLinearExpression(const LinearExpressionProto& expr,
198 const std::vector<Domain>& domains) {
199 int64_t value = expr.offset();
200 for (int i = 0; i < expr.vars_size(); ++i) {
201 const int ref = expr.vars(i);
202 const int64_t increment =
203 domains[PositiveRef(expr.vars(i))].Min() * expr.coeffs(i);
204 value += RefIsPositive(ref) ? increment : -increment;
205 }
206 return value;
207}

211// Compute the max of each expression, and assign it to the target expr. We only
212// support post-solving the case where whatever the value of all expression,
213// there will be a valid target.
214void PostsolveLinMax(const ConstraintProto& ct, std::vector<Domain>* domains) {
215 int64_t max_value = std::numeric_limits<int64_t>::min();
216 for (const LinearExpressionProto& expr : ct.lin_max().exprs()) {
217 // In most case all expression are fixed, except in the corner case where
218 // one of the expression refer to the target itself !
219 max_value = std::max(max_value, EvaluateLinearExpression(expr, *domains));
220 }
221
222 const LinearExpressionProto& target = ct.lin_max().target();
223 CHECK_EQ(target.vars().size(), 1);
224 CHECK(RefIsPositive(target.vars(0)));
225
226 max_value -= target.offset();
227 CHECK_EQ(max_value % target.coeffs(0), 0);
228 max_value /= target.coeffs(0);
229 (*domains)[target.vars(0)] = Domain(max_value);
230}

232// We only support 3 cases in the presolve currently.
233void PostsolveElement(const ConstraintProto& ct, std::vector<Domain>* domains) {
234 const int index_ref = ct.element().index();
235 const int index_var = PositiveRef(index_ref);
236 const int target_ref = ct.element().target();
237 const int target_var = PositiveRef(target_ref);
238
239 // Deal with non-fixed target and non-fixed index. This only happen if
240 // whatever the value of the index and selected variable, we can choose a
241 // valid target, so we just fix the index to its min value in this case.
242 if (!(*domains)[target_var].IsFixed() && !(*domains)[index_var].IsFixed()) {
243 const int64_t index_var_value = (*domains)[index_var].Min();
244 (*domains)[index_var] = Domain(index_var_value);
245
246 // If the selected variable is not fixed, we also need to fix it.
247 const int selected_ref = ct.element().vars(
248 RefIsPositive(index_ref) ? index_var_value : -index_var_value);
249 const int selected_var = PositiveRef(selected_ref);
250 if (!(*domains)[selected_var].IsFixed()) {
251 (*domains)[selected_var] = Domain((*domains)[selected_var].Min());
252 }
253 }
254
255 // Deal with fixed index.
256 if ((*domains)[index_var].IsFixed()) {
257 const int64_t index_var_value = (*domains)[index_var].FixedValue();
258 const int selected_ref = ct.element().vars(
259 RefIsPositive(index_ref) ? index_var_value : -index_var_value);
260 const int selected_var = PositiveRef(selected_ref);
261 if ((*domains)[selected_var].IsFixed()) {
262 const int64_t selected_value = (*domains)[selected_var].FixedValue();
263 (*domains)[target_var] = (*domains)[target_var].IntersectionWith(
264 Domain(RefIsPositive(target_ref) == RefIsPositive(selected_ref)
265 ? selected_value
266 : -selected_value));
267 DCHECK(!(*domains)[target_var].IsEmpty());
268 } else {
269 const bool same_sign =
270 (selected_var == selected_ref) == (target_var == target_ref);
271 const Domain target_domain = (*domains)[target_var];
272 const Domain selected_domain = same_sign
273 ? (*domains)[selected_var]
274 : (*domains)[selected_var].Negation();
275 const Domain final = target_domain.IntersectionWith(selected_domain);
276 const int64_t value = final.SmallestValue();
277 (*domains)[target_var] =
278 (*domains)[target_var].IntersectionWith(Domain(value));
279 (*domains)[selected_var] = (*domains)[selected_var].IntersectionWith(
280 Domain(same_sign ? value : -value));
281 DCHECK(!(*domains)[target_var].IsEmpty());
282 DCHECK(!(*domains)[selected_var].IsEmpty());
283 }
284 return;
285 }
286
287 // Deal with fixed target (and constant vars).
288 const int64_t target_value = (*domains)[target_var].FixedValue();
289 int selected_index_value = -1;
290 for (const int64_t v : (*domains)[index_var].Values()) {
291 const int64_t i = index_var == index_ref ? v : -v;
292 if (i < 0 || i >= ct.element().vars_size()) continue;
293
294 const int ref = ct.element().vars(i);
295 const int var = PositiveRef(ref);
296 const int64_t value = (*domains)[var].FixedValue();
297 if (RefIsPositive(target_ref) == RefIsPositive(ref)) {
298 if (value == target_value) {
299 selected_index_value = i;
300 break;
301 }
302 } else {
303 if (value == -target_value) {
304 selected_index_value = i;
305 break;
306 }
307 }
308 }
309
310 CHECK_NE(selected_index_value, -1);
311 (*domains)[index_var] = (*domains)[index_var].IntersectionWith(Domain(
312 RefIsPositive(index_ref) ? selected_index_value : -selected_index_value));
313 DCHECK(!(*domains)[index_var].IsEmpty());
314}

316// We only support assigning to an affine target.
317void PostsolveIntMod(const ConstraintProto& ct, std::vector<Domain>* domains) {
318 const int64_t exp = EvaluateLinearExpression(ct.int_mod().exprs(0), *domains);
319 const int64_t mod = EvaluateLinearExpression(ct.int_mod().exprs(1), *domains);
320 CHECK_NE(mod, 0);
321 const int64_t target_value = exp % mod;
322
323 const LinearExpressionProto& target = ct.int_mod().target();
324 CHECK_EQ(target.vars().size(), 1);
325 const int64_t term_value = target_value - target.offset();
326 CHECK_EQ(term_value % target.coeffs(0), 0);
327 const int64_t value = term_value / target.coeffs(0);
328 CHECK((*domains)[target.vars(0)].Contains(value));
329 (*domains)[target.vars(0)] = Domain(value);
330}

332void PostsolveResponse(const int64_t num_variables_in_original_model,
333 const CpModelProto& mapping_proto,
334 const std::vector<int>& postsolve_mapping,
335 std::vector<int64_t>* solution) {
336 CHECK_EQ(solution->size(), postsolve_mapping.size());
337
338 // Read the initial variable domains, either from the fixed solution of the
339 // presolved problems or from the mapping model.
340 std::vector<Domain> domains(mapping_proto.variables_size());
341 for (int i = 0; i < postsolve_mapping.size(); ++i) {
342 CHECK_LE(postsolve_mapping[i], domains.size());
343 domains[postsolve_mapping[i]] = Domain((*solution)[i]);
344 }
345 for (int i = 0; i < domains.size(); ++i) {
346 if (domains[i].IsEmpty()) {
347 domains[i] = ReadDomainFromProto(mapping_proto.variables(i));
348 }
349 CHECK(!domains[i].IsEmpty());
350 }
351
352 // Process the constraints in reverse order.
353 const int num_constraints = mapping_proto.constraints_size();
354 for (int i = num_constraints - 1; i >= 0; i--) {
355 const ConstraintProto& ct = mapping_proto.constraints(i);
356
357 // We ignore constraint with an enforcement literal set to false. If the
358 // enforcement is still unclear, we still process this constraint.
359 bool constraint_can_be_ignored = false;
360 for (const int enf : ct.enforcement_literal()) {
361 const int var = PositiveRef(enf);
362 const bool is_false =
363 domains[var].IsFixed() &&
364 RefIsPositive(enf) == (domains[var].FixedValue() == 0);
365 if (is_false) {
366 constraint_can_be_ignored = true;
367 break;
368 }
369 }
370 if (constraint_can_be_ignored) continue;
371
372 switch (ct.constraint_case()) {
373 case ConstraintProto::kBoolOr:
374 PostsolveClause(ct, &domains);
375 break;
376 case ConstraintProto::kExactlyOne:
377 PostsolveExactlyOne(ct, &domains);
378 break;
379 case ConstraintProto::kLinear:
380 PostsolveLinear(ct, &domains);
381 break;
382 case ConstraintProto::kLinMax:
383 PostsolveLinMax(ct, &domains);
384 break;
385 case ConstraintProto::kElement:
386 PostsolveElement(ct, &domains);
387 break;
388 case ConstraintProto::kIntMod:
389 PostsolveIntMod(ct, &domains);
390 break;
391 default:
392 // This should never happen as we control what kind of constraint we
393 // add to the mapping_proto;
394 LOG(FATAL) << "Unsupported constraint: " << ct.ShortDebugString();
395 }
396 }
397
398 // Fill the response.
399 // Maybe fix some still unfixed variable.
400 solution->clear();
401 CHECK_LE(num_variables_in_original_model, domains.size());
402 for (int i = 0; i < num_variables_in_original_model; ++i) {
403 solution->push_back(domains[i].SmallestValue());
404 }
405}
