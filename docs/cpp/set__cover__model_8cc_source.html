28void SetCoverModel::UpdateAllSubsetsList() {
29 const SubsetIndex new_size = columns_.size();
30 const SubsetIndex old_size(all_subsets_.size());
31 DCHECK_LE(old_size, new_size);
32 all_subsets_.resize(new_size.value());
33 for (SubsetIndex subset(old_size); subset < new_size; ++subset) {
34 all_subsets_[subset.value()] = subset;

60void SetCoverModel::SetSubsetCost(int subset, Cost cost) {
61 CHECK(std::isfinite(cost));
62 DCHECK_GE(subset, 0);
63 const SubsetIndex subset_index(subset);
64 const SubsetIndex num_subsets = columns_.size();
65 const SubsetIndex new_size = std::max(num_subsets, subset_index + 1);
66 columns_.resize(new_size, SparseColumn());
67 subset_costs_.resize(new_size, 0.0);
68 subset_costs_[subset_index] = cost;
69 UpdateAllSubsetsList();
70 row_view_is_valid_ = false; // Probably overkill, but better safe than sorry.
71}

102void SetCoverModel::CreateSparseRowView() {
103 if (row_view_is_valid_) {
104 return;
105 }
106 rows_.resize(num_elements_, SparseRow());
107 glop::StrictITIVector<ElementIndex, int> row_sizes(num_elements_, 0);
108 for (SubsetIndex subset(0); subset < columns_.size(); ++subset) {
109 std::sort(columns_[subset].begin(), columns_[subset].end());
110 for (const ElementIndex element : columns_[subset]) {
111 ++row_sizes[element];
112 }
113 }
114 for (ElementIndex element(0); element < num_elements_; ++element) {
115 rows_[element].reserve(EntryIndex(row_sizes[element]));
116 }
117 for (SubsetIndex subset(0); subset < columns_.size(); ++subset) {
118 for (const ElementIndex element : columns_[subset]) {
119 rows_[element].push_back(subset);
120 }
121 }
122 row_view_is_valid_ = true;
123}

125bool SetCoverModel::ComputeFeasibility() const {
126 CHECK_GT(num_elements_, 0);
127 CHECK_GT(columns_.size(), 0);
128 CHECK_EQ(columns_.size(), subset_costs_.size());
129
130 ElementToSubsetVector coverage(num_elements_, SubsetIndex(0));
131 for (const Cost cost : subset_costs_) {
132 CHECK_GT(cost, 0.0);
133 }
134 for (const SparseColumn& column : columns_) {
135 CHECK_GT(column.size(), 0);
136 for (const ElementIndex element : column) {
137 ++coverage[element];
138 }
139 }
140 for (ElementIndex element(0); element < num_elements_; ++element) {
141 CHECK_GE(coverage[element], 0);
142 if (coverage[element] == 0) {
143 return false;
144 }
145 }
146 VLOG(1) << "Max possible coverage = "
147 << *std::max_element(coverage.begin(), coverage.end());
148 for (SubsetIndex subset(0); subset < columns_.size(); ++subset) {
149 CHECK_EQ(all_subsets_[subset.value()], subset) << "subset = " << subset;
150 }
151 return true;
152}

154SetCoverProto SetCoverModel::ExportModelAsProto() {
155 SetCoverProto message;
156 for (SubsetIndex subset(0); subset < columns_.size(); ++subset) {
157 SetCoverProto::Subset* subset_proto = message.add_subset();
158 subset_proto->set_cost(subset_costs_[subset]);
159 std::sort(columns_[subset].begin(), columns_[subset].end());
160 for (const ElementIndex element : columns_[subset]) {
161 subset_proto->add_element(element.value());
162 }
163 }
164 return message;
165}

167void SetCoverModel::ImportModelFromProto(const SetCoverProto& message) {
168 columns_.clear();
169 subset_costs_.clear();
170 ReserveNumSubsets(message.subset_size());
171 SubsetIndex subset_index(0);
172 for (const SetCoverProto::Subset& subset_proto : message.subset()) {
173 subset_costs_[SubsetIndex(subset_index)] = subset_proto.cost();
174 if (subset_proto.element_size() > 0) {
175 columns_[subset_index].reserve(EntryIndex(subset_proto.element_size()));
176 for (auto element : subset_proto.element()) {
177 columns_[subset_index].push_back(ElementIndex(element));
178 num_elements_ =
179 ElementIndex(std::max(num_elements_.value(), element + 1));
180 }
181 ++subset_index;
182 }
183 }
184 UpdateAllSubsetsList();
185 CreateSparseRowView();
186}

189double StandardDeviation(const std::vector<ssize_t>& v) {
190 const ssize_t size = v.size();
191 double n = 0.0; // n is used in a calculation involving doubles.
192 double sum_of_squares = 0.0;
193 double sum = 0.0;
194 for (ssize_t i = 0; i < size; ++i) {
195 double sample = static_cast<double>(v[i]);
196 if (sample == 0.0) continue;
197 sum_of_squares += sample * sample;
198 sum += sample;
199 ++n;
200 }
201 return n == 0.0 ? 0.0 : sqrt((sum_of_squares - sum * sum / n) / n);
202}

204template <typename T>
205SetCoverModel::Stats ComputeStats(std::vector<T> sizes) {
206 SetCoverModel::Stats stats;
207 stats.min = *std::min_element(sizes.begin(), sizes.end());
208 stats.max = *std::max_element(sizes.begin(), sizes.end());
209 stats.mean = std::accumulate(sizes.begin(), sizes.end(), 0.0) / sizes.size();
210 std::nth_element(sizes.begin(), sizes.begin() + sizes.size() / 2,
211 sizes.end());
212 stats.median = sizes[sizes.size() / 2];
213 stats.stddev = StandardDeviation(sizes);
214 return stats;
215}

217std::vector<ssize_t> ComputeDeciles(std::vector<ssize_t> values) {
218 const int kNumDeciles = 10;
219 std::vector<ssize_t> deciles;
220 deciles.reserve(kNumDeciles);
221 for (int i = 1; i <= kNumDeciles; ++i) {
222 std::nth_element(values.begin(),
223 values.begin() + values.size() * i / kNumDeciles - 1,
224 values.end());
225 deciles.push_back(values[values.size() * i / kNumDeciles - 1]);
226 }
227 return deciles;
228}
