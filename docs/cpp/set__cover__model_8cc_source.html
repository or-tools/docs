74void SetCoverModel::CreateSparseRowView() {
75 if (row_view_is_valid_) {
76 return;
77 }
78 rows_.resize(num_elements_, SparseRow());
79 glop::StrictITIVector<ElementIndex, int> row_sizes(num_elements_, 0);
80 for (SubsetIndex subset(0); subset < columns_.size(); ++subset) {
81 std::sort(columns_[subset].begin(), columns_[subset].end());
82 for (const ElementIndex element : columns_[subset]) {
83 ++row_sizes[element];
84 }
85 }
86 for (ElementIndex element(0); element < num_elements_; ++element) {
87 rows_[element].reserve(EntryIndex(row_sizes[element]));
88 }
89 for (SubsetIndex subset(0); subset < columns_.size(); ++subset) {
90 for (const ElementIndex element : columns_[subset]) {
91 rows_[element].push_back(subset);
92 }
93 }
94 row_view_is_valid_ = true;
95}

97bool SetCoverModel::ComputeFeasibility() const {
98 CHECK_GT(num_elements_, 0);
99 CHECK_GT(columns_.size(), 0);
100 CHECK_EQ(columns_.size(), subset_costs_.size());
101
102 ElementToSubsetVector coverage(num_elements_, SubsetIndex(0));
103 for (const Cost cost : subset_costs_) {
104 CHECK_GT(cost, 0.0);
105 }
106 for (const SparseColumn& column : columns_) {
107 CHECK_GT(column.size(), 0);
108 for (const ElementIndex element : column) {
109 ++coverage[element];
110 }
111 }
112 for (ElementIndex element(0); element < num_elements_; ++element) {
113 CHECK_GE(coverage[element], 0);
114 if (coverage[element] == 0) {
115 return false;
116 }
117 }
118 VLOG(1) << "Max possible coverage = "
119 << *std::max_element(coverage.begin(), coverage.end());
120 return true;
121}
