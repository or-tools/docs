24void SetCoverModel::UpdateAllSubsetsList() {
25 const SubsetIndex new_size = columns_.size();
26 const SubsetIndex old_size(all_subsets_.size());
27 DCHECK_LE(old_size, new_size);
28 all_subsets_.resize(new_size.value());
29 for (SubsetIndex subset(old_size); subset < new_size; ++subset) {

55void SetCoverModel::SetSubsetCost(int subset, Cost cost) {
56 DCHECK_GE(subset, 0);
57 const SubsetIndex subset_index(subset);
58 const SubsetIndex num_subsets = columns_.size();
59 const SubsetIndex new_size = std::max(num_subsets, subset_index + 1);
60 columns_.resize(new_size, SparseColumn());
61 subset_costs_.resize(new_size, 0.0);
62 subset_costs_[subset_index] = cost;
63 UpdateAllSubsetsList();
64 row_view_is_valid_ = false; // Probably overkill, but better safe than sorry.
65}

95void SetCoverModel::CreateSparseRowView() {
96 if (row_view_is_valid_) {
97 return;
98 }
99 rows_.resize(num_elements_, SparseRow());
100 glop::StrictITIVector<ElementIndex, int> row_sizes(num_elements_, 0);
101 for (SubsetIndex subset(0); subset < columns_.size(); ++subset) {
102 std::sort(columns_[subset].begin(), columns_[subset].end());
103 for (const ElementIndex element : columns_[subset]) {
104 ++row_sizes[element];
105 }
106 }
107 for (ElementIndex element(0); element < num_elements_; ++element) {
108 rows_[element].reserve(EntryIndex(row_sizes[element]));
109 }
110 for (SubsetIndex subset(0); subset < columns_.size(); ++subset) {
111 for (const ElementIndex element : columns_[subset]) {
112 rows_[element].push_back(subset);
113 }
114 }
115 row_view_is_valid_ = true;
116}

118bool SetCoverModel::ComputeFeasibility() const {
119 CHECK_GT(num_elements_, 0);
120 CHECK_GT(columns_.size(), 0);
121 CHECK_EQ(columns_.size(), subset_costs_.size());
122
123 ElementToSubsetVector coverage(num_elements_, SubsetIndex(0));
124 for (const Cost cost : subset_costs_) {
125 CHECK_GT(cost, 0.0);
126 }
127 for (const SparseColumn& column : columns_) {
128 CHECK_GT(column.size(), 0);
129 for (const ElementIndex element : column) {
130 ++coverage[element];
131 }
132 }
133 for (ElementIndex element(0); element < num_elements_; ++element) {
134 CHECK_GE(coverage[element], 0);
135 if (coverage[element] == 0) {
136 return false;
137 }
138 }
139 VLOG(1) << "Max possible coverage = "
140 << *std::max_element(coverage.begin(), coverage.end());
141 for (SubsetIndex subset(0); subset < columns_.size(); ++subset) {
142 CHECK_EQ(all_subsets_[subset.value()], subset) << "subset = " << subset;
143 }
144 return true;
145}
