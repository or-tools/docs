186void ModelBuilderHelper::SafeAddConstraintTerm(int ct_index, int var_index,
187 double coeff) {
188 if (coeff == 0.0) return;
189 MPConstraintProto* ct_proto = model_.mutable_constraint(ct_index);
190 for (int i = 0; i < ct_proto->var_index_size(); ++i) {
191 if (ct_proto->var_index(i) == var_index) {
192 ct_proto->set_coefficient(i, coeff + ct_proto->coefficient(i));
193 return;
194 }
195 }
196 // If we reach this point, the variable does not exist in the constraint yet,
197 // so we add it to the constraint as a new term.
198 ct_proto->add_var_index(var_index);
199 ct_proto->add_coefficient(coeff);
200}

440std::optional<MPSolutionResponse> ModelSolverHelper::SolveRequest(
441 const MPModelRequest& request) {
442 if (!MPSolver::SupportsProblemType(
443 static_cast<MPSolver::OptimizationProblemType>(
444 request.solver_type()))) {
445 return std::nullopt;
446 }
447 MPSolutionResponse temp;
448 MPSolver::SolveWithProto(request, &temp, &interrupt_solve_);
449 return temp;
450}

453SolveStatus MPSolverResponseStatusToSolveStatus(MPSolverResponseStatus s) {
454 switch (s) {
455 case MPSOLVER_OPTIMAL:
456 return SolveStatus::OPTIMAL;
457 case MPSOLVER_FEASIBLE:
458 return SolveStatus::FEASIBLE;
459 case MPSOLVER_INFEASIBLE:
460 return SolveStatus::INFEASIBLE;
461 case MPSOLVER_UNBOUNDED:
462 return SolveStatus::UNBOUNDED;
463 case MPSOLVER_ABNORMAL:
464 return SolveStatus::ABNORMAL;
465 case MPSOLVER_NOT_SOLVED:
466 return SolveStatus::NOT_SOLVED;
467 case MPSOLVER_MODEL_IS_VALID:
468 return SolveStatus::MODEL_IS_VALID;
469 case MPSOLVER_CANCELLED_BY_USER:
470 return SolveStatus::CANCELLED_BY_USER;
471 case MPSOLVER_UNKNOWN_STATUS:
472 return SolveStatus::UNKNOWN_STATUS;
473 case MPSOLVER_MODEL_INVALID:
474 return SolveStatus::MODEL_INVALID;
475 case MPSOLVER_MODEL_INVALID_SOLUTION_HINT:
476 return SolveStatus::MODEL_INVALID;
477 case MPSOLVER_MODEL_INVALID_SOLVER_PARAMETERS:
478 return SolveStatus::INVALID_SOLVER_PARAMETERS;
479 case MPSOLVER_SOLVER_TYPE_UNAVAILABLE:
480 return SolveStatus::SOLVER_TYPE_UNAVAILABLE;
481 case MPSOLVER_INCOMPATIBLE_OPTIONS:
482 return SolveStatus::INCOMPATIBLE_OPTIONS;
483 default:
484 return SolveStatus::UNKNOWN_STATUS;
485 }
486}

489ModelSolverHelper::ModelSolverHelper(const std::string& solver_name) {
490 if (solver_name.empty()) return;
491 MPSolver::OptimizationProblemType parsed_type;
492 if (!MPSolver::ParseSolverType(solver_name, &parsed_type)) {
493 VLOG(1) << "Unsupported type " << solver_name;
494 } else {
495 solver_type_ = static_cast<MPModelRequest::SolverType>(parsed_type);
496 }
497}

499bool ModelSolverHelper::SolverIsSupported() const {
500 if (!solver_type_.has_value()) return false;
501 if (solver_type_.value() == MPModelRequest::GLOP_LINEAR_PROGRAMMING) {
502 return true;
503 }
504#ifdef USE_PDLP
505 if (solver_type_.value() == MPModelRequest::PDLP_LINEAR_PROGRAMMING) {
506 return true;
507 }
508#endif // USE_PDLP
509 if (solver_type_.value() == MPModelRequest::SAT_INTEGER_PROGRAMMING) {
510 return true;
511 }
512#ifdef USE_SCIP
513 if (solver_type_.value() == MPModelRequest::SCIP_MIXED_INTEGER_PROGRAMMING) {
514 return true;
515 }
516#endif // USE_SCIP
517 return false;
518}

520void ModelSolverHelper::Solve(const ModelBuilderHelper& model) {
521 response_.reset();
522 if (!solver_type_.has_value()) {
523 response_->set_status(
524 MPSolverResponseStatus::MPSOLVER_SOLVER_TYPE_UNAVAILABLE);
525 return;
526 }
527
528 MPModelRequest request;
529 *request.mutable_model() = model.model();
530 request.set_solver_type(solver_type_.value());
531 request.set_enable_internal_solver_output(solver_output_);
532 if (time_limit_in_second_.has_value()) {
533 request.set_solver_time_limit_seconds(time_limit_in_second_.value());
534 }
535 if (!solver_specific_parameters_.empty()) {
536 request.set_solver_specific_parameters(solver_specific_parameters_);
537 }
538 switch (solver_type_.value()) {
539 case MPModelRequest::GLOP_LINEAR_PROGRAMMING: {
540 // TODO(user): Enable log_callback support.
541 MPSolutionResponse temp;
542 MPSolver::SolveWithProto(request, &temp, &interrupt_solve_);
543 response_ = std::move(temp);
544 break;
545 }
546 case MPModelRequest::SAT_INTEGER_PROGRAMMING: {
547 const auto temp =
548 SatSolveProto(request, &interrupt_solve_, log_callback_, nullptr);
549 if (temp.ok()) {
550 response_ = std::move(temp.value());
551 }
552 break;
553 }
554#if defined(USE_SCIP)
555 case MPModelRequest::SCIP_MIXED_INTEGER_PROGRAMMING: {
556 // TODO(user): Enable log_callback support.
557 // TODO(user): Enable interrupt_solve.
558 const auto temp = ScipSolveProto(request);
559 if (temp.ok()) {
560 response_ = std::move(temp.value());
561 }
562 break;
563 }
564#endif // defined(USE_SCIP)
565#if defined(USE_PDLP)
566 case MPModelRequest::PDLP_LINEAR_PROGRAMMING: {
567 const auto temp = PdlpSolveProto(request);
568 if (temp.ok()) {
569 response_ = std::move(temp.value());
570 }
571 break;
572 }
573#endif // defined(USE_PDLP)
574 default: {
575 response_->set_status(
576 MPSolverResponseStatus::MPSOLVER_SOLVER_TYPE_UNAVAILABLE);
577 }
578 }
579 if (response_->status() == MPSOLVER_OPTIMAL ||
580 response_->status() == MPSOLVER_FEASIBLE) {
581 model_of_last_solve_ = &model.model();
582 activities_.assign(model.num_constraints(),
583 std::numeric_limits<double>::quiet_NaN());
584 } else {
585 activities_.clear();
586 }
587}

657double ModelSolverHelper::activity(int ct_index) {
658 if (!has_response() || ct_index >= activities_.size() ||
659 !model_of_last_solve_.has_value()) {
660 return 0.0;
661 }
662
663 if (std::isnan(activities_[ct_index])) {
664 const MPConstraintProto& ct_proto =
665 model_of_last_solve_.value()->constraint(ct_index);
666 double result = 0.0;
