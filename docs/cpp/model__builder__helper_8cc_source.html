232SolveStatus MPSolverResponseStatusToSolveStatus(MPSolverResponseStatus s) {
233 switch (s) {
234 case MPSOLVER_OPTIMAL:
235 return SolveStatus::OPTIMAL;
236 case MPSOLVER_FEASIBLE:
237 return SolveStatus::FEASIBLE;
238 case MPSOLVER_INFEASIBLE:
239 return SolveStatus::INFEASIBLE;
240 case MPSOLVER_UNBOUNDED:
241 return SolveStatus::UNBOUNDED;
242 case MPSOLVER_ABNORMAL:
243 return SolveStatus::ABNORMAL;
244 case MPSOLVER_NOT_SOLVED:
245 return SolveStatus::NOT_SOLVED;
246 case MPSOLVER_MODEL_IS_VALID:
247 return SolveStatus::MODEL_IS_VALID;
248 case MPSOLVER_CANCELLED_BY_USER:
249 return SolveStatus::CANCELLED_BY_USER;
250 case MPSOLVER_UNKNOWN_STATUS:
251 return SolveStatus::UNKNOWN_STATUS;
252 case MPSOLVER_MODEL_INVALID:
253 return SolveStatus::MODEL_INVALID;
254 case MPSOLVER_MODEL_INVALID_SOLUTION_HINT:
255 return SolveStatus::MODEL_INVALID;
256 case MPSOLVER_MODEL_INVALID_SOLVER_PARAMETERS:
257 return SolveStatus::INVALID_SOLVER_PARAMETERS;
258 case MPSOLVER_SOLVER_TYPE_UNAVAILABLE:
259 return SolveStatus::SOLVER_TYPE_UNAVAILABLE;
260 case MPSOLVER_INCOMPATIBLE_OPTIONS:
261 return SolveStatus::INCOMPATIBLE_OPTIONS;
262 default:
263 return SolveStatus::UNKNOWN_STATUS;
264 }
265}

268ModelSolverHelper::ModelSolverHelper(const std::string& solver_name) {
269 if (solver_name.empty()) return;
270 MPSolver::OptimizationProblemType parsed_type;
271 if (!MPSolver::ParseSolverType(solver_name, &parsed_type)) {
272 VLOG(1) << "Unsupported type " << solver_name;
273 } else {
274 solver_type_ = static_cast<MPModelRequest::SolverType>(parsed_type);
275 }
276}

279 return solver_type_.has_value();

282void ModelSolverHelper::Solve(const ModelBuilderHelper& model) {
283 response_.reset();
284 if (!solver_type_.has_value()) {
285 response_->set_status(
286 MPSolverResponseStatus::MPSOLVER_SOLVER_TYPE_UNAVAILABLE);
287 return;
288 }
289
290 MPModelRequest request;
291 *request.mutable_model() = model.model();
292 request.set_solver_type(solver_type_.value());
293 request.set_enable_internal_solver_output(solver_output_);
294 if (time_limit_in_second_.has_value()) {
295 request.set_solver_time_limit_seconds(time_limit_in_second_.value());
296 }
297 if (!solver_specific_parameters_.empty()) {
298 request.set_solver_specific_parameters(solver_specific_parameters_);
299 }
300 switch (solver_type_.value()) {
301 case MPModelRequest::GLOP_LINEAR_PROGRAMMING: {
302 // TODO(user): Enable log_callback support.
303 MPSolutionResponse temp;
304 MPSolver::SolveWithProto(request, &temp, &interrupt_solve_);
305 response_ = std::move(temp);
306 break;
307 }
308 case MPModelRequest::SAT_INTEGER_PROGRAMMING: {
309 const auto temp =
310 SatSolveProto(request, &interrupt_solve_, log_callback_, nullptr);
311 if (temp.ok()) {
312 response_ = std::move(temp.value());
313 }
314 break;
315 }
316#if defined(USE_SCIP)
317 case MPModelRequest::SCIP_MIXED_INTEGER_PROGRAMMING: {
318 // TODO(user): Enable log_callback support.
319 // TODO(user): Enable interrupt_solve.
320 const auto temp = ScipSolveProto(request);
321 if (temp.ok()) {
322 response_ = std::move(temp.value());
323 }
324 break;
325 }
326#endif // defined(USE_SCIP)
327 default: {
328 response_->set_status(
329 MPSolverResponseStatus::MPSOLVER_SOLVER_TYPE_UNAVAILABLE);
330 }
331 }
332}
