146void ModelBuilderHelper::AddConstraintTerm(int ct_index, int var_index,
147 double coeff) {
148 if (coeff == 0.0) return;
149 MPConstraintProto* ct_proto = model_.mutable_constraint(ct_index);
150 ct_proto->add_var_index(var_index);
151 ct_proto->add_coefficient(coeff);
152}

154void ModelBuilderHelper::SafeAddConstraintTerm(int ct_index, int var_index,
155 double coeff) {
156 if (coeff == 0.0) return;
157 MPConstraintProto* ct_proto = model_.mutable_constraint(ct_index);
158 for (int i = 0; i < ct_proto->var_index_size(); ++i) {
159 if (ct_proto->var_index(i) == var_index) {
160 ct_proto->set_coefficient(i, coeff + ct_proto->coefficient(i));
161 return;
162 }
163 }
164 // If we reach this point, the variable does not exist in the constraint yet,
165 // so we add it to the constraint as a new term.
166 ct_proto->add_var_index(var_index);
167 ct_proto->add_coefficient(coeff);
168}

277SolveStatus MPSolverResponseStatusToSolveStatus(MPSolverResponseStatus s) {
278 switch (s) {
279 case MPSOLVER_OPTIMAL:
280 return SolveStatus::OPTIMAL;
281 case MPSOLVER_FEASIBLE:
282 return SolveStatus::FEASIBLE;
283 case MPSOLVER_INFEASIBLE:
284 return SolveStatus::INFEASIBLE;
285 case MPSOLVER_UNBOUNDED:
286 return SolveStatus::UNBOUNDED;
287 case MPSOLVER_ABNORMAL:
288 return SolveStatus::ABNORMAL;
289 case MPSOLVER_NOT_SOLVED:
290 return SolveStatus::NOT_SOLVED;
291 case MPSOLVER_MODEL_IS_VALID:
292 return SolveStatus::MODEL_IS_VALID;
293 case MPSOLVER_CANCELLED_BY_USER:
294 return SolveStatus::CANCELLED_BY_USER;
295 case MPSOLVER_UNKNOWN_STATUS:
296 return SolveStatus::UNKNOWN_STATUS;
297 case MPSOLVER_MODEL_INVALID:
298 return SolveStatus::MODEL_INVALID;
299 case MPSOLVER_MODEL_INVALID_SOLUTION_HINT:
300 return SolveStatus::MODEL_INVALID;
301 case MPSOLVER_MODEL_INVALID_SOLVER_PARAMETERS:
302 return SolveStatus::INVALID_SOLVER_PARAMETERS;

313ModelSolverHelper::ModelSolverHelper(const std::string& solver_name) {
314 if (solver_name.empty()) return;
315 MPSolver::OptimizationProblemType parsed_type;
316 if (!MPSolver::ParseSolverType(solver_name, &parsed_type)) {
317 VLOG(1) << "Unsupported type " << solver_name;
318 } else {
319 solver_type_ = static_cast<MPModelRequest::SolverType>(parsed_type);
320 }
321}
322
323bool ModelSolverHelper::SolverIsSupported() const {
324 if (!solver_type_.has_value()) return false;
325 if (solver_type_.value() == MPModelRequest::GLOP_LINEAR_PROGRAMMING) {
326 return true;
327 }
328#ifdef USE_PDLP
329 if (solver_type_.value() == MPModelRequest::PDLP_LINEAR_PROGRAMMING) {
330 return true;
331 }
332#endif // USE_PDLP
333 if (solver_type_.value() == MPModelRequest::SAT_INTEGER_PROGRAMMING) {
334 return true;
335 }
336#ifdef USE_SCIP
337 if (solver_type_.value() == MPModelRequest::SCIP_MIXED_INTEGER_PROGRAMMING) {
338 return true;
339 }
340#endif // USE_SCIP
341 return false;
342}

344void ModelSolverHelper::Solve(const ModelBuilderHelper& model) {
345 response_.reset();
346 if (!solver_type_.has_value()) {
347 response_->set_status(
348 MPSolverResponseStatus::MPSOLVER_SOLVER_TYPE_UNAVAILABLE);
349 return;
350 }
351
352 MPModelRequest request;
353 *request.mutable_model() = model.model();
354 request.set_solver_type(solver_type_.value());
355 request.set_enable_internal_solver_output(solver_output_);
356 if (time_limit_in_second_.has_value()) {
357 request.set_solver_time_limit_seconds(time_limit_in_second_.value());
358 }
359 if (!solver_specific_parameters_.empty()) {
360 request.set_solver_specific_parameters(solver_specific_parameters_);
361 }
362 switch (solver_type_.value()) {
363 case MPModelRequest::GLOP_LINEAR_PROGRAMMING: {
364 // TODO(user): Enable log_callback support.
365 MPSolutionResponse temp;
366 MPSolver::SolveWithProto(request, &temp, &interrupt_solve_);
367 response_ = std::move(temp);
368 break;
369 }
370 case MPModelRequest::SAT_INTEGER_PROGRAMMING: {
371 const auto temp =
372 SatSolveProto(request, &interrupt_solve_, log_callback_, nullptr);
373 if (temp.ok()) {
374 response_ = std::move(temp.value());
375 }
376 break;
377 }
378#if defined(USE_SCIP)
379 case MPModelRequest::SCIP_MIXED_INTEGER_PROGRAMMING: {
380 // TODO(user): Enable log_callback support.
381 // TODO(user): Enable interrupt_solve.
382 const auto temp = ScipSolveProto(request);
383 if (temp.ok()) {
384 response_ = std::move(temp.value());
385 }
386 break;
387 }
388#endif // defined(USE_SCIP)
389#if defined(USE_PDLP)
390 case MPModelRequest::PDLP_LINEAR_PROGRAMMING: {
391 const auto temp = PdlpSolveProto(request);
392 if (temp.ok()) {
393 response_ = std::move(temp.value());
394 }
395 break;
396 }
397#endif // defined(USE_PDLP)
398 default: {
399 response_->set_status(
400 MPSolverResponseStatus::MPSOLVER_SOLVER_TYPE_UNAVAILABLE);
401 }
402 }
403 if (response_->status() == MPSOLVER_OPTIMAL ||
404 response_->status() == MPSOLVER_FEASIBLE) {
405 model_of_last_solve_ = &model.model();
406 activities_.assign(model.num_constraints(),
407 std::numeric_limits<double>::quiet_NaN());
408 } else {
409 activities_.clear();
410 }
411}

481double ModelSolverHelper::activity(int ct_index) {
482 if (!has_response() || ct_index >= activities_.size() ||
483 !model_of_last_solve_.has_value()) {
484 return 0.0;
485 }
486
487 if (std::isnan(activities_[ct_index])) {
488 const MPConstraintProto& ct_proto =
489 model_of_last_solve_.value()->constraint(ct_index);
490 double result = 0.0;
491 for (int i = 0; i < ct_proto.var_index_size(); ++i) {
492 result += response_->variable_value(ct_proto.var_index(i)) *
493 ct_proto.coefficient(i);
494 }
495 activities_[ct_index] = result;
496 }
497 return activities_[ct_index];
498}
