72bool ModelBuilderHelper::ReadModelFromProtoFile(const std::string& filename) {
73 if (file::GetTextProto(filename, &model_, file::Defaults()).ok() ||
74 file::GetBinaryProto(filename, &model_, file::Defaults()).ok()) {
75 return true;
76 }
77 MPModelRequest request;
78 if (file::GetTextProto(filename, &request, file::Defaults()).ok() ||
79 file::GetBinaryProto(filename, &request, file::Defaults()).ok()) {
80 model_ = request.model();
81 return true;
82 }
83 return false;
84}

211void ModelBuilderHelper::SafeAddConstraintTerm(int ct_index, int var_index,
212 double coeff) {
213 if (coeff == 0.0) return;
214 MPConstraintProto* ct_proto = model_.mutable_constraint(ct_index);
215 for (int i = 0; i < ct_proto->var_index_size(); ++i) {
216 if (ct_proto->var_index(i) == var_index) {
217 ct_proto->set_coefficient(i, coeff + ct_proto->coefficient(i));
218 return;
219 }
220 }
221 // If we reach this point, the variable does not exist in the constraint yet,
222 // so we add it to the constraint as a new term.
223 ct_proto->add_var_index(var_index);
224 ct_proto->add_coefficient(coeff);
225}

476SolveStatus MPSolverResponseStatusToSolveStatus(MPSolverResponseStatus s) {
477 switch (s) {
478 case MPSOLVER_OPTIMAL:
479 return SolveStatus::OPTIMAL;
480 case MPSOLVER_FEASIBLE:
481 return SolveStatus::FEASIBLE;
482 case MPSOLVER_INFEASIBLE:
483 return SolveStatus::INFEASIBLE;

485 return SolveStatus::UNBOUNDED;
486 case MPSOLVER_ABNORMAL:
487 return SolveStatus::ABNORMAL;
488 case MPSOLVER_NOT_SOLVED:
489 return SolveStatus::NOT_SOLVED;
490 case MPSOLVER_MODEL_IS_VALID:
491 return SolveStatus::MODEL_IS_VALID;
492 case MPSOLVER_CANCELLED_BY_USER:
493 return SolveStatus::CANCELLED_BY_USER;
494 case MPSOLVER_UNKNOWN_STATUS:
495 return SolveStatus::UNKNOWN_STATUS;
496 case MPSOLVER_MODEL_INVALID:
497 return SolveStatus::MODEL_INVALID;
498 case MPSOLVER_MODEL_INVALID_SOLUTION_HINT:
499 return SolveStatus::MODEL_INVALID;

505 return SolveStatus::INCOMPATIBLE_OPTIONS;

512ModelSolverHelper::ModelSolverHelper(const std::string& solver_name) {
513 if (solver_name.empty()) return;
514 MPSolver::OptimizationProblemType parsed_type;
515 if (!MPSolver::ParseSolverType(solver_name, &parsed_type)) {
516 VLOG(1) << "Unsupported type " << solver_name;
517 } else {
518 solver_type_ = static_cast<MPModelRequest::SolverType>(parsed_type);
519 }
520}

522bool ModelSolverHelper::SolverIsSupported() const {
523 if (!solver_type_.has_value()) return false;
524 if (solver_type_.value() == MPModelRequest::GLOP_LINEAR_PROGRAMMING) {
525 return true;
526 }
527#ifdef USE_PDLP
528 if (solver_type_.value() == MPModelRequest::PDLP_LINEAR_PROGRAMMING) {
529 return true;
530 }
531#endif // USE_PDLP
532 if (solver_type_.value() == MPModelRequest::SAT_INTEGER_PROGRAMMING) {
533 return true;
534 }
535#ifdef USE_SCIP
536 if (solver_type_.value() == MPModelRequest::SCIP_MIXED_INTEGER_PROGRAMMING) {
537 return true;
538 }
539#endif // USE_SCIP
540#ifdef USE_HIGHS
541 if (solver_type_.value() == MPModelRequest::HIGHS_LINEAR_PROGRAMMING ||
542 solver_type_.value() == MPModelRequest::HIGHS_MIXED_INTEGER_PROGRAMMING) {
543 return true;
544 }
545#endif // USE_HIGHS
546 if (solver_type_.value() ==
547 MPModelRequest::GUROBI_MIXED_INTEGER_PROGRAMMING ||
548 solver_type_.value() == MPModelRequest::GUROBI_LINEAR_PROGRAMMING) {
549 return GurobiIsCorrectlyInstalled();
550 }
551 if (solver_type_.value() ==
552 MPModelRequest::XPRESS_MIXED_INTEGER_PROGRAMMING ||
553 solver_type_.value() == MPModelRequest::XPRESS_LINEAR_PROGRAMMING) {
554 return XpressIsCorrectlyInstalled();
555 }
556 return false;
557}

559void ModelSolverHelper::Solve(const ModelBuilderHelper& model) {
560 response_.reset();
561 if (!solver_type_.has_value()) {
562 response_->set_status(
563 MPSolverResponseStatus::MPSOLVER_SOLVER_TYPE_UNAVAILABLE);
564 return;
565 }
566
567 MPModelRequest request;
568 *request.mutable_model() = model.model();
569 request.set_solver_type(solver_type_.value());
570 request.set_enable_internal_solver_output(solver_output_);
571 if (time_limit_in_second_.has_value()) {
572 request.set_solver_time_limit_seconds(time_limit_in_second_.value());
573 }
574 if (!solver_specific_parameters_.empty()) {
575 request.set_solver_specific_parameters(solver_specific_parameters_);
576 }
577 switch (solver_type_.value()) {
578 case MPModelRequest::GLOP_LINEAR_PROGRAMMING: {
579 response_ =
580 GlopSolveProto(std::move(request), &interrupt_solve_, log_callback_);
581 break;
582 }
583 case MPModelRequest::SAT_INTEGER_PROGRAMMING: {
584 response_ = SatSolveProto(std::move(request), &interrupt_solve_,
585 log_callback_, nullptr);
586 break;
587 }
588#if defined(USE_SCIP)
589 case MPModelRequest::SCIP_MIXED_INTEGER_PROGRAMMING: {
590 // TODO(user): Enable log_callback support.
591 // TODO(user): Enable interrupt_solve.
592 const auto temp = ScipSolveProto(std::move(request));
593 if (temp.ok()) {
594 response_ = std::move(temp.value());
595 }
596 break;
597 }
598#endif // defined(USE_SCIP)
599#if defined(USE_PDLP)
600 case MPModelRequest::PDLP_LINEAR_PROGRAMMING: {
601 const auto temp = PdlpSolveProto(std::move(request));
602 if (temp.ok()) {
603 response_ = std::move(temp.value());
604 }
605 break;
606 }
607#endif // defined(USE_PDLP)
608 case MPModelRequest::
609 GUROBI_LINEAR_PROGRAMMING: // ABSL_FALLTHROUGH_INTENDED
610 case MPModelRequest::GUROBI_MIXED_INTEGER_PROGRAMMING: {
611 const auto temp = GurobiSolveProto(std::move(request));
612 if (temp.ok()) {
613 response_ = std::move(temp.value());
614 }
615 break;
616 }
617#if defined(USE_HIGHS)
618 case MPModelRequest::HIGHS_LINEAR_PROGRAMMING: // ABSL_FALLTHROUGH_INTENDED
619 case MPModelRequest::HIGHS_MIXED_INTEGER_PROGRAMMING: {
620 // TODO(user): Enable log_callback support.
621 // TODO(user): Enable interrupt_solve.
622 const auto temp = HighsSolveProto(std::move(request));
623 if (temp.ok()) {
624 response_ = std::move(temp.value());
625 }
626 break;
627 }
628#endif // defined(USE_HIGHS)
629 case MPModelRequest::
630 XPRESS_LINEAR_PROGRAMMING: // ABSL_FALLTHROUGH_INTENDED
631 case MPModelRequest::XPRESS_MIXED_INTEGER_PROGRAMMING: {
632 response_ = XPressSolveProto(request);
633 break;
634 }
635 default: {
636 response_->set_status(
637 MPSolverResponseStatus::MPSOLVER_SOLVER_TYPE_UNAVAILABLE);
638 }
639 }
640 if (response_->status() == MPSOLVER_OPTIMAL ||
641 response_->status() == MPSOLVER_FEASIBLE) {
642 model_of_last_solve_ = &model.model();

719double ModelSolverHelper::activity(int ct_index) {
720 if (!has_response() || ct_index >= activities_.size() ||
721 !model_of_last_solve_.has_value()) {
722 return 0.0;
723 }
724
725 if (std::isnan(activities_[ct_index])) {
726 const MPConstraintProto& ct_proto =
727 model_of_last_solve_.value()->constraint(ct_index);
728 double result = 0.0;
729 for (int i = 0; i < ct_proto.var_index_size(); ++i) {
730 result += response_->variable_value(ct_proto.var_index(i)) *
731 ct_proto.coefficient(i);
732 }
733 activities_[ct_index] = result;
734 }
735 return activities_[ct_index];
736}
