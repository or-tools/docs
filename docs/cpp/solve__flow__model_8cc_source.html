52// See http://lpsolve.sourceforge.net/5.5/DIMACS_mcf.htm for the dimacs file
53// format of a min cost flow problem.
54//
55// TODO(user): This currently only works for min cost flow problem.
56void ConvertFlowModelToDimacs(const FlowModelProto& flow_model,
57 std::string* dimacs) {
58 CHECK_EQ(FlowModelProto::MIN_COST_FLOW, flow_model.problem_type());
59 dimacs->clear();
60 dimacs->append("c Min-Cost flow problem generated from a FlowModelProto.\n");
61
62 // We need to compute the num_nodes from the nodes appearing in the arcs.
63 int64_t num_nodes = 0;
64 for (int64_t i = 0; i < flow_model.arcs_size(); ++i) {
65 num_nodes = std::max(num_nodes, flow_model.arcs(i).tail() + 1);
66 num_nodes = std::max(num_nodes, flow_model.arcs(i).head() + 1);
67 }
68
69 // Problem size and type.
70 const int64_t num_arcs = flow_model.arcs_size();
71 dimacs->append("c\nc Problem line (problem_type, num nodes, num arcs).\n");
72 dimacs->append(absl::StrFormat("p min %d %d\n", num_nodes, num_arcs));
73
74 // Nodes.
75 dimacs->append("c\nc Node descriptor lines (id, supply/demand).\n");
76 for (int64_t i = 0; i < flow_model.nodes_size(); ++i) {
77 const int64_t id = flow_model.nodes(i).id() + 1;
78 const int64_t supply = flow_model.nodes(i).supply();
79 if (supply != 0) {
80 dimacs->append(absl::StrFormat("n %d %d\n", id, supply));
81 }
82 }
83
84 // Arcs.
85 dimacs->append(
86 "c\nc Arc descriptor Lines (tail, head, minflow, maxflow, cost).\n");
87 for (int64_t i = 0; i < flow_model.arcs_size(); ++i) {
88 const int64_t tail = flow_model.arcs(i).tail() + 1;
89 const int64_t head = flow_model.arcs(i).head() + 1;
90 const int64_t unit_cost = flow_model.arcs(i).unit_cost();
91 const int64_t capacity = flow_model.arcs(i).capacity();
92 dimacs->append(absl::StrFormat("a %d %d %d %d %d\n", tail, head, int64_t{0},
93 capacity, unit_cost));
94 }
95 dimacs->append("c\n");
96}

98// Note(user): Going from Dimacs to flow adds an extra copy, but for now we
99// don't really care of the Dimacs file reading performance.
100// Returns true if the file was converted correctly.
101bool ConvertDimacsToFlowModel(const std::string& file,
102 FlowModelProto* flow_model) {
103 flow_model->Clear();
104 FlowModelProto::ProblemType problem_type;
105 for (const std::string& line : FileLines(file)) {
106 if (line.empty()) continue;
107 if (line[0] == 'p') {
108 if (absl::StartsWith(line, "p min")) {
109 problem_type = FlowModelProto::MIN_COST_FLOW;
110 } else if (absl::StartsWith(line, "p max")) {
111 problem_type = FlowModelProto::MAX_FLOW;
112 } else {
113 LOG(ERROR) << "Unknown dimacs problem format.";
114 return false;
115 }
116 flow_model->set_problem_type(problem_type);
117 }
118 if (line[0] == 'n') {
119 int64_t id;
120 int64_t supply;
121 std::stringstream ss(line.substr(1));
122 switch (problem_type) {
123 case FlowModelProto::MIN_COST_FLOW:
124 ss >> id >> supply;
125 break;
126 case FlowModelProto::MAX_FLOW: {
127 std::string type;
128 ss >> id >> type;
129 supply = (type == "s" ? 1 : -1);
130 break;
131 }
132 default:
133 LOG(ERROR) << "Node line before the problem type definition.";
134 return false;
135 }
136 FlowNodeProto* node = flow_model->add_nodes();
137 node->set_id(id - 1);
138 node->set_supply(supply);
139 }
140 if (line[0] == 'a') {
141 int64_t tail;
142 int64_t head;
143 int64_t min_capacity = 0;
144 int64_t max_capacity = 0;
145 int64_t unit_cost = 0;
146 std::stringstream ss(line.substr(1));
147 switch (problem_type) {
148 case FlowModelProto::MIN_COST_FLOW:
149 ss >> tail >> head >> min_capacity >> max_capacity >> unit_cost;
150 break;
151 case FlowModelProto::MAX_FLOW:
152 ss >> tail >> head >> max_capacity;
153 break;
154 default:
155 LOG(ERROR) << "Arc line before the problem type definition.";
156 return false;
157 }
158 FlowArcProto* arc = flow_model->add_arcs();
159 arc->set_tail(tail - 1);
160 arc->set_head(head - 1);
161 arc->set_capacity(max_capacity);
162 arc->set_unit_cost(unit_cost);
163 if (min_capacity != 0) {
164 LOG(ERROR) << "We do not support minimum capacity.";
165 return false;
166 }
167 }
168 }
169 return true;
170}

175// Loads a FlowModelProto proto into the MinCostFlow class and solves it.
176void SolveMinCostFlow(const FlowModelProto& flow_model, double* loading_time,
177 double* solving_time) {
178 WallTimer timer;
179 timer.Start();
180
181 // Compute the number of nodes.
182 int64_t num_nodes = 0;
183 for (int i = 0; i < flow_model.arcs_size(); ++i) {
184 num_nodes = std::max(num_nodes, flow_model.arcs(i).tail() + 1);
185 num_nodes = std::max(num_nodes, flow_model.arcs(i).head() + 1);
186 }
187
188 // Build the graph.
189 Graph graph(num_nodes, flow_model.arcs_size());
190 for (int i = 0; i < flow_model.arcs_size(); ++i) {
191 graph.AddArc(flow_model.arcs(i).tail(), flow_model.arcs(i).head());
192 }
193 std::vector<Graph::ArcIndex> permutation;
194 graph.Build(&permutation);
195
196 GenericMinCostFlow<Graph> min_cost_flow(&graph);
197 for (int i = 0; i < flow_model.arcs_size(); ++i) {
198 const Graph::ArcIndex image = i < permutation.size() ? permutation[i] : i;
199 min_cost_flow.SetArcUnitCost(image, flow_model.arcs(i).unit_cost());
200 min_cost_flow.SetArcCapacity(image, flow_model.arcs(i).capacity());
201 }
202 for (int i = 0; i < flow_model.nodes_size(); ++i) {
203 min_cost_flow.SetNodeSupply(flow_model.nodes(i).id(),
204 flow_model.nodes(i).supply());
205 }
206
207 *loading_time = timer.Get();
208 absl::PrintF("%f,", *loading_time);
209 fflush(stdout);
210
211 timer.Start();
212 CHECK(min_cost_flow.Solve());
213 CHECK_EQ(GenericMinCostFlow<Graph>::OPTIMAL, min_cost_flow.status());
214 *solving_time = timer.Get();
215 absl::PrintF("%f,", *solving_time);
216 absl::PrintF("%d", min_cost_flow.GetOptimalCost());
217 fflush(stdout);
218}

220// Loads a FlowModelProto proto into the MaxFlow class and solves it.
221void SolveMaxFlow(const FlowModelProto& flow_model, double* loading_time,
222 double* solving_time) {
223 WallTimer timer;
224 timer.Start();
225
226 // Compute the number of nodes.
227 int64_t num_nodes = 0;
228 for (int i = 0; i < flow_model.arcs_size(); ++i) {
229 num_nodes = std::max(num_nodes, flow_model.arcs(i).tail() + 1);
230 num_nodes = std::max(num_nodes, flow_model.arcs(i).head() + 1);
231 }
232
233 // Build the graph.
234 Graph graph(num_nodes, flow_model.arcs_size());
235 for (int i = 0; i < flow_model.arcs_size(); ++i) {
236 graph.AddArc(flow_model.arcs(i).tail(), flow_model.arcs(i).head());
237 }
238 std::vector<Graph::ArcIndex> permutation;
239 graph.Build(&permutation);
240
241 // Find source & sink.
242 Graph::NodeIndex source = -1;
243 Graph::NodeIndex sink = -1;
244 CHECK_EQ(2, flow_model.nodes_size());
245 for (int i = 0; i < flow_model.nodes_size(); ++i) {
246 if (flow_model.nodes(i).supply() > 0) {
247 source = flow_model.nodes(i).id();
248 }
249 if (flow_model.nodes(i).supply() < 0) {
250 sink = flow_model.nodes(i).id();
251 }
252 }
253 CHECK_NE(source, -1);
254 CHECK_NE(sink, -1);
255
256 // Create the max flow instance and set the arc capacities.
257 GenericMaxFlow<Graph> max_flow(&graph, source, sink);
258 for (int i = 0; i < flow_model.arcs_size(); ++i) {
259 const Graph::ArcIndex image = i < permutation.size() ? permutation[i] : i;
260 max_flow.SetArcCapacity(image, flow_model.arcs(i).capacity());
261 }
262
263 *loading_time = timer.Get();
264 absl::PrintF("%f,", *loading_time);
265 fflush(stdout);
266
267 timer.Start();
268 CHECK(max_flow.Solve());
269 CHECK_EQ(GenericMaxFlow<Graph>::OPTIMAL, max_flow.status());
270 *solving_time = timer.Get();
271 absl::PrintF("%f,", *solving_time);
272 absl::PrintF("%d", max_flow.GetOptimalFlow());
273 fflush(stdout);
274}

283int main(int argc, char** argv) {
284 InitGoogle(
285 "Runs the OR-tools min-cost flow on a pattern of files given by --input. "
286 "The files must be in Dimacs text format or in binary FlowModelProto "
287 "format.",
288 &argc, &argv, true);
289 absl::SetFlag(&FLAGS_stderrthreshold, 0);
290 if (absl::GetFlag(FLAGS_input).empty()) {
291 LOG(FATAL) << "Please specify input pattern via --input=...";
292 }
293 std::vector<std::string> file_list;
294 file::Match(absl::GetFlag(FLAGS_input), &file_list, file::Defaults())
295 .IgnoreError();
296
297 TimeDistribution parsing_time_distribution("Parsing time summary");
298 TimeDistribution loading_time_distribution("Loading time summary");
299 TimeDistribution solving_time_distribution("Solving time summary");
300
301 absl::PrintF(
302 "file_name, parsing_time, loading_time, solving_time, optimal_cost\n");
303 for (int i = 0; i < file_list.size(); ++i) {
304 const std::string file_name = file_list[i];
305 absl::PrintF("%s,", file_name);
306 fflush(stdout);
307
308 // Parse the input as a proto.
309 double parsing_time = 0;
310 operations_research::FlowModelProto proto;
311 if (absl::EndsWith(file_name, ".bin")) {
312 ScopedWallTime timer(&parsing_time);
313 std::string raw_data;
314 CHECK_OK(file::GetContents(file_name, &raw_data, file::Defaults()));
315 proto.ParseFromString(raw_data);
316 } else {
317 ScopedWallTime timer(&parsing_time);
318 if (!ConvertDimacsToFlowModel(file_name, &proto)) continue;
319 }
320 absl::PrintF("%f,", parsing_time);
321 fflush(stdout);
322
323 // TODO(user): improve code to convert many files.
324 if (!absl::GetFlag(FLAGS_output_dimacs).empty()) {
325 LOG(INFO) << "Converting first input file to dimacs format.";
326 double time = 0;
327 {
328 ScopedWallTime timer(&time);
329 std::string dimacs;
330 ConvertFlowModelToDimacs(proto, &dimacs);
331 CHECK_OK(file::SetContents(absl::GetFlag(FLAGS_output_dimacs), dimacs,
332 file::Defaults()));
333 }
334 LOG(INFO) << "Done: " << time << "s.";
335 return EXIT_SUCCESS;
336 }
337
338 double loading_time = 0;
339 double solving_time = 0;
340 switch (proto.problem_type()) {
341 case FlowModelProto::MIN_COST_FLOW:
342 SolveMinCostFlow(proto, &loading_time, &solving_time);
343 break;
344 case FlowModelProto::MAX_FLOW:
345 SolveMaxFlow(proto, &loading_time, &solving_time);
346 break;
347 default:
348 LOG(ERROR) << "Problem type not supported: " << proto.problem_type();
349 }
350 absl::PrintF("\n");
351
352 parsing_time_distribution.AddTimeInSec(parsing_time);
353 loading_time_distribution.AddTimeInSec(loading_time);
354 solving_time_distribution.AddTimeInSec(solving_time);
355 }
356 absl::PrintF("%s", parsing_time_distribution.StatString());
357 absl::PrintF("%s", loading_time_distribution.StatString());
358 absl::PrintF("%s", solving_time_distribution.StatString());
359 return EXIT_SUCCESS;
360}
