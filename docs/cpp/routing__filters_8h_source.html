135class BasePathFilter : public IntVarLocalSearchFilter {
136 public:
137 BasePathFilter(const std::vector<IntVar*>& nexts, int next_domain_size);
138 ~BasePathFilter() override {}
139 bool Accept(const Assignment* delta, const Assignment* deltadelta,
140 int64_t objective_min, int64_t objective_max) override;
141 void OnSynchronize(const Assignment* delta) override;
142
143 protected:
144 static const int64_t kUnassigned;
145
146 int64_t GetNext(int64_t node) const {
147 return (new_nexts_[node] == kUnassigned)
148 ? (IsVarSynced(node) ? Value(node) : kUnassigned)
149 : new_nexts_[node];
150 }
151 int NumPaths() const { return starts_.size(); }
152 int64_t Start(int i) const { return starts_[i]; }
153 int GetPath(int64_t node) const { return paths_[node]; }
154 int Rank(int64_t node) const { return ranks_[node]; }
155 bool IsDisabled() const { return status_ == DISABLED; }
156 const std::vector<int64_t>& GetTouchedPathStarts() const {
157 return touched_paths_.PositionsSetAtLeastOnce();
158 }
159 bool PathStartTouched(int64_t start) const { return touched_paths_[start]; }
160 const std::vector<int64_t>& GetNewSynchronizedUnperformedNodes() const {
161 return new_synchronized_unperformed_nodes_.PositionsSetAtLeastOnce();
162 }
163
164 bool lns_detected() const { return lns_detected_; }
165
166 private:
167 enum Status { UNKNOWN, ENABLED, DISABLED };
168
169 virtual bool DisableFiltering() const { return false; }
170 virtual void OnBeforeSynchronizePaths() {}
171 virtual void OnAfterSynchronizePaths() {}
172 virtual void OnSynchronizePathFromStart(int64_t start) {}
173 virtual bool InitializeAcceptPath() { return true; }
174 virtual bool AcceptPath(int64_t path_start, int64_t chain_start,
175 int64_t chain_end) = 0;
176 virtual bool FinalizeAcceptPath(int64_t objective_min,
177 int64_t objective_max) {
178 return true;
179 }
181 void ComputePathStarts(std::vector<int64_t>* path_starts,
182 std::vector<int>* index_to_path);
183 bool HavePathsChanged();
184 void SynchronizeFullAssignment();
185 void UpdateAllRanks();
186 void UpdatePathRanksFromStart(int start);
187
188 std::vector<int64_t> node_path_starts_;
189 std::vector<int64_t> starts_;
190 std::vector<int> paths_;
191 SparseBitset<int64_t> new_synchronized_unperformed_nodes_;
192 std::vector<int64_t> new_nexts_;
193 std::vector<int> delta_touched_;
194 SparseBitset<> touched_paths_;
195 // clang-format off
196 std::vector<std::pair<int64_t, int64_t> > touched_path_chain_start_ends_;
197 // clang-format on
198 std::vector<int> ranks_;
199
200 Status status_;
201 bool lns_detected_;
202};
