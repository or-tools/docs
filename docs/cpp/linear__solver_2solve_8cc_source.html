131MPModelRequest ReadMipModel(const std::string& input) {
132 MPModelRequest request_proto;
133 MPModelProto model_proto;
134 if (absl::EndsWith(input, ".lp")) {
135#if defined(USE_LP_PARSER)
136 std::string data;
137 CHECK_OK(file::GetContents(input, &data, file::Defaults()));
138 absl::StatusOr<MPModelProto> result = ModelProtoFromLpFormat(data);
139 CHECK_OK(result);
140 model_proto = std::move(result).value();
141#else // !defined(USE_LP_PARSER)
142 LOG(FATAL) << "Support for parsing LP format is not compiled in.";
143#endif // !defined(USE_LP_PARSER)
144 } else if (absl::EndsWith(input, ".mps") ||
145 absl::EndsWith(input, ".mps.gz")) {
146 QCHECK_OK(glop::MPSReader().ParseFile(input, &model_proto))
147 << "Error while parsing the mps file '" << input << "'.";
148 } else {
149 ReadFileToProto(input, &model_proto);
150 ReadFileToProto(input, &request_proto);
151 }
152 // If the input is a proto in binary format, both ReadFileToProto could
153 // return true. Instead use the actual number of variables found to test the
154 // correct format of the input.
155 const bool is_model_proto = model_proto.variable_size() > 0;
156 const bool is_request_proto =
157 request_proto.model().variable_size() > 0 ||
158 !request_proto.model_delta().baseline_model_file_path().empty();
159 if (!is_model_proto && !is_request_proto) {
160 LOG(FATAL) << "Failed to parse '" << input
161 << "' as an MPModelProto or an MPModelRequest.";
162 } else {
163 CHECK(!(is_model_proto && is_request_proto));
164 }
165 if (is_request_proto) {
166 LOG(INFO) << "Read input proto as an MPModelRequest.";
167 } else {
168 LOG(INFO) << "Read input proto as an MPModelProto.";
169 model_proto.Swap(request_proto.mutable_model());
170 }
171 return request_proto;
172}

174MPSolutionResponse LocalSolve(const MPModelRequest& request_proto) {
175 // TODO(or-core-team): Why doesn't this use MPSolver::SolveWithProto() ?
176
177 // Create the solver, we use the name of the model as the solver name.
178 MPSolver solver(request_proto.model().name(),
179 static_cast<MPSolver::OptimizationProblemType>(
180 request_proto.solver_type()));
181 const absl::Status set_num_threads_status =
182 solver.SetNumThreads(absl::GetFlag(FLAGS_num_threads));
183 if (set_num_threads_status.ok()) {
184 LOG(INFO) << "Set number of threads to " << absl::GetFlag(FLAGS_num_threads)
185 << ".";
186 } else if (absl::GetFlag(FLAGS_num_threads) != 1) {
187 LOG(ERROR) << "Failed to set number of threads due to: "
188 << set_num_threads_status.message() << ". Using 1 as default.";
189 }
190 solver.EnableOutput();
191
192 if (request_proto.has_solver_specific_parameters()) {
193 CHECK(solver.SetSolverSpecificParametersAsString(
194 request_proto.solver_specific_parameters()))
195 << "Wrong solver_specific_parameters (bad --params or --params_file ?)";
196 }
197
198 MPSolutionResponse response;
199
200 // Load the model proto into the solver.
201 {
202 std::string error_message;
203 const MPSolverResponseStatus status =
204 solver.LoadModelFromProtoWithUniqueNamesOrDie(request_proto.model(),
205 &error_message);
206 // Note, the underlying MPSolver treats time limit equal to 0 as no limit.
207 if (status != MPSOLVER_MODEL_IS_VALID) {
208 // HACK(user): For SAT solves, when the model is invalid we directly
209 // exit here.
210 if (request_proto.solver_type() ==
211 MPModelRequest::SAT_INTEGER_PROGRAMMING) {
212 sat::CpSolverResponse sat_response;
213 sat_response.set_status(sat::CpSolverStatus::MODEL_INVALID);
214 LOG(INFO) << sat::CpSolverResponseStats(sat_response);
215 exit(1);
216 }
217 response.set_status(status);
218 response.set_status_str(error_message);
219 return response;
220 }
221 }
222 if (request_proto.has_solver_time_limit_seconds()) {
223 solver.SetTimeLimit(
224 absl::Seconds(request_proto.solver_time_limit_seconds()));
225 }
226
227 // Register a signal handler to interrupt the solve when the user presses ^C.
228 // Note that we ignore all previously registered handler here. If SCIP is
229 // used, this handler will be overridden by the one of SCIP that does the same
230 // thing.
231 SigintHandler handler;
232 handler.Register([&solver] { solver.InterruptSolve(); });
233
234 // Solve.
235 const MPSolver::ResultStatus status = solver.Solve();
236
237 // If --verify_solution is true, we already verified it. If not, we add
238 // a verification step here.
239 if ((status == MPSolver::OPTIMAL || status == MPSolver::FEASIBLE) &&
240 !absl::GetFlag(FLAGS_verify_solution)) {
241 const bool verified =
242 solver.VerifySolution(/*tolerance=*/MPSolverParameters().GetDoubleParam(
243 MPSolverParameters::PRIMAL_TOLERANCE),
244 absl::GetFlag(FLAGS_log_verification_errors));
245 LOG(INFO) << "The solution "
246 << (verified ? "was verified." : "didn't pass verification.");
247 }
248
249 // If the solver is a MIP, print the number of nodes.
250 // TODO(user): add the number of nodes to the response, and move this code
251 // to the main Run().
252 if (SolverTypeIsMip(request_proto.solver_type())) {
253 absl::PrintF("%-12s: %d\n", "Nodes", solver.nodes());
254 }
255
256 // Fill and return the response proto.
257 solver.FillSolutionResponseProto(&response);
258 return response;
259}

261void Run() {
262 QCHECK(!absl::GetFlag(FLAGS_input).empty()) << "--input is required";
263 QCHECK_GE(absl::GetFlag(FLAGS_time_limit), absl::ZeroDuration())
264 << "--time_limit must be given a positive duration";
265
266 MPSolver::OptimizationProblemType type;
267 CHECK(MPSolver::ParseSolverType(absl::GetFlag(FLAGS_solver), &type))
268 << "Unsupported --solver: " << absl::GetFlag(FLAGS_solver);
269
270 MPModelRequest request_proto = ReadMipModel(absl::GetFlag(FLAGS_input));
271
272 if (!absl::GetFlag(FLAGS_sol_hint).empty()) {
273 const auto read_sol =
274 ParseSolFile(absl::GetFlag(FLAGS_sol_hint), request_proto.model());
275 CHECK_OK(read_sol.status());
276 const MPSolutionResponse sol = read_sol.value();
277 if (request_proto.model().has_solution_hint()) {
278 LOG(WARNING) << "Overwriting solution hint found in the request with "
279 << "solution from " << absl::GetFlag(FLAGS_sol_hint);
280 }
281 request_proto.mutable_model()->clear_solution_hint();
282 for (int i = 0; i < sol.variable_value_size(); ++i) {
283 request_proto.mutable_model()->mutable_solution_hint()->add_var_index(i);
284 request_proto.mutable_model()->mutable_solution_hint()->add_var_value(
285 sol.variable_value(i));
286 }
287 }
288
289 printf("%-12s: '%s'\n", "File", absl::GetFlag(FLAGS_input).c_str());
290
291 // Detect format to dump protos.
292 operations_research::ProtoWriteFormat write_format;
293 if (absl::GetFlag(FLAGS_dump_format) == "text") {
294 write_format = ProtoWriteFormat::kProtoText;
295 } else if (absl::GetFlag(FLAGS_dump_format) == "binary") {
296 write_format = ProtoWriteFormat::kProtoBinary;
297 } else if (absl::GetFlag(FLAGS_dump_format) == "json") {
298 write_format = ProtoWriteFormat::kJson;
299 } else {
300 LOG(FATAL) << "Unsupported --dump_format: "
301 << absl::GetFlag(FLAGS_dump_format);
302 }
303
304 // Set or override request proto options from the command line flags.
305 request_proto.set_solver_type(static_cast<MPModelRequest::SolverType>(type));
306 if (absl::GetFlag(FLAGS_time_limit) != absl::InfiniteDuration()) {
307 LOG(INFO) << "Setting a time limit of " << absl::GetFlag(FLAGS_time_limit);
308 request_proto.set_solver_time_limit_seconds(
309 absl::ToDoubleSeconds(absl::GetFlag(FLAGS_time_limit)));
310 }
311 if (absl::GetFlag(FLAGS_linear_solver_enable_verbose_output)) {
312 request_proto.set_enable_internal_solver_output(true);
313 }
314 if (!absl::GetFlag(FLAGS_params_file).empty()) {
315 CHECK(absl::GetFlag(FLAGS_params).empty())
316 << "--params and --params_file are incompatible";
317 std::string file_contents;
318 CHECK_OK(file::GetContents(absl::GetFlag(FLAGS_params_file), &file_contents,
319 file::Defaults()))
320 << "Could not read parameters file.";
321 request_proto.set_solver_specific_parameters(file_contents);
322 }
323 if (!absl::GetFlag(FLAGS_params).empty()) {
324 request_proto.set_solver_specific_parameters(absl::GetFlag(FLAGS_params));
325 }
326
327 // If requested, save the model and/or request to file.
328 if (!absl::GetFlag(FLAGS_dump_model).empty()) {
329 CHECK(WriteProtoToFile(absl::GetFlag(FLAGS_dump_model),
330 request_proto.model(), write_format,
331 absl::GetFlag(FLAGS_dump_gzip)));
332 }
333 if (!absl::GetFlag(FLAGS_dump_request).empty()) {
334 CHECK(WriteProtoToFile(absl::GetFlag(FLAGS_dump_request), request_proto,
335 write_format, absl::GetFlag(FLAGS_dump_gzip)));
336 }
337
338 absl::PrintF(
339 "%-12s: %s\n", "Solver",
340 MPModelRequest::SolverType_Name(request_proto.solver_type()).c_str());
341 absl::PrintF("%-12s: %s\n", "Parameters", absl::GetFlag(FLAGS_params));
342 absl::PrintF("%-12s: %d x %d\n", "Dimension",
343 request_proto.model().constraint_size(),
344 request_proto.model().variable_size());
345
346 const absl::Time solve_start_time = absl::Now();
347
348 const MPSolutionResponse response = LocalSolve(request_proto);
349
350 const absl::Duration solving_time = absl::Now() - solve_start_time;
351 const bool has_solution = response.status() == MPSOLVER_OPTIMAL ||
352 response.status() == MPSOLVER_FEASIBLE;
353 absl::PrintF("%-12s: %s\n", "Status",
354 MPSolverResponseStatus_Name(
355 static_cast<MPSolverResponseStatus>(response.status()))
356 .c_str());
357 absl::PrintF("%-12s: %15.15e\n", "Objective",
358 has_solution ? response.objective_value() : 0.0);
359 absl::PrintF("%-12s: %15.15e\n", "BestBound",
360 has_solution ? response.best_objective_bound() : 0.0);
361 absl::PrintF("%-12s: %s\n", "StatusString", response.status_str());
362 absl::PrintF("%-12s: %-6.4g s\n", "Time",
363 absl::ToDoubleSeconds(solving_time));
364
365 // If requested, write the solution, in .sol format (--sol_file), proto
366 // format and/or csv format.
367 if (!absl::GetFlag(FLAGS_sol_file).empty() && has_solution) {
368 std::string sol_string;
369 absl::StrAppend(&sol_string, "=obj= ", response.objective_value(), "\n");
370 for (int i = 0; i < response.variable_value().size(); ++i) {
371 absl::StrAppend(&sol_string, request_proto.model().variable(i).name(),
372 " ", response.variable_value(i), "\n");
373 }
374 LOG(INFO) << "Writing .sol solution to '" << absl::GetFlag(FLAGS_sol_file)
375 << "'.\n";
376 CHECK_OK(file::SetContents(absl::GetFlag(FLAGS_sol_file), sol_string,
377 file::Defaults()));
378 }
379 if (!absl::GetFlag(FLAGS_dump_response).empty() && has_solution) {
380 CHECK(WriteProtoToFile(absl::GetFlag(FLAGS_dump_response), response,
381 write_format, absl::GetFlag(FLAGS_dump_gzip)));
382 }
383 if (!absl::GetFlag(FLAGS_output_csv).empty() && has_solution) {
384 std::string csv_file;
385 for (int i = 0; i < response.variable_value_size(); ++i) {
386 csv_file +=
387 absl::StrFormat("%s,%e\n", request_proto.model().variable(i).name(),
388 response.variable_value(i));
389 }
390 CHECK_OK(file::SetContents(absl::GetFlag(FLAGS_output_csv), csv_file,
391 file::Defaults()));
392 }
393}
