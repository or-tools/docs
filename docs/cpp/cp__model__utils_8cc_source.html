42void SetToNegatedLinearExpression(const LinearExpressionProto& input_expr,
43 LinearExpressionProto* output_negated_expr) {
44 output_negated_expr->Clear();
45 for (int i = 0; i < input_expr.vars_size(); ++i) {
46 output_negated_expr->add_vars(NegatedRef(input_expr.vars(i)));
47 output_negated_expr->add_coeffs(input_expr.coeffs(i));
48 }
49 output_negated_expr->set_offset(-input_expr.offset());
50}

52IndexReferences GetReferencesUsedByConstraint(const ConstraintProto& ct) {
53 IndexReferences output;
54 switch (ct.constraint_case()) {
55 case ConstraintProto::ConstraintCase::kBoolOr:
56 AddIndices(ct.bool_or().literals(), &output.literals);
57 break;
58 case ConstraintProto::ConstraintCase::kBoolAnd:
59 AddIndices(ct.bool_and().literals(), &output.literals);
60 break;
61 case ConstraintProto::ConstraintCase::kAtMostOne:
62 AddIndices(ct.at_most_one().literals(), &output.literals);
63 break;
64 case ConstraintProto::ConstraintCase::kExactlyOne:
65 AddIndices(ct.exactly_one().literals(), &output.literals);
66 break;
67 case ConstraintProto::ConstraintCase::kBoolXor:
68 AddIndices(ct.bool_xor().literals(), &output.literals);
69 break;
70 case ConstraintProto::ConstraintCase::kIntDiv:
71 AddIndices(ct.int_div().target().vars(), &output.variables);
72 for (const LinearExpressionProto& expr : ct.int_div().exprs()) {
73 AddIndices(expr.vars(), &output.variables);
74 }
75 break;
76 case ConstraintProto::ConstraintCase::kIntMod:
77 AddIndices(ct.int_mod().target().vars(), &output.variables);
78 for (const LinearExpressionProto& expr : ct.int_mod().exprs()) {
79 AddIndices(expr.vars(), &output.variables);
80 }
81 break;
82 case ConstraintProto::ConstraintCase::kLinMax: {
83 AddIndices(ct.lin_max().target().vars(), &output.variables);
84 for (const LinearExpressionProto& expr : ct.lin_max().exprs()) {
85 AddIndices(expr.vars(), &output.variables);
86 }
87 break;
88 }
89 case ConstraintProto::ConstraintCase::kIntProd:
90 AddIndices(ct.int_prod().target().vars(), &output.variables);
91 for (const LinearExpressionProto& expr : ct.int_prod().exprs()) {
92 AddIndices(expr.vars(), &output.variables);
93 }
94 break;
95 case ConstraintProto::ConstraintCase::kLinear:
96 AddIndices(ct.linear().vars(), &output.variables);
97 break;
98 case ConstraintProto::ConstraintCase::kAllDiff:
99 for (const LinearExpressionProto& expr : ct.all_diff().exprs()) {
100 AddIndices(expr.vars(), &output.variables);
101 }
102 break;
103 case ConstraintProto::ConstraintCase::kDummyConstraint:
104 AddIndices(ct.dummy_constraint().vars(), &output.variables);
105 break;
106 case ConstraintProto::ConstraintCase::kElement:
107 output.variables.push_back(ct.element().index());
108 output.variables.push_back(ct.element().target());
109 AddIndices(ct.element().vars(), &output.variables);
110 break;
111 case ConstraintProto::ConstraintCase::kCircuit:
112 AddIndices(ct.circuit().literals(), &output.literals);
113 break;
114 case ConstraintProto::ConstraintCase::kRoutes:
115 AddIndices(ct.routes().literals(), &output.literals);
116 break;
117 case ConstraintProto::ConstraintCase::kInverse:
118 AddIndices(ct.inverse().f_direct(), &output.variables);
119 AddIndices(ct.inverse().f_inverse(), &output.variables);
120 break;
121 case ConstraintProto::ConstraintCase::kReservoir:
122 for (const LinearExpressionProto& time : ct.reservoir().time_exprs()) {
123 AddIndices(time.vars(), &output.variables);
124 }
125 for (const LinearExpressionProto& level :
126 ct.reservoir().level_changes()) {
127 AddIndices(level.vars(), &output.variables);
128 }
129 AddIndices(ct.reservoir().active_literals(), &output.literals);
130 break;
131 case ConstraintProto::ConstraintCase::kTable:
132 AddIndices(ct.table().vars(), &output.variables);
133 break;
134 case ConstraintProto::ConstraintCase::kAutomaton:
135 AddIndices(ct.automaton().vars(), &output.variables);
136 break;
137 case ConstraintProto::ConstraintCase::kInterval:
138 AddIndices(ct.interval().start().vars(), &output.variables);
139 AddIndices(ct.interval().size().vars(), &output.variables);
140 AddIndices(ct.interval().end().vars(), &output.variables);
141 break;
142 case ConstraintProto::ConstraintCase::kNoOverlap:
143 break;
144 case ConstraintProto::ConstraintCase::kNoOverlap2D:
145 break;
146 case ConstraintProto::ConstraintCase::kCumulative:
147 AddIndices(ct.cumulative().capacity().vars(), &output.variables);
148 for (const LinearExpressionProto& demand : ct.cumulative().demands()) {
149 AddIndices(demand.vars(), &output.variables);
150 }
151 break;
152 case ConstraintProto::ConstraintCase::CONSTRAINT_NOT_SET:
153 break;
154 }
155 return output;
156}

256 }
257 break;
258 case ConstraintProto::ConstraintCase::kLinMax:
259 APPLY_TO_REPEATED_FIELD(lin_max, target()->mutable_vars);
260 for (int i = 0; i < ct->lin_max().exprs_size(); ++i) {
261 APPLY_TO_REPEATED_FIELD(lin_max, exprs(i)->mutable_vars);
262 }
263 break;
264 case ConstraintProto::ConstraintCase::kIntProd:
265 APPLY_TO_REPEATED_FIELD(int_prod, target()->mutable_vars);
266 for (int i = 0; i < ct->int_prod().exprs_size(); ++i) {
267 APPLY_TO_REPEATED_FIELD(int_prod, exprs(i)->mutable_vars);
268 }
269 break;
270 case ConstraintProto::ConstraintCase::kLinear:
271 APPLY_TO_REPEATED_FIELD(linear, vars);
272 break;
273 case ConstraintProto::ConstraintCase::kAllDiff:
274 for (int i = 0; i < ct->all_diff().exprs_size(); ++i) {
275 APPLY_TO_REPEATED_FIELD(all_diff, exprs(i)->mutable_vars);
276 }
277 break;
278 case ConstraintProto::ConstraintCase::kDummyConstraint:
279 APPLY_TO_REPEATED_FIELD(dummy_constraint, vars);
280 break;
281 case ConstraintProto::ConstraintCase::kElement:
282 APPLY_TO_SINGULAR_FIELD(element, index);
283 APPLY_TO_SINGULAR_FIELD(element, target);
284 APPLY_TO_REPEATED_FIELD(element, vars);
285 break;
286 case ConstraintProto::ConstraintCase::kCircuit:
287 break;
288 case ConstraintProto::ConstraintCase::kRoutes:
289 break;
290 case ConstraintProto::ConstraintCase::kInverse:
291 APPLY_TO_REPEATED_FIELD(inverse, f_direct);
292 APPLY_TO_REPEATED_FIELD(inverse, f_inverse);
293 break;

296 APPLY_TO_REPEATED_FIELD(reservoir, time_exprs(i)->mutable_vars);
297 }
298 for (int i = 0; i < ct->reservoir().level_changes_size(); ++i) {
299 APPLY_TO_REPEATED_FIELD(reservoir, level_changes(i)->mutable_vars);
300 }

304 break;
305 case ConstraintProto::ConstraintCase::kAutomaton:
306 APPLY_TO_REPEATED_FIELD(automaton, vars);
307 break;
308 case ConstraintProto::ConstraintCase::kInterval:
309 APPLY_TO_REPEATED_FIELD(interval, start()->mutable_vars);
310 APPLY_TO_REPEATED_FIELD(interval, size()->mutable_vars);
311 APPLY_TO_REPEATED_FIELD(interval, end()->mutable_vars);

314 break;
315 case ConstraintProto::ConstraintCase::kNoOverlap2D:
316 break;
317 case ConstraintProto::ConstraintCase::kCumulative:
318 APPLY_TO_REPEATED_FIELD(cumulative, capacity()->mutable_vars);

320 for (int& r :
321 *ct->mutable_cumulative()->mutable_demands(i)->mutable_vars()) {
322 f(&r);
323 }
324 }
325 break;
326 case ConstraintProto::ConstraintCase::CONSTRAINT_NOT_SET:

446std::vector<int> UsedVariables(const ConstraintProto& ct) {
447 IndexReferences references = GetReferencesUsedByConstraint(ct);
448 for (int& ref : references.variables) {
449 ref = PositiveRef(ref);
450 }
451 for (const int lit : references.literals) {
452 references.variables.push_back(PositiveRef(lit));
453 }
454 for (const int lit : ct.enforcement_literal()) {
455 references.variables.push_back(PositiveRef(lit));
456 }
457 gtl::STLSortAndRemoveDuplicates(&references.variables);
458 return references.variables;
459}

521int64_t ComputeInnerObjective(const CpObjectiveProto& objective,
522 absl::Span<const int64_t> solution) {
523 int64_t objective_value = 0;
524 for (int i = 0; i < objective.vars_size(); ++i) {
525 int64_t coeff = objective.coeffs(i);
526 const int ref = objective.vars(i);
527 const int var = PositiveRef(ref);
528 if (!RefIsPositive(ref)) coeff = -coeff;
529 objective_value += coeff * solution[var];
530 }
531 return objective_value;
532}

534bool ExpressionContainsSingleRef(const LinearExpressionProto& expr) {
535 return expr.offset() == 0 && expr.vars_size() == 1 &&
536 std::abs(expr.coeffs(0)) == 1;
537}
538
539bool ExpressionIsAffine(const LinearExpressionProto& expr) {
540 return expr.vars_size() <= 1;
541}

543// Returns the reference the expression can be reduced to. It will DCHECK that
544// ExpressionContainsSingleRef(expr) is true.
545int GetSingleRefFromExpression(const LinearExpressionProto& expr) {
546 DCHECK(ExpressionContainsSingleRef(expr));
547 return expr.coeffs(0) == 1 ? expr.vars(0) : NegatedRef(expr.vars(0));
548}

550void AddLinearExpressionToLinearConstraint(const LinearExpressionProto& expr,
551 int64_t coefficient,
552 LinearConstraintProto* linear) {
553 for (int i = 0; i < expr.vars_size(); ++i) {
554 linear->add_vars(expr.vars(i));
555 linear->add_coeffs(expr.coeffs(i) * coefficient);
556 }
557 DCHECK(!linear->domain().empty());
558 const int64_t shift = coefficient * expr.offset();
559 if (shift != 0) {
560 FillDomainInProto(ReadDomainFromProto(*linear).AdditionWith(Domain(-shift)),
561 linear);
562 }
563}

565bool LinearExpressionProtosAreEqual(const LinearExpressionProto& a,
566 const LinearExpressionProto& b,
567 int64_t b_scaling) {
568 if (a.vars_size() != b.vars_size()) return false;
569 if (a.offset() != b.offset() * b_scaling) return false;
570 absl::flat_hash_map<int, int64_t> coeffs;
571 for (int i = 0; i < a.vars_size(); ++i) {
572 coeffs[a.vars(i)] += a.coeffs(i);
573 coeffs[b.vars(i)] += -b.coeffs(i) * b_scaling;
574 }
575
576 for (const auto [var, coeff] : coeffs) {
577 if (coeff != 0) return false;
578 }
579 return true;
580}

582uint64_t FingerprintExpression(const LinearExpressionProto& lin,
583 uint64_t seed) {
584 uint64_t fp = seed;
585 if (!lin.vars().empty()) {
586 fp = FingerprintRepeatedField(lin.vars(), fp);
587 fp = FingerprintRepeatedField(lin.coeffs(), fp);
588 }
589 fp = FingerprintSingleField(lin.offset(), fp);
590 return fp;
591}

593uint64_t FingerprintModel(const CpModelProto& model, uint64_t seed) {
594 uint64_t fp = seed;
595 for (const IntegerVariableProto& var_proto : model.variables()) {
596 fp = FingerprintRepeatedField(var_proto.domain(), fp);
597 }
598 for (const ConstraintProto& ct : model.constraints()) {
599 if (!ct.enforcement_literal().empty()) {
600 fp = FingerprintRepeatedField(ct.enforcement_literal(), fp);
601 }
602 switch (ct.constraint_case()) {
603 case ConstraintProto::ConstraintCase::kBoolOr:
604 fp = FingerprintRepeatedField(ct.bool_or().literals(), fp);
605 break;
606 case ConstraintProto::ConstraintCase::kBoolAnd:
607 fp = FingerprintRepeatedField(ct.bool_and().literals(), fp);
608 break;
609 case ConstraintProto::ConstraintCase::kAtMostOne:
610 fp = FingerprintRepeatedField(ct.at_most_one().literals(), fp);
611 break;
612 case ConstraintProto::ConstraintCase::kExactlyOne:
613 fp = FingerprintRepeatedField(ct.exactly_one().literals(), fp);
614 break;
615 case ConstraintProto::ConstraintCase::kBoolXor:
616 fp = FingerprintRepeatedField(ct.bool_xor().literals(), fp);
617 break;
618 case ConstraintProto::ConstraintCase::kIntDiv:
619 fp = FingerprintExpression(ct.int_div().target(), fp);
620 for (const LinearExpressionProto& expr : ct.int_div().exprs()) {
621 fp = FingerprintExpression(expr, fp);
622 }
623 break;
624 case ConstraintProto::ConstraintCase::kIntMod:
625 fp = FingerprintExpression(ct.int_mod().target(), fp);
626 for (const LinearExpressionProto& expr : ct.int_mod().exprs()) {
627 fp = FingerprintExpression(expr, fp);
628 }
629 break;
630 case ConstraintProto::ConstraintCase::kLinMax: {
631 fp = FingerprintExpression(ct.lin_max().target(), fp);
632 for (const LinearExpressionProto& expr : ct.lin_max().exprs()) {
633 fp = FingerprintExpression(expr, fp);
634 }
635 break;
636 }
637 case ConstraintProto::ConstraintCase::kIntProd:
638 fp = FingerprintExpression(ct.int_prod().target(), fp);
639 for (const LinearExpressionProto& expr : ct.int_prod().exprs()) {
640 fp = FingerprintExpression(expr, fp);
641 }
642 break;
643 case ConstraintProto::ConstraintCase::kLinear:
644 fp = FingerprintRepeatedField(ct.linear().vars(), fp);
645 fp = FingerprintRepeatedField(ct.linear().coeffs(), fp);
646 fp = FingerprintRepeatedField(ct.linear().domain(), fp);
647 break;
648 case ConstraintProto::ConstraintCase::kAllDiff:
649 for (const LinearExpressionProto& expr : ct.all_diff().exprs()) {
650 fp = FingerprintExpression(expr, fp);
651 }
652 break;
653 case ConstraintProto::ConstraintCase::kDummyConstraint:
654 break;
655 case ConstraintProto::ConstraintCase::kElement:
656 fp = FingerprintSingleField(ct.element().index(), fp);
657 fp = FingerprintSingleField(ct.element().target(), fp);
658 fp = FingerprintRepeatedField(ct.element().vars(), fp);
659 break;
660 case ConstraintProto::ConstraintCase::kCircuit:
661 fp = FingerprintRepeatedField(ct.circuit().heads(), fp);
662 fp = FingerprintRepeatedField(ct.circuit().tails(), fp);
663 fp = FingerprintRepeatedField(ct.circuit().literals(), fp);
664 break;
665 case ConstraintProto::ConstraintCase::kRoutes:
666 fp = FingerprintRepeatedField(ct.routes().heads(), fp);
667 fp = FingerprintRepeatedField(ct.routes().tails(), fp);
668 fp = FingerprintRepeatedField(ct.routes().literals(), fp);
669 break;
670 case ConstraintProto::ConstraintCase::kInverse:
671 fp = FingerprintRepeatedField(ct.inverse().f_direct(), fp);
672 fp = FingerprintRepeatedField(ct.inverse().f_inverse(), fp);
673 break;
674 case ConstraintProto::ConstraintCase::kReservoir:
675 fp = FingerprintSingleField(ct.reservoir().min_level(), fp);
676 fp = FingerprintSingleField(ct.reservoir().max_level(), fp);
677 for (const LinearExpressionProto& expr : ct.reservoir().time_exprs()) {
678 fp = FingerprintExpression(expr, fp);
679 }
680 for (const LinearExpressionProto& expr :
681 ct.reservoir().level_changes()) {
682 fp = FingerprintExpression(expr, fp);
683 }
684 break;
685 case ConstraintProto::ConstraintCase::kTable:
686 fp = FingerprintRepeatedField(ct.table().vars(), fp);
687 fp = FingerprintRepeatedField(ct.table().values(), fp);
688 fp = FingerprintSingleField(ct.table().negated(), fp);
689 break;
690 case ConstraintProto::ConstraintCase::kAutomaton:
691 fp = FingerprintSingleField(ct.automaton().starting_state(), fp);
692 fp = FingerprintRepeatedField(ct.automaton().final_states(), fp);
693 fp = FingerprintRepeatedField(ct.automaton().transition_tail(), fp);
694 fp = FingerprintRepeatedField(ct.automaton().transition_head(), fp);
695 fp = FingerprintRepeatedField(ct.automaton().transition_label(), fp);
696 fp = FingerprintRepeatedField(ct.automaton().vars(), fp);
697 break;
698 case ConstraintProto::ConstraintCase::kInterval:
699 fp = FingerprintExpression(ct.interval().start(), fp);
700 fp = FingerprintExpression(ct.interval().size(), fp);
701 fp = FingerprintExpression(ct.interval().end(), fp);
702 break;
703 case ConstraintProto::ConstraintCase::kNoOverlap:
704 fp = FingerprintRepeatedField(ct.no_overlap().intervals(), fp);
705 break;
706 case ConstraintProto::ConstraintCase::kNoOverlap2D:
707 fp = FingerprintRepeatedField(ct.no_overlap_2d().x_intervals(), fp);
708 fp = FingerprintRepeatedField(ct.no_overlap_2d().y_intervals(), fp);
709 break;
710 case ConstraintProto::ConstraintCase::kCumulative:
711 fp = FingerprintRepeatedField(ct.cumulative().intervals(), fp);
712 fp = FingerprintExpression(ct.cumulative().capacity(), fp);
713 for (const LinearExpressionProto& demand : ct.cumulative().demands()) {
714 fp = FingerprintExpression(demand, fp);
715 }
716 break;
717 case ConstraintProto::ConstraintCase::CONSTRAINT_NOT_SET:
718 break;
719 }
720 }
721
722 // Fingerprint the objective.
723 if (model.has_objective()) {
724 fp = FingerprintRepeatedField(model.objective().vars(), fp);
725 fp = FingerprintRepeatedField(model.objective().coeffs(), fp);
726 fp = FingerprintSingleField(model.objective().offset(), fp);
727 fp = FingerprintSingleField(model.objective().scaling_factor(), fp);
728 fp = FingerprintRepeatedField(model.objective().domain(), fp);
729 } else if (model.has_floating_point_objective()) {
730 fp = FingerprintRepeatedField(model.floating_point_objective().vars(), fp);
731 fp =
732 FingerprintRepeatedField(model.floating_point_objective().coeffs(), fp);
733 fp = FingerprintSingleField(model.floating_point_objective().offset(), fp);
734 fp =
735 FingerprintSingleField(model.floating_point_objective().maximize(), fp);
736 }
737
738 if (model.has_solution_hint()) {
739 fp = FingerprintRepeatedField(model.solution_hint().vars(), fp);
740 fp = FingerprintRepeatedField(model.solution_hint().values(), fp);
741 }
742
743 // TODO(user): Should we fingerprint decision strategies?
744
745 return fp;
746}
