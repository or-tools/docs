51// We restart the shared tree 10 times after 2 restarts per worker. After that
52// we restart when the tree reaches the maximum allowable number of nodes, but
53// still at most once per 2 restarts per worker.
54const int kSyncsPerWorkerPerRestart = 2;
55const int kNumInitialRestarts = 10;
56
57// If you build a tree by expanding the nodes with minimal depth+discrepancy,
58// the number of leaves when all nodes with a given value have been split
59// follows the fibonacci sequence:
60// num_leaves(0) := 2;
61// num_leaves(1) := 3;
62// num_leaves(n) := num_leaves(n-1) + num_leaves(n-2)
63// This function returns f(n) := min({i | num_leaves(i) >= n})
64int MaxAllowedDiscrepancyPlusDepth(int num_leaves) {
65 int i = 0;
66 int a = 1;
67 int b = 2;
68 while (b < num_leaves) {
69 std::tie(a, b) = std::make_pair(b, a + b);
70 ++i;
71 }
72 return i;
73}

76Literal ProtoLiteral::Decode(CpModelMapping* mapping,
77 IntegerEncoder* encoder) const {
78 DCHECK_LT(proto_var_, mapping->NumProtoVariables());
79 if (mapping->IsBoolean(proto_var_)) {
80 return mapping->Literal(proto_var_);
81 }
82 return encoder->GetOrCreateAssociatedLiteral(DecodeInteger(mapping));
83}
84
85IntegerLiteral ProtoLiteral::DecodeInteger(CpModelMapping* mapping) const {
86 const int positive_var = PositiveRef(proto_var_);
87 if (!mapping->IsInteger(positive_var)) {
88 return IntegerLiteral();
89 }
90 if (proto_var_ < 0) {
91 return IntegerLiteral::LowerOrEqual(mapping->Integer(positive_var), -lb_);
92 }
93 return IntegerLiteral::GreaterOrEqual(mapping->Integer(positive_var), lb_);
94}

96std::optional<ProtoLiteral> ProtoLiteral::EncodeInteger(
97 IntegerLiteral literal, CpModelMapping* mapping) {
98 IntegerVariable positive_var = PositiveVariable(literal.var);
99 const int model_var =
100 mapping->GetProtoVariableFromIntegerVariable(positive_var);
101 if (model_var == -1) {
102 return std::nullopt;
103 }
104 ProtoLiteral result{
105 literal.var == positive_var ? model_var : NegatedRef(model_var),
106 literal.bound};
107 DCHECK_EQ(result.DecodeInteger(mapping), literal);
108 DCHECK_EQ(result.Negated().DecodeInteger(mapping), literal.Negated());
109 return result;
110}
111std::optional<ProtoLiteral> ProtoLiteral::Encode(Literal literal,
112 CpModelMapping* mapping,
113 IntegerEncoder* encoder) {
114 if (literal.Index() == kNoLiteralIndex) {
115 return std::nullopt;
116 }
117 int model_var =
118 mapping->GetProtoVariableFromBooleanVariable(literal.Variable());
119 if (model_var != -1) {
120 CHECK(mapping->IsBoolean(model_var));
121 ProtoLiteral result{
122 literal.IsPositive() ? model_var : NegatedRef(model_var),
123 literal.IsPositive() ? 1 : 0};
124 DCHECK_EQ(result.Decode(mapping, encoder), literal);
125 DCHECK_EQ(result.Negated().Decode(mapping, encoder), literal.Negated());
126 return result;
127 }
128 for (auto int_lit : encoder->GetIntegerLiterals(literal)) {
129 auto result = EncodeInteger(int_lit, mapping);
130 if (result.has_value()) {
131 DCHECK_EQ(result->DecodeInteger(mapping), int_lit);
132 DCHECK_EQ(result->Negated().DecodeInteger(mapping), int_lit.Negated());
133 return result;
134 }
135 }
136 return std::nullopt;
137}

139void ProtoTrail::PushLevel(const ProtoLiteral& decision,
140 IntegerValue objective_lb, int node_id) {
141 CHECK_GT(node_id, 0);
142 decision_indexes_.push_back(literals_.size());
143 literals_.push_back(decision);
144 node_ids_.push_back(node_id);
145 if (!level_to_objective_lbs_.empty()) {
146 objective_lb = std::max(level_to_objective_lbs_.back(), objective_lb);
147 }
148 level_to_objective_lbs_.push_back(objective_lb);
149}
150
151void ProtoTrail::SetLevelImplied(int level) {
152 DCHECK_GE(level, 1);
153 DCHECK_LE(level, decision_indexes_.size());
154 SetObjectiveLb(level - 1, ObjectiveLb(level));
155 decision_indexes_.erase(decision_indexes_.begin() + level - 1);
156 level_to_objective_lbs_.erase(level_to_objective_lbs_.begin() + level - 1);
157}
158
159void ProtoTrail::Clear() {
160 decision_indexes_.clear();
161 literals_.clear();
162 level_to_objective_lbs_.clear();
163 node_ids_.clear();
164}
165
166void ProtoTrail::SetObjectiveLb(int level, IntegerValue objective_lb) {
167 if (level == 0) return;
168 level_to_objective_lbs_[level - 1] =
169 std::max(objective_lb, level_to_objective_lbs_[level - 1]);
170}

172absl::Span<const int> ProtoTrail::NodeIds(int level) const {
173 DCHECK_LE(level, decision_indexes_.size());
174 int start = level == 0 ? 0 : decision_indexes_[level - 1];
175 int end = level == decision_indexes_.size() ? node_ids_.size()
176 : decision_indexes_[level];
177 return absl::MakeSpan(node_ids_.data() + start, end - start);
178}
179
180absl::Span<const ProtoLiteral> ProtoTrail::Implications(int level) const {
181 if (level > decision_indexes_.size()) {
182 return absl::MakeSpan(literals_.data(), 0);
183 }
184 int start = level == 0 ? 0 : decision_indexes_[level - 1] + 1;
185 int end = level == decision_indexes_.size() ? node_ids_.size()
186 : decision_indexes_[level];
187 return absl::MakeSpan(literals_.data() + start, end - start);
188}

190SharedTreeManager::SharedTreeManager(Model* model)
191 : params_(*model->GetOrCreate<SatParameters>()),
192 num_workers_(std::max(1, params_.shared_tree_num_workers())),
193 shared_response_manager_(model->GetOrCreate<SharedResponseManager>()),
194 num_splits_wanted_(num_workers_ - 1),
195 max_nodes_(params_.shared_tree_max_nodes_per_worker() >=
196 std::numeric_limits<int>::max() / num_workers_
197 ? std::numeric_limits<int>::max()
198 : num_workers_ *
199 params_.shared_tree_max_nodes_per_worker()) {
200 // Create the root node with a fake literal.
201 nodes_.push_back(
202 {.literal = ProtoLiteral(),
203 .objective_lb =
204 shared_response_manager_->GetInnerObjectiveLowerBound()});
205 unassigned_leaves_.reserve(num_workers_);
206 unassigned_leaves_.push_back(&nodes_.back());
207}

220bool SharedTreeManager::SyncTree(ProtoTrail& path) {
221 absl::MutexLock mutex_lock(&mu_);
222 std::vector<std::pair<Node*, int>> nodes = GetAssignedNodes(path);
223 if (!IsValid(path) || nodes.back().first->closed) {
224 path.Clear();
225 return false;
226 }
227 // We don't rely on these being empty, but we expect them to be.
228 DCHECK(to_close_.empty());
229 DCHECK(to_update_.empty());
230 int prev_level = -1;
231 for (const auto& [node, level] : nodes) {
232 if (level == prev_level) {
233 to_close_.push_back(GetSibling(node));
234 }
235 if (level > 0 && node->objective_lb < path.ObjectiveLb(level)) {
236 node->objective_lb = path.ObjectiveLb(level);
237 to_update_.push_back(node->parent);
238 }
239 prev_level = level;
240 }
241 ProcessNodeChanges();
242 // Restart after processing updates - we might learn a new objective bound.
243 if (++num_syncs_since_restart_ / num_workers_ > kSyncsPerWorkerPerRestart &&
244 (num_restarts_ < kNumInitialRestarts || nodes_.size() >= max_nodes_)) {
245 RestartLockHeld();
246 path.Clear();
247 return false;
248 }
249 return true;
250}

252void SharedTreeManager::ProposeSplit(ProtoTrail& path, ProtoLiteral decision) {
253 absl::MutexLock mutex_lock(&mu_);
254 if (!IsValid(path)) return;
255 std::vector<std::pair<Node*, int>> nodes = GetAssignedNodes(path);
256 if (nodes.back().first->children[0] != nullptr) {
257 LOG_IF(WARNING, nodes.size() > 1)
258 << "Cannot resplit previously split node @ " << nodes.back().second
259 << "/" << nodes.size();
260 return;
261 }
262 if (nodes_.size() >= max_nodes_) {
263 VLOG(1) << "Too many nodes to accept split";
264 return;
265 }
266 if (num_splits_wanted_ <= 0) {
267 VLOG(1) << "Enough splits for now";
268 return;
269 }
270 if (params_.shared_tree_split_strategy() ==
271 SatParameters::SPLIT_STRATEGY_DISCREPANCY ||
272 params_.shared_tree_split_strategy() ==
273 SatParameters::SPLIT_STRATEGY_AUTO) {
274 int discrepancy = 0;
275 for (const auto& [node, level] : nodes) {
276 if (node->parent == nullptr || node->implied) continue;
277 IntegerValue sibling_bound = GetSibling(node)->objective_lb;
278 discrepancy += (node->objective_lb == sibling_bound
279 ? node != node->parent->children[0]
280 : node->objective_lb > sibling_bound);
281 }
282 // TODO(user): Need to write up the shape this creates.
283 // This rule will allow twice as many leaves in the preferred subtree.
284 if (discrepancy + path.MaxLevel() >
285 MaxAllowedDiscrepancyPlusDepth(num_workers_)) {
286 VLOG(1) << "Too high discrepancy to accept split";
287 return;
288 }
289 } else if (params_.shared_tree_split_strategy() ==
290 SatParameters::SPLIT_STRATEGY_OBJECTIVE_LB) {
291 if (nodes.back().first->objective_lb > nodes.front().first->objective_lb) {
292 VLOG(1) << "Can only split nodes with minimum objective lb, "
293 << nodes.back().first->objective_lb << " > "
294 << nodes.front().first->objective_lb;
295 return;
296 }
297 } else if (params_.shared_tree_split_strategy() ==
298 SatParameters::SPLIT_STRATEGY_BALANCED_TREE) {
299 if (path.MaxLevel() + 1 > log2(num_workers_)) {
300 VLOG(1) << "Tree too unbalanced to accept split";
301 return;
302 }
303 }
304 VLOG_EVERY_N(1, 10) << unassigned_leaves_.size() << " unassigned leaves, "
305 << nodes_.size() << " subtrees, " << num_splits_wanted_
306 << " splits wanted";
307 Split(nodes, decision);
308 auto [new_leaf, level] = nodes.back();
309 path.PushLevel(new_leaf->literal, new_leaf->objective_lb, new_leaf->id);
310}

312void SharedTreeManager::ReplaceTree(ProtoTrail& path) {
313 absl::MutexLock mutex_lock(&mu_);
314 std::vector<std::pair<Node*, int>> nodes = GetAssignedNodes(path);
315 if (nodes.back().first->children[0] == nullptr &&
316 !nodes.back().first->closed && nodes.size() > 1) {
317 VLOG(1) << "Returning leaf to be replaced";
318 unassigned_leaves_.push_back(nodes.back().first);
319 }
320 path.Clear();
321 while (!unassigned_leaves_.empty()) {
322 const int i = num_leaves_assigned_++ % unassigned_leaves_.size();
323 std::swap(unassigned_leaves_[i], unassigned_leaves_.back());
324 Node* leaf = unassigned_leaves_.back();
325 unassigned_leaves_.pop_back();
326 if (!leaf->closed && leaf->children[0] == nullptr) {
327 AssignLeaf(path, leaf);
328 return;
329 }
330 }
331 VLOG(1) << "Assigning root because no unassigned leaves are available";
332 // TODO(user): Investigate assigning a random leaf so workers can still
333 // improve shared tree bounds.
334}

336SharedTreeManager::Node* SharedTreeManager::GetSibling(Node* node) {
337 if (node == nullptr || node->parent == nullptr) return nullptr;
338 if (node->parent->children[0] != node) {
339 return node->parent->children[0];
340 }
341 return node->parent->children[1];
342}

344void SharedTreeManager::Split(std::vector<std::pair<Node*, int>>& nodes,
345 ProtoLiteral lit) {
346 const auto [parent, level] = nodes.back();
347 DCHECK_EQ(parent->children[0], nullptr);
348 DCHECK_EQ(parent->children[1], nullptr);
349 parent->children[0] = MakeSubtree(parent, lit);
350 parent->children[1] = MakeSubtree(parent, lit.Negated());
351 nodes.push_back(std::make_pair(parent->children[0], level + 1));
352 unassigned_leaves_.push_back(parent->children[1]);
353 --num_splits_wanted_;
354}

356SharedTreeManager::Node* SharedTreeManager::MakeSubtree(Node* parent,
357 ProtoLiteral literal) {
358 nodes_.push_back(
359 Node{.literal = literal,
360 .objective_lb = parent->objective_lb,
361 .parent = parent,
362 .id = static_cast<int>(nodes_.size() + node_id_offset_)});
363 return &nodes_.back();
364}

366void SharedTreeManager::ProcessNodeChanges() {
367 while (!to_close_.empty() || !to_update_.empty()) {
368 while (!to_close_.empty()) {
369 Node* node = to_close_.back();
370 CHECK_NE(node, nullptr);
371 to_close_.pop_back();
372 if (node->closed) continue;
373 node->closed = true;
374 // If we are closing a leaf, try to maintain the same number of leaves;
375 num_splits_wanted_ += (node->children[0] == nullptr);
376 for (Node* child : node->children) {
377 if (child == nullptr || child->closed) continue;
378 to_close_.push_back(child);
379 }
380 if (node->parent != nullptr) {
381 to_update_.push_back(node->parent);
382 GetSibling(node)->implied = true;
383 } else {
384 shared_response_manager_->NotifyThatImprovingProblemIsInfeasible(
385 ShortStatus());
386 }
387 }
388 if (to_update_.empty()) break;
389 Node* node = to_update_.back();
390 to_update_.pop_back();
391 while (node != nullptr && !node->closed && node->children[0] != nullptr) {
392 bool has_open_child = false;
393 IntegerValue child_bound = kMaxIntegerValue;
394 for (const Node* child : node->children) {
395 if (child->closed) continue;
396 has_open_child = true;
397 child_bound = std::min(child->objective_lb, child_bound);
398 }
399 if (!has_open_child) {
400 to_close_.push_back(node);
401 } else if (child_bound > node->objective_lb) {
402 node->objective_lb = child_bound;
403 if (node->parent == nullptr) {
404 IntegerValue root_lb =
405 shared_response_manager_->GetInnerObjectiveLowerBound();
406 if (node->objective_lb > root_lb) {
407 shared_response_manager_->UpdateInnerObjectiveBounds(
408 ShortStatus(), node->objective_lb, kMaxIntegerValue);
409 } else {
410 node->objective_lb = root_lb;
411 }
412 }
413 } else {
414 break;
415 }
416 node = node->parent;
417 }
418 }
419}

421std::vector<std::pair<SharedTreeManager::Node*, int>>
422SharedTreeManager::GetAssignedNodes(const ProtoTrail& path) {
423 std::vector<std::pair<Node*, int>> nodes({std::make_pair(&nodes_[0], 0)});
424 if (!IsValid(path)) {
425 // Restart has happened, nodes in this path are no longer valid, but the
426 // root is equivalent.
427 return nodes;
428 }
429 for (int i = 0; i <= path.MaxLevel(); ++i) {
430 for (int id : path.NodeIds(i)) {
431 const int index = id - node_id_offset_;
432 CHECK_GE(index, 0) << " in path.NodeIds(" << i
433 << "), max_level=" << path.MaxLevel();
434 CHECK_LT(index, nodes_.size());
435 DCHECK_EQ(nodes.back().first, nodes_[index].parent);
436 nodes.push_back(std::make_pair(&nodes_[index], i));
437 }
438 }
439 return nodes;
440}

442void SharedTreeManager::CloseTree(ProtoTrail& path, int level) {
443 absl::MutexLock mutex_lock(&mu_);
444 const int node_id_to_close = path.NodeIds(level).front();
445 path.Clear();
446 if (node_id_to_close < node_id_offset_) return;
447 Node* node = &nodes_[node_id_to_close - node_id_offset_];
448 VLOG(1) << "Closing subtree at level " << level;
449 DCHECK(to_close_.empty());
450 to_close_.push_back(node);
451 ProcessNodeChanges();
452}

454void SharedTreeManager::AssignLeaf(ProtoTrail& path, Node* leaf) {
455 path.Clear();
456 std::vector<Node*> reversed_path;
457 while (leaf != &nodes_[0]) {
458 reversed_path.push_back(&nodes_[leaf->id - node_id_offset_]);
459 leaf = leaf->parent;
460 }
461 while (!reversed_path.empty()) {
462 Node* leaf = reversed_path.back();
463 reversed_path.pop_back();
464 path.PushLevel(leaf->literal, leaf->objective_lb, leaf->id);
465 if (leaf->implied) {
466 path.SetLevelImplied(path.MaxLevel());
467 }
468 }
469}

471bool SharedTreeManager::IsValid(const ProtoTrail& path) const {
472 auto node_ids = path.NodeIds(path.MaxLevel());
473 if (node_ids.empty()) return true;
474 if (node_ids.back() < node_id_offset_) return false;
475 return true;
476}

478void SharedTreeManager::RestartLockHeld() {
479 node_id_offset_ += nodes_.size();
480 nodes_.resize(1);
481 nodes_[0].id = node_id_offset_;
482 nodes_[0].children = {nullptr, nullptr};
483 unassigned_leaves_.clear();
484 num_splits_wanted_ = num_workers_ - 1;
485 num_restarts_ += 1;
486 num_syncs_since_restart_ = 0;
487}

518bool SharedTreeWorker::AddDecisionImplication(Literal lit, int level) {
519 CHECK_NE(lit.Index(), kNoLiteralIndex);
520 CHECK(!sat_solver_->Assignment().LiteralIsTrue(lit));
521 if (sat_solver_->Assignment().LiteralIsFalse(lit)) {
522 VLOG(1) << "Closing subtree via impl at " << level + 1
523 << " assigned=" << assigned_tree_.MaxLevel();
524 integer_trail_->ReportConflict(DecisionReason(level), {});
525 manager_->CloseTree(assigned_tree_, level);
526 assigned_tree_literals_.clear();
527 return false;
528 }
529 integer_trail_->EnqueueLiteral(lit, DecisionReason(level), {});
530 VLOG(1) << "Learned shared clause";
531 return true;
532}
533
534bool SharedTreeWorker::AddImplications(
535 absl::Span<const ProtoLiteral> implied_literals) {
536 const int level = sat_solver_->CurrentDecisionLevel();
537 // Level 0 implications are unit clauses and are synced elsewhere.
538 if (level == 0) return false;
539 if (level > assigned_tree_.MaxLevel()) {
540 return false;
541 }
542 bool added_clause = false;
543 for (const ProtoLiteral& impl : implied_literals) {
544 Literal lit(DecodeDecision(impl));
545 if (sat_solver_->Assignment().LiteralIsTrue(lit)) continue;
546 added_clause = true;
547 if (!AddDecisionImplication(lit, level)) return true;
548 }
549 if (objective_ != nullptr &&
550 objective_->objective_var != kNoIntegerVariable) {
551 const IntegerValue obj_lb =
552 integer_trail_->LowerBound(objective_->objective_var);
553 assigned_tree_.SetObjectiveLb(level, obj_lb);
554 const Literal obj_lit =
555 encoder_->GetOrCreateAssociatedLiteral(IntegerLiteral::GreaterOrEqual(
556 objective_->objective_var, assigned_tree_.ObjectiveLb(level)));
557 if (!sat_solver_->Assignment().LiteralIsTrue(obj_lit)) {
558 AddDecisionImplication(obj_lit, level);
559 return true;
560 }
561 }
562 return added_clause;
563}

565bool SharedTreeWorker::SyncWithLocalTrail() {
566 while (true) {
567 if (!sat_solver_->FinishPropagation()) return false;
568 // Ensure we are at fixed point w.r.t. implications in the tree up to the
569 // current level.
570 if (AddImplications(
571 assigned_tree_.Implications(sat_solver_->CurrentDecisionLevel()))) {
572 continue;
573 }
574 if (!helper_->BeforeTakingDecision()) return false;
575 const int level = sat_solver_->CurrentDecisionLevel();
576 if (level >= assigned_tree_.MaxLevel()) break;
577 if (level == assigned_tree_.MaxLevel()) break;
578 // The next decision is assigned, make sure it makes sense.
579 const Literal next_decision = assigned_tree_literals_[level];
580 if (!sat_solver_->Assignment().LiteralIsAssigned(next_decision)) break;
581 if (sat_solver_->Assignment().LiteralIsFalse(next_decision)) {
582 // Next assigned decision is impossible.
583 VLOG(1) << "Closing subtree at " << level + 1
584 << " assigned=" << assigned_tree_.MaxLevel();
585 manager_->CloseTree(assigned_tree_, level + 1);
586 assigned_tree_literals_.clear();
587 } else {
588 // The next level is implied by the current one.
589 assigned_tree_.SetLevelImplied(level + 1);
590 assigned_tree_literals_.erase(assigned_tree_literals_.begin() + level);
591 }
592 }
593 return true;
594}

596LiteralIndex SharedTreeWorker::NextDecision() {
597 const auto& decision_policy =
598 heuristics_->decision_policies[heuristics_->policy_index];
599 const int next_level = sat_solver_->CurrentDecisionLevel() + 1;
600 CHECK_EQ(assigned_tree_literals_.size(), assigned_tree_.MaxLevel());
601 if (next_level <= assigned_tree_.MaxLevel()) {
602 VLOG(1) << "Following shared trail depth=" << next_level << " "
603 << parameters_->name();
604 const Literal decision = assigned_tree_literals_[next_level - 1];
605 CHECK(!sat_solver_->Assignment().LiteralIsFalse(decision))
606 << " at depth " << next_level << " " << parameters_->name();
607 CHECK(!sat_solver_->Assignment().LiteralIsTrue(decision));
608 return decision.Index();
609 }
610 if (objective_ == nullptr ||
611 objective_->objective_var == kNoIntegerVariable) {
612 return helper_->GetDecision(decision_policy);
613 }
614 // If the current node is close to the global lower bound, maybe try to
615 // improve it.
616 const IntegerValue root_obj_lb =
617 integer_trail_->LevelZeroLowerBound(objective_->objective_var);
618 const IntegerValue root_obj_ub =
619 integer_trail_->LevelZeroUpperBound(objective_->objective_var);
620 const IntegerValue obj_split =
621 root_obj_lb + absl::LogUniform<int64_t>(
622 *random_, 0, (root_obj_ub - root_obj_lb).value());
623 const double objective_split_probability =
624 parameters_->shared_tree_worker_objective_split_probability();
625 return helper_->GetDecision([&]() -> BooleanOrIntegerLiteral {
626 IntegerValue obj_lb = integer_trail_->LowerBound(objective_->objective_var);
627 IntegerValue obj_ub = integer_trail_->UpperBound(objective_->objective_var);
628 if (obj_lb > obj_split || obj_ub <= obj_split ||
629 next_level > assigned_tree_.MaxLevel() + 1 ||
630 absl::Bernoulli(*random_, 1 - objective_split_probability)) {
631 return decision_policy();
632 }
633 return BooleanOrIntegerLiteral(
634 IntegerLiteral::LowerOrEqual(objective_->objective_var, obj_split));
635 });
636}

638void SharedTreeWorker::MaybeProposeSplit() {
639 if (splits_wanted_ == 0 ||
640 sat_solver_->CurrentDecisionLevel() != assigned_tree_.MaxLevel() + 1) {
641 return;
642 }
643 const Literal split_decision =
644 sat_solver_->Decisions()[assigned_tree_.MaxLevel()].literal;
645 const std::optional<ProtoLiteral> encoded = EncodeDecision(split_decision);
646 if (encoded.has_value()) {
647 CHECK_EQ(assigned_tree_literals_.size(), assigned_tree_.MaxLevel());
648 manager_->ProposeSplit(assigned_tree_, *encoded);
649 if (assigned_tree_.MaxLevel() > assigned_tree_literals_.size()) {
650 assigned_tree_literals_.push_back(split_decision);
651 --splits_wanted_;
652 CHECK_EQ(assigned_tree_literals_.size(), assigned_tree_.MaxLevel());
653 }
654 CHECK_EQ(assigned_tree_literals_.size(), assigned_tree_.MaxLevel());
655 }
656}

658void SharedTreeWorker::SyncWithSharedTree() {
659 splits_wanted_ = manager_->SplitsToGeneratePerWorker();
660 VLOG(1) << "Splits wanted: " << splits_wanted_ << " " << parameters_->name();
661 manager_->SyncTree(assigned_tree_);
662 // If we have no assignment, try to get one.
663 // We also want to ensure unassigned nodes have their lower bounds bumped
664 // periodically, so workers need to occasionally replace open trees.
665 // TODO(user): Ideally we should use some metric to replace a
666 // subtree when the worker is doing badly.
667 if (assigned_tree_.MaxLevel() == 0 || absl::Bernoulli(*random_, 1e-2)) {
668 manager_->ReplaceTree(assigned_tree_);
669 }
670 VLOG(1) << "Assigned level: " << assigned_tree_.MaxLevel() << " "
671 << parameters_->name();
672 assigned_tree_literals_.clear();
673 for (int i = 1; i <= assigned_tree_.MaxLevel(); ++i) {
674 assigned_tree_literals_.push_back(
675 DecodeDecision(assigned_tree_.Decision(i)));
676 }
677}

679SatSolver::Status SharedTreeWorker::Search(
680 const std::function<void()>& feasible_solution_observer) {
681 // Inside GetAssociatedLiteral if a literal becomes fixed at level 0 during
682 // Search,the code checks it is at level 0 when decoding the literal, but
683 // the fixed literals are cached, so we can create them now to avoid a
684 // crash.
685 sat_solver_->Backtrack(0);
686 encoder_->GetTrueLiteral();
687 encoder_->GetFalseLiteral();
688 const bool has_objective =
689 objective_ != nullptr && objective_->objective_var != kNoIntegerVariable;
690 std::vector<Literal> clause;
691 while (!time_limit_->LimitReached()) {
692 if (!sat_solver_->FinishPropagation()) {
693 return sat_solver_->UnsatStatus();
694 }
695 if (heuristics_->restart_policies[heuristics_->policy_index]()) {
696 heuristics_->policy_index = (heuristics_->policy_index + 1) %
697 heuristics_->decision_policies.size();
698 sat_solver_->Backtrack(0);
699 SyncWithSharedTree();
700 }
701 if (!SyncWithLocalTrail()) return sat_solver_->UnsatStatus();
702 Literal decision(NextDecision());
703 if (time_limit_->LimitReached()) return SatSolver::LIMIT_REACHED;
704 if (decision.Index() == kNoLiteralIndex) {
705 feasible_solution_observer();
706 if (!has_objective) return SatSolver::FEASIBLE;
707 const IntegerValue objective =
708 integer_trail_->LowerBound(objective_->objective_var);
709 sat_solver_->Backtrack(0);
710 if (!integer_trail_->Enqueue(
711 IntegerLiteral::LowerOrEqual(objective_->objective_var,
712 objective - 1),
713 {}, {})) {
714 return SatSolver::INFEASIBLE;
715 }
716
717 continue;
718 }
719 CHECK(!sat_solver_->Assignment().LiteralIsFalse(decision));
720 CHECK(!sat_solver_->Assignment().LiteralIsTrue(decision));
721 if (!helper_->TakeDecision(decision)) {
722 return sat_solver_->UnsatStatus();
723 }
724 MaybeProposeSplit();
725 }
726
727 return SatSolver::LIMIT_REACHED;
728}
