48void BuildModelFromSparseData(
49 const Eigen::Ref<const VectorXd>& variable_lower_bounds,
50 const Eigen::Ref<const VectorXd>& variable_upper_bounds,
51 const Eigen::Ref<const VectorXd>& objective_coefficients,
52 const Eigen::Ref<const VectorXd>& constraint_lower_bounds,
53 const Eigen::Ref<const VectorXd>& constraint_upper_bounds,
54 const SparseMatrix<double, Eigen::RowMajor>& constraint_matrix,
55 MPModelProto* model_proto) {
56 const int num_variables = variable_lower_bounds.size();
57 const int num_constraints = constraint_lower_bounds.size();
58
59 if (variable_upper_bounds.size() != num_variables) {
60 throw std::invalid_argument(
61 absl::StrCat("Invalid size ", variable_upper_bounds.size(),
62 " for variable_upper_bounds. Expected: ", num_variables));
63 }
64 if (objective_coefficients.size() != num_variables) {
65 throw std::invalid_argument(absl::StrCat(
66 "Invalid size ", objective_coefficients.size(),
67 " for linear_objective_coefficients. Expected: ", num_variables));
68 }
69 if (constraint_upper_bounds.size() != num_constraints) {
70 throw std::invalid_argument(absl::StrCat(
71 "Invalid size ", constraint_upper_bounds.size(),
72 " for constraint_upper_bounds. Expected: ", num_constraints));
73 }
74 if (constraint_matrix.cols() != num_variables) {
75 throw std::invalid_argument(
76 absl::StrCat("Invalid number of columns ", constraint_matrix.cols(),
77 " in constraint_matrix. Expected: ", num_variables));
78 }
79 if (constraint_matrix.rows() != num_constraints) {
80 throw std::invalid_argument(
81 absl::StrCat("Invalid number of rows ", constraint_matrix.rows(),
82 " in constraint_matrix. Expected: ", num_constraints));
83 }
84
85 for (int i = 0; i < num_variables; ++i) {
86 MPVariableProto* variable = model_proto->add_variable();
87 variable->set_lower_bound(variable_lower_bounds[i]);
88 variable->set_upper_bound(variable_upper_bounds[i]);
89 variable->set_objective_coefficient(objective_coefficients[i]);
90 }
91
92 for (int row = 0; row < num_constraints; ++row) {
93 MPConstraintProto* constraint = model_proto->add_constraint();
94 constraint->set_lower_bound(constraint_lower_bounds[row]);
95 constraint->set_upper_bound(constraint_upper_bounds[row]);
96 for (SparseMatrix<double, Eigen::RowMajor>::InnerIterator it(
97 constraint_matrix, row);
98 it; ++it) {
99 constraint->add_coefficient(it.value());
100 constraint->add_var_index(it.col());
101 }
102 }
103}
