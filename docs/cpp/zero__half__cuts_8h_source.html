28// Heuristic to find a good sums of rows from the LP (with coeff -1, +1) that
29// can lead to a violated zero-half cut (i.e. after integer rounding with a
30// divisor 2).
31//
32// For this, all that matter is the parity of the coefficients and the rhs in
33// the linear combination of the original problem constraint. So this class
34// maintain a copy of the LP matrix modulo 2 on which simplification and
35// heuristic are performed to find good cut candidates(s).
36//
37// Most of what is done here is described in the paper "Algorithms to Separate
38// {0, 1/2}-Chv√°tal-Gomory Cuts", Arie M. C. A. Koster, Adrian Zymolka, Manuel
39// Kutschka.
40class ZeroHalfCutHelper {
41 public:
42 // Public API: ProcessVariables() must be called first and then constraints
43 // can be added one by one. Finally GetZeroHalfInterestingCuts() will return a
44 // set of good candidates.
45 //
46 // TODO(user): This is a first implementation, both the heuristic and the
47 // code performance can probably be improved uppon.
48 void ProcessVariables(const std::vector<double>& lp_values,
49 const std::vector<IntegerValue>& lower_bounds,
50 const std::vector<IntegerValue>& upper_bounds);
51 void AddOneConstraint(
52 glop::RowIndex,
53 const std::vector<std::pair<glop::ColIndex, IntegerValue>>& terms,
54 IntegerValue lb, IntegerValue ub);
55 std::vector<std::vector<std::pair<glop::RowIndex, IntegerValue>>>
56 InterestingCandidates(ModelRandomGenerator* random);
57
58 // Visible for testing.
59 void Reset(int size);
60
61 // Visible for testing.
62 //
63 // Boolean matrix. Each column correspond to one variable (col indices).
64 // Each row to a sum of the initial problem constraints. We store the
65 // coefficient modulo 2, so only the positions of the ones.
66 struct CombinationOfRows {
67 // How this row was formed from the initial problem constraints.
68 std::vector<std::pair<glop::RowIndex, IntegerValue>> multipliers;
69
70 // The index of the odd coefficient of this combination.
71 std::vector<int> cols;
72
73 // The parity of the rhs (1 for odd).
74 int rhs_parity;
75
76 // How tight this constraints is under the current LP solution.
77 double slack;
78 };
79 void AddBinaryRow(const CombinationOfRows& binary_row);
80 const CombinationOfRows& MatrixRow(int row) const { return rows_[row]; }
81 const std::vector<int>& MatrixCol(int col) const { return col_to_rows_[col]; }
82
83 // Visible for testing.
84 //
85 // Adds the given row to all other rows having an odd cofficient on the given
86 // column. This then eliminate the entry (col, row) that is now a singleton by
87 // incresing the slack of the given row.
88 void EliminateVarUsingRow(int col, int row);
89
90 // Visible for testing.
91 //
92 // Like std::set_symmetric_difference, but use a vector<bool> instead of sort.
93 // This assumes tmp_marked_ to be all false. We don't DCHECK it here for
94 // speed, but it DCHECKed on each EliminateVarUsingRow() call. In addition,
95 // the result is filtered using the extra_condition function.
96 void SymmetricDifference(std::function<bool(int)> extra_condition,
97 const std::vector<int>& a, std::vector<int>* b);
98
99 private:
100 void ProcessSingletonColumns();
101
102 // As we combine rows, when the activity of a combination get too far away
103 // from its bound, we just discard it. Note that the row will still be there
104 // but its index will not appear in the col-wise representation of the matrix.
105 const double kSlackThreshold = 0.5;
106 const int kMaxAggregationSize = 100;
107
108 // We don't consider long constraint or constraint with high magnitude, since
109 // the highest violation we can hope for is 1, and if the magnitude is large
110 // then the cut efficacity will not be great.
111 const int kMaxInputConstraintSize = 100;
112 const double kMaxInputConstraintMagnitude = 1e6;
113
114 // Variable information.
115 std::vector<double> lp_values_;
116 std::vector<double> shifted_lp_values_;
117 std::vector<int> bound_parity_;
118
119 // Binary matrix.
120 //
121 // Note that as we combine rows, we never move their indices. So after initial
122 // creation rows_ will always have the same size.
123 std::vector<CombinationOfRows> rows_;
124 std::vector<std::vector<int>> col_to_rows_;
125 std::vector<int> singleton_cols_;
126
127 // Temporary vector used by SymmetricDifference().
128 std::vector<bool> tmp_marked_;
129};
