17// A very simple flattened 2D array of fixed size. It's movable, copyable.
18// It can also be assigned.
19// This was originally made to replace uses of vector<vector<...>> where each
20// vector had a fixed size: vector<vector<>> has much worse performance in a
21// highly concurrent setting, because it does a lot of memory allocations.
22
23#include <vector>
24
25#include "absl/types/span.h"
26
27namespace operations_research {

29// NOTE(user): T=bool is not yet supported (the [] operator doesn't work).
30template <typename T>
31class FlatMatrix {
32 public:
33 FlatMatrix() : num_rows_(0), num_cols_(0) {}
34 FlatMatrix(size_t num_rows, size_t num_cols)
35 : num_rows_(num_rows),
36 num_cols_(num_cols),
37 array_(num_rows_ * num_cols_) {}
38 FlatMatrix(size_t num_rows, size_t num_cols, const T& elem)
39 : num_rows_(num_rows),
40 num_cols_(num_cols),
41 array_(num_rows_ * num_cols_, elem) {}
42
43 size_t num_rows() const { return num_rows_; }
44 size_t num_cols() const { return num_cols_; }
45
46 absl::Span<T> operator[](size_t row) {
47 return absl::Span<T>(array_.data() + row * num_cols_, num_cols_);
48 }
49 absl::Span<const T> operator[](size_t row) const {
50 return {array_.data() + row * num_cols_, num_cols_};
51 }
52
53 // All the elements of the FlatMatrix.
54 absl::Span<T> all_elements() { return absl::MakeSpan(array_); }
55 absl::Span<const T> all_elements() const {
56 return absl::MakeConstSpan(array_);
57 }

59 // To iterate over the rows of the FlatMatrix. Example:
60 // FlatMatrix<double> matrix(23, 45);
61 // for (absl::Span<const double> row : matrix.rows()) {
62 // LOG(INFO) << DUMP_VARS(row);
63 // }
64 class ConstRowsIterator {
65 public:
66 typedef T value_type;
67 ConstRowsIterator(const T* ptr, size_t row, size_t row_size);
68 absl::Span<const T> operator*() const { return {ptr_, row_size_}; }
69 ConstRowsIterator& operator++();
70 bool operator!=(const ConstRowsIterator& rhs) const;
71
72 private:
73 const T* ptr_;
74 size_t row_;
75 const size_t row_size_;
76 };
77 struct ConstRows {
78 typedef absl::Span<const T> value_type;
79 typedef ConstRowsIterator const_iterator;
80 ConstRowsIterator begin() const;
81 ConstRowsIterator end() const;
82 FlatMatrix* matrix = nullptr;
83 };
84 ConstRows rows() { return {.matrix = this}; }
85
86 private:
87 // Those are non-const only to support the assignment operators.
88 size_t num_rows_;
89 size_t num_cols_;
90 // NOTE(user): We could use a simpler unique_ptr<T[]> or even a self-managed
91 // memory block, but we'd need to define the copy constructor.
92 std::vector<T> array_;
93};

96template <typename T>
97FlatMatrix<T>::ConstRowsIterator::ConstRowsIterator(const T* ptr, size_t row,
98 size_t row_size)
99 : ptr_(ptr), row_(row), row_size_(row_size) {}

101template <typename T>
102typename FlatMatrix<T>::ConstRowsIterator&
103FlatMatrix<T>::ConstRowsIterator::operator++() {
104 ptr_ += row_size_;
105 ++row_;
106 return *this;
107}
108
109template <typename T>
110bool FlatMatrix<T>::ConstRowsIterator::operator!=(
111 const ConstRowsIterator& rhs) const {
112 return std::tie(ptr_, row_, row_size_) !=
113 std::tie(rhs.ptr_, rhs.row_, rhs.row_size_);
114}

116template <typename T>
117typename FlatMatrix<T>::ConstRowsIterator FlatMatrix<T>::ConstRows::begin()
118 const {
119 return ConstRowsIterator(matrix->array_.data(), /*row=*/0,
120 /*row_size=*/matrix->num_cols_);
121}

123template <typename T>
124typename FlatMatrix<T>::ConstRowsIterator FlatMatrix<T>::ConstRows::end()
125 const {
126 return ConstRowsIterator(matrix->array_.data() + matrix->array_.size(),
127 /*row=*/matrix->num_rows_,
128 /*row_size=*/matrix->num_cols_);
129}
