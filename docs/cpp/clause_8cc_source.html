52// Returns true if the given watcher list contains the given clause.
53template <typename Watcher>
54bool WatcherListContains(const std::vector<Watcher>& list,
55 const SatClause& candidate) {
56 for (const Watcher& watcher : list) {
57 if (watcher.clause == &candidate) return true;
58 }
59 return false;
60}

62// A simple wrapper to simplify the erase(std::remove_if()) pattern.
63template <typename Container, typename Predicate>
64void RemoveIf(Container c, Predicate p) {
65 c->erase(std::remove_if(c->begin(), c->end(), p), c->end());

105bool ClauseManager::PropagateOnFalse(Literal false_literal, Trail* trail) {
106 SCOPED_TIME_STAT(&stats_);
107 DCHECK(is_clean_);
108 std::vector<Watcher>& watchers = watchers_on_false_[false_literal];
109 const auto assignment = AssignmentView(trail->Assignment());
110
111 // Note(user): It sounds better to inspect the list in order, this is because
112 // small clauses like binary or ternary clauses will often propagate and thus
113 // stay at the beginning of the list.
114 auto new_it = watchers.begin();
115 const auto end = watchers.end();
116 while (new_it != end && assignment.LiteralIsTrue(new_it->blocking_literal)) {
117 ++new_it;
118 }
119 for (auto it = new_it; it != end; ++it) {
120 // Don't even look at the clause memory if the blocking literal is true.
121 if (assignment.LiteralIsTrue(it->blocking_literal)) {
122 *new_it++ = *it;
123 continue;
124 }
125 ++num_inspected_clauses_;
126
127 // If the other watched literal is true, just change the blocking literal.
128 // Note that we use the fact that the first two literals of the clause are
129 // the ones currently watched.
130 Literal* literals = it->clause->literals();
131 const Literal other_watched_literal(
132 LiteralIndex(literals[0].Index().value() ^ literals[1].Index().value() ^
133 false_literal.Index().value()));
134 if (assignment.LiteralIsTrue(other_watched_literal)) {
135 *new_it = *it;
136 new_it->blocking_literal = other_watched_literal;
137 ++new_it;
138 ++num_inspected_clause_literals_;
139 continue;
140 }
141
142 // Look for another literal to watch. We go through the list in a cyclic
143 // fashion from start. The first two literals can be ignored as they are the
144 // watched ones.
145 {
146 const int start = it->start_index;
147 const int size = it->clause->size();
148 DCHECK_GE(start, 2);
149
150 int i = start;
151 while (i < size && assignment.LiteralIsFalse(literals[i])) ++i;
152 num_inspected_clause_literals_ += i - start + 2;
153 if (i >= size) {
154 i = 2;
155 while (i < start && assignment.LiteralIsFalse(literals[i])) ++i;
156 num_inspected_clause_literals_ += i - 2;
157 if (i >= start) i = size;
158 }
159 if (i < size) {
160 // literal[i] is unassigned or true, it's now the new literal to watch.
161 // Note that by convention, we always keep the two watched literals at
162 // the beginning of the clause.
163 literals[0] = other_watched_literal;
164 literals[1] = literals[i];
165 literals[i] = false_literal;
166 watchers_on_false_[literals[1]].emplace_back(
167 it->clause, other_watched_literal, i + 1);
168 continue;
169 }
170 }
171
172 // At this point other_watched_literal is either false or unassigned, all
173 // other literals are false.
174 if (assignment.LiteralIsFalse(other_watched_literal)) {
175 // Conflict: All literals of it->clause are false.
176 //
177 // Note(user): we could avoid a copy here, but the conflict analysis
178 // complexity will be a lot higher than this anyway.
179 trail->MutableConflict()->assign(it->clause->begin(), it->clause->end());
180 trail->SetFailingSatClause(it->clause);
181 num_inspected_clause_literals_ += it - watchers.begin() + 1;
182 watchers.erase(new_it, it);
183 return false;
184 } else {
185 // Propagation: other_watched_literal is unassigned, set it to true and
186 // put it at position 0. Note that the position 0 is important because
187 // we will need later to recover the literal that was propagated from the
188 // clause using this convention.
189 literals[0] = other_watched_literal;
190 literals[1] = false_literal;
191 reasons_[trail->Index()] = it->clause;
192 trail->Enqueue(other_watched_literal, propagator_id_);
193 *new_it++ = *it;
194 }
195 }
196 num_inspected_clause_literals_ += watchers.size(); // The blocking ones.
197 watchers.erase(new_it, end);
198 return true;
199}

201bool ClauseManager::Propagate(Trail* trail) {
202 const int old_index = trail->Index();
203 while (trail->Index() == old_index && propagation_trail_index_ < old_index) {
204 const Literal literal = (*trail)[propagation_trail_index_++];
205 if (!PropagateOnFalse(literal.Negated(), trail)) return false;
206 }
207 return true;
208}

238// Sets up the 2-watchers data structure. It selects two non-false literals
239// and attaches the clause to the event: one of the watched literals become
240// false. It returns false if the clause only contains literals assigned to
241// false. If only one literals is not false, it propagates it to true if it
242// is not already assigned.
243bool ClauseManager::AttachAndPropagate(SatClause* clause, Trail* trail) {
244 SCOPED_TIME_STAT(&stats_);
245
246 const int size = clause->size();
247 Literal* literals = clause->literals();
248
249 // Select the first two literals that are not assigned to false and put them
250 // on position 0 and 1.
251 int num_literal_not_false = 0;
252 for (int i = 0; i < size; ++i) {
253 if (!trail->Assignment().LiteralIsFalse(literals[i])) {
254 std::swap(literals[i], literals[num_literal_not_false]);
255 ++num_literal_not_false;
256 if (num_literal_not_false == 2) {
257 break;
258 }
259 }
260 }
261
262 // Returns false if all the literals were false.
263 // This should only happen on an UNSAT problem, and there is no need to attach
264 // the clause in this case.
265 if (num_literal_not_false == 0) return false;
266
267 if (num_literal_not_false == 1) {
268 // To maintain the validity of the 2-watcher algorithm, we need to watch
269 // the false literal with the highest decision level.
270 int max_level = trail->Info(literals[1].Variable()).level;
271 for (int i = 2; i < size; ++i) {
272 const int level = trail->Info(literals[i].Variable()).level;
273 if (level > max_level) {
274 max_level = level;
275 std::swap(literals[1], literals[i]);
276 }
277 }
278
279 // Propagates literals[0] if it is unassigned.
280 if (!trail->Assignment().LiteralIsTrue(literals[0])) {
281 reasons_[trail->Index()] = clause;
282 trail->Enqueue(literals[0], propagator_id_);
283 }
284 }
285
286 ++num_watched_clauses_;
287 AttachOnFalse(literals[0], literals[1], clause);
288 AttachOnFalse(literals[1], literals[0], clause);
289 return true;
290}

292void ClauseManager::Attach(SatClause* clause, Trail* trail) {
293 Literal* literals = clause->literals();
294 DCHECK(!trail->Assignment().LiteralIsAssigned(literals[0]));
295 DCHECK(!trail->Assignment().LiteralIsAssigned(literals[1]));
296
297 ++num_watched_clauses_;
298 AttachOnFalse(literals[0], literals[1], clause);
299 AttachOnFalse(literals[1], literals[0], clause);
300}

302void ClauseManager::InternalDetach(SatClause* clause) {
303 --num_watched_clauses_;
304 const size_t size = clause->size();
305 if (drat_proof_handler_ != nullptr && size > 2) {

319void ClauseManager::Detach(SatClause* clause) {
320 InternalDetach(clause);
321 for (const Literal l : {clause->FirstLiteral(), clause->SecondLiteral()}) {
322 needs_cleaning_.Clear(l);
323 RemoveIf(&(watchers_on_false_[l]), [](const Watcher& watcher) {
324 return watcher.clause->IsRemoved();
325 });
326 }

329void ClauseManager::DetachAllClauses() {
330 if (!all_clauses_are_attached_) return;
331 all_clauses_are_attached_ = false;
332
333 // This is easy, and this allows to reset memory if some watcher lists where

340void ClauseManager::AttachAllClauses() {
341 if (all_clauses_are_attached_) return;
342 all_clauses_are_attached_ = true;
343
344 needs_cleaning_.ClearAll(); // This doesn't resize it.
345 watchers_on_false_.resize(needs_cleaning_.size().value());
346
347 DeleteRemovedClauses();
348 for (SatClause* clause : clauses_) {
349 ++num_watched_clauses_;
350 DCHECK_GE(clause->size(), 2);
351 AttachOnFalse(clause->FirstLiteral(), clause->SecondLiteral(), clause);
352 AttachOnFalse(clause->SecondLiteral(), clause->FirstLiteral(), clause);
353 }
354}

356// This one do not need the clause to be detached.
357bool ClauseManager::InprocessingFixLiteral(Literal true_literal) {
358 DCHECK_EQ(trail_->CurrentDecisionLevel(), 0);
359 if (drat_proof_handler_ != nullptr) {
360 drat_proof_handler_->AddClause({true_literal});
361 }
362 // TODO(user): remove the test when the DRAT issue with fixed literal is
363 // resolved.
364 if (!trail_->Assignment().LiteralIsTrue(true_literal)) {
365 trail_->EnqueueWithUnitReason(true_literal);
366
367 // Even when all clauses are detached, we can propagate the implication
368 // graph and we do that right away.
369 return implication_graph_->Propagate(trail_);
370 }
371 return true;
372}

385bool ClauseManager::InprocessingRewriteClause(
386 SatClause* clause, absl::Span<const Literal> new_clause) {
387 if (new_clause.empty()) return false; // UNSAT.
388
389 if (DEBUG_MODE) {
390 for (const Literal l : new_clause) {
391 DCHECK(!trail_->Assignment().LiteralIsAssigned(l));
392 }
393 }
394
395 if (new_clause.size() == 1) {
396 if (!InprocessingFixLiteral(new_clause[0])) return false;
397 InprocessingRemoveClause(clause);
398 return true;
399 }
400
401 if (new_clause.size() == 2) {
402 CHECK(implication_graph_->AddBinaryClause(new_clause[0], new_clause[1]));
403 InprocessingRemoveClause(clause);
404 return true;
405 }
406
407 if (drat_proof_handler_ != nullptr) {
408 // We must write the new clause before we delete the old one.
409 drat_proof_handler_->AddClause(new_clause);
410 drat_proof_handler_->DeleteClause(clause->AsSpan());
411 }
412
413 if (all_clauses_are_attached_) {
414 // We can still rewrite the clause, but it is inefficient. We first
415 // detach it in a non-lazy way.
416 --num_watched_clauses_;
417 clause->Clear();
418 for (const Literal l : {clause->FirstLiteral(), clause->SecondLiteral()}) {
419 needs_cleaning_.Clear(l);
420 RemoveIf(&(watchers_on_false_[l]), [](const Watcher& watcher) {
421 return watcher.clause->IsRemoved();
422 });
423 }
424 }
425
426 clause->Rewrite(new_clause);
427
428 // And we reattach it.
429 if (all_clauses_are_attached_) Attach(clause, trail_);
430 return true;
431}

433SatClause* ClauseManager::InprocessingAddClause(
434 absl::Span<const Literal> new_clause) {
435 DCHECK(!new_clause.empty());
436 DCHECK(!all_clauses_are_attached_);
437 if (DEBUG_MODE) {
438 for (const Literal l : new_clause) {
439 DCHECK(!trail_->Assignment().LiteralIsAssigned(l));
440 }
441 }
442
443 if (new_clause.size() == 1) {
444 // TODO(user): We should return false...
445 if (!InprocessingFixLiteral(new_clause[0])) return nullptr;
446 return nullptr;
447 }
448
449 if (new_clause.size() == 2) {
450 implication_graph_->AddBinaryClause(new_clause[0], new_clause[1]);
451 return nullptr;
452 }
453
454 SatClause* clause = SatClause::Create(new_clause);
455 clauses_.push_back(clause);
456 return clause;
457}

459void ClauseManager::CleanUpWatchers() {
460 SCOPED_TIME_STAT(&stats_);
461 for (const LiteralIndex index : needs_cleaning_.PositionsSetAtLeastOnce()) {
462 DCHECK(needs_cleaning_[index]);
463 RemoveIf(&(watchers_on_false_[index]), [](const Watcher& watcher) {
464 return watcher.clause->IsRemoved();
465 });
466 needs_cleaning_.Clear(index);
467 }
468 needs_cleaning_.NotifyAllClear();
469 is_clean_ = true;
470}

472// We also update to_minimize_index_/to_probe_index_ correctly.
473//
474// TODO(user): If more indices are needed, generalize the code to a vector of
475// indices.
476void ClauseManager::DeleteRemovedClauses() {
477 DCHECK(is_clean_);
478
479 int new_size = 0;
480 const int old_size = clauses_.size();
481 for (int i = 0; i < old_size; ++i) {
482 if (i == to_minimize_index_) to_minimize_index_ = new_size;
483 if (i == to_probe_index_) to_probe_index_ = new_size;
484 if (clauses_[i]->IsRemoved()) {
485 delete clauses_[i];
486 } else {
487 clauses_[new_size++] = clauses_[i];
488 }
489 }
490 clauses_.resize(new_size);
491
492 if (to_minimize_index_ > new_size) to_minimize_index_ = new_size;
493 if (to_probe_index_ > new_size) to_probe_index_ = new_size;
494}

524// TODO(user): Not all of the solver knows about representative literal, do
525// use them here to maintains invariant? Explore this when we start cleaning our
526// clauses using equivalence during search. We can easily do it for every
527// conflict we learn instead of here.
528bool BinaryImplicationGraph::AddBinaryClause(Literal a, Literal b) {
529 SCOPED_TIME_STAT(&stats_);
530
531 // Tricky: If this is the first clause, the propagator will be added and
532 // assumed to be in a "propagated" state. This makes sure this is the case.
533 if (IsEmpty()) propagation_trail_index_ = trail_->Index();
534
535 if (drat_proof_handler_ != nullptr) {
536 // TODO(user): Like this we will duplicate all binary clause from the
537 // problem. However this leads to a simpler API (since we don't need to
538 // special case the loading of the original clauses) and we mainly use drat
539 // proof for testing anyway.
540 drat_proof_handler_->AddClause({a, b});
541 }
542
543 if (is_redundant_[a.Index()]) a = Literal(representative_of_[a.Index()]);
544 if (is_redundant_[b.Index()]) b = Literal(representative_of_[b.Index()]);
545 if (a == b.Negated()) return true;
546 if (a == b) return FixLiteral(a);
547
548 DCHECK(!is_removed_[a]);
549 DCHECK(!is_removed_[b]);
550 estimated_sizes_[a.NegatedIndex()]++;
551 estimated_sizes_[b.NegatedIndex()]++;
552 implications_[a.NegatedIndex()].push_back(b);
553 implications_[b.NegatedIndex()].push_back(a);
554 implies_something_.Set(a.NegatedIndex());
555 implies_something_.Set(b.NegatedIndex());
556 NotifyPossibleDuplicate(a);
557 NotifyPossibleDuplicate(b);
558 is_dag_ = false;
559 num_implications_ += 2;
560
561 if (enable_sharing_ && add_callback_ != nullptr) add_callback_(a, b);
562
563 const auto& assignment = trail_->Assignment();
564 if (trail_->CurrentDecisionLevel() == 0) {
565 DCHECK(!assignment.LiteralIsAssigned(a));
566 DCHECK(!assignment.LiteralIsAssigned(b));
567 } else {
568 if (assignment.LiteralIsFalse(a)) {
569 if (assignment.LiteralIsAssigned(b)) {
570 if (assignment.LiteralIsFalse(b)) return false;
571 } else {
572 reasons_[trail_->Index()] = a;
573 trail_->Enqueue(b, propagator_id_);
574 }
575 } else if (assignment.LiteralIsFalse(b)) {
576 if (!assignment.LiteralIsAssigned(a)) {
577 reasons_[trail_->Index()] = b;
578 trail_->Enqueue(a, propagator_id_);
579 }
580 }
581 }
582
583 return true;
584}

586bool BinaryImplicationGraph::AddAtMostOne(
587 absl::Span<const Literal> at_most_one) {
588 DCHECK_EQ(trail_->CurrentDecisionLevel(), 0);
589 if (at_most_one.size() <= 1) return true;
590
591 // Temporarily copy the at_most_one constraint at the end of
592 // at_most_one_buffer_. It will be cleaned up and added by
593 // CleanUpAndAddAtMostOnes().
594 const int base_index = at_most_one_buffer_.size();
595 at_most_one_buffer_.push_back(Literal(LiteralIndex(at_most_one.size())));
596 at_most_one_buffer_.insert(at_most_one_buffer_.end(), at_most_one.begin(),
597 at_most_one.end());
598
599 is_dag_ = false;
600 return CleanUpAndAddAtMostOnes(base_index);
601}

607bool BinaryImplicationGraph::FixLiteral(Literal true_literal) {
608 CHECK_EQ(trail_->CurrentDecisionLevel(), 0);
609 if (trail_->Assignment().LiteralIsTrue(true_literal)) return true;
610 if (trail_->Assignment().LiteralIsFalse(true_literal)) return false;
611

613 drat_proof_handler_->AddClause({true_literal});
614 }
615
616 trail_->EnqueueWithUnitReason(true_literal);
617 return Propagate(trail_);
618}

620// This works by doing a linear scan on the at_most_one_buffer_ and
621// cleaning/copying the at most ones on the fly to the beginning of the same
622// buffer.
623bool BinaryImplicationGraph::CleanUpAndAddAtMostOnes(int base_index) {
624 const VariablesAssignment& assignment = trail_->Assignment();
625 int local_end = base_index;
626 const int buffer_size = at_most_one_buffer_.size();
627 for (int i = base_index; i < buffer_size;) {
628 // Process a new at most one.
629 // It will be copied into buffer[local_start + 1, local_end].
630 // With its size at buffer[local_start].
631 const int local_start = local_end;
632
633 // Update the iterator now. Even if the current at_most_one is reduced away,
634 // local_start will still point to the next one, or to the end of the
635 // buffer.
636 if (i == at_most_one_iterator_) {
637 at_most_one_iterator_ = local_start;
638 }
639
640 // We have an at_most_one starting at i, and we increment i to the next one.
641 const absl::Span<const Literal> initial_amo = AtMostOne(i);
642 i += initial_amo.size() + 1;
643
644 // Reserve space for size.
645 local_end++;
646 bool set_all_left_to_false = false;
647 for (const Literal l : initial_amo) {
648 if (assignment.LiteralIsFalse(l)) continue;
649 if (is_removed_[l]) continue;
650 if (!set_all_left_to_false && assignment.LiteralIsTrue(l)) {
651 set_all_left_to_false = true;
652 continue;
653 }
654 at_most_one_buffer_[local_end++] = RepresentativeOf(l);
655 }
656 at_most_one_buffer_[local_start] =
657 Literal(LiteralIndex(local_end - local_start - 1));
658
659 // Deal with duplicates.
660 // Any duplicate in an "at most one" must be false.
661 bool some_duplicates = false;
662 if (!set_all_left_to_false) {
663 // Sort the copied amo.
664 // We only want to expose a const AtMostOne() so we sort directly here.
665 Literal* pt = &at_most_one_buffer_[local_start + 1];
666 std::sort(pt, pt + AtMostOne(local_start).size());
667
668 LiteralIndex previous = kNoLiteralIndex;
669 bool remove_previous = false;
670 int new_local_end = local_start + 1;
671 for (const Literal l : AtMostOne(local_start)) {
672 if (l.Index() == previous) {
673 if (assignment.LiteralIsTrue(l)) return false;
674 if (!assignment.LiteralIsFalse(l)) {
675 if (!FixLiteral(l.Negated())) return false;
676 }
677 remove_previous = true;
678 some_duplicates = true;
679 continue;
680 }
681
682 // We need to pay attention to triplet or more of equal elements, so
683 // it is why we need this boolean and can't just remove it right away.
684 if (remove_previous) {
685 --new_local_end;
686 remove_previous = false;
687 }
688 previous = l.Index();
689 at_most_one_buffer_[new_local_end++] = l;
690 }
691 if (remove_previous) --new_local_end;
692
693 // Update local end and the amo size.
694 local_end = new_local_end;
695 at_most_one_buffer_[local_start] =
696 Literal(LiteralIndex(local_end - local_start - 1));
697 }
698
699 // If there was some duplicates, we need to rescan to see if a literal
700 // didn't become true because its negation was appearing twice!
701 if (some_duplicates) {
702 int new_local_end = local_start + 1;
703 for (const Literal l : AtMostOne(local_start)) {
704 if (assignment.LiteralIsFalse(l)) continue;
705 if (!set_all_left_to_false && assignment.LiteralIsTrue(l)) {
706 set_all_left_to_false = true;
707 continue;
708 }
709 at_most_one_buffer_[new_local_end++] = l;
710 }
711
712 // Update local end and the amo size.
713 local_end = new_local_end;
714 at_most_one_buffer_[local_start] =
715 Literal(LiteralIndex(local_end - local_start - 1));
716 }
717
718 // Deal with all false.
719 if (set_all_left_to_false) {
720 for (const Literal l : AtMostOne(local_start)) {
721 if (assignment.LiteralIsFalse(l)) continue;
722 if (assignment.LiteralIsTrue(l)) return false;
723 if (!FixLiteral(l.Negated())) return false;
724 }
725
726 // Erase this at_most_one.
727 local_end = local_start;
728 continue;
729 }
730
731 // Create a Span<> to simplify the code below.
732 const absl::Span<const Literal> at_most_one = AtMostOne(local_start);
733
734 // We expand small sizes into implications.
735 // TODO(user): Investigate what the best threshold is.
736 if (at_most_one.size() <= std::max(2, at_most_one_max_expansion_size_)) {
737 for (const Literal a : at_most_one) {
738 for (const Literal b : at_most_one) {
739 if (a == b) continue;
740 implications_[a].push_back(b.Negated());
741 implies_something_.Set(a);
742 NotifyPossibleDuplicate(a);
743 }
744 }
745 num_implications_ += at_most_one.size() * (at_most_one.size() - 1);
746
747 // This will erase the at_most_one from the buffer.
748 local_end = local_start;
749 continue;
750 }
751
752 // Index the new at most one.
753 for (const Literal l : at_most_one) {
754 if (l.Index() >= at_most_ones_.size()) {
755 at_most_ones_.resize(l.Index().value() + 1);
756 }
757 DCHECK(!is_redundant_[l]);
758 at_most_ones_[l].push_back(local_start);
759 implies_something_.Set(l);
760 }
761 }
762
763 at_most_one_buffer_.resize(local_end);
764 return true;
765}

767bool BinaryImplicationGraph::PropagateOnTrue(Literal true_literal,
768 Trail* trail) {
769 SCOPED_TIME_STAT(&stats_);
770
771 const auto assignment = AssignmentView(trail->Assignment());
772 DCHECK(assignment.LiteralIsTrue(true_literal));
773
774 // Note(user): This update is not exactly correct because in case of conflict
775 // we don't inspect that much clauses. But doing ++num_inspections_ inside the
776 // loop does slow down the code by a few percent.
777 num_inspections_ += implications_[true_literal].size();
778
779 for (const Literal literal : implications_[true_literal]) {
780 if (assignment.LiteralIsTrue(literal)) {
781 // Note(user): I tried to update the reason here if the literal was
782 // enqueued after the true_literal on the trail. This property is
783 // important for ComputeFirstUIPConflict() to work since it needs the
784 // trail order to be a topological order for the deduction graph.
785 // But the performance was not too good...
786 continue;
787 }
788
789 ++num_propagations_;
790 if (assignment.LiteralIsFalse(literal)) {
791 // Conflict.
792 *(trail->MutableConflict()) = {true_literal.Negated(), literal};
793 return false;
794 } else {
795 // Propagation.
796 reasons_[trail->Index()] = true_literal.Negated();
797 trail->FastEnqueue(literal);
798 }
799 }
800
801 // Propagate the at_most_one constraints.
802 if (true_literal.Index() < at_most_ones_.size()) {
803 for (const int start : at_most_ones_[true_literal]) {
804 bool seen = false;
805 for (const Literal literal : AtMostOne(start)) {
806 ++num_inspections_;
807 if (literal == true_literal) {
808 if (DEBUG_MODE) {
809 DCHECK(!seen);
810 seen = true;
811 }
812 continue;
813 }
814 if (assignment.LiteralIsFalse(literal)) continue;
815
816 ++num_propagations_;
817 if (assignment.LiteralIsTrue(literal)) {
818 // Conflict.
819 *(trail->MutableConflict()) = {true_literal.Negated(),
820 literal.Negated()};
821 return false;
822 } else {
823 // Propagation.
824 reasons_[trail->Index()] = true_literal.Negated();
825 trail->FastEnqueue(literal.Negated());
826 }
827 }
828 }
829 }
830
831 return true;
832}

852// Here, we remove all the literal whose negation are implied by the negation of
853// the 1-UIP literal (which always appear first in the given conflict). Note
854// that this algorithm is "optimal" in the sense that it leads to a minimized
855// conflict with a backjump level as low as possible. However, not all possible
856// literals are removed.
857//
858// TODO(user): Also consider at most one?
859void BinaryImplicationGraph::MinimizeConflictWithReachability(
860 std::vector<Literal>* conflict) {
861 SCOPED_TIME_STAT(&stats_);
862 dfs_stack_.clear();
863
864 // Compute the reachability from the literal "not(conflict->front())" using
865 // an iterative dfs.
866 const LiteralIndex root_literal_index = conflict->front().NegatedIndex();
867 is_marked_.ClearAndResize(LiteralIndex(implications_.size()));
868 is_marked_.Set(root_literal_index);
869
870 // TODO(user): This sounds like a good idea, but somehow it seems better not
871 // to do that even though it is almost for free. Investigate more.
872 //
873 // The idea here is that since we already compute the reachability from the
874 // root literal, we can use this computation to remove any implication
875 // root_literal => b if there is already root_literal => a and b is reachable
876 // from a.
877 const bool also_prune_direct_implication_list = false;
878
879 // We treat the direct implications differently so we can also remove the
880 // redundant implications from this list at the same time.
881 auto& direct_implications = implications_[root_literal_index];
882 for (const Literal l : direct_implications) {
883 if (is_marked_[l]) continue;
884 dfs_stack_.push_back(l);
885 while (!dfs_stack_.empty()) {
886 const LiteralIndex index = dfs_stack_.back().Index();
887 dfs_stack_.pop_back();
888 if (!is_marked_[index]) {
889 is_marked_.Set(index);
890 for (const Literal implied : implications_[index]) {
891 if (!is_marked_[implied]) dfs_stack_.push_back(implied);
892 }
893 }
894 }
895
896 // The "trick" is to unmark 'l'. This way, if we explore it twice, it means
897 // that this l is reachable from some other 'l' from the direct implication
898 // list. Remarks:
899 // - We don't loose too much complexity when this happen since a literal
900 // can be unmarked only once, so in the worst case we loop twice over its
901 // children. Moreover, this literal will be pruned for later calls.
902 // - This is correct, i.e. we can't prune too many literals because of a
903 // strongly connected component. Proof by contradiction: If we take the
904 // first (in direct_implications) literal from a removed SCC, it must
905 // have marked all the others. But because they are marked, they will not
906 // be explored again and so can't mark the first literal.
907 if (also_prune_direct_implication_list) {
908 is_marked_.Clear(l);
909 }
910 }
911
912 // Now we can prune the direct implications list and make sure are the
913 // literals there are marked.
914 if (also_prune_direct_implication_list) {
915 int new_size = 0;
916 for (const Literal l : direct_implications) {
917 if (!is_marked_[l]) {
918 is_marked_.Set(l);
919 direct_implications[new_size] = l;
920 ++new_size;
921 }
922 }
923 if (new_size < direct_implications.size()) {
924 num_redundant_implications_ += direct_implications.size() - new_size;
925 direct_implications.resize(new_size);
926 }
927 }
928
929 RemoveRedundantLiterals(conflict);
930}

932// Same as MinimizeConflictWithReachability() but also mark (in the given
933// SparseBitset) the reachable literal already assigned to false. These literals
934// will be implied if the 1-UIP literal is assigned to false, and the classic
935// minimization algorithm can take advantage of that.
936void BinaryImplicationGraph::MinimizeConflictFirst(
937 const Trail& trail, std::vector<Literal>* conflict,
938 SparseBitset<BooleanVariable>* marked) {
939 SCOPED_TIME_STAT(&stats_);
940 DCHECK(!conflict->empty());
941 is_marked_.ClearAndResize(LiteralIndex(implications_.size()));
942 MarkDescendants(conflict->front().Negated());
943 for (const LiteralIndex i : is_marked_.PositionsSetAtLeastOnce()) {
944 // TODO(user): if this is false, then we actually have a conflict of size 2.
945 // This can only happen if the binary clause was not propagated properly
946 // if for instance we do chronological bactracking without re-enqueing the
947 // consequence of a binary clause.
948 if (trail.Assignment().LiteralIsTrue(Literal(i))) {
949 marked->Set(Literal(i).Variable());
950 }
951 }
952 RemoveRedundantLiterals(conflict);
953}

955// Same as MinimizeConflictFirst() but take advantage of this reachability
956// computation to remove redundant implication in the implication list of the
957// first UIP conflict.
958void BinaryImplicationGraph::MinimizeConflictFirstWithTransitiveReduction(
959 const Trail& /*trail*/, std::vector<Literal>* conflict,
960 absl::BitGenRef random) {
961 SCOPED_TIME_STAT(&stats_);
962 const LiteralIndex root_literal_index = conflict->front().NegatedIndex();
963 is_marked_.ClearAndResize(LiteralIndex(implications_.size()));
964 is_marked_.Set(root_literal_index);
965
966 int new_size = 0;
967 auto& direct_implications = implications_[root_literal_index];
968
969 // The randomization allow to find more redundant implication since to find
970 // a => b and remove b, a must be before b in direct_implications. Note that
971 // a std::reverse() could work too. But randomization seems to work better.
972 // Probably because it has other impact on the search tree.
973 std::shuffle(direct_implications.begin(), direct_implications.end(), random);
974 dfs_stack_.clear();
975 for (const Literal l : direct_implications) {
976 if (is_marked_[l]) {
977 // The literal is already marked! so it must be implied by one of the
978 // previous literal in the direct_implications list. We can safely remove
979 // it.
980 continue;
981 }
982 direct_implications[new_size++] = l;
983 dfs_stack_.push_back(l);
984 while (!dfs_stack_.empty()) {
985 const LiteralIndex index = dfs_stack_.back().Index();
986 dfs_stack_.pop_back();
987 if (!is_marked_[index]) {
988 is_marked_.Set(index);
989 for (const Literal implied : implications_[index]) {
990 if (!is_marked_[implied]) dfs_stack_.push_back(implied);
991 }
992 }
993 }
994 }
995 if (new_size < direct_implications.size()) {
996 num_redundant_implications_ += direct_implications.size() - new_size;
997 direct_implications.resize(new_size);
998 }
999 RemoveRedundantLiterals(conflict);
1000}

1002void BinaryImplicationGraph::RemoveRedundantLiterals(
1003 std::vector<Literal>* conflict) {
1004 SCOPED_TIME_STAT(&stats_);
1005 int new_index = 1;
1006 for (int i = 1; i < conflict->size(); ++i) {
1007 if (!is_marked_[(*conflict)[i].NegatedIndex()]) {
1008 (*conflict)[new_index] = (*conflict)[i];
1009 ++new_index;
1010 }
1011 }
1012 if (new_index < conflict->size()) {
1013 ++num_minimization_;
1014 num_literals_removed_ += conflict->size() - new_index;
1015 conflict->resize(new_index);
1016 }
1017}

1020void BinaryImplicationGraph::MinimizeConflictExperimental(
1021 const Trail& trail, std::vector<Literal>* conflict) {
1022 SCOPED_TIME_STAT(&stats_);
1023 is_marked_.ClearAndResize(LiteralIndex(implications_.size()));
1024 is_simplified_.ClearAndResize(LiteralIndex(implications_.size()));
1025 for (const Literal lit : *conflict) {
1026 is_marked_.Set(lit);
1027 }
1028
1029 // Identify and remove the redundant literals from the given conflict.
1030 // 1/ If a -> b then a can be removed from the conflict clause.
1031 // This is because not b -> not a.
1032 // 2/ a -> b can only happen if level(a) <= level(b).
1033 // 3/ Because of 2/, cycles can appear only at the same level.
1034 // The vector is_simplified_ is used to avoid removing all elements of a
1035 // cycle. Note that this is not optimal in the sense that we may not remove
1036 // a literal that can be removed.
1037 //
1038 // Note that there is no need to explore the unique literal of the highest
1039 // decision level since it can't be removed. Because this is a conflict, such
1040 // literal is always at position 0, so we start directly at 1.
1041 int index = 1;
1042 for (int i = 1; i < conflict->size(); ++i) {
1043 const Literal lit = (*conflict)[i];
1044 const int lit_level = trail.Info(lit.Variable()).level;
1045 bool keep_literal = true;
1046 for (const Literal implied : implications_[lit]) {
1047 if (is_marked_[implied]) {
1048 DCHECK_LE(lit_level, trail.Info(implied.Variable()).level);
1049 if (lit_level == trail.Info(implied.Variable()).level &&
1050 is_simplified_[implied]) {
1051 continue;
1052 }
1053 keep_literal = false;
1054 break;
1055 }
1056 }
1057 if (keep_literal) {
1058 (*conflict)[index] = lit;
1059 ++index;
1060 } else {
1061 is_simplified_.Set(lit);
1062 }
1063 }
1064 if (index < conflict->size()) {
1065 ++num_minimization_;
1066 num_literals_removed_ += conflict->size() - index;
1067 conflict->erase(conflict->begin() + index, conflict->end());
1068 }
1069}

1071void BinaryImplicationGraph::RemoveFixedVariables() {
1072 SCOPED_TIME_STAT(&stats_);
1073 DCHECK_EQ(trail_->CurrentDecisionLevel(), 0);
1074 if (IsEmpty()) return;
1075
1076 // Nothing to do if nothing changed since last call.
1077 const int new_num_fixed = trail_->Index();
1078 DCHECK_EQ(propagation_trail_index_, new_num_fixed);
1079 if (num_processed_fixed_variables_ == new_num_fixed) return;
1080
1081 const VariablesAssignment& assignment = trail_->Assignment();
1082 is_marked_.ClearAndResize(LiteralIndex(implications_.size()));
1083 for (; num_processed_fixed_variables_ < new_num_fixed;
1084 ++num_processed_fixed_variables_) {
1085 const Literal true_literal = (*trail_)[num_processed_fixed_variables_];
1086 if (DEBUG_MODE) {
1087 // The code assumes that everything is already propagated.
1088 // Otherwise we will remove implications that didn't propagate yet!
1089 for (const Literal lit : implications_[true_literal]) {
1090 DCHECK(trail_->Assignment().LiteralIsTrue(lit));
1091 }
1092 }
1093
1094 // If b is true and a -> b then because not b -> not a, all the
1095 // implications list that contains b will be marked by this process.
1096 // And the ones that contains not(b) should correspond to a false literal!
1097 //
1098 // TODO(user): This might not be true if we remove implication by
1099 // transitive reduction and the process was aborted due to the computation
1100 // limit. I think it will be good to maintain that invariant though,
1101 // otherwise fixed literals might never be removed from these lists...
1102 for (const Literal lit : implications_[true_literal.NegatedIndex()]) {
1103 if (lit.NegatedIndex() < representative_of_.size() &&
1104 representative_of_[lit.Negated()] != kNoLiteralIndex) {
1105 // We mark its representative instead.
1106 is_marked_.Set(representative_of_[lit.Negated()]);
1107 } else {
1108 is_marked_.Set(lit.NegatedIndex());
1109 }
1110 }
1111 gtl::STLClearObject(&(implications_[true_literal]));
1112 gtl::STLClearObject(&(implications_[true_literal.NegatedIndex()]));
1113
1114 if (true_literal.Index() < at_most_ones_.size()) {
1115 gtl::STLClearObject(&(at_most_ones_[true_literal]));
1116 }
1117 if (true_literal.NegatedIndex() < at_most_ones_.size()) {
1118 gtl::STLClearObject(&(at_most_ones_[true_literal.NegatedIndex()]));
1119 }
1120 }
1121 for (const LiteralIndex i : is_marked_.PositionsSetAtLeastOnce()) {
1122 RemoveIf(&implications_[i], [&assignment](const Literal& lit) {
1123 return assignment.LiteralIsTrue(lit);
1124 });
1125 }
1126
1127 // TODO(user): This might be a bit slow. Do not call all the time if needed,
1128 // this shouldn't change the correctness of the code.
1129 at_most_ones_.clear();
1130 CleanUpAndAddAtMostOnes(/*base_index=*/0);
1131 DCHECK(InvariantsAreOk());
1132}

1134class SccGraph {
1135 public:
1136 using Implications =
1137 absl::StrongVector<LiteralIndex, absl::InlinedVector<Literal, 6>>;
1138 using AtMostOnes =
1139 absl::StrongVector<LiteralIndex, absl::InlinedVector<int32_t, 6>>;
1140 using SccFinder =
1141 StronglyConnectedComponentsFinder<int32_t, SccGraph,
1142 CompactVectorVector<int32_t, int32_t>>;
1143
1144 explicit SccGraph(SccFinder* finder, Implications* graph,
1145 AtMostOnes* at_most_ones,
1146 std::vector<Literal>* at_most_one_buffer)
1147 : finder_(*finder),
1148 implications_(*graph),
1149 at_most_ones_(*at_most_ones),
1150 at_most_one_buffer_(*at_most_one_buffer) {}
1151
1152 const std::vector<int32_t>& operator[](int32_t node) const {
1153 tmp_.clear();
1154 for (const Literal l : implications_[LiteralIndex(node)]) {
1155 tmp_.push_back(l.Index().value());
1156 if (finder_.NodeIsInCurrentDfsPath(l.NegatedIndex().value())) {
1157 to_fix_.push_back(l);
1158 }
1159 }
1160 if (node < at_most_ones_.size()) {
1161 for (const int start : at_most_ones_[LiteralIndex(node)]) {
1162 if (start >= at_most_one_already_explored_.size()) {
1163 at_most_one_already_explored_.resize(start + 1, false);
1164 previous_node_to_explore_at_most_one_.resize(start + 1);
1165 }
1166
1167 // In the presence of at_most_ones_ constraints, expanding them
1168 // implicitly to implications in the SCC computation can result in a
1169 // quadratic complexity rather than a linear one in term of the input
1170 // data structure size. So this test here is critical on problem with
1171 // large at_most ones like the "ivu06-big.mps.gz" where without it, the
1172 // full FindStronglyConnectedComponents() take more than on hour instead
1173 // of less than a second!
1174 if (at_most_one_already_explored_[start]) {
1175 // We never expand a node twice.
1176 const int first_node = previous_node_to_explore_at_most_one_[start];
1177 DCHECK_NE(node, first_node);
1178
1179 if (finder_.NodeIsInCurrentDfsPath(first_node)) {
1180 // If the first node is not settled, then we do explore the
1181 // at_most_one constraint again. In "Mixed-Integer-Programming:
1182 // Analyzing 12 years of progress", Tobias Achterberg and Roland
1183 // Wunderling explains that an at most one need to be looped over at
1184 // most twice. I am not sure exactly how that works, so for now we
1185 // are not fully linear, but on actual instances, we only rarely
1186 // run into this case.
1187 //
1188 // Note that we change the previous node to explore at most one
1189 // since the current node will be settled before the old ones.
1190 //
1191 // TODO(user): avoid looping more than twice on the same at most one
1192 // constraints? Note that the second time we loop we have x => y =>
1193 // not(x), so we can already detect that x must be false which we
1194 // detect below.
1195 previous_node_to_explore_at_most_one_[start] = node;
1196 } else {
1197 // The first node is already settled and so are all its child. Only
1198 // not(first_node) might still need exploring.
1199 tmp_.push_back(
1200 Literal(LiteralIndex(first_node)).NegatedIndex().value());
1201 continue;
1202 }
1203 } else {
1204 at_most_one_already_explored_[start] = true;
1205 previous_node_to_explore_at_most_one_[start] = node;
1206 }
1207
1208 const absl::Span<const Literal> amo =
1209 absl::MakeSpan(&at_most_one_buffer_[start + 1],
1210 at_most_one_buffer_[start].Index().value());
1211 for (const Literal l : amo) {
1212 if (l.Index() == node) continue;
1213 tmp_.push_back(l.NegatedIndex().value());
1214 if (finder_.NodeIsInCurrentDfsPath(l.Index().value())) {
1215 to_fix_.push_back(l.Negated());
1216 }
1217 }
1218 }
1219 }
1220 work_done_ += tmp_.size();
1221 return tmp_;
1222 }
1223
1224 // All these literals where detected to be true during the SCC computation.
1225 mutable std::vector<Literal> to_fix_;
1226
1227 // For the deterministic time.
1228 mutable int64_t work_done_ = 0;
1229
1230 private:
1231 const SccFinder& finder_;
1232 const Implications& implications_;
1233 const AtMostOnes& at_most_ones_;
1234 const std::vector<Literal>& at_most_one_buffer_;
1235
1236 mutable std::vector<int32_t> tmp_;
1237
1238 // Used to get a non-quadratic complexity in the presence of at most ones.
1239 mutable std::vector<bool> at_most_one_already_explored_;
1240 mutable std::vector<int> previous_node_to_explore_at_most_one_;
1241};

1243bool BinaryImplicationGraph::DetectEquivalences(bool log_info) {
1244 // This was already called, and no new constraint where added. Note that new
1245 // fixed variable cannot create new equivalence, only new binary clauses do.
1246 if (is_dag_) return true;
1247 WallTimer wall_timer;
1248 wall_timer.Start();
1249 log_info |= VLOG_IS_ON(1);
1250
1251 // Lets remove all fixed variables first.
1252 if (!Propagate(trail_)) return false;
1253 RemoveFixedVariables();
1254 const VariablesAssignment& assignment = trail_->Assignment();
1255 DCHECK(InvariantsAreOk());
1256
1257 // TODO(user): We could just do it directly though.
1258 const int32_t size(implications_.size());
1259 CompactVectorVector<int32_t, int32_t> scc;
1260 scc.reserve(size);
1261 double dtime = 0.0;
1262 {
1263 SccGraph::SccFinder finder;
1264 SccGraph graph(&finder, &implications_, &at_most_ones_,
1265 &at_most_one_buffer_);
1266 finder.FindStronglyConnectedComponents(size, graph, &scc);
1267 dtime += 4e-8 * graph.work_done_;
1268
1269 for (const Literal l : graph.to_fix_) {
1270 if (assignment.LiteralIsFalse(l)) return false;
1271 if (assignment.LiteralIsTrue(l)) continue;
1272 if (!FixLiteral(l)) return false;
1273 }
1274 }
1275
1276 // The old values will still be valid.
1277 representative_of_.resize(size, kNoLiteralIndex);
1278 is_redundant_.resize(size);
1279
1280 int num_equivalences = 0;
1281 reverse_topological_order_.clear();
1282 for (int index = 0; index < scc.size(); ++index) {
1283 const absl::Span<int32_t> component = scc[index];
1284
1285 // If one is fixed then all must be fixed. Note that the reason why the
1286 // propagation didn't already do that and we don't always get fixed
1287 // component of size 1 is because of the potential newly fixed literals
1288 // above.
1289 //
1290 // In any case, all fixed literals are marked as redundant.
1291 {
1292 bool all_fixed = false;
1293 bool all_true = false;
1294 for (const int32_t i : component) {
1295 const Literal l = Literal(LiteralIndex(i));
1296 if (trail_->Assignment().LiteralIsAssigned(l)) {
1297 all_fixed = true;
1298 all_true = trail_->Assignment().LiteralIsTrue(l);
1299 break;
1300 }
1301 }
1302 if (all_fixed) {
1303 for (const int32_t i : component) {
1304 const Literal l = Literal(LiteralIndex(i));
1305 if (!is_redundant_[l]) {
1306 ++num_redundant_literals_;
1307 is_redundant_.Set(l);
1308 representative_of_[l] = l.Index();
1309 }
1310 const Literal to_fix = all_true ? l : l.Negated();
1311 if (assignment.LiteralIsFalse(to_fix)) return false;
1312 if (assignment.LiteralIsTrue(to_fix)) continue;
1313 if (!FixLiteral(l)) return false;
1314 }
1315
1316 // Next component.
1317 continue;
1318 }
1319 }
1320
1321 // We ignore variable that appear in no constraints.
1322 if (component.size() == 1 && is_removed_[LiteralIndex(component[0])]) {
1323 continue;
1324 }
1325
1326 // We always take the smallest literal index (which also corresponds to the
1327 // smallest BooleanVariable index) as a representative. This make sure that
1328 // the representative of a literal l and the one of not(l) will be the
1329 // negation of each other. There is also reason to think that it is
1330 // heuristically better to use a BooleanVariable that was created first.
1331 std::sort(component.begin(), component.end());
1332 const LiteralIndex representative(component[0]);
1333 reverse_topological_order_.push_back(representative);
1334
1335 if (component.size() == 1) {
1336 // Note that because we process list in reverse topological order, this
1337 // is only needed if there is any equivalence before this point.
1338 if (num_equivalences > 0) {
1339 auto& representative_list = implications_[representative];
1340 for (Literal& ref : representative_list) {
1341 const LiteralIndex rep = representative_of_[ref];
1342 if (rep == representative) continue;
1343 if (rep == kNoLiteralIndex) continue;
1344 ref = Literal(rep);
1345 }
1346 gtl::STLSortAndRemoveDuplicates(&representative_list);
1347 }
1348 continue;
1349 }
1350
1351 // Sets the representative.
1352 for (int i = 1; i < component.size(); ++i) {
1353 const Literal literal = Literal(LiteralIndex(component[i]));
1354 if (!is_redundant_[literal]) {
1355 ++num_redundant_literals_;
1356 is_redundant_.Set(literal);
1357 }
1358 representative_of_[literal] = representative;
1359
1360 // Detect if x <=> not(x) which means unsat. Note that we relly on the
1361 // fact that when sorted, they will both be consecutive in the list.
1362 if (Literal(LiteralIndex(component[i - 1])).Negated() == literal) {
1363 LOG_IF(INFO, log_info) << "Trivially UNSAT in DetectEquivalences()";
1364 return false;
1365 }
1366 }
1367
1368 // Merge all the lists in implications_[representative].
1369 // Note that we do not want representative in its own list.
1370 auto& representative_list = implications_[representative];
1371 int new_size = 0;
1372 for (const Literal l : representative_list) {
1373 const Literal rep = RepresentativeOf(l);
1374 if (rep.Index() == representative) continue;
1375 representative_list[new_size++] = rep;
1376 }
1377 representative_list.resize(new_size);
1378 for (int i = 1; i < component.size(); ++i) {
1379 const Literal literal = Literal(LiteralIndex(component[i]));
1380 auto& ref = implications_[literal];
1381 for (const Literal l : ref) {
1382 const Literal rep = RepresentativeOf(l);
1383 if (rep.Index() != representative) representative_list.push_back(rep);
1384 }
1385
1386 // Add representative <=> literal.
1387 //
1388 // Remark: this relation do not need to be added to a DRAT proof since
1389 // the redundant variables should never be used again for a pure SAT
1390 // problem.
1391 representative_list.push_back(literal);
1392 ref.clear();
1393 ref.push_back(Literal(representative));
1394 }
1395 gtl::STLSortAndRemoveDuplicates(&representative_list);
1396 num_equivalences += component.size() - 1;
1397 }
1398
1399 is_dag_ = true;
1400 if (num_equivalences != 0) {
1401 // Remap all at most ones. Remove fixed variables, process duplicates. Note
1402 // that this might result in more implications when we expand small at most
1403 // one.
1404 at_most_ones_.clear();
1405 CleanUpAndAddAtMostOnes(/*base_index=*/0);
1406
1407 num_implications_ = 0;
1408 for (LiteralIndex i(0); i < size; ++i) {
1409 num_implications_ += implications_[i].size();
1410 }
1411 dtime += 2e-8 * num_implications_;
1412 }
1413
1414 time_limit_->AdvanceDeterministicTime(dtime);
1415 const int num_fixed_during_scc =
1416 trail_->Index() - num_processed_fixed_variables_;
1417 RemoveFixedVariables();
1418 DCHECK(InvariantsAreOk());
1419 LOG_IF(INFO, log_info) << "SCC. " << num_equivalences
1420 << " redundant equivalent literals. "
1421 << num_fixed_during_scc << " fixed. "
1422 << num_implications_ << " implications left. "
1423 << implications_.size() << " literals."
1424 << " size of at_most_one buffer = "
1425 << at_most_one_buffer_.size() << "."
1426 << " dtime: " << dtime
1427 << " wtime: " << wall_timer.Get();
1428 return true;
1429}

1431// Note that as a side effect this also do a full "failed literal probing"
1432// using the binary implication graph only.
1433//
1434// TODO(user): Track which literal have new implications, and only process
1435// the antecedants of these.
1436bool BinaryImplicationGraph::ComputeTransitiveReduction(bool log_info) {
1437 DCHECK_EQ(trail_->CurrentDecisionLevel(), 0);
1438 if (!DetectEquivalences()) return false;
1439
1440 // TODO(user): the situation with fixed variable is not really "clean".
1441 // Simplify the code so we are sure we don't run into issue or have to deal
1442 // with any of that here.
1443 if (!Propagate(trail_)) return false;
1444 RemoveFixedVariables();
1445 DCHECK(InvariantsAreOk());
1446
1447 log_info |= VLOG_IS_ON(1);
1448 WallTimer wall_timer;
1449 wall_timer.Start();
1450
1451 int64_t num_fixed = 0;
1452 int64_t num_new_redundant_implications = 0;
1453 bool aborted = false;
1454 tmp_removed_.clear();
1455 work_done_in_mark_descendants_ = 0;
1456 int marked_index = 0;
1457
1458 // For each node we do a graph traversal and only keep the literals
1459 // at maximum distance 1. This only works because we have a DAG when ignoring
1460 // the "redundant" literal marked by DetectEquivalences(). Note that we also
1461 // need no duplicates in the implications list for correctness which is also
1462 // guaranteed by DetectEquivalences().
1463 //
1464 // TODO(user): We should be able to reuse some propagation like it is done for
1465 // tree-look. Once a node is processed, we just need to process a node that
1466 // implies it. Test if we can make this faster. Alternatively, only clear
1467 // a part of is_marked_ (after the first child of root in reverse topo order).
1468 //
1469 // TODO(user): Can we exploit the fact that the implication graph is a
1470 // skew-symmetric graph (isomorphic to its transposed) so that we do less
1471 // work?
1472 const LiteralIndex size(implications_.size());
1473 LiteralIndex previous = kNoLiteralIndex;
1474 for (const LiteralIndex root : reverse_topological_order_) {
1475 // In most situation reverse_topological_order_ contains no redundant, fixed
1476 // or removed variables. But the reverse_topological_order_ is only
1477 // recomputed when new binary are added to the graph, not when new variable
1478 // are fixed.
1479 if (is_redundant_[root]) continue;
1480 if (trail_->Assignment().LiteralIsAssigned(Literal(root))) continue;
1481
1482 auto& direct_implications = implications_[root];
1483 if (direct_implications.empty()) continue;
1484
1485 // This is a "poor" version of the tree look stuff, but it does show good
1486 // improvement. If we just processed one of the child of root, we don't
1487 // need to re-explore it.
1488 //
1489 // TODO(user): Another optim we can do is that we never need to expand
1490 // any node with a reverse topo order smaller or equal to the min of the
1491 // ones in this list.
1492 bool clear_previous_reachability = true;
1493 for (const Literal direct_child : direct_implications) {
1494 if (direct_child.Index() == previous) {
1495 clear_previous_reachability = false;
1496 is_marked_.Clear(previous);
1497 break;
1498 }
1499 }
1500 if (clear_previous_reachability) {
1501 is_marked_.ClearAndResize(size);
1502 marked_index = 0;
1503 }
1504 previous = root;
1505
1506 for (const Literal direct_child : direct_implications) {
1507 if (is_redundant_[direct_child]) continue;
1508 if (is_marked_[direct_child]) continue;
1509
1510 // This is a corner case where because of equivalent literal, root
1511 // appear in implications_[root], we will remove it below.
1512 if (direct_child.Index() == root) continue;
1513
1514 // When this happens, then root must be false, we handle this just after
1515 // the loop.
1516 if (direct_child.NegatedIndex() == root) {
1517 is_marked_.Set(direct_child);
1518 break;
1519 }
1520
1521 MarkDescendants(direct_child);
1522
1523 // We have a DAG, so direct_child could only be marked first.
1524 is_marked_.Clear(direct_child);
1525 }
1526 DCHECK(!is_marked_[root])
1527 << "DetectEquivalences() should have removed cycles!";
1528 is_marked_.Set(root);
1529
1530 // Also mark all the ones reachable through the root AMOs.
1531 if (root < at_most_ones_.size()) {
1532 for (const int start : at_most_ones_[root]) {
1533 for (const Literal l : AtMostOne(start)) {
1534 if (l.Index() == root) continue;
1535 if (!is_marked_[l.Negated()] && !is_redundant_[l.Negated()]) {
1536 is_marked_.SetUnsafe(l.Negated());
1537 MarkDescendants(l.Negated());
1538 }
1539 }
1540 }
1541 }
1542
1543 // Failed literal probing. If both x and not(x) are marked then root must be
1544 // false. Note that because we process "roots" in reverse topological order,
1545 // we will fix the LCA of x and not(x) first.
1546 const auto& marked_positions = is_marked_.PositionsSetAtLeastOnce();
1547 for (; marked_index < marked_positions.size(); ++marked_index) {
1548 const LiteralIndex i = marked_positions[marked_index];
1549 if (is_marked_[Literal(i).NegatedIndex()]) {
1550 // We tested that at the beginning.
1551 DCHECK(!trail_->Assignment().LiteralIsAssigned(Literal(root)));
1552
1553 // We propagate right away the binary implications so that we do not
1554 // need to consider all antecedents of root in the transitive
1555 // reduction.
1556 ++num_fixed;
1557 if (!FixLiteral(Literal(root).Negated())) return false;
1558 break;
1559 }
1560 }
1561
1562 // Note that direct_implications will be cleared by
1563 // RemoveFixedVariables() that will need to inspect it to completely
1564 // remove Literal(root) from all lists.
1565 if (trail_->Assignment().LiteralIsAssigned(Literal(root))) continue;
1566
1567 // Only keep the non-marked literal (and the redundant one which are never
1568 // marked). We mark root to remove it in the corner case where it was
1569 // there.
1570 int new_size = 0;
1571 for (const Literal l : direct_implications) {
1572 if (!is_marked_[l]) {
1573 direct_implications[new_size++] = l;
1574 } else {
1575 tmp_removed_.push_back({Literal(root), l});
1576 DCHECK(!is_redundant_[l]);
1577 }
1578 }
1579 const int diff = direct_implications.size() - new_size;
1580 direct_implications.resize(new_size);
1581 direct_implications.shrink_to_fit();
1582 num_new_redundant_implications += diff;
1583 num_implications_ -= diff;
1584
1585 // Abort if the computation involved is too big.
1586 if (work_done_in_mark_descendants_ > 1e8) {
1587 aborted = true;
1588 break;
1589 }
1590 }
1591
1592 is_marked_.ClearAndResize(size);
1593
1594 // If we aborted early, we might no longer have both a=>b and not(b)=>not(a).
1595 // This is not desirable has some algo relies on this invariant. We fix this
1596 // here.
1597 if (aborted) {
1598 absl::flat_hash_set<std::pair<LiteralIndex, LiteralIndex>> removed;
1599 for (const auto [a, b] : tmp_removed_) {
1600 removed.insert({a.Index(), b.Index()});
1601 }
1602 for (LiteralIndex i(0); i < implications_.size(); ++i) {
1603 int new_size = 0;
1604 const LiteralIndex negated_i = Literal(i).NegatedIndex();
1605 auto& implication = implications_[i];
1606 for (const Literal l : implication) {
1607 if (removed.contains({l.NegatedIndex(), negated_i})) continue;
1608 implication[new_size++] = l;
1609 }
1610 implication.resize(new_size);
1611 }
1612 }
1613 if (num_fixed > 0) {
1614 RemoveFixedVariables();
1615 }
1616 DCHECK(InvariantsAreOk());
1617
1618 gtl::STLClearObject(&tmp_removed_);
1619 const double dtime = 1e-8 * work_done_in_mark_descendants_;
1620 time_limit_->AdvanceDeterministicTime(dtime);
1621 num_redundant_implications_ += num_new_redundant_implications;
1622 LOG_IF(INFO, log_info) << "Transitive reduction removed "
1623 << num_new_redundant_implications << " literals. "
1624 << num_fixed << " fixed. " << num_implications_
1625 << " implications left. " << implications_.size()
1626 << " literals." << " dtime: " << dtime
1627 << " wtime: " << wall_timer.Get()
1628 << (aborted ? " Aborted." : "");
1629 return true;
1630}

1634int ElementInIntersectionOrMinusOne(absl::Span<const int> a,
1635 absl::Span<const int> b) {
1636 DCHECK(std::is_sorted(a.begin(), a.end()));
1637 DCHECK(std::is_sorted(b.begin(), b.end()));
1638 if (a.empty() || b.empty()) return -1;
1639 int i = 0;
1640 int j = 0;
1641 while (true) {
1642 if (a[i] == b[j]) return a[i];
1643 if (a[i] < b[j]) {
1644 if (++i == a.size()) return -1;
1645 } else {
1646 if (++j == b.size()) return -1;
1647 }
1648 }

1653bool BinaryImplicationGraph::TransformIntoMaxCliques(
1654 std::vector<std::vector<Literal>>* at_most_ones,
1655 int64_t max_num_explored_nodes) {
1656 // The code below assumes a DAG.
1657 if (!DetectEquivalences()) return false;
1658 work_done_in_mark_descendants_ = 0;
1659
1660 int num_extended = 0;
1661 int num_removed = 0;
1662 int num_added = 0;
1663
1664 // Data to detect inclusion of base amo into extend amo.
1665 std::vector<int> detector_clique_index;
1666 CompactVectorVector<int> storage;
1667 InclusionDetector detector(storage);
1668 detector.SetWorkLimit(1e9);
1669
1670 std::vector<int> dense_index_to_index;
1671 absl::StrongVector<LiteralIndex, std::vector<int>> max_cliques_containing(
1672 implications_.size());
1673
1674 // We starts by processing larger constraints first.
1675 // But we want the output order to be stable.
1676 std::vector<std::pair<int, int>> index_size_vector;
1677 index_size_vector.reserve(at_most_ones->size());
1678 for (int index = 0; index < at_most_ones->size(); ++index) {
1679 std::vector<Literal>& clique = (*at_most_ones)[index];
1680 if (clique.size() <= 1) continue;
1681
1682 // Note(user): Because we always use literal with the smallest variable
1683 // indices as representative, this make sure that if possible, we express
1684 // the clique in term of user provided variable (that are always created
1685 // first).
1686 //
1687 // Remap the clique to only use representative.
1688 for (Literal& ref : clique) {
1689 DCHECK_LT(ref.Index(), representative_of_.size());
1690 const LiteralIndex rep = representative_of_[ref];
1691 if (rep == kNoLiteralIndex) continue;
1692 ref = Literal(rep);
1693 }
1694
1695 // We skip anything that can be presolved further as the code below do
1696 // not handle duplicate well.
1697 //
1698 // TODO(user): Shall we presolve it here?
1699 bool skip = false;
1700 std::sort(clique.begin(), clique.end());
1701 for (int i = 1; i < clique.size(); ++i) {
1702 if (clique[i] == clique[i - 1] || clique[i] == clique[i - i].Negated()) {
1703 skip = true;
1704 break;
1705 }
1706 }
1707 if (skip) continue;
1708
1709 index_size_vector.push_back({index, clique.size()});
1710 }
1711 std::stable_sort(
1712 index_size_vector.begin(), index_size_vector.end(),
1713 [](const std::pair<int, int> a, const std::pair<int, int>& b) {
1714 return a.second > b.second;
1715 });
1716
1717 absl::flat_hash_set<int> cannot_be_removed;
1718 std::vector<bool> was_extended(at_most_ones->size(), false);
1719 for (const auto& [index, old_size] : index_size_vector) {
1720 std::vector<Literal>& clique = (*at_most_ones)[index];
1721 if (time_limit_->LimitReached()) break;
1722
1723 // Special case for clique of size 2, we don't expand them if they
1724 // are included in an already added clique.
1725 if (clique.size() == 2) {
1726 DCHECK_NE(clique[0], clique[1]);
1727 const int dense_index = ElementInIntersectionOrMinusOne(
1728 max_cliques_containing[clique[0]], max_cliques_containing[clique[1]]);
1729 if (dense_index >= 0) {
1730 const int superset_index = dense_index_to_index[dense_index];
1731 if (was_extended[superset_index])
1732 cannot_be_removed.insert(superset_index);
1733 ++num_removed;
1734 clique.clear();
1735 continue;
1736 }
1737 }
1738
1739 // Save the non-extended version as possible subset.
1740 // TODO(user): Detect on the fly is superset already exist.
1741 detector_clique_index.push_back(index);
1742 detector.AddPotentialSubset(storage.AddLiterals(clique));
1743
1744 // We only expand the clique as long as we didn't spend too much time.
1745 if (work_done_in_mark_descendants_ < max_num_explored_nodes) {
1746 clique = ExpandAtMostOne(clique, max_num_explored_nodes);
1747 }
1748
1749 // Save the extended version as possible superset.
1750 detector_clique_index.push_back(index);
1751 detector.AddPotentialSuperset(storage.AddLiterals(clique));
1752
1753 // Also index clique for size 2 quick lookup.
1754 const int dense_index = dense_index_to_index.size();
1755 dense_index_to_index.push_back(index);
1756 for (const Literal l : clique) {
1757 max_cliques_containing[l].push_back(dense_index);
1758 }
1759
1760 if (clique.size() > old_size) {
1761 was_extended[index] = true;
1762 ++num_extended;
1763 }
1764 ++num_added;
1765 }
1766
1767 // Remove clique (before extension) that are included in an extended one.
1768 absl::flat_hash_set<int> cannot_be_superset;
1769 detector.DetectInclusions([&](int subset, int superset) {
1770 const int subset_index = detector_clique_index[subset];
1771 const int superset_index = detector_clique_index[superset];
1772 if (subset_index == superset_index) return;
1773
1774 // Abort if one was already deleted.
1775 if ((*at_most_ones)[subset_index].empty()) return;
1776 if ((*at_most_ones)[superset_index].empty()) return;
1777
1778 // If an extended clique already cover a deleted one, we cannot try to
1779 // remove it by looking at its non-extended version.
1780 if (cannot_be_removed.contains(subset_index)) return;
1781
1782 ++num_removed;
1783 (*at_most_ones)[subset_index].clear();
1784 if (was_extended[superset_index]) cannot_be_removed.insert(superset_index);
1785 });
1786
1787 if (num_extended > 0 || num_removed > 0 || num_added > 0) {
1788 VLOG(1) << "Clique Extended: " << num_extended
1789 << " Removed: " << num_removed << " Added: " << num_added
1790 << (work_done_in_mark_descendants_ > max_num_explored_nodes
1791 ? " (Aborted)"
1792 : "");
1793 }
1794 return true;
1795}

1797template <bool use_weight>
1798std::vector<Literal> BinaryImplicationGraph::ExpandAtMostOneWithWeight(
1799 const absl::Span<const Literal> at_most_one,
1800 const absl::StrongVector<LiteralIndex, bool>& can_be_included,
1801 const absl::StrongVector<LiteralIndex, double>& expanded_lp_values) {
1802 std::vector<Literal> clique(at_most_one.begin(), at_most_one.end());
1803 std::vector<LiteralIndex> intersection;
1804 double clique_weight = 0.0;
1805 const int64_t old_work = work_done_in_mark_descendants_;
1806 if (use_weight) {
1807 for (const Literal l : clique) {
1808 clique_weight += expanded_lp_values[l];
1809 }
1810 }
1811 for (int i = 0; i < clique.size(); ++i) {
1812 // Do not spend too much time here.
1813 if (work_done_in_mark_descendants_ - old_work > 1e8) break;
1814
1815 is_marked_.ClearAndResize(LiteralIndex(implications_.size()));
1816 MarkDescendants(clique[i]);
1817 if (i == 0) {
1818 for (const LiteralIndex index : is_marked_.PositionsSetAtLeastOnce()) {
1819 if (can_be_included[Literal(index).NegatedIndex()]) {
1820 intersection.push_back(index);
1821 }
1822 }
1823 for (const Literal l : clique) is_marked_.Clear(l.NegatedIndex());
1824 }
1825
1826 int new_size = 0;
1827 double intersection_weight = 0.0;
1828 is_marked_.Clear(clique[i]);
1829 is_marked_.Clear(clique[i].NegatedIndex());
1830 for (const LiteralIndex index : intersection) {
1831 if (!is_marked_[index]) continue;
1832 intersection[new_size++] = index;
1833 if (use_weight) {
1834 intersection_weight += expanded_lp_values[index];
1835 }
1836 }
1837 intersection.resize(new_size);
1838 if (intersection.empty()) break;
1839
1840 // We can't generate a violated cut this way. This is because intersection
1841 // contains all the possible ways to extend the current clique.
1842 if (use_weight && clique_weight + intersection_weight <= 1.0) {
1843 clique.clear();
1844 return clique;
1845 }
1846
1847 // Expand? The negation of any literal in the intersection is a valid way
1848 // to extend the clique.
1849 if (i + 1 == clique.size()) {
1850 // Heuristic: use literal with largest lp value. We randomize slightly.
1851 int index = -1;
1852 double max_lp = 0.0;
1853 for (int j = 0; j < intersection.size(); ++j) {
1854 // If we don't use weight, we prefer variable that comes first.
1855 const double lp =
1856 use_weight ? 1.0 - expanded_lp_values[intersection[j]] +
1857 absl::Uniform<double>(*random_, 0.0, 1e-4)
1858 : can_be_included.size() - intersection[j].value();
1859 if (index == -1 || lp > max_lp) {
1860 index = j;
1861 max_lp = lp;
1862 }
1863 }
1864 if (index != -1) {
1865 clique.push_back(Literal(intersection[index]).Negated());
1866 std::swap(intersection.back(), intersection[index]);
1867 intersection.pop_back();
1868 if (use_weight) {

1887const std::vector<std::vector<Literal>>&
1888BinaryImplicationGraph::GenerateAtMostOnesWithLargeWeight(
1889 const std::vector<Literal>& literals,
1890 const std::vector<double>& lp_values) {
1891 // We only want to generate a cut with literals from the LP, not extra ones.
1892 const int num_literals = implications_.size();
1893 absl::StrongVector<LiteralIndex, bool> can_be_included(num_literals, false);
1894 absl::StrongVector<LiteralIndex, double> expanded_lp_values(num_literals,
1895 0.0);
1896 const int size = literals.size();
1897 for (int i = 0; i < size; ++i) {
1898 const Literal l = literals[i];
1899 can_be_included[l] = true;
1900 can_be_included[l.NegatedIndex()] = true;
1901
1902 const double value = lp_values[i];
1903 expanded_lp_values[l] = value;
1904 expanded_lp_values[l.NegatedIndex()] = 1.0 - value;
1905 }
1906
1907 // We want highest sum first.
1908 struct Candidate {
1909 Literal a;
1910 Literal b;
1911 double sum;
1912 bool operator<(const Candidate& other) const { return sum > other.sum; }
1913 };
1914 std::vector<Candidate> candidates;
1915
1916 // First heuristic. Currently we only consider violated at most one of size 2,
1917 // and extend them. Right now, the code is a bit slow to try too many at every
1918 // LP node so it is why we are defensive like this. Note also that because we
1919 // currently still statically add the initial implications, this will only add
1920 // cut based on newly learned binary clause. Or the one that were not added
1921 // to the relaxation in the first place.
1922 for (int i = 0; i < size; ++i) {
1923 Literal current_literal = literals[i];
1924 double current_value = lp_values[i];
1925 if (trail_->Assignment().LiteralIsAssigned(current_literal)) continue;
1926 if (is_redundant_[current_literal]) continue;
1927
1928 if (current_value < 0.5) {
1929 current_literal = current_literal.Negated();
1930 current_value = 1.0 - current_value;
1931 }
1932
1933 // We consider only one candidate for each current_literal.
1934 LiteralIndex best = kNoLiteralIndex;
1935 double best_value = 0.0;
1936 for (const Literal l : implications_[current_literal]) {
1937 if (!can_be_included[l]) continue;
1938 const double activity =
1939 current_value + expanded_lp_values[l.NegatedIndex()];
1940 if (activity <= 1.01) continue;
1941 const double v = activity + absl::Uniform<double>(*random_, 0.0, 1e-4);
1942 if (best == kNoLiteralIndex || v > best_value) {
1943 best_value = v;
1944 best = l.NegatedIndex();
1945 }
1946 }
1947 if (best != kNoLiteralIndex) {
1948 const double activity = current_value + expanded_lp_values[best];
1949 candidates.push_back({current_literal, Literal(best), activity});
1950 }
1951 }
1952
1953 // Do not genate too many cut at once.
1954 const int kMaxNumberOfCutPerCall = 10;
1955 std::sort(candidates.begin(), candidates.end());
1956 if (candidates.size() > kMaxNumberOfCutPerCall) {
1957 candidates.resize(kMaxNumberOfCutPerCall);
1958 }
1959
1960 // Expand to a maximal at most one each candidates before returning them.
1961 // Note that we only expand using literal from the LP.
1962 tmp_cuts_.clear();
1963 std::vector<Literal> at_most_one;
1964 for (const Candidate& candidate : candidates) {
1965 at_most_one = ExpandAtMostOneWithWeight(
1966 {candidate.a, candidate.b}, can_be_included, expanded_lp_values);
1967 if (!at_most_one.empty()) tmp_cuts_.push_back(at_most_one);
1968 }
1969 return tmp_cuts_;
1970}

1975std::vector<absl::Span<const Literal>>
1976BinaryImplicationGraph::HeuristicAmoPartition(std::vector<Literal>* literals) {
1977 std::vector<absl::Span<const Literal>> result;
1978
1979 absl::StrongVector<LiteralIndex, bool> to_consider(implications_.size(),
1980 false);
1981 for (const Literal l : *literals) to_consider[l] = true;
1982
1983 // Priority queue of (intersection_size, start_of_amo).
1984 std::priority_queue<std::pair<int, int>> pq;
1985
1986 // Compute for each at most one that might overlap, its overlaping size and
1987 // stores its start in the at_most_one_buffer_.
1988 //
1989 // This is in O(num_literal in amo).
1990 absl::flat_hash_set<int> explored_amo;
1991 for (const Literal l : *literals) {
1992 if (l.Index() >= at_most_ones_.size()) continue;
1993 for (const int start : at_most_ones_[l]) {
1994 const auto [_, inserted] = explored_amo.insert(start);
1995 if (!inserted) continue;
1996
1997 int intersection_size = 0;
1998 for (const Literal l : AtMostOne(start)) {
1999 if (to_consider[l]) ++intersection_size;
2000 }
2001 if (intersection_size > 1) {
2002 pq.push({intersection_size, start});
2003 }
2004
2005 // Abort early if we are done.
2006 if (intersection_size == literals->size()) break;
2007 }
2008 }
2009
2010 // Consume AMOs, update size.
2011 int num_processed = 0;
2012 while (!pq.empty()) {
2013 const auto [old_size, start] = pq.top();
2014 pq.pop();
2015
2016 // Recompute size.
2017 int intersection_size = 0;
2018 for (const Literal l : AtMostOne(start)) {
2019 if (to_consider[l]) ++intersection_size;
2020 }
2021 if (intersection_size != old_size) {
2022 // re-add with new size.
2023 if (intersection_size > 1) {
2024 pq.push({intersection_size, start});
2025 }
2026 continue;
2027 }
2028
2029 // Mark the literal of that at most one at false.
2030 for (const Literal l : AtMostOne(start)) {
2031 to_consider[l] = false;
2032 }
2033
2034 // Extract the intersection by moving it in
2035 // [num_processed, num_processed + intersection_size).
2036 const int span_start = num_processed;
2037 for (int i = num_processed; i < literals->size(); ++i) {
2038 if (to_consider[(*literals)[i]]) continue;
2039 std::swap((*literals)[num_processed], (*literals)[i]);
2040 ++num_processed;
2041 }
2042 result.push_back(absl::MakeSpan(literals->data() + span_start,
2043 num_processed - span_start));
2044 }
2045 return result;
2046}

2048void BinaryImplicationGraph::MarkDescendants(Literal root) {
2049 bfs_stack_.resize(implications_.size());
2050 auto* const stack = bfs_stack_.data();
2051 const int amo_size = static_cast<int>(at_most_ones_.size());
2052 auto is_marked = is_marked_.const_view();
2053 auto is_redundant = is_redundant_.const_view();
2054 if (is_redundant[root]) return;
2055
2056 int stack_size = 1;
2057 stack[0] = root;
2058 is_marked_.Set(root);
2059 for (int j = 0; j < stack_size; ++j) {
2060 const Literal current = stack[j];
2061 if (!implies_something_[current]) continue;
2062
2063 for (const Literal l : implications_[current]) {
2064 if (!is_marked[l] && !is_redundant[l]) {
2065 is_marked_.SetUnsafe(l);
2066 stack[stack_size++] = l;
2067 }
2068 }
2069
2070 if (current.Index() >= amo_size) continue;
2071 for (const int start : at_most_ones_[current]) {
2072 for (const Literal l : AtMostOne(start)) {
2073 if (l == current) continue;
2074 if (!is_marked[l.NegatedIndex()] && !is_redundant[l.NegatedIndex()]) {
2075 is_marked_.SetUnsafe(l.NegatedIndex());
2076 stack[stack_size++] = l.Negated();
2077 }
2078 }
2079 }
2080 }
2081 work_done_in_mark_descendants_ += stack_size;
2082}
