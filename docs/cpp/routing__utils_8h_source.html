24// Tracks whether bins constrained by several nonnegative dimensions can contain
25// items added incrementally. Also tracks soft violation costs.
26class BinCapacities {
27 public:
28 explicit BinCapacities(int num_bins)
29 : num_bins_(num_bins),
30 load_per_bin_(num_bins),
31 load_limits_per_bin_(num_bins),
32 total_cost_(0) {}
33
34 // Represents the limits of a bin with respect to a dimension:
35 // - max_load is a max total load, can cause TryAddItemToBin() to return false
36 // if load would exceed max_load.
37 // - soft_max_load is a max load that can be exceeded, causing the TotalCost()
38 // to increase. Initial value *may* be negative, to help with modelling.
39 // - cost_above_soft_max_load is the cost incurred per unit by which load
40 // exceeds soft_max_load.
41 struct LoadLimit {
42 int64_t max_load;
43 int64_t soft_max_load;
44 int64_t cost_above_soft_max_load;
45 };
46
47 void AddDimension(
48 std::function<int64_t(int, int)> load_demand_of_item_for_bin,
49 std::vector<LoadLimit> load_limit_per_bin);
50 int NumDimensions() const { return load_demands_per_dimension_.size(); }
51
52 // Checks whether adding item(s) is feasible w.r.t. dimensions.
53 bool CheckAdditionFeasibility(int item, int bin) const;
54 bool CheckAdditionsFeasibility(const std::vector<int>& items, int bin) const;
55 // Adds item to bin, returns whether the bin is feasible.
56 // The item is still added even when infeasible.
57 bool AddItemToBin(int item, int bin);
58 // Removes item from bin, return whether the bin is feasible.
59 bool RemoveItemFromBin(int item, int bin);
60 // Returns the total cost incurred by violating soft costs.
61 int64_t TotalCost() const { return total_cost_; }
62 // Removes all items from given bin, resets the total cost.
63 void ClearItemsOfBin(int bin);
64 // Removes all items from all bins.
65 void ClearItems();
66
67 private:
68 const int num_bins_;
69 // load_per_bin_[bin][dimension]
70 std::vector<std::vector<int64_t>> load_per_bin_;
71 // load_limits_per_bin_[bin][dimension].
72 std::vector<std::vector<LoadLimit>> load_limits_per_bin_;
73 // load_demands_per_dimension_[dimension](item, bin).
74 std::vector<std::function<int64_t(int, int)>> load_demands_per_dimension_;
75 int64_t total_cost_;
76};
