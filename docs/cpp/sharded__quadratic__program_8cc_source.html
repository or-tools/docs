34// Logs a warning if `matrix` has more than `density_limit` non-zeros in
35// a single column.
36void WarnIfMatrixUnbalanced(
37 const Eigen::SparseMatrix<double, Eigen::ColMajor, int64_t>& matrix,
38 absl::string_view matrix_name, int64_t density_limit) {
39 if (matrix.cols() == 0) return;
40 int64_t worst_column = 0;
41 for (int64_t col = 0; col < matrix.cols(); ++col) {
42 if (matrix.col(col).nonZeros() > matrix.col(worst_column).nonZeros()) {
43 worst_column = col;
44 }
45 }
46 if (matrix.col(worst_column).nonZeros() > density_limit) {
47 // TODO(user): fix this automatically in presolve instead of asking the
48 // user to do it.
49 LOG(WARNING)
50 << "The " << matrix_name << " has "
51 << matrix.col(worst_column).nonZeros() << " non-zeros in its "
52 << worst_column
53 << "th column. For best parallelization all rows and columns should "
54 "have at most order "
55 << density_limit
56 << " non-zeros. Consider rewriting the QP to split the corresponding "
57 "variable or constraint.";
58 }
59}

63ShardedQuadraticProgram::ShardedQuadraticProgram(QuadraticProgram qp,
64 const int num_threads,
65 const int num_shards)
66 : qp_(std::move(qp)),
67 transposed_constraint_matrix_(qp_.constraint_matrix.transpose()),
68 thread_pool_(num_threads == 1
69 ? nullptr
70 : std::make_unique<ThreadPool>("PDLP", num_threads)),
71 constraint_matrix_sharder_(qp_.constraint_matrix, num_shards,
72 thread_pool_.get()),
73 transposed_constraint_matrix_sharder_(transposed_constraint_matrix_,
74 num_shards, thread_pool_.get()),
75 primal_sharder_(qp_.variable_lower_bounds.size(), num_shards,
76 thread_pool_.get()),
77 dual_sharder_(qp_.constraint_lower_bounds.size(), num_shards,
78 thread_pool_.get()) {
79 CHECK_GE(num_threads, 1);
80 CHECK_GE(num_shards, num_threads);
81 if (num_threads > 1) {
82 thread_pool_->StartWorkers();
83 const int64_t work_per_iteration = qp_.constraint_matrix.nonZeros() +
84 qp_.variable_lower_bounds.size() +
85 qp_.constraint_lower_bounds.size();
86 const int64_t column_density_limit = work_per_iteration / num_threads;
87 WarnIfMatrixUnbalanced(qp_.constraint_matrix, "constraint matrix",
88 column_density_limit);
89 WarnIfMatrixUnbalanced(transposed_constraint_matrix_,
90 "transposed constraint matrix",
91 column_density_limit);
92 }

97// Multiply each entry of `matrix` by the corresponding element of
98// `row_scaling_vec` and `col_scaling_vec`, i.e.,
99// `matrix[i,j] *= row_scaling_vec[i] * col_scaling_vec[j]`.
100void ScaleMatrix(
101 const Eigen::VectorXd& col_scaling_vec,
102 const Eigen::VectorXd& row_scaling_vec, const Sharder& sharder,
103 Eigen::SparseMatrix<double, Eigen::ColMajor, int64_t>& matrix) {
104 CHECK_EQ(matrix.cols(), col_scaling_vec.size());
105 CHECK_EQ(matrix.rows(), row_scaling_vec.size());
106 sharder.ParallelForEachShard([&](const Sharder::Shard& shard) {
107 auto matrix_shard = shard(matrix);
108 auto col_scaling_vec_shard = shard(col_scaling_vec);
109 for (int64_t col_num = 0; col_num < shard(matrix).outerSize(); ++col_num) {
110 for (decltype(matrix_shard)::InnerIterator it(matrix_shard, col_num); it;
111 ++it) {
112 it.valueRef() *=
113 row_scaling_vec[it.row()] * col_scaling_vec_shard[it.col()];
114 }
115 }
116 });
117}

119void ReplaceLargeValuesWithInfinity(const double threshold,
120 const Sharder& sharder,
121 Eigen::VectorXd& vector) {
122 constexpr double kInfinity = std::numeric_limits<double>::infinity();
123 sharder.ParallelForEachShard([&](const Sharder::Shard& shard) {
124 auto vector_shard = shard(vector);
125 for (int64_t i = 0; i < vector_shard.size(); ++i) {
126 if (vector_shard[i] <= -threshold) vector_shard[i] = -kInfinity;
127 if (vector_shard[i] >= threshold) vector_shard[i] = kInfinity;
128 }
129 });
130}

134void ShardedQuadraticProgram::RescaleQuadraticProgram(
135 const Eigen::VectorXd& col_scaling_vec,
136 const Eigen::VectorXd& row_scaling_vec) {
137 CHECK_EQ(PrimalSize(), col_scaling_vec.size());
138 CHECK_EQ(DualSize(), row_scaling_vec.size());
139 primal_sharder_.ParallelForEachShard([&](const Sharder::Shard& shard) {
140 CHECK((shard(col_scaling_vec).array() > 0.0).all());
141 shard(qp_.objective_vector) =
142 shard(qp_.objective_vector).cwiseProduct(shard(col_scaling_vec));
143 shard(qp_.variable_lower_bounds) =
144 shard(qp_.variable_lower_bounds).cwiseQuotient(shard(col_scaling_vec));
145 shard(qp_.variable_upper_bounds) =
146 shard(qp_.variable_upper_bounds).cwiseQuotient(shard(col_scaling_vec));
147 if (!IsLinearProgram(qp_)) {
148 shard(qp_.objective_matrix->diagonal()) =
149 shard(qp_.objective_matrix->diagonal())
150 .cwiseProduct(
151 shard(col_scaling_vec).cwiseProduct(shard(col_scaling_vec)));
152 }
153 });
154 dual_sharder_.ParallelForEachShard([&](const Sharder::Shard& shard) {
155 CHECK((shard(row_scaling_vec).array() > 0.0).all());
156 shard(qp_.constraint_lower_bounds) =
157 shard(qp_.constraint_lower_bounds).cwiseProduct(shard(row_scaling_vec));
158 shard(qp_.constraint_upper_bounds) =
159 shard(qp_.constraint_upper_bounds).cwiseProduct(shard(row_scaling_vec));
160 });
161
162 ScaleMatrix(col_scaling_vec, row_scaling_vec, constraint_matrix_sharder_,
163 qp_.constraint_matrix);
164 ScaleMatrix(row_scaling_vec, col_scaling_vec,
165 transposed_constraint_matrix_sharder_,
166 transposed_constraint_matrix_);
167}

169void ShardedQuadraticProgram::ReplaceLargeConstraintBoundsWithInfinity(
170 const double threshold) {
171 ReplaceLargeValuesWithInfinity(threshold, DualSharder(),
172 qp_.constraint_lower_bounds);
173 ReplaceLargeValuesWithInfinity(threshold, DualSharder(),
174 qp_.constraint_upper_bounds);

177void ShardedQuadraticProgram::SetConstraintBounds(
178 int64_t constraint_index, std::optional<double> lower_bound,
179 std::optional<double> upper_bound) {
180 CHECK_LT(constraint_index, DualSize());
181 CHECK_GE(constraint_index, 0);

184 }
185 if (upper_bound.has_value()) {
