33// If we hardcode shift_amount to 32, the 32-bit formula is:
34// magic_number = 2 ^ 64 / d
35// value / d = value * magic_number >> 64
36//
37// One caveat is that for d == 1, magic_number takes 65 bits overflowing a
38// uint64_t. So, we again disallow inputs with d == 1.
39ConstantDivisor<uint32_t>::ConstantDivisor(value_type d)
40 : ConstantDivisorBase((std::numeric_limits<MagicValueType>::max() / d) + 1,
41 d) {
42 CHECK_GT(d, 1) << "ConstantDivisor<uint32_t> only supports denominators > 1.";
43}
