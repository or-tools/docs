79void IntegerEncoder::FullyEncodeVariable(IntegerVariable var) {
80 if (VariableIsFullyEncoded(var)) return;
81
82 CHECK_EQ(0, sat_solver_->CurrentDecisionLevel());
83 var = PositiveVariable(var);
84 const PositiveOnlyIndex index = GetPositiveOnlyIndex(var);
85 CHECK(!domains_[index].IsEmpty()); // UNSAT. We don't deal with that here.
86 CHECK_LT(domains_[index].Size(), 100000)
87 << "Domain too large for full encoding.";
88
89 // TODO(user): Maybe we can optimize the literal creation order and their
90 // polarity as our default SAT heuristics initially depends on this.
91 //
92 // TODO(user): Currently, in some corner cases,
93 // GetOrCreateLiteralAssociatedToEquality() might trigger some propagation
94 // that update the domain of var, so we need to cache the values to not read
95 // garbage. Note that it is okay to call the function on values no longer
96 // reachable, as this will just do nothing.
97 tmp_values_.clear();
98 for (const int64_t v : domains_[index].Values()) {
99 tmp_values_.push_back(IntegerValue(v));
100 }
101 for (const IntegerValue v : tmp_values_) {
102 GetOrCreateLiteralAssociatedToEquality(var, v);
103 }
104
105 // Mark var and Negation(var) as fully encoded.
106 DCHECK_LT(GetPositiveOnlyIndex(var), is_fully_encoded_.size());
107 is_fully_encoded_[GetPositiveOnlyIndex(var)] = true;
108}

110bool IntegerEncoder::VariableIsFullyEncoded(IntegerVariable var) const {
111 const PositiveOnlyIndex index = GetPositiveOnlyIndex(var);
112 if (index >= is_fully_encoded_.size()) return false;
113
114 // Once fully encoded, the status never changes.
115 if (is_fully_encoded_[index]) return true;
116 var = PositiveVariable(var);
117
118 // TODO(user): Cache result as long as equality_by_var_[index] is unchanged?
119 // It might not be needed since if the variable is not fully encoded, then
120 // PartialDomainEncoding() will filter unreachable values, and so the size
121 // check will be false until further value have been encoded.
122 const int64_t initial_domain_size = domains_[index].Size();
123 if (equality_by_var_[index].size() < initial_domain_size) return false;
124
125 // This cleans equality_by_var_[index] as a side effect and in particular,
126 // sorts it by values.
127 PartialDomainEncoding(var);
128
129 // TODO(user): Comparing the size might be enough, but we want to be always
130 // valid even if either (*domains_[var]) or PartialDomainEncoding(var) are
131 // not properly synced because the propagation is not finished.
132 const auto& ref = equality_by_var_[index];
133 int i = 0;
134 for (const int64_t v : domains_[index].Values()) {
135 if (i < ref.size() && v == ref[i].value) {
136 i++;
137 }
138 }
139 if (i == ref.size()) {
140 is_fully_encoded_[index] = true;
141 }
142 return is_fully_encoded_[index];
143}

151const std::vector<ValueLiteralPair>& IntegerEncoder::PartialDomainEncoding(
152 IntegerVariable var) const {
153 const PositiveOnlyIndex index = GetPositiveOnlyIndex(var);
154 if (index >= equality_by_var_.size()) {
155 partial_encoding_.clear();
156 return partial_encoding_;
157 }
158
159 int new_size = 0;
160 partial_encoding_.assign(equality_by_var_[index].begin(),
161 equality_by_var_[index].end());
162 for (int i = 0; i < partial_encoding_.size(); ++i) {
163 const ValueLiteralPair pair = partial_encoding_[i];
164 if (sat_solver_->Assignment().LiteralIsFalse(pair.literal)) continue;
165 if (sat_solver_->Assignment().LiteralIsTrue(pair.literal)) {
166 partial_encoding_.clear();
167 partial_encoding_.push_back(pair);
168 new_size = 1;
169 break;
170 }
171 partial_encoding_[new_size++] = pair;
172 }
173 partial_encoding_.resize(new_size);
174 std::sort(partial_encoding_.begin(), partial_encoding_.end(),
175 ValueLiteralPair::CompareByValue());
176
177 if (trail_->CurrentDecisionLevel() == 0) {
178 // We can cleanup the current encoding in this case.
179 equality_by_var_[index].assign(partial_encoding_.begin(),
180 partial_encoding_.end());
181 }
182
183 if (!VariableIsPositive(var)) {
184 std::reverse(partial_encoding_.begin(), partial_encoding_.end());
185 for (ValueLiteralPair& ref : partial_encoding_) ref.value = -ref.value;
186 }
187 return partial_encoding_;
188}

190// Note that by not inserting the literal in "order" we can in the worst case
191// use twice as much implication (2 by literals) instead of only one between
192// consecutive literals.
193void IntegerEncoder::AddImplications(
194 const absl::btree_map<IntegerValue, Literal>& map,
195 absl::btree_map<IntegerValue, Literal>::const_iterator it,
196 Literal associated_lit) {
197 if (!add_implications_) return;
198 DCHECK_EQ(it->second, associated_lit);
199
200 // Literal(after) => associated_lit
201 auto after_it = it;
202 ++after_it;
203 if (after_it != map.end()) {
204 sat_solver_->AddClauseDuringSearch(
205 {after_it->second.Negated(), associated_lit});
206 }
207
208 // associated_lit => Literal(before)
209 if (it != map.begin()) {
210 auto before_it = it;
211 --before_it;
212 sat_solver_->AddClauseDuringSearch(
213 {associated_lit.Negated(), before_it->second});
214 }
215}

217void IntegerEncoder::AddAllImplicationsBetweenAssociatedLiterals() {
218 CHECK_EQ(0, sat_solver_->CurrentDecisionLevel());
219 add_implications_ = true;
220
221 // This is tricky: AddBinaryClause() might trigger propagation that causes the
222 // encoding to be filtered. So we make a copy...
223 const int num_vars = encoding_by_var_.size();
224 for (PositiveOnlyIndex index(0); index < num_vars; ++index) {
225 LiteralIndex previous = kNoLiteralIndex;
226 const IntegerVariable var(2 * index.value());
227 for (const auto [unused, literal] : PartialGreaterThanEncoding(var)) {
228 if (previous != kNoLiteralIndex) {
229 // literal => previous.
230 sat_solver_->AddBinaryClause(literal.Negated(), Literal(previous));
231 }
232 previous = literal.Index();
233 }
234 }

237std::pair<IntegerLiteral, IntegerLiteral> IntegerEncoder::Canonicalize(
238 IntegerLiteral i_lit) const {
239 const bool positive = VariableIsPositive(i_lit.var);
240 if (!positive) i_lit = i_lit.Negated();
241
242 const IntegerVariable var(i_lit.var);
243 const PositiveOnlyIndex index = GetPositiveOnlyIndex(var);
244 IntegerValue after(i_lit.bound);
245 IntegerValue before(i_lit.bound - 1);
246 DCHECK_GE(before, domains_[index].Min());
247 DCHECK_LE(after, domains_[index].Max());
248 int64_t previous = std::numeric_limits<int64_t>::min();
249 for (const ClosedInterval& interval : domains_[index]) {
250 if (before > previous && before < interval.start) before = previous;
251 if (after > previous && after < interval.start) after = interval.start;
252 if (after <= interval.end) break;
253 previous = interval.end;
254 }
255 if (positive) {
256 return {IntegerLiteral::GreaterOrEqual(var, after),
257 IntegerLiteral::LowerOrEqual(var, before)};
258 } else {
259 return {IntegerLiteral::LowerOrEqual(var, before),
260 IntegerLiteral::GreaterOrEqual(var, after)};
261 }

264Literal IntegerEncoder::GetOrCreateAssociatedLiteral(IntegerLiteral i_lit) {
265 // Remove trivial literal.
266 {
267 const PositiveOnlyIndex index = GetPositiveOnlyIndex(i_lit.var);
268 if (VariableIsPositive(i_lit.var)) {
269 if (i_lit.bound <= domains_[index].Min()) return GetTrueLiteral();
270 if (i_lit.bound > domains_[index].Max()) return GetFalseLiteral();
271 } else {
272 const IntegerValue bound = -i_lit.bound;
273 if (bound >= domains_[index].Max()) return GetTrueLiteral();
274 if (bound < domains_[index].Min()) return GetFalseLiteral();
275 }
276 }
277
278 // Canonicalize and see if we have an equivalent literal already.
279 const auto canonical_lit = Canonicalize(i_lit);
280 if (VariableIsPositive(i_lit.var)) {
281 const LiteralIndex index = GetAssociatedLiteral(canonical_lit.first);
282 if (index != kNoLiteralIndex) return Literal(index);
283 } else {
284 const LiteralIndex index = GetAssociatedLiteral(canonical_lit.second);
285 if (index != kNoLiteralIndex) return Literal(index).Negated();
286 }
287
288 ++num_created_variables_;
289 const Literal literal(sat_solver_->NewBooleanVariable(), true);
290 AssociateToIntegerLiteral(literal, canonical_lit.first);
291
292 // TODO(user): on some problem this happens. We should probably make sure that
293 // we don't create extra fixed Boolean variable for no reason.
294 if (sat_solver_->Assignment().LiteralIsAssigned(literal)) {
295 VLOG(1) << "Created a fixed literal for no reason!";
296 }
297 return literal;
298}

318Literal IntegerEncoder::GetOrCreateLiteralAssociatedToEquality(
319 IntegerVariable var, IntegerValue value) {
320 {
321 const auto it =
322 equality_to_associated_literal_.find(PositiveVarKey(var, value));
323 if (it != equality_to_associated_literal_.end()) {
324 return it->second;
325 }
326 }
327
328 // Check for trivial true/false literal to avoid creating variable for no
329 // reasons.
330 const Domain& domain = domains_[GetPositiveOnlyIndex(var)];
331 if (!domain.Contains(VariableIsPositive(var) ? value.value()
332 : -value.value())) {
333 return GetFalseLiteral();
334 }
335 if (domain.IsFixed()) {
336 AssociateToIntegerEqualValue(GetTrueLiteral(), var, value);
337 return GetTrueLiteral();
338 }
339
340 ++num_created_variables_;
341 const Literal literal(sat_solver_->NewBooleanVariable(), true);
342 AssociateToIntegerEqualValue(literal, var, value);
343
344 // TODO(user): this happens on some problem. We should probably
345 // make sure that we don't create extra fixed Boolean variable for no reason.
346 // Note that here we could detect the case before creating the literal. The
347 // initial domain didn't contain it, but maybe the one of (>= value) or (<=
348 // value) is false?
349 if (sat_solver_->Assignment().LiteralIsAssigned(literal)) {
350 VLOG(1) << "Created a fixed literal for no reason!";
351 }
352 return literal;
353}

355void IntegerEncoder::AssociateToIntegerLiteral(Literal literal,
356 IntegerLiteral i_lit) {
357 // Always transform to positive variable.
358 if (!VariableIsPositive(i_lit.var)) {
359 i_lit = i_lit.Negated();
360 literal = literal.Negated();
361 }
362
363 const PositiveOnlyIndex index = GetPositiveOnlyIndex(i_lit.var);
364 const Domain& domain = domains_[index];
365 const IntegerValue min(domain.Min());
366 const IntegerValue max(domain.Max());
367 if (i_lit.bound <= min) {
368 return (void)sat_solver_->AddUnitClause(literal);
369 }
370 if (i_lit.bound > max) {
371 return (void)sat_solver_->AddUnitClause(literal.Negated());
372 }
373
374 if (index >= encoding_by_var_.size()) {
375 encoding_by_var_.resize(index.value() + 1);
376 }
377 auto& var_encoding = encoding_by_var_[index];
378
379 // We just insert the part corresponding to the literal with positive
380 // variable.
381 const auto canonical_pair = Canonicalize(i_lit);
382 const auto [it, inserted] =
383 var_encoding.insert({canonical_pair.first.bound, literal});
384 if (!inserted) {
385 const Literal associated(it->second);
386 if (associated != literal) {
387 DCHECK_EQ(sat_solver_->CurrentDecisionLevel(), 0);
388 sat_solver_->AddClauseDuringSearch({literal, associated.Negated()});
389 sat_solver_->AddClauseDuringSearch({literal.Negated(), associated});
390 }
391 return;
392 }
393 AddImplications(var_encoding, it, literal);
394
395 // Corner case if adding implication cause this to be fixed.
396 if (sat_solver_->CurrentDecisionLevel() == 0) {
397 if (sat_solver_->Assignment().LiteralIsTrue(literal)) {
398 delayed_to_fix_->integer_literal_to_fix.push_back(canonical_pair.first);
399 }
400 if (sat_solver_->Assignment().LiteralIsFalse(literal)) {
401 delayed_to_fix_->integer_literal_to_fix.push_back(canonical_pair.second);
402 }
403 }
404
405 // Resize reverse encoding.
406 const int new_size =
407 1 + std::max(literal.Index().value(), literal.NegatedIndex().value());
408 if (new_size > reverse_encoding_.size()) {
409 reverse_encoding_.resize(new_size);
410 }
411 reverse_encoding_[literal].push_back(canonical_pair.first);
412 reverse_encoding_[literal.NegatedIndex()].push_back(canonical_pair.second);
413
414 // Detect the case >= max or <= min and properly register them. Note that
415 // both cases will happen at the same time if there is just two possible
416 // value in the domain.
417 if (canonical_pair.first.bound == max) {
418 AssociateToIntegerEqualValue(literal, i_lit.var, max);
419 }
420 if (-canonical_pair.second.bound == min) {
421 AssociateToIntegerEqualValue(literal.Negated(), i_lit.var, min);
422 }
423}

425void IntegerEncoder::AssociateToIntegerEqualValue(Literal literal,
426 IntegerVariable var,
427 IntegerValue value) {
428 // The function is symmetric and we only deal with positive variable.
429 if (!VariableIsPositive(var)) {
430 var = NegationOf(var);
431 value = -value;
432 }
433
434 // Detect literal view. Note that the same literal can be associated to more
435 // than one variable, and thus already have a view. We don't change it in
436 // this case.
437 const PositiveOnlyIndex index = GetPositiveOnlyIndex(var);
438 const Domain& domain = domains_[index];
439 if (value == 1 && domain.Min() >= 0 && domain.Max() <= 1) {
440 if (literal.Index() >= literal_view_.size()) {
441 literal_view_.resize(literal.Index().value() + 1, kNoIntegerVariable);
442 literal_view_[literal] = var;
443 } else if (literal_view_[literal] == kNoIntegerVariable) {
444 literal_view_[literal] = var;
445 }
446 }
447 if (value == -1 && domain.Min() >= -1 && domain.Max() <= 0) {
448 if (literal.Index() >= literal_view_.size()) {
449 literal_view_.resize(literal.Index().value() + 1, kNoIntegerVariable);
450 literal_view_[literal] = NegationOf(var);
451 } else if (literal_view_[literal] == kNoIntegerVariable) {
452 literal_view_[literal] = NegationOf(var);
453 }
454 }
455
456 // We use the "do not insert if present" behavior of .insert() to do just one
457 // lookup.
458 const auto insert_result = equality_to_associated_literal_.insert(
459 {PositiveVarKey(var, value), literal});
460 if (!insert_result.second) {
461 // If this key is already associated, make the two literals equal.
462 const Literal representative = insert_result.first->second;
463 if (representative != literal) {
464 sat_solver_->AddClauseDuringSearch({literal, representative.Negated()});
465 sat_solver_->AddClauseDuringSearch({literal.Negated(), representative});
466 }
467 return;
468 }
469
470 // Fix literal for value outside the domain.
471 if (!domain.Contains(value.value())) {
472 return (void)sat_solver_->AddUnitClause(literal.Negated());
473 }
474
475 // Update equality_by_var. Note that due to the
476 // equality_to_associated_literal_ hash table, there should never be any
477 // duplicate values for a given variable.
478 if (index >= equality_by_var_.size()) {
479 equality_by_var_.resize(index.value() + 1);
480 is_fully_encoded_.resize(index.value() + 1);
481 }
482 equality_by_var_[index].push_back({value, literal});
483
484 // Fix literal for constant domain.
485 if (domain.IsFixed()) {
486 return (void)sat_solver_->AddUnitClause(literal);
487 }
488
489 const IntegerLiteral ge = IntegerLiteral::GreaterOrEqual(var, value);
490 const IntegerLiteral le = IntegerLiteral::LowerOrEqual(var, value);
491
492 // Special case for the first and last value.
493 if (value == domain.Min()) {
494 // Note that this will recursively call AssociateToIntegerEqualValue() but
495 // since equality_to_associated_literal_[] is now set, the recursion will
496 // stop there. When a domain has just 2 values, this allows to call just
497 // once AssociateToIntegerEqualValue() and also associate the other value to
498 // the negation of the given literal.
499 AssociateToIntegerLiteral(literal, le);
500 return;
501 }
502 if (value == domain.Max()) {
503 AssociateToIntegerLiteral(literal, ge);
504 return;
505 }
506
507 // (var == value) <=> (var >= value) and (var <= value).
508 const Literal a(GetOrCreateAssociatedLiteral(ge));
509 const Literal b(GetOrCreateAssociatedLiteral(le));
510 sat_solver_->AddClauseDuringSearch({a, literal.Negated()});
511 sat_solver_->AddClauseDuringSearch({b, literal.Negated()});
512 sat_solver_->AddClauseDuringSearch({a.Negated(), b.Negated(), literal});
513
514 // Update reverse encoding.
515 const int new_size = 1 + literal.Index().value();
516 if (new_size > reverse_equality_encoding_.size()) {
517 reverse_equality_encoding_.resize(new_size);
518 }
519 reverse_equality_encoding_[literal].push_back({var, value});
520}

542// Note that we assume the input literal is canonicalized and do not fall into
543// a hole. Otherwise, this work but will likely return a literal before and
544// not one equivalent to it (which can be after!).
545LiteralIndex IntegerEncoder::SearchForLiteralAtOrBefore(
546 IntegerLiteral i_lit, IntegerValue* bound) const {
547 const PositiveOnlyIndex index = GetPositiveOnlyIndex(i_lit.var);
548 if (index >= encoding_by_var_.size()) return kNoLiteralIndex;
549 const auto& encoding = encoding_by_var_[index];
550 if (VariableIsPositive(i_lit.var)) {
551 // We need the entry at or before.
552 // We take the element before the upper_bound() which is either the encoding
553 // of i if it already exists, or the encoding just before it.
554 auto after_it = encoding.upper_bound(i_lit.bound);
555 if (after_it == encoding.begin()) return kNoLiteralIndex;
556 --after_it;
557 *bound = after_it->first;
558 return after_it->second.Index();
559 } else {
560 // We ask for who is implied by -var >= -bound, so we look for
561 // the var >= value with value > bound and take its negation.
562 auto after_it = encoding.upper_bound(-i_lit.bound);
563 if (after_it == encoding.end()) return kNoLiteralIndex;
564
565 // Compute tight bound if there are holes, we have X <= candidate.
566 const Domain& domain = domains_[index];
567 if (after_it->first <= domain.Min()) return kNoLiteralIndex;
568 *bound = -domain.ValueAtOrBefore(after_it->first.value() - 1);
569 return after_it->second.NegatedIndex();
570 }

573ABSL_MUST_USE_RESULT bool IntegerEncoder::LiteralOrNegationHasView(
574 Literal lit, IntegerVariable* view, bool* view_is_direct) const {
575 const IntegerVariable direct_var = GetLiteralView(lit);
576 const IntegerVariable opposite_var = GetLiteralView(lit.Negated());
577 // If a literal has both views, we want to always keep the same
578 // representative: the smallest IntegerVariable.
579 if (direct_var != kNoIntegerVariable &&
580 (opposite_var == kNoIntegerVariable || direct_var <= opposite_var)) {
581 if (view != nullptr) *view = direct_var;
582 if (view_is_direct != nullptr) *view_is_direct = true;
583 return true;
584 }
585 if (opposite_var != kNoIntegerVariable) {
586 if (view != nullptr) *view = opposite_var;
587 if (view_is_direct != nullptr) *view_is_direct = false;
588 return true;
589 }
590 return false;
591}

593std::vector<ValueLiteralPair> IntegerEncoder::PartialGreaterThanEncoding(
594 IntegerVariable var) const {
595 std::vector<ValueLiteralPair> result;
596 const PositiveOnlyIndex index = GetPositiveOnlyIndex(var);
597 if (index >= encoding_by_var_.size()) return result;
598 if (VariableIsPositive(var)) {
599 for (const auto [value, literal] : encoding_by_var_[index]) {
600 result.push_back({value, literal});
601 }
602 return result;
603 }
604
605 // Tricky: we need to account for holes.
606 const Domain& domain = domains_[index];
607 if (domain.IsEmpty()) return result;
608 int i = 0;
609 int64_t previous;
610 const int num_intervals = domain.NumIntervals();
611 for (const auto [value, literal] : encoding_by_var_[index]) {
612 while (value > domain[i].end) {
613 previous = domain[i].end;
614 ++i;
615 if (i == num_intervals) break;
616 }
617 if (i == num_intervals) break;
618 if (value <= domain[i].start) {
619 if (i == 0) continue;
620 result.push_back({-previous, literal.Negated()});
621 } else {
622 result.push_back({-value + 1, literal.Negated()});
623 }
624 }
625 std::reverse(result.begin(), result.end());
626 return result;
627}

629bool IntegerEncoder::UpdateEncodingOnInitialDomainChange(IntegerVariable var,
630 Domain domain) {
631 DCHECK(VariableIsPositive(var));
632 const PositiveOnlyIndex index = GetPositiveOnlyIndex(var);
633 if (index >= encoding_by_var_.size()) return true;
634
635 // Fix >= literal that can be fixed.
636 // We filter and canonicalize the encoding.
637 int i = 0;
638 int num_fixed = 0;
639 tmp_encoding_.clear();
640 for (const auto [value, literal] : encoding_by_var_[index]) {
641 while (i < domain.NumIntervals() && value > domain[i].end) ++i;
642 if (i == domain.NumIntervals()) {
643 // We are past the end, so always false.
644 if (trail_->Assignment().LiteralIsTrue(literal)) return false;
645 if (trail_->Assignment().LiteralIsFalse(literal)) continue;
646 ++num_fixed;
647 trail_->EnqueueWithUnitReason(literal.Negated());
648 continue;
649 }
650 if (i == 0 && value <= domain[0].start) {
651 // We are at or before the beginning, so always true.
652 if (trail_->Assignment().LiteralIsTrue(literal)) continue;
653 if (trail_->Assignment().LiteralIsFalse(literal)) return false;
654 ++num_fixed;
655 trail_->EnqueueWithUnitReason(literal);
656 continue;
657 }
658
659 // Note that we canonicalize the literal if it fall into a hole.
660 tmp_encoding_.push_back(
661 {std::max<IntegerValue>(value, domain[i].start), literal});
662 }
663 encoding_by_var_[index].clear();
664 for (const auto [value, literal] : tmp_encoding_) {
665 encoding_by_var_[index].insert({value, literal});
666 }
667
668 // Same for equality encoding.
669 // This will be lazily cleaned on the next PartialDomainEncoding() call.
670 i = 0;
671 for (const ValueLiteralPair pair : PartialDomainEncoding(var)) {
672 while (i < domain.NumIntervals() && pair.value > domain[i].end) ++i;
673 if (i == domain.NumIntervals() || pair.value < domain[i].start) {
674 if (trail_->Assignment().LiteralIsTrue(pair.literal)) return false;
675 if (trail_->Assignment().LiteralIsFalse(pair.literal)) continue;
676 ++num_fixed;
677 trail_->EnqueueWithUnitReason(pair.literal.Negated());
678 }
679 }
680
681 if (num_fixed > 0) {
682 VLOG(1) << "Domain intersection fixed " << num_fixed
683 << " encoding literals";
684 }
685
686 return true;
687}

696bool IntegerTrail::Propagate(Trail* trail) {
697 const int level = trail->CurrentDecisionLevel();
698 for (ReversibleInterface* rev : reversible_classes_) rev->SetLevel(level);
699
700 // Make sure that our internal "integer_search_levels_" size matches the
701 // sat decision levels. At the level zero, integer_search_levels_ should
702 // be empty.
703 if (level > integer_search_levels_.size()) {
704 integer_search_levels_.push_back(integer_trail_.size());
705 reason_decision_levels_.push_back(literals_reason_starts_.size());
706 CHECK_EQ(level, integer_search_levels_.size());
707 }
708
709 // This is required because when loading a model it is possible that we add
710 // (literal <-> integer literal) associations for literals that have already
711 // been propagated here. This often happens when the presolve is off
712 // and many variables are fixed.
713 //
714 // TODO(user): refactor the interaction IntegerTrail <-> IntegerEncoder so
715 // that we can just push right away such literal. Unfortunately, this is is
716 // a big chunk of work.
717 if (level == 0) {
718 for (const IntegerLiteral i_lit : delayed_to_fix_->integer_literal_to_fix) {
719 if (IsCurrentlyIgnored(i_lit.var)) continue;
720
721 // Note that we do not call Enqueue here but directly the update domain
722 // function so that we do not abort even if the level zero bounds were
723 // up to date.
724 const IntegerValue lb =
725 std::max(LevelZeroLowerBound(i_lit.var), i_lit.bound);
726 const IntegerValue ub = LevelZeroUpperBound(i_lit.var);
727 if (!UpdateInitialDomain(i_lit.var, Domain(lb.value(), ub.value()))) {
728 sat_solver_->NotifyThatModelIsUnsat();
729 return false;
730 }
731 }
732 delayed_to_fix_->integer_literal_to_fix.clear();
733
734 for (const Literal lit : delayed_to_fix_->literal_to_fix) {
735 if (trail_->Assignment().LiteralIsFalse(lit)) {
736 sat_solver_->NotifyThatModelIsUnsat();
737 return false;
738 }
739 if (trail_->Assignment().LiteralIsTrue(lit)) continue;
740 trail_->EnqueueWithUnitReason(lit);
741 }
742 delayed_to_fix_->literal_to_fix.clear();
743 }
744
745 // Process all the "associated" literals and Enqueue() the corresponding
746 // bounds.
747 while (propagation_trail_index_ < trail->Index()) {
748 const Literal literal = (*trail)[propagation_trail_index_++];
749 for (const IntegerLiteral i_lit : encoder_->GetIntegerLiterals(literal)) {
750 if (IsCurrentlyIgnored(i_lit.var)) continue;
751
752 // The reason is simply the associated literal.
753 if (!EnqueueAssociatedIntegerLiteral(i_lit, literal)) {
754 return false;
755 }
756 }
757 }
758
759 return true;
760}

762void IntegerTrail::Untrail(const Trail& trail, int literal_trail_index) {
763 ++num_untrails_;
764 conditional_lbs_.clear();
765 const int level = trail.CurrentDecisionLevel();
766 propagation_trail_index_ =
767 std::min(propagation_trail_index_, literal_trail_index);
768
769 if (level < first_level_without_full_propagation_) {
770 first_level_without_full_propagation_ = -1;
771 }
772
773 // Note that if a conflict was detected before Propagate() of this class was
774 // even called, it is possible that there is nothing to backtrack.
775 if (level >= integer_search_levels_.size()) return;
776 const int target = integer_search_levels_[level];
777 integer_search_levels_.resize(level);
778 CHECK_GE(target, vars_.size());
779 CHECK_LE(target, integer_trail_.size());
780
781 for (int index = integer_trail_.size() - 1; index >= target; --index) {
782 const TrailEntry& entry = integer_trail_[index];
783 if (entry.var < 0) continue; // entry used by EnqueueLiteral().
784 vars_[entry.var].current_trail_index = entry.prev_trail_index;
785 vars_[entry.var].current_bound =
786 integer_trail_[entry.prev_trail_index].bound;
787 }
788 integer_trail_.resize(target);
789
790 // Clear reason.
791 const int old_size = reason_decision_levels_[level];
792 reason_decision_levels_.resize(level);
793 if (old_size < literals_reason_starts_.size()) {
794 literals_reason_buffer_.resize(literals_reason_starts_[old_size]);
795
796 const int bound_start = bounds_reason_starts_[old_size];
797 bounds_reason_buffer_.resize(bound_start);
798 if (bound_start < trail_index_reason_buffer_.size()) {
799 trail_index_reason_buffer_.resize(bound_start);
800 }
801
802 literals_reason_starts_.resize(old_size);
803 bounds_reason_starts_.resize(old_size);
804 }
805
806 // We notify the new level once all variables have been restored to their
807 // old value.
808 for (ReversibleInterface* rev : reversible_classes_) rev->SetLevel(level);
809}

873// Note that we don't support optional variable here. Or at least if you set
874// the domain of an optional variable to zero, the problem will be declared
875// unsat.
876bool IntegerTrail::UpdateInitialDomain(IntegerVariable var, Domain domain) {
877 CHECK_EQ(trail_->CurrentDecisionLevel(), 0);
878 if (!VariableIsPositive(var)) {
879 var = NegationOf(var);
880 domain = domain.Negation();
881 }
882
883 const PositiveOnlyIndex index = GetPositiveOnlyIndex(var);
884 const Domain& old_domain = (*domains_)[index];
885 domain = domain.IntersectionWith(old_domain);
886 if (old_domain == domain) return true;
887
888 if (domain.IsEmpty()) return false;
889 const bool lb_changed = domain.Min() > old_domain.Min();
890 const bool ub_changed = domain.Max() < old_domain.Max();
891 (*domains_)[index] = domain;
892
893 // Update directly the level zero bounds.
894 DCHECK(
895 ReasonIsValid(IntegerLiteral::LowerOrEqual(var, domain.Max()), {}, {}));
896 DCHECK(
897 ReasonIsValid(IntegerLiteral::GreaterOrEqual(var, domain.Min()), {}, {}));
898 DCHECK_GE(domain.Min(), LowerBound(var));
899 DCHECK_LE(domain.Max(), UpperBound(var));
900 vars_[var].current_bound = domain.Min();
901 integer_trail_[var.value()].bound = domain.Min();
902 vars_[NegationOf(var)].current_bound = -domain.Max();
903 integer_trail_[NegationOf(var).value()].bound = -domain.Max();
904
905 // Do not forget to update the watchers.
906 for (SparseBitset<IntegerVariable>* bitset : watchers_) {
907 if (lb_changed) bitset->Set(var);
908 if (ub_changed) bitset->Set(NegationOf(var));
909 }
910
911 // Update the encoding.
912 return encoder_->UpdateEncodingOnInitialDomainChange(var, domain);
913}

936int IntegerTrail::FindTrailIndexOfVarBefore(IntegerVariable var,
937 int threshold) const {
938 // Optimization. We assume this is only called when computing a reason, so we
939 // can ignore this trail_index if we already need a more restrictive reason
940 // for this var.
941 const int index_in_queue = tmp_var_to_trail_index_in_queue_[var];
942 if (threshold <= index_in_queue) {
943 if (index_in_queue != std::numeric_limits<int32_t>::max())
944 has_dependency_ = true;
945 return -1;
946 }
947
948 DCHECK_GE(threshold, vars_.size());
949 int trail_index = vars_[var].current_trail_index;
950
951 // Check the validity of the cached index and use it if possible.
952 if (trail_index > threshold) {
953 const int cached_index = var_trail_index_cache_[var];
954 if (cached_index >= threshold && cached_index < trail_index &&
955 integer_trail_[cached_index].var == var) {
956 trail_index = cached_index;
957 }
958 }
959
960 while (trail_index >= threshold) {
961 trail_index = integer_trail_[trail_index].prev_trail_index;
962 if (trail_index >= var_trail_index_cache_threshold_) {
963 var_trail_index_cache_[var] = trail_index;
964 }
965 }
966
967 const int num_vars = vars_.size();
968 return trail_index < num_vars ? -1 : trail_index;
969}

971int IntegerTrail::FindLowestTrailIndexThatExplainBound(
972 IntegerLiteral i_lit) const {
973 DCHECK_LE(i_lit.bound, vars_[i_lit.var].current_bound);
974 if (i_lit.bound <= LevelZeroLowerBound(i_lit.var)) return -1;
975 int trail_index = vars_[i_lit.var].current_trail_index;
976
977 // Check the validity of the cached index and use it if possible. This caching
978 // mechanism is important in case of long chain of propagation on the same
979 // variable. Because during conflict resolution, we call
980 // FindLowestTrailIndexThatExplainBound() with lowest and lowest bound, this
981 // cache can transform a quadratic complexity into a linear one.
982 {
983 const int cached_index = var_trail_index_cache_[i_lit.var];
984 if (cached_index < trail_index) {
985 const TrailEntry& entry = integer_trail_[cached_index];
986 if (entry.var == i_lit.var && entry.bound >= i_lit.bound) {
987 trail_index = cached_index;
988 }
989 }
990 }
991
992 int prev_trail_index = trail_index;
993 while (true) {
994 if (trail_index >= var_trail_index_cache_threshold_) {
995 var_trail_index_cache_[i_lit.var] = trail_index;
996 }
997 const TrailEntry& entry = integer_trail_[trail_index];
998 if (entry.bound == i_lit.bound) return trail_index;
999 if (entry.bound < i_lit.bound) return prev_trail_index;
1000 prev_trail_index = trail_index;
1001 trail_index = entry.prev_trail_index;
1002 }
1003}

1043void IntegerTrail::RelaxLinearReason(IntegerValue slack,
1044 absl::Span<const IntegerValue> coeffs,
1045 std::vector<int>* trail_indices) const {
1046 DCHECK_GT(slack, 0);
1047 DCHECK(relax_heap_.empty());
1048
1049 // We start by filtering *trail_indices:
1050 // - remove all level zero entries.
1051 // - keep the one that cannot be relaxed.
1052 // - move the other one to the relax_heap_ (and creating the heap).
1053 int new_size = 0;
1054 const int size = coeffs.size();
1055 const int num_vars = vars_.size();
1056 for (int i = 0; i < size; ++i) {
1057 const int index = (*trail_indices)[i];
1058
1059 // We ignore level zero entries.
1060 if (index < num_vars) continue;
1061
1062 // If the coeff is too large, we cannot relax this entry.
1063 const IntegerValue coeff = coeffs[i];
1064 if (coeff > slack) {
1065 (*trail_indices)[new_size++] = index;
1066 continue;
1067 }
1068
1069 // This is a bit hacky, but when it is used from MergeReasonIntoInternal(),
1070 // we never relax a reason that will not be expanded because it is already
1071 // part of the current conflict.
1072 const TrailEntry& entry = integer_trail_[index];
1073 if (entry.var != kNoIntegerVariable &&
1074 index <= tmp_var_to_trail_index_in_queue_[entry.var]) {
1075 (*trail_indices)[new_size++] = index;
1076 continue;
1077 }
1078
1079 // Note that both terms of the product are positive.
1080 const TrailEntry& previous_entry = integer_trail_[entry.prev_trail_index];
1081 const int64_t diff =
1082 CapProd(coeff.value(), (entry.bound - previous_entry.bound).value());
1083 if (diff > slack) {
1084 (*trail_indices)[new_size++] = index;
1085 continue;
1086 }
1087
1088 relax_heap_.push_back({index, coeff, diff});
1089 }
1090 trail_indices->resize(new_size);
1091 std::make_heap(relax_heap_.begin(), relax_heap_.end());
1092
1093 while (slack > 0 && !relax_heap_.empty()) {
1094 const RelaxHeapEntry heap_entry = relax_heap_.front();
1095 std::pop_heap(relax_heap_.begin(), relax_heap_.end());
1096 relax_heap_.pop_back();
1097
1098 // The slack might have changed since the entry was added.
1099 if (heap_entry.diff > slack) {
1100 trail_indices->push_back(heap_entry.index);
1101 continue;
1102 }
1103
1104 // Relax, and decide what to do with the new value of index.
1105 slack -= heap_entry.diff;
1106 const int index = integer_trail_[heap_entry.index].prev_trail_index;
1107
1108 // Same code as in the first block.
1109 if (index < num_vars) continue;
1110 if (heap_entry.coeff > slack) {
1111 trail_indices->push_back(index);
1112 continue;
1113 }
1114 const TrailEntry& entry = integer_trail_[index];
1115 if (entry.var != kNoIntegerVariable &&
1116 index <= tmp_var_to_trail_index_in_queue_[entry.var]) {
1117 trail_indices->push_back(index);
1118 continue;
1119 }
1120
1121 const TrailEntry& previous_entry = integer_trail_[entry.prev_trail_index];
1122 const int64_t diff = CapProd(heap_entry.coeff.value(),
1123 (entry.bound - previous_entry.bound).value());
1124 if (diff > slack) {
1125 trail_indices->push_back(index);
1126 continue;
1127 }
1128 relax_heap_.push_back({index, heap_entry.coeff, diff});
1129 std::push_heap(relax_heap_.begin(), relax_heap_.end());
1130 }
1131
1132 // If we aborted early because of the slack, we need to push all remaining
1133 // indices back into the reason.
1134 for (const RelaxHeapEntry& entry : relax_heap_) {
1135 trail_indices->push_back(entry.index);
1136 }
1137 relax_heap_.clear();
1138}

1167 }

1257bool IntegerTrail::ConditionalEnqueue(
1258 Literal lit, IntegerLiteral i_lit, std::vector<Literal>* literal_reason,
1259 std::vector<IntegerLiteral>* integer_reason) {
1260 const VariablesAssignment& assignment = trail_->Assignment();
1261 if (assignment.LiteralIsFalse(lit)) return true;
1262
1263 // We can always push var if the optional literal is the same.
1264 //
1265 // TODO(user): we can also push lit.var if its presence implies lit.
1266 if (lit.Index() == OptionalLiteralIndex(i_lit.var)) {
1267 return Enqueue(i_lit, *literal_reason, *integer_reason);
1268 }
1269
1270 if (assignment.LiteralIsTrue(lit)) {
1271 literal_reason->push_back(lit.Negated());
1272 return Enqueue(i_lit, *literal_reason, *integer_reason);
1273 }
1274
1275 if (IntegerLiteralIsFalse(i_lit)) {
1276 integer_reason->push_back(
1277 IntegerLiteral::LowerOrEqual(i_lit.var, i_lit.bound - 1));
1278 EnqueueLiteral(lit.Negated(), *literal_reason, *integer_reason);
1279 return true;
1280 }
1281
1282 // We can't push anything in this case.
1283 //
1284 // We record it for this propagation phase (until the next untrail) as this
1285 // is relatively fast and heuristics can exploit this.
1286 //
1287 // Note that currently we only use ConditionalEnqueue() in scheduling
1288 // propagator, and these propagator are quite slow so this is not visible.
1289 //
1290 // TODO(user): We could even keep the reason and maybe do some reasoning using
1291 // at_least_one constraint on a set of the Boolean used here.
1292 const auto [it, inserted] =
1293 conditional_lbs_.insert({{lit.Index(), i_lit.var}, i_lit.bound});
1294 if (!inserted) {
1295 it->second = std::max(it->second, i_lit.bound);
1296 }
1297
1298 return true;
1299}

1314bool IntegerTrail::ReasonIsValid(
1315 absl::Span<const Literal> literal_reason,
1316 absl::Span<const IntegerLiteral> integer_reason) {
1317 const VariablesAssignment& assignment = trail_->Assignment();
1318 for (const Literal lit : literal_reason) {
1319 if (!assignment.LiteralIsFalse(lit)) return false;
1320 }
1321 for (const IntegerLiteral i_lit : integer_reason) {
1322 if (i_lit.IsAlwaysTrue()) continue;
1323 if (i_lit.IsAlwaysFalse()) {
1324 LOG(INFO) << "Reason has a constant false literal!";
1325 return false;
1326 }
1327 if (i_lit.bound > vars_[i_lit.var].current_bound) {
1328 if (IsOptional(i_lit.var)) {
1329 const Literal is_ignored = IsIgnoredLiteral(i_lit.var);
1330 LOG(INFO) << "Reason " << i_lit << " is not true!"
1331 << " optional variable:" << i_lit.var
1332 << " present:" << assignment.LiteralIsFalse(is_ignored)
1333 << " absent:" << assignment.LiteralIsTrue(is_ignored)
1334 << " current_lb:" << vars_[i_lit.var].current_bound;
1335 } else {
1336 LOG(INFO) << "Reason " << i_lit << " is not true!"
1337 << " non-optional variable:" << i_lit.var
1338 << " current_lb:" << vars_[i_lit.var].current_bound;
1339 }
1340 return false;
1341 }
1342 }
1343
1344 // This may not indicate an incorectness, but just some propagators that
1345 // didn't reach a fixed-point at level zero.
1346 if (!integer_search_levels_.empty()) {
1347 int num_literal_assigned_after_root_node = 0;
1348 for (const Literal lit : literal_reason) {
1349 if (trail_->Info(lit.Variable()).level > 0) {
1350 num_literal_assigned_after_root_node++;
1351 }
1352 }
1353 for (const IntegerLiteral i_lit : integer_reason) {
1354 if (i_lit.IsAlwaysTrue()) continue;
1355 if (LevelZeroLowerBound(i_lit.var) < i_lit.bound) {
1356 num_literal_assigned_after_root_node++;
1357 }
1358 }
1359 if (num_literal_assigned_after_root_node == 0) {
1360 VLOG(2) << "Propagating a literal with no reason at a positive level!\n"
1361 << "level:" << integer_search_levels_.size() << " "
1362 << ReasonDebugString(literal_reason, integer_reason) << "\n"
1363 << DebugString();
1364 }
1365 }
1366
1367 return true;
1368}

1415void IntegerTrail::EnqueueLiteralInternal(
1416 Literal literal, LazyReasonFunction lazy_reason,
1417 absl::Span<const Literal> literal_reason,
1418 absl::Span<const IntegerLiteral> integer_reason) {
1419 DCHECK(!trail_->Assignment().LiteralIsAssigned(literal));
1420 DCHECK(lazy_reason != nullptr ||
1421 ReasonIsValid(literal, literal_reason, integer_reason));
1422 if (integer_search_levels_.empty()) {
1423 // Level zero. We don't keep any reason.
1424 trail_->EnqueueWithUnitReason(literal);
1425 return;
1426 }
1427
1428 // If we are fixing something at a positive level, remember it.
1429 if (!integer_search_levels_.empty() && integer_reason.empty() &&
1430 literal_reason.empty() && lazy_reason == nullptr) {
1431 delayed_to_fix_->literal_to_fix.push_back(literal);
1432 }
1433
1434 const int trail_index = trail_->Index();
1435 if (trail_index >= boolean_trail_index_to_integer_one_.size()) {
1436 boolean_trail_index_to_integer_one_.resize(trail_index + 1);
1437 }
1438 boolean_trail_index_to_integer_one_[trail_index] = integer_trail_.size();
1439
1440 int reason_index = literals_reason_starts_.size();
1441 if (lazy_reason != nullptr) {
1442 if (integer_trail_.size() >= lazy_reasons_.size()) {
1443 lazy_reasons_.resize(integer_trail_.size() + 1, nullptr);
1444 }
1445 lazy_reasons_[integer_trail_.size()] = lazy_reason;
1446 reason_index = -1;
1447 } else {
1448 // Copy the reason.
1449 literals_reason_starts_.push_back(literals_reason_buffer_.size());
1450 literals_reason_buffer_.insert(literals_reason_buffer_.end(),
1451 literal_reason.begin(),
1452 literal_reason.end());
1453 bounds_reason_starts_.push_back(bounds_reason_buffer_.size());
1454 bounds_reason_buffer_.insert(bounds_reason_buffer_.end(),
1455 integer_reason.begin(), integer_reason.end());
1456 }
1457
1458 integer_trail_.push_back({/*bound=*/IntegerValue(0),
1459 /*var=*/kNoIntegerVariable,
1460 /*prev_trail_index=*/-1,
1461 /*reason_index=*/reason_index});
1462
1463 trail_->Enqueue(literal, propagator_id_);
1464}

1485// We try to select a variable with a large domain that was propagated a lot
1486// already.
1487IntegerVariable IntegerTrail::NextVariableToBranchOnInPropagationLoop() const {
1488 CHECK(InPropagationLoop());
1489 ++num_decisions_to_break_loop_;
1490 std::vector<IntegerVariable> vars;
1491 for (int i = integer_search_levels_.back(); i < integer_trail_.size(); ++i) {
1492 const IntegerVariable var = integer_trail_[i].var;
1493 if (var == kNoIntegerVariable) continue;
1494 if (UpperBound(var) - LowerBound(var) <= 100) continue;
1495 vars.push_back(var);
1496 }
1497 if (vars.empty()) return kNoIntegerVariable;
1498 std::sort(vars.begin(), vars.end());
1499 IntegerVariable best_var = vars[0];
1500 int best_count = 1;
1501 int count = 1;
1502 for (int i = 1; i < vars.size(); ++i) {
1503 if (vars[i] != vars[i - 1]) {
1504 count = 1;
1505 } else {
1506 ++count;
1507 if (count > best_count) {
1508 best_count = count;
1509 best_var = vars[i];
1510 }
1511 }
1512 }
1513 return best_var;
1514}

1539bool IntegerTrail::EnqueueInternal(
1540 IntegerLiteral i_lit, LazyReasonFunction lazy_reason,
1541 absl::Span<const Literal> literal_reason,
1542 absl::Span<const IntegerLiteral> integer_reason,
1543 int trail_index_with_same_reason) {
1544 DCHECK(lazy_reason != nullptr ||
1545 ReasonIsValid(i_lit, literal_reason, integer_reason));
1546 const IntegerVariable var(i_lit.var);
1547
1548 // No point doing work if the variable is already ignored.
1549 if (IsCurrentlyIgnored(var)) return true;
1550
1551 // Nothing to do if the bound is not better than the current one.
1552 // TODO(user): Change this to a CHECK? propagator shouldn't try to push such
1553 // bound and waste time explaining it.
1554 if (i_lit.bound <= vars_[var].current_bound) return true;
1555 ++num_enqueues_;
1556
1557 // If the domain of var is not a single intervals and i_lit.bound fall into a
1558 // "hole", we increase it to the next possible value. This ensure that we
1559 // never Enqueue() non-canonical literals. See also Canonicalize().
1560 //
1561 // Note: The literals in the reason are not necessarily canonical, but then
1562 // we always map these to enqueued literals during conflict resolution.
1563 CanonicalizeLiteralIfNeeded(&i_lit);
1564
1565 // Check if the integer variable has an empty domain.
1566 if (i_lit.bound > UpperBound(var)) {
1567 // We relax the upper bound as much as possible to still have a conflict.
1568 const auto ub_reason = IntegerLiteral::LowerOrEqual(var, i_lit.bound - 1);
1569
1570 if (!IsOptional(var) || trail_->Assignment().LiteralIsFalse(
1571 Literal(is_ignored_literals_[var]))) {
1572 // Note that we want only one call to MergeReasonIntoInternal() for
1573 // efficiency and a potential smaller reason.
1574 auto* conflict = InitializeConflict(i_lit, lazy_reason, literal_reason,
1575 integer_reason);
1576 if (IsOptional(var)) {
1577 conflict->push_back(Literal(is_ignored_literals_[var]));
1578 }
1579 {
1580 const int trail_index = FindLowestTrailIndexThatExplainBound(ub_reason);
1581 const int num_vars = vars_.size(); // must be signed.
1582 if (trail_index >= num_vars) tmp_queue_.push_back(trail_index);
1583 }
1584 MergeReasonIntoInternal(conflict);
1585 return false;
1586 } else {
1587 // Note(user): We never make the bound of an optional literal cross. We
1588 // used to have a bug where we propagated these bounds and their
1589 // associated literals, and we were reaching a conflict while propagating
1590 // the associated literal instead of setting is_ignored below to false.
1591 const Literal is_ignored = Literal(is_ignored_literals_[var]);
1592 if (integer_search_levels_.empty()) {
1593 trail_->EnqueueWithUnitReason(is_ignored);
1594 } else {
1595 // Here we currently expand any lazy reason because we need to add
1596 // to it the reason for the upper bound.
1597 // TODO(user): A possible solution would be to support the two types
1598 // of reason (lazy and not) at the same time and use the union of both?
1599 if (lazy_reason != nullptr) {
1600 lazy_reason(i_lit, integer_trail_.size(), &lazy_reason_literals_,
1601 &lazy_reason_trail_indices_);
1602 std::vector<IntegerLiteral> temp;
1603 for (const int trail_index : lazy_reason_trail_indices_) {
1604 const TrailEntry& entry = integer_trail_[trail_index];
1605 temp.push_back(IntegerLiteral(entry.var, entry.bound));
1606 }
1607 EnqueueLiteral(is_ignored, lazy_reason_literals_, temp);
1608 } else {
1609 EnqueueLiteral(is_ignored, literal_reason, integer_reason);
1610 }
1611
1612 // Hack, we add the upper bound reason here.
1613 bounds_reason_buffer_.push_back(ub_reason);
1614 }
1615 return true;
1616 }
1617 }
1618
1619 // Stop propagating if we detect a propagation loop. The search heuristic will
1620 // then take an appropriate next decision. Note that we do that after checking
1621 // for a potential conflict if the two bounds of a variable cross. This is
1622 // important, so that in the corner case where all variables are actually
1623 // fixed, we still make sure no propagator detect a conflict.
1624 //
1625 // TODO(user): Some propagation code have CHECKS in place and not like when
1626 // something they just pushed is not reflected right away. They must be aware
1627 // of that, which is a bit tricky.
1628 if (InPropagationLoop()) {
1629 // Note that we still propagate "big" push as it seems better to do that
1630 // now rather than to delay to the next decision.
1631 const IntegerValue lb = LowerBound(i_lit.var);
1632 const IntegerValue ub = UpperBound(i_lit.var);
1633 if (i_lit.bound - lb < (ub - lb) / 2) {
1634 if (first_level_without_full_propagation_ == -1) {
1635 first_level_without_full_propagation_ = trail_->CurrentDecisionLevel();
1636 }
1637 return true;
1638 }
1639 }
1640
1641 // Notify the watchers.
1642 for (SparseBitset<IntegerVariable>* bitset : watchers_) {
1643 bitset->Set(i_lit.var);
1644 }
1645
1646 // Enqueue the strongest associated Boolean literal implied by this one.
1647 // Because we linked all such literal with implications, all the one before
1648 // will be propagated by the SAT solver.
1649 //
1650 // Important: It is possible that such literal or even stronger ones are
1651 // already true! This is because we might push stuff while Propagate() haven't
1652 // been called yet. Maybe we should call it?
1653 //
1654 // TODO(user): It might be simply better and more efficient to simply enqueue
1655 // all of them here. We have also more liberty to choose the explanation we
1656 // want. A drawback might be that the implications might not be used in the
1657 // binary conflict minimization algo.
1658 IntegerValue bound;
1659 const LiteralIndex literal_index =
1660 encoder_->SearchForLiteralAtOrBefore(i_lit, &bound);
1661 if (literal_index != kNoLiteralIndex) {
1662 const Literal to_enqueue = Literal(literal_index);
1663 if (trail_->Assignment().LiteralIsFalse(to_enqueue)) {
1664 auto* conflict = InitializeConflict(i_lit, lazy_reason, literal_reason,
1665 integer_reason);
1666 conflict->push_back(to_enqueue);
1667 MergeReasonIntoInternal(conflict);
1668 return false;
1669 }
1670
1671 // If the associated literal exactly correspond to i_lit, then we push
1672 // it first, and then we use it as a reason for i_lit. We do that so that
1673 // MergeReasonIntoInternal() will not unecessarily expand further the reason
1674 // for i_lit.
1675 if (bound >= i_lit.bound) {
1676 DCHECK_EQ(bound, i_lit.bound);
1677 if (!trail_->Assignment().LiteralIsTrue(to_enqueue)) {
1678 EnqueueLiteralInternal(to_enqueue, lazy_reason, literal_reason,
1679 integer_reason);
1680 }
1681 return EnqueueAssociatedIntegerLiteral(i_lit, to_enqueue);
1682 }
1683
1684 if (!trail_->Assignment().LiteralIsTrue(to_enqueue)) {
1685 if (integer_search_levels_.empty()) {
1686 trail_->EnqueueWithUnitReason(to_enqueue);
1687 } else {
1688 // Subtle: the reason is the same as i_lit, that we will enqueue if no
1689 // conflict occur at position integer_trail_.size(), so we just refer to
1690 // this index here.
1691 const int trail_index = trail_->Index();
1692 if (trail_index >= boolean_trail_index_to_integer_one_.size()) {
1693 boolean_trail_index_to_integer_one_.resize(trail_index + 1);
1694 }
1695 boolean_trail_index_to_integer_one_[trail_index] =
1696 trail_index_with_same_reason;
1697 trail_->Enqueue(to_enqueue, propagator_id_);
1698 }
1699 }
1700 }
1701
1702 // Special case for level zero.
1703 if (integer_search_levels_.empty()) {
1704 ++num_level_zero_enqueues_;
1705 vars_[i_lit.var].current_bound = i_lit.bound;
1706 integer_trail_[i_lit.var.value()].bound = i_lit.bound;
1707
1708 // We also update the initial domain. If this fail, since we are at level
1709 // zero, we don't care about the reason.
1710 trail_->MutableConflict()->clear();
1711 return UpdateInitialDomain(
1712 i_lit.var,
1713 Domain(LowerBound(i_lit.var).value(), UpperBound(i_lit.var).value()));
1714 }
1715 DCHECK_GT(trail_->CurrentDecisionLevel(), 0);
1716
1717 // If we are not at level zero but there is not reason, we have a root level
1718 // deduction. Remember it so that we don't forget on the next restart.
1719 if (!integer_search_levels_.empty() && integer_reason.empty() &&
1720 literal_reason.empty() && lazy_reason == nullptr &&
1721 trail_index_with_same_reason >= integer_trail_.size()) {
1722 if (!RootLevelEnqueue(i_lit)) return false;
1723 }
1724
1725 int reason_index = literals_reason_starts_.size();
1726 if (lazy_reason != nullptr) {
1727 if (integer_trail_.size() >= lazy_reasons_.size()) {
1728 lazy_reasons_.resize(integer_trail_.size() + 1, nullptr);
1729 }
1730 lazy_reasons_[integer_trail_.size()] = lazy_reason;
1731 reason_index = -1;
1732 } else if (trail_index_with_same_reason >= integer_trail_.size()) {
1733 // Save the reason into our internal buffers.
1734 literals_reason_starts_.push_back(literals_reason_buffer_.size());
1735 if (!literal_reason.empty()) {
1736 literals_reason_buffer_.insert(literals_reason_buffer_.end(),
1737 literal_reason.begin(),
1738 literal_reason.end());
1739 }
1740 bounds_reason_starts_.push_back(bounds_reason_buffer_.size());
1741 if (!integer_reason.empty()) {
1742 bounds_reason_buffer_.insert(bounds_reason_buffer_.end(),
1743 integer_reason.begin(),
1744 integer_reason.end());
1745 }
1746 } else {
1747 reason_index = integer_trail_[trail_index_with_same_reason].reason_index;
1748 }
1749
1750 const int prev_trail_index = vars_[i_lit.var].current_trail_index;
1751 integer_trail_.push_back({/*bound=*/i_lit.bound,
1752 /*var=*/i_lit.var,
1753 /*prev_trail_index=*/prev_trail_index,
1754 /*reason_index=*/reason_index});
1755
1756 vars_[i_lit.var].current_bound = i_lit.bound;
1757 vars_[i_lit.var].current_trail_index = integer_trail_.size() - 1;
1758 return true;
1759}
1760
1761bool IntegerTrail::EnqueueAssociatedIntegerLiteral(IntegerLiteral i_lit,
1762 Literal literal_reason) {
1763 DCHECK(ReasonIsValid(i_lit, {literal_reason.Negated()}, {}));
1764 DCHECK(!IsCurrentlyIgnored(i_lit.var));
1765
1766 // Nothing to do if the bound is not better than the current one.
1767 if (i_lit.bound <= vars_[i_lit.var].current_bound) return true;
1768 ++num_enqueues_;
1769
1770 // Make sure we do not fall into a hole.
1771 CanonicalizeLiteralIfNeeded(&i_lit);
1772
1773 // Check if the integer variable has an empty domain. Note that this should
1774 // happen really rarely since in most situation, pushing the upper bound would
1775 // have resulted in this literal beeing false. Because of this we revert to
1776 // the "generic" Enqueue() to avoid some code duplication.
1777 if (i_lit.bound > UpperBound(i_lit.var)) {
1778 return Enqueue(i_lit, {literal_reason.Negated()}, {});
1779 }
1780
1781 // Notify the watchers.
1782 for (SparseBitset<IntegerVariable>* bitset : watchers_) {
1783 bitset->Set(i_lit.var);
1784 }
1785
1786 // Special case for level zero.
1787 if (integer_search_levels_.empty()) {
1788 vars_[i_lit.var].current_bound = i_lit.bound;
1789 integer_trail_[i_lit.var.value()].bound = i_lit.bound;
1790
1791 // We also update the initial domain. If this fail, since we are at level
1792 // zero, we don't care about the reason.
1793 trail_->MutableConflict()->clear();
1794 return UpdateInitialDomain(
1795 i_lit.var,
1796 Domain(LowerBound(i_lit.var).value(), UpperBound(i_lit.var).value()));
1797 }
1798 DCHECK_GT(trail_->CurrentDecisionLevel(), 0);
1799
1800 const int reason_index = literals_reason_starts_.size();
1801 CHECK_EQ(reason_index, bounds_reason_starts_.size());
1802 literals_reason_starts_.push_back(literals_reason_buffer_.size());
1803 bounds_reason_starts_.push_back(bounds_reason_buffer_.size());
1804 literals_reason_buffer_.push_back(literal_reason.Negated());
1805
1806 const int prev_trail_index = vars_[i_lit.var].current_trail_index;
1807 integer_trail_.push_back({/*bound=*/i_lit.bound,
1808 /*var=*/i_lit.var,
1809 /*prev_trail_index=*/prev_trail_index,
1810 /*reason_index=*/reason_index});
1811
1812 vars_[i_lit.var].current_bound = i_lit.bound;
1813 vars_[i_lit.var].current_trail_index = integer_trail_.size() - 1;
1814 return true;
1815}

1833 const int start = bounds_reason_starts_[reason_index];
1834 const int end = reason_index + 1 < bounds_reason_starts_.size()
1835 ? bounds_reason_starts_[reason_index + 1]
1836 : bounds_reason_buffer_.size();
1837 if (start == end) return {};
1838
1839 // Cache the result if not already computed. Remark, if the result was never
1840 // computed then the span trail_index_reason_buffer_[start, end) will either
1841 // be non-existent or full of -1.
1842 //
1843 // TODO(user): For empty reason, we will always recompute them.
1844 if (end > trail_index_reason_buffer_.size()) {
1845 trail_index_reason_buffer_.resize(end, -1);
1846 }
1847 if (trail_index_reason_buffer_[start] == -1) {
1848 int new_end = start;
1849 const int num_vars = vars_.size();
1850 for (int i = start; i < end; ++i) {
1851 const int dep =
1852 FindLowestTrailIndexThatExplainBound(bounds_reason_buffer_[i]);
1853 if (dep >= num_vars) {
1854 trail_index_reason_buffer_[new_end++] = dep;
1855 }
1856 }
1857 return absl::Span<const int>(&trail_index_reason_buffer_[start],
1858 new_end - start);
1859 } else {
1860 // TODO(user): We didn't store new_end in a previous call, so end might be
1861 // larger. That is a bit annoying since we have to test for -1 while
1862 // iterating.
1863 return absl::Span<const int>(&trail_index_reason_buffer_[start],
1864 end - start);
1865 }
1866}

1901void IntegerTrail::MergeReasonInto(absl::Span<const IntegerLiteral> literals,
1902 std::vector<Literal>* output) const {
1903 DCHECK(tmp_queue_.empty());
1904 const int num_vars = vars_.size();
1905 for (const IntegerLiteral& literal : literals) {
1906 if (literal.IsAlwaysTrue()) continue;
1907 const int trail_index = FindLowestTrailIndexThatExplainBound(literal);
1908
1909 // Any indices lower than that means that there is no reason needed.
1910 // Note that it is important for size to be signed because of -1 indices.
1911 if (trail_index >= num_vars) tmp_queue_.push_back(trail_index);
1912 }
1913 return MergeReasonIntoInternal(output);

1916// This will expand the reason of the IntegerLiteral already in tmp_queue_ until
1917// everything is explained in term of Literal.
1918void IntegerTrail::MergeReasonIntoInternal(std::vector<Literal>* output) const {
1919 // All relevant trail indices will be >= vars_.size(), so we can safely use
1920 // zero to means that no literal referring to this variable is in the queue.
1921 DCHECK(std::all_of(tmp_var_to_trail_index_in_queue_.begin(),
1922 tmp_var_to_trail_index_in_queue_.end(),
1923 [](int v) { return v == 0; }));
1924
1925 added_variables_.ClearAndResize(BooleanVariable(trail_->NumVariables()));
1926 for (const Literal l : *output) {
1927 added_variables_.Set(l.Variable());
1928 }
1929
1930 // During the algorithm execution, all the queue entries that do not match the
1931 // content of tmp_var_to_trail_index_in_queue_[] will be ignored.
1932 for (const int trail_index : tmp_queue_) {
1933 DCHECK_GE(trail_index, vars_.size());
1934 DCHECK_LT(trail_index, integer_trail_.size());
1935 const TrailEntry& entry = integer_trail_[trail_index];
1936 tmp_var_to_trail_index_in_queue_[entry.var] =
1937 std::max(tmp_var_to_trail_index_in_queue_[entry.var], trail_index);
1938 }
1939
1940 // We manage our heap by hand so that we can range iterate over it above, and
1941 // this initial heapify is faster.
1942 std::make_heap(tmp_queue_.begin(), tmp_queue_.end());
1943
1944 // We process the entries by highest trail_index first. The content of the
1945 // queue will always be a valid reason for the literals we already added to
1946 // the output.
1947 tmp_to_clear_.clear();
1948 while (!tmp_queue_.empty()) {
1949 const int trail_index = tmp_queue_.front();
1950 const TrailEntry& entry = integer_trail_[trail_index];
1951 std::pop_heap(tmp_queue_.begin(), tmp_queue_.end());
1952 tmp_queue_.pop_back();
1953
1954 // Skip any stale queue entry. Amongst all the entry referring to a given
1955 // variable, only the latest added to the queue is valid and we detect it
1956 // using its trail index.
1957 if (tmp_var_to_trail_index_in_queue_[entry.var] != trail_index) {
1958 continue;
1959 }
1960
1961 // Set the cache threshold. Since we process trail indices in decreasing
1962 // order and we only have single linked list, we only want to advance the
1963 // "cache" up to this threshold.
1964 var_trail_index_cache_threshold_ = trail_index;
1965
1966 // If this entry has an associated literal, then it should always be the
1967 // one we used for the reason. This code DCHECK that.
1968 if (DEBUG_MODE) {
1969 const LiteralIndex associated_lit =
1970 encoder_->GetAssociatedLiteral(IntegerLiteral::GreaterOrEqual(
1971 IntegerVariable(entry.var), entry.bound));
1972 if (associated_lit != kNoLiteralIndex) {
1973 // We check that the reason is the same!
1974 const int reason_index = integer_trail_[trail_index].reason_index;
1975 CHECK_NE(reason_index, -1);
1976 {
1977 const int start = literals_reason_starts_[reason_index];
1978 const int end = reason_index + 1 < literals_reason_starts_.size()
1979 ? literals_reason_starts_[reason_index + 1]
1980 : literals_reason_buffer_.size();
1981 CHECK_EQ(start + 1, end);
1982
1983 // Because we can update initial domains, an associated literal might
1984 // fall in a domain hole and can be different when canonicalized.
1985 //
1986 // TODO(user): Make the contract clearer, it is messy right now.
1987 if (/*DISABLES_CODE*/ (false)) {
1988 CHECK_EQ(literals_reason_buffer_[start],
1989 Literal(associated_lit).Negated());
1990 }
1991 }
1992 {
1993 const int start = bounds_reason_starts_[reason_index];
1994 const int end = reason_index + 1 < bounds_reason_starts_.size()
1995 ? bounds_reason_starts_[reason_index + 1]
1996 : bounds_reason_buffer_.size();
1997 CHECK_EQ(start, end);
1998 }
1999 }
2000 }
2001
2002 // Process this entry. Note that if any of the next expansion include the
2003 // variable entry.var in their reason, we must process it again because we
2004 // cannot easily detect if it was needed to infer the current entry.
2005 //
2006 // Important: the queue might already contains entries referring to the same
2007 // variable. The code act like if we deleted all of them at this point, we
2008 // just do that lazily. tmp_var_to_trail_index_in_queue_[var] will
2009 // only refer to newly added entries.
2010 tmp_var_to_trail_index_in_queue_[entry.var] = 0;
2011 has_dependency_ = false;
2012
2013 ComputeLazyReasonIfNeeded(trail_index);
2014 AppendLiteralsReason(trail_index, output);
2015 for (const int next_trail_index : Dependencies(trail_index)) {
2016 if (next_trail_index < 0) break;
2017 DCHECK_LT(next_trail_index, trail_index);
2018 const TrailEntry& next_entry = integer_trail_[next_trail_index];
2019
2020 // Only add literals that are not "implied" by the ones already present.
2021 // For instance, do not add (x >= 4) if we already have (x >= 7). This
2022 // translate into only adding a trail index if it is larger than the one
2023 // in the queue referring to the same variable.
2024 const int index_in_queue =
2025 tmp_var_to_trail_index_in_queue_[next_entry.var];
2026 if (index_in_queue != std::numeric_limits<int32_t>::max())
2027 has_dependency_ = true;
2028 if (next_trail_index > index_in_queue) {
2029 tmp_var_to_trail_index_in_queue_[next_entry.var] = next_trail_index;
2030 tmp_queue_.push_back(next_trail_index);
2031 std::push_heap(tmp_queue_.begin(), tmp_queue_.end());
2032 }
2033 }
2034
2035 // Special case for a "leaf", we will never need this variable again.
2036 if (!has_dependency_) {
2037 tmp_to_clear_.push_back(entry.var);
2038 tmp_var_to_trail_index_in_queue_[entry.var] =
2039 std::numeric_limits<int32_t>::max();
2040 }
2041 }
2042
2043 // clean-up.
2044 for (const IntegerVariable var : tmp_to_clear_) {
2045 tmp_var_to_trail_index_in_queue_[var] = 0;
2046 }
2047}

2051absl::Span<const Literal> IntegerTrail::Reason(const Trail& trail,
2052 int trail_index) const {
2053 const int index = boolean_trail_index_to_integer_one_[trail_index];
2054 std::vector<Literal>* reason = trail.GetEmptyVectorToStoreReason(trail_index);

2066 return *reason;

2114 // Process any new Literal on the trail.
2115 while (propagation_trail_index_ < trail->Index()) {
2116 const Literal literal = (*trail)[propagation_trail_index_++];
2117 if (literal.Index() >= literal_to_watcher_.size()) continue;
2118 for (const auto entry : literal_to_watcher_[literal]) {
2119 if (!in_queue_[entry.id]) {
2120 in_queue_[entry.id] = true;
2121 queue_by_priority_[id_to_priority_[entry.id]].push_back(entry.id);
2122 }
2123 if (entry.watch_index >= 0) {
2124 id_to_watch_indices_[entry.id].push_back(entry.watch_index);
2125 }
2126 }
2127 }
2128
2129 // Process the newly changed variables lower bounds.
2130 for (const IntegerVariable var : modified_vars_.PositionsSetAtLeastOnce()) {
2131 if (var.value() >= var_to_watcher_.size()) continue;
2132 for (const auto entry : var_to_watcher_[var]) {
2133 if (!in_queue_[entry.id]) {
2134 in_queue_[entry.id] = true;
2135 queue_by_priority_[id_to_priority_[entry.id]].push_back(entry.id);
2136 }
2137 if (entry.watch_index >= 0) {
2138 id_to_watch_indices_[entry.id].push_back(entry.watch_index);
2139 }
2140 }
2141 }
2142
2143 if (trail->CurrentDecisionLevel() == 0 &&
2144 !level_zero_modified_variable_callback_.empty()) {
2145 modified_vars_for_callback_.Resize(modified_vars_.size());
2146 for (const IntegerVariable var : modified_vars_.PositionsSetAtLeastOnce()) {
2147 modified_vars_for_callback_.Set(var);
2148 }
2149 }
2150
2151 modified_vars_.ClearAndResize(integer_trail_->NumIntegerVariables());
2152}

2154bool GenericLiteralWatcher::Propagate(Trail* trail) {
2155 // Only once per call to Propagate(), if we are at level zero, we might want
2156 // to call propagators even if the bounds didn't change.
2157 const int level = trail->CurrentDecisionLevel();
2158 if (level == 0) {
2159 for (const int id : propagator_ids_to_call_at_level_zero_) {
2160 if (in_queue_[id]) continue;
2161 in_queue_[id] = true;
2162 queue_by_priority_[id_to_priority_[id]].push_back(id);
2163 }
2164 }
2165
2166 UpdateCallingNeeds(trail);
2167
2168 // Increase the deterministic time depending on some basic fact about our
2169 // propagation.
2170 int64_t num_propagate_calls = 0;
2171 const int64_t old_enqueue = integer_trail_->num_enqueues();
2172 auto cleanup =
2173 ::absl::MakeCleanup([&num_propagate_calls, old_enqueue, this]() {
2174 const int64_t diff = integer_trail_->num_enqueues() - old_enqueue;
2175 time_limit_->AdvanceDeterministicTime(1e-8 * num_propagate_calls +
2176 1e-7 * diff);
2177 });
2178
2179 // Note that the priority may be set to -1 inside the loop in order to restart
2180 // at zero.
2181 int test_limit = 0;
2182 for (int priority = 0; priority < queue_by_priority_.size(); ++priority) {
2183 // We test the time limit from time to time. This is in order to return in
2184 // case of slow propagation.
2185 //
2186 // TODO(user): The queue will not be emptied, but I am not sure the solver
2187 // will be left in an usable state. Fix if it become needed to resume
2188 // the solve from the last time it was interrupted.
2189 if (test_limit > 100) {
2190 test_limit = 0;
2191 if (time_limit_->LimitReached()) break;
2192 }
2193 if (stop_propagation_callback_ != nullptr && stop_propagation_callback_()) {
2194 integer_trail_->NotifyThatPropagationWasAborted();
2195 break;
2196 }
2197
2198 std::deque<int>& queue = queue_by_priority_[priority];
2199 while (!queue.empty()) {
2200 const int id = queue.front();
2201 current_id_ = id;
2202 queue.pop_front();
2203
2204 // Before we propagate, make sure any reversible structure are up to date.
2205 // Note that we never do anything expensive more than once per level.
2206 {
2207 const int low =
2208 id_to_greatest_common_level_since_last_call_[IdType(id)];
2209 const int high = id_to_level_at_last_call_[id];
2210 if (low < high || level > low) { // Equivalent to not all equal.
2211 id_to_level_at_last_call_[id] = level;
2212 id_to_greatest_common_level_since_last_call_.MutableRef(IdType(id)) =
2213 level;
2214 for (ReversibleInterface* rev : id_to_reversible_classes_[id]) {
2215 if (low < high) rev->SetLevel(low);
2216 if (level > low) rev->SetLevel(level);
2217 }
2218 for (int* rev_int : id_to_reversible_ints_[id]) {
2219 rev_int_repository_->SaveState(rev_int);
2220 }
2221 }
2222 }
2223
2224 // This is needed to detect if the propagator propagated anything or not.
2225 const int64_t old_integer_timestamp = integer_trail_->num_enqueues();
2226 const int64_t old_boolean_timestamp = trail->Index();
2227
2228 // TODO(user): Maybe just provide one function Propagate(watch_indices) ?
2229 ++num_propagate_calls;
2230 std::vector<int>& watch_indices_ref = id_to_watch_indices_[id];
2231 const bool result =
2232 watch_indices_ref.empty()
2233 ? watchers_[id]->Propagate()
2234 : watchers_[id]->IncrementalPropagate(watch_indices_ref);
2235 if (!result) {
2236 watch_indices_ref.clear();
2237 in_queue_[id] = false;
2238 return false;
2239 }
2240
2241 // Update the propagation queue. At this point, the propagator has been
2242 // removed from the queue but in_queue_ is still true.
2243 if (id_to_idempotence_[id]) {
2244 // If the propagator is assumed to be idempotent, then we set
2245 // in_queue_ to false after UpdateCallingNeeds() so this later
2246 // function will never add it back.
2247 UpdateCallingNeeds(trail);
2248 watch_indices_ref.clear();
2249 in_queue_[id] = false;
2250 } else {
2251 // Otherwise, we set in_queue_ to false first so that
2252 // UpdateCallingNeeds() may add it back if the propagator modified any
2253 // of its watched variables.
2254 watch_indices_ref.clear();
2255 in_queue_[id] = false;
2256 UpdateCallingNeeds(trail);
2257 }
2258
2259 // If the propagator pushed a literal, we exit in order to rerun all SAT
2260 // only propagators first. Note that since a literal was pushed we are
2261 // guaranteed to be called again, and we will resume from priority 0.
2262 if (trail->Index() > old_boolean_timestamp) {
2263 // Important: for now we need to re-run the clauses propagator each time
2264 // we push a new literal because some propagator like the arc consistent
2265 // all diff relies on this.
2266 //
2267 // TODO(user): However, on some problem, it seems to work better to not
2268 // do that. One possible reason is that the reason of a "natural"
2269 // propagation might be better than one we learned.
2270 return true;
2271 }
2272
2273 // If the propagator pushed an integer bound, we revert to priority = 0.
2274 if (integer_trail_->num_enqueues() > old_integer_timestamp) {
2275 ++test_limit;
2276 priority = -1; // Because of the ++priority in the for loop.
2277 break;
2278 }
2279 }
2280 }
2281
2282 // We wait until we reach the fix point before calling the callback.
2283 if (trail->CurrentDecisionLevel() == 0) {
2284 const std::vector<IntegerVariable>& modified_vars =
2285 modified_vars_for_callback_.PositionsSetAtLeastOnce();
2286 for (const auto& callback : level_zero_modified_variable_callback_) {
2287 callback(modified_vars);
2288 }
2289 modified_vars_for_callback_.ClearAndResize(
2290 integer_trail_->NumIntegerVariables());
2291 }
2292
2293 return true;
2294}

2370ExcludeCurrentSolutionWithoutIgnoredVariableAndBacktrack() {
2371 return [=](Model* model) {
2372 SatSolver* sat_solver = model->GetOrCreate<SatSolver>();
2373 IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
2374 IntegerEncoder* encoder = model->GetOrCreate<IntegerEncoder>();

2378 clause_to_exclude_solution.reserve(current_level);

2380 bool include_decision = true;
2381 const Literal decision = sat_solver->Decisions()[i].literal;
2382
2383 // Tests if this decision is associated to a bound of an ignored variable
2384 // in the current assignment.
2385 const InlinedIntegerLiteralVector& associated_literals =
2386 encoder->GetIntegerLiterals(decision);
2387 for (const IntegerLiteral bound : associated_literals) {
2388 if (integer_trail->IsCurrentlyIgnored(bound.var)) {
2389 // In this case we replace the decision (which is a bound on an
2390 // ignored variable) with the fact that the integer variable was
2391 // ignored. This works because the only impact a bound of an ignored
2392 // variable can have on the rest of the model is through the
2393 // is_ignored literal.
2394 clause_to_exclude_solution.push_back(
2395 integer_trail->IsIgnoredLiteral(bound.var).Negated());
2396 include_decision = false;
2397 }
2398 }
2399
2400 if (include_decision) {
2401 clause_to_exclude_solution.push_back(decision.Negated());
2402 }
2403 }
2404
2405 // Note that it is okay to add duplicates literals in ClauseConstraint(),
2406 // the clause will be preprocessed correctly.
