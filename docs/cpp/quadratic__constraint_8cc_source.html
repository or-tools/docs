26BoundedQuadraticExpression QuadraticConstraint::AsBoundedQuadraticExpression()
27 const {
28 QuadraticExpression expression;
29 const QuadraticConstraintData& data = storage()->constraint_data(id_);
30 for (const auto [var, coeff] : data.linear_terms.terms()) {
31 expression += coeff * Variable(storage(), var);
32 }
33 for (const auto& [first_var, second_var, coeff] :
34 data.quadratic_terms.Terms()) {
35 expression += coeff * Variable(storage(), first_var) *
36 Variable(storage(), second_var);
37 }
38 return data.lower_bound <= std::move(expression) <= data.upper_bound;
39}
