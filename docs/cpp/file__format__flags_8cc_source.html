56std::string AbslUnparseFlag(const FileFormat f) {
57 switch (f) {
58 case FileFormat::kMathOptBinary:
59 return "mathopt";
60 case FileFormat::kMathOptText:
61 return "mathopt_txt";
62 case FileFormat::kLinearSolverBinary:
63 return "linear_solver";
64 case FileFormat::kLinearSolverText:
65 return "linear_solver_txt";
66 case FileFormat::kMPS:
67 return "mps";
68 case FileFormat::kLP:
69 return "lp";
70 }
71}

78bool AbslParseFlag(const absl::string_view text, FileFormat* const f,
79 std::string* const error) {
80 for (const FileFormat candidate : AllFileFormats()) {
81 if (text == AbslUnparseFlag(candidate)) {
82 *f = candidate;
83 return true;
84 }
85 }
86
87 absl::StrAppend(error, "not a known value");
88 return false;
89}

104std::optional<FileFormat> FormatFromFilePath(absl::string_view file_path) {
105 const absl::flat_hash_map<absl::string_view, FileFormat> extensions =
106 ExtensionToFileFormat();
107
108 // Sort extensions in reverse lexicographic order. The point here would be to
109 // consider ".pb.txt" before we text for ".txt".
110 using ExtensionPair = std::pair<absl::string_view, FileFormat>;
111 std::vector<ExtensionPair> sorted_extensions(extensions.begin(),
112 extensions.end());
113 absl::c_sort(sorted_extensions,
114 [](const ExtensionPair& lhs, const ExtensionPair& rhs) {
115 return lhs > rhs;
116 });
117
118 for (const auto& [extension, format] : sorted_extensions) {
119 if (absl::EndsWith(file_path, extension)) {
120 return format;
121 }
122 }
123 return std::nullopt;
124}

126std::optional<FileFormat> FormatFromFlagOrFilePath(

130 return *format_flag_value;

132 return FormatFromFilePath(file_path);

142std::string OptionalFormatFlagPossibleValuesList() {
143 // Get the lines for each format and the introduction doc.
144 std::string list = FormatFlagPossibleValuesList();
145
146 // Add the doc of what happen when format is not specified.
147 absl::StrAppend(&list, "\n", kListLinePrefix, "<unset>",
148 ": to guess the format from the file extension:");
149
150 // Builds a map from formats to their extensions.
151 absl::flat_hash_map<FileFormat, std::vector<absl::string_view>>
152 format_extensions;
153 for (const auto& [extension, format] : ExtensionToFileFormat()) {
154 format_extensions[format].push_back(extension);
155 }
156 for (auto& [format, extensions] : format_extensions) {
157 absl::c_sort(extensions);
158 }
159
160 // Here we iterate on all formats so that we list them in the same order as in
161 // the enum.
162 for (const FileFormat format : AllFileFormats()) {
163 // Here we use operator[] as it is not an issue to create new empty entries
164 // in the map.
165 const std::vector<absl::string_view>& extensions =
166 format_extensions[format];
167 if (extensions.empty()) {
168 continue;
169 }
170
171 absl::StrAppend(&list, "\n", kSubListLinePrefix,
172 absl::StrJoin(extensions, ", "), ": ",
173 AbslUnparseFlag(format));
174 }
175 return list;
176}

178std::string FormatFlagPossibleValuesList() {

180 {FileFormat::kMathOptBinary, "for a MathOpt ModelProto in binary"},

191 format_help.at(format));
192 }
193 return list;

196absl::StatusOr<std::pair<ModelProto, std::optional<SolutionHintProto>>>
197ReadModel(const absl::string_view file_path, const FileFormat format) {
198 switch (format) {
199 case FileFormat::kMathOptBinary: {
200 ASSIGN_OR_RETURN(ModelProto model, file::GetBinaryProto<ModelProto>(
201 file_path, file::Defaults()));
202 return std::make_pair(std::move(model), std::nullopt);
203 }
204 case FileFormat::kMathOptText: {
205 ASSIGN_OR_RETURN(ModelProto model, file::GetTextProto<ModelProto>(
206 file_path, file::Defaults()));
207 return std::make_pair(std::move(model), std::nullopt);
208 }
209 case FileFormat::kLinearSolverBinary:
210 case FileFormat::kLinearSolverText: {
211 ASSIGN_OR_RETURN(
212 const MPModelProto linear_solver_model,
213 format == FileFormat::kLinearSolverBinary
214 ? file::GetBinaryProto<MPModelProto>(file_path, file::Defaults())
215 : file::GetTextProto<MPModelProto>(file_path, file::Defaults()));
216 ASSIGN_OR_RETURN(ModelProto model,
217 MPModelProtoToMathOptModel(linear_solver_model));
218 ASSIGN_OR_RETURN(
219 std::optional<SolutionHintProto> hint,
220 MPModelProtoSolutionHintToMathOptHint(linear_solver_model));
221 return std::make_pair(std::move(model), std::move(hint));
222 }
223 case FileFormat::kMPS: {
224 ASSIGN_OR_RETURN(ModelProto model, ReadMpsFile(file_path));
225 return std::make_pair(std::move(model), std::nullopt);
226 }
227 case FileFormat::kLP: {
228 return absl::UnimplementedError(
229 "MathOpt does not yet support reading .lp files; only writing them");
230 }
231 }
232}

234absl::Status WriteModel(const absl::string_view file_path,
235 const ModelProto& model_proto,
236 const std::optional<SolutionHintProto>& hint_proto,
237 const FileFormat format) {
238 switch (format) {
239 case FileFormat::kMathOptBinary:
240 return file::SetBinaryProto(file_path, model_proto, file::Defaults());
241 case FileFormat::kMathOptText:
242 return file::SetTextProto(file_path, model_proto, file::Defaults());
243 case FileFormat::kLinearSolverBinary:
244 case FileFormat::kLinearSolverText: {
245 ASSIGN_OR_RETURN(const MPModelProto linear_solver_model,
246 MathOptModelToMPModelProto(model_proto));
247 if (hint_proto.has_value()) {
248 LOG(WARNING) << "support for converting a MathOpt hint to MPModelProto "
249 "is not yet supported thus the hint has been lost";
250 }
251 return format == FileFormat::kLinearSolverBinary
252 ? file::SetBinaryProto(file_path, linear_solver_model,
253 file::Defaults())
254 : file::SetTextProto(file_path, linear_solver_model,
255 file::Defaults());
256 }
257 case FileFormat::kMPS: {
258 ASSIGN_OR_RETURN(const std::string mps_data,
259 ModelProtoToMps(model_proto));
260 return file::SetContents(file_path, mps_data, file::Defaults());
261 }
262 case FileFormat::kLP: {
263 ASSIGN_OR_RETURN(const std::string lp_data, ModelProtoToLp(model_proto));
264 return file::SetContents(file_path, lp_data, file::Defaults());
265 }
266 }
267}
