48absl::StatusOr<SolveResult> CallSolve(
49 Solver& solver, const ModelStorage* const expected_storage,
50 const SolveArguments& arguments) {
51 RETURN_IF_ERROR(arguments.CheckModelStorageAndCallback(expected_storage));
52
53 Solver::Callback cb = nullptr;
54 absl::Mutex mutex;
55 absl::Status cb_status; // Guarded by `mutex`.
56 if (arguments.callback != nullptr) {
57 cb = [&](const CallbackDataProto& callback_data_proto) {
58 const CallbackData data(expected_storage, callback_data_proto);
59 // TODO(b/249995436): offer a way for user-callback to return a Status as
60 // well and somehow label it as "user error" (annotation + payload?).
61 const CallbackResult result = arguments.callback(data);
62
63 if (const absl::Status status =
64 result.CheckModelStorage(expected_storage);
65 !status.ok()) {
66 // Note that we use util::StatusBuilder() here as util::Annotate() is
67 // not available in open-source code.
68 util::StatusBuilder builder(status);
69 builder << "invalid CallbackResult returned by user callback";
70
71 const absl::MutexLock lock(&mutex);
72 cb_status.Update(builder);
73
74 // Trigger early termination of the solve.
75 CallbackResultProto result_proto;
76 result_proto.set_terminate(true);
77 return result_proto;
78 }
79
80 return result.Proto();
81 };
82 }
83 ASSIGN_OR_RETURN(
84 const SolveResultProto solve_result,
85 solver.Solve(
86 {.parameters = arguments.parameters.Proto(),
87 .model_parameters = arguments.model_parameters.Proto(),
88 .message_callback = arguments.message_callback,
89 .callback_registration = arguments.callback_registration.Proto(),
90 .user_cb = std::move(cb),
91 .interrupter = arguments.interrupter}));
92
93 const absl::MutexLock lock(&mutex);
94 RETURN_IF_ERROR(cb_status);
95
96 return SolveResult::FromProto(expected_storage, solve_result);
97}

101absl::StatusOr<SolveResult> Solve(const Model& model,
102 const SolverType solver_type,
103 const SolveArguments& solve_args,
104 const SolverInitArguments& init_args) {
105 ASSIGN_OR_RETURN(const std::unique_ptr<Solver> solver,
106 Solver::New(EnumToProto(solver_type), model.ExportModel(),
107 ToSolverInitArgs(init_args)));
108 return CallSolve(*solver, model.storage(), solve_args);
109}

111absl::StatusOr<std::unique_ptr<IncrementalSolver>> IncrementalSolver::New(
112 Model* const model, const SolverType solver_type,
113 SolverInitArguments arguments) {
114 if (model == nullptr) {
115 return absl::InvalidArgumentError("input model can't be null");
116 }
117 std::unique_ptr<UpdateTracker> update_tracker = model->NewUpdateTracker();
118 ASSIGN_OR_RETURN(const ModelProto model_proto, update_tracker->ExportModel());
119 ASSIGN_OR_RETURN(std::unique_ptr<Solver> solver,
120 Solver::New(EnumToProto(solver_type), model_proto,
121 ToSolverInitArgs(arguments)));
122 return absl::WrapUnique<IncrementalSolver>(
123 new IncrementalSolver(solver_type, std::move(arguments), model->storage(),
124 std::move(update_tracker), std::move(solver)));
125}

144absl::StatusOr<UpdateResult> IncrementalSolver::Update() {
145 ASSIGN_OR_RETURN(std::optional<ModelUpdateProto> model_update,
146 update_tracker_->ExportModelUpdate());
147 if (!model_update) {
148 return UpdateResult(true, std::move(model_update));
149 }
150
151 OR_ASSIGN_OR_RETURN3(const bool did_update, solver_->Update(*model_update),
152 _ << "update failed");
153 RETURN_IF_ERROR(update_tracker_->AdvanceCheckpoint());
154
155 if (did_update) {
156 return UpdateResult(true, std::move(model_update));
157 }
158
159 ASSIGN_OR_RETURN(const ModelProto model_proto,
160 update_tracker_->ExportModel());
161 OR_ASSIGN_OR_RETURN3(solver_,
162 Solver::New(EnumToProto(solver_type_), model_proto,
163 ToSolverInitArgs(init_args_)),
164 _ << "solver re-creation failed");
165
166 return UpdateResult(false, std::move(model_update));
167}
