40// For an optimization problem, this contains the internal integer objective
41// to minimize and information on how to display it correctly in the logs.
42struct ObjectiveDefinition {
43 double scaling_factor = 1.0;
44 double offset = 0.0;
45 IntegerVariable objective_var = kNoIntegerVariable;
46
47 // The objective linear expression that should be equal to objective_var.
48 // If not all proto variable have an IntegerVariable view, then some vars
49 // will be set to kNoIntegerVariable. In practice, when this is used, we make
50 // sure there is a view though.
51 std::vector<IntegerVariable> vars;
52 std::vector<IntegerValue> coeffs;
53
54 // List of variable that when set to their lower bound should help getting a
55 // better objective. This is used by some search heuristic to preferably
56 // assign any of the variable here to their lower bound first.
57 absl::flat_hash_set<IntegerVariable> objective_impacting_variables;
58
59 double ScaleIntegerObjective(IntegerValue value) const {
60 return (ToDouble(value) + offset) * scaling_factor;
61 }
62};

64// Holds the mapping between CpModel proto indices and the sat::model ones.
65//
66// This also holds some information used when loading a CpModel proto.
67class CpModelMapping {
68 public:
69 // Returns true if the given CpModelProto variable reference refers to a
70 // Boolean variable. Such variable will always have an associated Literal(),
71 // but not always an associated Integer().
72 bool IsBoolean(int ref) const {
73 DCHECK_LT(PositiveRef(ref), booleans_.size());
74 return booleans_[PositiveRef(ref)] != kNoBooleanVariable;
75 }
76
77 bool IsInteger(int ref) const {
78 DCHECK_LT(PositiveRef(ref), integers_.size());
79 return integers_[PositiveRef(ref)] != kNoIntegerVariable;
80 }
81
82 sat::Literal Literal(int ref) const {
83 DCHECK(IsBoolean(ref));
84 return sat::Literal(booleans_[PositiveRef(ref)], RefIsPositive(ref));
85 }
86
87 IntegerVariable Integer(int ref) const {
88 DCHECK(IsInteger(ref));
89 const IntegerVariable var = integers_[PositiveRef(ref)];
90 return RefIsPositive(ref) ? var : NegationOf(var);
91 }
92
93 // TODO(user): We could "easily" create an intermediate variable for more
94 // complex linear expression. We could also identify duplicate expressions to
95 // not create two identical integer variable.
96 AffineExpression Affine(const LinearExpressionProto& exp) const {
97 CHECK_LE(exp.vars().size(), 1);
98 if (exp.vars().empty()) {
99 return AffineExpression(IntegerValue(exp.offset()));
100 }
101 return AffineExpression(Integer(exp.vars(0)), IntegerValue(exp.coeffs(0)),
102 IntegerValue(exp.offset()));
103 }
104
105 IntervalVariable Interval(int i) const {
106 CHECK_GE(i, 0);
107 CHECK_LT(i, intervals_.size());
108 CHECK_NE(intervals_[i], kNoIntervalVariable);
109 return intervals_[i];
110 }
111
112 template <typename List>
113 std::vector<IntegerVariable> Integers(const List& list) const {
114 std::vector<IntegerVariable> result;
115 for (const auto i : list) result.push_back(Integer(i));
116 return result;
117 }
118
119 template <typename ProtoIndices>
120 std::vector<sat::Literal> Literals(const ProtoIndices& indices) const {
121 std::vector<sat::Literal> result;
122 for (const int i : indices) result.push_back(CpModelMapping::Literal(i));
123 return result;
124 }
125
126 template <typename List>
127 std::vector<AffineExpression> Affines(const List& list) const {
128 std::vector<AffineExpression> result;
129 for (const auto& i : list) result.push_back(Affine(i));
130 return result;
131 }
132
133 template <typename ProtoIndices>
134 std::vector<IntervalVariable> Intervals(const ProtoIndices& indices) const {
135 std::vector<IntervalVariable> result;
136 for (const int i : indices) result.push_back(Interval(i));
137 return result;
138 }
139
140 // Depending on the option, we will load constraints in stages. This is used
141 // to detect constraints that are already loaded. For instance the interval
142 // constraints and the linear constraint of size 1 (encodings) are usually
143 // loaded first.
144 bool ConstraintIsAlreadyLoaded(const ConstraintProto* ct) const {
145 return already_loaded_ct_.contains(ct);
146 }
147
148 // Returns true if the given constraint is a "half-encoding" constraint. That
149 // is, if it is of the form (b => size 1 linear) but there is no (<=) side in
150 // the model. Such constraint are detected while we extract integer encoding
151 // and are cached here so that we can deal properly with them during the
152 // linear relaxation.
153 bool IsHalfEncodingConstraint(const ConstraintProto* ct) const {
154 return is_half_encoding_ct_.contains(ct);
155 }
156
157 // Note that both these functions returns positive reference or -1.
158 int GetProtoVariableFromBooleanVariable(BooleanVariable var) const {
159 if (var.value() >= reverse_boolean_map_.size()) return -1;
160 return reverse_boolean_map_[var];
161 }
162 int GetProtoVariableFromIntegerVariable(IntegerVariable var) const {
163 if (var.value() >= reverse_integer_map_.size()) return -1;
164 return reverse_integer_map_[var];
165 }
166
167 const std::vector<IntegerVariable>& GetVariableMapping() const {
168 return integers_;
169 }
170
171 LinearExpression GetExprFromProto(
172 const LinearExpressionProto& expr_proto) const {
173 LinearExpression expr;
174 expr.vars = Integers(expr_proto.vars());
175 for (int j = 0; j < expr_proto.coeffs_size(); ++j) {
176 expr.coeffs.push_back(IntegerValue(expr_proto.coeffs(j)));
177 }
178 expr.offset = IntegerValue(expr_proto.offset());
179 return CanonicalizeExpr(expr);
180 }
181
182 // For logging only, these are not super efficient.
183 int NumIntegerVariables() const {
184 int result = 0;
185 for (const IntegerVariable var : integers_) {
186 if (var != kNoIntegerVariable) result++;
187 }
188 return result;
189 }
190 int NumBooleanVariables() const {
191 int result = 0;
192 for (const BooleanVariable var : booleans_) {
193 if (var != kNoBooleanVariable) result++;
194 }
195 return result;
196 }
197
198 // Returns the number of variables in the loaded proto.
199 int NumProtoVariables() const { return integers_.size(); }
200
201 private:
202 friend void LoadVariables(const CpModelProto& model_proto,
203 bool view_all_booleans_as_integers, Model* m);
204 friend void ExtractEncoding(const CpModelProto& model_proto, Model* m);
205
206 // Note that only the variables used by at least one constraint will be
207 // created, the other will have a kNo[Integer,Interval,Boolean]VariableValue.
208 std::vector<IntegerVariable> integers_;
209 std::vector<IntervalVariable> intervals_;
210 std::vector<BooleanVariable> booleans_;
211
212 // Recover from a IntervalVariable/BooleanVariable its associated CpModelProto
213 // index. The value of -1 is used to indicate that there is no correspondence
214 // (i.e. this variable is only used internally).
215 absl::StrongVector<BooleanVariable, int> reverse_boolean_map_;
216 absl::StrongVector<IntegerVariable, int> reverse_integer_map_;
217
218 // Set of constraints to ignore because they were already dealt with by
219 // ExtractEncoding().
220 absl::flat_hash_set<const ConstraintProto*> already_loaded_ct_;
221 absl::flat_hash_set<const ConstraintProto*> is_half_encoding_ct_;
222};
