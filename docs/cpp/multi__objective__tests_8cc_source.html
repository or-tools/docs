165// For a univariate optimization problem with two objectives.
166struct SimpleMultiObjectiveSolveResult {
167 TerminationReason termination = TerminationReason::kOtherError;
168 double solution = 0.0;
169 double priority_0_objective_value = 0.0;
170 double priority_1_objective_value = 0.0;
171};
172
173enum class ObjectiveType { kPrimary, kAuxiliary };
174enum class ToleranceType { kAbsolute, kRelative };
175
176// We consider the two objective model
177// priority 0: max(x)
178// priority 1: min(x)
179// s.t. 0 <= x <= 2
180// x is integer
181//
182// The optimal solution is x^* = 2 with objective values (2, 2). We test the
183// degradation tolerances by setting them on the priority 0 objective such that
184// the optimal solution, (up to tolerances), is x^* = 1 with objective values
185// (1, 1). We can accomplish this with an absolute tolerance of 1 or a relative
186// tolerance of 0.5.
187absl::StatusOr<SimpleMultiObjectiveSolveResult> SolveWithObjectiveDegradation(
188 const SolverType solver_type, const SolveParameters& parameters,
189 const ObjectiveType priority_0_type, const ObjectiveType priority_1_type,
190 const ToleranceType tolerance_type) {
191 if (priority_0_type == ObjectiveType::kPrimary &&
192 priority_1_type == ObjectiveType::kPrimary) {
193 return absl::InvalidArgumentError("Both objectives cannot both be primary");
194 }
195 Model model;
196 const Variable x = model.AddIntegerVariable(0.0, 2.0, "x");
197 const Objective priority_0 = [&]() {
198 switch (priority_0_type) {
199 case ObjectiveType::kPrimary:
200 model.Maximize(x);
201 model.set_objective_priority(model.primary_objective(), 0);
202 return model.primary_objective();
203 case ObjectiveType::kAuxiliary:
204 return model.AddMaximizationObjective(x, /*priority=*/0);
205 }
206 }();
207 const Objective priority_1 = [&]() {
208 switch (priority_1_type) {
209 case ObjectiveType::kPrimary:
210 model.Minimize(x);
211 model.set_objective_priority(model.primary_objective(), 1);
212 return model.primary_objective();
213 case ObjectiveType::kAuxiliary:
214 return model.AddMinimizationObjective(x, /*priority=*/1);
215 }
216 }();
217 ModelSolveParameters model_parameters;
218 switch (tolerance_type) {
219 case ToleranceType::kAbsolute:
220 model_parameters.objective_parameters[priority_0]
221 .objective_degradation_absolute_tolerance = 1.0;
222 break;
223 case ToleranceType::kRelative:
224 model_parameters.objective_parameters[priority_0]
225 .objective_degradation_relative_tolerance = 0.5;
226 }
227 ASSIGN_OR_RETURN(
228 const SolveResult result,
229 Solve(model, solver_type,
230 {.parameters = parameters, .model_parameters = model_parameters}));
231 if (!result.has_primal_feasible_solution()) {
232 return absl::InternalError("No feasible solution found");
233 }
234 return SimpleMultiObjectiveSolveResult{
235 .termination = result.termination.reason,
236 .solution = result.best_primal_solution().variable_values.at(x),
237 .priority_0_objective_value = result.objective_value(priority_0),
238 .priority_1_objective_value = result.objective_value(priority_1)};
239}

241TEST_P(SimpleMultiObjectiveTest, PrimaryObjectiveDegradationAbsoluteTolerance) {
242 if (!GetParam().supports_auxiliary_objectives) {
243 GTEST_SKIP() << kNoMultiObjectiveSupportMessage;
244 }
245 ASSERT_OK_AND_ASSIGN(const SimpleMultiObjectiveSolveResult result,
246 SolveWithObjectiveDegradation(
247 GetParam().solver_type, GetParam().parameters,
248 /*priority_0_type=*/ObjectiveType::kPrimary,

250 ToleranceType::kAbsolute));

252 EXPECT_THAT(result.solution, DoubleNear(1.0, kTolerance));

254 EXPECT_THAT(result.priority_1_objective_value, DoubleNear(1.0, kTolerance));
255}

347// We start with the single objective model:
348// max x
349// s.t. x + y <= 1.5
350// 0 <= x, y <= 1
351//
352// The optimal objective value is 1, and (x^*, y^*) = (1, a) is optimal for any
353// value a in [0, 1].
354//
355// We then add the secondary objective
356//
357// max {x, x + 3*y + 2}
358// s.t. x + y <= 1.5
359// 0 <= x, y <= 1
360//
361// This has the unique optimal solution (x^*, y^*) = (1, 0.5) with objective
362// values (1, 4.5).
363TEST_P(IncrementalMultiObjectiveTest, SingleToMultiObjectiveModel) {
364 Model model;
365 const Variable x = model.AddContinuousVariable(0.0, 1.0, "x");
366 const Variable y = model.AddContinuousVariable(0.0, 1.0, "y");
367 model.AddLinearConstraint(x + y <= 1.5);
368 model.Maximize(x);
369
370 ASSERT_OK_AND_ASSIGN(
371 const auto solver,
372 IncrementalSolver::New(&model, GetParam().solver_type, {}));
373 // Since there are multiple optimal solutions we do not match against the
374 // solution value.
375 ASSERT_THAT(solver->Solve({.parameters = GetParam().parameters}),
376 IsOkAndHolds(IsOptimal(1.0)));
377
378 const Objective o =
379 model.AddMaximizationObjective(x + 3.0 * y + 2.0, /*priority=*/1);
380
381 if (!GetParam().supports_auxiliary_objectives) {
382 // Here we test that solvers that don't support auxiliary objectives return
383 // false in SolverInterface::CanUpdate(). Thus they should fail in their
384 // factory function instead of failing in their SolverInterface::Update()
385 // function. To assert we rely on status annotations added by
386 // IncrementalSolver::Update() to the returned status of Solver::Update()
387 // and Solver::New().
388 EXPECT_THAT(
389 solver->Update(),
390 StatusIs(AnyOf(absl::StatusCode::kInvalidArgument,
391 absl::StatusCode::kUnimplemented),
392 AllOf(HasSubstr("multiple objective"),
393 // Sub-string expected for Solver::Update() error.
394 Not(HasSubstr("update failed")),
395 // Sub-string expected for Solver::New() error.
396 HasSubstr("solver re-creation failed"))));
397 return;
398 }
399
400 ASSERT_THAT(
401 solver->Update(),
402 IsOkAndHolds(GetParam().supports_incremental_objective_add_and_delete
403 ? DidUpdate()
404 : Not(DidUpdate())));
405 ASSERT_OK_AND_ASSIGN(
406 const SolveResult result,
407 solver->SolveWithoutUpdate({.parameters = GetParam().parameters}));
408 ASSERT_THAT(result, IsOptimalWithSolution(1.0, {{x, 1.0}, {y, 0.5}}));
409 EXPECT_EQ(result.objective_value(o), 4.5);
410}
