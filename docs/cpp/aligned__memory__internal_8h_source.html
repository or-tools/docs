27template <typename T, size_t alignment_bytes, size_t misalignment_bytes>
28struct AllocatorWithAlignment : public std::allocator<T> {
29 // Allocates memory for num_items items of type T. The memory must be freed
30 // using deallocate(); using it with free() or `delete` might cause unexpected
31 // behavior when misalignment is used.
32 T* allocate(size_t num_items) {
33 // Having misalignment_bytes >= alignment_bytes is useless, because all
34 // misalignments are equivalent modulo `alignment_bytes`. Disallowing it
35 // allows us to simplify the code below.
36 static_assert(alignment_bytes == 0 || misalignment_bytes < alignment_bytes);
37
38 // `std::aligned_alloc(alignment, size)` requires that `size` is a multiple
39 // of `alignment`, and might return a nullptr when this is not respected. To
40 // be safe, we round the number of bytes up to alignment.
41 const size_t num_required_bytes =
42 misalignment_bytes + num_items * sizeof(T);
43
44 const size_t num_allocated_bytes =
45 MathUtil::RoundUpTo(num_required_bytes, alignment_bytes);
46
47 std::uintptr_t ptr = reinterpret_cast<std::uintptr_t>(
48 std::aligned_alloc(alignment_bytes, num_allocated_bytes));
49 return reinterpret_cast<T*>(ptr + misalignment_bytes);
50 }
51 // A version of allocate() that takes a hint; we just ignore the hint.
52 T* allocate(size_t n, const void*) { return allocate(n); }
53
54 // Frees memory allocated by allocate().
55 void deallocate(T* p, size_t) {
56 std::uintptr_t aligned_pointer =
57 reinterpret_cast<std::uintptr_t>(p) - misalignment_bytes;
58 free(reinterpret_cast<void*>(aligned_pointer));
59 }
60
61 // Rebind must be specialized to produce AllocatorWithAlignment and not
62 // std::allocator. It uses the same alignment and misalignment as its source.
63 template <typename U>
64 struct rebind {
65 using other =
66 AllocatorWithAlignment<U, alignment_bytes, misalignment_bytes>;
67 };
68};
