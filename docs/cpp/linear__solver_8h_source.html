192 enum OptimizationProblemType {
193 // Linear programming problems.
194 // ----------------------------
195 CLP_LINEAR_PROGRAMMING = 0,
196 GLPK_LINEAR_PROGRAMMING = 1,

198 // In-house linear programming solver based on the primal-dual hybrid
199 // gradient method. Sometimes faster than Glop for medium-size problems and
200 // scales to much larger problems than Glop.
201 PDLP_LINEAR_PROGRAMMING = 8,
202
203 // Integer programming problems.
204 // -----------------------------
205 SCIP_MIXED_INTEGER_PROGRAMMING = 3, // Recommended default value.

213 CPLEX_MIXED_INTEGER_PROGRAMMING = 11,
214 XPRESS_LINEAR_PROGRAMMING = 101,
215 XPRESS_MIXED_INTEGER_PROGRAMMING = 102,
216
217 // Boolean optimization problem (requires only integer variables and works
218 // best with only Boolean variables).
219 BOP_INTEGER_PROGRAMMING = 12,

221 // SAT based solver (requires only integer and Boolean variables).
222 // If you pass it mixed integer problems, it will scale coefficients to

226 // Dedicated knapsack solvers.
