198 // ----------------------------

208 // Integer programming problems.
209 // -----------------------------

219 CPLEX_MIXED_INTEGER_PROGRAMMING = 11,
220 XPRESS_LINEAR_PROGRAMMING = 101,
221 XPRESS_MIXED_INTEGER_PROGRAMMING = 102,

224 // Boolean optimization problem (requires only integer variables and works
225 // best with only Boolean variables).
