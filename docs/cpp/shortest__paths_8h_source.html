14// This file contains functions to compute shortest paths on Ebert graphs using
15// Dijkstra's algorithm,
16// E.W. Dijkstra, "A note on two problems in connexion with graphs". Numerische
17// Mathematik 1:269â€“271, 1959. See for example:
18// http://www.springerlink.com/content/uu8608u0u27k7256/fulltext.pdf.
19// More information can also be found on Wikipedia:
20// http://en.wikipedia.org/wiki/Dijkstra's_algorithm
21//
22// This is a unidirectional implementation of Dijkstra's algorithm. A
23// bidirectional is available in bidirectional_dijkstra.h for specific use
24// cases.
25//
26// Each 1-to-many shortest path computation is run in a separate thread. Users
27// should select the number of threads to use according to the number of cores
28// available (each thread will use up one core). However, increasing the number
29// of threads also increases temporary memory used by each 1-to-many
30// computation.
31//
32// Also included are classes to store path data resulting from shortest path
33// computations (cf. PathContainer).
34//
35// Usage example computing all-pair shortest paths on a graph:
36// StarGraph graph(...,...);
37// ZVector<uint32_t> arc_lengths(...,...);
38// ... populate graph and arc lengths ...
39// PathContainer container;
40// PathContainer::BuildInMemoryCompactPathContainer(&container);
41// ComputeAllToAllShortestPathsWithMultipleThreads(graph,
42// arc_lengths,
43// /*num_threads=*/4,
44// &container);
45//
46// Usage example computing shortest paths between a subset of graph nodes:
47// StarGraph graph(...,...);
48// ZVector<uint32_t> arc_lengths(...,...);
49// ... populate graph and arc lengths ...
50// vector<NodeIndex> sources;
51// vector<NodeIndex> sinks;
52// ... fill sources and sinks ...
53// PathContainer container;
54// PathContainer::BuildInMemoryCompactPathContainer(&container);
55// ComputeManyToManyShortestPathsWithMultipleThreads(graph,
56// arc_lengths,
57// sources,
58// sinks,
59// /*num_threads=*/4,
60// &container);

91// Container class storing paths and distances along the paths. It is used in
92// shortest path computation functions to store resulting shortest paths.
93// Usage example iterating on the path between nodes from and to:
94// PathContainer path_container;
95// PathContainer::BuildInMemoryCompactPathContainer(&path_container);
96// ... fill up container ...
97// const NodeIndex from =...;
98// NodeIndex to =...;
99// while (to != from) {
100// LOG(INFO) << to;
101// to = path_container.GetPenultimateNodeInPath(from, to);
102// }
103class PathContainer {
104 public:
105 PathContainer();
106
107 // This type is neither copyable nor movable.
108 PathContainer(const PathContainer&) = delete;
109 PathContainer& operator=(const PathContainer&) = delete;
110
111 ~PathContainer();
112
113 // Returns the distance between node 'from' and node 'to' following the path
114 // out of 'from' and into 'to'. Note that if 'from' == 'to', the distance is
115 // not necessarily 0 if the path out of 'to' and back into 'to' has a distance
116 // greater than 0. If you do require the distance to be 0 in this case, add to
117 // the graph an arc from 'to' to itself with a length of 0.
118 // If nodes are not connected, returns kDisconnectedPathDistance.
119 PathDistance GetDistance(NodeIndex from, NodeIndex to) const;
120
121 // Returns the penultimate node on the path out of node 'from' into node 'to'
122 // (the direct predecessor of node 'to' on the path).
123 // If 'from' == 'to', the penultimate node is 'to' only if the shortest path
124 // from 'to' to itself is composed of the arc ('to, 'to'), which might not be
125 // the case if either this arc doesn't exist or if the length of this arc is
126 // greater than the distance of an alternate path.
127 // If nodes are not connected, returns StarGraph::kNilNode.
128 NodeIndex GetPenultimateNodeInPath(NodeIndex from, NodeIndex to) const;
129
130 // Returns path nodes from node "from" to node "to" in a ordered vector.
131 // The vector starts with 'from' and ends with 'to', if both nodes are
132 // connected (otherwise an empty vector is returned).
133 void GetPath(NodeIndex from, NodeIndex to,
134 std::vector<NodeIndex>* path) const;
135
136 // For internal use only. Returns the internal container implementation.
137 PathContainerImpl* GetImplementation() const;
138
139 // Builds a path container which only stores distances between path nodes.
140 static void BuildPathDistanceContainer(PathContainer* path_container);
141
142 // Builds a path container which stores explicit paths and distances between
143 // path nodes in a memory-compact representation.
144 // In this case GetPenultimateNodeInPath() is O(log(path_tree_size)),
145 // path_tree_size being the size of a tree of paths from a source node (in
146 // practice it is equal to the number of nodes in the graph if all nodes
147 // are strongly connected).
148 // GetPath is O(log(path_tree_size) + path_size), where path_size is the
149 // size of the resulting path; note this is faster than successive calls
150 // to GetPenultimateNodeInPath() which would result in
151 // O(log(path_tree_size) * path_size).
152 static void BuildInMemoryCompactPathContainer(PathContainer* path_container);
153
154 // TODO(user): Add save-to-disk container.
155 // TODO(user): Add BuildInMemoryFastPathContainer(), which does
156 // GetPenultimateNodeInPath() in O(1).
157
158 private:
159 std::unique_ptr<PathContainerImpl> container_;
160};

162// Utility function which returns a vector containing all nodes of a graph.
163template <class GraphType>
164void GetGraphNodes(const GraphType& graph, std::vector<NodeIndex>* nodes) {
165 CHECK(nodes != nullptr);
166 nodes->clear();
167 nodes->reserve(graph.num_nodes());
168 for (typename GraphType::NodeIterator iterator(graph); iterator.Ok();
169 iterator.Next()) {
170 nodes->push_back(iterator.Index());
171 }
172}
173
174template <class GraphType>
175void GetGraphNodesFromGraph(const GraphType& graph,
176 std::vector<typename GraphType::NodeIndex>* nodes) {
177 CHECK(nodes != nullptr);
178 nodes->clear();
179 nodes->reserve(graph.num_nodes());
180 for (const typename GraphType::NodeIndex node : graph.AllNodes()) {
181 nodes->push_back(node);
182 }
183}
