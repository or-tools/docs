70LinearConstraint Model::AddLinearConstraint(
71 const BoundedLinearExpression& bounded_expr, absl::string_view name) {
72 CheckOptionalModel(bounded_expr.expression.storage());
73
74 const LinearConstraintId constraint = storage()->AddLinearConstraint(
75 bounded_expr.lower_bound_minus_offset(),
76 bounded_expr.upper_bound_minus_offset(), name);
77 for (const auto& [variable, coef] : bounded_expr.expression.terms()) {
78 storage()->set_linear_constraint_coefficient(constraint,
79 variable.typed_id(), coef);
80 }
81 return LinearConstraint(storage(), constraint);
82}

206LinearExpression Model::ObjectiveAsLinearExpression() const {
207 CHECK_EQ(storage()->num_quadratic_objective_terms(kPrimaryObjectiveId), 0)
208 << "The objective function contains quadratic terms and cannot be "
209 "represented as a LinearExpression";
210 LinearExpression result = storage()->objective_offset(kPrimaryObjectiveId);
211 for (const auto& [v, coef] :
212 storage()->linear_objective(kPrimaryObjectiveId)) {
213 result += Variable(storage(), v) * coef;
214 }
215 return result;
216}
217
218QuadraticExpression Model::ObjectiveAsQuadraticExpression() const {
219 QuadraticExpression result = storage()->objective_offset(kPrimaryObjectiveId);
220 for (const auto& [v, coef] :
221 storage()->linear_objective(kPrimaryObjectiveId)) {
222 result += Variable(storage(), v) * coef;
223 }
224 for (const auto& [v1, v2, coef] :
225 storage()->quadratic_objective_terms(kPrimaryObjectiveId)) {
226 result +=
227 QuadraticTerm(Variable(storage(), v1), Variable(storage(), v2), coef);
228 }
229 return result;
230}

286std::ostream& operator<<(std::ostream& ostr, const Model& model) {
287 ostr << "Model";
288 if (!model.name().empty()) ostr << " " << model.name();
289 ostr << ":\n";
290
291 if (model.num_auxiliary_objectives() == 0) {
292 ostr << " Objective:\n"
293 << (model.is_maximize() ? " maximize " : " minimize ")
294 << model.ObjectiveAsQuadraticExpression() << "\n";
295 } else {
296 ostr << " Objectives:\n";
297 const auto stream_objective = [](std::ostream& ostr, const Objective obj) {
298 ostr << " " << obj << " (priority " << obj.priority()
299 << "): " << (obj.maximize() ? "maximize " : "minimize ")
300 << obj.AsQuadraticExpression() << "\n";
301 };
302 stream_objective(ostr, model.primary_objective());
303 for (const Objective obj : model.SortedAuxiliaryObjectives()) {
304 stream_objective(ostr, obj);
305 }
306 }
307
308 ostr << " Linear constraints:\n";
309 for (const LinearConstraint constraint : model.SortedLinearConstraints()) {
310 ostr << " " << constraint << ": " << constraint.AsBoundedLinearExpression()
311 << "\n";
312 }
313
314 if (model.num_quadratic_constraints() > 0) {
315 ostr << " Quadratic constraints:\n";
316 for (const QuadraticConstraint constraint :
317 model.SortedQuadraticConstraints()) {
318 ostr << " " << constraint << ": "
319 << constraint.AsBoundedQuadraticExpression() << "\n";
320 }
321 }
322
323 if (model.num_second_order_cone_constraints() > 0) {
324 ostr << " Second-order cone constraints:\n";
325 for (const SecondOrderConeConstraint constraint :
326 model.SortedSecondOrderConeConstraints()) {
327 ostr << " " << constraint << ": " << constraint.ToString() << "\n";
328 }
329 }
330
331 if (model.num_sos1_constraints() > 0) {
332 ostr << " SOS1 constraints:\n";
333 for (const Sos1Constraint constraint : model.SortedSos1Constraints()) {
334 ostr << " " << constraint << ": " << constraint.ToString() << "\n";
335 }
336 }
337
338 if (model.num_sos2_constraints() > 0) {
339 ostr << " SOS2 constraints:\n";
340 for (const Sos2Constraint constraint : model.SortedSos2Constraints()) {
341 ostr << " " << constraint << ": " << constraint.ToString() << "\n";
342 }
343 }
344
345 if (model.num_indicator_constraints() > 0) {
346 ostr << " Indicator constraints:\n";
347 for (const IndicatorConstraint constraint :
348 model.SortedIndicatorConstraints()) {
349 ostr << " " << constraint << ": " << constraint.ToString() << "\n";
350 }
351 }
352
353 ostr << " Variables:\n";
354 for (const Variable v : model.SortedVariables()) {
355 ostr << " " << v;
356 if (v.is_integer()) {
357 if (v.lower_bound() == 0 && v.upper_bound() == 1) {
358 ostr << " (binary)\n";
359 continue;
360 }
361 ostr << " (integer)";
362 }
363 ostr << " in ";
364 if (v.lower_bound() == -kInf) {
365 ostr << "(-∞";
366 } else {
367 ostr << "[" << RoundTripDoubleFormat(v.lower_bound());
368 }
369 ostr << ", ";
370 if (v.upper_bound() == kInf) {
371 ostr << "+∞)";
372 } else {
373 ostr << RoundTripDoubleFormat(v.upper_bound()) << "]";
374 }
375 ostr << "\n";
376 }
377 return ostr;
378}

382QuadraticConstraint Model::AddQuadraticConstraint(
383 const BoundedQuadraticExpression& bounded_expr,
384 const absl::string_view name) {
385 CheckOptionalModel(bounded_expr.expression.storage());
386 SparseCoefficientMap linear_terms;
387 for (const auto [var, coeff] : bounded_expr.expression.linear_terms()) {
388 linear_terms.set(var.typed_id(), coeff);
389 }
390 SparseSymmetricMatrix quadratic_terms;
391 for (const auto& [var_ids, coeff] :
392 bounded_expr.expression.quadratic_terms()) {
393 quadratic_terms.set(var_ids.typed_id().first, var_ids.typed_id().second,
394 coeff);
395 }
396 const QuadraticConstraintId id =
397 storage()->AddAtomicConstraint(QuadraticConstraintData{
398 .lower_bound = bounded_expr.lower_bound_minus_offset(),
399 .upper_bound = bounded_expr.upper_bound_minus_offset(),
400 .linear_terms = std::move(linear_terms),
401 .quadratic_terms = std::move(quadratic_terms),
402 .name = std::string(name),
403 });
404 return QuadraticConstraint(storage(), id);
405}

409SecondOrderConeConstraint Model::AddSecondOrderConeConstraint(
410 const std::vector<LinearExpression>& arguments_to_norm,
411 const LinearExpression& upper_bound, const absl::string_view name) {
412 CheckOptionalModel(upper_bound.storage());
413 std::vector<LinearExpressionData> arguments_to_norm_data;
414 arguments_to_norm_data.reserve(arguments_to_norm.size());
415 for (const LinearExpression& expr : arguments_to_norm) {
416 CheckOptionalModel(expr.storage());
417 arguments_to_norm_data.push_back(FromLinearExpression(expr));
418 }
419 const SecondOrderConeConstraintId id =
420 storage()->AddAtomicConstraint(SecondOrderConeConstraintData{
421 .upper_bound = FromLinearExpression(upper_bound),
422 .arguments_to_norm = std::move(arguments_to_norm_data),
423 .name = std::string(name),
424 });
425 return SecondOrderConeConstraint(storage(), id);
426}

432template <typename SosData>
433SosData MakeSosData(const std::vector<LinearExpression>& expressions,
434 std::vector<double> weights, const absl::string_view name) {
435 std::vector<LinearExpressionData> storage_expressions;
436 storage_expressions.reserve(expressions.size());
437 for (const LinearExpression& expr : expressions) {
438 LinearExpressionData& storage_expr = storage_expressions.emplace_back();
439 storage_expr.offset = expr.offset();
440 for (const auto& [var, coeff] : expr.terms()) {
441 storage_expr.coeffs.set(var.typed_id(), coeff);
442 }
443 }
444 return SosData(std::move(storage_expressions), std::move(weights),
445 std::string(name));
446}

474// --------------------------- Indicator constraints ---------------------------
475
476IndicatorConstraint Model::AddIndicatorConstraint(
477 const Variable indicator_variable,
478 const BoundedLinearExpression& implied_constraint,
479 const bool activate_on_zero, const absl::string_view name) {
480 CheckModel(indicator_variable.storage());
481 CheckOptionalModel(implied_constraint.expression.storage());
482 // We ignore the offset while unpacking here; instead, we account for it below
483 // by using the `{lower,upper}_bound_minus_offset` member functions.
484 auto [expr, _] = FromLinearExpression(implied_constraint.expression);
485 const IndicatorConstraintId id =
486 storage()->AddAtomicConstraint(IndicatorConstraintData{
487 .lower_bound = implied_constraint.lower_bound_minus_offset(),
488 .upper_bound = implied_constraint.upper_bound_minus_offset(),
489 .linear_terms = std::move(expr),
490 .indicator = indicator_variable.typed_id(),
491 .activate_on_zero = activate_on_zero,
492 .name = std::string(name),
493 });
494 return IndicatorConstraint(storage(), id);
495}
