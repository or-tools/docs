115double SatSolver::deterministic_time() const {
116 // Each of these counters mesure really basic operations. The weight are just
117 // an estimate of the operation complexity. Note that these counters are never
118 // reset to zero once a SatSolver is created.
119 //
120 // TODO(user): Find a better procedure to fix the weight than just educated
121 // guess.
122 return 1e-8 * (8.0 * trail_->NumberOfEnqueues() +
123 1.0 * binary_implication_graph_->num_inspections() +
124 4.0 * clauses_propagator_->num_inspected_clauses() +
125 1.0 * clauses_propagator_->num_inspected_clause_literals() +
126
127 // Here there is a factor 2 because of the untrail.
128 20.0 * pb_constraints_->num_constraint_lookups() +
129 2.0 * pb_constraints_->num_threshold_updates() +
130 1.0 * pb_constraints_->num_inspected_constraint_literals());
131}

159bool SatSolver::AddClauseDuringSearch(absl::Span<const Literal> literals) {
160 if (model_is_unsat_) return false;
161 const int index = trail_->Index();
162 if (literals.empty()) return SetModelUnsat();
163 if (literals.size() == 1) return AddUnitClause(literals[0]);
164 if (literals.size() == 2) {
165 // TODO(user): We generate in some corner cases clauses with
166 // literals[0].Variable() == literals[1].Variable(). Avoid doing that and
167 // adding such binary clauses to the graph?
168 if (!binary_implication_graph_->AddBinaryClauseDuringSearch(literals[0],
169 literals[1])) {
170 CHECK_EQ(CurrentDecisionLevel(), 0);
171 return SetModelUnsat();
172 }
173 } else {
174 if (!clauses_propagator_->AddClause(literals)) {
175 CHECK_EQ(CurrentDecisionLevel(), 0);
176 return SetModelUnsat();
177 }
178 }
179
180 // Tricky: Even if nothing new is propagated, calling Propagate() might, via
181 // the LP, deduce new things. This is problematic because some code assumes
182 // that when we create newly associated literals, nothing else changes.
183 if (trail_->Index() == index) return true;
184 return FinishPropagation();
185}

199// Note(user): we assume there is no duplicate literals in the clauses added
200// here if is_safe is true. Most of the code works, but some advanced algo might
201// be wrong/suboptimal if this is the case. So even when presolve is off we need
202// some "cleanup" to enforce this invariant. Alternatively we could have robut
203// algo in all the stack, but that seems a worse design.
204bool SatSolver::AddProblemClause(absl::Span<const Literal> literals,
205 bool is_safe) {
206 SCOPED_TIME_STAT(&stats_);
207 CHECK_EQ(CurrentDecisionLevel(), 0);
208 if (model_is_unsat_) return false;
209
210 // Filter already assigned literals.
211 literals_scratchpad_.clear();
212 for (const Literal l : literals) {
213 if (trail_->Assignment().LiteralIsTrue(l)) return true;
214 if (trail_->Assignment().LiteralIsFalse(l)) continue;
215 literals_scratchpad_.push_back(l);
216 }
217
218 if (!is_safe) {
219 gtl::STLSortAndRemoveDuplicates(&literals_scratchpad_);
220 for (int i = 0; i + 1 < literals_scratchpad_.size(); ++i) {
221 if (literals_scratchpad_[i] == literals_scratchpad_[i + 1].Negated()) {
222 return true;
223 }
224 }
225 }
226
227 AddProblemClauseInternal(literals_scratchpad_);
228
229 // Tricky: The PropagationIsDone() condition shouldn't change anything for a
230 // pure SAT problem, however in the CP-SAT context, calling Propagate() can
231 // tigger computation (like the LP) even if no domain changed since the last
232 // call. We do not want to do that.
233 if (!PropagationIsDone() && !Propagate()) {
234 return SetModelUnsat();
235 }
236 return true;
237}

239bool SatSolver::AddProblemClauseInternal(absl::Span<const Literal> literals) {
240 SCOPED_TIME_STAT(&stats_);
241 if (DEBUG_MODE) {
242 CHECK_EQ(CurrentDecisionLevel(), 0);
243 for (const Literal l : literals) {
244 CHECK(!trail_->Assignment().LiteralIsAssigned(l));
245 }
246 }
247
248 if (literals.empty()) return SetModelUnsat();
249
250 if (literals.size() == 1) {
251 if (drat_proof_handler_ != nullptr) {
252 // Note that we will output problem unit clauses twice, but that is a
253 // small price to pay for having a single variable fixing API.
254 drat_proof_handler_->AddClause({literals[0]});
255 }
256 trail_->EnqueueWithUnitReason(literals[0]);
257 } else if (literals.size() == 2) {
258 // TODO(user): Make sure the presolve do not generate such clauses.
259 if (literals[0] == literals[1]) {
260 // Literal must be true.
261 trail_->EnqueueWithUnitReason(literals[0]);
262 } else if (literals[0] == literals[1].Negated()) {
263 // Always true.
264 return true;
265 } else {
266 AddBinaryClauseInternal(literals[0], literals[1],
267 /*export_clause=*/false);
268 }
269 } else {
270 if (!clauses_propagator_->AddClause(literals, trail_)) {
271 return SetModelUnsat();
272 }
273 }
274
275 return true;
276}

278bool SatSolver::AddLinearConstraintInternal(
279 const std::vector<LiteralWithCoeff>& cst, Coefficient rhs,
280 Coefficient max_value) {
281 SCOPED_TIME_STAT(&stats_);
282 DCHECK(BooleanLinearExpressionIsCanonical(cst));
283 if (rhs < 0) return SetModelUnsat(); // Unsatisfiable constraint.
284 if (rhs >= max_value) return true; // Always satisfied constraint.
285
286 // The case "rhs = 0" will just fix variables, so there is no need to
287 // updates the weighted sign.
288 if (rhs > 0) decision_policy_->UpdateWeightedSign(cst, rhs);
289
290 // Since the constraint is in canonical form, the coefficients are sorted.
291 const Coefficient min_coeff = cst.front().coefficient;
292 const Coefficient max_coeff = cst.back().coefficient;
293
294 // A linear upper bounded constraint is a clause if the only problematic
295 // assignment is the one where all the literals are true.
296 if (max_value - min_coeff <= rhs) {
297 // This constraint is actually a clause. It is faster to treat it as one.
298 literals_scratchpad_.clear();
299 for (const LiteralWithCoeff& term : cst) {
300 literals_scratchpad_.push_back(term.literal.Negated());
301 }
302 return AddProblemClauseInternal(literals_scratchpad_);
303 }
304
305 // Detect at most one constraints. Note that this use the fact that the
306 // coefficient are sorted.
307 if (!parameters_->use_pb_resolution() && max_coeff <= rhs &&
308 2 * min_coeff > rhs) {
309 literals_scratchpad_.clear();
310 for (const LiteralWithCoeff& term : cst) {
311 literals_scratchpad_.push_back(term.literal);
312 }
313 if (!binary_implication_graph_->AddAtMostOne(literals_scratchpad_)) {
314 return SetModelUnsat();
315 }
316 return true;
317 }
318
319 problem_is_pure_sat_ = false;
320
321 // TODO(user): If this constraint forces all its literal to false (when rhs is
322 // zero for instance), we still add it. Optimize this?
323 return pb_constraints_->AddConstraint(cst, rhs, trail_);
324}

326void SatSolver::CanonicalizeLinear(std::vector<LiteralWithCoeff>* cst,
327 Coefficient* bound_shift,
328 Coefficient* max_value) {
329 // This block removes assigned literals from the constraint.
330 Coefficient fixed_variable_shift(0);
331 {
332 int index = 0;
333 for (const LiteralWithCoeff& term : *cst) {
334 if (trail_->Assignment().LiteralIsFalse(term.literal)) continue;
335 if (trail_->Assignment().LiteralIsTrue(term.literal)) {
336 CHECK(SafeAddInto(-term.coefficient, &fixed_variable_shift));
337 continue;
338 }
339 (*cst)[index] = term;
340 ++index;
341 }
342 cst->resize(index);
343 }
344
345 // Now we canonicalize.
346 // TODO(user): fix variables that must be true/false and remove them.
347 Coefficient bound_delta(0);
348 CHECK(ComputeBooleanLinearExpressionCanonicalForm(cst, &bound_delta,
349 max_value));
350
351 CHECK(SafeAddInto(bound_delta, bound_shift));
352 CHECK(SafeAddInto(fixed_variable_shift, bound_shift));
353}

355bool SatSolver::AddLinearConstraint(bool use_lower_bound,
356 Coefficient lower_bound,
357 bool use_upper_bound,
358 Coefficient upper_bound,
359 std::vector<LiteralWithCoeff>* cst) {
360 SCOPED_TIME_STAT(&stats_);
361 CHECK_EQ(CurrentDecisionLevel(), 0);
362 if (model_is_unsat_) return false;
363
364 Coefficient bound_shift(0);
365
366 if (use_upper_bound) {
367 Coefficient max_value(0);
368 CanonicalizeLinear(cst, &bound_shift, &max_value);
369 const Coefficient rhs =
370 ComputeCanonicalRhs(upper_bound, bound_shift, max_value);
371 if (!AddLinearConstraintInternal(*cst, rhs, max_value)) {
372 return SetModelUnsat();
373 }
374 }
375
376 if (use_lower_bound) {
377 // We need to "re-canonicalize" in case some literal were fixed while we
378 // processed one direction.
379 Coefficient max_value(0);
380 CanonicalizeLinear(cst, &bound_shift, &max_value);
381
382 // We transform the constraint into an upper-bounded one.
383 for (int i = 0; i < cst->size(); ++i) {
384 (*cst)[i].literal = (*cst)[i].literal.Negated();
385 }
386 const Coefficient rhs =
387 ComputeNegatedCanonicalRhs(lower_bound, bound_shift, max_value);
388 if (!AddLinearConstraintInternal(*cst, rhs, max_value)) {
389 return SetModelUnsat();
390 }
391 }
392
393 // Tricky: The PropagationIsDone() condition shouldn't change anything for a
394 // pure SAT problem, however in the CP-SAT context, calling Propagate() can
395 // tigger computation (like the LP) even if no domain changed since the last
396 // call. We do not want to do that.
397 if (!PropagationIsDone() && !Propagate()) {
398 return SetModelUnsat();
399 }
400 return true;
401}

403int SatSolver::AddLearnedClauseAndEnqueueUnitPropagation(
404 const std::vector<Literal>& literals, bool is_redundant) {
405 SCOPED_TIME_STAT(&stats_);
406
407 if (literals.size() == 1) {
408 // A length 1 clause fix a literal for all the search.
409 // ComputeBacktrackLevel() should have returned 0.
410 CHECK_EQ(CurrentDecisionLevel(), 0);
411 trail_->EnqueueWithUnitReason(literals[0]);
412 return /*lbd=*/1;
413 }
414
415 if (literals.size() == 2) {
416 if (track_binary_clauses_) {
417 // This clause MUST be knew, otherwise something is wrong.
418 CHECK(binary_clauses_.Add(BinaryClause(literals[0], literals[1])));
419 }
420 if (shared_binary_clauses_callback_ != nullptr) {
421 shared_binary_clauses_callback_(literals[0], literals[1]);
422 }
423 CHECK(binary_implication_graph_->AddBinaryClauseDuringSearch(literals[0],
424 literals[1]));
425 return /*lbd=*/2;
426 }
427
428 CleanClauseDatabaseIfNeeded();
429
430 // Important: Even though the only literal at the last decision level has
431 // been unassigned, its level was not modified, so ComputeLbd() works.
432 const int lbd = ComputeLbd(literals);
433 if (is_redundant && lbd > parameters_->clause_cleanup_lbd_bound()) {
434 --num_learned_clause_before_cleanup_;
435
436 SatClause* clause =
437 clauses_propagator_->AddRemovableClause(literals, trail_);
438
439 // BumpClauseActivity() must be called after clauses_info_[clause] has
440 // been created or it will have no effect.
441 (*clauses_propagator_->mutable_clauses_info())[clause].lbd = lbd;
442 BumpClauseActivity(clause);
443 } else {
444 CHECK(clauses_propagator_->AddClause(literals, trail_));
445 }
446 return lbd;
447}

535// Returns true iff 'b' is subsumed by 'a' (i.e 'a' is included in 'b').
536// This is slow and only meant to be used in DCHECKs.
537bool ClauseSubsumption(const std::vector<Literal>& a, SatClause* b) {
538 std::vector<Literal> superset(b->begin(), b->end());
539 std::vector<Literal> subset(a.begin(), a.end());
540 std::sort(superset.begin(), superset.end());
541 std::sort(subset.begin(), subset.end());
542 return std::includes(superset.begin(), superset.end(), subset.begin(),
543 subset.end());
544}

548int SatSolver::EnqueueDecisionAndBackjumpOnConflict(Literal true_literal) {
549 SCOPED_TIME_STAT(&stats_);
550 if (model_is_unsat_) return kUnsatTrailIndex;
551 DCHECK(PropagationIsDone());
552
553 // We should never enqueue before the assumptions_.
554 if (DEBUG_MODE && !assumptions_.empty()) {
555 CHECK_GE(current_decision_level_, assumption_level_);
556 }
557
558 EnqueueNewDecision(true_literal);
559 if (!FinishPropagation()) return kUnsatTrailIndex;
560 return last_decision_or_backtrack_trail_index_;
561}

563bool SatSolver::RestoreSolverToAssumptionLevel() {
564 if (model_is_unsat_) return false;
565 if (CurrentDecisionLevel() > assumption_level_) {
566 Backtrack(assumption_level_);
567 return true;
568 }
569 if (!FinishPropagation()) return false;
570 return ReapplyAssumptionsIfNeeded();

573bool SatSolver::FinishPropagation() {
574 if (model_is_unsat_) return false;
575 while (true) {
576 const int old_decision_level = current_decision_level_;
577 if (!PropagateAndStopAfterOneConflictResolution()) {
578 if (model_is_unsat_) return false;
579 if (current_decision_level_ == old_decision_level) {
580 CHECK(!assumptions_.empty());
581 return false;
582 }
583 continue;
584 }
585 break;
586 }
587 CHECK(PropagationIsDone());
588 return true;
589}

599bool SatSolver::ResetWithGivenAssumptions(
600 const std::vector<Literal>& assumptions) {
601 if (!ResetToLevelZero()) return false;
602 if (assumptions.empty()) return true;
603
604 // For assumptions and core-based search, it is really important to add as
605 // many binary clauses as possible. This is because we do not wan to miss any
606 // early core of size 2.
607 ProcessNewlyFixedVariables();
608
609 DCHECK(assumptions_.empty());
610 assumption_level_ = 1;
611 assumptions_ = assumptions;
612 return ReapplyAssumptionsIfNeeded();
613}

615// Note that we do not count these as "branches" for a reporting purpose.
616bool SatSolver::ReapplyAssumptionsIfNeeded() {
617 if (model_is_unsat_) return false;
618 if (CurrentDecisionLevel() >= assumption_level_) return true;
619
620 if (CurrentDecisionLevel() == 0 && !assumptions_.empty()) {
621 // When assumptions_ is not empty, the first "decision" actually contains
622 // multiple one, and we should never use its literal.
623 CHECK_EQ(current_decision_level_, 0);
624 last_decision_or_backtrack_trail_index_ = trail_->Index();
625 decisions_[0] = Decision(trail_->Index(), Literal());
626
627 ++current_decision_level_;
628 trail_->SetDecisionLevel(current_decision_level_);
629
630 // We enqueue all assumptions at once at decision level 1.
631 int num_decisions = 0;
632 for (const Literal lit : assumptions_) {
633 if (Assignment().LiteralIsTrue(lit)) continue;
634 if (Assignment().LiteralIsFalse(lit)) {
635 // See GetLastIncompatibleDecisions().
636 *trail_->MutableConflict() = {lit.Negated(), lit};
637 if (num_decisions == 0) {
638 // This is needed to avoid an empty level that cause some CHECK fail.
639 current_decision_level_ = 0;
640 trail_->SetDecisionLevel(0);
641 }
642 return false;
643 }
644 ++num_decisions;
645 trail_->EnqueueSearchDecision(lit);
646 }
647
648 // Corner case: all assumptions are fixed at level zero, we ignore them.
649 if (num_decisions == 0) {
650 current_decision_level_ = 0;
651 trail_->SetDecisionLevel(0);
652 return ResetToLevelZero();
653 }
654
655 // Now that everything is enqueued, we propagate.
656 return FinishPropagation();
657 }
658
659 DCHECK(assumptions_.empty());
660 const int64_t old_num_branches = counters_.num_branches;
661 const SatSolver::Status status = ReapplyDecisionsUpTo(assumption_level_ - 1);
662 counters_.num_branches = old_num_branches;
663 assumption_level_ = CurrentDecisionLevel();
664 return (status == SatSolver::FEASIBLE);
665}

667bool SatSolver::PropagateAndStopAfterOneConflictResolution() {
668 SCOPED_TIME_STAT(&stats_);
669 if (Propagate()) return true;
670 if (model_is_unsat_) return false;
671
672 ++counters_.num_failures;
673 const int conflict_trail_index = trail_->Index();
674 const int conflict_decision_level = current_decision_level_;
675
676 // A conflict occurred, compute a nice reason for this failure.
677 same_reason_identifier_.Clear();
678 const int max_trail_index = ComputeMaxTrailIndex(trail_->FailingClause());
679 if (!assumptions_.empty() && !trail_->FailingClause().empty()) {
680 // If the failing clause only contains literal at the assumptions level,
681 // we cannot use the ComputeFirstUIPConflict() code as we might have more
682 // than one decision.
683 //
684 // TODO(user): We might still want to "learn" the clause, especially if
685 // it reduces to only one literal in which case we can just fix it.
686 const int highest_level =
687 DecisionLevel((*trail_)[max_trail_index].Variable());
688 if (highest_level == 1) return false;
689 }
690
691 ComputeFirstUIPConflict(max_trail_index, &learned_conflict_,
692 &reason_used_to_infer_the_conflict_,
693 &subsumed_clauses_);
694
695 // An empty conflict means that the problem is UNSAT.
696 if (learned_conflict_.empty()) return SetModelUnsat();
697 DCHECK(IsConflictValid(learned_conflict_));
698 DCHECK(ClauseIsValidUnderDebugAssignment(learned_conflict_));
699
700 // Update the activity of all the variables in the first UIP clause.
701 // Also update the activity of the last level variables expanded (and
702 // thus discarded) during the first UIP computation. Note that both
703 // sets are disjoint.
704 decision_policy_->BumpVariableActivities(learned_conflict_);
705 decision_policy_->BumpVariableActivities(reason_used_to_infer_the_conflict_);
706 if (parameters_->also_bump_variables_in_conflict_reasons()) {
707 ComputeUnionOfReasons(learned_conflict_, &extra_reason_literals_);
708 decision_policy_->BumpVariableActivities(extra_reason_literals_);
709 }
710
711 // Bump the clause activities.
712 // Note that the activity of the learned clause will be bumped too
713 // by AddLearnedClauseAndEnqueueUnitPropagation().
714 if (trail_->FailingSatClause() != nullptr) {
715 BumpClauseActivity(trail_->FailingSatClause());
716 }
717 BumpReasonActivities(reason_used_to_infer_the_conflict_);
718
719 // Decay the activities.
720 decision_policy_->UpdateVariableActivityIncrement();
721 UpdateClauseActivityIncrement();
722 pb_constraints_->UpdateActivityIncrement();
723
724 // Hack from Glucose that seems to perform well.
725 const int period = parameters_->glucose_decay_increment_period();
726 const double max_decay = parameters_->glucose_max_decay();
727 if (counters_.num_failures % period == 0 &&
728 parameters_->variable_activity_decay() < max_decay) {
729 parameters_->set_variable_activity_decay(
730 parameters_->variable_activity_decay() +
731 parameters_->glucose_decay_increment());
732 }
733
734 // PB resolution.
735 // There is no point using this if the conflict and all the reasons involved
736 // in its resolution were clauses.
737 bool compute_pb_conflict = false;
738 if (parameters_->use_pb_resolution()) {
739 compute_pb_conflict = (pb_constraints_->ConflictingConstraint() != nullptr);
740 if (!compute_pb_conflict) {
741 for (Literal lit : reason_used_to_infer_the_conflict_) {
742 if (ReasonPbConstraintOrNull(lit.Variable()) != nullptr) {
743 compute_pb_conflict = true;
744 break;
745 }
746 }
747 }
748 }
749
750 // TODO(user): Note that we use the clause above to update the variable
751 // activities and not the pb conflict. Experiment.
752 if (compute_pb_conflict) {
753 pb_conflict_.ClearAndResize(num_variables_.value());
754 Coefficient initial_slack(-1);
755 if (pb_constraints_->ConflictingConstraint() == nullptr) {
756 // Generic clause case.
757 Coefficient num_literals(0);
758 for (Literal literal : trail_->FailingClause()) {
759 pb_conflict_.AddTerm(literal.Negated(), Coefficient(1.0));
760 ++num_literals;
761 }
762 pb_conflict_.AddToRhs(num_literals - 1);
763 } else {
764 // We have a pseudo-Boolean conflict, so we start from there.
765 pb_constraints_->ConflictingConstraint()->AddToConflict(&pb_conflict_);
766 pb_constraints_->ClearConflictingConstraint();
767 initial_slack =
768 pb_conflict_.ComputeSlackForTrailPrefix(*trail_, max_trail_index + 1);
769 }
770
771 int pb_backjump_level;
772 ComputePBConflict(max_trail_index, initial_slack, &pb_conflict_,
773 &pb_backjump_level);
774 if (pb_backjump_level == -1) return SetModelUnsat();
775
776 // Convert the conflict into the vector<LiteralWithCoeff> form.
777 std::vector<LiteralWithCoeff> cst;
778 pb_conflict_.CopyIntoVector(&cst);
779 DCHECK(PBConstraintIsValidUnderDebugAssignment(cst, pb_conflict_.Rhs()));
780
781 // Check if the learned PB conflict is just a clause:
782 // all its coefficient must be 1, and the rhs must be its size minus 1.
783 bool conflict_is_a_clause = (pb_conflict_.Rhs() == cst.size() - 1);
784 if (conflict_is_a_clause) {
785 for (LiteralWithCoeff term : cst) {
786 if (term.coefficient != Coefficient(1)) {
787 conflict_is_a_clause = false;
788 break;
789 }
790 }
791 }
792
793 if (!conflict_is_a_clause) {
794 // Use the PB conflict.
795 DCHECK_GT(pb_constraints_->NumberOfConstraints(), 0);
796 CHECK_LT(pb_backjump_level, CurrentDecisionLevel());
797 Backtrack(pb_backjump_level);
798 CHECK(pb_constraints_->AddLearnedConstraint(cst, pb_conflict_.Rhs(),
799 trail_));
800 CHECK_GT(trail_->Index(), last_decision_or_backtrack_trail_index_);
801 counters_.num_learned_pb_literals += cst.size();
802 return false;
803 }
804
805 // Continue with the normal clause flow, but use the PB conflict clause
806 // if it has a lower backjump level.
807 if (pb_backjump_level < ComputeBacktrackLevel(learned_conflict_)) {
808 subsumed_clauses_.clear(); // Because the conflict changes.
809 learned_conflict_.clear();
810 is_marked_.ClearAndResize(num_variables_);
811 int max_level = 0;
812 int max_index = 0;
813 for (LiteralWithCoeff term : cst) {
814 DCHECK(Assignment().LiteralIsTrue(term.literal));
815 DCHECK_EQ(term.coefficient, 1);
816 const int level = trail_->Info(term.literal.Variable()).level;
817 if (level == 0) continue;
818 if (level > max_level) {
819 max_level = level;
820 max_index = learned_conflict_.size();
821 }
822 learned_conflict_.push_back(term.literal.Negated());
823
824 // The minimization functions below expect the conflict to be marked!
825 // TODO(user): This is error prone, find a better way?
826 is_marked_.Set(term.literal.Variable());
827 }
828 CHECK(!learned_conflict_.empty());
829 std::swap(learned_conflict_.front(), learned_conflict_[max_index]);
830 DCHECK(IsConflictValid(learned_conflict_));
831 }
832 }
833
834 // Minimizing the conflict with binary clauses first has two advantages.
835 // First, there is no need to compute a reason for the variables eliminated
836 // this way. Second, more variables may be marked (in is_marked_) and
837 // MinimizeConflict() can take advantage of that. Because of this, the
838 // LBD of the learned conflict can change.
839 DCHECK(ClauseIsValidUnderDebugAssignment(learned_conflict_));
840 if (!binary_implication_graph_->IsEmpty()) {
841 if (parameters_->binary_minimization_algorithm() ==
842 SatParameters::BINARY_MINIMIZATION_FIRST) {
843 binary_implication_graph_->MinimizeConflictFirst(
844 *trail_, &learned_conflict_, &is_marked_);
845 } else if (parameters_->binary_minimization_algorithm() ==
846 SatParameters::
847 BINARY_MINIMIZATION_FIRST_WITH_TRANSITIVE_REDUCTION) {
848 binary_implication_graph_->MinimizeConflictFirstWithTransitiveReduction(
849 *trail_, &learned_conflict_,
850 *model_->GetOrCreate<ModelRandomGenerator>());
851 }
852 DCHECK(IsConflictValid(learned_conflict_));
853 }
854
855 // Minimize the learned conflict.
856 MinimizeConflict(&learned_conflict_, &reason_used_to_infer_the_conflict_);
857
858 // Minimize it further with binary clauses?
859 if (!binary_implication_graph_->IsEmpty()) {
860 // Note that on the contrary to the MinimizeConflict() above that
861 // just uses the reason graph, this minimization can change the
862 // clause LBD and even the backtracking level.
863 switch (parameters_->binary_minimization_algorithm()) {
864 case SatParameters::NO_BINARY_MINIMIZATION:
865 ABSL_FALLTHROUGH_INTENDED;
866 case SatParameters::BINARY_MINIMIZATION_FIRST:
867 ABSL_FALLTHROUGH_INTENDED;
868 case SatParameters::BINARY_MINIMIZATION_FIRST_WITH_TRANSITIVE_REDUCTION:
869 break;
870 case SatParameters::BINARY_MINIMIZATION_WITH_REACHABILITY:
871 binary_implication_graph_->MinimizeConflictWithReachability(
872 &learned_conflict_);
873 break;
874 case SatParameters::EXPERIMENTAL_BINARY_MINIMIZATION:
875 binary_implication_graph_->MinimizeConflictExperimental(
876 *trail_, &learned_conflict_);
877 break;
878 }
879 DCHECK(IsConflictValid(learned_conflict_));
880 }
881
882 // We notify the decision before backtracking so that we can save the phase.
883 // The current heuristic is to try to take a trail prefix for which there is
884 // currently no conflict (hence just before the last decision was taken).
885 //
886 // TODO(user): It is unclear what the best heuristic is here. Both the current
887 // trail index or the trail before the current decision perform well, but
888 // using the full trail seems slightly better even though it will contain the
889 // current conflicting literal.
890 decision_policy_->BeforeConflict(trail_->Index());
891
892 // Backtrack and add the reason to the set of learned clause.
893 counters_.num_literals_learned += learned_conflict_.size();
894 Backtrack(ComputeBacktrackLevel(learned_conflict_));
895 DCHECK(ClauseIsValidUnderDebugAssignment(learned_conflict_));
896
897 // Note that we need to output the learned clause before cleaning the clause
898 // database. This is because we already backtracked and some of the clauses
899 // that were needed to infer the conflict may not be "reasons" anymore and
900 // may be deleted.
901 if (drat_proof_handler_ != nullptr) {
902 drat_proof_handler_->AddClause(learned_conflict_);
903 }
904
905 // Detach any subsumed clause. They will actually be deleted on the next
906 // clause cleanup phase.
907 bool is_redundant = true;
908 if (!subsumed_clauses_.empty() &&
909 parameters_->subsumption_during_conflict_analysis()) {
910 for (SatClause* clause : subsumed_clauses_) {
911 DCHECK(ClauseSubsumption(learned_conflict_, clause));
912 if (!clauses_propagator_->IsRemovable(clause)) {
913 is_redundant = false;
914 }
915 clauses_propagator_->LazyDetach(clause);
916 }
917 clauses_propagator_->CleanUpWatchers();
918 counters_.num_subsumed_clauses += subsumed_clauses_.size();
919 }
920
921 // Create and attach the new learned clause.
922 const int conflict_lbd = AddLearnedClauseAndEnqueueUnitPropagation(
923 learned_conflict_, is_redundant);
924 restart_->OnConflict(conflict_trail_index, conflict_decision_level,
925 conflict_lbd);
926 return false;
927}

929SatSolver::Status SatSolver::ReapplyDecisionsUpTo(
930 int max_level, int* first_propagation_index) {
931 SCOPED_TIME_STAT(&stats_);
932 DCHECK(assumptions_.empty());
933 int decision_index = current_decision_level_;
934 while (decision_index <= max_level) {
935 DCHECK_GE(decision_index, current_decision_level_);
936 const Literal previous_decision = decisions_[decision_index].literal;
937 ++decision_index;
938 if (Assignment().LiteralIsTrue(previous_decision)) {
939 // Note that this particular position in decisions_ will be overridden,
940 // but that is fine since this is a consequence of the previous decision,
941 // so we will never need to take it into account again.
942 continue;
943 }
944 if (Assignment().LiteralIsFalse(previous_decision)) {
945 // See GetLastIncompatibleDecisions().
946 *trail_->MutableConflict() = {previous_decision.Negated(),
947 previous_decision};
948 return ASSUMPTIONS_UNSAT;
949 }
950
951 // Not assigned, we try to take it.
952 const int old_level = current_decision_level_;
953 const int index = EnqueueDecisionAndBackjumpOnConflict(previous_decision);
954 if (first_propagation_index != nullptr) {
955 *first_propagation_index = std::min(*first_propagation_index, index);
956 }
957 if (index == kUnsatTrailIndex) return INFEASIBLE;
958 if (current_decision_level_ <= old_level) {
959 // A conflict occurred which backjumped to an earlier decision level.
960 // We potentially backjumped over some valid decisions, so we need to
961 // continue the loop and try to re-enqueue them.
962 //
963 // Note that there is no need to update max_level, because when we will
964 // try to reapply the current "previous_decision" it will result in a
965 // conflict. IMPORTANT: we can't actually optimize this and abort the loop
966 // earlier though, because we need to check that it is conflicting because
967 // it is already propagated to false. There is no guarantee of this
968 // because we learn the first-UIP conflict. If it is not the case, we will
969 // then learn a new conflict, backjump, and continue the loop.
970 decision_index = current_decision_level_;
971 }
972 }
973 return FEASIBLE;
974}

978 SCOPED_TIME_STAT(&stats_);

982 DCHECK_LT(CurrentDecisionLevel(), decisions_.size());
983 decisions_[CurrentDecisionLevel()].literal = true_literal;
984 if (first_propagation_index != nullptr) {
985 *first_propagation_index = trail_->Index();
986 }
987 return ReapplyDecisionsUpTo(CurrentDecisionLevel(), first_propagation_index);

1005void SatSolver::Backtrack(int target_level) {
1006 SCOPED_TIME_STAT(&stats_);
1007 // TODO(user): The backtrack method should not be called when the model is
1008 // unsat. Add a DCHECK to prevent that, but before fix the
1009 // bop::BopOptimizerBase architecture.
1010
1011 // Do nothing if the CurrentDecisionLevel() is already correct.
1012 // This is needed, otherwise target_trail_index below will remain at zero and
1013 // that will cause some problems. Note that we could forbid a user to call
1014 // Backtrack() with the current level, but that is annoying when you just
1015 // want to reset the solver with Backtrack(0).
1016 if (CurrentDecisionLevel() == target_level) return;
1017 DCHECK_GE(target_level, 0);
1018 DCHECK_LE(target_level, CurrentDecisionLevel());
1019
1020 // Any backtrack to the root from a positive one is counted as a restart.
1021 if (target_level == 0) counters_.num_restarts++;
1022
1023 // Per the SatPropagator interface, this is needed before calling Untrail.
1024 trail_->SetDecisionLevel(target_level);
1025
1026 current_decision_level_ = target_level;
1027 const int target_trail_index =
1028 decisions_[current_decision_level_].trail_index;
1029
1030 Untrail(target_trail_index);
1031 last_decision_or_backtrack_trail_index_ = trail_->Index();
1032}

1096void SatSolver::KeepAllClauseUsedToInfer(BooleanVariable variable) {
1097 CHECK(Assignment().VariableIsAssigned(variable));
1098 if (trail_->Info(variable).level == 0) return;
1099 int trail_index = trail_->Info(variable).trail_index;
1100 std::vector<bool> is_marked(trail_index + 1, false); // move to local member.
1101 is_marked[trail_index] = true;
1102 int num = 1;
1103 for (; num > 0 && trail_index >= 0; --trail_index) {
1104 if (!is_marked[trail_index]) continue;
1105 is_marked[trail_index] = false;
1106 --num;
1107
1108 const BooleanVariable var = (*trail_)[trail_index].Variable();
1109 SatClause* clause = ReasonClauseOrNull(var);
1110 if (clause != nullptr) {
1111 clauses_propagator_->mutable_clauses_info()->erase(clause);
1112 }
1113 for (const Literal l : trail_->Reason(var)) {
1114 const AssignmentInfo& info = trail_->Info(l.Variable());
1115 if (info.level == 0) continue;
1116 if (!is_marked[info.trail_index]) {
1117 is_marked[info.trail_index] = true;
1118 ++num;
1119 }
1120 }
1121 }
1122}

1124// TODO(user): this is really an in-processing stuff and should be moved out
1125// of here. I think the name for that (or similar) technique is called vivify.
1126// Ideally this should be scheduled after other faster in-processing technique.
1127void SatSolver::TryToMinimizeClause(SatClause* clause) {
1128 CHECK_EQ(CurrentDecisionLevel(), 0);
1129 ++counters_.minimization_num_clauses;
1130
1131 absl::btree_set<LiteralIndex> moved_last;
1132 std::vector<Literal> candidate(clause->begin(), clause->end());
1133 while (!model_is_unsat_) {
1134 // We want each literal in candidate to appear last once in our propagation
1135 // order. We want to do that while maximizing the reutilization of the
1136 // current assignment prefix, that is minimizing the number of
1137 // decision/progagation we need to perform.
1138 const int target_level = MoveOneUnprocessedLiteralLast(
1139 moved_last, CurrentDecisionLevel(), &candidate);
1140 if (target_level == -1) break;
1141 Backtrack(target_level);
1142 while (CurrentDecisionLevel() < candidate.size()) {
1143 const int level = CurrentDecisionLevel();
1144 const Literal literal = candidate[level];
1145 if (Assignment().LiteralIsFalse(literal)) {
1146 candidate.erase(candidate.begin() + level);
1147 continue;
1148 } else if (Assignment().LiteralIsTrue(literal)) {
1149 const int variable_level =
1150 LiteralTrail().Info(literal.Variable()).level;
1151 if (variable_level == 0) {
1152 ProcessNewlyFixedVariablesForDratProof();
1153 counters_.minimization_num_true++;
1154 counters_.minimization_num_removed_literals += clause->size();
1155 Backtrack(0);
1156 clauses_propagator_->Detach(clause);
1157 return;
1158 }
1159
1160 // If literal (at true) wasn't propagated by this clause, then we
1161 // know that this clause is subsumed by other clauses in the database,
1162 // so we can remove it. Note however that we need to make sure we will
1163 // never remove the clauses that subsumes it later.
1164 if (ReasonClauseOrNull(literal.Variable()) != clause) {
1165 counters_.minimization_num_subsumed++;
1166 counters_.minimization_num_removed_literals += clause->size();
1167
1168 // TODO(user): do not do that if it make us keep too many clauses?
1169 KeepAllClauseUsedToInfer(literal.Variable());
1170 Backtrack(0);
1171 clauses_propagator_->Detach(clause);
1172 return;
1173 } else {
1174 // Simplify. Note(user): we could only keep in clause the literals
1175 // responsible for the propagation, but because of the subsumption
1176 // above, this is not needed.
1177 if (variable_level + 1 < candidate.size()) {
1178 candidate.resize(variable_level);
1179 candidate.push_back(literal);
1180 }
1181 }
1182 break;
1183 } else {
1184 ++counters_.minimization_num_decisions;
1185 EnqueueDecisionAndBackjumpOnConflict(literal.Negated());
1186 if (!clause->IsAttached()) {
1187 Backtrack(0);
1188 return;
1189 }
1190 if (model_is_unsat_) return;
1191 }
1192 }
1193 if (candidate.empty()) {
1194 model_is_unsat_ = true;
1195 return;
1196 }
1197 moved_last.insert(candidate.back().Index());
1198 }
1199
1200 // Returns if we don't have any minimization.
1201 Backtrack(0);
1202 if (candidate.size() == clause->size()) return;
1203
1204 if (candidate.size() == 1) {
1205 if (drat_proof_handler_ != nullptr) {
1206 drat_proof_handler_->AddClause(candidate);
1207 }
1208 if (!Assignment().VariableIsAssigned(candidate[0].Variable())) {
1209 counters_.minimization_num_removed_literals += clause->size();
1210 trail_->EnqueueWithUnitReason(candidate[0]);
1211 FinishPropagation();
1212 }
1213 return;
1214 }
1215
1216 if (candidate.size() == 2) {
1217 counters_.minimization_num_removed_literals += clause->size() - 2;
1218
1219 // The order is important for the drat proof.
1220 AddBinaryClauseInternal(candidate[0], candidate[1], /*export_clause=*/true);
1221 clauses_propagator_->Detach(clause);
1222
1223 // This is needed in the corner case where this was the first binary clause
1224 // of the problem so that PropagationIsDone() returns true on the newly
1225 // created BinaryImplicationGraph.
1226 FinishPropagation();
1227 return;
1228 }
1229
1230 counters_.minimization_num_removed_literals +=
1231 clause->size() - candidate.size();
1232
1233 // TODO(user): If the watched literal didn't change, we could just rewrite
1234 // the clause while keeping the two watched literals at the beginning.
1235 if (!clauses_propagator_->InprocessingRewriteClause(clause, candidate)) {
1236 model_is_unsat_ = true;
1237 }
1238}

1240SatSolver::Status SatSolver::SolveInternal(TimeLimit* time_limit,
1241 int64_t max_number_of_conflicts) {
1242 SCOPED_TIME_STAT(&stats_);
1243 if (model_is_unsat_) return INFEASIBLE;
1244
1245 // TODO(user): Because the counter are not reset to zero, this cause the
1246 // metrics / sec to be completely broken except when the solver is used
1247 // for exactly one Solve().
1248 timer_.Restart();
1249
1250 // Display initial statistics.
1251 if (logger_->LoggingIsEnabled()) {
1252 SOLVER_LOG(logger_, "Initial memory usage: ", MemoryUsage());
1253 SOLVER_LOG(logger_, "Number of variables: ", num_variables_.value());
1254 SOLVER_LOG(logger_, "Number of clauses (size > 2): ",
1255 clauses_propagator_->num_clauses());
1256 SOLVER_LOG(logger_, "Number of binary clauses: ",
1257 binary_implication_graph_->num_implications());
1258 SOLVER_LOG(logger_, "Number of linear constraints: ",
1259 pb_constraints_->NumberOfConstraints());
1260 SOLVER_LOG(logger_, "Number of fixed variables: ", trail_->Index());
1261 SOLVER_LOG(logger_, "Number of watched clauses: ",
1262 clauses_propagator_->num_watched_clauses());
1263 SOLVER_LOG(logger_, "Parameters: ", ProtobufShortDebugString(*parameters_));
1264 }
1265
1266 // Used to trigger clause minimization via propagation.
1267 int64_t next_minimization_num_restart =
1268 restart_->NumRestarts() +
1269 parameters_->minimize_with_propagation_restart_period();
1270
1271 // Variables used to show the search progress.
1272 const int64_t kDisplayFrequency = 10000;
1273 int64_t next_display = parameters_->log_search_progress()
1274 ? NextMultipleOf(num_failures(), kDisplayFrequency)
1275 : std::numeric_limits<int64_t>::max();
1276
1277 // Variables used to check the memory limit every kMemoryCheckFrequency.
1278 const int64_t kMemoryCheckFrequency = 10000;
1279 int64_t next_memory_check =
1280 NextMultipleOf(num_failures(), kMemoryCheckFrequency);
1281
1282 // The max_number_of_conflicts is per solve but the counter is for the whole
1283 // solver.
1284 const int64_t kFailureLimit =
1285 max_number_of_conflicts == std::numeric_limits<int64_t>::max()
1286 ? std::numeric_limits<int64_t>::max()
1287 : counters_.num_failures + max_number_of_conflicts;
1288
1289 // Starts search.
1290 for (;;) {
1291 // Test if a limit is reached.
1292 if (time_limit != nullptr) {
1293 AdvanceDeterministicTime(time_limit);
1294 if (time_limit->LimitReached()) {
1295 SOLVER_LOG(logger_, "The time limit has been reached. Aborting.");
1296 return StatusWithLog(LIMIT_REACHED);
1297 }
1298 }
1299 if (num_failures() >= kFailureLimit) {
1300 SOLVER_LOG(logger_, "The conflict limit has been reached. Aborting.");
1301 return StatusWithLog(LIMIT_REACHED);
1302 }
1303
1304 // The current memory checking takes time, so we only execute it every
1305 // kMemoryCheckFrequency conflict. We use >= because counters_.num_failures
1306 // may augment by more than one at each iteration.
1307 //
1308 // TODO(user): Find a better way.
1309 if (counters_.num_failures >= next_memory_check) {
1310 next_memory_check = NextMultipleOf(num_failures(), kMemoryCheckFrequency);
1311 if (IsMemoryLimitReached()) {
1312 SOLVER_LOG(logger_, "The memory limit has been reached. Aborting.");
1313 return StatusWithLog(LIMIT_REACHED);
1314 }
1315 }
1316
1317 // Display search progression. We use >= because counters_.num_failures may
1318 // augment by more than one at each iteration.
1319 if (counters_.num_failures >= next_display) {
1320 SOLVER_LOG(logger_, RunningStatisticsString());
1321 next_display = NextMultipleOf(num_failures(), kDisplayFrequency);
1322 }
1323
1324 const int old_level = current_decision_level_;
1325 if (!PropagateAndStopAfterOneConflictResolution()) {
1326 // A conflict occurred, continue the loop.
1327 if (model_is_unsat_) return StatusWithLog(INFEASIBLE);
1328 if (old_level == current_decision_level_) {
1329 CHECK(!assumptions_.empty());
1330 return StatusWithLog(ASSUMPTIONS_UNSAT);
1331 }
1332 } else {
1333 // We need to reapply any assumptions that are not currently applied.
1334 if (!ReapplyAssumptionsIfNeeded()) return StatusWithLog(UnsatStatus());
1335
1336 // At a leaf?
1337 if (trail_->Index() == num_variables_.value()) {
1338 return StatusWithLog(FEASIBLE);
1339 }
1340
1341 if (restart_->ShouldRestart()) {
1342 Backtrack(assumption_level_);
1343 }
1344
1345 // Clause minimization using propagation.
1346 if (CurrentDecisionLevel() == 0 &&
1347 restart_->NumRestarts() >= next_minimization_num_restart) {
1348 next_minimization_num_restart =
1349 restart_->NumRestarts() +
1350 parameters_->minimize_with_propagation_restart_period();
1351 MinimizeSomeClauses(
1352 parameters_->minimize_with_propagation_num_decisions());
1353
1354 // Corner case: the minimization above being based on propagation may
1355 // fix the remaining variables or prove UNSAT.
1356 if (model_is_unsat_) return StatusWithLog(INFEASIBLE);
1357 if (trail_->Index() == num_variables_.value()) {
1358 return StatusWithLog(FEASIBLE);
1359 }
1360 }
1361
1362 DCHECK_GE(CurrentDecisionLevel(), assumption_level_);
1363 EnqueueNewDecision(decision_policy_->NextBranch());
1364 }
1365 }
1366}

1368void SatSolver::MinimizeSomeClauses(int decisions_budget) {
1369 // Tricky: we don't want TryToMinimizeClause() to delete to_minimize
1370 // while we are processing it.
1371 block_clause_deletion_ = true;
1372
1373 const int64_t target_num_branches = counters_.num_branches + decisions_budget;
1374 while (counters_.num_branches < target_num_branches &&
1375 (time_limit_ == nullptr || !time_limit_->LimitReached())) {
1376 SatClause* to_minimize = clauses_propagator_->NextClauseToMinimize();
1377 if (to_minimize != nullptr) {
1378 TryToMinimizeClause(to_minimize);
1379 if (model_is_unsat_) return;
1380 } else {
1381 if (to_minimize == nullptr) {
1382 VLOG(1) << "Minimized all clauses, restarting from first one.";
1383 clauses_propagator_->ResetToMinimizeIndex();
1384 }
1385 break;
1386 }
1387 }
1388
1389 block_clause_deletion_ = false;
1390 clauses_propagator_->DeleteRemovedClauses();
1391}

1393std::vector<Literal> SatSolver::GetLastIncompatibleDecisions() {
1394 SCOPED_TIME_STAT(&stats_);
1395 std::vector<Literal> unsat_assumptions;
1396
1397 is_marked_.ClearAndResize(num_variables_);
1398
1399 int trail_index = 0;
1400 int num_true = 0;
1401 for (const Literal lit : trail_->FailingClause()) {
1402 CHECK(Assignment().LiteralIsAssigned(lit));
1403 if (Assignment().LiteralIsTrue(lit)) {
1404 // literal at true in the conflict must be decision/assumptions that could
1405 // not be taken.
1406 ++num_true;
1407 unsat_assumptions.push_back(lit.Negated());
1408 continue;
1409 }
1410 trail_index =
1411 std::max(trail_index, trail_->Info(lit.Variable()).trail_index);
1412 is_marked_.Set(lit.Variable());
1413 }
1414 CHECK_LE(num_true, 1);
1415
1416 // We just expand the conflict until we only have decisions.
1417 const int limit =
1418 CurrentDecisionLevel() > 0 ? decisions_[0].trail_index : trail_->Index();
1419 CHECK_LT(trail_index, trail_->Index());
1420 while (true) {
1421 // Find next marked literal to expand from the trail.
1422 while (trail_index >= limit &&
1423 !is_marked_[(*trail_)[trail_index].Variable()]) {
1424 --trail_index;
1425 }
1426 if (trail_index < limit) break;
1427 const Literal marked_literal = (*trail_)[trail_index];
1428 --trail_index;
1429
1430 if (trail_->AssignmentType(marked_literal.Variable()) ==
1431 AssignmentType::kSearchDecision) {
1432 unsat_assumptions.push_back(marked_literal);
1433 } else {
1434 // Marks all the literals of its reason.
1435 for (const Literal literal : trail_->Reason(marked_literal.Variable())) {
1436 const BooleanVariable var = literal.Variable();
1437 const int level = DecisionLevel(var);
1438 if (level > 0 && !is_marked_[var]) is_marked_.Set(var);
1439 }
1440 }
1441 }
1442
1443 // We reverse the assumptions so they are in the same order as the one in
1444 // which the decision were made.
1445 std::reverse(unsat_assumptions.begin(), unsat_assumptions.end());
1446 return unsat_assumptions;
1447}

1470void SatSolver::BumpClauseActivity(SatClause* clause) {
1471 // We only bump the activity of the clauses that have some info. So if we know
1472 // that we will keep a clause forever, we don't need to create its Info. More
1473 // than the speed, this allows to limit as much as possible the activity
1474 // rescaling.
1475 auto it = clauses_propagator_->mutable_clauses_info()->find(clause);
1476 if (it == clauses_propagator_->mutable_clauses_info()->end()) return;
1477
1478 // Check if the new clause LBD is below our threshold to keep this clause
1479 // indefinitely. Note that we use a +1 here because the LBD of a newly learned
1480 // clause decrease by 1 just after the backjump.
1481 const int new_lbd = ComputeLbd(*clause);
1482 if (new_lbd + 1 <= parameters_->clause_cleanup_lbd_bound()) {
1483 clauses_propagator_->mutable_clauses_info()->erase(clause);
1484 return;
1485 }
1486
1487 // Eventually protect this clause for the next cleanup phase.
1488 switch (parameters_->clause_cleanup_protection()) {
1489 case SatParameters::PROTECTION_NONE:
1490 break;
1491 case SatParameters::PROTECTION_ALWAYS:
1492 it->second.protected_during_next_cleanup = true;
1493 break;
1494 case SatParameters::PROTECTION_LBD:
1495 // This one is similar to the one used by the Glucose SAT solver.
1496 //
1497 // TODO(user): why the +1? one reason may be that the LBD of a conflict
1498 // decrease by 1 just after the backjump...
1499 if (new_lbd + 1 < it->second.lbd) {
1500 it->second.protected_during_next_cleanup = true;
1501 it->second.lbd = new_lbd;
1502 }
1503 }
1504
1505 // Increase the activity.
1506 const double activity = it->second.activity += clause_activity_increment_;
1507 if (activity > parameters_->max_clause_activity_value()) {
1508 RescaleClauseActivities(1.0 / parameters_->max_clause_activity_value());
1509 }
1510}

1512void SatSolver::RescaleClauseActivities(double scaling_factor) {
1513 SCOPED_TIME_STAT(&stats_);
1514 clause_activity_increment_ *= scaling_factor;
1515 for (auto& entry : *clauses_propagator_->mutable_clauses_info()) {
1516 entry.second.activity *= scaling_factor;
1517 }
1518}

1522 clause_activity_increment_ *= 1.0 / parameters_->clause_activity_decay();

1536int SatSolver::ComputeBacktrackLevel(const std::vector<Literal>& literals) {
1537 SCOPED_TIME_STAT(&stats_);
1538 DCHECK_GT(CurrentDecisionLevel(), 0);
1539
1540 // We want the highest decision level among literals other than the first one.
1541 // Note that this level will always be smaller than that of the first literal.
1542 //
1543 // Note(user): if the learned clause is of size 1, we backtrack all the way to
1544 // the beginning. It may be possible to follow another behavior, but then the
1545 // code require some special cases in
1546 // AddLearnedClauseAndEnqueueUnitPropagation() to fix the literal and not
1547 // backtrack over it. Also, subsequent propagated variables may not have a
1548 // correct level in this case.
1549 int backtrack_level = 0;
1550 for (int i = 1; i < literals.size(); ++i) {
1551 const int level = DecisionLevel(literals[i].Variable());
1552 backtrack_level = std::max(backtrack_level, level);
1553 }
1554 DCHECK_LT(backtrack_level, DecisionLevel(literals[0].Variable()));
1555 DCHECK_LE(DecisionLevel(literals[0].Variable()), CurrentDecisionLevel());
1556 return backtrack_level;
1557}

1559template <typename LiteralList>
1560int SatSolver::ComputeLbd(const LiteralList& literals) {
1561 SCOPED_TIME_STAT(&stats_);
1562 const int limit =
1563 parameters_->count_assumption_levels_in_lbd() ? 0 : assumption_level_;
1564
1565 // We know that the first literal is always of the highest level.
1566 is_level_marked_.ClearAndResize(
1567 SatDecisionLevel(DecisionLevel(literals.begin()->Variable()) + 1));
1568 for (const Literal literal : literals) {
1569 const SatDecisionLevel level(DecisionLevel(literal.Variable()));
1570 DCHECK_GE(level, 0);
1571 if (level > limit && !is_level_marked_[level]) {
1572 is_level_marked_.Set(level);
1573 }
1574 }
1575 return is_level_marked_.NumberOfSetCallsWithDifferentArguments();
1576}

1655void SatSolver::ProcessNewlyFixedVariablesForDratProof() {
1656 if (drat_proof_handler_ == nullptr) return;
1657 if (CurrentDecisionLevel() != 0) return;
1658
1659 // We need to output the literals that are fixed so we can remove all
1660 // clauses that contains them. Note that this doesn't seems to be needed
1661 // for drat-trim.
1662 //
1663 // TODO(user): Ideally we could output such literal as soon as they are fixed,
1664 // but this is not that easy to do. Spend some time to find a cleaner
1665 // alternative? Currently this works, but:
1666 // - We will output some fixed literals twice since we already output learnt
1667 // clauses of size one.
1668 // - We need to call this function when needed.
1669 Literal temp;
1670 for (; drat_num_processed_fixed_variables_ < trail_->Index();
1671 ++drat_num_processed_fixed_variables_) {
1672 temp = (*trail_)[drat_num_processed_fixed_variables_];
1673 drat_proof_handler_->AddClause({&temp, 1});
1674 }
1675}

1677void SatSolver::ProcessNewlyFixedVariables() {
1678 SCOPED_TIME_STAT(&stats_);
1679 DCHECK_EQ(CurrentDecisionLevel(), 0);
1680 int num_detached_clauses = 0;
1681 int num_binary = 0;
1682
1683 ProcessNewlyFixedVariablesForDratProof();
1684
1685 // We remove the clauses that are always true and the fixed literals from the
1686 // others. Note that none of the clause should be all false because we should
1687 // have detected a conflict before this is called.
1688 for (SatClause* clause : clauses_propagator_->AllClausesInCreationOrder()) {
1689 if (!clause->IsAttached()) continue;
1690
1691 const size_t old_size = clause->size();
1692 if (clause->RemoveFixedLiteralsAndTestIfTrue(trail_->Assignment())) {
1693 // The clause is always true, detach it.
1694 clauses_propagator_->LazyDetach(clause);
1695 ++num_detached_clauses;
1696 continue;
1697 }
1698
1699 const size_t new_size = clause->size();
1700 if (new_size == old_size) continue;
1701
1702 if (drat_proof_handler_ != nullptr) {
1703 CHECK_GT(new_size, 0);
1704 drat_proof_handler_->AddClause({clause->begin(), new_size});
1705 drat_proof_handler_->DeleteClause({clause->begin(), old_size});
1706 }
1707
1708 if (new_size == 2) {
1709 // This clause is now a binary clause, treat it separately. Note that
1710 // it is safe to do that because this clause can't be used as a reason
1711 // since we are at level zero and the clause is not satisfied.
1712 AddBinaryClauseInternal(clause->FirstLiteral(), clause->SecondLiteral(),
1713 /*export_clause=*/true);
1714 clauses_propagator_->LazyDetach(clause);
1715 ++num_binary;
1716 continue;
1717 }
1718 }
1719
1720 // Note that we will only delete the clauses during the next database cleanup.
1721 clauses_propagator_->CleanUpWatchers();
1722 if (num_detached_clauses > 0 || num_binary > 0) {
1723 VLOG(1) << trail_->Index() << " fixed variables at level 0. "
1724 << "Detached " << num_detached_clauses << " clauses. " << num_binary
1725 << " converted to binary.";
1726 }
1727
1728 // We also clean the binary implication graph.
1729 // Tricky: If we added the first binary clauses above, the binary graph
1730 // is not in "propagated" state as it should be, so we call Propagate() so
1731 // all the checks are happy.
1732 CHECK(binary_implication_graph_->Propagate(trail_));
1733 binary_implication_graph_->RemoveFixedVariables();
1734 num_processed_fixed_variables_ = trail_->Index();
1735 deterministic_time_of_last_fixed_variables_cleanup_ = deterministic_time();
1736}

1738bool SatSolver::PropagationIsDone() const {
1739 for (SatPropagator* propagator : propagators_) {
1740 if (propagator->IsEmpty()) continue;
1741 if (!propagator->PropagationIsDone(*trail_)) return false;
1742 }
1743 return true;
1744}
1745
1746// TODO(user): Support propagating only the "first" propagators. That can
1747// be useful for probing/in-processing, so we can control if we do only the SAT
1748// part or the full integer part...
1749bool SatSolver::Propagate() {
1750 SCOPED_TIME_STAT(&stats_);
1751
1752 // Because we might potentially iterate often on this list below, we remove
1753 // empty propagators.
1754 //
1755 // TODO(user): This might not really be needed.
1756 non_empty_propagators_.clear();
1757 for (SatPropagator* propagator : propagators_) {
1758 if (!propagator->IsEmpty()) {
1759 non_empty_propagators_.push_back(propagator);
1760 }
1761 }
1762
1763 while (true) {
1764 // The idea here is to abort the inspection as soon as at least one
1765 // propagation occurs so we can loop over and test again the highest
1766 // priority constraint types using the new information.
1767 //
1768 // Note that the first propagators_ should be the binary_implication_graph_
1769 // and that its Propagate() functions will not abort on the first
1770 // propagation to be slightly more efficient.
1771 const int old_index = trail_->Index();
1772 for (SatPropagator* propagator : non_empty_propagators_) {
1773 DCHECK(propagator->PropagatePreconditionsAreSatisfied(*trail_));
1774 if (!propagator->Propagate(trail_)) return false;
1775 if (trail_->Index() > old_index) break;
1776 }
1777 if (trail_->Index() == old_index) break;
1778 }
1779 return true;
1780}

1795bool SatSolver::ResolvePBConflict(BooleanVariable var,
1796 MutableUpperBoundedLinearConstraint* conflict,
1797 Coefficient* slack) {
1798 const int trail_index = trail_->Info(var).trail_index;
1799
1800 // This is the slack of the conflict < trail_index
1801 DCHECK_EQ(*slack, conflict->ComputeSlackForTrailPrefix(*trail_, trail_index));
1802
1803 // Pseudo-Boolean case.
1804 UpperBoundedLinearConstraint* pb_reason = ReasonPbConstraintOrNull(var);
1805 if (pb_reason != nullptr) {
1806 pb_reason->ResolvePBConflict(*trail_, var, conflict, slack);
1807 return false;
1808 }
1809
1810 // Generic clause case.
1811 Coefficient multiplier(1);
1812
1813 // TODO(user): experiment and choose the "best" algo.
1814 const int algorithm = 1;
1815 switch (algorithm) {
1816 case 1:
1817 // We reduce the conflict slack to 0 before adding the clause.
1818 // The advantage of this method is that the coefficients stay small.
1819 conflict->ReduceSlackTo(*trail_, trail_index, *slack, Coefficient(0));
1820 break;
1821 case 2:
1822 // No reduction, we add the lower possible multiple.
1823 multiplier = *slack + 1;
1824 break;
1825 default:
1826 // No reduction, the multiple is chosen to cancel var.
1827 multiplier = conflict->GetCoefficient(var);
1828 }
1829
1830 Coefficient num_literals(1);
1831 conflict->AddTerm(
1832 trail_->Assignment().GetTrueLiteralForAssignedVariable(var).Negated(),
1833 multiplier);
1834 for (Literal literal : trail_->Reason(var)) {
1835 DCHECK_NE(literal.Variable(), var);
1836 DCHECK(Assignment().LiteralIsFalse(literal));
1837 conflict->AddTerm(literal.Negated(), multiplier);
1838 ++num_literals;
1839 }
1840 conflict->AddToRhs((num_literals - 1) * multiplier);
1841
1842 // All the algorithms above result in a new slack of -1.
1843 *slack = -1;

1913// This method will compute a first UIP conflict
1914// http://www.cs.tau.ac.il/~msagiv/courses/ATP/iccad2001_final.pdf
1915// http://gauss.ececs.uc.edu/SAT/articles/FAIA185-0131.pdf
1916void SatSolver::ComputeFirstUIPConflict(
1917 int max_trail_index, std::vector<Literal>* conflict,
1918 std::vector<Literal>* reason_used_to_infer_the_conflict,
1919 std::vector<SatClause*>* subsumed_clauses) {
1920 SCOPED_TIME_STAT(&stats_);
1921
1922 // This will be used to mark all the literals inspected while we process the
1923 // conflict and the reasons behind each of its variable assignments.
1924 is_marked_.ClearAndResize(num_variables_);
1925
1926 conflict->clear();
1927 reason_used_to_infer_the_conflict->clear();
1928 subsumed_clauses->clear();
1929 if (max_trail_index == -1) return;
1930
1931 // max_trail_index is the maximum trail index appearing in the failing_clause
1932 // and its level (Which is almost always equals to the CurrentDecisionLevel(),
1933 // except for symmetry propagation).
1934 DCHECK_EQ(max_trail_index, ComputeMaxTrailIndex(trail_->FailingClause()));
1935 int trail_index = max_trail_index;
1936 const int highest_level = DecisionLevel((*trail_)[trail_index].Variable());
1937 if (highest_level == 0) return;
1938
1939 // To find the 1-UIP conflict clause, we start by the failing_clause, and
1940 // expand each of its literal using the reason for this literal assignment to
1941 // false. The is_marked_ set allow us to never expand the same literal twice.
1942 //
1943 // The expansion is not done (i.e. stop) for literals that were assigned at a
1944 // decision level below the current one. If the level of such literal is not
1945 // zero, it is added to the conflict clause.
1946 //
1947 // Now, the trick is that we use the trail to expand the literal of the
1948 // current level in a very specific order. Namely the reverse order of the one
1949 // in which they were inferred. We stop as soon as
1950 // num_literal_at_highest_level_that_needs_to_be_processed is exactly one.
1951 //
1952 // This last literal will be the first UIP because by definition all the
1953 // propagation done at the current level will pass though it at some point.
1954 absl::Span<const Literal> clause_to_expand = trail_->FailingClause();
1955 SatClause* sat_clause = trail_->FailingSatClause();
1956 DCHECK(!clause_to_expand.empty());
1957 int num_literal_at_highest_level_that_needs_to_be_processed = 0;
1958 while (true) {
1959 int num_new_vars_at_positive_level = 0;
1960 int num_vars_at_positive_level_in_clause_to_expand = 0;
1961 for (const Literal literal : clause_to_expand) {
1962 const BooleanVariable var = literal.Variable();
1963 const int level = DecisionLevel(var);
1964 if (level > 0) ++num_vars_at_positive_level_in_clause_to_expand;
1965 if (!is_marked_[var]) {
1966 is_marked_.Set(var);
1967 if (level == highest_level) {
1968 ++num_new_vars_at_positive_level;
1969 ++num_literal_at_highest_level_that_needs_to_be_processed;
1970 } else if (level > 0) {
1971 ++num_new_vars_at_positive_level;
1972 // Note that all these literals are currently false since the clause
1973 // to expand was used to infer the value of a literal at this level.
1974 DCHECK(trail_->Assignment().LiteralIsFalse(literal));
1975 conflict->push_back(literal);
1976 } else {
1977 reason_used_to_infer_the_conflict->push_back(literal);
1978 }
1979 }
1980 }
1981
1982 // If there is new variables, then all the previously subsumed clauses are
1983 // not subsumed anymore.
1984 if (num_new_vars_at_positive_level > 0) {
1985 // TODO(user): We could still replace all these clauses with the current
1986 // conflict.
1987 subsumed_clauses->clear();
1988 }
1989
1990 // This check if the new conflict is exactly equal to clause_to_expand.
1991 // Since we just performed an union, comparing the size is enough. When this
1992 // is true, then the current conflict subsumes the reason whose underlying
1993 // clause is given by sat_clause.
1994 if (sat_clause != nullptr &&
1995 num_vars_at_positive_level_in_clause_to_expand ==
1996 conflict->size() +
1997 num_literal_at_highest_level_that_needs_to_be_processed) {
1998 subsumed_clauses->push_back(sat_clause);
1999 }
2000
2001 // Find next marked literal to expand from the trail.
2002 DCHECK_GT(num_literal_at_highest_level_that_needs_to_be_processed, 0);
2003 while (!is_marked_[(*trail_)[trail_index].Variable()]) {
2004 --trail_index;
2005 DCHECK_GE(trail_index, 0);
2006 DCHECK_EQ(DecisionLevel((*trail_)[trail_index].Variable()),
2007 highest_level);
2008 }
2009
2010 if (num_literal_at_highest_level_that_needs_to_be_processed == 1) {
2011 // We have the first UIP. Add its negation to the conflict clause.
2012 // This way, after backtracking to the proper level, the conflict clause
2013 // will be unit, and infer the negation of the UIP that caused the fail.
2014 conflict->push_back((*trail_)[trail_index].Negated());
2015
2016 // To respect the function API move the first UIP in the first position.
2017 std::swap(conflict->back(), conflict->front());
2018 break;
2019 }
2020
2021 const Literal literal = (*trail_)[trail_index];
2022 reason_used_to_infer_the_conflict->push_back(literal);
2023
2024 // If we already encountered the same reason, we can just skip this literal
2025 // which is what setting clause_to_expand to the empty clause do.
2026 if (same_reason_identifier_.FirstVariableWithSameReason(
2027 literal.Variable()) != literal.Variable()) {
2028 clause_to_expand = {};
2029 } else {
2030 clause_to_expand = trail_->Reason(literal.Variable());
2031 }
2032 sat_clause = ReasonClauseOrNull(literal.Variable());
2033
2034 --num_literal_at_highest_level_that_needs_to_be_processed;
2035 --trail_index;
2036 }
2037}

2056// TODO(user): Remove the literals assigned at level 0.
2057void SatSolver::ComputePBConflict(int max_trail_index,
2058 Coefficient initial_slack,
2059 MutableUpperBoundedLinearConstraint* conflict,
2060 int* pb_backjump_level) {
2061 SCOPED_TIME_STAT(&stats_);
2062 int trail_index = max_trail_index;
2063
2064 // First compute the slack of the current conflict for the assignment up to
2065 // trail_index. It must be negative since this is a conflict.
2066 Coefficient slack = initial_slack;
2067 DCHECK_EQ(slack,
2068 conflict->ComputeSlackForTrailPrefix(*trail_, trail_index + 1));
2069 CHECK_LT(slack, 0) << "We don't have a conflict!";
2070
2071 // Iterate backward over the trail.
2072 int backjump_level = 0;
2073 while (true) {
2074 const BooleanVariable var = (*trail_)[trail_index].Variable();
2075 --trail_index;
2076
2077 if (conflict->GetCoefficient(var) > 0 &&
2078 trail_->Assignment().LiteralIsTrue(conflict->GetLiteral(var))) {
2079 if (parameters_->minimize_reduction_during_pb_resolution()) {
2080 // When this parameter is true, we don't call ReduceCoefficients() at
2081 // every loop. However, it is still important to reduce the "current"
2082 // variable coefficient, because this can impact the value of the new
2083 // slack below.
2084 conflict->ReduceGivenCoefficient(var);
2085 }
2086
2087 // This is the slack one level before (< Info(var).trail_index).
2088 slack += conflict->GetCoefficient(var);
2089
2090 // This can't happen at the beginning, but may happen later.
2091 // It means that even without var assigned, we still have a conflict.
2092 if (slack < 0) continue;
2093
2094 // At this point, just removing the last assignment lift the conflict.
2095 // So we can abort if the true assignment before that is at a lower level
2096 // TODO(user): Somewhat inefficient.
2097 // TODO(user): We could abort earlier...
2098 const int current_level = DecisionLevel(var);
2099 int i = trail_index;
2100 while (i >= 0) {
2101 const BooleanVariable previous_var = (*trail_)[i].Variable();
2102 if (conflict->GetCoefficient(previous_var) > 0 &&
2103 trail_->Assignment().LiteralIsTrue(
2104 conflict->GetLiteral(previous_var))) {
2105 break;
2106 }
2107 --i;
2108 }
2109 if (i < 0 || DecisionLevel((*trail_)[i].Variable()) < current_level) {
2110 backjump_level = i < 0 ? 0 : DecisionLevel((*trail_)[i].Variable());
2111 break;
2112 }
2113
2114 // We can't abort, So resolve the current variable.
2115 DCHECK_NE(trail_->AssignmentType(var), AssignmentType::kSearchDecision);
2116 const bool clause_used = ResolvePBConflict(var, conflict, &slack);
2117
2118 // At this point, we have a negative slack. Note that ReduceCoefficients()
2119 // will not change it. However it may change the slack value of the next
2120 // iteration (when we will no longer take into account the true literal
2121 // with highest trail index).
2122 //
2123 // Note that the trail_index has already been decremented, it is why
2124 // we need the +1 in the slack computation.
2125 const Coefficient slack_only_for_debug =
2126 DEBUG_MODE
2127 ? conflict->ComputeSlackForTrailPrefix(*trail_, trail_index + 1)
2128 : Coefficient(0);
2129
2130 if (clause_used) {
2131 // If a clause was used, we know that slack has the correct value.
2132 if (!parameters_->minimize_reduction_during_pb_resolution()) {
2133 conflict->ReduceCoefficients();
2134 }
2135 } else {
2136 // TODO(user): The function below can take most of the running time on
2137 // some instances. The goal is to have slack updated to its new value
2138 // incrementally, but we are not here yet.
2139 if (parameters_->minimize_reduction_during_pb_resolution()) {
2140 slack =
2141 conflict->ComputeSlackForTrailPrefix(*trail_, trail_index + 1);
2142 } else {
2143 slack = conflict->ReduceCoefficientsAndComputeSlackForTrailPrefix(
2144 *trail_, trail_index + 1);
2145 }
2146 }
2147 DCHECK_EQ(slack, slack_only_for_debug);
2148 CHECK_LT(slack, 0);
2149 if (conflict->Rhs() < 0) {
2150 *pb_backjump_level = -1;
2151 return;
2152 }
2153 }
2154 }
2155
2156 // Reduce the conflit coefficients if it is not already done.
2157 // This is important to avoid integer overflow.
2158 if (!parameters_->minimize_reduction_during_pb_resolution()) {
2159 conflict->ReduceCoefficients();
2160 }
2161
2162 // Double check.
2163 // The sum of the literal with level <= backjump_level must propagate.
2164 std::vector<Coefficient> sum_for_le_level(backjump_level + 2, Coefficient(0));
2165 std::vector<Coefficient> max_coeff_for_ge_level(backjump_level + 2,
2166 Coefficient(0));
2167 int size = 0;
2168 Coefficient max_sum(0);
2169 for (BooleanVariable var : conflict->PossibleNonZeros()) {
2170 const Coefficient coeff = conflict->GetCoefficient(var);
2171 if (coeff == 0) continue;
2172 max_sum += coeff;
2173 ++size;
2174 if (!trail_->Assignment().VariableIsAssigned(var) ||
2175 DecisionLevel(var) > backjump_level) {
2176 max_coeff_for_ge_level[backjump_level + 1] =
2177 std::max(max_coeff_for_ge_level[backjump_level + 1], coeff);
2178 } else {
2179 const int level = DecisionLevel(var);
2180 if (trail_->Assignment().LiteralIsTrue(conflict->GetLiteral(var))) {
2181 sum_for_le_level[level] += coeff;
2182 }
2183 max_coeff_for_ge_level[level] =
2184 std::max(max_coeff_for_ge_level[level], coeff);
2185 }
2186 }
2187
2188 // Compute the cumulative version.
2189 for (int i = 1; i < sum_for_le_level.size(); ++i) {
2190 sum_for_le_level[i] += sum_for_le_level[i - 1];
2191 }
2192 for (int i = max_coeff_for_ge_level.size() - 2; i >= 0; --i) {
2193 max_coeff_for_ge_level[i] =
2194 std::max(max_coeff_for_ge_level[i], max_coeff_for_ge_level[i + 1]);
2195 }
2196
2197 // Compute first propagation level. -1 means that the problem is UNSAT.
2198 // Note that the first propagation level may be < backjump_level!
2199 if (sum_for_le_level[0] > conflict->Rhs()) {
2200 *pb_backjump_level = -1;
2201 return;
2202 }
2203 for (int i = 0; i <= backjump_level; ++i) {
2204 const Coefficient level_sum = sum_for_le_level[i];
2205 CHECK_LE(level_sum, conflict->Rhs());
2206 if (conflict->Rhs() - level_sum < max_coeff_for_ge_level[i + 1]) {
2207 *pb_backjump_level = i;
2208 return;
2209 }
2210 }
2211 LOG(FATAL) << "The code should never reach here.";
2212}

2243// other literals of the conflict. It is directly infered if the literals of its
2244// reason clause are either from level 0 or from the conflict itself.

2250 SCOPED_TIME_STAT(&stats_);

2252

2254 // since the first literal of the conflict is the 1-UIP literal.
2255 int index = 1;

2257 const BooleanVariable var = (*conflict)[i].Variable();

2281// This is similar to MinimizeConflictSimple() except that for each literal of
2282// the conflict, the literals of its reason are recursively expanded using their
2283// reason and so on. The recursion loops until we show that the initial literal
2284// can be infered from the conflict variables alone, or if we show that this is
2285// not the case. The result of any variable expansion will be cached in order
2286// not to be expended again.
2287void SatSolver::MinimizeConflictRecursively(std::vector<Literal>* conflict) {
2288 SCOPED_TIME_STAT(&stats_);
2289
2290 // is_marked_ will contains all the conflict literals plus the literals that
2291 // have been shown to depends only on the conflict literals. is_independent_
2292 // will contains the literals that have been shown NOT to depends only on the
2293 // conflict literals. The too set are exclusive for non-conflict literals, but
2294 // a conflict literal (which is always marked) can be independent if we showed
2295 // that it can't be removed from the clause.
2296 //
2297 // Optimization: There is no need to call is_marked_.ClearAndResize() or to
2298 // mark the conflict literals since this was already done by
2299 // ComputeFirstUIPConflict().
2300 is_independent_.ClearAndResize(num_variables_);
2301
2302 // min_trail_index_per_level_ will always be reset to all
2303 // std::numeric_limits<int>::max() at the end. This is used to prune the
2304 // search because any literal at a given level with an index smaller or equal
2305 // to min_trail_index_per_level_[level] can't be redundant.
2306 if (CurrentDecisionLevel() >= min_trail_index_per_level_.size()) {
2307 min_trail_index_per_level_.resize(CurrentDecisionLevel() + 1,
2308 std::numeric_limits<int>::max());
2309 }
2310
2311 // Compute the number of variable at each decision levels. This will be used
2312 // to pruned the DFS because we know that the minimized conflict will have at
2313 // least one variable of each decision levels. Because such variable can't be
2314 // eliminated using lower decision levels variable otherwise it will have been
2315 // propagated.
2316 //
2317 // Note(user): Because is_marked_ may actually contains literals that are
2318 // implied if the 1-UIP literal is false, we can't just iterate on the
2319 // variables of the conflict here.
2320 for (BooleanVariable var : is_marked_.PositionsSetAtLeastOnce()) {
2321 const int level = DecisionLevel(var);
2322 min_trail_index_per_level_[level] = std::min(
2323 min_trail_index_per_level_[level], trail_->Info(var).trail_index);
2324 }
2325
2326 // Remove the redundant variable from the conflict. That is the ones that can
2327 // be infered by some other variables in the conflict.
2328 // Note that we can skip the first position since this is the 1-UIP.
2329 int index = 1;
2330 for (int i = 1; i < conflict->size(); ++i) {
2331 const BooleanVariable var = (*conflict)[i].Variable();
2332 const AssignmentInfo& info = trail_->Info(var);
2333 if (time_limit_->LimitReached() ||
2334 info.type == AssignmentType::kSearchDecision ||
2335 info.trail_index <= min_trail_index_per_level_[info.level] ||
2336 !CanBeInferedFromConflictVariables(var)) {
2337 // Mark the conflict variable as independent. Note that is_marked_[var]
2338 // will still be true.
2339 is_independent_.Set(var);
2340 (*conflict)[index] = (*conflict)[i];
2341 ++index;
2342 }
2343 }
2344 conflict->resize(index);
2345
2346 // Reset min_trail_index_per_level_. We use the sparse version only if it
2347 // involves less than half the size of min_trail_index_per_level_.
2348 const int threshold = min_trail_index_per_level_.size() / 2;
2349 if (is_marked_.PositionsSetAtLeastOnce().size() < threshold) {
2350 for (BooleanVariable var : is_marked_.PositionsSetAtLeastOnce()) {
2351 min_trail_index_per_level_[DecisionLevel(var)] =
2352 std::numeric_limits<int>::max();
2353 }
2354 } else {
2355 min_trail_index_per_level_.clear();
2356 }
2357}

2359bool SatSolver::CanBeInferedFromConflictVariables(BooleanVariable variable) {
2360 // Test for an already processed variable with the same reason.
2361 {
2362 DCHECK(is_marked_[variable]);
2363 const BooleanVariable v =
2364 same_reason_identifier_.FirstVariableWithSameReason(variable);
2365 if (v != variable) return !is_independent_[v];
2366 }
2367
2368 // This function implement an iterative DFS from the given variable. It uses
2369 // the reason clause as adjacency lists. dfs_stack_ can be seens as the
2370 // recursive call stack of the variable we are currently processing. All its
2371 // adjacent variable will be pushed into variable_to_process_, and we will
2372 // then dequeue them one by one and process them.
2373 //
2374 // Note(user): As of 03/2014, --cpu_profile seems to indicate that using
2375 // dfs_stack_.assign(1, variable) is slower. My explanation is that the
2376 // function call is not inlined.
2377 dfs_stack_.clear();
2378 dfs_stack_.push_back(variable);
2379 variable_to_process_.clear();
2380 variable_to_process_.push_back(variable);
2381
2382 // First we expand the reason for the given variable.
2383 for (const Literal literal : trail_->Reason(variable)) {
2384 const BooleanVariable var = literal.Variable();
2385 DCHECK_NE(var, variable);
2386 if (is_marked_[var]) continue;
2387 const AssignmentInfo& info = trail_->Info(var);
2388 if (info.level == 0) {
2389 // Note that this is not needed if the solver is not configured to produce
2390 // an unsat proof. However, the (level == 0) test should always be false
2391 // in this case because there will never be literals of level zero in any
2392 // reason when we don't want a proof.
2393 is_marked_.Set(var);
2394 continue;
2395 }
2396 if (info.trail_index <= min_trail_index_per_level_[info.level] ||
2397 info.type == AssignmentType::kSearchDecision || is_independent_[var]) {
2398 return false;
2399 }
2400 variable_to_process_.push_back(var);
2401 }
2402
2403 // Then we start the DFS.
2404 while (!variable_to_process_.empty()) {
2405 const BooleanVariable current_var = variable_to_process_.back();
2406 if (current_var == dfs_stack_.back()) {
2407 // We finished the DFS of the variable dfs_stack_.back(), this can be seen
2408 // as a recursive call terminating.
2409 if (dfs_stack_.size() > 1) {
2410 DCHECK(!is_marked_[current_var]);
2411 is_marked_.Set(current_var);
2412 }
2413 variable_to_process_.pop_back();
2414 dfs_stack_.pop_back();
2415 continue;
2416 }
2417
2418 // If this variable became marked since the we pushed it, we can skip it.
2419 if (is_marked_[current_var]) {
2420 variable_to_process_.pop_back();
2421 continue;
2422 }
2423
2424 // This case will never be encountered since we abort right away as soon
2425 // as an independent variable is found.
2426 DCHECK(!is_independent_[current_var]);
2427
2428 // Test for an already processed variable with the same reason.
2429 {
2430 const BooleanVariable v =
2431 same_reason_identifier_.FirstVariableWithSameReason(current_var);
2432 if (v != current_var) {
2433 if (is_independent_[v]) break;
2434 DCHECK(is_marked_[v]);
2435 variable_to_process_.pop_back();
2436 continue;
2437 }
2438 }
2439
2440 // Expand the variable. This can be seen as making a recursive call.
2441 dfs_stack_.push_back(current_var);
2442 bool abort_early = false;
2443 for (Literal literal : trail_->Reason(current_var)) {
2444 const BooleanVariable var = literal.Variable();
2445 DCHECK_NE(var, current_var);
2446 const AssignmentInfo& info = trail_->Info(var);
2447 if (info.level == 0 || is_marked_[var]) continue;
2448 if (info.trail_index <= min_trail_index_per_level_[info.level] ||
2449 info.type == AssignmentType::kSearchDecision ||
2450 is_independent_[var]) {
2451 abort_early = true;
2452 break;
2453 }
2454 variable_to_process_.push_back(var);
2455 }
2456 if (abort_early) break;
2457 }
2458
2459 // All the variable left on the dfs_stack_ are independent.
2460 for (const BooleanVariable var : dfs_stack_) {
2461 is_independent_.Set(var);
2462 }
2463 return dfs_stack_.empty();
2464}

2486// This function allows a conflict variable to be replaced by another variable
2487// not originally in the conflict. Greater reduction and backtracking can be
2488// achieved this way, but the effect of this is not clear.
2489//
2490// TODO(user): More investigation needed. This seems to help on the Hanoi
2491// problems, but degrades performance on others.
2492//
2493// TODO(user): Find a reference for this? neither minisat nor glucose do that,
2494// they just do MinimizeConflictRecursively() with a different implementation.
2495// Note that their behavior also make more sense with the way they (and we) bump
2496// the variable activities.
2497void SatSolver::MinimizeConflictExperimental(std::vector<Literal>* conflict) {
2498 SCOPED_TIME_STAT(&stats_);
2499
2500 // First, sort the variables in the conflict by decreasing decision levels.
2501 // Also initialize is_marked_ to true for all conflict variables.
2502 is_marked_.ClearAndResize(num_variables_);
2503 const int current_level = CurrentDecisionLevel();
2504 std::vector<WeightedVariable> variables_sorted_by_level;
2505 for (Literal literal : *conflict) {
2506 const BooleanVariable var = literal.Variable();
2507 is_marked_.Set(var);
2508 const int level = DecisionLevel(var);
2509 if (level < current_level) {
2510 variables_sorted_by_level.push_back(WeightedVariable(var, level));
2511 }
2512 }
2513 std::sort(variables_sorted_by_level.begin(), variables_sorted_by_level.end(),
2514 VariableWithLargerWeightFirst());
2515
2516 // Then process the reason of the variable with highest level first.
2517 std::vector<BooleanVariable> to_remove;
2518 for (WeightedVariable weighted_var : variables_sorted_by_level) {
2519 const BooleanVariable var = weighted_var.var;
2520
2521 // A nullptr reason means that this was a decision variable from the
2522 // previous levels.
2523 const absl::Span<const Literal> reason = trail_->Reason(var);
2524 if (reason.empty()) continue;
2525
2526 // Compute how many and which literals from the current reason do not appear
2527 // in the current conflict. Level 0 literals are ignored.
2528 std::vector<Literal> not_contained_literals;
2529 for (const Literal reason_literal : reason) {
2530 const BooleanVariable reason_var = reason_literal.Variable();
2531
2532 // We ignore level 0 variables.
2533 if (DecisionLevel(reason_var) == 0) continue;
2534
2535 // We have a reason literal whose variable is not yet seen.
2536 // If there is more than one, break right away, we will not minimize the
2537 // current conflict with this variable.
2538 if (!is_marked_[reason_var]) {
2539 not_contained_literals.push_back(reason_literal);
2540 if (not_contained_literals.size() > 1) break;
2541 }
2542 }
2543 if (not_contained_literals.empty()) {
2544 // This variable will be deleted from the conflict. Note that we don't
2545 // unmark it. This is because this variable can be infered from the other
2546 // variables in the conflict, so it is okay to skip it when processing the
2547 // reasons of other variables.
2548 to_remove.push_back(var);
2549 } else if (not_contained_literals.size() == 1) {
2550 // Replace the literal from variable var with the only
2551 // not_contained_literals from the current reason.
2552 to_remove.push_back(var);
2553 is_marked_.Set(not_contained_literals.front().Variable());
2554 conflict->push_back(not_contained_literals.front());
2555 }
2556 }
2557
2558 // Unmark the variable that should be removed from the conflict.
2559 for (BooleanVariable var : to_remove) {
2560 is_marked_.Clear(var);
2561 }
2562
2563 // Remove the now unmarked literals from the conflict.
2564 int index = 0;
2565 for (int i = 0; i < conflict->size(); ++i) {
2566 const Literal literal = (*conflict)[i];
2567 if (is_marked_[literal.Variable()]) {
2568 (*conflict)[index] = literal;
2569 ++index;
2570 }
2571 }
2572 conflict->erase(conflict->begin() + index, conflict->end());
2573}
