14// Generates strongly typed integer types.
15//
16// StrongIndex is a simple template class mechanism for defining "logical"
17// index-like class types that support some of the same functionalities
18// as int, but which prevent assignment, construction, and
19// other operations from other similar integer-like types. Essentially, the
20// template class StrongIndex<StrongIndexName> has the additional
21// property that it cannot be assigned to or constructed from another
22// StrongIndex with a different StrongIndexName.
23//
24// Usage
25// DEFINE_STRONG_INDEX_TYPE(name);
26// where name is the desired (unique) name for the "logical" index type.
27//
28// StrongInt64 is a more general strong integer class based on int64_t.
29// It has the same general type safeness, and supports more integer operators.
30//
31// Usage
32// DEFINE_STRONG_INT64_TYPE(name);
33// where name is the desired (unique) name for the "logical" int64_t type.
34//
35// SUPPORTED OPERATIONS --------------------------------------------------------
36//
37// The StrongIndex type is limited and only supports following operators are
38// supported: unary: ++ (both prefix and postfix), comparison: ==, !=, <, <=, >,
39// >=; assignment: =, +=, -=,; stream: <<. Each operator allows the same
40// StrongIndexName and the int to be used on both left- and right-hand sides.
41//
42// The StrongInt64 type supports all integer operators across StrongInt64
43// with the same StrongIntegerName and int64_t.
44//
45// Both support an accessor value() returning the stored value.
46//
47// The classes also define hash functors that allows the strong types to be used
48// as key to hashable containers.

67// Note: The struct index_type_name ## _index_tag_ trickery is needed to ensure

91// Note: we need two classes as it is the only way to have different set of
92// operators on each class.
93// Note: We use to class to easily define a different set of operators for the
94// index and int64_t type.
95
96#define STRONG_ASSIGNMENT_OP(StrongClass, IntType, op) \
97 ThisType& operator op(const ThisType & arg_value) { \
98 value_ op arg_value.value(); \
99 return *this; \
100 } \
101 ThisType& operator op(IntType arg_value) { \
102 value_ op arg_value; \
103 return *this; \
104 }
105
106#define INCREMENT_AND_DECREMENT_OPERATORS \
107 ThisType& operator++() { \
108 ++value_; \
109 return *this; \
110 } \
111 const ThisType operator++(int) { \
112 ThisType temp(*this); \
113 ++value_; \
114 return temp; \
115 } \
116 ThisType& operator--() { \
117 --value_; \
118 return *this; \
119 } \
120 const ThisType operator--(int) { \
121 ThisType temp(*this); \
122 --value_; \
123 return temp; \
124 }

126// Holds an int value and behaves as an int by exposing assignment,
127// unary, comparison, and arithmetic operators.
128//
129// The template parameter StrongIndexName defines the name for the int type and
130// must be unique within a binary (the convenient DEFINE_STRONG_INDEX_TYPE macro
131// at the start of the file generates a unique StrongIndexName).
132//
133// This class is NOT thread-safe.
134template <typename StrongIndexName>
135class StrongIndex {
136 public:
137 typedef int ValueType; // Needed for StrongVector.
138 typedef StrongIndex<StrongIndexName> ThisType; // Syntactic sugar.
139
140 static constexpr absl::string_view TypeName() {
141 return StrongIndexName::TypeName();
142 }
143
144 struct ABSL_DEPRECATED("Use absl::Hash instead") Hasher {
145 size_t operator()(const StrongIndex& x) const {
146 return static_cast<size_t>(x.value());
147 }
148 };
149
150 constexpr StrongIndex() : value_(0) {}
151
152 explicit constexpr StrongIndex(int value) : value_(value) {}
153
154 StrongIndex& operator=(int arg_value) {
155 value_ = arg_value;
156 return *this;
157 }
158
159 // The class provides a value() accessor returning the stored int value_
160 // as well as a templatized accessor that is just a syntactic sugar for
161 // static_cast<T>(var.value());
162 constexpr int value() const { return value_; }
163
164 template <typename ValType> // Needed for StrongVector.
165 constexpr ValType value() const {
166 return static_cast<ValType>(value_);
167 }
168
169 constexpr ThisType operator+() const { return ThisType(value_); }
170 constexpr ThisType operator-() const { return ThisType(-value_); }
171
172 INCREMENT_AND_DECREMENT_OPERATORS;
173
174 STRONG_ASSIGNMENT_OP(StrongIndex, int, +=);
175 STRONG_ASSIGNMENT_OP(StrongIndex, int, -=);
176
177 private:
178 int value_;
179};
