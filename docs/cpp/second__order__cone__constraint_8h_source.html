33// A value type that references a second-order cone constraint from
34// ModelStorage. Usually this type is passed by copy.
35class SecondOrderConeConstraint {
36 public:
37 // The typed integer used for ids.
38 using IdType = SecondOrderConeConstraintId;

40 inline SecondOrderConeConstraint(const ModelStorage* storage,
41 SecondOrderConeConstraintId id);

46 inline const ModelStorage* storage() const;
47
48 inline absl::string_view name() const;

50 // Returns "upper_bound" with respect to a constraint of the form
51 // ||arguments_to_norm||₂ ≤ upper_bound.
52 LinearExpression UpperBound() const;
53
54 // Returns "arguments_to_norm" with respect to a constraint of the form
55 // ||arguments_to_norm||₂ ≤ upper_bound.

59 // a nonzero coefficient. Order is not defined.

62 // Returns a detailed string description of the contents of the constraint
63 // (not its name, use `<<` for that instead).

75 private:
