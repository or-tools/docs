26absl::Status ValidateConstraint(const SosConstraintProto& constraint,
27 const IdNameBiMap& variable_universe) {
28 if (!constraint.weights().empty() &&
29 constraint.weights_size() != constraint.expressions_size()) {
30 return util::InvalidArgumentErrorBuilder()
31 << "Length mismatch between weights and expressions: "
32 << constraint.weights_size() << " vs. "
33 << constraint.expressions_size();
34 }
35 for (const LinearExpressionProto& expression : constraint.expressions()) {
36 RETURN_IF_ERROR(ValidateLinearExpression(expression, variable_universe))
37 << "Invalid SOS expression";
38 }
39 for (const double weight : constraint.weights()) {
40 RETURN_IF_ERROR(CheckScalarNoNanNoInf(weight)) << "Invalid SOS weight";
41 }
42 return absl::OkStatus();
43}
