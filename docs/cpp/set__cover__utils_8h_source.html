30// Element used for AdjustablePriorityQueue. It's an implementation detail.
31class SubsetPriority {
32 public:
33 SubsetPriority()
34 : heap_index_(-1),
35 subset_(0),
36 priority_(std::numeric_limits<Cost>::infinity()) {}
37
38 SubsetPriority(SubsetIndex s, Cost cost)
39 : heap_index_(s.value()), subset_(s), priority_(cost) {}
40
41 void SetHeapIndex(int h) { heap_index_ = h; }
42 int GetHeapIndex() const { return heap_index_; }
43
44 // The priority queue maintains the max element. This comparator breaks ties
45 // between subsets using their cardinalities.
46 bool operator<(const SubsetPriority& other) const {
47 return priority_ < other.priority_ ||
48 (priority_ == other.priority_ && subset_ < other.subset_);
49 }
50
51 SubsetIndex GetSubset() const { return subset_; }
52 void SetPriority(Cost priority) { priority_ = priority; }
53 Cost GetPriority() const { return priority_; }
54
55 private:
56 int heap_index_;
57 SubsetIndex subset_;
58 Cost priority_;
59};

112// A Tabu list is a fixed-sized circular array of small size, usually a few
113// dozens of elements.
114template <typename T>
115class TabuList {
116 public:
117 explicit TabuList(T size) : array_(0), fill_(0), index_(0) {
118 array_.resize(size.value(), T(-1));
119 }
120
121 // Returns the size of the array.
122 int size() const { return array_.size(); }
123
124 // Initializes the array of the Tabu list.
125 void Init(int size) {
126 array_.resize(size, T(-1));
127 fill_ = 0;
128 index_ = 0;
129 }
130
131 // Adds t to the array. When the end of the array is reached, re-start at 0.
132 void Add(T t) {
133 const int size = array_.size();
134 array_[index_] = t;
135 ++index_;
136 if (index_ >= size) {
137 index_ = 0;
138 }
139 if (fill_ < size) {
140 ++fill_;
141 }
142 }
143
144 // Returns true if t is in the array. This is O(size), but small.
145 bool Contains(T t) const {
146 for (int i = 0; i < fill_; ++i) {
147 if (t == array_[i]) {
148 return true;
149 }
150 }
151 return false;
152 }
153
154 private:
155 std::vector<T> array_;
156 int fill_;
157 int index_;
158};
