29// A contiguous block of memory that contains `size` values of type `Value`, and
30// the first value is aligned to `alignment` bytes.
31template <typename Value, size_t size, size_t alignment = sizeof(Value) * size>
32struct alignas(alignment) AlignedBlock {
33 Value values[size] = {};
34
35 Value Sum() const {
36 alignas(alignment) Value sum[size];
37 std::copy(std::begin(values), std::end(values), std::begin(sum));
38 for (int i = size; i > 1; i /= 2) {
39 int middle = i / 2;
40 for (int j = 0; j < middle; ++j) {
41 sum[j] += sum[middle + j];
42 }
43 }
44 return sum[0];
45 }
46};

48// In-place addition for two AlignedBlock values. Adds `in` to `out`, storing
49// the value in `out`. Unless something steps in, this compiles into a single
50// `*addp*` instruction.
51template <typename Value, size_t size>
52void AddInPlace(AlignedBlock<Value, size>& out,
53 const AlignedBlock<Value, size>& in) {
54 for (int i = 0; i < size; ++i) {
55 out.values[i] += in.values[i];
56 }
57}

59// Computes the sum of `num_blocks` aligned blocks. Proceeds in three phases:
60// 1. Parallel sum with N = `num_blocks_per_iteration` independent block-sized
61// accumulators. At the end, accumulator i contains partial sums at indices
62// i, i + N, i + 2*N, ..., i + M*N, where M is the largest number of blocks
63// such that i+M*N < num_blocks for all i.
64// 2. Parallel addition of remaining blocks. All remaining blocks are added to
65// accumulators 0, ..., num_remaining_blocks - 1.
66// 3. Sum of accumulators: all accumulators are added together. The result is a
67// single block-sized accumulator that is returned to the caller.
68//
69// The code of the function was specifically tuned for 32-bit floating point
70// values, and works best with block_size = 4 and num_blocks_per_iteration = 4.
71// It will likely work with other types, but may require extra care and possibly
72// different parameters.
73//
74// NOTE(user): As of 2023-04-28, Clang's auto-vectorizer is *very* brittle
75// and most attempts to reduce the last accumulator from step 3 into a single
76// value prevents the rest of the function from being vectorized. To mitigate
77// this behavior we return the whole block (which should normally fit into a
78// single vector register). We also mark the function with
79// `ABSL_ATTRIBUTE_NOINLINE` to make prevent the inliner from merging this
80// function with any additional code that would prevent vectorization.
81template <size_t block_size, size_t num_blocks_per_iteration, typename Value>
82AlignedBlock<Value, block_size> ABSL_ATTRIBUTE_NOINLINE AlignedBlockSum(
83 const AlignedBlock<Value, block_size>* blocks, size_t num_blocks) {
84 using Block = AlignedBlock<Value, block_size>;
85 static_assert(sizeof(Block[2]) == sizeof(Block::values) * 2,
86 "The values in the block are not packed.");
87
88 AlignedBlock<Value, block_size> sum[num_blocks_per_iteration];
89
90 const int leftover_blocks = num_blocks % num_blocks_per_iteration;
91 const int packed_blocks = num_blocks - leftover_blocks;
92
93 const AlignedBlock<Value, block_size>* aligned_block_end =
94 blocks + packed_blocks;
95
96 // Phase 1: Parallel sum of the bulk of the data.
97 if (packed_blocks >= num_blocks_per_iteration) {
98 std::copy(blocks, blocks + num_blocks_per_iteration, sum);
99 }
100 for (int i = num_blocks_per_iteration; i < packed_blocks;
101 i += num_blocks_per_iteration) {
102 for (int j = 0; j < num_blocks_per_iteration; ++j) {
103 AddInPlace(sum[j], blocks[i + j]);
104 }
105 }
106
107 // Phase 2: Semi-parallel sum of the remaining up to
108 // num_blocks_per_iteration - 1 blocks.
109 for (int i = 0; i < leftover_blocks; ++i) {
110 AddInPlace(sum[i], aligned_block_end[i]);
111 }
112
113 // Phase 3: Reduce the accumulator blocks to a single block.
114 // NOTE(user): When this code is auto-vectorized correctly, the initial
115 // copy is a no-op, and the for loop below translates to
116 // num_blocks_per_iteration - 1 vector add instructions. In 2023-05, I
117 // experimented with other versions (e.g. using sum[0] as the target or making
118 // res a const reference to sum[0], but in most cases they broke vectorization
119 // of the whole function).
120 AlignedBlock<Value, block_size> res = sum[0];
121 for (int i = 1; i < num_blocks_per_iteration; ++i) {
122 AddInPlace(res, sum[i]);
123 }
124
125 return res;
126}

128// Computes the sum of values in `values`, by adding `num_blocks_per_iteration`
129// blocks of `block_size` values.
130// By default, the sum does not make any assumptions about the size or alignment
131// of `values`. When the first item of `values` is known to be aligned to
132// `block_size * sizeof(Value)` bytes, `assume_aligned_at_start` can be used to
133// save a small amount of time.
134template <size_t block_size = 4, size_t num_blocks_per_iteration = 4,
135 bool assume_aligned_at_start = false, typename Value>
136Value VectorSum(absl::Span<const Value> values) {
137 using Block = AlignedBlock<Value, block_size>;
138 const Value* start_ptr = values.data();
139 const int size = values.size();
140 // With less than two blocks, there's not a lot to vectorize, and a simple
141 // sequential sum is usually faster.
142 if (size == 0) return Value{0};
143 if (size < 2 * block_size) {
144 return std::accumulate(start_ptr + 1, start_ptr + size, *start_ptr);
145 }
146
147 if (assume_aligned_at_start) {
148 ABSL_ASSUME(reinterpret_cast<std::uintptr_t>(start_ptr) % alignof(Block) ==
149 0);
150 }
151 const Value* aligned_start_ptr =
152 assume_aligned_at_start ? start_ptr : AlignUp<alignof(Block)>(start_ptr);
153 const Block* blocks = reinterpret_cast<const Block*>(aligned_start_ptr);
154 const Value* end_ptr = start_ptr + size;
155 const Value* aligned_end_ptr = AlignDown<alignof(Block)>(end_ptr);
156 ABSL_ASSUME(aligned_start_ptr <= aligned_end_ptr);
157 const size_t num_blocks = (aligned_end_ptr - aligned_start_ptr) / block_size;
158 ABSL_ASSUME(
159 reinterpret_cast<std::uintptr_t>(aligned_end_ptr) % alignof(Block) == 0);
160
161 Value leading_items_sum{};
162 if (!assume_aligned_at_start) {
163 leading_items_sum = std::accumulate(start_ptr, aligned_start_ptr, Value{});
164 }
165 Block res =
166 AlignedBlockSum<block_size, num_blocks_per_iteration>(blocks, num_blocks);
167 Value block_sum = res.Sum();
168 return std::accumulate(aligned_end_ptr, end_ptr,
169 block_sum + leading_items_sum);
170}
