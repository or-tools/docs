39// Parameters to control a single solve that are specific to the input
40// model (see SolveParametersProto for model independent parameters).
41struct ModelSolveParameters {
42 // Returns the parameters that empty DualSolution and DualRay, only keep the
43 // values of all variables in PrimalSolution and PrimalRay.
44 //
45 // This is a shortcut method that is equivalent to setting the dual filters
46 // with MakeSkipAllFilter().
47 static ModelSolveParameters OnlyPrimalVariables();
48
49 // Returns the parameters that empty DualSolution and DualRay, only keep the
50 // values of the specified variables in PrimalSolution and PrimalRay.
51 //
52 // The input Collection must be usable in a for-range loop with Variable
53 // values. This will be typically a std::vector<Variable> or and
54 // std::initializer_list<Variable> (see the other overload).
55 //
56 // This is a shortcut method that is equivalent to setting the dual filters
57 // with MakeSkipAllFilter() and the variable_values_filter with
58 // MakeKeepKeysFilter(variables).
59 //
60 // Example:
61 // std::vector<Variable> decision_vars = ...;
62 // const auto params =
63 // ModelSolveParameters::OnlySomePrimalVariables(decision_vars);
64 template <typename Collection>
65 static ModelSolveParameters OnlySomePrimalVariables(
66 const Collection& variables);
67
68 // Returns the parameters that empty DualSolution and DualRay, only keeping
69 // the values of the specified variables in PrimalSolution and PrimalRay.
70 //
71 // See the other overload's documentation for details. This overload is needed
72 // since C++ can't guess the type when using an initializer list expression.
73 //
74 // Example:
75 // const Variable a = ...;
76 // const Variable b = ...;
77 // const auto params =
78 // ModelSolveParameters::OnlySomePrimalVariables({a, b});
79 static ModelSolveParameters OnlySomePrimalVariables(
80 std::initializer_list<Variable> variables);
81
82 // The filter that is applied to variable_values of both PrimalSolution and
83 // PrimalRay.
84 MapFilter<Variable> variable_values_filter;
85
86 // The filter that is applied to dual_values of DualSolution and DualRay.
87 MapFilter<LinearConstraint> dual_values_filter;
88
89 // The filter that is applied to reduced_costs of DualSolution and DualRay.
90 MapFilter<Variable> reduced_costs_filter;
91
92 // Optional initial basis for warm starting simplex LP solvers. If set, it is
93 // expected to be valid.
94 std::optional<Basis> initial_basis;
95
96 // A suggested starting solution for the solver.
97 //
98 // MIP solvers generally only want primal information (`variable_values`),
99 // while LP solvers want both primal and dual information (`dual_values`).
100 //
101 // Many MIP solvers can work with: (1) partial solutions that do not specify
102 // all variables or (2) infeasible solutions. In these cases, solvers
103 // typically solve a sub-MIP to complete/correct the hint.
104 //
105 // How the hint is used by the solver, if at all, is highly dependent on the
106 // solver, the problem type, and the algorithm used. The most reliable way to
107 // ensure your hint has an effect is to read the underlying solvers logs with
108 // and without the hint.
109 //
110 // Simplex-based LP solvers typically prefer an initial basis to a solution
111 // hint (they need to crossover to convert the hint to a basic feasible
112 // solution otherwise).
113 struct SolutionHint {
114 // Values should be finite and not NaN (otherwise, `Solve()` will return a
115 // `Status` error).
116 VariableMap<double> variable_values;
117
118 // Values should be finite and not NaN (otherwise, `Solve()` will return a
119 // `Status` error).
120 LinearConstraintMap<double> dual_values;
121
122 // Returns a failure if the referenced variables and constraints don't
123 // belong to the input expected_storage (which must not be nullptr).
124 absl::Status CheckModelStorage(const ModelStorage* expected_storage) const;
125
126 // Returns the proto equivalent of this object.
127 //
128 // The caller should use CheckModelStorage() as this function does not check
129 // internal consistency of the referenced variables and constraints.
130 SolutionHintProto Proto() const;
131
132 // Returns the hint corresponding to this proto and the given model.
133 //
134 // This can be useful when loading a hint from another format, e.g. with
135 // MPModelProtoSolutionHintToMathOptHint().
136 static absl::StatusOr<SolutionHint> FromProto(
137 const Model& model, const SolutionHintProto& hint_proto);
138 };
139
140 // Optional solution hints. If the underlying solver only accepts a single
141 // hint, the first hint is used.
142 std::vector<SolutionHint> solution_hints;
143
144 // Optional branching priorities. Variables with higher values will be
145 // branched on first. Variables for which priorities are not set get the
146 // solver's default priority (usually zero).
147 VariableMap<int32_t> branching_priorities;
148
149 // Returns a failure if the referenced variables don't belong to the input
150 // expected_storage (which must not be nullptr).
151 absl::Status CheckModelStorage(const ModelStorage* expected_storage) const;
152
153 // Returns the proto equivalent of this object.
154 //
155 // The caller should use CheckModelStorage() as this function does not check
156 // internal consistency of the referenced variables and constraints.
157 ModelSolveParametersProto Proto() const;
158};

164template <typename Collection>
165ModelSolveParameters ModelSolveParameters::OnlySomePrimalVariables(
166 const Collection& variables) {
167 ModelSolveParameters parameters = OnlyPrimalVariables();
168 parameters.variable_values_filter = MakeKeepKeysFilter(variables);
169 return parameters;
170}
