31// During a simplex iteration, when the basis 'leaving_row' has been
32// selected, one of the main quantities needed in the primal or dual simplex
33// algorithm is called the update row.
34//
35// By definition, update_row[col] is the coefficient at position
36// 'leaving_row' in the current basis of the column 'col' of the matrix A.
37//
38// One efficient way to compute it is to compute the left inverse by B of the
39// unit vector with a one at the given leaving_row, and then to take the
40// scalar product of this left inverse with all the columns of A:
41// update_row[col] = (unit_{leaving_row} . B^{-1}) . A_col
42class UpdateRow {
43 public:
44 // Takes references to the linear program data we need.
45 UpdateRow(const CompactSparseMatrix& matrix,
46 const CompactSparseMatrix& transposed_matrix,
47 const VariablesInfo& variables_info, const RowToColMapping& basis,
48 const BasisFactorization& basis_factorization);
49
50 // This type is neither copyable nor movable.
51 UpdateRow(const UpdateRow&) = delete;
52 UpdateRow& operator=(const UpdateRow&) = delete;
53
54 // Invalidates the current update row and unit_row_left_inverse so the next
55 // call to ComputeUpdateRow() will recompute everything and not just return
56 // right away.
57 void Invalidate();
58
59 // Computes the left inverse of the given unit row, and stores it in
60 // unit_row_left_inverse_. The result is computed only once if leaving_row do
61 // not change, this until the next Invalidate() call.
62 void ComputeUnitRowLeftInverse(RowIndex leaving_row);
63
64 // Computes the relevant coefficients (See GetIsRelevantBitRow() in
65 // VariablesInfo) of the update row. The result is only computed once
66 // if leaving_row do not change, this until the next Invalidate() call.
67 void ComputeUpdateRow(RowIndex leaving_row);
68
69 // Returns the left inverse of the unit row as computed by the last call to
70 // ComputeUpdateRow().
71 const ScatteredRow& GetUnitRowLeftInverse() const;
72
73 // Returns true if ComputeUpdateRow() was called since the last Invalidate().
74 bool IsComputedFor(RowIndex leaving_row) const {
75 return update_row_computed_for_ == leaving_row;
76 }
77
78 // Returns the update coefficients and non-zero positions corresponding to the
79 // last call to ComputeUpdateRow().
80 //
81 // TODO(user): Consider returning a packed vector of coefficient parallel to
82 // GetNonZeroPositions() instead. It should be fast to compute and iteration
83 // later should be quicker.
84 const DenseRow& GetCoefficients() const;
85 const ColIndexVector& GetNonZeroPositions() const;
86 Fractional GetCoefficient(ColIndex col) const { return coefficient_[col]; }
87
88 // Computes the update row including all position and fill output with it.
89 // We only use this when ComputeUnitRowLeftInverse() has already been called
90 // and we CHECK that.
91 void ComputeFullUpdateRow(RowIndex leaving_row, DenseRow* output) const;
92
93 // Sets the algorithm parameters.
94 void SetParameters(const GlopParameters& parameters);
95
96 // Returns statistics about this class as a string.
97 std::string StatString() const { return stats_.StatString(); }

99 // Only used for testing.
100 // Computes as the update row the product 'lhs' times the linear program
101 // matrix given at construction. Only the relevant columns matter (see
102 // VariablesInfo) and 'algorithm' can be one of "column", "row" or
103 // "row_hypersparse".
104 void ComputeUpdateRowForBenchmark(const DenseRow& lhs,
105 const std::string& algorithm);

107 // Deterministic time used by the scalar product computation of this class.

120 void ComputeUpdatesRowWiseHypersparse();
121 void ComputeUpdatesColumnWise();
