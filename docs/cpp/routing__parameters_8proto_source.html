137 // --- Inter-route operators ---
138 // Operator which moves a single node to another position.
139 // Possible neighbors for the path 1 -> 2 -> 3 -> 4 -> 5
140 // (where (1, 5) are first and last nodes of the path and can therefore not
141 // be moved):
142 // 1 -> 3 -> [2] -> 4 -> 5

144 // 1 -> 2 -> 4 -> [3] -> 5
145 // 1 -> [4] -> 2 -> 3 -> 5
146 OptionalBoolean use_relocate = 1;
147 // Operator which moves a pair of pickup and delivery nodes to another
148 // position where the first node of the pair must be before the second node
149 // on the same path. Compared to the light_relocate_pair operator, tries all
150 // possible positions of insertion of a pair (not only after another pair).
151 // Possible neighbors for the path 1 -> A -> B -> 2 -> 3 (where (1, 3) are
152 // first and last nodes of the path and can therefore not be moved, and
153 // (A, B) is a pair of nodes):
154 // 1 -> [A] -> 2 -> [B] -> 3
155 // 1 -> 2 -> [A] -> [B] -> 3
156 OptionalBoolean use_relocate_pair = 2;
157 // Operator which moves a pair of pickup and delivery nodes after another
158 // pair.
159 // Possible neighbors for paths 1 -> A -> B -> 2, 3 -> C -> D -> 4 (where
160 // (1, 2) and (3, 4) are first and last nodes of paths and can therefore not
161 // be moved, and (A, B) and (C, D) are pair of nodes):
162 // 1 -> 2, 3 -> C -> [A] -> D -> [B] -> 4
163 // 1 -> A -> [C] -> B -> [D] -> 2, 3 -> 4
164 OptionalBoolean use_light_relocate_pair = 24;
165 // Relocate neighborhood which moves chains of neighbors.
166 // The operator starts by relocating a node n after a node m, then continues
167 // moving nodes which were after n as long as the "cost" added is less than
168 // the "cost" of the arc (m, n). If the new chain doesn't respect the domain
169 // of next variables, it will try reordering the nodes until it finds a
170 // valid path.
171 // Possible neighbors for path 1 -> A -> B -> C -> D -> E -> 2 (where (1, 2)
172 // are first and last nodes of the path and can therefore not be moved, A
173 // must be performed before B, and A, D and E are located at the same
174 // place):
175 // 1 -> A -> C -> [B] -> D -> E -> 2
176 // 1 -> A -> C -> D -> [B] -> E -> 2
177 // 1 -> A -> C -> D -> E -> [B] -> 2
178 // 1 -> A -> B -> D -> [C] -> E -> 2
179 // 1 -> A -> B -> D -> E -> [C] -> 2
180 // 1 -> A -> [D] -> [E] -> B -> C -> 2
181 // 1 -> A -> B -> [D] -> [E] -> C -> 2
182 // 1 -> A -> [E] -> B -> C -> D -> 2
183 // 1 -> A -> B -> [E] -> C -> D -> 2
184 // 1 -> A -> B -> C -> [E] -> D -> 2
185 // This operator is extremely useful to move chains of nodes which are
186 // located at the same place (for instance nodes part of a same stop).
187 OptionalBoolean use_relocate_neighbors = 3;
188 // Relocate neighborhood that moves subpaths all pickup and delivery
189 // pairs have both pickup and delivery inside the subpath or both outside
190 // the subpath. For instance, for given paths:
191 // 0 -> A -> B -> A' -> B' -> 5 -> 6 -> 8
192 // 7 -> 9
193 // Pairs (A,A') and (B,B') are interleaved, so the expected neighbors are:
194 // 0 -> 5 -> A -> B -> A' -> B' -> 6 -> 8
195 // 7 -> 9
196 //
197 // 0 -> 5 -> 6 -> A -> B -> A' -> B' -> 8
198 // 7 -> 9
199 //
200 // 0 -> 5 -> 6 -> 8
201 // 7 -> A -> B -> A' -> B' -> 9
202 OptionalBoolean use_relocate_subtrip = 25;
203 // Operator which exchanges the positions of two nodes.
204 // Possible neighbors for the path 1 -> 2 -> 3 -> 4 -> 5
205 // (where (1, 5) are first and last nodes of the path and can therefore not
206 // be moved):
207 // 1 -> [3] -> [2] -> 4 -> 5
208 // 1 -> [4] -> 3 -> [2] -> 5
209 // 1 -> 2 -> [4] -> [3] -> 5
210 OptionalBoolean use_exchange = 4;
211 // Operator which exchanges the positions of two pair of nodes. Pairs
212 // correspond to the pickup and delivery pairs defined in the routing model.
213 // Possible neighbor for the paths
214 // 1 -> A -> B -> 2 -> 3 and 4 -> C -> D -> 5
215 // (where (1, 3) and (4, 5) are first and last nodes of the paths and can
216 // therefore not be moved, and (A, B) and (C,D) are pairs of nodes):
217 // 1 -> [C] -> [D] -> 2 -> 3, 4 -> [A] -> [B] -> 5

219 // Operator which exchanges subtrips associated to two pairs of nodes,
220 // see use_relocate_subtrip for a definition of subtrips.
221 OptionalBoolean use_exchange_subtrip = 26;
222 // Operator which cross exchanges the starting chains of 2 paths, including

224 // First and last nodes are not moved.

226 // (where (1, 5) and (6, 8) are first and last nodes of the paths and can
227 // therefore not be moved):
228 // 1 -> [7] -> 3 -> 4 -> 5 6 -> [2] -> 8

238 // 3 -> A ------> B --> C -----> D -> 4 (where A -> B and C -> D are the 2
239 // most expensive arcs, and the chain resulting from breaking them is
240 // B -> C):
241 // 1 -> [B -> C] -> 2 3 -> A -> D -> 4
242 // 1 -> 2 3 -> [B -> C] -> A -> D -> 4
243 // 1 -> 2 3 -> A -> D -> [B -> C] -> 4

245 // --- Intra-route operators ---
246 // Operator which reverses a subchain of a path. It is called TwoOpt
247 // because it breaks two arcs on the path; resulting paths are called
248 // two-optimal.
249 // Possible neighbors for the path 1 -> 2 -> 3 -> 4 -> 5
250 // (where (1, 5) are first and last nodes of the path and can therefore not
251 // be moved):
252 // 1 -> [3 -> 2] -> 4 -> 5
253 // 1 -> [4 -> 3 -> 2] -> 5
254 // 1 -> 2 -> [4 -> 3] -> 5
255 OptionalBoolean use_two_opt = 7;
256 // Operator which moves sub-chains of a path of length 1, 2 and 3 to another
257 // position in the same path.
258 // When the length of the sub-chain is 1, the operator simply moves a node
259 // to another position.
260 // Possible neighbors for the path 1 -> 2 -> 3 -> 4 -> 5, for a sub-chain
261 // length of 2 (where (1, 5) are first and last nodes of the path and can
262 // therefore not be moved):
263 // 1 -> 4 -> [2 -> 3] -> 5
264 // 1 -> [3 -> 4] -> 2 -> 5
265 // The OR_OPT operator is a limited version of 3-Opt (breaks 3 arcs on a
266 // path).
267 OptionalBoolean use_or_opt = 8;
268 // Lin-Kernighan operator.
269 // While the accumulated local gain is positive, performs a 2-OPT or a 3-OPT
270 // move followed by a series of 2-OPT moves. Returns a neighbor for which
271 // the global gain is positive.
272 OptionalBoolean use_lin_kernighan = 9;
273 // Sliding TSP operator.
274 // Uses an exact dynamic programming algorithm to solve the TSP
275 // corresponding to path sub-chains.
276 // For a subchain 1 -> 2 -> 3 -> 4 -> 5 -> 6, solves the TSP on
277 // nodes A, 2, 3, 4, 5, where A is a merger of nodes 1 and 6 such that
278 // cost(A,i) = cost(1,i) and cost(i,A) = cost(i,6).
279 OptionalBoolean use_tsp_opt = 10;
