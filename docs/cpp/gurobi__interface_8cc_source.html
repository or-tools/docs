14// Gurobi backend to MPSolver.
15//
16// Implementation Notes:
17//
18// Incrementalism (last updated June 29, 2020): For solving both LPs and MIPs,
19// Gurobi attempts to reuse information from previous solves, potentially
20// giving a faster solve time. MPSolver supports this for the following problem
21// modification types:
22// * Adding a variable,
23// * Adding a linear constraint,
24// * Updating a variable bound,
25// * Updating an objective coefficient or the objective offset (note that in
26// Gurobi 7.5 LP solver, there is a bug if you update only the objective
27// offset and nothing else).
28// * Updating a coefficient in the constraint matrix.
29// * Updating the type of variable (integer, continuous)
30// * Changing the optimization direction.
31// Updates of the following types will force a resolve from scratch:
32// * Updating the upper or lower bounds of a linear constraint. Note that in
33// MPSolver's model, this includes updating the sense (le, ge, eq, range) of
34// a linear constraint.
35// * Clearing a constraint
36// Any model containing indicator constraints is considered "non-incremental"
37// and will always solve from scratch.
38//
39// The above limitations are largely due MPSolver and this file, not Gurobi.
40//
41// Warning(rander): the interactions between callbacks and incrementalism are
42// poorly tested, proceed with caution.
43//

424double GurobiMPCallbackContext::VariableValue(const MPVariable* variable) {
425 CHECK(variable != nullptr);
426 if (!variable_values_extracted_) {
427 const MPCallbackEvent where = Event();
428 CHECK(where == MPCallbackEvent::kMipSolution ||
429 where == MPCallbackEvent::kMipNode)
430 << "You can only call VariableValue at "
431 << ToString(MPCallbackEvent::kMipSolution) << " or "
432 << ToString(MPCallbackEvent::kMipNode)
433 << " but called from: " << ToString(where);
434 const int gurobi_get_var_param = where == MPCallbackEvent::kMipNode
435 ? GRB_CB_MIPNODE_REL
436 : GRB_CB_MIPSOL_SOL;
437
438 gurobi_variable_values_.resize(num_gurobi_vars_);
439 CheckedGurobiCall(GRBcbget(
440 current_gurobi_internal_callback_context_.gurobi_internal_callback_data,
441 current_gurobi_internal_callback_context_.where, gurobi_get_var_param,
442 static_cast<void*>(gurobi_variable_values_.data())));
443 variable_values_extracted_ = true;
444 }
445 return gurobi_variable_values_[mp_var_to_gurobi_var_->at(variable->index())];
446}

448template <typename GRBConstraintFunction>
449void GurobiMPCallbackContext::AddGeneratedConstraint(
450 const LinearRange& linear_range,
451 GRBConstraintFunction grb_constraint_function) {
452 std::vector<int> variable_indices;
453 std::vector<double> variable_coefficients;
454 const int num_terms = linear_range.linear_expr().terms().size();
455 variable_indices.reserve(num_terms);
456 variable_coefficients.reserve(num_terms);
457 for (const auto& var_coef_pair : linear_range.linear_expr().terms()) {
458 variable_indices.push_back(
459 mp_var_to_gurobi_var_->at(var_coef_pair.first->index()));
460 variable_coefficients.push_back(var_coef_pair.second);
461 }
462 if (std::isfinite(linear_range.upper_bound())) {
463 CheckedGurobiCall(grb_constraint_function(
464 current_gurobi_internal_callback_context_.gurobi_internal_callback_data,
465 variable_indices.size(), variable_indices.data(),
466 variable_coefficients.data(), GRB_LESS_EQUAL,
467 linear_range.upper_bound()));
468 }
469 if (std::isfinite(linear_range.lower_bound())) {
470 CheckedGurobiCall(grb_constraint_function(
471 current_gurobi_internal_callback_context_.gurobi_internal_callback_data,
472 variable_indices.size(), variable_indices.data(),
473 variable_coefficients.data(), GRB_GREATER_EQUAL,
474 linear_range.lower_bound()));
475 }
476}

499double GurobiMPCallbackContext::SuggestSolution(
500 const absl::flat_hash_map<const MPVariable*, double>& solution) {
501 const MPCallbackEvent where = Event();
502 CHECK(where == MPCallbackEvent::kMipNode)
503 << "Feasible solutions can only be added at MIP_NODE, tried to add "
504 "solution at: "
505 << ToString(where);
506
507 std::vector<double> full_solution(num_gurobi_vars_, GRB_UNDEFINED);
508 for (const auto& variable_value : solution) {
509 const MPVariable* var = variable_value.first;
510 full_solution[mp_var_to_gurobi_var_->at(var->index())] =
511 variable_value.second;
512 }
513
514 double objval;
515 CheckedGurobiCall(GRBcbsolution(
516 current_gurobi_internal_callback_context_.gurobi_internal_callback_data,
517 full_solution.data(), &objval));
518
519 return objval;
520}

527// NOTE(user): This function must have this exact API, because we are passing
528// it to Gurobi as a callback.

530 void* gurobi_internal_callback_data, int where,
531 void* raw_model_and_callback) {
532 MPCallbackWithGurobiContext* const callback_with_context =
533 static_cast<MPCallbackWithGurobiContext*>(raw_model_and_callback);
534 CHECK(callback_with_context != nullptr);
535 CHECK(callback_with_context->context != nullptr);
536 CHECK(callback_with_context->callback != nullptr);

538 model, gurobi_internal_callback_data, where};
539 callback_with_context->context->UpdateFromGurobiState(
540 gurobi_internal_context);
541 callback_with_context->callback->RunCallback(callback_with_context->context);
542 return 0;
543}

638void GurobiInterface::Reset() {
639 // We hold calls to GRBterminate() until the new model_ is ready.
640 const absl::MutexLock lock(&hold_interruptions_mutex_);
641
642 GRBmodel* old_model = model_;
643 CheckedGurobiCall(GRBnewmodel(global_env_, &model_, solver_->name_.c_str(),
644 0, // numvars
645 nullptr, // obj
646 nullptr, // lb
647 nullptr, // ub
648 nullptr, // vtype
649 nullptr)); // varnames

651 // Copy all existing parameters from the previous model to the new one. This
652 // ensures that if a user calls multiple times
653 // SetSolverSpecificParametersAsString() and then Reset() is called, we still
654 // take into account all parameters.
655 //
656 // The current code only reapplies the parameters stored in
657 // solver_specific_parameter_string_ at the start of the solve; other
658 // parameters set by previous calls are only kept in the Gurobi model.
659 CheckedGurobiCall(GRBcopyparams(GRBgetenv(model_), GRBgetenv(old_model)));
660
661 CheckedGurobiCall(GRBfreemodel(old_model));
662 old_model = nullptr;
663
664 ResetExtractionInformation();
665 mp_var_to_gurobi_var_.clear();
666 mp_cons_to_gurobi_linear_cons_.clear();
667 num_gurobi_vars_ = 0;
668 num_gurobi_linear_cons_ = 0;
669 had_nonincremental_change_ = false;

837MPSolver::BasisStatus GurobiInterface::TransformGRBConstraintBasisStatus(
838 int gurobi_basis_status, int constraint_index) const {
839 const int grb_index = mp_cons_to_gurobi_linear_cons_.at(constraint_index);
840 if (grb_index < 0) {
841 LOG(DFATAL) << "Basis status not available for nonlinear constraints.";
842 return MPSolver::FREE;
843 }
844 switch (gurobi_basis_status) {
845 case GRB_BASIC:
846 return MPSolver::BASIC;
847 default: {
848 // Non basic.
849 double tolerance = 0.0;
850 CheckedGurobiCall(GRBgetdblparam(GRBgetenv(model_),
851 GRB_DBL_PAR_FEASIBILITYTOL, &tolerance));
852 const double slack = GetDoubleAttrElement(GRB_DBL_ATTR_SLACK, grb_index);
853 const char sense = GetCharAttrElement(GRB_CHAR_ATTR_SENSE, grb_index);
854 VLOG(4) << "constraint " << constraint_index << " , slack = " << slack
855 << " , sense = " << sense;
856 if (fabs(slack) <= tolerance) {
857 switch (sense) {
858 case GRB_EQUAL:
859 case GRB_LESS_EQUAL:
860 return MPSolver::AT_UPPER_BOUND;
861 case GRB_GREATER_EQUAL:
862 return MPSolver::AT_LOWER_BOUND;
863 default:
864 return MPSolver::FREE;
865 }
866 } else {
867 return MPSolver::FREE;
868 }
869 }
870 }

914// Extracts new variables.
915void GurobiInterface::ExtractNewVariables() {
916 const int total_num_vars = solver_->variables_.size();
917 if (total_num_vars > last_variable_index_) {
918 // Define new variables.
919 for (int j = last_variable_index_; j < total_num_vars; ++j) {
920 const MPVariable* const var = solver_->variables_.at(j);
921 set_variable_as_extracted(var->index(), true);
922 CheckedGurobiCall(GRBaddvar(
923 model_, 0, // numnz
924 nullptr, // vind
925 nullptr, // vval
926 solver_->objective_->GetCoefficient(var), var->lb(), var->ub(),
927 var->integer() && mip_ ? GRB_INTEGER : GRB_CONTINUOUS,
928 var->name().empty() ? nullptr : var->name().c_str()));
929 mp_var_to_gurobi_var_.push_back(num_gurobi_vars_++);
930 }
931 CheckedGurobiCall(GRBupdatemodel(model_));
932 // Add new variables to existing constraints.
933 std::vector<int> grb_cons_ind;
934 std::vector<int> grb_var_ind;
935 std::vector<double> coef;
936 for (int i = 0; i < last_constraint_index_; ++i) {
937 // If there was a nonincremental change/the model is not incremental (e.g.
938 // there is an indicator constraint), we should never enter this loop, as
939 // last_variable_index_ will be reset to zero before ExtractNewVariables()
940 // is called.
941 MPConstraint* const ct = solver_->constraints_[i];
942 const int grb_ct_idx = mp_cons_to_gurobi_linear_cons_.at(ct->index());
943 DCHECK_GE(grb_ct_idx, 0);
944 DCHECK(ct->indicator_variable() == nullptr);
945 for (const auto& entry : ct->coefficients_) {
946 const int var_index = entry.first->index();
947 DCHECK(variable_is_extracted(var_index));
948
949 if (var_index >= last_variable_index_) {
950 grb_cons_ind.push_back(grb_ct_idx);
951 grb_var_ind.push_back(mp_var_to_gurobi_var_.at(var_index));
952 coef.push_back(entry.second);
953 }
954 }
955 }
956 if (!grb_cons_ind.empty()) {
957 CheckedGurobiCall(GRBchgcoeffs(model_, grb_cons_ind.size(),
958 grb_cons_ind.data(), grb_var_ind.data(),
959 coef.data()));
960 }
961 }
962 CheckedGurobiCall(GRBupdatemodel(model_));
963 DCHECK_EQ(GetIntAttr(GRB_INT_ATTR_NUMVARS), num_gurobi_vars_);
964}
965
966void GurobiInterface::ExtractNewConstraints() {
967 int total_num_rows = solver_->constraints_.size();
968 if (last_constraint_index_ < total_num_rows) {
969 // Add each new constraint.
970 for (int row = last_constraint_index_; row < total_num_rows; ++row) {
971 MPConstraint* const ct = solver_->constraints_[row];
972 set_constraint_as_extracted(row, true);
973 const int size = ct->coefficients_.size();
974 std::vector<int> grb_vars;
975 std::vector<double> coefs;
976 grb_vars.reserve(size);
977 coefs.reserve(size);
978 for (const auto& entry : ct->coefficients_) {
979 const int var_index = entry.first->index();
980 CHECK(variable_is_extracted(var_index));
981 grb_vars.push_back(mp_var_to_gurobi_var_.at(var_index));
982 coefs.push_back(entry.second);
983 }
984 char* const name =
985 ct->name().empty() ? nullptr : const_cast<char*>(ct->name().c_str());
986 if (ct->indicator_variable() != nullptr) {
987 const int grb_ind_var =
988 mp_var_to_gurobi_var_.at(ct->indicator_variable()->index());
989 if (ct->lb() > -std::numeric_limits<double>::infinity()) {
990 CheckedGurobiCall(GRBaddgenconstrIndicator(
991 model_, name, grb_ind_var, ct->indicator_value(), size,
992 grb_vars.data(), coefs.data(),
993 ct->ub() == ct->lb() ? GRB_EQUAL : GRB_GREATER_EQUAL, ct->lb()));
994 }
995 if (ct->ub() < std::numeric_limits<double>::infinity() &&
996 ct->lb() != ct->ub()) {
997 CheckedGurobiCall(GRBaddgenconstrIndicator(
998 model_, name, grb_ind_var, ct->indicator_value(), size,
999 grb_vars.data(), coefs.data(), GRB_LESS_EQUAL, ct->ub()));
1000 }
1001 mp_cons_to_gurobi_linear_cons_.push_back(-1);
1002 } else {
1003 // Using GRBaddrangeconstr for constraints that don't require it adds
1004 // a slack which is not always removed by presolve.
1005 if (ct->lb() == ct->ub()) {
1006 CheckedGurobiCall(GRBaddconstr(model_, size, grb_vars.data(),
1007 coefs.data(), GRB_EQUAL, ct->lb(),
1008 name));
1009 } else if (ct->lb() == -std::numeric_limits<double>::infinity()) {
1010 CheckedGurobiCall(GRBaddconstr(model_, size, grb_vars.data(),
1011 coefs.data(), GRB_LESS_EQUAL, ct->ub(),
1012 name));
1013 } else if (ct->ub() == std::numeric_limits<double>::infinity()) {
1014 CheckedGurobiCall(GRBaddconstr(model_, size, grb_vars.data(),
1015 coefs.data(), GRB_GREATER_EQUAL,
1016 ct->lb(), name));
1017 } else {
1018 CheckedGurobiCall(GRBaddrangeconstr(model_, size, grb_vars.data(),
1019 coefs.data(), ct->lb(), ct->ub(),
1020 name));
1021 // NOTE(user): range constraints implicitly add an extra variable
1022 // to the model.
1023 num_gurobi_vars_++;
1024 }
1025 mp_cons_to_gurobi_linear_cons_.push_back(num_gurobi_linear_cons_++);
1026 }
1027 }
1028 }
1029 CheckedGurobiCall(GRBupdatemodel(model_));
1030 DCHECK_EQ(GetIntAttr(GRB_INT_ATTR_NUMCONSTRS), num_gurobi_linear_cons_);
1031}

1156MPSolver::ResultStatus GurobiInterface::Solve(const MPSolverParameters& param) {
1157 WallTimer timer;
1158 timer.Start();
1159
1160 if (param.GetIntegerParam(MPSolverParameters::INCREMENTALITY) ==
1161 MPSolverParameters::INCREMENTALITY_OFF ||
1162 ModelIsNonincremental() || had_nonincremental_change_) {
1163 Reset();
1164 }
1165
1166 // Set log level.
1167 CheckedGurobiCall(
1168 GRBsetintparam(GRBgetenv(model_), GRB_INT_PAR_OUTPUTFLAG, !quiet_));
1169
1170 ExtractModel();
1171 // Sync solver.
1172 CheckedGurobiCall(GRBupdatemodel(model_));
1173 VLOG(1) << absl::StrFormat("Model built in %s.",
1174 absl::FormatDuration(timer.GetDuration()));
1175
1176 // Set solution hints if any.
1177 for (const std::pair<const MPVariable*, double>& p :
1178 solver_->solution_hint_) {
1179 SetDoubleAttrElement(GRB_DBL_ATTR_START,
1180 mp_var_to_gurobi_var_.at(p.first->index()), p.second);
1181 }
1182
1183 // Pass branching priority annotations if at least one has been updated.
1184 if (update_branching_priorities_) {
1185 for (const MPVariable* var : solver_->variables_) {
1186 SetIntAttrElement(GRB_INT_ATTR_BRANCHPRIORITY,
1187 mp_var_to_gurobi_var_.at(var->index()),
1188 var->branching_priority());
1189 }
1190 update_branching_priorities_ = false;
1191 }
1192
1193 // Time limit.
1194 if (solver_->time_limit() != 0) {
1195 VLOG(1) << "Setting time limit = " << solver_->time_limit() << " ms.";
1196 CheckedGurobiCall(GRBsetdblparam(GRBgetenv(model_), GRB_DBL_PAR_TIMELIMIT,
1197 solver_->time_limit_in_secs()));
1198 }
1199
1200 // We first set our internal MPSolverParameters from 'param' and then set
1201 // any user-specified internal solver parameters via
1202 // solver_specific_parameter_string_.
1203 // Default MPSolverParameters can override custom parameters (for example for
1204 // presolving) and therefore we apply MPSolverParameters first.
1205 SetParameters(param);
1206 solver_->SetSolverSpecificParametersAsString(
1207 solver_->solver_specific_parameter_string_);
1208
1209 std::unique_ptr<GurobiMPCallbackContext> gurobi_context;
1210 MPCallbackWithGurobiContext mp_callback_with_context;
1211 int gurobi_precrush = 0;
1212 int gurobi_lazy_constraint = 0;
1213 if (callback_ == nullptr) {
1214 CheckedGurobiCall(GRBsetcallbackfunc(model_, nullptr, nullptr));
1215 } else {
1216 gurobi_context = std::make_unique<GurobiMPCallbackContext>(
1217 global_env_, &mp_var_to_gurobi_var_, num_gurobi_vars_,

1220 mp_callback_with_context.callback = callback_;
1221 CheckedGurobiCall(GRBsetcallbackfunc(
1222 model_, CallbackImpl, static_cast<void*>(&mp_callback_with_context)));
1223 gurobi_precrush = callback_->might_add_cuts();
1224 gurobi_lazy_constraint = callback_->might_add_lazy_constraints();
1225 }
1226 CheckedGurobiCall(
1227 GRBsetintparam(GRBgetenv(model_), GRB_INT_PAR_PRECRUSH, gurobi_precrush));
1228 CheckedGurobiCall(GRBsetintparam(
1229 GRBgetenv(model_), GRB_INT_PAR_LAZYCONSTRAINTS, gurobi_lazy_constraint));
1230
1231 // Logs all parameters not at default values in the model environment.
1232 if (!quiet()) {
1233 std::cout << GurobiParamInfoForLogging(GRBgetenv(model_),
1234 /*one_liner_output=*/true);
1235 }
1236
1237 // Solve
1238 timer.Restart();
1239 const int status = GRBoptimize(model_);
1240
1241 if (status) {
1242 VLOG(1) << "Failed to optimize MIP." << GRBgeterrormsg(global_env_);
1243 } else {
1244 VLOG(1) << absl::StrFormat("Solved in %s.",
1245 absl::FormatDuration(timer.GetDuration()));
1246 }
1247
1248 // Get the status.
1249 const int optimization_status = GetIntAttr(GRB_INT_ATTR_STATUS);
1250 VLOG(1) << absl::StrFormat("Solution status %d.\n", optimization_status);
1251 const int solution_count = SolutionCount();
1252
1253 switch (optimization_status) {
1254 case GRB_OPTIMAL:
1255 result_status_ = MPSolver::OPTIMAL;
1256 break;
1257 case GRB_INFEASIBLE:
1258 result_status_ = MPSolver::INFEASIBLE;
1259 break;
1260 case GRB_UNBOUNDED:
1261 result_status_ = MPSolver::UNBOUNDED;
1262 break;
1263 case GRB_INF_OR_UNBD:
1264 // TODO(user): We could introduce our own "infeasible or
1265 // unbounded" status.
1266 result_status_ = MPSolver::INFEASIBLE;
1267 break;
1268 default: {
1269 if (solution_count > 0) {
1270 result_status_ = MPSolver::FEASIBLE;
1271 } else {
1272 result_status_ = MPSolver::NOT_SOLVED;
1273 }
1274 break;
1275 }
1276 }
1277
1278 if (IsMIP() && (result_status_ != MPSolver::UNBOUNDED &&
1279 result_status_ != MPSolver::INFEASIBLE)) {
1280 const int error =
1281 GRBgetdblattr(model_, GRB_DBL_ATTR_OBJBOUND, &best_objective_bound_);
1282 LOG_IF(WARNING, error != 0)
1283 << "Best objective bound is not available, error=" << error
1284 << ", message=" << GRBgeterrormsg(global_env_);
1285 VLOG(1) << "best bound = " << best_objective_bound_;
1286 }
1287
1288 if (solution_count > 0 && (result_status_ == MPSolver::FEASIBLE ||
1289 result_status_ == MPSolver::OPTIMAL)) {
1290 current_solution_index_ = 0;
1291 // Get the results.
1292 objective_value_ = GetDoubleAttr(GRB_DBL_ATTR_OBJVAL);
1293 VLOG(1) << "objective = " << objective_value_;
1294
1295 {
1296 const std::vector<double> grb_variable_values =
1297 GetDoubleAttrArray(GRB_DBL_ATTR_X, num_gurobi_vars_);
1298 for (int i = 0; i < solver_->variables_.size(); ++i) {
1299 MPVariable* const var = solver_->variables_[i];
1300 const double val = grb_variable_values.at(mp_var_to_gurobi_var_.at(i));
1301 var->set_solution_value(val);
1302 VLOG(3) << var->name() << ", value = " << val;
1303 }
1304 }
1305 if (!mip_) {
1306 {
1307 const std::vector<double> grb_reduced_costs =
1308 GetDoubleAttrArray(GRB_DBL_ATTR_RC, num_gurobi_vars_);
1309 for (int i = 0; i < solver_->variables_.size(); ++i) {
1310 MPVariable* const var = solver_->variables_[i];
1311 const double rc = grb_reduced_costs.at(mp_var_to_gurobi_var_.at(i));
1312 var->set_reduced_cost(rc);
1313 VLOG(4) << var->name() << ", reduced cost = " << rc;
1314 }
1315 }
1316
1317 {
1318 std::vector<double> grb_dual_values =
1319 GetDoubleAttrArray(GRB_DBL_ATTR_PI, num_gurobi_linear_cons_);
1320 for (int i = 0; i < solver_->constraints_.size(); ++i) {
1321 MPConstraint* const ct = solver_->constraints_[i];
1322 const double dual_value =
1323 grb_dual_values.at(mp_cons_to_gurobi_linear_cons_.at(i));
1324 ct->set_dual_value(dual_value);
1325 VLOG(4) << "row " << ct->index() << ", dual value = " << dual_value;
1326 }
1327 }
1328 }
1329 }
1330
1331 sync_status_ = SOLUTION_SYNCHRONIZED;
1332 GRBresetparams(GRBgetenv(model_));
1333 return result_status_;
1334}

1359 // Next solution only supported for MIP
1360 if (!mip_) return false;
1361
1362 // Make sure we have successfully solved the problem and not modified it.
1363 if (!CheckSolutionIsSynchronizedAndExists()) {
1364 return false;
1365 }

1367 if (current_solution_index_ + 1 >= SolutionCount()) {

1370 current_solution_index_++;
1371
1372 CheckedGurobiCall(GRBsetintparam(
1373 GRBgetenv(model_), GRB_INT_PAR_SOLUTIONNUMBER, current_solution_index_));
1374
1375 objective_value_ = GetDoubleAttr(GRB_DBL_ATTR_POOLOBJVAL);
1376 const std::vector<double> grb_variable_values =
1377 GetDoubleAttrArray(GRB_DBL_ATTR_XN, num_gurobi_vars_);
1378
1379 for (int i = 0; i < solver_->variables_.size(); ++i) {
1380 MPVariable* const var = solver_->variables_[i];

1383 }
1384 // TODO(user): This reset may not be necessary, investigate.
1385 GRBresetparams(GRBgetenv(model_));
