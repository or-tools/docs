17// An object oriented wrapper for variables in ModelStorage (used internally by
18// Model) with support for arithmetic operations to build linear expressions and
19// express linear constraints.
20//
21// Types are:
22// - Variable: a reference to a variable of an ModelStorage.
23//
24// - LinearExpression: a weighted sum of variables with an optional offset;
25// something like `3*x + 2*y + 5`.
26//
27// - LinearTerm: a term of a linear expression, something like `2*x`. It is
28// used as an intermediate in the arithmetic operations that builds linear
29// expressions.
30//
31// - (Lower|Upper)BoundedLinearExpression: two classes representing the result
32// of the comparison of a LinearExpression with a constant. For example `3*x
33// + 2*y + 5 >= 3`.
34//
35// - BoundedLinearExpression: the result of the comparison of a linear
36// expression with two bounds, an upper bound and a lower bound. For example
37// `2 <= 3*x + 2*y + 5 <= 3`; or `4 >= 3*x + 2*y + 5 >= 1`.
38//
39// - QuadraticTermKey: a key used internally to represent a pair of Variables.
40//
41// - QuadraticTerm: a term representing the product of a scalar coefficient
42// and two Variables (possibly the same); something like `2*x*y` or `3*x*x`.
43// It is used as an intermediate in the arithmetic operations that build
44// quadratic expressions.
45//
46// - QuadraticExpression: a sum of a quadratic terms, linear terms, and a
47// scalar offset; something like `3*x*y + 2*x*x + 4x + 5`.
48//
49// - VariablesEquality: the result of comparing two Variable instances with
50// the == operator. For example `a == b`. This intermediate class support
51// implicit conversion to both bool and BoundedLinearExpression types. This
52// enables using variables as key of maps (using the conversion to bool)
53// without preventing adding constraints of variable equality.
54//
55// The basic arithmetic operators are overloaded for those types so that we can
56// write math expressions with variables to build linear expressions. The >=, <=
57// and == comparison operators are overloaded to produce BoundedLinearExpression
58// that can be used to build constraints.
59//
60// For example we can have:
61// const Variable x = ...;
62// const Variable y = ...;
63// const LinearExpression expr = 2 * x + 3 * y - 2;
64// const BoundedLinearExpression bounded_expr = 1 <= 2 * x + 3 * y - 2 <= 10;
65//
66// To making working with containers of doubles/Variables/LinearExpressions
67// easier, the template methods Sum() and InnerProduct() are provided, e.g.
68// const std::vector<int> ints = ...;
69// const std::vector<double> doubles = ...;
70// const std::vector<Variable> vars = ...;
71// const std::vector<LinearTerm> terms = ...;
72// const std::vector<LinearExpression> exprs = ...;
73// const LinearExpression s1 = Sum(ints);
74// const LinearExpression s2 = Sum(doubles);
75// const LinearExpression s3 = Sum(vars);
76// const LinearExpression s4 = Sum(terms);
77// const LinearExpression s5 = Sum(exprs);
78// const LinearExpression p1 = InnerProduct(ints, vars);
79// const LinearExpression p2 = InnerProduct(terms, doubles);
80// const LinearExpression p3 = InnerProduct(doubles, exprs);
81// These methods work on any iterable type (defining begin() and end()). For
82// InnerProduct, the inputs must be of equal size, and a compile time error will
83// be generated unless at least one input is a container of a type implicitly
84// convertible to double.
85//
86// Pre C++20, avoid the use of std::accumulate and std::inner_product with
87// LinearExpression, they cause a quadratic blowup in running time.
88//
89// While there is some complexity in the source, users typically should not need
90// to look at types other than Variable and LinearExpression too closely. Their
91// code usually will only refer to those types.

265 // Adds each element of items to this.
266 //
267 // Specifically, letting
268 // (i_1, i_2, ..., i_n) = items
269 // adds
270 // i_1 + i_2 + ... + i_n
271 // to this.
272 //
273 // Example:
274 // const Variable a = ...;
275 // const Variable b = ...;
276 // const std::vector<Variable> vars = {a, b};
277 // LinearExpression expr(8.0);
278 // expr.AddSum(vars);
279 // Results in expr having the value a + b + 8.0.
280 //
281 // Compile time requirements:
282 // * Iterable is a sequence (an array or object with begin() and end()).
283 // * The type of an element of items is one of double, Variable, LinearTerm
284 // or LinearExpression (or is implicitly convertible to one of these types,
285 // e.g. int).
286 //
287 // Note: The implementation is equivalent to:
288 // for(const auto item : items) {
289 // *this += item;
290 // }
291 template <typename Iterable>
292 inline void AddSum(const Iterable& items);
293
294 // Creates a new LinearExpression object equal to the sum. The implementation
295 // is equivalent to:
296 // LinearExpression expr;
297 // expr.AddSum(items);
298 template <typename Iterable>
299 static inline LinearExpression Sum(const Iterable& items);

301 // Adds the inner product of left and right to this.
302 //
303 // Specifically, letting
304 // (l_1, l_2 ..., l_n) = left,
305 // (r_1, r_2, ..., r_n) = right,
306 // adds
307 // l_1 * r_1 + l_2 * r_2 + ... + l_n * r_n
308 // to this.
309 //
310 // Example:
311 // const Variable a = ...;
312 // const Variable b = ...;
313 // const std::vector<Variable> left = {a, b};
314 // const std::vector<double> right = {10.0, 2.0};
315 // LinearExpression expr(3.0);
316 // expr.AddInnerProduct(left, right)
317 // Results in expr having the value 10.0 * a + 2.0 * b + 3.0.
318 //
319 // Compile time requirements:
320 // * LeftIterable and RightIterable are both sequences (arrays or objects
321 // with begin() and end())
322 // * For both left and right, their elements a type of either double,
323 // Variable, LinearTerm or LinearExpression (or type implicitly convertible
324 // to one of these types, e.g. int).
325 // * At least one of left or right has elements with type double (or a type
326 // implicitly convertible, e.g. int).
327 // Runtime requirements (or CHECK fails):
328 // * left and right have an equal number of elements.
329 //
330 // Note: The implementation is equivalent to the following pseudocode:
331 // for(const auto& [l, r] : zip(left, right)) {
332 // *this += l * r;
333 // }
334 // In particular, the multiplication will be performed on the types of the
335 // elements in left and right (take care with low precision types), but the
336 // addition will always use double precision.
337 template <typename LeftIterable, typename RightIterable>
338 inline void AddInnerProduct(const LeftIterable& left,
339 const RightIterable& right);
340
341 // Creates a new LinearExpression object equal to the inner product. The
342 // implementation is equivalent to:
343 // LinearExpression expr;
344 // expr.AddInnerProduct(left, right);
345 template <typename LeftIterable, typename RightIterable>
346 static inline LinearExpression InnerProduct(const LeftIterable& left,
347 const RightIterable& right);

791 // Adds each element of items to this.
792 //
793 // Specifically, letting
794 // (i_1, i_2, ..., i_n) = items
795 // adds
796 // i_1 + i_2 + ... + i_n
797 // to this.
798 //
799 // Example:
800 // const Variable a = ...;
801 // const Variable b = ...;
802 // const std::vector<Variable> vars = {a, b};
803 // const std::vector<QuadraticTerm> terms = {2 * a * b};
804 // QuadraticExpression expr = 8;
805 // expr.AddSum(vars);
806 // expr.AddSum(terms);
807 // Results in expr having the value 2 * a * b + a + b + 8.0.
808 //
809 // Compile time requirements:
810 // * Iterable is a sequence (an array or object with begin() and end()).
811 // * The type of an element of items is one of double, Variable, LinearTerm,
812 // LinearExpression, QuadraticTerm, or QuadraticExpression (or is
813 // implicitly convertible to one of these types, e.g. int).
814 //
815 // Note: The implementation is equivalent to:
816 // for(const auto item : items) {
817 // *this += item;
818 // }
819 template <typename Iterable>
820 inline void AddSum(const Iterable& items);

822 // Returns the sum of the elements of items.
823 //
824 // Specifically, letting
825 // (i_1, i_2, ..., i_n) = items
826 // returns
827 // i_1 + i_2 + ... + i_n.
828 //
829 // Example:
830 // const Variable a = ...;
831 // const Variable b = ...;
832 // const std::vector<QuadraticTerm> terms = {a * a, 2 * a * b, 3 * b * a};
833 // QuadraticExpression::Sum(vars)
834 // => a^2 + 5 a * b
835 // Note, instead of:
836 // QuadraticExpression expr(3.0);
837 // expr += QuadraticExpression::Sum(items);
838 // Prefer:
839 // expr.AddSum(items);
840 //
841 // See QuadraticExpression::AddSum() for a precise contract on the type
842 // Iterable.
843 template <typename Iterable>
844 static inline QuadraticExpression Sum(const Iterable& items);

846 // Adds the inner product of left and right to this.
847 //
848 // Specifically, letting
849 // (l_1, l_2 ..., l_n) = left,
850 // (r_1, r_2, ..., r_n) = right,
851 // adds
852 // l_1 * r_1 + l_2 * r_2 + ... + l_n * r_n
853 // to this.
854 //
855 // Example:
856 // const Variable a = ...;
857 // const Variable b = ...;
858 // const std::vector<Variable> vars = {a, b};
859 // const std::vector<double> coeffs = {10.0, 2.0};
860 // QuadraticExpression expr = 3.0;
861 // expr.AddInnerProduct(coeffs, vars);
862 // expr.AddInnerProduct(vars, vars);
863 // Results in expr having the value a^2 + b^2 + 10.0 * a + 2.0 * b + 3.0.
864 //
865 // Compile time requirements:
866 // * LeftIterable and RightIterable are both sequences (arrays or objects
867 // with begin() and end())
868 // * For both left and right, their elements are of type double, Variable,
869 // LinearTerm, LinearExpression, QuadraticTerm, or QuadraticExpression (or
870 // is implicitly convertible to one of these types, e.g. int).
871 // Runtime requirements (or CHECK fails):
872 // * The inner product value, and its constitutive intermediate terms, can be
873 // represented as a QuadraticExpression (potentially through an implicit
874 // conversion).
875 // * left and right have an equal number of elements.
876 //
877 // Note: The implementation is equivalent to the following pseudocode:
878 // for(const auto& [l, r] : zip(left, right)) {
879 // *this += l * r;
880 // }
881 // In particular, the multiplication will be performed on the types of the
882 // elements in left and right (take care with low precision types), but the
883 // addition will always use double precision.
884 template <typename LeftIterable, typename RightIterable>
885 inline void AddInnerProduct(const LeftIterable& left,
886 const RightIterable& right);

888 // Returns the inner product of left and right.
889 //
890 // Specifically, letting
891 // (l_1, l_2 ..., l_n) = left,
892 // (r_1, r_2, ..., r_n) = right,
893 // returns
894 // l_1 * r_1 + l_2 * r_2 + ... + l_n * r_n.
895 //
896 // Example:
897 // const Variable a = ...;
898 // const Variable b = ...;
899 // const std::vector<Variable> left = {a, a};
900 // const std::vector<Variable> left = {a, b};
901 // QuadraticExpression::InnerProduct(left, right);

904 // QuadraticExpression expr(3.0);

906 // Prefer:
907 // expr.AddInnerProduct(left, right);

910 // QuadraticExpression::AddInnerProduct() for a precise contract on template
911 // types.

1667void AddInnerProduct(const LeftIterable& left, const RightIterable& right,
1668 Expression& expr) {

1670 using std::end;
1671 auto l = begin(left);
1672 auto r = begin(right);
1673 const auto l_end = end(left);
1674 const auto r_end = end(right);

1678 CHECK(l == l_end)
1679 << "left had more elements than right, sizes should be equal";
1680 CHECK(r == r_end)
