48// This is how the SatSolver stores a clause. A clause is just a disjunction of
49// literals. In many places, we just use vector<literal> to encode one. But in
50// the critical propagation code, we use this class to remove one memory
51// indirection.
52class SatClause {
53 public:
54 // Creates a sat clause. There must be at least 2 literals.
55 // Clause with one literal fix variable directly and are never constructed.
56 // Note that in practice, we use BinaryImplicationGraph for the clause of size
57 // 2, so this is used for size at least 3.
58 static SatClause* Create(absl::Span<const Literal> literals);
59
60 // Non-sized delete because this is a tail-padded class.
61 void operator delete(void* p) {
62 ::operator delete(p); // non-sized delete
63 }
64
65 // Number of literals in the clause.
66 int size() const { return size_; }
67
68 // We re-use the size to lazily remove clause and notify that they need to be
69 // deleted. It is why this is not called empty() to emphasis that fact. Note
70 // that we never create an initially empty clause, so there is no confusion
71 // with an infeasible model with an empty clause inside.
72 int IsRemoved() const { return size_ == 0; }
73
74 // Allows for range based iteration: for (Literal literal : clause) {}.
75 const Literal* begin() const { return &(literals_[0]); }
76 const Literal* end() const { return &(literals_[size_]); }
77
78 // Returns the first and second literals. These are always the watched
79 // literals if the clause is attached in the LiteralWatchers.
80 Literal FirstLiteral() const { return literals_[0]; }
81 Literal SecondLiteral() const { return literals_[1]; }
82
83 // Returns the literal that was propagated to true. This only works for a
84 // clause that just propagated this literal. Otherwise, this will just returns
85 // a literal of the clause.
86 Literal PropagatedLiteral() const { return literals_[0]; }
87
88 // Returns the reason for the last unit propagation of this clause. The
89 // preconditions are the same as for PropagatedLiteral(). Note that we don't
90 // need to include the propagated literal.
91 absl::Span<const Literal> PropagationReason() const {
92 return absl::Span<const Literal>(&(literals_[1]), size_ - 1);
93 }
94
95 // Returns a Span<> representation of the clause.
96 absl::Span<const Literal> AsSpan() const {
97 return absl::Span<const Literal>(&(literals_[0]), size_);
98 }
99
100 // Removes literals that are fixed. This should only be called at level 0
101 // where a literal is fixed iff it is assigned. Aborts and returns true if
102 // they are not all false.
103 //
104 // Note that the removed literal can still be accessed in the portion [size,
105 // old_size) of literals().
106 bool RemoveFixedLiteralsAndTestIfTrue(const VariablesAssignment& assignment);
107
108 // Returns true if the clause is satisfied for the given assignment. Note that
109 // the assignment may be partial, so false does not mean that the clause can't
110 // be satisfied by completing the assignment.
111 bool IsSatisfied(const VariablesAssignment& assignment) const;
112
113 std::string DebugString() const;
114
115 private:
116 // The manager needs to permute the order of literals in the clause and
117 // call Clear()/Rewrite.
118 friend class ClauseManager;
119
120 Literal* literals() { return &(literals_[0]); }
121
122 // Marks the clause so that the next call to CleanUpWatchers() can identify it
123 // and actually remove it. We use size_ = 0 for this since the clause will
124 // never be used afterwards.
125 void Clear() { size_ = 0; }
126
127 // Rewrites a clause with another shorter one. Note that the clause shouldn't
128 // be attached when this is called.
129 void Rewrite(absl::Span<const Literal> new_clause) {
130 size_ = 0;
131 for (const Literal l : new_clause) literals_[size_++] = l;
132 }
133
134 int32_t size_;
135
136 // This class store the literals inline, and literals_ mark the starts of the
137 // variable length portion.
138 Literal literals_[0];
139};

246 // Round-robbing selection of the next clause to minimize/probe.
247 // Note that for minimization we only look at clause kept forever.
248 //
249 // TODO(user): If more indices are needed, switch to a generic API.
250 SatClause* NextClauseToMinimize() {
251 for (; to_minimize_index_ < clauses_.size(); ++to_minimize_index_) {
252 if (clauses_[to_minimize_index_]->IsRemoved()) continue;
253 if (!IsRemovable(clauses_[to_minimize_index_])) {
254 return clauses_[to_minimize_index_++];
255 }
256 }
257 return nullptr;
258 }
259 SatClause* NextClauseToProbe() {
260 for (; to_probe_index_ < clauses_.size(); ++to_probe_index_) {
261 if (clauses_[to_probe_index_]->IsRemoved()) continue;
262 return clauses_[to_probe_index_++];
263 }
264 return nullptr;
265 }

297 // Contains, for each literal, the list of clauses that need to be inspected
298 // when the corresponding literal becomes false.
299 struct Watcher {
300 Watcher() = default;
301 Watcher(SatClause* c, Literal b, int i = 2)
302 : blocking_literal(b), start_index(i), clause(c) {}
303
304 // Optimization. A literal from the clause that sometimes allow to not even
305 // look at the clause memory when true.
306 Literal blocking_literal;
307
308 // Optimization. An index in the clause. Instead of looking for another
309 // literal to watch from the start, we will start from here instead, and
310 // loop around if needed. This allows to avoid bad quadratic corner cases
311 // and lead to an "optimal" complexity. See "Optimal Implementation of
312 // Watched Literals and more General Techniques", Ian P. Gent.
313 //
314 // Note that ideally, this should be part of a SatClause, so it can be
315 // shared across watchers. However, since we have 32 bits for "free" here
316 // because of the struct alignment, we store it here instead.
317 int32_t start_index;
318
319 SatClause* clause;
320 };

392// A simple class to manage a set of binary clauses.
393class BinaryClauseManager {
394 public:
395 BinaryClauseManager() = default;
396
397 // This type is neither copyable nor movable.
398 BinaryClauseManager(const BinaryClauseManager&) = delete;
399 BinaryClauseManager& operator=(const BinaryClauseManager&) = delete;
400
401 int NumClauses() const { return set_.size(); }
402
403 // Adds a new binary clause to the manager and returns true if it wasn't
404 // already present.
405 bool Add(BinaryClause c) {
406 std::pair<int, int> p(c.a.SignedValue(), c.b.SignedValue());
407 if (p.first > p.second) std::swap(p.first, p.second);
408 if (set_.find(p) == set_.end()) {
409 set_.insert(p);
410 newly_added_.push_back(c);
411 return true;
412 }
413 return false;
414 }
415
416 // Returns the newly added BinaryClause since the last ClearNewlyAdded() call.
417 const std::vector<BinaryClause>& newly_added() const { return newly_added_; }
418 void ClearNewlyAdded() { newly_added_.clear(); }
419
420 private:
421 absl::flat_hash_set<std::pair<int, int>> set_;
422 std::vector<BinaryClause> newly_added_;
423};

425// Special class to store and propagate clauses of size 2 (i.e. implication).
426// Such clauses are never deleted. Together, they represent the 2-SAT part of
427// the problem. Note that 2-SAT satisfiability is a polynomial problem, but
428// W2SAT (weighted 2-SAT) is NP-complete.
429//
430// TODO(user): Most of the note below are done, but we currently only applies
431// the reduction before the solve. We should consider doing more in-processing.
432// The code could probably still be improved too.
433//
434// Note(user): All the variables in a strongly connected component are
435// equivalent and can be thus merged as one. This is relatively cheap to compute
436// from time to time (linear complexity). We will also get contradiction (a <=>
437// not a) this way. This is done by DetectEquivalences().
438//
439// Note(user): An implication (a => not a) implies that a is false. I am not
440// sure it is worth detecting that because if the solver assign a to true, it
441// will learn that right away. I don't think we can do it faster.
442//
443// Note(user): The implication graph can be pruned. This is called the
444// transitive reduction of a graph. For instance If a => {b,c} and b => {c},
445// then there is no need to store a => {c}. The transitive reduction is unique
446// on an acyclic graph. Computing it will allow for a faster propagation and
447// memory reduction. It is however not cheap. Maybe simple lazy heuristics to
448// remove redundant arcs are better. Note that all the learned clauses we add
449// will never be redundant (but they could introduce cycles). This is done
450// by ComputeTransitiveReduction().
451//
452// Note(user): This class natively support at most one constraints. This is
453// a way to reduced significantly the memory and size of some 2-SAT instances.
454// However, it is not fully exploited for pure SAT problems. See
455// TransformIntoMaxCliques().
456//
457// Note(user): Add a preprocessor to remove duplicates in the implication lists.
458// Note that all the learned clauses we add will never create duplicates.
459//
460// References for most of the above and more:
461// - Brafman RI, "A simplifier for propositional formulas with many binary
462// clauses", IEEE Trans Syst Man Cybern B Cybern. 2004 Feb;34(1):52-9.
463// http://citeseerx.ist.psu.edu/viewdoc/summary?doi=10.1.1.28.4911
464// - Marijn J. H. Heule, Matti Jrvisalo, Armin Biere, "Efficient CNF
465// Simplification Based on Binary Implication Graphs", Theory and Applications
466// of Satisfiability Testing - SAT 2011, Lecture Notes in Computer Science
467// Volume 6695, 2011, pp 201-215
468// http://www.cs.helsinki.fi/u/mjarvisa/papers/heule-jarvisalo-biere.sat11.pdf
469class BinaryImplicationGraph : public SatPropagator {
470 public:
471 explicit BinaryImplicationGraph(Model* model)
472 : SatPropagator("BinaryImplicationGraph"),
473 stats_("BinaryImplicationGraph"),
474 time_limit_(model->GetOrCreate<TimeLimit>()),
475 random_(model->GetOrCreate<ModelRandomGenerator>()),
476 trail_(model->GetOrCreate<Trail>()),
477 at_most_one_max_expansion_size_(
478 model->GetOrCreate<SatParameters>()
479 ->at_most_one_max_expansion_size()) {
480 trail_->RegisterPropagator(this);
481 }

689 // Extract all the binary clauses managed by this class. The Output type must
690 // support an AddBinaryClause(Literal a, Literal b) function.
691 //
692 // Important: This currently does NOT include at most one constraints.
693 //
694 // TODO(user): When extracting to cp_model.proto we could be more efficient
695 // by extracting bool_and constraint with many lhs terms.
696 template <typename Output>
697 void ExtractAllBinaryClauses(Output* out) const {
698 // TODO(user): Ideally we should just never have duplicate clauses in this
699 // class. But it seems we do in some corner cases, so lets not output them
700 // twice.
701 absl::flat_hash_set<std::pair<LiteralIndex, LiteralIndex>>
702 duplicate_detection;
703 for (LiteralIndex i(0); i < implications_.size(); ++i) {
704 const Literal a = Literal(i).Negated();
705 for (const Literal b : implications_[i]) {
706 // Note(user): We almost always have both a => b and not(b) => not(a) in
707 // our implications_ database. Except if ComputeTransitiveReduction()
708 // was aborted early, but in this case, if only one is present, the
709 // other could be removed, so we shouldn't need to output it.
710 if (a < b && duplicate_detection.insert({a, b}).second) {
711 out->AddBinaryClause(a, b);
712 }
713 }
714 }
715 }
