44bool IsNanOrAbsGreaterThanOrEqual(double value, double abs_value_threshold) {
45 return std::isnan(value) || std::abs(value) >= abs_value_threshold;
46}

48// Internal method to detect errors in bounds. The object passed as parameter
49// must have "lower_bound" and "upper_bound" fields.
50template <typename BoundedElement>
51std::string FindErrorInBounds(const BoundedElement& element,
52 double abs_value_threshold,
53 const bool accept_trivially_infeasible_bounds) {
54 if (std::isnan(element.lower_bound()) || std::isnan(element.upper_bound()) ||
55 element.lower_bound() >= abs_value_threshold ||
56 element.upper_bound() <= -abs_value_threshold ||
57 (!accept_trivially_infeasible_bounds &&
58 element.lower_bound() > element.upper_bound())) {
59 return absl::StrFormat("Infeasible bounds: [%f, %f]", element.lower_bound(),
60 element.upper_bound());
61 }
62 return "";
63}

65// Internal method to detect errors in a single variable.
66std::string FindErrorInMPVariable(
67 const MPVariableProto& variable, double abs_value_threshold,
68 const bool accept_trivially_infeasible_bounds) {
69 const std::string bound_error = FindErrorInBounds(
70 variable, abs_value_threshold, accept_trivially_infeasible_bounds);
71 if (!bound_error.empty()) return bound_error;
72
73 if (!accept_trivially_infeasible_bounds && variable.is_integer() &&
74 ceil(variable.lower_bound()) > floor(variable.upper_bound())) {
75 return absl::StrCat(
76 "Infeasible bounds for integer variable: [", (variable.lower_bound()),
77 ", ", (variable.upper_bound()), "]", " translate to the empty set");
78 }
79 if (IsNanOrAbsGreaterThanOrEqual(variable.objective_coefficient(),
80 abs_value_threshold)) {
81 return absl::StrCat("Invalid objective_coefficient: ",
82 (variable.objective_coefficient()));
83 }
84 return std::string();
85}

87// Returns an error message if 'var_indices' contains a duplicate index.
88template <typename Iterable>
89std::string FindDuplicateVarIndex(const Iterable& var_indices,
90 std::vector<bool>* var_mask) {
91 int duplicate_var_index = -1;
92 for (const int var_index : var_indices) {
93 if ((*var_mask)[var_index]) duplicate_var_index = var_index;
94 (*var_mask)[var_index] = true;
95 }
96 // Reset "var_mask" to all false, sparsely.
97 for (const int var_index : var_indices) {
98 (*var_mask)[var_index] = false;
99 }
100 if (duplicate_var_index >= 0) {
101 return absl::StrCat("var_index #", duplicate_var_index,
102 " appears several times");
103 }
104 return "";
105}

107// Internal method to detect errors in a single constraint.
108// "var_mask" is a vector<bool> whose size is the number of variables in
109// the model, and it will be all set to false before and after the call.
110std::string FindErrorInMPConstraint(
111 const MPConstraintProto& constraint, std::vector<bool>* var_mask,
112 double abs_value_threshold, const bool accept_trivially_infeasible_bounds) {
113 const std::string bound_error = FindErrorInBounds(
114 constraint, abs_value_threshold, accept_trivially_infeasible_bounds);
115 if (!bound_error.empty()) return bound_error;
116
117 // TODO(user): clarify explicitly, at least in a comment, whether we want
118 // to accept empty constraints (i.e. without variables).
119
120 const int num_vars_in_model = var_mask->size();
121 const int num_vars_in_ct = constraint.var_index_size();
122 const int num_coeffs_in_ct = constraint.coefficient_size();
123 if (num_vars_in_ct != num_coeffs_in_ct) {
124 return absl::StrCat("var_index_size() != coefficient_size() (",
125 num_vars_in_ct, " VS ", num_coeffs_in_ct);
126 }
127 for (int i = 0; i < num_vars_in_ct; ++i) {
128 const int var_index = constraint.var_index(i);
129 if (var_index >= num_vars_in_model || var_index < 0) {
130 return absl::StrCat("var_index(", i, ")=", var_index,
131 " is out of bounds");
132 }
133 const double coeff = constraint.coefficient(i);
134 if (IsNanOrAbsGreaterThanOrEqual(coeff, abs_value_threshold)) {
135 return absl::StrCat("coefficient(", i, ")=", (coeff), " is invalid");
136 }
137 }
138
139 const std::string error =
140 FindDuplicateVarIndex(constraint.var_index(), var_mask);
141 if (!error.empty()) return error;
142
143 // We found no error, all is fine.
144 return std::string();
145}

147std::string CroppedConstraintDebugString(const MPConstraintProto& constraint) {
148 const int kMaxPrintedVars = 10;
149
150 MPConstraintProto constraint_light = constraint;
151 std::string suffix_str;
152 if (constraint.var_index_size() > kMaxPrintedVars) {
153 constraint_light.mutable_var_index()->Truncate(kMaxPrintedVars);
154 absl::StrAppend(&suffix_str,
155 " (var_index cropped; size=", constraint.var_index_size(),
156 ").");
157 }
158 if (constraint.coefficient_size() > kMaxPrintedVars) {
159 constraint_light.mutable_coefficient()->Truncate(kMaxPrintedVars);
160 absl::StrAppend(&suffix_str, " (coefficient cropped; size=",
161 constraint.coefficient_size(), ").");
162 }
163 return absl::StrCat("Constraint proto: ",
164 ProtobufShortDebugString(constraint_light), suffix_str);
165}

167bool IsBoolean(const MPVariableProto& variable) {
168 if (variable.lower_bound() < 0) return false;
169 if (variable.upper_bound() > 1) return false;
170 return variable.is_integer();

173std::string FindErrorInMPIndicatorConstraint(
174 const MPModelProto& model, const MPIndicatorConstraint& indicator,
175 std::vector<bool>* var_mask, double abs_value_threshold,
176 bool accept_trivially_infeasible_bounds) {
177 if (!indicator.has_var_index()) {
178 return "var_index is required.";
179 }
180 const int var_index = indicator.var_index();
181 if (var_index < 0 || var_index >= model.variable_size()) {
182 return absl::StrCat("var_index=", var_index, " is out of bounds.");
183 }
184 if (!IsBoolean(model.variable(var_index))) {
185 return absl::StrCat("var_index=", var_index, " is not Boolean.");
186 }
187 const int var_value = indicator.var_value();
188 if (var_value < 0 || var_value > 1) {
189 return absl::StrCat("var_value=", var_value, " must be 0 or 1.");
190 }
191 const MPConstraintProto& constraint = indicator.constraint();
192 std::string error =
193 FindErrorInMPConstraint(constraint, var_mask, abs_value_threshold,
194 accept_trivially_infeasible_bounds);
195 if (!error.empty()) {
196 // Constraint protos can be huge, theoretically. So we guard against
197 // that.
198 return absl::StrCat(error, " in constraint ",
199 CroppedConstraintDebugString(constraint));
200 }
201 return "";
202}

204std::string FindErrorInMPSosConstraint(const MPModelProto& model,
205 const MPSosConstraint& sos,
206 std::vector<bool>* var_mask,
207 double abs_value_threshold) {
208 if (sos.weight_size() != 0 && sos.weight_size() != sos.var_index_size()) {
209 return "weight_size() > 0 and var_index_size() != weight_size()";
210 }
211 for (const int var_index : sos.var_index()) {
212 if (var_index < 0 || var_index >= model.variable_size()) {
213 return absl::StrCat("var_index=", var_index, " is out of bounds.");
214 }
215 }
216 for (int i = 0; i < sos.weight_size(); ++i) {
217 if (IsNanOrAbsGreaterThanOrEqual(sos.weight(i), abs_value_threshold)) {
218 return absl::StrCat("Invalid weight: ", sos.weight(i));
219 }
220 if (i == 0) continue;
221 if (sos.weight(i - 1) >= sos.weight(i)) {
222 return "SOS weights must be strictly increasing";
223 }
224 }
225
226 const std::string error = FindDuplicateVarIndex(sos.var_index(), var_mask);
227 if (!error.empty()) return error;
228
229 return "";
230}

232std::string FindErrorInMPQuadraticConstraint(
233 const MPModelProto& model, const MPQuadraticConstraint& qcst,
234 std::vector<bool>* var_mask, double abs_value_threshold,
235 bool accept_trivially_infeasible_bounds) {
236 const int num_vars = model.variable_size();
237
238 if (qcst.var_index_size() != qcst.coefficient_size()) {
239 return "var_index_size() != coefficient_size()";
240 }
241
242 const std::string bound_error = FindErrorInBounds(
243 qcst, abs_value_threshold, accept_trivially_infeasible_bounds);
244 if (!bound_error.empty()) return bound_error;
245
246 for (int i = 0; i < qcst.var_index_size(); ++i) {
247 if (qcst.var_index(i) < 0 || qcst.var_index(i) >= num_vars) {
248 return absl::StrCat("var_index(", i, ")=", qcst.var_index(i),
249 " is invalid.", " It must be in [0, ", num_vars, ")");
250 }
251
252 if (IsNanOrAbsGreaterThanOrEqual(qcst.coefficient(i),
253 abs_value_threshold)) {
254 return absl::StrCat("coefficient(", i, ")=", qcst.coefficient(i),
255 " is invalid");
256 }
257 }
258 const std::string duplicate_error =
259 FindDuplicateVarIndex(qcst.var_index(), var_mask);
260 if (!duplicate_error.empty()) return duplicate_error;
261
262 if (qcst.qvar1_index_size() != qcst.qvar2_index_size() ||
263 qcst.qvar1_index_size() != qcst.qcoefficient_size()) {
264 return "quadratic indices and coefficients must have the same size";
265 }
266 for (int i = 0; i < qcst.qvar1_index_size(); ++i) {
267 if (qcst.qvar1_index(i) >= num_vars || qcst.qvar1_index(i) < 0) {
268 return absl::StrCat("qvar1_index(", i, ")=", qcst.qvar1_index(i),
269 " is invalid.", " It must be in [0, ", num_vars, ")");
270 }
271
272 if (qcst.qvar2_index(i) >= num_vars || qcst.qvar2_index(i) < 0) {
273 return absl::StrCat("qvar2_index(", i, ")=", qcst.qvar2_index(i),
274 " is invalid.", " It must be in [0, ", num_vars, ")");
275 }
276
277 if (IsNanOrAbsGreaterThanOrEqual(qcst.qcoefficient(i),
278 abs_value_threshold)) {
279 return absl::StrCat("qcoefficient(", i, ")=", qcst.qcoefficient(i),
280 " is invalid");
281 }
282 }
283
284 return "";
285}

287std::string FindErrorInMPAbsConstraint(const MPModelProto& model,
288 const MPAbsConstraint& abs) {
289 if (!abs.has_var_index()) {
290 return "var_index is required.";
291 }
292 if (!abs.has_resultant_var_index()) {
293 return "resultant_var_index is required.";
294 }

297 if (abs.var_index() < 0 || abs.var_index() >= num_vars) {
298 return absl::StrCat("var_index=", abs.var_index(), " is invalid.",
299 " It must be in [0, ", num_vars, ")");
300 }
301 if (abs.resultant_var_index() < 0 || abs.resultant_var_index() >= num_vars) {
302 return absl::StrCat("var_index=", abs.resultant_var_index(), " is invalid.",
303 " It must be in [0, ", num_vars, ")");
304 }
305 return "";

308std::string FindErrorInMPAndOrConstraint(const MPModelProto& model,
309 const MPArrayConstraint& and_or) {
310 if (and_or.var_index_size() == 0) {
311 return "var_index cannot be empty.";
312 }
313 if (!and_or.has_resultant_var_index()) {
314 return "resultant_var_index is required.";
315 }
316
317 const int num_vars = model.variable_size();
318 for (int i = 0; i < and_or.var_index_size(); ++i) {
319 if (and_or.var_index(i) < 0 || and_or.var_index(i) >= num_vars) {
320 return absl::StrCat("var_index(", i, ")=", and_or.var_index(i),
321 " is invalid.", " It must be in [0, ", num_vars, ")");
322 }
323 if (!IsBoolean(model.variable(and_or.var_index(i)))) {
324 return absl::StrCat("var_index=", i, " is not Boolean.");
325 }
326 }
327 if (and_or.resultant_var_index() < 0 ||
328 and_or.resultant_var_index() >= num_vars) {
329 return absl::StrCat("resultant_var_index=", and_or.resultant_var_index(),
330 " is invalid.", " It must be in [0, ", num_vars, ")");
331 }
332 if (!IsBoolean(model.variable(and_or.resultant_var_index()))) {
333 return absl::StrCat("resultant_var_index is not Boolean.");
334 }
335 return "";

395std::string FindErrorInSolutionHint(
396 const PartialVariableAssignment& solution_hint, int num_vars,
397 double abs_value_threshold) {
398 if (solution_hint.var_index_size() != solution_hint.var_value_size()) {
399 return absl::StrCat("var_index_size() != var_value_size() [",
400 solution_hint.var_index_size(), " VS ",
401 solution_hint.var_value_size());
402 }
403 std::vector<bool> var_in_hint(num_vars, false);
404 for (int i = 0; i < solution_hint.var_index_size(); ++i) {
405 const int var_index = solution_hint.var_index(i);
406 if (var_index >= num_vars || var_index < 0) {
407 return absl::StrCat("var_index(", i, ")=", var_index, " is invalid.",
408 " It must be in [0, ", num_vars, ")");
409 }
410 if (var_in_hint[var_index]) {
411 return absl::StrCat("Duplicate var_index = ", var_index);
412 }
413 var_in_hint[var_index] = true;
414 if (IsNanOrAbsGreaterThanOrEqual(solution_hint.var_value(i),
415 abs_value_threshold)) {
416 return absl::StrCat("var_value(", i, ")=", (solution_hint.var_value(i)),
417 " is invalid");
418 }
419 }
420 return std::string();
421}

424// Maps the names of variables (or constraints, or other entities) to their
425// index in the MPModelProto. Non-unique names are supported, but are singled
426// out as such, by setting their index (the 'value' of the map entry) to -1.
427template <class NamedEntity>
428absl::flat_hash_map<std::string, int> BuildNameToIndexMap(
429 const google::protobuf::RepeatedPtrField<NamedEntity>& entities) {
430 absl::flat_hash_map<std::string, int> out;
431 for (int i = 0; i < entities.size(); ++i) {
432 int& index = gtl::LookupOrInsert(&out, entities.Get(i).name(), i);
433 if (index != i) index = -1;
434 }
435 return out;
436}

438class LazyMPModelNameToIndexMaps {
439 public:
440 explicit LazyMPModelNameToIndexMaps(const MPModelProto& model)
441 : model_(model) {}
442
443 absl::StatusOr<int> LookupName(
444 MPModelProto::Annotation::TargetType target_type,
445 const std::string& name) {
446 const absl::flat_hash_map<std::string, int>* map = nullptr;
447 switch (target_type) {
448 case MPModelProto::Annotation::VARIABLE_DEFAULT:
449 if (!variable_name_to_index_) {
450 variable_name_to_index_ = BuildNameToIndexMap(model_.variable());
451 }
452 map = &variable_name_to_index_.value();
453 break;
454 case MPModelProto::Annotation::CONSTRAINT:
455 if (!constraint_name_to_index_) {
456 constraint_name_to_index_ = BuildNameToIndexMap(model_.constraint());
457 }
458 map = &constraint_name_to_index_.value();
459 break;
460 case MPModelProto::Annotation::GENERAL_CONSTRAINT:
461 if (!general_constraint_name_to_index_) {
462 general_constraint_name_to_index_ =
463 BuildNameToIndexMap(model_.general_constraint());
464 }
465 map = &general_constraint_name_to_index_.value();
466 break;
467 }
468 const int index = gtl::FindWithDefault(*map, name, -2);
469 if (index == -2) return absl::NotFoundError("name not found");
470 if (index == -1) return absl::InvalidArgumentError("name is not unique");
471 return index;
472 }
473
474 private:
475 const MPModelProto& model_;
476 std::optional<absl::flat_hash_map<std::string, int>> variable_name_to_index_;
477 std::optional<absl::flat_hash_map<std::string, int>>
478 constraint_name_to_index_;
479 std::optional<absl::flat_hash_map<std::string, int>>
480 general_constraint_name_to_index_;
481};
482} // namespace

484std::string FindErrorInAnnotation(const MPModelProto::Annotation& annotation,
485 const MPModelProto& model,
486 LazyMPModelNameToIndexMaps* name_maps) {
487 // Checks related to the 'target' fields.
488 if (!annotation.has_target_index() && !annotation.has_target_name()) {
489 return "One of target_index or target_name must be set";
490 }
491 if (!MPModelProto::Annotation::TargetType_IsValid(annotation.target_type())) {
492 return "Invalid target_type";
493 }
494 int num_entitities = -1;
495 switch (annotation.target_type()) {
496 case MPModelProto::Annotation::VARIABLE_DEFAULT:
497 num_entitities = model.variable_size();
498 break;
499 case MPModelProto::Annotation::CONSTRAINT:
500 num_entitities = model.constraint_size();
501 break;
502 case MPModelProto::Annotation::GENERAL_CONSTRAINT:
503 num_entitities = model.general_constraint_size();
504 break;
505 }
506 int target_index = -1;
507 if (annotation.has_target_index()) {
508 target_index = annotation.target_index();
509 if (target_index < 0 || target_index >= num_entitities) {
510 return "Invalid target_index";
511 }
512 }
513 if (annotation.has_target_name()) {
514 if (annotation.has_target_index()) {
515 // No need to build the name lookup maps to verify consistency: we can
516 // even accept a name that is not unique, as long as the pointed entity
517 // (identified by its index) has the right name.
518 std::string name;
519 switch (annotation.target_type()) {
520 case MPModelProto::Annotation::VARIABLE_DEFAULT:
521 name = model.variable(target_index).name();
522 break;
523 case MPModelProto::Annotation::CONSTRAINT:
524 name = model.constraint(target_index).name();
525 break;
526 case MPModelProto::Annotation::GENERAL_CONSTRAINT:
527 name = model.general_constraint(target_index).name();
528 break;
529 }
530 if (annotation.target_name() != name) {
531 return absl::StrFormat(
532 "target_name='%s' doesn't match the name '%s' of target_index=%d",
533 annotation.target_name(), name, target_index);
534 }
535 } else { // !annotation.has_target_index()
536 const absl::StatusOr<int> index_or = name_maps->LookupName(
537 annotation.target_type(), annotation.target_name());
538 if (!index_or.ok()) {
539 return absl::StrCat("Bad target_name: ", index_or.status().message());
540 }
541 target_index = index_or.value();
542 }
543 }
544
545 // As of 2022-02, there are no checks related to the 'payload' fields. They
546 // can be set, unset, everything goes.
547 return "";
548}

552std::string FindErrorInMPModelProto(
553 const MPModelProto& model, double abs_value_threshold,
554 const bool accept_trivially_infeasible_bounds) {
555 // NOTE(user): Empty models are considered fine by this function, although
556 // it is not clear whether MPSolver::Solve() will always respond in the same
557 // way, depending on the solvers.
558 if (abs_value_threshold == 0.0) {
559 abs_value_threshold = absl::GetFlag(FLAGS_model_validator_infinity);
560 }
561
562 if (IsNanOrAbsGreaterThanOrEqual(model.objective_offset(),
563 abs_value_threshold)) {
564 return absl::StrCat("Invalid objective_offset: ",
565 (model.objective_offset()));
566 }
567 const int num_vars = model.variable_size();
568 const int num_cts = model.constraint_size();
569
570 // Validate variables.
571 std::string error;
572 for (int i = 0; i < num_vars; ++i) {
573 error = FindErrorInMPVariable(model.variable(i), abs_value_threshold,
574 accept_trivially_infeasible_bounds);
575 if (!error.empty()) {
576 return absl::StrCat("In variable #", i, ": ", error, ". Variable proto: ",
577 ProtobufShortDebugString(model.variable(i)));
578 }
579 }
580
581 // Validate constraints.
582 std::vector<bool> variable_appears(num_vars, false);
583 for (int i = 0; i < num_cts; ++i) {
584 const MPConstraintProto& constraint = model.constraint(i);
585 error = FindErrorInMPConstraint(constraint, &variable_appears,
586 abs_value_threshold,
587 accept_trivially_infeasible_bounds);
588 if (!error.empty()) {
589 // Constraint protos can be huge, theoretically. So we guard against that.
590 return absl::StrCat("In constraint #", i, ": ", error, ". ",
591 CroppedConstraintDebugString(constraint));
592 }
593 }
594
595 // Validate general constraints.
596 for (int i = 0; i < model.general_constraint_size(); ++i) {
597 const MPGeneralConstraintProto& gen_constraint =
598 model.general_constraint(i);
599 std::string error;
600 switch (gen_constraint.general_constraint_case()) {
601 case MPGeneralConstraintProto::kIndicatorConstraint:
602 error = FindErrorInMPIndicatorConstraint(
603 model, gen_constraint.indicator_constraint(), &variable_appears,
604 abs_value_threshold, accept_trivially_infeasible_bounds);
605 break;
606
607 case MPGeneralConstraintProto::kSosConstraint:
608 error =
609 FindErrorInMPSosConstraint(model, gen_constraint.sos_constraint(),
610 &variable_appears, abs_value_threshold);
611 break;
612
613 case MPGeneralConstraintProto::kQuadraticConstraint:
614 error = FindErrorInMPQuadraticConstraint(
615 model, gen_constraint.quadratic_constraint(), &variable_appears,
616 abs_value_threshold, accept_trivially_infeasible_bounds);
617 break;
618
619 case MPGeneralConstraintProto::kAbsConstraint:
620 error =
621 FindErrorInMPAbsConstraint(model, gen_constraint.abs_constraint());
622 break;
623
624 case MPGeneralConstraintProto::kAndConstraint:
625 error = FindErrorInMPAndOrConstraint(model,
626 gen_constraint.and_constraint());
627 break;
628
629 case MPGeneralConstraintProto::kOrConstraint:
630 error =
631 FindErrorInMPAndOrConstraint(model, gen_constraint.or_constraint());
632 break;
633
634 case MPGeneralConstraintProto::kMinConstraint:
635 error = FindErrorInMPMinMaxConstraint(
636 model, gen_constraint.min_constraint(), abs_value_threshold);
637 break;
638
639 case MPGeneralConstraintProto::kMaxConstraint:
640 error = FindErrorInMPMinMaxConstraint(
641 model, gen_constraint.max_constraint(), abs_value_threshold);
642 break;
643 default:
644 return absl::StrCat("Unknown general constraint type ",
645 gen_constraint.general_constraint_case());
646 }
647 if (!error.empty()) {
648 return absl::StrCat("In general constraint #", i, ": ", error);
649 }
650 }
651
652 // Validate objectives.
653 if (model.has_quadratic_objective()) {
654 error = FindErrorInQuadraticObjective(model.quadratic_objective(), num_vars,
655 abs_value_threshold);
656 if (!error.empty()) return absl::StrCat("In quadratic_objective: ", error);
657 }
658
659 // Validate the solution hint.
660 error = FindErrorInSolutionHint(model.solution_hint(), num_vars,
661 abs_value_threshold);
662 if (!error.empty()) {
663 return absl::StrCat("In solution_hint(): ", error);
664 }
665
666 // Validate the annotations.
667 {
668 LazyMPModelNameToIndexMaps name_maps(model);
669 for (int a = 0; a < model.annotation_size(); ++a) {
670 error = FindErrorInAnnotation(model.annotation(a), model, &name_maps);
671 if (!error.empty()) {
672 return absl::StrCat("In annotation #", a, ": ", error);
673 }
674 }
675 }
676
677 return std::string();
678}

680absl::optional<LazyMutableCopy<MPModelProto>>
681ExtractValidMPModelOrPopulateResponseStatus(const MPModelRequest& request,
682 MPSolutionResponse* response) {
683 CHECK(response != nullptr);
684
685 if (!request.has_model() && !request.has_model_delta()) {
686 response->set_status(MPSOLVER_OPTIMAL);
687 response->set_status_str("Requests without model are considered OPTIMAL");
688 return absl::nullopt;
689 }
690 if (request.has_model() && request.has_model_delta()) {
691 response->set_status(MPSOLVER_MODEL_INVALID);
692 response->set_status_str(
693 "Fields 'model' and 'model_delta' are mutually exclusive");
694 return absl::nullopt;
695 }
696
697 // Extract the baseline model.
698 LazyMutableCopy<MPModelProto> model(request.model());
699 if (request.has_model_delta()) {
700 // NOTE(user): This library needs to be portable, so we can't include
701 // ortools/base/file.h; see ../port/file.h.
702 std::string contents;
703 const absl::Status file_read_status = PortableFileGetContents(
704 request.model_delta().baseline_model_file_path(), &contents);
705 if (!file_read_status.ok()) {
706 response->set_status(MPSOLVER_MODEL_INVALID);
707 response->set_status_str(
708 "Error when reading model_delta.baseline_model_file_path: '" +
709 file_read_status.ToString());
710 return absl::nullopt;
711 }
712 if (!model.get_mutable()->ParseFromString(contents)) {
713 response->set_status(MPSOLVER_MODEL_INVALID);
714 response->set_status_str(
715 absl::StrFormat("The contents of baseline model file '%s' couldn't "
716 "be parsed as a raw serialized MPModelProto",
717 request.model_delta().baseline_model_file_path()));
718 return absl::nullopt;
719 }
720 }
721
722 // Validate the baseline model.
723 std::string error = FindErrorInMPModelProto(model.get());
724
725 // If the baseline is valid and we have a model delta, validate the delta,
726 // then apply it.
727 if (error.empty() && request.has_model_delta()) {
728 const MPModelDeltaProto& delta = request.model_delta();
729 error = FindErrorInMPModelDeltaProto(delta, model.get());
730 if (error.empty()) ApplyVerifiedMPModelDelta(delta, model.get_mutable());
731 }
732
733 // Deal with errors.
734 if (!error.empty()) {
735 if (request.enable_internal_solver_output()) {
736 LOG(ERROR) << absl::StrCat("Invalid model: ", error);
737 }
738 response->set_status(absl::StrContains(error, "Infeasible")
739 ? MPSOLVER_INFEASIBLE
740 : MPSOLVER_MODEL_INVALID);
741 response->set_status_str(error);
742 return absl::nullopt;
743 }
744
745 if (model.get().variable_size() == 0 && model.get().constraint_size() == 0 &&
746 model.get().general_constraint_size() == 0) {
747 response->set_status(MPSOLVER_OPTIMAL);
748 response->set_objective_value(model.get().objective_offset());
749 response->set_best_objective_bound(response->objective_value());
750 response->set_status_str(
751 "Requests without variables and constraints are considered OPTIMAL");
752 return absl::nullopt;
753 }
754
755 return std::move(model);
756}

769// TODO(user): Add a general FindFeasibilityErrorInSolution() and factor out the
770// common code.
771std::string FindFeasibilityErrorInSolutionHint(const MPModelProto& model,
772 double tolerance) {
773 const int num_vars = model.variable_size();
774
775 // First, we validate the solution hint.
776 std::string error =
777 FindErrorInSolutionHint(model.solution_hint(), num_vars,
778 absl::GetFlag(FLAGS_model_validator_infinity));
779 if (!error.empty()) return absl::StrCat("Invalid solution_hint: ", error);
780
781 // Special error message for the empty case.
782 if (num_vars > 0 && model.solution_hint().var_index_size() == 0) {
783 return "Empty solution_hint.";
784 }
785
786 // To be feasible, the hint must not be partial.
787 if (model.solution_hint().var_index_size() != num_vars) {
788 return absl::StrCat("Partial solution_hint: only ",
789 model.solution_hint().var_index_size(), " out of the ",
790 num_vars, " problem variables are set.");
791 }
792
793 // All the values must be exactly in the variable bounds.
794 std::vector<double> var_value(num_vars);
795 for (int i = 0; i < model.solution_hint().var_index_size(); ++i) {
796 const int var_index = model.solution_hint().var_index(i);
797 const double value = model.solution_hint().var_value(i);
798 var_value[var_index] = value;
799 const double lb = model.variable(var_index).lower_bound();
800 const double ub = model.variable(var_index).upper_bound();
801 if (!IsSmallerWithinTolerance(value, ub, tolerance) ||
802 !IsSmallerWithinTolerance(lb, value, tolerance)) {
803 return absl::StrCat("Variable '", model.variable(var_index).name(),
804 "' is set to ", (value),
805 " which is not in the variable bounds [", (lb), ", ",
806 (ub), "] modulo a tolerance of ", (tolerance), ".");
807 }
808 }
809
810 // All the constraints must be satisfiable.
811 for (int cst_index = 0; cst_index < model.constraint_size(); ++cst_index) {
812 const MPConstraintProto& constraint = model.constraint(cst_index);
813 AccurateSum<double> activity;
814 for (int j = 0; j < constraint.var_index_size(); ++j) {
815 activity.Add(constraint.coefficient(j) *
816 var_value[constraint.var_index(j)]);
817 }
818 const double lb = model.constraint(cst_index).lower_bound();
819 const double ub = model.constraint(cst_index).upper_bound();
820 if (!IsSmallerWithinTolerance(activity.Value(), ub, tolerance) ||
821 !IsSmallerWithinTolerance(lb, activity.Value(), tolerance)) {
822 return absl::StrCat(
823 "Constraint '", model.constraint(cst_index).name(), "' has activity ",
824 (activity.Value()), " which is not in the constraint bounds [", (lb),
825 ", ", (ub), "] modulo a tolerance of ", (tolerance), ".");
826 }
827 }
828
829 return "";
830}

832std::string FindErrorInMPModelDeltaProto(const MPModelDeltaProto& delta,
833 const MPModelProto& model) {
834 const double abs_value_threshold =
835 absl::GetFlag(FLAGS_model_validator_infinity);
836 int num_vars = model.variable_size();
837 // Validate delta variables.
838 std::string error;
839 absl::flat_hash_set<int> new_var_indices;
840 int max_var_index = num_vars - 1;
841 MPVariableProto tmp_var_proto;
842 for (const auto& pair : delta.variable_overrides()) {
843 const int var_index = pair.first;
844 const MPVariableProto& var_override_proto = pair.second;
845 if (var_index < 0) {
846 error = "Invalid key";
847 } else if (var_index >= num_vars) {
848 max_var_index = std::max(max_var_index, var_index);
849 new_var_indices.insert(var_index);
850 error =
851 FindErrorInMPVariable(var_override_proto, abs_value_threshold,
852 /*accept_trivially_infeasible_bounds=*/false);
853 } else {
854 tmp_var_proto = model.variable(var_index);
855 // NOTE(user): It is OK for the override proto to be empty, i.e. be a
856 // non-override.
857 tmp_var_proto.MergeFrom(var_override_proto);
858 error =
859 FindErrorInMPVariable(tmp_var_proto, abs_value_threshold,
860 /*accept_trivially_infeasible_bounds=*/false);
861 }
862 if (!error.empty()) {
863 return absl::StrFormat(
864 "variable_overrides with key (eg. var index) = %d: %s", var_index,
865 error);
866 }
867 }
868 if (max_var_index != num_vars + new_var_indices.size() - 1) {
869 return absl::StrFormat(
870 "The added and existing variable indices do not form a dense integer "
871 "interval: oldmax=%d, max=%d, num added=%d",
872 num_vars - 1, max_var_index, new_var_indices.size());
873 }
874 // Now we "officially" add the new variables to "num_vars".
875 num_vars += new_var_indices.size();
876
877 // Validate delta constraints. We can avoid going over the full
878 // var_index/coefficient of the original constraint, since the overrides are
879 // self-sufficient (i.e. the override var_index/coefficients are valid iff
880 // they would be valid in a standalone, new constraint). So we use a partial
881 // proto merger to avoid those in the baseline constraint.
882 std::vector<bool> variable_appears(num_vars, false);
883 MPConstraintProto tmp_constraint_proto;
884 const int num_constraints = model.constraint_size();
885 absl::flat_hash_set<int> new_ct_indices;
886 int max_ct_index = num_constraints - 1;
887 for (const auto& pair : delta.constraint_overrides()) {
888 const int ct_index = pair.first;
889 const MPConstraintProto& constraint_override_proto = pair.second;
890 if (ct_index < 0) {
891 error = "Invalid constraint index";
892 } else if (ct_index >= num_constraints) {
893 max_ct_index = std::max(max_ct_index, ct_index);
894 new_ct_indices.insert(ct_index);
895 error = FindErrorInMPConstraint(
896 constraint_override_proto, &variable_appears, abs_value_threshold,
897 /*accept_trivially_infeasible_bounds=*/false);
898 } else {
899 // NOTE(user): We don't need to do the merging of var_index/coefficient:
900 // that part of the merged constraint will be valid iff the override is
901 // valid as a standalone var_index/coefficient map.
902 // So we simply validate a reduced version of the actual "merged"
903 // constraint, by removing the var_index/coefficient of the baseline.
904 // Benefit: the complexity is O(|constraint override|) even if the
905 // baseline constraint was huge.
906 tmp_constraint_proto.Clear();
907 MergeMPConstraintProtoExceptTerms(model.constraint(ct_index),
908 &tmp_constraint_proto);
909 tmp_constraint_proto.MergeFrom(constraint_override_proto);
910 error = FindErrorInMPConstraint(
911 tmp_constraint_proto, &variable_appears, abs_value_threshold,
912 /*accept_trivially_infeasible_bounds=*/false);
913 }
914 if (!error.empty()) {
915 return absl::StrFormat(
916 "constraint_overrides with key (eg. constraint index) = %d: %s",
917 ct_index, error);
918 }
919 }
920 if (max_ct_index != num_constraints + new_ct_indices.size() - 1) {
921 return absl::StrFormat(
922 "The added and existing constraint indices do not form a dense integer "
923 "interval: oldmax=%d, max=%d, num added=%d",
924 num_constraints - 1, max_ct_index, new_ct_indices.size());
925 }
926
927 return "";
928}

942void PruneZeroTermsInMpConstraint(MPConstraintProto* ct) {
943 // Optimize the fast path (when no term is pruned) by doing a first quick scan
944 // until the first zero.
945 int first_zero = 0;
946 while (first_zero < ct->var_index_size() &&
947 ct->coefficient(first_zero) != 0.0) {
948 ++first_zero;
949 }
950 int num_kept = first_zero;
951 for (int i = first_zero; i < ct->var_index_size(); ++i) {
952 if (ct->coefficient(i) == 0.0) continue;
953 if (num_kept != i) {
954 ct->set_var_index(num_kept, ct->var_index(i));
955 ct->set_coefficient(num_kept, ct->coefficient(i));
956 }
957 ++num_kept;
958 }
959 ct->mutable_var_index()->Truncate(num_kept);
960 ct->mutable_coefficient()->Truncate(num_kept);
961}

973void ApplyVerifiedMPModelDelta(const MPModelDeltaProto& delta,
974 MPModelProto* model) {
975 // Apply the delta to the variables: first, resize the variable array.
976 int max_var_index = -1;
977 for (const auto& p : delta.variable_overrides()) {
978 max_var_index = std::max(max_var_index, p.first);
979 }
980 if (max_var_index >= model->variable_size()) {
981 ExtendRepeatedPtrFieldToSize(max_var_index + 1, model->mutable_variable());
982 }
983 // Then, apply the variable overrides.
984 for (const auto& p : delta.variable_overrides()) {
985 model->mutable_variable(p.first)->MergeFrom(p.second);
986 }
987
988 // Apply the delta to the constraints: first, resize the constraint array.
989 int max_ct_index = -1;
990 for (const auto& p : delta.constraint_overrides()) {
991 max_ct_index = std::max(max_ct_index, p.first);
992 }
993 const int old_num_constraints = model->constraint_size();
994 if (max_ct_index >= old_num_constraints) {
995 ExtendRepeatedPtrFieldToSize(max_ct_index + 1, model->mutable_constraint());
996 }
997 // Then, apply the constraint overrides.
998 for (const auto& p : delta.constraint_overrides()) {
999 const MPConstraintProto& override_ct = p.second;
1000 MPConstraintProto* baseline = model->mutable_constraint(p.first);
1001 // Fast path for added constraints.
1002 if (p.first >= old_num_constraints) {
1003 *baseline = override_ct;
1004 continue;
1005 }
1006 MergeMPConstraintProtoExceptTerms(/*from=*/override_ct, /*to=*/baseline);
1007 // Special case: the override is neutralized.
1008 if (override_ct.has_lower_bound() &&
1009 override_ct.lower_bound() <=
1010 -absl::GetFlag(FLAGS_model_validator_infinity) &&
1011 override_ct.has_upper_bound() &&
1012 override_ct.upper_bound() >=
1013 absl::GetFlag(FLAGS_model_validator_infinity)) {
1014 baseline->clear_var_index();
1015 baseline->clear_coefficient();
1016 continue;
1017 }
1018 // Otherwise we have to apply the term overrides. We can't do that in less
1019 // than O(|baseline| + |override_ct|) because the baseline doesn't have a
1020 // lookup-friendly data structure. But we still try to do it as efficiently
1021 // as possible. In particular, we only use O(|override_ct|) extra memory.
1022 absl::flat_hash_map<int, double> term_overrides;
1023 term_overrides.reserve(override_ct.var_index_size());
1024 for (int i = 0; i < override_ct.var_index_size(); ++i) {
1025 term_overrides[override_ct.var_index(i)] = override_ct.coefficient(i);
1026 }
1027 for (int i = 0; i < baseline->var_index_size(); ++i) {
1028 auto it = term_overrides.find(baseline->var_index(i));
1029 if (it == term_overrides.end()) continue;
1030 baseline->set_coefficient(i, it->second);
1031 it->second = 0.0; // To mark this term override as 'has been applied'.
1032 }
1033 PruneZeroTermsInMpConstraint(baseline);
1034 // Add the term overrides which haven't been used: those are added terms.
1035 for (const auto& p : term_overrides) {
1036 if (p.second != 0.0) {
1037 baseline->add_var_index(p.first);
1038 baseline->add_coefficient(p.second);
1039 }
1040 }
1041 }
1042}
