28// SetCoverLedger does the bookkeeping for a solution to the
29// SetCoverModel passed as argument.
30// The state of a SetCoverLedger instance is uniquely de fined by a
31// SubsetBoolVector representing whether a subset is selected in the solution
32// or not.
33// A SetCoverLedger is (relatively) small:
34// is_selected_, a partial solution, vector of Booleans of size #subsets.
35// From this, the following can be computed:
36// coverage_, the number of times an elememt is covered;
37// marginal_impacts_, the number of elements of a subset still uncovered;
38// is_removable_, whether a subset can be removed from the solution.
39// Note that is_removable_[subset] implies is_selected_[subset], and thus
40// (is_removable_[subset] <= is_selected_[subset]) == true.
41class SetCoverLedger {
42 public:
43 // Constructs an empty weighted set covering solver state.
44 // The model may not change after the ledger was built.
45 explicit SetCoverLedger(SetCoverModel* m) : model_(m) { Initialize(); }
46
47 // Initializes the solver once the data is set. The model cannot be changed
48 // afterwards.
49 void Initialize();
50
51 // Recomputes all the invariants for the current solution.
52 void MakeDataConsistent() {
53 cost_ = ComputeCost(is_selected_);
54 coverage_ = ComputeCoverage(is_selected_);
55 is_removable_ = ComputeIsRemovable(coverage_);
56 marginal_impacts_ = ComputeMarginalImpacts(coverage_);
57 num_elements_covered_ = ComputeNumElementsCovered(coverage_);
58 }
59
60 // Returns the weighted set covering model to which the state applies.
61 SetCoverModel* model() const { return model_; }
62
63 // Returns the cost of current solution.
64 Cost cost() const { return cost_; }
65
66 // Returns whether subset is selected in the solution.
67 bool is_selected(SubsetIndex subset) const { return is_selected_[subset]; }
68
69 // Returns the number of elements in each subset that are not covered in the
70 // current solution.
71 ElementIndex marginal_impacts(SubsetIndex subset) const {
72 return marginal_impacts_[subset];
73 }
74
75 // Returns the number of subsets covering each element.
76 SubsetIndex coverage(ElementIndex subset) const { return coverage_[subset]; }
77
78 // Returns whether subset can be removed from the solution.
79 bool is_removable(SubsetIndex subset) const { return is_removable_[subset]; }
80
81 // Returns the number of elements covered.
82 ElementIndex num_elements_covered() const { return num_elements_covered_; }
83
84 // Stores the solution and recomputes the data in the ledger.
85 void LoadSolution(const SubsetBoolVector& c);
86
87 // Returns the current solution.
88 SubsetBoolVector GetSolution() const { return is_selected_; }
89
90 // Returns true if the data stored in the ledger is consistent.
91 bool CheckConsistency() const;
92
93 // Computes is_removable_ from scratch for every subset.
94 // TODO(user): reconsider exposing this.
95 void RecomputeIsRemovable() { is_removable_ = ComputeIsRemovable(coverage_); }
96
97 // Returns the subsets that share at least one element with subset.
98 // TODO(user): is it worth to precompute this?
99 std::vector<SubsetIndex> ComputeImpactedSubsets(SubsetIndex subset) const;
100
101 // Updates is_removable_ for each subset in impacted_subsets.
102 void UpdateIsRemovable(const std::vector<SubsetIndex>& impacted_subsets);
103
104 // Updates marginal_impacts_ for each subset in impacted_subsets.
105 void UpdateMarginalImpacts(const std::vector<SubsetIndex>& impacted_subsets);
106
107 // Toggles is_selected_[subset] to value, and incrementally updates the
108 // ledger.
109 // Returns a vector of subsets impacted by the change, in case they need
110 // to be reconsidered in a solution geneator or a local search algorithm.
111 // Calls UnsafeToggle, with the added checks:
112 // If value is true, DCHECKs that subset is removable.
113 // If value is true, DCHECKs that marginal impact of subset is removable.
114 std::vector<SubsetIndex> Toggle(SubsetIndex subset, bool value);
115
116 // Same as Toggle, with less DCHECKS.
117 // Useful for some meta-heuristics that allow to go through infeasible
118 // solutions.
119 // Only checks that value is different from is_selected_[subset].
120 std::vector<SubsetIndex> UnsafeToggle(SubsetIndex subset, bool value);
121
122 // Update coverage_ for subset when setting is_selected_[subset] to value.
123 void UpdateCoverage(SubsetIndex subset, bool value);
124
125 // Returns true if the elements selected in the current solution cover all
126 // the elements of the set.
127 bool CheckSolution() const;
128
129 // Checks that coverage_ and marginal_impacts_ are consistent with choices.
130 bool CheckCoverageAndMarginalImpacts(const SubsetBoolVector& choices) const;
131
132 private:
133 // Recomputes the cost from scratch from c.
134 Cost ComputeCost(const SubsetBoolVector& c) const;
135
136 // Computes is_removable based on a coverage cvrg.
137 SubsetBoolVector ComputeIsRemovable(const ElementToSubsetVector& cvrg) const;
138
139 // Computes marginal impacts based on a coverage cvrg.
140 SubsetToElementVector ComputeMarginalImpacts(
141 const ElementToSubsetVector& cvrg) const;
142
143 // Computes the number of elements covered based on coverage vector 'cvrg'.
144 ElementIndex ComputeNumElementsCovered(
145 const ElementToSubsetVector& cvrg) const;
146
147 // Returns true if subset can be removed from the solution, i.e. it is
148 // redundant to cover all the elements.
149 // This function is used to check that is_removable[subset] is consistent.
150 bool ComputeIsRemovable(SubsetIndex subset) const;
151
152 // Returns the number of elements currently covered by subset.
153 ElementToSubsetVector ComputeSingleSubsetCoverage(SubsetIndex subset) const;
154
155 // Returns a vector containing the number of subsets covering each element.
156 ElementToSubsetVector ComputeCoverage(const SubsetBoolVector& choices) const;
157
158 // Checks that the value of coverage_ is correct by recomputing and comparing.
159 bool CheckSingleSubsetCoverage(SubsetIndex subset) const;
160
161 // Checks that coverage_ is consistent with choices.
162 bool CheckCoverageAgainstSolution(const SubsetBoolVector& choices) const;
163
164 // Returns true if is_removable_ is consistent.
165 bool CheckIsRemovable() const;
166
167 // The weighted set covering model on which the solver is run.
168 SetCoverModel* model_;
169
170 // Current cost.
171 Cost cost_;
172
173 // The number of elements covered in the current solution.
174 ElementIndex num_elements_covered_;
175
176 // Current assignment.
177 SubsetBoolVector is_selected_;
178
179 // The marginal impact of a subset is the number of elements in that subset
180 // that are not covered in the current solution.
181 SubsetToElementVector marginal_impacts_;
182
183 // The coverage of an element is the number of used subsets which contains
184 // the said element.
185 ElementToSubsetVector coverage_;
186
187 // True if the subset can be removed from the solution without making it
188 // infeasible.
189 SubsetBoolVector is_removable_;
190};
