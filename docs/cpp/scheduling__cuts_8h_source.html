31// For a given set of intervals and demands, we compute the energy of
32// each task and make sure their sum fits in the span of the intervals * its
33// capacity.
34//
35// If an interval is optional, it contributes
36// min_demand * min_size * presence_literal
37// amount of total energy.
38//
39// If an interval is performed, we use the linear energy formulation (if
40// defined, that is if different from a constant -1), or the McCormick
41// relaxation of the product size * demand if not defined.
42//
43// The maximum energy is capacity * span of intervals at level 0.
44CutGenerator CreateCumulativeEnergyCutGenerator(
45 SchedulingConstraintHelper* helper, SchedulingDemandHelper* demands_helper,
46 const AffineExpression& capacity,
47 const std::optional<AffineExpression>& makespan, Model* model);
48
49// For a given set of intervals and demands, we first compute the mandatory part
50// of the interval as [start_max , end_min]. We use this to calculate mandatory
51// demands for each start_max time points for eligible intervals.
52// Since the sum of these mandatory demands must be smaller or equal to the
53// capacity, we create a cut representing that.
54//
55// If an interval is optional, it contributes min_demand * presence_literal
56// amount of demand to the mandatory demands sum. So the final cut is generated
57// as follows:
58// sum(demands of always present intervals)
59// + sum(presence_literal * min_of_demand) <= capacity.
60CutGenerator CreateCumulativeTimeTableCutGenerator(
61 SchedulingConstraintHelper* helper, SchedulingDemandHelper* demands_helper,
62 const AffineExpression& capacity, Model* model);

64// Completion time cuts for the cumulative constraint. It is a simple relaxation
65// where we replace a cumulative task with demand k and duration d by a
66// no_overlap task with duration d * k / capacity_max.
67CutGenerator CreateCumulativeCompletionTimeCutGenerator(
68 SchedulingConstraintHelper* helper, SchedulingDemandHelper* demands_helper,
69 const AffineExpression& capacity, Model* model);

83// Energetic cuts for the no_overlap_2d constraint.
84//
85// For a given set of rectangles, we compute the area of each rectangle
86// and make sure their sum is less than the area of the bounding interval.
87//
88// If an interval is optional, it contributes
89// min_size_x * min_size_y * presence_literal
90// amount of total area.
91//
92// If an interval is performed, we use the linear area formulation (if
93// possible), or the McCormick relaxation of the size_x * size_y.
94//
95// The maximum area is the area of the bounding rectangle of each intervals
96// at level 0.
97CutGenerator CreateNoOverlap2dEnergyCutGenerator(
98 SchedulingConstraintHelper* x_helper, SchedulingConstraintHelper* y_helper,
99 SchedulingDemandHelper* x_demands_helper,
100 SchedulingDemandHelper* y_demands_helper,
101 const std::vector<std::vector<LiteralValueValue>>& energies, Model* model);

103// For a given set of intervals, we first compute the min and max of all
104// intervals. Then we create a cut that indicates that all intervals must fit
105// in that span.
106//
107// If an interval is optional, it contributes min_size * presence_literal
108// amount of demand to the mandatory demands sum. So the final cut is generated
109// as follows:
110// sum(sizes of always present intervals)
111// + sum(presence_literal * min_of_size) <= span of all intervals.
112CutGenerator CreateNoOverlapEnergyCutGenerator(
113 SchedulingConstraintHelper* helper,
114 const std::optional<AffineExpression>& makespan, Model* model);

152// Stores the event for a rectangle along the two axis x and y.
153// For a no_overlap constraint, y is always of size 1 between 0 and 1.
154// For a cumulative constraint, y is the demand that must be between 0 and
155// capacity_max.
156// For a no_overlap_2d constraint, y the other dimension of the rect.
157struct CtEvent : BaseEvent {
158 CtEvent(int t, SchedulingConstraintHelper* x_helper);
159
160 // The lp value of the end of the x interval.
161 AffineExpression x_end;
162 double x_lp_end;
163
164 // Indicates if the events used the optional energy information from the
165 // model.
166 bool use_energy = false;
167
168 // Indicates if the cut is lifted, that is if it includes tasks that are not
169 // strictly contained in the current time window.
170 bool lifted = false;
171
172 // If we know that the size on y is fixed, we can use some heuristic to
173 // compute the maximum subset sums under the capacity and use that instead
174 // of the full capacity.
175 bool y_size_is_fixed = false;
176
177 std::string DebugString() const;
178};

180// Computes the minimum sum of the end min and the minimum sum of the end min
181// weighted by y_size_min of all events. It returns false if no permatutation is
182// valid w.r.t. the range of x_start.
183//
184// Note that this is an exhaustive algorithm, so the number of events should be
185// small, like <= 10. They should also starts in index order.
186//
187// Optim: If both sums are proven <= to the corresponding threshold, we abort.
188struct PermutableEvent {
189 PermutableEvent(int i, CtEvent e)
190 : index(i),
191 x_start_min(e.x_start_min),
192 x_start_max(e.x_start_max),
193 x_size_min(e.x_size_min),
194 y_size_min(e.y_size_min) {}
195 bool operator<(const PermutableEvent& o) const { return index < o.index; }
196
197 int index; // for < to be used by std::next_permutation().
198 IntegerValue x_start_min;
199 IntegerValue x_start_max;
200 IntegerValue x_size_min;
201 IntegerValue y_size_min;
202};
