29// For a given set of intervals and demands, we compute the energy of
30// each task and make sure their sum fits in the span of the intervals * its
31// capacity.
32//
33// If an interval is optional, it contributes
34// min_demand * min_size * presence_literal
35// amount of total energy.
36//
37// If an interval is performed, we use the linear energy formulation (if
38// defined, that is if different from a constant -1), or the McCormick
39// relaxation of the product size * demand if not defined.
40//
41// The maximum energy is capacity * span of intervals at level 0.
42CutGenerator CreateCumulativeEnergyCutGenerator(
43 SchedulingConstraintHelper* helper, SchedulingDemandHelper* demands_helper,
44 const AffineExpression& capacity,
45 const std::optional<AffineExpression>& makespan, Model* model);

47// For a given set of intervals and demands, we first compute the mandatory part
48// of the interval as [start_max , end_min]. We use this to calculate mandatory
49// demands for each start_max time points for eligible intervals.
50// Since the sum of these mandatory demands must be smaller or equal to the
51// capacity, we create a cut representing that.
52//
53// If an interval is optional, it contributes min_demand * presence_literal
54// amount of demand to the mandatory demands sum. So the final cut is generated
55// as follows:
56// sum(demands of always present intervals)
57// + sum(presence_literal * min_of_demand) <= capacity.
58CutGenerator CreateCumulativeTimeTableCutGenerator(
59 SchedulingConstraintHelper* helper, SchedulingDemandHelper* demands_helper,
60 const AffineExpression& capacity, Model* model);

62// Completion time cuts for the cumulative constraint. It is a simple relaxation

81// Energetic cuts for the no_overlap_2d constraint.
82//
83// For a given set of rectangles, we compute the area of each rectangle
84// and make sure their sum is less than the area of the bounding interval.
85//
86// If an interval is optional, it contributes
87// min_size_x * min_size_y * presence_literal
88// amount of total area.
89//
90// If an interval is performed, we use the linear area formulation (if
91// possible), or the McCormick relaxation of the size_x * size_y.
92//
93// The maximum area is the area of the bounding rectangle of each intervals
94// at level 0.
95CutGenerator CreateNoOverlap2dEnergyCutGenerator(
96 const std::vector<IntervalVariable>& x_intervals,
97 const std::vector<IntervalVariable>& y_intervals, Model* model);

99// For a given set of intervals, we first compute the min and max of all
100// intervals. Then we create a cut that indicates that all intervals must fit
101// in that span.
102//
103// If an interval is optional, it contributes min_size * presence_literal
104// amount of demand to the mandatory demands sum. So the final cut is generated
105// as follows:
106// sum(sizes of always present intervals)
107// + sum(presence_literal * min_of_size) <= span of all intervals.
108CutGenerator CreateNoOverlapEnergyCutGenerator(
109 SchedulingConstraintHelper* helper,
110 const std::optional<AffineExpression>& makespan, Model* model);

148// Stores the event for a rectangle along the two axis x and y.
149// For a no_overlap constraint, y is always of size 1 between 0 and 1.
150// For a cumulative constraint, y is the demand that must be between 0 and
151// capacity_max.
152// For a no_overlap_2d constraint, y the other dimension of the rect.
153struct CtEvent : BaseEvent {
154 CtEvent(int t, SchedulingConstraintHelper* x_helper);
155
156 // The lp value of the end of the x interval.
157 AffineExpression x_end;
158 double x_lp_end;
159
160 // Indicates if the events used the optional energy information from the
161 // model.
162 bool use_energy = false;
163
164 // Indicates if the cut is lifted, that is if it includes tasks that are not
165 // strictly contained in the current time window.
166 bool lifted = false;
167
168 // If we know that the size on y is fixed, we can use some heuristic to
169 // compute the maximum subset sums under the capacity and use that instead
170 // of the full capacity.
171 bool y_size_is_fixed = false;
172
173 std::string DebugString() const;
174};

176// Computes the minimum sum of the end min and the minimum sum of the end min
177// weighted by y_size_min of all events. It returns false if no permatutation is
178// valid w.r.t. the range of x_start.
179//
180// Note that this is an exhaustive algorithm, so the number of events should be
181// small, like <= 10. They should also starts in index order.
182//
183// Optim: If both sums are proven <= to the corresponding threshold, we abort.
184struct PermutableEvent {
185 PermutableEvent(int i, CtEvent e)
186 : index(i),
187 x_start_min(e.x_start_min),
188 x_start_max(e.x_start_max),
189 x_size_min(e.x_size_min),
190 y_size_min(e.y_size_min) {}
191 bool operator<(const PermutableEvent& o) const { return index < o.index; }
192
193 int index; // for < to be used by std::next_permutation().
194 IntegerValue x_start_min;
195 IntegerValue x_start_max;
196 IntegerValue x_size_min;
197 IntegerValue y_size_min;
198};
