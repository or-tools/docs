35struct ClosedInterval {
36 ClosedInterval() {}
37 ClosedInterval(int64_t s, int64_t e) : start(s), end(e) {
38 DLOG_IF(DFATAL, s > e) << "Invalid ClosedInterval(" << s << ", " << e
39 << ")";
40 }
41
42 std::string DebugString() const;
43 bool operator==(const ClosedInterval& other) const {
44 return start == other.start && end == other.end;
45 }
46
47 // Because we mainly manipulate vector of disjoint intervals, we only need to
48 // sort by the start. We do not care about the order in which interval with
49 // the same start appear since they will always be merged into one interval.
50 bool operator<(const ClosedInterval& other) const {
51 return start < other.start;
52 }
53
54 template <typename H>
55 friend H AbslHashValue(H h, const ClosedInterval& interval) {
56 return H::combine(std::move(h), interval.start, interval.end);
57 }
58
59 int64_t start = 0; // Inclusive.
60 int64_t end = 0; // Inclusive.
61};
