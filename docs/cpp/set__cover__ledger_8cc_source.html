28void SetCoverLedger::Initialize() {
29 DCHECK(model_->ComputeFeasibility());
30 model_->CreateSparseRowView();
31
32 const SubsetIndex num_subsets(model_->num_subsets());
33 is_selected_.assign(num_subsets, false);
34 is_removable_.assign(num_subsets, false);
35 marginal_impacts_.assign(num_subsets, ElementIndex(0));
36
37 const SparseColumnView& columns = model_->columns();
38 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
39 marginal_impacts_[subset] = columns[subset].size().value();
40 }
41 const ElementIndex num_elements(model_->num_elements());
42 coverage_.assign(num_elements, SubsetIndex(0));
43 cost_ = 0.0;
44 num_elements_covered_ = ElementIndex(0);
45}

58bool SetCoverLedger::CheckSolution() const {
59 bool is_ok = true;
60
61 const ElementToSubsetVector cvrg = ComputeCoverage(is_selected_);
62 const ElementIndex num_elements(model_->num_elements());
63 for (ElementIndex element(0); element < num_elements; ++element) {
64 if (cvrg[element] == 0) {
65 LOG(ERROR) << "Recomputed coverage_ for element " << element << " = 0";
66 is_ok = false;
67 }
68 }
69
70 const Cost recomputed_cost = ComputeCost(is_selected_);
71 if (cost_ != recomputed_cost) {
72 LOG(ERROR) << "Cost = " << cost_
73 << ", while recomputed cost_ = " << recomputed_cost;
74 is_ok = false;
75 }
76 return is_ok;
77}

79bool SetCoverLedger::CheckCoverageAgainstSolution(
80 const SubsetBoolVector& choices) const {
81 const SubsetIndex num_subsets(model_->num_subsets());
82 DCHECK_EQ(num_subsets, choices.size());
83 const ElementToSubsetVector cvrg = ComputeCoverage(choices);
84 bool is_ok = true;
85 const ElementIndex num_elements(model_->num_elements());
86 for (ElementIndex element(0); element < num_elements; ++element) {
87 if (coverage_[element] != cvrg[element]) {
88 LOG(ERROR) << "Recomputed coverage_ for element " << element << " = "
89 << cvrg[element]
90 << ", while updated coverage_ = " << coverage_[element];
91 is_ok = false;
92 }
93 }
94 return is_ok;

97bool SetCoverLedger::CheckCoverageAndMarginalImpacts(
98 const SubsetBoolVector& choices) const {
99 const SubsetIndex num_subsets(model_->num_subsets());
100 DCHECK_EQ(num_subsets, choices.size());
101 const ElementToSubsetVector cvrg = ComputeCoverage(choices);
102 bool is_ok = CheckCoverageAgainstSolution(choices);
103 const SubsetToElementVector mrgnl_impcts = ComputeMarginalImpacts(cvrg);
104 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
105 if (marginal_impacts_[subset] != mrgnl_impcts[subset]) {
106 LOG(ERROR) << "Recomputed marginal impact for subset " << subset << " = "
107 << mrgnl_impcts[subset] << ", while updated marginal impact = "
108 << marginal_impacts_[subset];
109 is_ok = false;
110 }
111 }
112 return is_ok;
113}

115// Used only once, for testing. TODO(user): Merge with
116// CheckCoverageAndMarginalImpacts.
117SubsetToElementVector SetCoverLedger::ComputeMarginalImpacts(
118 const ElementToSubsetVector& cvrg) const {
119 const ElementIndex num_elements(model_->num_elements());
120 DCHECK_EQ(num_elements, cvrg.size());
121 const SparseColumnView& columns = model_->columns();
122 const SubsetIndex num_subsets(model_->num_subsets());
123 SubsetToElementVector mrgnl_impcts(num_subsets, ElementIndex(0));
124 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
125 for (ElementIndex element : columns[subset]) {
126 if (cvrg[element] == 0) {
127 ++mrgnl_impcts[subset];
128 }
129 }
130 DCHECK_LE(mrgnl_impcts[subset], columns[subset].size().value());
131 DCHECK_GE(mrgnl_impcts[subset], 0);
132 }
133 return mrgnl_impcts;
134}

139 const SubsetIndex num_subsets(model_->num_subsets());

160ElementToSubsetVector SetCoverLedger::ComputeCoverage(
161 const SubsetBoolVector& choices) const {
162 const ElementIndex num_elements(model_->num_elements());
163 const SparseRowView& rows = model_->rows();
164 // Use "crypterse" naming to avoid confusing with coverage_.
165 ElementToSubsetVector cvrg(num_elements, SubsetIndex(0));
166 for (ElementIndex element(0); element < num_elements; ++element) {
167 for (SubsetIndex subset : rows[element]) {
168 if (choices[subset]) {
169 ++cvrg[element];
170 }
171 }
172 DCHECK_LE(cvrg[element], rows[element].size().value());
173 DCHECK_GE(cvrg[element], 0);
174 }
175 return cvrg;
176}

191 const SparseColumnView& columns = model_->columns();
192 const ElementIndex num_elements(model_->num_elements());
193 // Use "crypterse" naming to avoid confusing with cvrg.
194 ElementToSubsetVector cvrg(num_elements, SubsetIndex(0));
195 const SparseRowView& rows = model_->rows();
196 for (const ElementIndex element : columns[subset]) {
197 for (SubsetIndex subset : rows[element]) {

203 DCHECK_GE(cvrg[element], 0);

208std::vector<SubsetIndex> SetCoverLedger::Toggle(SubsetIndex subset,
209 bool value) {
210 // Note: "if p then q" is also "not(p) or q", or p <= q (p LE q).
211 // If selected, then is_removable, to make sure we still have a solution.
212 DCHECK(is_selected_[subset] <= is_removable_[subset]);
213 // If value, then marginal_impact > 0, to not increase the cost.
214 DCHECK((value <= (marginal_impacts_[subset] > 0)));
215 return UnsafeToggle(subset, value);
216}

218std::vector<SubsetIndex> SetCoverLedger::UnsafeToggle(SubsetIndex subset,
219 bool value) {
220 // We allow to deselect a non-removable subset, but at least the
221 // Toggle should be a real change.
222 DCHECK_NE(is_selected_[subset], value);
223 // If selected, then marginal_impact == 0.
224 DCHECK((is_selected_[subset] <= (marginal_impacts_[subset] == 0)));
225 DVLOG(1) << (value ? "S" : "Des") << "electing subset " << subset;
226 const SubsetCostVector& subset_costs = model_->subset_costs();
227 cost_ += value ? subset_costs[subset] : -subset_costs[subset];
228 is_selected_[subset] = value;
229 UpdateCoverage(subset, value);
230 const std::vector<SubsetIndex> impacted_subsets =
231 ComputeImpactedSubsets(subset);
232 UpdateIsRemovable(impacted_subsets);
233 UpdateMarginalImpacts(impacted_subsets);
234 DCHECK((is_selected_[subset] <= (marginal_impacts_[subset] == 0)));
235 return impacted_subsets;
236}

238void SetCoverLedger::UpdateCoverage(SubsetIndex subset, bool value) {
239 const SparseColumnView& columns = model_->columns();
240 const SparseRowView& rows = model_->rows();
241 const int delta = value ? 1 : -1;
242 for (const ElementIndex element : columns[subset]) {
243 DVLOG(2) << "Coverage of element " << element << " changed from "
244 << coverage_[element] << " to " << coverage_[element] + delta;
245 coverage_[element] += delta;
246 DCHECK_GE(coverage_[element], 0);
247 DCHECK_LE(coverage_[element], rows[element].size().value());
248 if (coverage_[element] == 1) {
249 ++num_elements_covered_;
250 } else if (coverage_[element] == 0) {
251 --num_elements_covered_;
252 }
253 }
254 DCHECK(CheckSingleSubsetCoverage(subset));
255}

257// Compute the impact of the change in the assignment for each subset
258// containing element. Store this in a flat_hash_set so as to buffer the
259// change.
260std::vector<SubsetIndex> SetCoverLedger::ComputeImpactedSubsets(
261 SubsetIndex subset) const {
262 const SparseColumnView& columns = model_->columns();
263 const SparseRowView& rows = model_->rows();
264 absl::flat_hash_set<SubsetIndex> impacted_subsets_collection;
265 for (const ElementIndex element : columns[subset]) {
266 for (const SubsetIndex subset : rows[element]) {
267 impacted_subsets_collection.insert(subset);
268 }
269 }
270 DCHECK(impacted_subsets_collection.find(subset) !=
271 impacted_subsets_collection.end());
272 std::vector<SubsetIndex> impacted_subsets(impacted_subsets_collection.begin(),
273 impacted_subsets_collection.end());
274 DCHECK_LE(impacted_subsets.size(), model_->num_subsets());
275 std::sort(impacted_subsets.begin(), impacted_subsets.end());
276 return impacted_subsets;
277}

279bool SetCoverLedger::ComputeIsRemovable(SubsetIndex subset) const {
280 DCHECK(CheckSingleSubsetCoverage(subset));
281 const SparseColumnView& columns = model_->columns();
282 for (const ElementIndex element : columns[subset]) {
283 if (coverage_[element] <= 1) {
284 return false;
285 }
286 }
287 return true;

297SubsetBoolVector SetCoverLedger::ComputeIsRemovable(
298 const ElementToSubsetVector& cvrg) const {
299 DCHECK(CheckCoverageAgainstSolution(is_selected_));
300 const SubsetIndex num_subsets(model_->num_subsets());
301 SubsetBoolVector is_rmvble(num_subsets, true);
302 const SparseRowView& rows = model_->rows();
303 for (ElementIndex element(0); element < rows.size(); ++element) {
304 if (cvrg[element] <= 1) {
305 for (const SubsetIndex subset : rows[element]) {
306 is_rmvble[subset] = false;
307 }
308 }
309 }
310 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
311 DCHECK_EQ(is_rmvble[subset], ComputeIsRemovable(subset));
312 }
313 return is_rmvble;
314}

316bool SetCoverLedger::CheckIsRemovable() const {
317 const SubsetBoolVector is_rmvble = ComputeIsRemovable(coverage_);
318 const SubsetIndex num_subsets(model_->num_subsets());
319 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
320 DCHECK_EQ(is_rmvble[subset], ComputeIsRemovable(subset));
321 }
322 return true;

325void SetCoverLedger::UpdateMarginalImpacts(
326 const std::vector<SubsetIndex>& impacted_subsets) {
327 const SparseColumnView& columns = model_->columns();
328 for (const SubsetIndex subset : impacted_subsets) {
329 ElementIndex impact(0);
330 for (const ElementIndex element : columns[subset]) {
331 if (coverage_[element] == 0) {
332 ++impact;
333 }
334 }
335 DVLOG(2) << "Changing impact of subset " << subset << " from "
336 << marginal_impacts_[subset] << " to " << impact;
337 marginal_impacts_[subset] = impact;
338 DCHECK_LE(marginal_impacts_[subset], columns[subset].size().value());
339 DCHECK_GE(marginal_impacts_[subset], 0);
340 }
341 DCHECK(CheckCoverageAndMarginalImpacts(is_selected_));
342}
