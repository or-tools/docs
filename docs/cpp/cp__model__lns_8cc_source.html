85void NeighborhoodGeneratorHelper::Synchronize() {
86 if (shared_bounds_ != nullptr) {
87 std::vector<int> model_variables;
88 std::vector<int64_t> new_lower_bounds;
89 std::vector<int64_t> new_upper_bounds;
90 shared_bounds_->GetChangedBounds(shared_bounds_id_, &model_variables,
91 &new_lower_bounds, &new_upper_bounds);
92
93 bool new_variables_have_been_fixed = false;
94
95 {
96 absl::MutexLock domain_lock(&domain_mutex_);
97
98 for (int i = 0; i < model_variables.size(); ++i) {
99 const int var = model_variables[i];
100 const int64_t new_lb = new_lower_bounds[i];
101 const int64_t new_ub = new_upper_bounds[i];
102 if (VLOG_IS_ON(3)) {
103 const auto& domain =
104 model_proto_with_only_variables_.variables(var).domain();
105 const int64_t old_lb = domain.Get(0);
106 const int64_t old_ub = domain.Get(domain.size() - 1);
107 VLOG(3) << "Variable: " << var << " old domain: [" << old_lb << ", "
108 << old_ub << "] new domain: [" << new_lb << ", " << new_ub
109 << "]";
110 }
111 const Domain old_domain = ReadDomainFromProto(
112 model_proto_with_only_variables_.variables(var));
113 const Domain new_domain =
114 old_domain.IntersectionWith(Domain(new_lb, new_ub));
115 if (new_domain.IsEmpty()) {
116 // This can mean two things:
117 // 1/ This variable is a normal one and the problem is UNSAT or
118 // 2/ This variable is optional, and its associated literal must be
119 // set to false.
120 //
121 // Currently, we wait for any full solver to pick the crossing bounds
122 // and do the correct stuff on their own. We do not want to have empty
123 // domain in the proto as this would means INFEASIBLE. So we just
124 // ignore such bounds here.
125 //
126 // TODO(user): We could set the optional literal to false directly in
127 // the bound sharing manager. We do have to be careful that all the
128 // different solvers have the same optionality definition though.
129 continue;
130 }
131 FillDomainInProto(
132 new_domain,
133 model_proto_with_only_variables_.mutable_variables(var));
134 new_variables_have_been_fixed |= new_domain.IsFixed();
135 }
136 }
137
138 // Only trigger the computation if needed.
139 if (new_variables_have_been_fixed) {
140 RecomputeHelperData();
141 }
142 }
143}

145bool NeighborhoodGeneratorHelper::ObjectiveDomainIsConstraining() const {
146 if (!model_proto_.has_objective()) return false;
147 if (model_proto_.objective().domain().empty()) return false;
148
149 int64_t min_activity = 0;
150 int64_t max_activity = 0;
151 const int num_terms = model_proto_.objective().vars().size();
152 for (int i = 0; i < num_terms; ++i) {
153 const int var = PositiveRef(model_proto_.objective().vars(i));
154 const int64_t coeff = model_proto_.objective().coeffs(i);
155 const auto& var_domain =
156 model_proto_with_only_variables_.variables(var).domain();
157 const int64_t v1 = coeff * var_domain[0];
158 const int64_t v2 = coeff * var_domain[var_domain.size() - 1];
159 min_activity += std::min(v1, v2);
160 max_activity += std::max(v1, v2);
161 }
162
163 const Domain obj_domain = ReadDomainFromProto(model_proto_.objective());
164 const Domain inferred_domain =
165 Domain(min_activity, max_activity)
166 .IntersectionWith(
167 Domain(std::numeric_limits<int64_t>::min(), obj_domain.Max()));
168 return !inferred_domain.IsIncludedIn(obj_domain);
169}

172 type_to_constraints_.clear();
173 const int num_constraints = model_proto_.constraints_size();
174 for (int c = 0; c < num_constraints; ++c) {
175 const int type = model_proto_.constraints(c).constraint_case();
176 if (type >= type_to_constraints_.size()) {
177 type_to_constraints_.resize(type + 1);
178 }
179 type_to_constraints_[type].push_back(c);
180 }
181
182 const int num_variables = model_proto_.variables().size();
183 is_in_objective_.resize(num_variables, false);
184 if (model_proto_.has_objective()) {
185 for (const int ref : model_proto_.objective().vars()) {
186 is_in_objective_[PositiveRef(ref)] = true;
187 }
188 }

191// Recompute all the data when new variables have been fixed. Note that this
192// shouldn't be called if there is no change as it is in O(problem size).
193void NeighborhoodGeneratorHelper::RecomputeHelperData() {
194 absl::MutexLock graph_lock(&graph_mutex_);
195 absl::ReaderMutexLock domain_lock(&domain_mutex_);
196
197 // Do basic presolving to have a more precise graph.
198 // Here we just remove trivially true constraints.
199 //
200 // Note(user): We do that each time a new variable is fixed. It might be too
201 // much, but on the miplib and in 1200s, we do that only about 1k time on the
202 // worst case problem.
203 //
204 // TODO(user): Change API to avoid a few copy?
205 // TODO(user): We could keep the context in the class.
206 // TODO(user): We can also start from the previous simplified model instead.
207 {
208 Model local_model;
209 CpModelProto mapping_proto;
210 simplied_model_proto_.Clear();
211 *simplied_model_proto_.mutable_variables() =
212 model_proto_with_only_variables_.variables();
213 PresolveContext context(&local_model, &simplied_model_proto_,
214 &mapping_proto);
215 ModelCopy copier(&context);
216
217 // TODO(user): Not sure what to do if the model is UNSAT.
218 // This shouldn't matter as it should be dealt with elsewhere.
219 copier.ImportAndSimplifyConstraints(model_proto_, {});
220 }
221
222 // Compute the constraint <-> variable graph.
223 //
224 // TODO(user): Remove duplicate constraints?
225 const auto& constraints = simplied_model_proto_.constraints();
226 constraint_to_var_.clear();
227 constraint_to_var_.reserve(constraints.size());
228 for (int ct_index = 0; ct_index < constraints.size(); ++ct_index) {
229 // We remove the interval constraints since we should have an equivalent
230 // linear constraint somewhere else. This is not the case if we have a fixed
231 // size optional interval variable. But it should not matter as the
232 // intervals are replaced by their underlying variables in the scheduling
233 // constraints.
234 if (constraints[ct_index].constraint_case() == ConstraintProto::kInterval) {
235 continue;
236 }
237
238 tmp_row_.clear();
239 for (const int var : UsedVariables(constraints[ct_index])) {
240 if (IsConstant(var)) continue;
241 tmp_row_.push_back(var);
242 }
243
244 // We replace intervals by their underlying integer variables. Note that
245 // this is needed for a correct decomposition into independent part.
246 bool need_sort = false;
247 for (const int interval : UsedIntervals(constraints[ct_index])) {
248 need_sort = true;
249 for (const int var : UsedVariables(constraints[interval])) {
250 if (IsConstant(var)) continue;
251 tmp_row_.push_back(var);
252 }
253 }
254
255 // We remove constraint of size 0 and 1 since they are not useful for LNS
256 // based on this graph.
257 if (tmp_row_.size() <= 1) {
258 continue;
259 }
260
261 // Keep this constraint.
262 if (need_sort) {
263 gtl::STLSortAndRemoveDuplicates(&tmp_row_);
264 }
265 constraint_to_var_.Add(tmp_row_);
266 }
267
268 // Initialize var to constraints, and make sure it has an entry for all
269 // variables.
270 var_to_constraint_.ResetFromTranspose(
271 constraint_to_var_,
272 /*min_transpose_size=*/model_proto_.variables().size());
273
274 // We mark as active all non-constant variables.
275 // Non-active variable will never be fixed in standard LNS fragment.
276 active_variables_.clear();
277 const int num_variables = model_proto_.variables_size();
278 active_variables_set_.assign(num_variables, false);
279 for (int i = 0; i < num_variables; ++i) {
280 if (!IsConstant(i)) {
281 active_variables_.push_back(i);
282 active_variables_set_[i] = true;
283 }
284 }
285
286 active_objective_variables_.clear();
287 for (const int var : model_proto_.objective().vars()) {
288 DCHECK(RefIsPositive(var));
289 if (active_variables_set_[var]) {
290 active_objective_variables_.push_back(var);
291 }
292 }
293
294 // Compute connected components.
295 // Note that fixed variable are just ignored.
296 DenseConnectedComponentsFinder union_find;
297 union_find.SetNumberOfNodes(num_variables);
298 for (int c = 0; c < constraint_to_var_.size(); ++c) {
299 const auto row = constraint_to_var_[c];
300 if (row.size() <= 1) continue;
301 for (int i = 1; i < row.size(); ++i) {
302 union_find.AddEdge(row[0], row[i]);
303 }
304 }
305
306 // If we have a lower bound on the objective, then this "objective constraint"
307 // might link components together.
308 if (ObjectiveDomainIsConstraining()) {
309 const auto& refs = model_proto_.objective().vars();
310 const int num_terms = refs.size();
311 for (int i = 1; i < num_terms; ++i) {
312 union_find.AddEdge(PositiveRef(refs[0]), PositiveRef(refs[i]));
313 }
314 }
315
316 // Compute all components involving non-fixed variables.
317 //
318 // TODO(user): If a component has no objective, we can fix it to any feasible
319 // solution. This will automatically be done by LNS fragment covering such
320 // component though.
321 components_.clear();
322 var_to_component_index_.assign(num_variables, -1);
323 for (int var = 0; var < num_variables; ++var) {
324 if (IsConstant(var)) continue;
325 const int root = union_find.FindRoot(var);
326 DCHECK_LT(root, var_to_component_index_.size());
327 int& index = var_to_component_index_[root];
328 if (index == -1) {
329 index = components_.size();
330 components_.push_back({});
331 }
332 var_to_component_index_[var] = index;
333 components_[index].push_back(var);
334 }
335
336 // Display information about the reduced problem.
337 //
338 // TODO(user): Exploit connected component while generating fragments.
339 // TODO(user): Do not generate fragment not touching the objective.
340 if (!shared_response_->LoggingIsEnabled()) return;
341
342 std::vector<int> component_sizes;
343 for (const std::vector<int>& component : components_) {
344 component_sizes.push_back(component.size());
345 }
346 std::sort(component_sizes.begin(), component_sizes.end(),
347 std::greater<int>());
348 std::string compo_message;
349 if (component_sizes.size() > 1) {
350 if (component_sizes.size() <= 10) {
351 compo_message =
352 absl::StrCat(" compo:", absl::StrJoin(component_sizes, ","));
353 } else {
354 component_sizes.resize(10);
355 compo_message =
356 absl::StrCat(" compo:", absl::StrJoin(component_sizes, ","), ",...");
357 }
358 }
359
360 // TODO(user): This is not ideal, as if two reductions appears in a row and
361 // nothing else is done for a while, we will never see the "latest" size
362 // in the log until it is reduced again.
363 shared_response_->LogMessageWithThrottling(
364 "Model",
365 absl::StrCat("var:", active_variables_.size(), "/", num_variables,
366 " constraints:", simplied_model_proto_.constraints().size(),
367 "/", model_proto_.constraints().size(), compo_message));
368}

398bool NeighborhoodGeneratorHelper::IntervalIsActive(
399 int index, const CpSolverResponse& initial_solution) const {
400 const ConstraintProto& interval_ct = ModelProto().constraints(index);
401 // We only look at intervals that are performed in the solution. The
402 // unperformed intervals should be automatically freed during the generation
403 // phase.
404 if (interval_ct.enforcement_literal().size() == 1) {
405 const int enforcement_ref = interval_ct.enforcement_literal(0);
406 const int enforcement_var = PositiveRef(enforcement_ref);
407 const int value = initial_solution.solution(enforcement_var);
408 if (RefIsPositive(enforcement_ref) == (value == 0)) return false;
409 }
410
411 for (const int v : interval_ct.interval().start().vars()) {
412 if (!IsConstant(v)) return true;
413 }
414 for (const int v : interval_ct.interval().size().vars()) {
415 if (!IsConstant(v)) return true;
416 }
417 for (const int v : interval_ct.interval().end().vars()) {
418 if (!IsConstant(v)) return true;
419 }
420 return false;
421}

458 active_rectangles.push_back({x_i, y_i});

460 }
461 }

463 return active_rectangles;

466std::vector<std::vector<int>>
467NeighborhoodGeneratorHelper::GetUniqueIntervalSets() const {
468 std::vector<std::vector<int>> intervals_in_constraints;
469 absl::flat_hash_set<std::vector<int>> added_intervals_sets;
470 const auto add_interval_list_only_once =
471 [&intervals_in_constraints,
472 &added_intervals_sets](const auto& intervals) {
473 std::vector<int> candidate({intervals.begin(), intervals.end()});
474 gtl::STLSortAndRemoveDuplicates(&candidate);
475 if (added_intervals_sets.insert(candidate).second) {
476 intervals_in_constraints.push_back(candidate);
477 }
478 };
479 for (const int ct_index : TypeToConstraints(ConstraintProto::kNoOverlap)) {
480 add_interval_list_only_once(
481 model_proto_.constraints(ct_index).no_overlap().intervals());
482 }
483 for (const int ct_index : TypeToConstraints(ConstraintProto::kCumulative)) {
484 add_interval_list_only_once(
485 model_proto_.constraints(ct_index).cumulative().intervals());
486 }
487 for (const int ct_index : TypeToConstraints(ConstraintProto::kNoOverlap2D)) {
488 add_interval_list_only_once(
489 model_proto_.constraints(ct_index).no_overlap_2d().x_intervals());
490 add_interval_list_only_once(
491 model_proto_.constraints(ct_index).no_overlap_2d().y_intervals());
492 }
493 return intervals_in_constraints;
494}

498int64_t GetLinearExpressionValue(const LinearExpressionProto& expr,
499 const CpSolverResponse& initial_solution) {
500 int64_t result = expr.offset();
501 for (int i = 0; i < expr.vars_size(); ++i) {
502 result += expr.coeffs(i) * initial_solution.solution(expr.vars(i));
503 }
504 return result;
505}

524// Selects all intervals in a random time window to meet the difficulty
525// requirement.
526TimePartition PartitionIndicesAroundRandomTimeWindow(
527 const std::vector<int>& intervals, const CpModelProto& model_proto,
528 const CpSolverResponse& initial_solution, double difficulty,
529 absl::BitGenRef random) {
530 std::vector<StartEndIndex> start_end_indices;
531 for (int index = 0; index < intervals.size(); ++index) {
532 const int interval = intervals[index];
533 const ConstraintProto& interval_ct = model_proto.constraints(interval);
534 const int64_t start_value = GetLinearExpressionValue(
535 interval_ct.interval().start(), initial_solution);
536 const int64_t end_value = GetLinearExpressionValue(
537 interval_ct.interval().end(), initial_solution);
538 start_end_indices.push_back(
539 {start_value, end_value, index, absl::Uniform(random, 0., 1.0)});
540 }
541
542 if (start_end_indices.empty()) return {};
543
544 std::sort(start_end_indices.begin(), start_end_indices.end());
545 const int relaxed_size = std::floor(difficulty * start_end_indices.size());
546
547 std::uniform_int_distribution<int> random_var(
548 0, start_end_indices.size() - relaxed_size - 1);
549 // TODO(user): Consider relaxing more than one time window
550 // intervals. This seems to help with Giza models.
551 const int random_start_index = random_var(random);
552
553 // We want to minimize the time window relaxed, so we now sort the interval
554 // after the first selected intervals by end value.
555 // TODO(user): We could do things differently (include all tasks <= some
556 // end). The difficulty is that the number of relaxed tasks will differ from
557 // the target. We could also tie break tasks randomly.
558 std::sort(start_end_indices.begin() + random_start_index,
559 start_end_indices.end(),
560 [](const StartEndIndex& a, const StartEndIndex& b) {
561 return std::tie(a.end, a.noise, a.index_in_input_vector) <
562 std::tie(b.end, b.noise, b.index_in_input_vector);
563 });
564 TimePartition result;
565 int i = 0;
566 for (; i < random_start_index; ++i) {
567 result.indices_before_selected.push_back(
568 start_end_indices[i].index_in_input_vector);
569 }
570 for (; i < random_start_index + relaxed_size; ++i) {
571 result.selected_indices.push_back(
572 start_end_indices[i].index_in_input_vector);
573 }
574 for (; i < start_end_indices.size(); ++i) {
575 result.indices_after_selected.push_back(
576 start_end_indices[i].index_in_input_vector);
577 }
578 return result;
579}

581struct Demand {
582 int interval_index;
583 int64_t start;
584 int64_t end;
585 int64_t height;
586
587 // Because of the binary splitting of the capacity in the procedure used to
588 // extract precedences out of a cumulative constraint, processing bigger
589 // heights first will decrease its probability of being split across the 2
590 // halves of the current split.
591 bool operator<(const Demand& other) const {
592 return std::tie(start, height, end) <
593 std::tie(other.start, other.height, other.end);
594 }
595
596 std::string DebugString() const {
597 return absl::StrCat("{i=", interval_index, " span=[", start, ",", end, "]",
598 " d=", height, "}");
599 }
600};

612bool IsPresent(const ConstraintProto& interval_ct,
613 const CpSolverResponse& initial_solution) {
614 if (interval_ct.enforcement_literal().size() != 1) return true;
615
616 const int enforcement_ref = interval_ct.enforcement_literal(0);
617 const int enforcement_var = PositiveRef(enforcement_ref);
618 const int64_t value = initial_solution.solution(enforcement_var);
619 return RefIsPositive(enforcement_ref) == (value == 1);
620}

622void InsertNoOverlapPrecedences(
623 const absl::flat_hash_set<int>& ignored_intervals,
624 const CpSolverResponse& initial_solution, const CpModelProto& model_proto,
625 int no_overlap_index,
626 absl::flat_hash_set<std::pair<int, int>>* precedences) {
627 std::vector<Demand> demands;
628 const NoOverlapConstraintProto& no_overlap =
629 model_proto.constraints(no_overlap_index).no_overlap();
630 for (const int interval_index : no_overlap.intervals()) {
631 if (ignored_intervals.contains(interval_index)) continue;
632 const ConstraintProto& interval_ct =
633 model_proto.constraints(interval_index);
634 if (!IsPresent(interval_ct, initial_solution)) continue;
635
636 const int64_t start_value = GetLinearExpressionValue(
637 interval_ct.interval().start(), initial_solution);
638 const int64_t end_value = GetLinearExpressionValue(
639 interval_ct.interval().end(), initial_solution);
640 DCHECK_LE(start_value, end_value);
641 demands.push_back({interval_index, start_value, end_value, 1});
642 }
643
644 // TODO(user): We actually only need interval_index, start.
645 // No need to fill the other fields here.
646 std::sort(demands.begin(), demands.end());
647 InsertPrecedencesFromSortedListOfNonOverlapingIntervals(demands, precedences);
648}

650void ProcessDemandListFromCumulativeConstraint(
651 const std::vector<Demand>& demands, int64_t capacity,
652 std::deque<std::pair<std::vector<Demand>, int64_t>>* to_process,
653 absl::BitGenRef random,
654 absl::flat_hash_set<std::pair<int, int>>* precedences) {
655 if (demands.size() <= 1) return;
656
657 // Checks if any pairs of tasks cannot overlap.
658 int64_t sum_of_min_two_capacities = 2;
659 if (capacity > 1) {
660 int64_t min1 = std::numeric_limits<int64_t>::max();
661 int64_t min2 = std::numeric_limits<int64_t>::max();
662 for (const Demand& demand : demands) {
663 if (demand.height <= min1) {
664 min2 = min1;
665 min1 = demand.height;
666 } else if (demand.height < min2) {
667 min2 = demand.height;
668 }
669 }
670 sum_of_min_two_capacities = min1 + min2;
671 }
672
673 DCHECK_GT(sum_of_min_two_capacities, 1);
674 if (sum_of_min_two_capacities > capacity) {
675 InsertPrecedencesFromSortedListOfNonOverlapingIntervals(demands,
676 precedences);
677 return;
678 }
679
680 std::vector<int64_t> unique_starts;
681 for (const Demand& demand : demands) {
682 DCHECK(unique_starts.empty() || demand.start >= unique_starts.back());
683 if (unique_starts.empty() || unique_starts.back() < demand.start) {
684 unique_starts.push_back(demand.start);
685 }
686 }
687 DCHECK(std::is_sorted(unique_starts.begin(), unique_starts.end()));
688 const int num_points = unique_starts.size();
689
690 // Split the capacity in 2 and dispatch all demands on the 2 parts.
691 const int64_t capacity1 = capacity / 2;
692 std::vector<int64_t> usage1(num_points);
693 std::vector<Demand> demands1;
694
695 const int64_t capacity2 = capacity - capacity1;
696 std::vector<int64_t> usage2(num_points);
697 std::vector<Demand> demands2;
698
699 int usage_index = 0;
700 for (const Demand& d : demands) {
701 // Since we process demand by increasing start, the usage_index only
702 // need to increase.
703 while (usage_index < num_points && unique_starts[usage_index] < d.start) {
704 usage_index++;
705 }
706 DCHECK_LT(usage_index, num_points);
707 DCHECK_EQ(unique_starts[usage_index], d.start);
708 const int64_t slack1 = capacity1 - usage1[usage_index];
709 const int64_t slack2 = capacity2 - usage2[usage_index];
710
711 // We differ from the ICAPS article. If it fits in both sub-cumulatives, We
712 // choose the smallest slack. If it fits into at most one, we choose the
713 // biggest slack. If both slacks are equal, we choose randomly.
714 const bool prefer2 =
715 slack1 == slack2
716 ? absl::Bernoulli(random, 0.5)
717 : (d.height <= std::min(slack1, slack2) ? slack2 < slack1
718 : slack2 > slack1);
719
720 auto& selected_usage = prefer2 ? usage2 : usage1;
721 auto& residual_usage = prefer2 ? usage1 : usage2;
722 std::vector<Demand>& selected_demands = prefer2 ? demands2 : demands1;
723 std::vector<Demand>& residual_demands = prefer2 ? demands1 : demands2;
724 const int64_t selected_slack = prefer2 ? slack2 : slack1;
725
726 const int64_t assigned_to_selected = std::min(selected_slack, d.height);
727 DCHECK_GT(assigned_to_selected, 0);
728 for (int i = usage_index; i < num_points; ++i) {
729 if (d.end <= unique_starts[i]) break;
730 selected_usage[i] += assigned_to_selected;
731 }
732 selected_demands.push_back(
733 {d.interval_index, d.start, d.end, assigned_to_selected});
734
735 if (d.height > selected_slack) {
736 const int64_t residual = d.height - selected_slack;
737 DCHECK_GT(residual, 0);
738 DCHECK_LE(residual, prefer2 ? slack1 : slack2);
739 for (int i = usage_index; i < num_points; ++i) {
740 if (d.end <= unique_starts[i]) break;
741 residual_usage[i] += residual;
742 }
743 residual_demands.push_back({d.interval_index, d.start, d.end, residual});
744 }
745 }
746
747 if (demands1.size() > 1) {
748 to_process->emplace_back(std::move(demands1), capacity1);
749 }
750 if (demands2.size() > 1) {
751 to_process->emplace_back(std::move(demands2), capacity2);
752 }
753}

755void InsertCumulativePrecedences(
756 const absl::flat_hash_set<int>& ignored_intervals,
757 const CpSolverResponse& initial_solution, const CpModelProto& model_proto,
758 int cumulative_index, absl::BitGenRef random,
759 absl::flat_hash_set<std::pair<int, int>>* precedences) {
760 const CumulativeConstraintProto& cumulative =
761 model_proto.constraints(cumulative_index).cumulative();
762
763 std::vector<Demand> demands;
764 for (int i = 0; i < cumulative.intervals().size(); ++i) {
765 const int interval_index = cumulative.intervals(i);
766 if (ignored_intervals.contains(interval_index)) continue;
767 const ConstraintProto& interval_ct =
768 model_proto.constraints(interval_index);
769 if (!IsPresent(interval_ct, initial_solution)) continue;
770
771 const int64_t start_value = GetLinearExpressionValue(
772 interval_ct.interval().start(), initial_solution);
773 const int64_t end_value = GetLinearExpressionValue(
774 interval_ct.interval().end(), initial_solution);
775 const int64_t demand_value =
776 GetLinearExpressionValue(cumulative.demands(i), initial_solution);
777 if (start_value == end_value || demand_value == 0) continue;
778
779 demands.push_back({interval_index, start_value, end_value, demand_value});
780 }
781 std::sort(demands.begin(), demands.end());
782
783 const int64_t capacity_value =
784 GetLinearExpressionValue(cumulative.capacity(), initial_solution);
785 DCHECK_GT(capacity_value, 0);
786
787 // Copying all these demands is memory intensive. Let's be careful here.
788 std::deque<std::pair<std::vector<Demand>, int64_t>> to_process;
789 to_process.emplace_back(std::move(demands), capacity_value);
790
791 while (!to_process.empty()) {
792 auto& next_task = to_process.front();
793 ProcessDemandListFromCumulativeConstraint(next_task.first,
794 /*capacity=*/next_task.second,
795 &to_process, random, precedences);
796 to_process.pop_front();
797 }
798}

914std::vector<std::vector<int>> NeighborhoodGeneratorHelper::GetRoutingPaths(
915 const CpSolverResponse& initial_solution) const {
916 struct HeadAndArcLiteral {
917 int head;
918 int literal;
919 };
920
921 std::vector<std::vector<int>> result;
922 absl::flat_hash_map<int, HeadAndArcLiteral> tail_to_head_and_arc_literal;
923
924 for (const int i : TypeToConstraints(ConstraintProto::kCircuit)) {
925 const CircuitConstraintProto& ct = ModelProto().constraints(i).circuit();
926
927 // Collect arcs.
928 int min_node = std::numeric_limits<int>::max();
929 tail_to_head_and_arc_literal.clear();
930 for (int i = 0; i < ct.literals_size(); ++i) {
931 const int literal = ct.literals(i);
932 const int head = ct.heads(i);
933 const int tail = ct.tails(i);
934 const int bool_var = PositiveRef(literal);
935 const int64_t value = initial_solution.solution(bool_var);
936 // Skip unselected arcs.
937 if (RefIsPositive(literal) == (value == 0)) continue;
938 // Ignore self loops.
939 if (head == tail) continue;
940 tail_to_head_and_arc_literal[tail] = {head, bool_var};
941 min_node = std::min(tail, min_node);
942 }
943 if (tail_to_head_and_arc_literal.empty()) continue;
944
945 // Unroll the path.
946 int current_node = min_node;
947 std::vector<int> path;
948 do {
949 auto it = tail_to_head_and_arc_literal.find(current_node);
950 CHECK(it != tail_to_head_and_arc_literal.end());
951 current_node = it->second.head;
952 path.push_back(it->second.literal);
953 } while (current_node != min_node);
954 result.push_back(std::move(path));
955 }
956
957 std::vector<HeadAndArcLiteral> route_starts;
958 for (const int i : TypeToConstraints(ConstraintProto::kRoutes)) {
959 const RoutesConstraintProto& ct = ModelProto().constraints(i).routes();
960 tail_to_head_and_arc_literal.clear();
961 route_starts.clear();
962
963 // Collect route starts and arcs.
964 for (int i = 0; i < ct.literals_size(); ++i) {
965 const int literal = ct.literals(i);
966 const int head = ct.heads(i);
967 const int tail = ct.tails(i);
968 const int bool_var = PositiveRef(literal);
969 const int64_t value = initial_solution.solution(bool_var);
970 // Skip unselected arcs.
971 if (RefIsPositive(literal) == (value == 0)) continue;
972 // Ignore self loops.
973 if (head == tail) continue;
974 if (tail == 0) {
975 route_starts.push_back({head, bool_var});
976 } else {
977 tail_to_head_and_arc_literal[tail] = {head, bool_var};
978 }
979 }
980
981 // Unroll all routes.
982 for (const HeadAndArcLiteral& head_var : route_starts) {
983 std::vector<int> path;
984 int current_node = head_var.head;
985 path.push_back(head_var.literal);
986 do {
987 auto it = tail_to_head_and_arc_literal.find(current_node);
988 CHECK(it != tail_to_head_and_arc_literal.end());
989 current_node = it->second.head;
990 path.push_back(it->second.literal);
991 } while (current_node != 0);
992 result.push_back(std::move(path));
993 }
994 }
995
996 return result;
997}

999Neighborhood NeighborhoodGeneratorHelper::FixGivenVariables(
1000 const CpSolverResponse& base_solution,
1001 const absl::flat_hash_set<int>& variables_to_fix) const {
1002 Neighborhood neighborhood;
1003
1004 // TODO(user): Maybe relax all variables in the objective when the number
1005 // is small or negligible compared to the number of variables.
1006 const int unique_objective_variable =
1007 model_proto_.has_objective() && model_proto_.objective().vars_size() == 1
1008 ? model_proto_.objective().vars(0)
1009 : -1;
1010
1011 // Fill in neighborhood.delta all variable domains.
1012 {
1013 absl::ReaderMutexLock domain_lock(&domain_mutex_);
1014
1015 const int num_variables =
1016 model_proto_with_only_variables_.variables().size();
1017 neighborhood.delta.mutable_variables()->Reserve(num_variables);
1018 for (int i = 0; i < num_variables; ++i) {
1019 const IntegerVariableProto& current_var =
1020 model_proto_with_only_variables_.variables(i);
1021 IntegerVariableProto* new_var = neighborhood.delta.add_variables();
1022
1023 // We only copy the name in debug mode.
1024 if (DEBUG_MODE) new_var->set_name(current_var.name());
1025
1026 const Domain domain = ReadDomainFromProto(current_var);
1027 const int64_t base_value = base_solution.solution(i);
1028
1029 if (variables_to_fix.contains(i) && i != unique_objective_variable) {
1030 if (domain.Contains(base_value)) {
1031 new_var->add_domain(base_value);
1032 new_var->add_domain(base_value);
1033 } else {
1034 // If under the updated domain, the base solution is no longer valid,
1035 // We should probably regenerate this neighborhood. But for now we
1036 // just do a best effort and take the closest value.
1037 int64_t closest_value = domain.Min();
1038 int64_t closest_dist = std::abs(closest_value - base_value);
1039 for (const ClosedInterval interval : domain) {
1040 for (const int64_t value : {interval.start, interval.end}) {
1041 const int64_t dist = std::abs(value - base_value);
1042 if (dist < closest_dist) {
1043 closest_value = value;
1044 closest_dist = dist;
1045 }
1046 }
1047 }
1048 FillDomainInProto(Domain(closest_value, closest_value), new_var);
1049 }
1050 } else {
1051 FillDomainInProto(domain, new_var);
1052 }
1053 }
1054 }
1055
1056 // Fill some statistic fields and detect if we cover a full component.
1057 //
1058 // TODO(user): If there is just one component, we can skip some computation.
1059 {
1060 absl::ReaderMutexLock graph_lock(&graph_mutex_);
1061 std::vector<int> count(components_.size(), 0);
1062 const int num_variables = neighborhood.delta.variables().size();
1063 for (int var = 0; var < num_variables; ++var) {
1064 const auto& domain = neighborhood.delta.variables(var).domain();
1065 if (domain.size() != 2 || domain[0] != domain[1]) {
1066 ++neighborhood.num_relaxed_variables;
1067 if (is_in_objective_[var]) {
1068 ++neighborhood.num_relaxed_variables_in_objective;
1069 }
1070 const int c = var_to_component_index_[var];
1071 if (c != -1) count[c]++;
1072 }
1073 }
1074
1075 for (int i = 0; i < components_.size(); ++i) {
1076 if (count[i] == components_[i].size()) {
1077 neighborhood.variables_that_can_be_fixed_to_local_optimum.insert(
1078 neighborhood.variables_that_can_be_fixed_to_local_optimum.end(),
1079 components_[i].begin(), components_[i].end());
1080 }
1081 }
1082 }
1083
1084 // If the objective domain might cut the optimal solution, we cannot exploit
1085 // the connected components. We compute this outside the mutex to avoid
1086 // any deadlock risk.
1087 //
1088 // TODO(user): We could handle some complex domain (size > 2).
1089 if (model_proto_.has_objective() &&
1090 (model_proto_.objective().domain().size() != 2 ||
1091 shared_response_->GetInnerObjectiveLowerBound() <
1092 model_proto_.objective().domain(0))) {
1093 neighborhood.variables_that_can_be_fixed_to_local_optimum.clear();
1094 }
1095
1096 AddSolutionHinting(base_solution, &neighborhood.delta);
1097
1098 neighborhood.is_generated = true;
1099 neighborhood.is_reduced = !variables_to_fix.empty();
1100 neighborhood.is_simple = true;
1101
1102 // TODO(user): force better objective? Note that this is already done when the
1103 // hint above is successfully loaded (i.e. if it passes the presolve
1104 // correctly) since the solver will try to find better solution than the
1105 // current one.
1106 return neighborhood;
1107}

1173double NeighborhoodGenerator::Synchronize() {
1174 absl::MutexLock mutex_lock(&generator_mutex_);
1175
1176 // To make the whole update process deterministic, we currently sort the
1177 // SolveData.
1178 std::sort(solve_data_.begin(), solve_data_.end());
1179
1180 // This will be used to update the difficulty of this neighborhood.
1181 int num_fully_solved_in_batch = 0;
1182 int num_not_fully_solved_in_batch = 0;
1183
1184 double total_dtime = 0.0;
1185 for (const SolveData& data : solve_data_) {
1186 ++num_calls_;
1187
1188 // INFEASIBLE or OPTIMAL means that we "fully solved" the local problem.
1189 // If we didn't, then we cannot be sure that there is no improving solution
1190 // in that neighborhood.
1191 if (data.status == CpSolverStatus::INFEASIBLE ||
1192 data.status == CpSolverStatus::OPTIMAL) {
1193 ++num_fully_solved_calls_;
1194 ++num_fully_solved_in_batch;
1195 } else {
1196 ++num_not_fully_solved_in_batch;
1197 }
1198
1199 // It seems to make more sense to compare the new objective to the base
1200 // solution objective, not the best one. However this causes issue in the
1201 // logic below because on some problems the neighborhood can always lead
1202 // to a better "new objective" if the base solution wasn't the best one.
1203 //
1204 // This might not be a final solution, but it does work ok for now.
1205 const IntegerValue best_objective_improvement = IntegerValue(CapSub(
1206 data.initial_best_objective.value(), data.new_objective.value()));
1207 if (best_objective_improvement > 0) {
1208 num_consecutive_non_improving_calls_ = 0;
1209 next_time_limit_bump_ = 50;
1210 } else {
1211 ++num_consecutive_non_improving_calls_;
1212 }
1213
1214 // Confusing: this one is however comparing to the base solution objective.
1215 if (data.base_objective > data.new_objective) {
1216 ++num_improving_calls_;
1217 }
1218
1219 // TODO(user): Weight more recent data.
1220 // degrade the current average to forget old learnings.
1221 const double gain_per_time_unit =
1222 std::max(0.0, static_cast<double>(best_objective_improvement.value())) /
1223 (1.0 + data.deterministic_time);
1224 if (num_calls_ <= 100) {
1225 current_average_ += (gain_per_time_unit - current_average_) / num_calls_;
1226 } else {
1227 current_average_ = 0.9 * current_average_ + 0.1 * gain_per_time_unit;
1228 }
1229
1230 total_dtime += data.deterministic_time;
1231 }
1232
1233 // Update the difficulty.
1234 difficulty_.Update(/*num_decreases=*/num_not_fully_solved_in_batch,
1235 /*num_increases=*/num_fully_solved_in_batch);
1236
1237 // Bump the time limit if we saw no better solution in the last few calls.
1238 // This means that as the search progress, we likely spend more and more time
1239 // trying to solve individual neighborhood.
1240 //
1241 // TODO(user): experiment with resetting the time limit if a solution is
1242 // found.
1243 if (num_consecutive_non_improving_calls_ > next_time_limit_bump_) {
1244 next_time_limit_bump_ = num_consecutive_non_improving_calls_ + 50;
1245 deterministic_limit_ *= 1.02;
1246
1247 // We do not want the limit to go to high. Intuitively, the goal is to try
1248 // out a lot of neighborhoods, not just spend a lot of time on a few.
1249 deterministic_limit_ = std::min(60.0, deterministic_limit_);
1250 }
1251
1252 solve_data_.clear();
1253 return total_dtime;
1254}

1332 std::vector<int> relaxed_variables;
1333 std::vector<int> visited_variables;
1334
1335 // It is important complexity wise to never scan a constraint twice!
1336 const int num_model_constraints = helper_.ModelProto().constraints_size();

1338
1339 std::vector<int> random_variables;
1340 {
1341 absl::ReaderMutexLock graph_lock(&helper_.graph_mutex_);
1342
1343 // The number of active variables can decrease asynchronously.

1345 const int num_active_vars =

1353 num_objective_variables > 0 // Prefer objective variables.
1354 ? helper_.ActiveObjectiveVariablesWhileHoldingLock()
1355 [absl::Uniform<int>(random, 0, num_objective_variables)]

1395// Note that even if difficulty means full neighborhood, we go through the
1396// generation process to never get out of a connected components.
1397Neighborhood ArcGraphNeighborhoodGenerator::Generate(
1398 const CpSolverResponse& initial_solution, double difficulty,
1399 absl::BitGenRef random) {
1400 const int num_model_vars = helper_.ModelProto().variables_size();
1401 if (num_model_vars == 0) return helper_.NoNeighborhood();
1402
1403 // We copy the full graph var <-> constraints so that we can:
1404 // - reduce it in place
1405 // - not hold the mutex too long.
1406 // TODO(user): should we compress it or use a different representation ?
1407 CompactVectorVector<int, int> vars_to_constraints;
1408 CompactVectorVector<int, int> constraints_to_vars;
1409 int num_active_vars = 0;
1410 std::vector<int> active_objective_vars;
1411 {
1412 absl::ReaderMutexLock graph_lock(&helper_.graph_mutex_);
1413 num_active_vars = helper_.ActiveVariablesWhileHoldingLock().size();
1414 active_objective_vars = helper_.ActiveObjectiveVariablesWhileHoldingLock();
1415 constraints_to_vars = helper_.ConstraintToVar();
1416 vars_to_constraints = helper_.VarToConstraint();
1417 }
1418
1419 const int target_size = std::ceil(difficulty * num_active_vars);
1420 if (target_size == 0) return helper_.NoNeighborhood();
1421
1422 // We pick a variable from the objective.
1423 const int num_objective_variables = active_objective_vars.size();
1424 if (num_objective_variables == 0) return helper_.NoNeighborhood();
1425 const int first_var = active_objective_vars[absl::Uniform<int>(
1426 random, 0, num_objective_variables)];
1427
1428 std::vector<bool> relaxed_variables_set(num_model_vars, false);
1429 std::vector<int> relaxed_variables;
1430 // Active vars are relaxed variables with some unexplored neighbors.
1431 std::vector<int> active_vars;
1432
1433 relaxed_variables_set[first_var] = true;
1434 relaxed_variables.push_back(first_var);
1435 active_vars.push_back(first_var);
1436
1437 while (relaxed_variables.size() < target_size) {
1438 if (active_vars.empty()) break; // We have exhausted our component.
1439
1440 const int tail_index = absl::Uniform<int>(random, 0, active_vars.size());
1441 const int tail_var = active_vars[tail_index];
1442 int head_var = tail_var;
1443 while (!vars_to_constraints[tail_var].empty() && head_var == tail_var) {
1444 const auto cts = vars_to_constraints[tail_var];
1445 const int pos_ct = absl::Uniform<int>(random, 0, cts.size());
1446 const int ct = cts[pos_ct];
1447 while (!constraints_to_vars[ct].empty() && head_var == tail_var) {
1448 const auto vars = constraints_to_vars[ct];
1449 const int pos_var = absl::Uniform<int>(random, 0, vars.size());
1450 const int candidate = vars[pos_var];
1451
1452 // We remove the variable as it is either already relaxed, or will be
1453 // relaxed.
1454 constraints_to_vars.RemoveBySwap(ct, pos_var);
1455 if (!relaxed_variables_set[candidate]) {
1456 head_var = candidate;
1457 }
1458 }
1459 if (constraints_to_vars[ct].empty()) {
1460 // This constraint has no more un-relaxed variables.
1461 vars_to_constraints.RemoveBySwap(tail_var, pos_ct);
1462 }
1463 }
1464
1465 // Variable is no longer active ?
1466 if (vars_to_constraints[tail_var].empty()) {
1467 std::swap(active_vars[tail_index], active_vars.back());
1468 active_vars.pop_back();
1469 }
1470
1471 if (head_var != tail_var) {
1472 relaxed_variables_set[head_var] = true;
1473 relaxed_variables.push_back(head_var);
1474 active_vars.push_back(head_var);
1475 }
1476 }
1477 return helper_.RelaxGivenVariables(initial_solution, relaxed_variables);
1478}

1480// Note that even if difficulty means full neighborhood, we go through the
1481// generation process to never get out of a connected components.
1482Neighborhood ConstraintGraphNeighborhoodGenerator::Generate(
1483 const CpSolverResponse& initial_solution, double difficulty,
1484 absl::BitGenRef random) {
1485 const int num_model_constraints = helper_.ModelProto().constraints_size();
1486 if (num_model_constraints == 0) {
1487 return helper_.FullNeighborhood();
1488 }
1489
1490 const int num_model_vars = helper_.ModelProto().variables_size();
1491 std::vector<bool> visited_variables_set(num_model_vars, false);
1492 std::vector<int> relaxed_variables;
1493
1494 std::vector<bool> added_constraints(num_model_constraints, false);
1495 std::vector<int> next_constraints;
1496
1497 std::vector<int> random_variables;
1498 {
1499 absl::ReaderMutexLock graph_lock(&helper_.graph_mutex_);
1500 const int num_active_vars =
1501 helper_.ActiveVariablesWhileHoldingLock().size();
1502 const int target_size = std::ceil(difficulty * num_active_vars);
1503 if (target_size == num_active_vars) return helper_.FullNeighborhood();
1504
1505 // Start by a random constraint.
1506 const int num_active_constraints = helper_.ConstraintToVar().size();
1507 if (num_active_constraints != 0) {
1508 next_constraints.push_back(
1509 absl::Uniform<int>(random, 0, num_active_constraints));
1510 added_constraints[next_constraints.back()] = true;
1511 }
1512
1513 while (relaxed_variables.size() < target_size) {
1514 // Stop if we have a full connected component.
1515 if (next_constraints.empty()) break;
1516
1517 // Pick a random unprocessed constraint.
1518 const int i = absl::Uniform<int>(random, 0, next_constraints.size());
1519 const int constraint_index = next_constraints[i];
1520 std::swap(next_constraints[i], next_constraints.back());
1521 next_constraints.pop_back();
1522
1523 // Add all the variable of this constraint and increase the set of next
1524 // possible constraints.
1525 DCHECK_LT(constraint_index, num_active_constraints);
1526 random_variables.assign(
1527 helper_.ConstraintToVar()[constraint_index].begin(),
1528 helper_.ConstraintToVar()[constraint_index].end());
1529 std::shuffle(random_variables.begin(), random_variables.end(), random);
1530 for (const int var : random_variables) {
1531 if (visited_variables_set[var]) continue;
1532 visited_variables_set[var] = true;
1533 if (helper_.IsActive(var)) {
1534 relaxed_variables.push_back(var);
1535 }
1536 if (relaxed_variables.size() >= target_size) break;
1537
1538 for (const int ct : helper_.VarToConstraint()[var]) {
1539 if (added_constraints[ct]) continue;
1540 added_constraints[ct] = true;
1541 next_constraints.push_back(ct);
1542 }
1543 }
1544 }
1545 }
1546
1547 return helper_.RelaxGivenVariables(initial_solution, relaxed_variables);
1548}

1550Neighborhood DecompositionGraphNeighborhoodGenerator::Generate(
1551 const CpSolverResponse& initial_solution, double difficulty,
1552 absl::BitGenRef random) {
1553 int max_width = 0;
1554 int size_at_min_width_after_100;
1555 int min_width_after_100 = std::numeric_limits<int>::max();
1556 int num_zero_score = 0;
1557 std::vector<int> relaxed_variables;
1558
1559 // Note(user): The algo is slower than the other graph generator, so we
1560 // might not want to lock the graph for so long? it is just a reader lock
1561 // though.
1562 {
1563 absl::ReaderMutexLock graph_lock(&helper_.graph_mutex_);
1564
1565 const int num_active_vars =
1566 helper_.ActiveVariablesWhileHoldingLock().size();
1567 const int target_size = std::ceil(difficulty * num_active_vars);
1568 if (target_size == num_active_vars) return helper_.FullNeighborhood();
1569
1570 const int num_vars = helper_.VarToConstraint().size();
1571 const int num_constraints = helper_.ConstraintToVar().size();
1572 if (num_constraints == 0 || num_vars == 0) {
1573 return helper_.FullNeighborhood();
1574 }
1575
1576 // We will grow this incrementally.
1577 // Index in the graph are first variables then constraints.
1578 const int num_nodes = num_vars + num_constraints;
1579 std::vector<bool> added(num_nodes, false);
1580 std::vector<bool> added_or_connected(num_nodes, false);
1581
1582 // We will process var/constraint node by minimum "score".
1583 struct QueueElement {
1584 int Index() const { return index; }
1585 bool operator<(const QueueElement& o) const {
1586 if (score == o.score) return tie_break < o.tie_break;
1587 return score < o.score;
1588 }
1589
1590 int index;
1591 int score = 0;
1592 double tie_break = 0.0;
1593 };
1594 std::vector<QueueElement> elements(num_nodes);
1595 IntegerPriorityQueue<QueueElement> pq(num_nodes);
1596
1597 // Initialize elements.
1598 for (int i = 0; i < num_nodes; ++i) {
1599 elements[i].index = i;
1600 elements[i].tie_break = absl::Uniform<double>(random, 0.0, 1.0);
1601 }
1602
1603 // We start by a random active variable.
1604 //
1605 // Note that while num_vars contains all variables, all the fixed variable
1606 // will have no associated constraint, so we don't want to start from a
1607 // random variable.
1608 //
1609 // TODO(user): Does starting by a constraint make sense too?
1610 const int first_index =
1611 helper_.ActiveVariablesWhileHoldingLock()[absl::Uniform<int>(
1612 random, 0, num_active_vars)];
1613 elements[first_index].score = helper_.VarToConstraint()[first_index].size();
1614 pq.Add(elements[first_index]);
1615 added_or_connected[first_index] = true;
1616
1617 // Pop max-degree from queue and update.
1618 std::vector<int> to_update;
1619 while (!pq.IsEmpty() && relaxed_variables.size() < target_size) {
1620 // Just for logging.
1621 if (relaxed_variables.size() > 100 && pq.Size() < min_width_after_100) {
1622 min_width_after_100 = pq.Size();
1623 size_at_min_width_after_100 = relaxed_variables.size();
1624 }
1625
1626 const int index = pq.Top().index;
1627 const int score = pq.Top().score;
1628 pq.Pop();
1629 added[index] = true;
1630
1631 // When the score is zero, we don't need to update anything since the
1632 // frontier does not grow.
1633 if (score == 0) {
1634 if (index < num_vars) relaxed_variables.push_back(index);
1635 ++num_zero_score;
1636 continue;
1637 }
1638
1639 // Note that while it might looks bad, the overall complexity of this is
1640 // in O(num_edge) since we scan each index once and each newly connected
1641 // vertex once.
1642 int num_added = 0;
1643 to_update.clear();
1644 if (index < num_vars) {
1645 relaxed_variables.push_back(index);
1646 for (const int c : helper_.VarToConstraint()[index]) {
1647 const int c_index = num_vars + c;
1648 if (added_or_connected[c_index]) continue;
1649 ++num_added;
1650 added_or_connected[c_index] = true;
1651 to_update.push_back(c_index);
1652 for (const int v : helper_.ConstraintToVar()[c]) {
1653 if (added[v]) continue;
1654 if (added_or_connected[v]) {
1655 to_update.push_back(v);
1656 elements[v].score--;
1657 } else {
1658 elements[c_index].score++;
1659 }
1660 }
1661 }
1662 } else {
1663 for (const int v : helper_.ConstraintToVar()[index - num_vars]) {
1664 if (added_or_connected[v]) continue;
1665 ++num_added;
1666 added_or_connected[v] = true;
1667 to_update.push_back(v);
1668 for (const int c : helper_.VarToConstraint()[v]) {
1669 if (added[num_vars + c]) continue;
1670 if (added_or_connected[num_vars + c]) {
1671 elements[num_vars + c].score--;
1672 to_update.push_back(num_vars + c);
1673 } else {
1674 elements[v].score++;
1675 }
1676 }
1677 }
1678 }
1679
1680 // The score is exactly the frontier increase in size.
1681 // This is the same as the min-degree heuristic for the elimination order.
1682 // Except we only consider connected nodes.
1683 CHECK_EQ(num_added, score);
1684
1685 gtl::STLSortAndRemoveDuplicates(&to_update);
1686 for (const int index : to_update) {
1687 DCHECK(!added[index]);
1688 if (pq.Contains(index)) {
1689 pq.ChangePriority(elements[index]);
1690 } else {
1691 pq.Add(elements[index]);
1692 }
1693 }
1694
1695 max_width = std::max(max_width, pq.Size());
1696 }
1697
1698 // Just for logging.
1699 if (pq.Size() < min_width_after_100) {
1700 min_width_after_100 = pq.Size();
1701 size_at_min_width_after_100 = relaxed_variables.size();
1702 }
1703
1704 VLOG(2) << "#relaxed " << relaxed_variables.size() << " #zero_score "
1705 << num_zero_score << " max_width " << max_width
1706 << " (size,min_width)_after_100 (" << size_at_min_width_after_100
1707 << "," << min_width_after_100 << ") " << " final_width "
1708 << pq.Size();
1709 }
1710
1711 return helper_.RelaxGivenVariables(initial_solution, relaxed_variables);
1712}

1716// Given a (sub)set of binary variables and their initial solution values,
1717// returns a local branching constraint over these variables, that is:
1718// sum_{i : s[i] == 0} x_i + sum_{i : s[i] == 1} (1 - x_i) <= k
1719// where s is the initial solution and k is the neighborhood size. Requires all
1720// variables and initial solution values to be binary.
1721ConstraintProto LocalBranchingConstraint(
1722 const std::vector<int>& variable_indices,
1723 const std::vector<int64_t>& initial_solution, const int neighborhood_size) {
1724 DCHECK_EQ(variable_indices.size(), initial_solution.size());
1725 DCHECK_GE(neighborhood_size, 0);
1726 ConstraintProto local_branching_constraint;
1727 local_branching_constraint.set_name("local_branching");
1728 LinearConstraintProto* linear = local_branching_constraint.mutable_linear();
1729 int lhs_constant_value = 0;
1730 for (int i = 0; i < variable_indices.size(); ++i) {
1731 if (initial_solution[i] == 0) {
1732 linear->add_coeffs(1);
1733 linear->add_vars(variable_indices[i]);
1734 } else {
1735 DCHECK_EQ(initial_solution[i], 1);
1736 linear->add_coeffs(-1);
1737 linear->add_vars(variable_indices[i]);
1738 lhs_constant_value++;
1739 }
1740 }
1741 linear->add_domain(-lhs_constant_value);
1742 linear->add_domain(-lhs_constant_value + neighborhood_size);
1743 return local_branching_constraint;
1744}

1748Neighborhood LocalBranchingLpBasedNeighborhoodGenerator::Generate(
1749 const CpSolverResponse& initial_solution, double difficulty,
1750 absl::BitGenRef random) {
1751 std::vector<int> active_variables = helper_.ActiveVariables();
1752
1753 // Collect active binary variables and corresponding initial solution values.
1754 // TODO(user): Extend to integer variables.
1755 std::vector<int> binary_var_indices;
1756 std::vector<int> non_binary_var_indices;
1757 std::vector<int64_t> binary_var_initial_solution;
1758 for (const int active_var_index : active_variables) {
1759 const IntegerVariableProto& var =
1760 helper_.ModelProto().variables(active_var_index);
1761 if (var.domain_size() == 2 && var.domain(0) == 0 && var.domain(1) == 1) {
1762 binary_var_indices.push_back(active_var_index);
1763 binary_var_initial_solution.push_back(
1764 initial_solution.solution(active_var_index));
1765 } else {
1766 non_binary_var_indices.push_back(active_var_index);
1767 }
1768 }
1769 if (binary_var_indices.empty()) {
1770 return helper_.NoNeighborhood();
1771 }
1772
1773 const int target_size =
1774 static_cast<int>(std::ceil(difficulty * binary_var_indices.size()));
1775
1776 // Create and solve local branching LP.
1777 CpModelProto local_branching_model = helper_.UpdatedModelProtoCopy();
1778 *local_branching_model.add_constraints() = LocalBranchingConstraint(
1779 binary_var_indices, binary_var_initial_solution, target_size);
1780 Model model("lb_relax_lns_lp");
1781 auto* const params = model.GetOrCreate<SatParameters>();
1782 // Parameters to enable solving the LP only.
1783 params->set_num_workers(1);
1784 params->set_linearization_level(2);
1785 params->set_stop_after_root_propagation(true);
1786 params->set_add_lp_constraints_lazily(false);
1787 // Parameters to attempt to speed up solve.
1788 params->set_cp_model_presolve(false);
1789 params->set_cp_model_probing_level(0);
1790 // Parameters to limit time spent in the solve. The max number of iterations
1791 // is relaxed from the default since we rely more on deterministic time.
1792 params->set_root_lp_iterations(100000);
1793 params->set_max_deterministic_time(10);
1794 if (global_time_limit_ != nullptr) {
1795 global_time_limit_->UpdateLocalLimit(model.GetOrCreate<TimeLimit>());
1796 }
1797 solve_callback_(local_branching_model, &model);
1798
1799 // Skip LNS if no (full) feasible solution was found for the LP.
1800 const auto lp_constraints =
1801 model.GetOrCreate<LinearProgrammingConstraintCollection>();
1802 for (const LinearProgrammingConstraint* lp_constraint : *lp_constraints) {
1803 if (!lp_constraint->HasSolution()) {
1804 return helper_.NoNeighborhood();
1805 }
1806 }
1807
1808 // Compute differences between LP solution and initial solution, with a small
1809 // random noise for tie breaking.
1810 const auto var_mapping = model.GetOrCreate<CpModelMapping>();
1811 const auto lp_solution = model.GetOrCreate<ModelLpValues>();
1812 std::vector<double> differences;
1813 for (int i = 0; i < binary_var_indices.size(); ++i) {
1814 double difference =
1815 std::abs(lp_solution->at(var_mapping->Integer(binary_var_indices[i])) -
1816 binary_var_initial_solution[i]);
1817 differences.push_back(difference +
1818 absl::Uniform<double>(random, 0.0, 1e-6));
1819 }
1820
1821 // Take the target_size variables with largest differences.
1822 std::vector<int> vars_to_relax(binary_var_indices.size());
1823 absl::c_iota(vars_to_relax, 0);
1824 absl::c_sort(vars_to_relax, [&differences](const int i, const int j) {
1825 return differences[i] > differences[j];
1826 });
1827 vars_to_relax.resize(target_size);
1828
1829 // For now, we include all non-binary variables in the relaxation, since their
1830 // values are likely tied to the binary values.
1831 vars_to_relax.insert(vars_to_relax.end(), non_binary_var_indices.begin(),
1832 non_binary_var_indices.end());
1833 return helper_.RelaxGivenVariables(initial_solution, vars_to_relax);

1861 neighborhood.is_reduced = !precedences.empty();
1862 if (!neighborhood.is_reduced) { // Returns the full neighborhood.
1863 helper.AddSolutionHinting(initial_solution, &neighborhood.delta);
1864 neighborhood.is_generated = true;

1866 }
1867
1868 // Collect seen intervals.
1869 absl::flat_hash_set<int> seen_intervals;
1870 for (const std::pair<int, int>& prec : precedences) {
1871 seen_intervals.insert(prec.first);
1872 seen_intervals.insert(prec.second);
1873 }
1874
1875 // Fix the presence/absence of unseen intervals.
1876 bool enforcement_literals_fixed = false;
1877 for (const int i : helper.TypeToConstraints(ConstraintProto::kInterval)) {
1878 if (seen_intervals.contains(i)) continue;

1880 const ConstraintProto& interval_ct = helper.ModelProto().constraints(i);
1881 if (interval_ct.enforcement_literal().empty()) continue;
1882
1883 DCHECK_EQ(interval_ct.enforcement_literal().size(), 1);
1884 const int enforcement_ref = interval_ct.enforcement_literal(0);
1885 const int enforcement_var = PositiveRef(enforcement_ref);
1886 const int value = initial_solution.solution(enforcement_var);
1887
1888 // If the interval is not enforced, we just relax it. If it belongs to an
1889 // exactly one constraint, and the enforced interval is not relaxed, then
1890 // propagation will force this interval to stay not enforced. Otherwise,
1891 // LNS will be able to change which interval will be enforced among all
1892 // alternatives.
1893 if (RefIsPositive(enforcement_ref) == (value == 0)) continue;
1894
1895 // Fix the value.
1896 neighborhood.delta.mutable_variables(enforcement_var)->clear_domain();
1897 neighborhood.delta.mutable_variables(enforcement_var)->add_domain(value);
1898 neighborhood.delta.mutable_variables(enforcement_var)->add_domain(value);
1899 enforcement_literals_fixed = true;
1900 }
1901
1902 for (const std::pair<int, int>& prec : precedences) {
1903 const LinearExpressionProto& before_end =
1904 helper.ModelProto().constraints(prec.first).interval().end();
1905 const LinearExpressionProto& after_start =
1906 helper.ModelProto().constraints(prec.second).interval().start();
1907 DCHECK_LE(GetLinearExpressionValue(before_end, initial_solution),
1908 GetLinearExpressionValue(after_start, initial_solution));
1909 AddPrecedence(before_end, after_start, &neighborhood.delta);
1910 }

1913 helper.AddSolutionHinting(initial_solution, &neighborhood.delta);

1921 absl::Span<const int> variables_to_fix,
1922 const CpSolverResponse& initial_solution, absl::BitGenRef random,
1923 const NeighborhoodGeneratorHelper& helper) {
1924 Neighborhood neighborhood = helper.FullNeighborhood();

1926 // We will extend the set with some interval that we cannot fix.
1927 absl::flat_hash_set<int> ignored_intervals(intervals_to_relax.begin(),
1928 intervals_to_relax.end());
1929
1930 // Fix the presence/absence of non-relaxed intervals.
1931 for (const int i : helper.TypeToConstraints(ConstraintProto::kInterval)) {
1932 DCHECK_GE(i, 0);
1933 if (ignored_intervals.contains(i)) continue;
1934
1935 const ConstraintProto& interval_ct = helper.ModelProto().constraints(i);
1936 if (interval_ct.enforcement_literal().empty()) continue;

1938 DCHECK_EQ(interval_ct.enforcement_literal().size(), 1);
1939 const int enforcement_ref = interval_ct.enforcement_literal(0);
1940 const int enforcement_var = PositiveRef(enforcement_ref);
1941 const int value = initial_solution.solution(enforcement_var);
1942

1944 // exactly one constraint, and the enforced interval is not relaxed, then
1945 // propagation will force this interval to stay not enforced. Otherwise,
1946 // LNS will be able to change which interval will be enforced among all
1947 // alternatives.
1948 if (RefIsPositive(enforcement_ref) == (value == 0)) {
1949 ignored_intervals.insert(i);
1950 continue;
1951 }
1952
1953 // Fix the value.
1954 neighborhood.delta.mutable_variables(enforcement_var)->clear_domain();
1955 neighborhood.delta.mutable_variables(enforcement_var)->add_domain(value);
1956 neighborhood.delta.mutable_variables(enforcement_var)->add_domain(value);
1957 }
1958
1959 if (ignored_intervals.size() >=
1960 helper.TypeToConstraints(ConstraintProto::kInterval)
1961 .size()) { // Returns the full neighborhood.
1962 helper.AddSolutionHinting(initial_solution, &neighborhood.delta);
1963 neighborhood.is_generated = true;
1964 return neighborhood;
1965 }
1966
1967 neighborhood.is_reduced = true;
1968
1969 // We differ from the ICAPS05 paper as we do not consider ignored intervals
1970 // when generating the precedence graph, instead of building the full graph,
1971 // then removing intervals, and reconstructing the precedence graph
1972 // heuristically after that.

1974 helper.GetSchedulingPrecedences(ignored_intervals, initial_solution,
1975 random);

1978 helper.ModelProto().constraints(prec.first).interval().end();
1979 const LinearExpressionProto& after_start =
1980 helper.ModelProto().constraints(prec.second).interval().start();

1982 GetLinearExpressionValue(after_start, initial_solution));
1983 AddPrecedence(before_end, after_start, &neighborhood.delta);
1984 }
1985
1986 // fix the extra variables passed as parameters.
1987 for (const int var : variables_to_fix) {
1988 const int value = initial_solution.solution(var);
1989 neighborhood.delta.mutable_variables(var)->clear_domain();
1990 neighborhood.delta.mutable_variables(var)->add_domain(value);
1991 neighborhood.delta.mutable_variables(var)->add_domain(value);
1992 }
1993
1994 // Set the current solution as a hint.
1995 helper.AddSolutionHinting(initial_solution, &neighborhood.delta);
1996 neighborhood.is_generated = true;
1997
1998 return neighborhood;
