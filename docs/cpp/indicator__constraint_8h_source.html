33// A value type that references an indicator constraint from ModelStorage.
34// Usually this type is passed by copy.
35class IndicatorConstraint {
36 public:
37 // The typed integer used for ids.
38 using IdType = IndicatorConstraintId;
39
40 inline IndicatorConstraint(const ModelStorage* storage,
41 IndicatorConstraintId id);
42
43 inline int64_t id() const;
44
45 inline IndicatorConstraintId typed_id() const;
46 inline const ModelStorage* storage() const;
47
48 inline absl::string_view name() const;
49
50 // Returns nullopt if the indicator variable is unset (this is a valid state,
51 // in which the constraint is functionally ignored).
52 inline std::optional<Variable> indicator_variable() const;
53
54 // The value the indicator variable takes to activate the implied constraint.
55 inline bool activate_on_zero() const;
56
57 BoundedLinearExpression ImpliedConstraint() const;
58
59 // Returns all variables that appear in the indicator constraint with a
60 // nonzero coefficient. Order is not defined.
61 inline std::vector<Variable> NonzeroVariables() const;
62
63 // Returns a detailed string description of the contents of the constraint
64 // (not its name, use `<<` for that instead).
65 std::string ToString() const;
66
67 friend inline bool operator==(const IndicatorConstraint& lhs,
68 const IndicatorConstraint& rhs);
69 friend inline bool operator!=(const IndicatorConstraint& lhs,
70 const IndicatorConstraint& rhs);
71 template <typename H>
72 friend H AbslHashValue(H h, const IndicatorConstraint& constraint);
73 friend std::ostream& operator<<(std::ostream& ostr,
74 const IndicatorConstraint& constraint);
75
76 private:
77 const ModelStorage* storage_;
78 IndicatorConstraintId id_;
79};

87// Inline function implementations

90int64_t IndicatorConstraint::id() const { return id_.value(); }
91
92IndicatorConstraintId IndicatorConstraint::typed_id() const { return id_; }
93
94const ModelStorage* IndicatorConstraint::storage() const { return storage_; }
95
96absl::string_view IndicatorConstraint::name() const {
97 if (storage_->has_constraint(id_)) {
98 return storage_->constraint_data(id_).name;
99 }
100 return kDeletedConstraintDefaultDescription;
101}
102
103std::optional<Variable> IndicatorConstraint::indicator_variable() const {
104 const std::optional<VariableId> maybe_indicator =
105 storage_->constraint_data(id_).indicator;
106 if (!maybe_indicator.has_value()) {
107 return std::nullopt;
108 }
109 return Variable(storage_, *maybe_indicator);
110}
111
112bool IndicatorConstraint::activate_on_zero() const {
113 return storage_->constraint_data(id_).activate_on_zero;
114}
115
116std::vector<Variable> IndicatorConstraint::NonzeroVariables() const {
117 return AtomicConstraintNonzeroVariables(*storage_, id_);
118}
119
120bool operator==(const IndicatorConstraint& lhs,
121 const IndicatorConstraint& rhs) {
122 return lhs.id_ == rhs.id_ && lhs.storage_ == rhs.storage_;
123}
124
125bool operator!=(const IndicatorConstraint& lhs,
126 const IndicatorConstraint& rhs) {
127 return !(lhs == rhs);
128}
129
130template <typename H>
131H AbslHashValue(H h, const IndicatorConstraint& constraint) {
132 return H::combine(std::move(h), constraint.id_.value(), constraint.storage_);
133}

135std::ostream& operator<<(std::ostream& ostr,
136 const IndicatorConstraint& constraint) {
137 // TODO(b/170992529): handle quoting of invalid characters in the name.
138 const absl::string_view name = constraint.name();
139 if (name.empty()) {

141 } else {
142 ostr << name;
143 }
144 return ostr;
145}
