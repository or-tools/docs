27// This class is used to decompose an existing LinearProgram into several
28// independent LinearPrograms. Problems are independent when none of their
29// variables are connected, i.e. appear in the same constraints.
30// Consider for instance the following problem:
31// min: x + 2 y + 3 z + 4 t + 5 u
32// c1: 0 <= x + z <= 1;
33// c2: 0 <= y + t <= 1;
34// c3: 0 <= x + u <= 1;
35// int: x, y, z, t, u
36// Variables x, z and u are connected by constraints c1 and c3.
37// Variables y and t are connected by constraints c2.
38// The problem can be decomposed into two independent problems:
39// min: x + 3 z + 5 u
40// c1: 0 <= x + z <= 1;
41// c3: 0 <= x + u <= 1;
42// int: x, z, u
43// and
44// min: 2 y + 4 t
45// c2: 0 <= y + t <= 1;
46// int: y, t
47//
48// Note that a solution to those two independent problems is a solution to the
49// original problem.
50class LPDecomposer {
51 public:
52 LPDecomposer();
53
54 // This type is neither copyable nor movable.
55 LPDecomposer(const LPDecomposer&) = delete;
56 LPDecomposer& operator=(const LPDecomposer&) = delete;
57
58 // Decomposes the problem into independent problems.
59 // Note that a pointer is kept (no copy) on the linear_problem, so the problem
60 // should not change during the life of the LPDecomposer object.
61 void Decompose(const LinearProgram* linear_problem)
62 ABSL_LOCKS_EXCLUDED(mutex_);
63
64 // Returns the number of independent problems generated by Decompose().
65 int GetNumberOfProblems() const ABSL_LOCKS_EXCLUDED(mutex_);
66
67 // Returns the original problem, i.e. as it was before any decomposition.
68 const LinearProgram& original_problem() const ABSL_LOCKS_EXCLUDED(mutex_);
69
70 // Fills lp with the problem_index^th independent problem generated by
71 // Decompose().
72 // Note that this method runs in O(num-entries-in-generated-problem).
73 void ExtractLocalProblem(int problem_index, LinearProgram* lp)
74 ABSL_LOCKS_EXCLUDED(mutex_);
75
76 // Returns an assignment to the original problem based on the assignments
77 // to the independent problems. Requires Decompose() to have been called.
78 DenseRow AggregateAssignments(const std::vector<DenseRow>& assignments) const
79 ABSL_LOCKS_EXCLUDED(mutex_);
80
81 // Returns an assignment to the given subproblem based on the assignment to
82 // the original problem. Requires Decompose() to have been called.
83 DenseRow ExtractLocalAssignment(int problem_index, const DenseRow& assignment)
84 ABSL_LOCKS_EXCLUDED(mutex_);
85
86 private:
87 const LinearProgram* original_problem_;
88 std::vector<std::vector<ColIndex>> clusters_;
89
90 mutable absl::Mutex mutex_;
91};
