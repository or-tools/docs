50QuadraticProgram ReadQuadraticProgramOrDie(const std::string& filename,
51 bool include_names) {
52 if (absl::EndsWith(filename, ".mps") || absl::EndsWith(filename, ".mps.gz") ||
53 absl::EndsWith(filename, ".mps.bz2")) {
54 return ReadMpsLinearProgramOrDie(filename, include_names);
55 }
56 if (absl::EndsWith(filename, ".pb") ||
57 absl::EndsWith(filename, ".textproto") ||
58 absl::EndsWith(filename, ".json") ||
59 absl::EndsWith(filename, ".json.gz")) {
60 return ReadMPModelProtoFileOrDie(filename, include_names);
61 }
62 LOG(QFATAL) << "Invalid filename suffix in " << filename
63 << ". Valid suffixes are .mps, .mps.gz, .pb, .textproto,"
64 << ".json, and .json.gz";
65}

67QuadraticProgram ReadMpsLinearProgramOrDie(const std::string& lp_file,
68 bool include_names) {
69 absl::StatusOr<MPModelProto> lp_proto =
70 operations_research::glop::MpsFileToMPModelProto(lp_file);
71 QCHECK_OK(lp_proto);
72 // `MpsFileToMPModelProto` sometimes fails silently if the file isn't read
73 // properly.
74 QCHECK_GT(lp_proto->variable_size(), 0)
75 << "No variables in LP. Error reading file? " << lp_file;
76 auto result = QpFromMpModelProto(*lp_proto, /*relax_integer_variables=*/true,
77 include_names);
78 QCHECK_OK(result);
79 return *std::move(result);
80}

82QuadraticProgram ReadMPModelProtoFileOrDie(
83 const std::string& mpmodel_proto_file, bool include_names) {
84 MPModelProto lp_proto;
85 QCHECK(ReadFileToProto(mpmodel_proto_file, &lp_proto)) << mpmodel_proto_file;
86 auto result = QpFromMpModelProto(lp_proto, /*relax_integer_variables=*/true,
87 include_names);
88 QCHECK_OK(result);
89 return *std::move(result);

92absl::Status WriteLinearProgramToMps(const QuadraticProgram& linear_program,
93 const std::string& mps_file) {
94 if (!IsLinearProgram(linear_program)) {
95 return absl::InvalidArgumentError(
96 "'linear_program' has a quadratic objective");
97 }
98 ASSIGN_OR_RETURN(MPModelProto proto, QpToMpModelProto(linear_program));
99 ASSIGN_OR_RETURN(std::string mps_export, ExportModelAsMpsFormat(proto));
100 File* file;
101 RETURN_IF_ERROR(file::Open(mps_file, "w", &file, file::Defaults()));
102 auto status = file::WriteString(file, mps_export, file::Defaults());
103 status.Update(file->Close(file::Defaults()));
104 return status;
105}
