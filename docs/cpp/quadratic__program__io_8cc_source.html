52QuadraticProgram ReadQuadraticProgramOrDie(const std::string& filename,
53 bool include_names) {
54 if (absl::EndsWith(filename, ".mps") || absl::EndsWith(filename, ".mps.gz") ||
55 absl::EndsWith(filename, ".mps.bz2")) {
56 return ReadMpsLinearProgramOrDie(filename, include_names);
57 }
58 if (absl::EndsWith(filename, ".pb") ||
59 absl::EndsWith(filename, ".textproto") ||
60 absl::EndsWith(filename, ".json") ||
61 absl::EndsWith(filename, ".json.gz")) {
62 return ReadMPModelProtoFileOrDie(filename, include_names);
63 }
64 LOG(QFATAL) << "Invalid filename suffix in " << filename
65 << ". Valid suffixes are .mps, .mps.gz, .pb, .textproto,"
66 << ".json, and .json.gz";
67}

69QuadraticProgram ReadMPModelProtoFileOrDie(

71 MPModelProto lp_proto;

79absl::Status WriteLinearProgramToMps(const QuadraticProgram& linear_program,
80 const std::string& mps_file) {
81 if (!IsLinearProgram(linear_program)) {
82 return absl::InvalidArgumentError(
83 "'linear_program' has a quadratic objective");
84 }
85 ASSIGN_OR_RETURN(MPModelProto proto, QpToMpModelProto(linear_program));
86 ASSIGN_OR_RETURN(std::string mps_export, ExportModelAsMpsFormat(proto));
87 File* file;
88 RETURN_IF_ERROR(file::Open(mps_file, "w", &file, file::Defaults()));
89 auto status = file::WriteString(file, mps_export, file::Defaults());
90 status.Update(file->Close(file::Defaults()));
91 return status;
92}

104// Class implementing the
105// `ortools/lp_data/mps_reader_template.h` interface that only
106// stores the names of rows and columns, and the number of non-zeros found.
107class MpsReaderDimensionAndNames {
108 public:
109 using IndexType = int64_t;
110 void SetUp() {
111 read_or_parse_failed_ = false;
112 col_name_to_index_.clear();
113 row_name_to_index_.clear();
114 added_non_zeros_ = 0;
115 }
116 void CleanUp() {}
117 double ConstraintLowerBound(IndexType index) { return 0; }
118 double ConstraintUpperBound(IndexType index) { return 0; }
119 IndexType FindOrCreateConstraint(absl::string_view row_name) {
120 const auto [it, success_unused] = row_name_to_index_.try_emplace(
121 row_name, static_cast<IndexType>(row_name_to_index_.size()));
122 return it->second;
123 }
124 IndexType FindOrCreateVariable(absl::string_view col_name) {
125 const auto [it, success_unused] = col_name_to_index_.try_emplace(
126 col_name, static_cast<IndexType>(col_name_to_index_.size()));
127 return it->second;
128 }
129 void SetConstraintBounds(IndexType row_index, double lower_bound,
130 double upper_bound) {}
131 void SetConstraintCoefficient(IndexType row_index, IndexType col_index,
132 double coefficient) {
133 ++added_non_zeros_;
134 }
135 void SetIsLazy(IndexType row_index) {}
136 void SetName(absl::string_view problem_name) {}
137 void SetObjectiveCoefficient(IndexType index, double coefficient) {}
138 void SetObjectiveDirection(bool maximize) {}
139 void SetObjectiveOffset(double offset) {}
140 void SetVariableTypeToInteger(IndexType index) {}
141 void SetVariableTypeToSemiContinuous(IndexType index) {
142 LOG_FIRST_N(WARNING, 1)
143 << "Semi-continuous variables not supported, failed to parse file";
144 read_or_parse_failed_ = true;
145 }
146 void SetVariableBounds(IndexType index, double lower_bound,
147 double upper_bound) {}
148 double VariableLowerBound(IndexType index) { return 0; }
149 double VariableUpperBound(IndexType index) { return 0; }
150 absl::Status CreateIndicatorConstraint(absl::string_view row_name,
151 IndexType col_index, bool var_value) {
152 absl::string_view message =
153 "Indicator constraints not supported, failed to parse file";
154 LOG_FIRST_N(WARNING, 1) << message;
155 return absl::InvalidArgumentError(message);
156 }
157
158 // Lookup constant functions. They assume that `row_name` or `col_name`,
159 // respectively, have been previously added to the internal set of rows or
160 // variables. The functions QFAIL if this is not true.
161 IndexType FindVariableOrDie(absl::string_view col_name) const {
162 const auto it = col_name_to_index_.find(col_name);
163 if (it == col_name_to_index_.end()) {
164 LOG(QFATAL) << absl::StrFormat(
165 "column `%s` not previously found in file, internal error?",
166 col_name);
167 }
168 return it->second;
169 }
170 IndexType FindConstraintOrDie(absl::string_view row_name) const {
171 const auto it = row_name_to_index_.find(row_name);
172 if (it == row_name_to_index_.end()) {
173 LOG(QFATAL) << absl::StrFormat(
174 "row `%s` not previously found in file, internal error?", row_name);
175 }
176 return it->second;
177 }
178
179 // Number of non-zeros added so-far.
180 int64_t AddedNonZeros() const { return added_non_zeros_; }
181
182 // Number of variables added so-far.
183 int64_t NumVariables() const { return col_name_to_index_.size(); }
184
185 // Number of constraints added so-far.
186 int64_t NumConstraints() const { return row_name_to_index_.size(); }
187
188 // Return `true` if an unsupported MPS section was found.
189 bool FailedToParse() const { return read_or_parse_failed_; }
190
191 // Return a const-hash map of col names to indices.
192 const absl::flat_hash_map<std::string, IndexType>& ColNameIndexMap() const {
193 return col_name_to_index_;
194 }
195
196 // Return a const-hash map of row names to indices.
197 const absl::flat_hash_map<std::string, IndexType>& RowNameIndexMap() const {
198 return row_name_to_index_;
199 }
200
201 private:
202 bool read_or_parse_failed_ = false;
203 absl::flat_hash_map<std::string, IndexType> col_name_to_index_;
204 absl::flat_hash_map<std::string, IndexType> row_name_to_index_;
205 int64_t added_non_zeros_ = 0;
206};

208// Class implementing the
209// `ortools/lp_data/mps_reader_template.h` interface.
210// It is intended to be used in conjunction with `MpsReaderDimensionAndNames`
211// as follows:
212//
213// // Retrieve dimension and name information from file.
214// MpsReaderDimensionAndNames dimension_and_name;
215// MPSReaderTemplate<MpsReaderDimensionAndNames> dimension_parser;
216// dimension_parser.ParseFile(file_name, &dimension_and_name);
217// // Store QP problem coefficients.
218// MpsReaderQpDataWrapper qp_wrapper(&dimension_and_name, include_names);
219// MPSReaderTemplate<MpsReaderQpDataWrapper> qp_parser;
220// qp_parser.ParseFile(file_name, &qp_wrapper);
221// // Retrieve fully assembled QP.
222// QuadraticProgram result = qp_wrapper.GetAndClearQuadraticProgram();
223class MpsReaderQpDataWrapper {
224 public:
225 using IndexType = int64_t;
226 // Constructor, `*dimension_and_names` must outlive the class, and be
227 // constant during the object's lifetime. If `include_names`=true, the
228 // resulting QuadraticProgram from `GetAndClearQuadraticProgram()` will
229 // include name information from the MPS file.
230 // NOTE: The code assumes that the file to be read is the same file already
231 // read using the `*dimension_and_names` argument.
232 MpsReaderQpDataWrapper(const MpsReaderDimensionAndNames* dimension_and_names,
233 bool include_names)
234 : include_names_{include_names},
235 dimension_and_names_{*dimension_and_names} {}
236
237 // Implements the `MPSReaderTemplate` interface.
238 void SetUp() {
239 const int64_t num_variables = dimension_and_names_.NumVariables();
240 const int64_t num_constraints = dimension_and_names_.NumConstraints();
241 triplets_.reserve(dimension_and_names_.AddedNonZeros());
242 quadratic_program_ = QuadraticProgram(/*num_variables=*/num_variables,
243 /*num_constraints=*/num_constraints);
244 // Default variables in MPS files have a zero lower bound, an infinity
245 // upper bound, and a zero objective; while default constraints are
246 // 'equal to zero' constraints.
247 quadratic_program_.constraint_lower_bounds =
248 Eigen::VectorXd::Zero(num_constraints);
249 quadratic_program_.constraint_upper_bounds =
250 Eigen::VectorXd::Zero(num_constraints);
251 quadratic_program_.variable_lower_bounds =
252 Eigen::VectorXd::Zero(num_variables);
253 }
254 void CleanUp() {
255 SetEigenMatrixFromTriplets(std::move(triplets_),
256 quadratic_program_.constraint_matrix);
257 // Deal with maximization problems.
258 if (quadratic_program_.objective_scaling_factor == -1) {
259 quadratic_program_.objective_offset *= -1;
260 for (double& objective_coefficient :
261 quadratic_program_.objective_vector) {
262 objective_coefficient *= -1;
263 }
264 }
265 if (include_names_) {
266 quadratic_program_.variable_names =
267 std::vector<std::string>(dimension_and_names_.NumVariables());
268 quadratic_program_.constraint_names =
269 std::vector<std::string>(dimension_and_names_.NumConstraints());
270 for (auto [name, index] : dimension_and_names_.ColNameIndexMap()) {
271 (*quadratic_program_.variable_names)[index] = name;
272 }
273 for (auto [name, index] : dimension_and_names_.RowNameIndexMap()) {
274 (*quadratic_program_.constraint_names)[index] = name;
275 }
276 }
277 }
278 double ConstraintLowerBound(IndexType index) {
279 return quadratic_program_.constraint_lower_bounds[index];
280 }
281 double ConstraintUpperBound(IndexType index) {
282 return quadratic_program_.constraint_upper_bounds[index];
283 }
284 IndexType FindOrCreateConstraint(absl::string_view row_name) {
285 return dimension_and_names_.FindConstraintOrDie(row_name);
286 }
287 IndexType FindOrCreateVariable(absl::string_view col_name) {
288 return dimension_and_names_.FindVariableOrDie(col_name);
289 }
290 void SetConstraintBounds(IndexType index, double lower_bound,
291 double upper_bound) {
292 quadratic_program_.constraint_lower_bounds[index] = lower_bound;
293 quadratic_program_.constraint_upper_bounds[index] = upper_bound;
294 }
295 void SetConstraintCoefficient(IndexType row_index, IndexType col_index,
296 double coefficient) {
297 DCHECK_LT(triplets_.size(), triplets_.capacity());
298 triplets_.emplace_back(row_index, col_index, coefficient);
299 }
300 void SetIsLazy(IndexType row_index) {
301 LOG_FIRST_N(WARNING, 1) << "Lazy constraint information lost, treated as "
302 "regular constraint instead";
303 }
304 void SetName(absl::string_view problem_name) {
305 if (include_names_) {
306 quadratic_program_.problem_name = std::string(problem_name);
307 }
308 }
309 void SetObjectiveCoefficient(IndexType index, double coefficient) {
310 quadratic_program_.objective_vector[index] = coefficient;
311 }
312 void SetObjectiveDirection(bool maximize) {
313 quadratic_program_.objective_scaling_factor = (maximize ? -1 : 1);
314 }
315 void SetObjectiveOffset(double offset) {
316 quadratic_program_.objective_offset = offset;
317 }
318 void SetVariableTypeToInteger(IndexType index) {
319 LOG_FIRST_N(WARNING, 1) << "Dropping integrality requirements, all "
320 "variables treated as continuous";
321 }
322 void SetVariableTypeToSemiContinuous(IndexType index) {
323 // This line should never execute, as the code must fail on
324 // `MpsReaderDimensionAndNames::SetVariableTypeToSemiContinuous` before.
325 LOG(QFATAL) << "Semi-continuous variables are unsupported, and should have "
326 "been detected before (in MpsReaderDimensionAndNames)";
327 }
328 void SetVariableBounds(IndexType index, double lower_bound,
329 double upper_bound) {
330 quadratic_program_.variable_lower_bounds[index] = lower_bound;
331 quadratic_program_.variable_upper_bounds[index] = upper_bound;
332 }
333 double VariableLowerBound(IndexType index) {
334 return quadratic_program_.variable_lower_bounds[index];
335 }
336 double VariableUpperBound(IndexType index) {
337 return quadratic_program_.variable_upper_bounds[index];
338 }
339 absl::Status CreateIndicatorConstraint(absl::string_view row_name,
340 IndexType col_index, bool var_value) {
341 // This function should never be called, as the code must fail on
342 // `MpsReaderDimensionAndNames::CreateIndicatorConstraint` before.
343 LOG(QFATAL) << "Indicator constraints are unsupported, and should have "
344 "been detected before (in MpsReaderDimensionAndNames)";
345 }
346
347 // Returns a `QuadraticProgram` holding all information read by the
348 // `mps_reader_template.h` interface. It leaves the internal quadratic
349 // program in an indeterminate state.
350 QuadraticProgram&& GetAndClearQuadraticProgram() {
351 return std::move(quadratic_program_);
352 }
353
354 private:
355 bool include_names_;
356 QuadraticProgram quadratic_program_;
357 const MpsReaderDimensionAndNames& dimension_and_names_;
358 std::vector<Eigen::Triplet<double, int64_t>> triplets_;
359};

363absl::StatusOr<QuadraticProgram> ReadMpsLinearProgram(
364 const std::string& lp_file, bool include_names) {
365 MpsReaderDimensionAndNames dimension_and_names;
366 // Collect MPS format, sizes and names.
367 MPSReaderTemplate<MpsReaderDimensionAndNames> pass_one_reader;
368 const absl::StatusOr<MPSReaderFormat> pass_one_format =
369 pass_one_reader.ParseFile(lp_file, &dimension_and_names,
370 MPSReaderFormat::kAutoDetect);
371 if (!pass_one_format.ok()) {
372 return util::StatusBuilder(pass_one_format.status()) << absl::StrFormat(
373 "Could not read or parse file `%s` as an MPS file", lp_file);
374 }
375 if (dimension_and_names.FailedToParse()) {
376 return absl::InvalidArgumentError(absl::StrFormat(
377 "Could not read or parse file `%s` as an MPS file, or unsupported "
378 "features/sections found",
379 lp_file));
380 }
381 DCHECK(*pass_one_format == MPSReaderFormat::kFixed ||
382 *pass_one_format == MPSReaderFormat::kFree);
383 // Populate the QP with pre-allocated sizes.
384 MpsReaderQpDataWrapper qp_data_wrapper(&dimension_and_names, include_names);
385 MPSReaderTemplate<MpsReaderQpDataWrapper> pass_two_reader;
386 const absl::StatusOr<MPSReaderFormat> pass_two_format =
387 pass_two_reader.ParseFile(lp_file, &qp_data_wrapper, *pass_one_format);
388 if (!pass_two_format.ok()) {
389 return util::StatusBuilder(pass_two_format.status()) << absl::StrFormat(
390 "Could not read or parse file `%s` as an MPS file "
391 "(maybe file changed between reads?)",
392 lp_file);
393 }
394 DCHECK(*pass_one_format == *pass_two_format);
395 return qp_data_wrapper.GetAndClearQuadraticProgram();
396}

398QuadraticProgram ReadMpsLinearProgramOrDie(const std::string& lp_file,
399 bool include_names) {

402 if (!result.ok()) {

405 }
406 return *result;
407}
