23// Adjustable k-ary heap for `Aggregate` classes containing a priority and an
24// index referring to an array where the relevant data is stored.
25//
26// Because the class uses indices and vectors, it is much faster than
27// AdjustablePriorityQueue, even in the binary heap case.
28//
29// k-ary heaps are useful when SiftDown() (aka Decrease) is called more often
30// than Pop() (aka Extract).
31//
32// Namely, Pop() has a complexity in O(k * log_k (n)), while SiftDown() is in
33// O(log_k(n)), even when k = 2. This explains the small gain.
34//
35// The `Aggregate` class must provide the following features. It's better
36// described with an example:
37//
38// class PriorityAggregate {
39// public:
40// using Index = int;
41// using Priority = float;
42// PriorityAggregate() = default
43// PriorityAggregate(Priority priority, Index index)
44// : index_(index), priority_(priority) {}
45// Priority priority() const { return priority_; }
46// Index index() const { return index_; }
47// bool operator<(const PriorityAggregate other) const {
48// if (other.priority() != priority()) {
49// return priority() < other.priority();
50// }
51// return index() < other.index();
52// }
53//
54// private:
55// Index index_;
56// Priority priority_;
57// };
58
59// Everything in the class above is mandatory, except the implementation
60// of operator< which contains here a way to break ties between two Aggregates.
61// It's important to expose the Index and Priority types.
62// The accessors index() and priority must also be provided, as they are used
63// by AdjustableKAryHeap.
64
65template <typename Aggregate, int Arity, bool IsMaxHeap>
66class AdjustableKAryHeap {
67 public:
68 using Priority = typename Aggregate::Priority;
69 using Index = typename Aggregate::Index;
70 using HeapIndex = Index;
71 static_assert(Arity >= 2, "arity must be at least 2");
72 static_assert(std::numeric_limits<Index>::is_integer,
73 "Index must be an integer");
74 static_assert(std::numeric_limits<Priority>::is_specialized,
75 "Priority must be an integer or floating-point type");
76 AdjustableKAryHeap() { Clear(); }
77
78 // Construct an n-heap from an existing vector, tracking original indices.
79 // `universe_size` is the maximum possible index in `elements`.
80 explicit AdjustableKAryHeap(const std::vector<Aggregate>& elements,
81 HeapIndex universe_size) {
82 Load(elements, universe_size);
83 }
84
85 void Clear() {
86 data_.clear();
87 heap_positions_.clear();
88 heap_size_ = 0;
89 }
90
91 void Load(const std::vector<Aggregate>& elements, HeapIndex universe_size) {
92 static_assert(Arity >= 2, "Arity must be at least 2");
93 data_.resize(elements.size());
94 heap_size_ = elements.size();
95 std::copy(elements.begin(), elements.end(), data_.begin());
96 heap_positions_.resize(universe_size, kNonExistent);
97 for (HeapIndex i = 0; i < data_.size(); ++i) {
98 heap_positions_[index(i)] = i;
99 }
100 BuildHeap();
101 }
102
103 // Returns the top element from the heap (smallest for min-heap, largest for
104 // max-heap), removes it, and rearranges the heap.
105 // This will CHECK-fail if the heap is empty (through Top()).
106 Aggregate Pop() {
107 const Aggregate top_element = Top();
108 CHECK(RemoveAtHeapPosition(0));
109 return top_element;
110 }
111
112 // Returns the top element, without modifying the heap.
113 // This will CHECK-fail if the heap is empty.
114 Aggregate Top() const {
115 CHECK(!IsEmpty());
116 return data_[0];
117 }
118
119 // Returns the number of elements in the heap.
120 HeapIndex heap_size() const { return heap_size_; }
121
122 // True iff the heap is empty.
123 bool IsEmpty() const { return heap_size() == 0; }
124
125 // Insert an element into the heap.
126 void Insert(Aggregate element) {
127 if (index(element) >= heap_positions_.size()) {
128 heap_positions_.resize(index(element) + 1, kNonExistent);
129 }
130 if (GetHeapPosition(index(element)) == kNonExistent) {
131 heap_positions_[index(element)] = data_.size();
132 data_.push_back(element);
133 ++heap_size_;
134 }
135 Update(element);
136 }
137
138 // Removes the element at index.
139 bool Remove(Index index) {
140 if (IsEmpty()) return false;
141 const HeapIndex heap_position = GetHeapPosition(index);
142 return heap_position != kNonExistent ? RemoveAtHeapPosition(heap_position)
143 : false;
144 }
145
146 // Change the value of an element.
147 void Update(Aggregate element) {
148 DCHECK(!IsEmpty());
149 const HeapIndex heap_position = GetHeapPosition(index(element));
150 DCHECK_GE(heap_position, 0);
151 DCHECK_LT(heap_position, heap_positions_.size());
152 data_[heap_position] = element;
153 if (HasPriority(heap_position, Parent(heap_position))) {
154 SiftUp(heap_position);
155 } else {
156 SiftDown(heap_position);
157 }
158 }
159
160 // Checks that the heap is well-formed.
161 bool CheckHeapProperty() const {
162 for (HeapIndex i = heap_size() - 1; i >= Arity; --i) {
163 CHECK(HasPriority(Parent(i), i))
164 << "Parent " << Parent(i) << " with priority " << priority(Parent(i))
165 << " does not have priority over " << i << " with priority "
166 << priority(i) << " , heap_size = " << heap_size()
167 << ", priority difference = " << priority(i) - priority(Parent(i));
168 }
169 CHECK_LE(heap_size(), heap_positions_.size());
170 CHECK_LE(heap_size(), data_.size());
171 return true;
172 }
173
174 private:
175 // Gets the current position of element with index i in the heap.
176 HeapIndex GetHeapPosition(Index i) const {
177 DCHECK_GE(i, 0);
178 DCHECK_LT(i, heap_positions_.size());
179 return heap_positions_[i];
180 }
181
182 // Removes an element at a given heap position.
183 bool RemoveAtHeapPosition(HeapIndex heap_index) {
184 DCHECK(!IsEmpty());
185 DCHECK_GE(heap_index, 0);
186 if (heap_index >= heap_size()) return false;
187 PerformSwap(heap_index, heap_size() - 1);
188 --heap_size_;
189 if (HasPriority(heap_index, Parent(heap_index))) {
190 SiftUp(heap_index);
191 } else {
192 SiftDown(heap_index);
193 }
194 return true;
195 }
196
197 // Maintains heap property by sifting down starting from the end,
198 void BuildHeap() {
199 for (HeapIndex i = Parent(heap_size()); i >= 0; --i) {
200 SiftDown(i);
201 }
202 DCHECK(CheckHeapProperty());
203 }
204
205 // Maintains heap property by sifting up an element.
206 void SiftUp(HeapIndex index) {
207 while (index > 0 && HasPriority(index, Parent(index))) {
208 PerformSwap(index, Parent(index));
209 index = Parent(index);
210 }
211 }
212
213 // Maintains heap property by sifting down an element.
214 void SiftDown(HeapIndex index) {
215 while (true) {
216 const HeapIndex highest_priority_child = GetHighestPriorityChild(index);
217 if (highest_priority_child == index) return;
218 PerformSwap(index, highest_priority_child);
219 index = highest_priority_child;
220 }
221 }
222
223 // Finds the child with the highest priority, i.e. the child with the
224 // smallest (resp. largest) key for a min- (resp. max-) heap.
225 // Returns index is there are no such children.
226 HeapIndex GetHighestPriorityChild(HeapIndex index) const {
227 const HeapIndex right_bound = std::min(RightChild(index) + 1, heap_size());
228 HeapIndex highest_priority_child = index;
229 for (HeapIndex i = LeftChild(index); i < right_bound; ++i) {
230 if (HasPriority(i, highest_priority_child)) {
231 highest_priority_child = i;
232 }
233 }
234 return highest_priority_child;
235 }
236
237 // Swaps two elements of data_, while also making sure heap_positions_ is
238 // properly maintained.
239 void PerformSwap(HeapIndex i, HeapIndex j) {
240 std::swap(data_[i], data_[j]);
241 std::swap(heap_positions_[index(i)], heap_positions_[index(j)]);
242 }
243
244 // Compares two entries by priority, then by value of the related integer
245 // value.
246 bool LessThan(HeapIndex i, HeapIndex j) const {
247 // The order "j then i" matters to minimize the number of comparison
248 // instructions generated.
249 if (priority(j) != priority(i)) {
250 // Again the order "i, then j" matters.
251 return priority(i) < priority(j);
252 }
253 return index(i) < index(j);
254 }
255
256 // Compares two elements based on whether we are dealing with a min- or a
257 // max-heap. Returns true if (data indexed by) i has more priority
258 // than j. Note that we only use operator::<.
259 bool HasPriority(HeapIndex i, HeapIndex j) const {
260 return IsMaxHeap ? data_[j] < data_[i] : data_[i] < data_[j];
261 }
262
263 const Index kNonExistent = -1;
264
265 // Since Arity is a (small) constant, we expect compilers to avoid
266 // multiplication instructions and use LEA instructions or a combination
267 // of shifts and arithmetic operations.
268 // Powers of 2 are guaranteed to be quick thanks to simple shifts.
269
270 // Gets the leftmost child index of a given node
271 HeapIndex LeftChild(HeapIndex index) const { return Arity * index + 1; }
272
273 // Gets the rightmost child index of a given node
274 HeapIndex RightChild(HeapIndex index) const { return Arity * (index + 1); }
275
276 // For division, the optimization is more uncertain, although a simple
277 // multiplication and a shift might be used by the compiler.
278 // Of course, powers of 2 are guaranteed to be quick thanks to simple shifts.
279
280 // Gets the parent index of a given index.
281 HeapIndex Parent(HeapIndex index) const { return (index - 1) / Arity; }
282
283 // Returns the index related to the element.
284 Index index(const Aggregate& element) const { return element.index(); }
285
286 // Returns the index of the element at position i in the heap.
287 Index index(HeapIndex i) const { return data_[i].index(); }
288
289 // Returns the index of the element at position i in the heap.
290 Priority priority(HeapIndex i) const { return data_[i].priority(); }
291
292 // The heap is stored as a vector.
293 std::vector<Aggregate> data_;
294
295 // Maps original heap_index to current heap position.
296 std::vector<Index> heap_positions_;
297
298 // The number of elements currently in the heap. This may be updated
299 // either when removing an element (which is not removed from data_), or
300 // adding a new one.
301 HeapIndex heap_size_ = 0;
302};
