45absl::Status VariablesValid(const VariablesProto& variables,
46 const bool check_names) {
47 RETURN_IF_ERROR(CheckIdsRangeAndStrictlyIncreasing(variables.ids()))
48 << "Bad variable ids";
49 RETURN_IF_ERROR(
50 CheckValues(MakeView(variables.ids(), variables.lower_bounds()),
51 {.allow_positive_infinity = false}, "lower_bounds"));
52 RETURN_IF_ERROR(
53 CheckValues(MakeView(variables.ids(), variables.upper_bounds()),
54 {.allow_negative_infinity = false}, "upper_bounds"));
55 RETURN_IF_ERROR(
56 CheckValues(MakeView(variables.ids(), variables.integers()), "integers"));
57 RETURN_IF_ERROR(CheckNameVector(MakeView(variables.ids(), variables.names()),
58 check_names));
59 return absl::OkStatus();
60}

62absl::Status VariableUpdatesValid(
63 const VariableUpdatesProto& variable_updates) {

90absl::Status ObjectiveValid(const ObjectiveProto& objective,
91 absl::Span<const int64_t> variable_ids) {
92 // 1. Validate offset
93 RETURN_IF_ERROR(CheckScalarNoNanNoInf(objective.offset()))
94 << "Objective offset invalid";
95 // 2. Validate linear terms
96 const auto linear_coefficients = MakeView(objective.linear_coefficients());
97 RETURN_IF_ERROR(CheckIdsAndValues(
98 linear_coefficients,
99 {.allow_positive_infinity = false, .allow_negative_infinity = false}))
100 << "Linear objective coefficients bad";
101 RETURN_IF_ERROR(CheckSortedIdsSubset(linear_coefficients.ids(), variable_ids))
102 << "Objective.linear_coefficients.ids not found in Variables.ids";
103 // 3. Validate quadratic terms
104 RETURN_IF_ERROR(SparseMatrixValid(objective.quadratic_coefficients(),
105 /*enforce_upper_triangular=*/true))
106 << "Objective.quadratic_coefficients invalid";
107 RETURN_IF_ERROR(SparseMatrixIdsAreKnown(objective.quadratic_coefficients(),
108 variable_ids, variable_ids))
109 << "Objective.quadratic_coefficients invalid";
110 return absl::OkStatus();
111}

113// NOTE: This method does not check requirements on the IDs
114absl::Status ObjectiveUpdatesValid(
115 const ObjectiveUpdatesProto& objective_updates) {
116 // 1. Validate offset
117 RETURN_IF_ERROR(CheckScalarNoNanNoInf(objective_updates.offset_update()))
118 << "Offset update invalid";
119 // 2. Validate linear terms
120 RETURN_IF_ERROR(CheckIdsAndValues(
121 MakeView(objective_updates.linear_coefficients()),
122 {.allow_positive_infinity = false, .allow_negative_infinity = false}))
123 << "Linear objective coefficients bad";
124 // 3. Validate quadratic terms
125 RETURN_IF_ERROR(SparseMatrixValid(objective_updates.quadratic_coefficients(),
126 /*enforce_upper_triangular=*/true))
127 << "Objective.quadratic_coefficients invalid";
128 return absl::OkStatus();
129}

131absl::Status ObjectiveUpdatesValidForModel(
132 const ObjectiveUpdatesProto& objective_updates,
133 const IdUpdateValidator& id_validator) {
134 RETURN_IF_ERROR(id_validator.CheckSortedIdsSubsetOfFinal(
135 objective_updates.linear_coefficients().ids()))
136 << "Linear coefficients ids not found in variable ids";
137 RETURN_IF_ERROR(id_validator.CheckSortedIdsSubsetOfFinal(
138 objective_updates.quadratic_coefficients().row_ids()))
139 << "Quadratic coefficient ids bad";
140 RETURN_IF_ERROR(id_validator.CheckIdsSubsetOfFinal(
141 objective_updates.quadratic_coefficients().column_ids()))
142 << "Quadratic coefficient ids bad";
143 return absl::OkStatus();
144}

206absl::Status ValidateModel(const ModelProto& model, const bool check_names) {
207 RETURN_IF_ERROR(VariablesValid(model.variables(), check_names))
208 << "Model.variables are invalid.";
209 RETURN_IF_ERROR(ObjectiveValid(model.objective(), model.variables().ids()))
210 << "Model.objective is invalid";
211 RETURN_IF_ERROR(
212 LinearConstraintsValid(model.linear_constraints(), check_names))
213 << "Model.linear_constraints are invalid";
214 RETURN_IF_ERROR(SparseMatrixValid(model.linear_constraint_matrix()))
215 << "Model.linear_constraint_matrix invalid";
216 RETURN_IF_ERROR(SparseMatrixIdsAreKnown(model.linear_constraint_matrix(),
217 model.linear_constraints().ids(),
218 model.variables().ids()))
219 << "Model.linear_constraint_matrix ids are inconsistent";
220 return absl::OkStatus();
221}

227absl::Status ValidateModelUpdate(const ModelUpdateProto& model_update,
228 const bool check_names) {
229 RETURN_IF_ERROR(CheckIdsRangeAndStrictlyIncreasing(
230 model_update.deleted_linear_constraint_ids()))
231 << "ModelUpdateProto.deleted_linear_constraint_ids invalid";
232 RETURN_IF_ERROR(
233 CheckIdsRangeAndStrictlyIncreasing(model_update.deleted_variable_ids()))
234 << "ModelUpdateProto.deleted_variable_ids invalid";
235 RETURN_IF_ERROR(VariableUpdatesValid(model_update.variable_updates()))
236 << "ModelUpdateProto.variable_updates invalid";
237 RETURN_IF_ERROR(
238 LinearConstraintUpdatesValid(model_update.linear_constraint_updates()))
239 << "ModelUpdateProto.linear_constraint_updates invalid";
240 RETURN_IF_ERROR(VariablesValid(model_update.new_variables(), check_names))
241 << "ModelUpdateProto.new_variables invalid";
242 RETURN_IF_ERROR(LinearConstraintsValid(model_update.new_linear_constraints(),
243 check_names))
244 << "ModelUpdateProto.new_linear_constraints invalid";
245 RETURN_IF_ERROR(ObjectiveUpdatesValid(model_update.objective_updates()))
246 << "ModelUpdateProto.objective_update invalid";
247 RETURN_IF_ERROR(
248 SparseMatrixValid(model_update.linear_constraint_matrix_updates()))
249 << "Model.linear_constraint_matrix_updates invalid";
250 return absl::OkStatus();
251}

253absl::Status ValidateModelUpdateAndSummary(const ModelUpdateProto& model_update,
254 const ModelSummary& model_summary,
255 const bool check_names) {
256 RETURN_IF_ERROR(ValidateModelUpdate(model_update));
257 const IdUpdateValidator variable_id_validator(
258 model_summary.variables, model_update.deleted_variable_ids(),
259 model_update.new_variables().ids());
260 RETURN_IF_ERROR(variable_id_validator.IsValid())
261 << "Invalid new or deleted variable id";
262 const IdUpdateValidator linear_constraint_id_validator(
263 model_summary.linear_constraints,
264 model_update.deleted_linear_constraint_ids(),
265 model_update.new_linear_constraints().ids());
266 RETURN_IF_ERROR(linear_constraint_id_validator.IsValid())
267 << "Invalid new or deleted linear constraint id";
268
269 RETURN_IF_ERROR(VariableUpdatesValidForState(model_update.variable_updates(),
270 variable_id_validator))
271 << "Invalid variable update";
272
273 RETURN_IF_ERROR(LinearConstraintUpdatesValidForState(
274 model_update.linear_constraint_updates(), linear_constraint_id_validator))
275 << "Invalid linear constraint update";
276
277 RETURN_IF_ERROR(ObjectiveUpdatesValidForModel(
278 model_update.objective_updates(), variable_id_validator))
279 << "Invalid objective update";
280 RETURN_IF_ERROR(LinearConstraintMatrixIdsValidForUpdate(
281 model_update.linear_constraint_matrix_updates(),
282 linear_constraint_id_validator, variable_id_validator))
283 << "Invalid linear constraint matrix update";
284 if (check_names && !model_update.new_variables().names().empty()) {
285 RETURN_IF_ERROR(
286 CheckNewNames(model_summary.variables,
287 MakeView(model_update.new_variables().ids(),
288 model_update.new_variables().names())))
289 << "Bad new variable names";
290 }
291
292 if (check_names && !model_update.new_linear_constraints().names().empty()) {
293 RETURN_IF_ERROR(
294 CheckNewNames(model_summary.linear_constraints,
295 MakeView(model_update.new_linear_constraints().ids(),
296 model_update.new_linear_constraints().names())))
297 << "Bad new linear constraint names";
298 }
299 return absl::OkStatus();
300}
