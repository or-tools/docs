39std::optional<absl::string_view> Enum<SolutionStatus>::ToOptString(
40 SolutionStatus value) {
41 switch (value) {
42 case SolutionStatus::kFeasible:
43 return "feasible";
44 case SolutionStatus::kInfeasible:
45 return "infeasible";
46 case SolutionStatus::kUndetermined:
47 return "undetermined";
48 }
49 return std::nullopt;
50}

61absl::StatusOr<PrimalSolution> PrimalSolution::FromProto(
62 const ModelStorage* model,
63 const PrimalSolutionProto& primal_solution_proto) {
64 PrimalSolution primal_solution;
65 OR_ASSIGN_OR_RETURN3(
66 primal_solution.variable_values,
67 VariableValuesFromProto(model, primal_solution_proto.variable_values()),
68 _ << "invalid variable_values");
69 primal_solution.objective_value = primal_solution_proto.objective_value();
70 OR_ASSIGN_OR_RETURN3(
71 primal_solution.auxiliary_objective_values,
72 AuxiliaryObjectiveValuesFromProto(
73 model, primal_solution_proto.auxiliary_objective_values()),
74 _ << "invalid auxiliary_objective_values");
75 const std::optional<SolutionStatus> feasibility_status =
76 EnumFromProto(primal_solution_proto.feasibility_status());
77 if (!feasibility_status.has_value()) {
78 return absl::InvalidArgumentError("feasibility_status must be specified");
79 }
80 primal_solution.feasibility_status = *feasibility_status;
81 return primal_solution;
82}

94double PrimalSolution::get_objective_value(const Objective objective) const {
95 if (!variable_values.empty()) {
96 // Here we assume all keys are in the same storage. As PrimalSolution is not
97 // properly encapsulated, we can't maintain a ModelStorage pointer and
98 // iterating on all keys would have a too high cost.
99 CHECK_EQ(variable_values.begin()->first.storage(), objective.storage());
100 }
101 if (!objective.is_primary()) {
102 CHECK(auxiliary_objective_values.contains(objective));
103 return auxiliary_objective_values.at(objective);
104 }
105 return objective_value;
106}

179absl::StatusOr<Basis> Basis::FromProto(const ModelStorage* model,
180 const BasisProto& basis_proto) {
181 Basis basis;

183 basis.constraint_status,
184 LinearConstraintBasisFromProto(model, basis_proto.constraint_status()),
185 _ << "invalid constraint_status");
186 OR_ASSIGN_OR_RETURN3(
187 basis.variable_status,
188 VariableBasisFromProto(model, basis_proto.variable_status()),
189 _ << "invalid variable_status");
190 const std::optional<SolutionStatus> basic_dual_feasibility =
191 EnumFromProto(basis_proto.basic_dual_feasibility());
192 if (!basic_dual_feasibility.has_value()) {
193 return absl::InvalidArgumentError(
194 "basic_dual_feasibility for a basis must be specified");
195 }
196 basis.basic_dual_feasibility = *basic_dual_feasibility;
197 return basis;
198}

200absl::Status Basis::CheckModelStorage(
201 const ModelStorage* const expected_storage) const {
202 for (const auto& [v, _] : variable_status) {
203 RETURN_IF_ERROR(
204 internal::CheckModelStorage(/*storage=*/v.storage(),
205 /*expected_storage=*/expected_storage))
206 << "invalid variable " << v << " in variable_status";
207 }
208 for (const auto& [c, _] : constraint_status) {
209 RETURN_IF_ERROR(
210 internal::CheckModelStorage(/*storage=*/c.storage(),
211 /*expected_storage=*/expected_storage))
212 << "invalid constraint " << c << " in constraint_status";
213 }
214 return absl::OkStatus();
215}

226absl::StatusOr<Solution> Solution::FromProto(
227 const ModelStorage* model, const SolutionProto& solution_proto) {
228 Solution solution;
229 if (solution_proto.has_primal_solution()) {
230 OR_ASSIGN_OR_RETURN3(
231 solution.primal_solution,
232 PrimalSolution::FromProto(model, solution_proto.primal_solution()),
233 _ << "invalid primal_solution");
234 }
235 if (solution_proto.has_dual_solution()) {
236 OR_ASSIGN_OR_RETURN3(
237 solution.dual_solution,
238 DualSolution::FromProto(model, solution_proto.dual_solution()),
239 _ << "invalid dual_solution");
240 }
241 if (solution_proto.has_basis()) {
242 OR_ASSIGN_OR_RETURN3(solution.basis,
243 Basis::FromProto(model, solution_proto.basis()),
244 _ << "invalid basis");
245 }
246 return solution;
247}
