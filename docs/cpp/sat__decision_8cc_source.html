42void SatDecisionPolicy::IncreaseNumVariables(int num_variables) {
43 const int old_num_variables = activities_.size();
44 DCHECK_GE(num_variables, activities_.size());
45
46 activities_.resize(num_variables, parameters_.initial_variables_activity());
47 tie_breakers_.resize(num_variables, 0.0);
48 num_bumps_.clear();
49 pq_need_update_for_var_at_trail_index_.IncreaseSize(num_variables);
50
51 weighted_sign_.resize(num_variables, 0.0);
52
53 has_forced_polarity_.resize(num_variables, false);
54 forced_polarity_.resize(num_variables);
55 has_target_polarity_.resize(num_variables, false);
56 target_polarity_.resize(num_variables);
57 var_polarity_.resize(num_variables);
58
59 ResetInitialPolarity(/*from=*/old_num_variables);
60
61 // Update the priority queue. Note that each addition is in O(1) because
62 // the priority is 0.0.
63 var_ordering_.Reserve(num_variables);
64 if (var_ordering_is_initialized_) {
65 for (BooleanVariable var(old_num_variables); var < num_variables; ++var) {
66 var_ordering_.Add({var, 0.0, activities_[var]});
67 }
68 }
69}

71void SatDecisionPolicy::BeforeConflict(int trail_index) {
72 if (parameters_.use_erwa_heuristic()) {
73 ++num_conflicts_;
74 num_conflicts_stack_.push_back({trail_.Index(), 1});
75 }
76
77 if (trail_index > target_length_) {
78 target_length_ = trail_index;
79 has_target_polarity_.assign(has_target_polarity_.size(), false);
80 for (int i = 0; i < trail_index; ++i) {
81 const Literal l = trail_[i];
82 has_target_polarity_[l.Variable()] = true;
83 target_polarity_[l.Variable()] = l.IsPositive();
84 }
85 }
86
87 if (trail_index > best_partial_assignment_.size()) {
88 best_partial_assignment_.assign(trail_.IteratorAt(0),
89 trail_.IteratorAt(trail_index));
90 }
91
92 --num_conflicts_until_rephase_;
93 RephaseIfNeeded();
94}

96void SatDecisionPolicy::RephaseIfNeeded() {
97 if (parameters_.polarity_rephase_increment() <= 0) return;
98 if (num_conflicts_until_rephase_ > 0) return;
99
100 VLOG(1) << "End of polarity phase " << polarity_phase_
101 << " target_length: " << target_length_
102 << " best_length: " << best_partial_assignment_.size();
103
104 ++polarity_phase_;
105 num_conflicts_until_rephase_ =
106 parameters_.polarity_rephase_increment() * (polarity_phase_ + 1);
107
108 // We always reset the target each time we change phase.
109 target_length_ = 0;
110 has_target_polarity_.assign(has_target_polarity_.size(), false);
111
112 // Cycle between different initial polarities. Note that we already start by
113 // the default polarity, and this code is reached the first time with a
114 // polarity_phase_ of 1.
115 switch (polarity_phase_ % 8) {
116 case 0:
117 ResetInitialPolarity(/*from=*/0);
118 break;
119 case 1:
120 UseLongestAssignmentAsInitialPolarity();
121 break;
122 case 2:
123 ResetInitialPolarity(/*from=*/0, /*inverted=*/true);
124 break;
125 case 3:
126 UseLongestAssignmentAsInitialPolarity();
127 break;
128 case 4:
129 RandomizeCurrentPolarity();
130 break;
131 case 5:
132 UseLongestAssignmentAsInitialPolarity();
133 break;
134 case 6:
135 FlipCurrentPolarity();
136 break;
137 case 7:
138 UseLongestAssignmentAsInitialPolarity();
139 break;
140 }
141}

143void SatDecisionPolicy::ResetDecisionHeuristic() {
144 const int num_variables = activities_.size();
145 variable_activity_increment_ = 1.0;
146 activities_.assign(num_variables, parameters_.initial_variables_activity());
147 tie_breakers_.assign(num_variables, 0.0);
148 num_bumps_.clear();
149 var_ordering_.Clear();
150
151 polarity_phase_ = 0;
152 num_conflicts_until_rephase_ = parameters_.polarity_rephase_increment();
153
154 ResetInitialPolarity(/*from=*/0);
155 has_target_polarity_.assign(num_variables, false);
156 has_forced_polarity_.assign(num_variables, false);
157 best_partial_assignment_.clear();
158
159 num_conflicts_ = 0;
160 num_conflicts_stack_.clear();
161
162 var_ordering_is_initialized_ = false;
163}

165void SatDecisionPolicy::ResetInitialPolarity(int from, bool inverted) {
166 // Sets the initial polarity.
167 //
168 // TODO(user): The WEIGHTED_SIGN one are currently slightly broken because the
169 // weighted_sign_ is updated after this has been called. It requires a call
170 // to ResetDecisionHeuristic() after all the constraint have been added. Fix.
171 // On another hand, this is only used with SolveWithRandomParameters() that
172 // does call this function.
173 const int num_variables = activities_.size();
174 for (BooleanVariable var(from); var < num_variables; ++var) {
175 switch (parameters_.initial_polarity()) {
176 case SatParameters::POLARITY_TRUE:
177 var_polarity_[var] = inverted ? false : true;
178 break;
179 case SatParameters::POLARITY_FALSE:
180 var_polarity_[var] = inverted ? true : false;
181 break;
182 case SatParameters::POLARITY_RANDOM:
183 var_polarity_[var] = std::uniform_int_distribution<int>(0, 1)(*random_);
184 break;
185 case SatParameters::POLARITY_WEIGHTED_SIGN:
186 var_polarity_[var] = weighted_sign_[var] > 0;
187 break;
188 case SatParameters::POLARITY_REVERSE_WEIGHTED_SIGN:
189 var_polarity_[var] = weighted_sign_[var] < 0;
190 break;
191 }
192 }
193}

195void SatDecisionPolicy::UseLongestAssignmentAsInitialPolarity() {
196 // In this special case, we just overwrite partially the current fixed
197 // polarity and reset the best best_partial_assignment_ for the next such
198 // phase.
199 for (const Literal l : best_partial_assignment_) {
200 var_polarity_[l.Variable()] = l.IsPositive();
201 }
202 best_partial_assignment_.clear();
203}

205void SatDecisionPolicy::FlipCurrentPolarity() {
206 const int num_variables = var_polarity_.size();
207 for (BooleanVariable var; var < num_variables; ++var) {
208 var_polarity_[var] = !var_polarity_[var];

219void SatDecisionPolicy::InitializeVariableOrdering() {
220 const int num_variables = activities_.size();
221
222 // First, extract the variables without activity, and add the other to the
223 // priority queue.
224 var_ordering_.Clear();
225 tmp_variables_.clear();
226 for (BooleanVariable var(0); var < num_variables; ++var) {
227 if (!trail_.Assignment().VariableIsAssigned(var)) {
228 if (activities_[var] > 0.0) {
229 var_ordering_.Add(
230 {var, static_cast<float>(tie_breakers_[var]), activities_[var]});
231 } else {
232 tmp_variables_.push_back(var);
233 }
234 }
235 }
236
237 // Set the order of the other according to the parameters_.
238 // Note that this is just a "preference" since the priority queue will kind
239 // of randomize this. However, it is more efficient than using the tie_breaker
240 // which add a big overhead on the priority queue.
241 //
242 // TODO(user): Experiment and come up with a good set of heuristics.
243 switch (parameters_.preferred_variable_order()) {
244 case SatParameters::IN_ORDER:
245 break;
246 case SatParameters::IN_REVERSE_ORDER:
247 std::reverse(tmp_variables_.begin(), tmp_variables_.end());
248 break;
249 case SatParameters::IN_RANDOM_ORDER:
250 std::shuffle(tmp_variables_.begin(), tmp_variables_.end(), *random_);
251 break;
252 }
253
254 // Add the variables without activity to the queue (in the default order)
255 for (const BooleanVariable var : tmp_variables_) {
256 var_ordering_.Add({var, static_cast<float>(tie_breakers_[var]), 0.0});
257 }
258
259 // Finish the queue initialization.
260 pq_need_update_for_var_at_trail_index_.ClearAndResize(num_variables);
261 pq_need_update_for_var_at_trail_index_.SetAllBefore(trail_.Index());
262 var_ordering_is_initialized_ = true;
263}

265void SatDecisionPolicy::SetAssignmentPreference(Literal literal,
266 double weight) {
267 if (!parameters_.use_optimization_hints()) return;
268 DCHECK_GE(weight, 0.0);
269 DCHECK_LE(weight, 1.0);
270
271 has_forced_polarity_[literal.Variable()] = true;
272 forced_polarity_[literal.Variable()] = literal.IsPositive();
273
274 // The tie_breaker is changed, so we need to reinitialize the priority queue.
275 // Note that this doesn't change the activity though.
276 tie_breakers_[literal.Variable()] = weight;
277 var_ordering_is_initialized_ = false;
278}

280std::vector<std::pair<Literal, double>> SatDecisionPolicy::AllPreferences()
281 const {
282 std::vector<std::pair<Literal, double>> prefs;
283 for (BooleanVariable var(0); var < var_polarity_.size(); ++var) {
284 // TODO(user): we currently assume that if the tie_breaker is zero then
285 // no preference was set (which is not 100% correct). Fix that.
286 const double value = var_ordering_.GetElement(var.value()).tie_breaker;
287 if (value > 0.0) {
288 prefs.push_back(std::make_pair(Literal(var, var_polarity_[var]), value));
289 }
290 }
291 return prefs;
292}

294void SatDecisionPolicy::UpdateWeightedSign(
295 const std::vector<LiteralWithCoeff>& terms, Coefficient rhs) {
296 for (const LiteralWithCoeff& term : terms) {
297 const double weight = static_cast<double>(term.coefficient.value()) /
298 static_cast<double>(rhs.value());
299 weighted_sign_[term.literal.Variable()] +=
300 term.literal.IsPositive() ? -weight : weight;
301 }

304void SatDecisionPolicy::BumpVariableActivities(
305 absl::Span<const Literal> literals) {
306 if (parameters_.use_erwa_heuristic()) {
307 if (num_bumps_.size() != activities_.size()) {
308 num_bumps_.resize(activities_.size(), 0);
309 }
310 for (const Literal literal : literals) {
311 // Note that we don't really need to bump level 0 variables since they
312 // will never be backtracked over. However it is faster to simply bump
313 // them.
314 ++num_bumps_[literal.Variable()];
315 }
316 return;
317 }
318
319 const double max_activity_value = parameters_.max_variable_activity_value();
320 for (const Literal literal : literals) {
321 const BooleanVariable var = literal.Variable();
322 const int level = trail_.Info(var).level;
323 if (level == 0) continue;
324 activities_[var] += variable_activity_increment_;
325 pq_need_update_for_var_at_trail_index_.Set(trail_.Info(var).trail_index);
326 if (activities_[var] > max_activity_value) {
327 RescaleVariableActivities(1.0 / max_activity_value);
328 }
329 }
330}

332void SatDecisionPolicy::RescaleVariableActivities(double scaling_factor) {
333 variable_activity_increment_ *= scaling_factor;
334 for (BooleanVariable var(0); var < activities_.size(); ++var) {
335 activities_[var] *= scaling_factor;
336 }
337
338 // When rescaling the activities of all the variables, the order of the
339 // active variables in the heap will not change, but we still need to update
340 // their weights so that newly inserted elements will compare correctly with
341 // already inserted ones.
342 //
343 // IMPORTANT: we need to reset the full heap from scratch because just
344 // multiplying the current weight by scaling_factor is not guaranteed to
345 // preserve the order. This is because the activity of two entries may go to
346 // zero and the tie-breaking ordering may change their relative order.
347 //
348 // InitializeVariableOrdering() will be called lazily only if needed.
349 var_ordering_is_initialized_ = false;
350}
351
352void SatDecisionPolicy::UpdateVariableActivityIncrement() {
353 variable_activity_increment_ *= 1.0 / parameters_.variable_activity_decay();
354}

356Literal SatDecisionPolicy::NextBranch() {
357 // Lazily initialize var_ordering_ if needed.
358 if (!var_ordering_is_initialized_) {
359 InitializeVariableOrdering();
360 }
361
362 // Choose the variable.
363 BooleanVariable var;
364 const double ratio = parameters_.random_branches_ratio();
365 auto zero_to_one = [this]() {
366 return std::uniform_real_distribution<double>()(*random_);
367 };
368 if (ratio != 0.0 && zero_to_one() < ratio) {
369 while (true) {
370 // TODO(user): This may not be super efficient if almost all the
371 // variables are assigned.
372 std::uniform_int_distribution<int> index_dist(0,
373 var_ordering_.Size() - 1);
374 var = var_ordering_.QueueElement(index_dist(*random_)).var;
375 if (!trail_.Assignment().VariableIsAssigned(var)) break;
376 pq_need_update_for_var_at_trail_index_.Set(trail_.Info(var).trail_index);
377 var_ordering_.Remove(var.value());
378 }
379 } else {
380 // The loop is done this way in order to leave the final choice in the heap.
381 DCHECK(!var_ordering_.IsEmpty());
382 var = var_ordering_.Top().var;
383 while (trail_.Assignment().VariableIsAssigned(var)) {
384 var_ordering_.Pop();
385 pq_need_update_for_var_at_trail_index_.Set(trail_.Info(var).trail_index);
386 DCHECK(!var_ordering_.IsEmpty());
387 var = var_ordering_.Top().var;
388 }
389 }
390
391 // Choose its polarity (i.e. True of False).
392 const double random_ratio = parameters_.random_polarity_ratio();
393 if (random_ratio != 0.0 && zero_to_one() < random_ratio) {
394 return Literal(var, std::uniform_int_distribution<int>(0, 1)(*random_));
395 }
396
397 if (has_forced_polarity_[var]) return Literal(var, forced_polarity_[var]);
398 if (in_stable_phase_ && has_target_polarity_[var]) {
399 return Literal(var, target_polarity_[var]);
400 }
401 return Literal(var, var_polarity_[var]);
402}

404void SatDecisionPolicy::PqInsertOrUpdate(BooleanVariable var) {
405 const WeightedVarQueueElement element{
406 var, static_cast<float>(tie_breakers_[var]), activities_[var]};
407 if (var_ordering_.Contains(var.value())) {
408 // Note that the new weight should always be higher than the old one.
409 var_ordering_.IncreasePriority(element);
410 } else {
411 var_ordering_.Add(element);
412 }
413}

415void SatDecisionPolicy::Untrail(int target_trail_index) {
416 // TODO(user): avoid looping twice over the trail?
417 if (maybe_enable_phase_saving_ && parameters_.use_phase_saving()) {
418 for (int i = target_trail_index; i < trail_.Index(); ++i) {
419 const Literal l = trail_[i];
420 var_polarity_[l.Variable()] = l.IsPositive();
421 }
422 }
423
424 DCHECK_LT(target_trail_index, trail_.Index());
425 if (parameters_.use_erwa_heuristic()) {
426 if (num_bumps_.size() != activities_.size()) {
427 num_bumps_.resize(activities_.size(), 0);
428 }
429
430 // The ERWA parameter between the new estimation of the learning rate and
431 // the old one. TODO(user): Expose parameters for these values.
432 const double alpha = std::max(0.06, 0.4 - 1e-6 * num_conflicts_);
433
434 // This counts the number of conflicts since the assignment of the variable
435 // at the current trail_index that we are about to untrail.
436 int num_conflicts = 0;
437 int next_num_conflicts_update =
438 num_conflicts_stack_.empty() ? -1
439 : num_conflicts_stack_.back().trail_index;
440
441 int trail_index = trail_.Index();
442 while (trail_index > target_trail_index) {
443 if (next_num_conflicts_update == trail_index) {
444 num_conflicts += num_conflicts_stack_.back().count;
445 num_conflicts_stack_.pop_back();
446 next_num_conflicts_update =
447 num_conflicts_stack_.empty()
448 ? -1
449 : num_conflicts_stack_.back().trail_index;
450 }
451 const BooleanVariable var = trail_[--trail_index].Variable();
452
453 // TODO(user): This heuristic can make this code quite slow because
454 // all the untrailed variable will cause a priority queue update.
455 if (num_conflicts > 0) {
456 const int64_t num_bumps = num_bumps_[var];
457 double new_rate = 0.0;
458 if (num_bumps > 0) {
459 num_bumps_[var] = 0;
460 new_rate = static_cast<double>(num_bumps) / num_conflicts;
461 }
462 activities_[var] = alpha * new_rate + (1 - alpha) * activities_[var];
463 }
464 if (var_ordering_is_initialized_) PqInsertOrUpdate(var);
465 }
466 if (num_conflicts > 0) {
467 if (!num_conflicts_stack_.empty() &&
468 num_conflicts_stack_.back().trail_index == trail_.Index()) {
469 num_conflicts_stack_.back().count += num_conflicts;
470 } else {
471 num_conflicts_stack_.push_back({trail_.Index(), num_conflicts});
472 }
473 }
474 } else {
475 if (!var_ordering_is_initialized_) return;
476
477 // Trail index of the next variable that will need a priority queue update.
478 int to_update = pq_need_update_for_var_at_trail_index_.Top();
479 while (to_update >= target_trail_index) {
480 DCHECK_LT(to_update, trail_.Index());
481 PqInsertOrUpdate(trail_[to_update].Variable());
482 pq_need_update_for_var_at_trail_index_.ClearTop();
483 to_update = pq_need_update_for_var_at_trail_index_.Top();
484 }
485 }
486
487 // Invariant.
488 if (DEBUG_MODE && var_ordering_is_initialized_) {
489 for (int trail_index = trail_.Index() - 1; trail_index > target_trail_index;
490 --trail_index) {
491 const BooleanVariable var = trail_[trail_index].Variable();
492 CHECK(var_ordering_.Contains(var.value()));
493 CHECK_EQ(activities_[var], var_ordering_.GetElement(var.value()).weight);
494 }
495 }
496}
