29// Implementation of Kruskal's mininumum spanning tree algorithm (c.f.
30// https://en.wikipedia.org/wiki/Kruskal%27s_algorithm).
31// Returns the index of the arcs appearing in the tree; will return a forest if
32// the graph is disconnected. Nodes without any arcs will be ignored.
33// Each arc of the graph is interpreted as an undirected arc.
34// Complexity of the algorithm is O(E * log(E)) where E is the number of arcs
35// in the graph. Memory usage is O(E * log(E)).
36
37// TODO(user): Add a global Minimum Spanning Tree API automatically switching
38// between Prim and Kruskal depending on problem size.
39
40// Version taking sorted graph arcs. Allows somewhat incremental recomputation
41// of minimum spanning trees as most of the processing time is spent sorting
42// arcs.
43// Usage:
44// ListGraph<int, int> graph(...);
45// std::vector<int> sorted_arcs = ...;
46// std::vector<int> mst = BuildKruskalMinimumSpanningTreeFromSortedArcs(
47// graph, sorted_arcs);
48//
49template <typename Graph>
50std::vector<typename Graph::ArcIndex>
51BuildKruskalMinimumSpanningTreeFromSortedArcs(
52 const Graph& graph,
53 absl::Span<const typename Graph::ArcIndex> sorted_arcs) {
54 using ArcIndex = typename Graph::ArcIndex;
55 using NodeIndex = typename Graph::NodeIndex;
56 const int num_arcs = graph.num_arcs();
57 int arc_index = 0;
58 std::vector<ArcIndex> tree_arcs;
59 if (graph.num_nodes() == 0) {
60 return tree_arcs;
61 }
62 const int expected_tree_size = graph.num_nodes() - 1;
63 tree_arcs.reserve(expected_tree_size);
64 DenseConnectedComponentsFinder components;
65 components.SetNumberOfNodes(graph.num_nodes());
66 while (tree_arcs.size() != expected_tree_size && arc_index < num_arcs) {
67 const ArcIndex arc = sorted_arcs[arc_index];
68 const auto tail = graph.Tail(arc);
69 const auto head = graph.Head(arc);
70 if (!components.Connected(tail, head)) {
71 components.AddEdge(tail, head);
72 tree_arcs.push_back(arc);
73 }
74 ++arc_index;
75 }
76 return tree_arcs;
77}

79// Version taking an arc comparator to sort graph arcs.
80// Usage:
81// ListGraph<int, int> graph(...);
82// const auto arc_cost = [&graph](int arc) {
83// return f(graph.Tail(arc), graph.Head(arc));
84// };
85// std::vector<int> mst = BuildKruskalMinimumSpanningTree(
86// graph,
87// [&arc_cost](int a, int b) { return arc_cost(a) < arc_cost(b); });
88//
89template <typename Graph, typename ArcComparator>
90std::vector<typename Graph::ArcIndex> BuildKruskalMinimumSpanningTree(
91 const Graph& graph, const ArcComparator& arc_comparator) {
92 using ArcIndex = typename Graph::ArcIndex;
93 std::vector<ArcIndex> sorted_arcs(graph.num_arcs());
94 for (const ArcIndex arc : graph.AllForwardArcs()) {
95 sorted_arcs[arc] = arc;
96 }
97 std::sort(sorted_arcs.begin(), sorted_arcs.end(), arc_comparator);
98 return BuildKruskalMinimumSpanningTreeFromSortedArcs(graph, sorted_arcs);
99}

101// Implementation of Prim's mininumum spanning tree algorithm (c.f.
102// https://en.wikipedia.org/wiki/Prim's_algorithm) on undirected connected
103// graphs.
104// Returns the index of the arcs appearing in the tree.
105// Complexity of the algorithm is O(E * log(V)) where E is the number of arcs
106// in the graph, V is the number of vertices. Memory usage is O(V) + memory
107// taken by the graph.
108// Usage:
109// ListGraph<int, int> graph(...);
110// const auto arc_cost = [&graph](int arc) -> int64_t {
111// return f(graph.Tail(arc), graph.Head(arc));
112// };
113// std::vector<int> mst = BuildPrimMinimumSpanningTree(graph, arc_cost);
114//
115template <typename Graph, typename ArcValue>
116std::vector<typename Graph::ArcIndex> BuildPrimMinimumSpanningTree(
117 const Graph& graph, const ArcValue& arc_value) {
118 using ArcIndex = typename Graph::ArcIndex;
119 using NodeIndex = typename Graph::NodeIndex;
120 using ArcValueType = decltype(arc_value(0));
121 std::vector<ArcIndex> tree_arcs;
122 if (graph.num_nodes() == 0) {
123 return tree_arcs;
124 }
125 const int expected_tree_size = graph.num_nodes() - 1;
126 tree_arcs.reserve(expected_tree_size);
127 std::vector<ArcIndex> node_neighbor(graph.num_nodes(), Graph::kNilArc);
128 std::vector<bool> node_active(graph.num_nodes(), true);
129
130 // This struct represents entries in the adjustable priority queue which
131 // maintains active nodes (not added to the tree yet) in decreasing insertion
132 // cost order. AdjustablePriorityQueue requires the existence of the
133 // SetHeapIndex and GetHeapIndex methods.
134 struct Entry {
135 void SetHeapIndex(int index) { heap_index = index; }
136 int GetHeapIndex() const { return heap_index; }
137 bool operator<(const Entry& other) const { return value > other.value; }
138
139 NodeIndex node;
140 ArcValueType value;
141 int heap_index;
142 };
143
144 AdjustablePriorityQueue<Entry> pq;
145 std::vector<Entry> entries;
146 std::vector<bool> touched_entry(graph.num_nodes(), false);
147 for (NodeIndex node : graph.AllNodes()) {
148 entries.push_back({node, std::numeric_limits<ArcValueType>::max(), -1});
149 }
150 entries[0].value = 0;
151 pq.Add(&entries[0]);
152 while (!pq.IsEmpty() && tree_arcs.size() != expected_tree_size) {
153 const Entry* best = pq.Top();
154 const NodeIndex node = best->node;
155 pq.Pop();
156 node_active[node] = false;
157 if (node_neighbor[node] != Graph::kNilArc) {
158 tree_arcs.push_back(node_neighbor[node]);
159 }
160 for (const ArcIndex arc : graph.OutgoingArcs(node)) {
161 const NodeIndex neighbor = graph.Head(arc);
162 if (node_active[neighbor]) {
163 const ArcValueType value = arc_value(arc);
164 Entry& entry = entries[neighbor];
165 if (value < entry.value || !touched_entry[neighbor]) {
166 node_neighbor[neighbor] = arc;
167 entry.value = value;
168 touched_entry[neighbor] = true;
169 if (pq.Contains(&entry)) {
170 pq.NoteChangedPriority(&entry);
171 } else {
172 pq.Add(&entry);
173 }
174 }
175 }
176 }
177 }
178 return tree_arcs;
179}
