37LinearExpr& LinearExpr::operator+=(const LinearExpr& rhs) {
38 for (const auto& kv : rhs.terms_) {
39 terms_[kv.first] += kv.second;

45LinearExpr& LinearExpr::operator-=(const LinearExpr& rhs) {

53LinearExpr& LinearExpr::operator*=(double rhs) {
54 if (rhs == 0) {
55 terms_.clear();
56 offset_ = 0;
57 } else if (rhs != 1) {
58 for (auto& kv : terms_) {
59 kv.second *= rhs;
60 }
61 offset_ *= rhs;
62 }
63 return *this;
64}

90void AppendTerm(const double coef, const std::string& var_name,
91 const bool is_first, std::string* s) {
92 if (is_first) {
93 if (coef == 1.0) {
94 absl::StrAppend(s, var_name);
95 } else if (coef == -1.0) {
96 absl::StrAppend(s, "-", var_name);
97 } else {
98 absl::StrAppend(s, coef, "*", var_name);
99 }
100 } else {
101 const std::string op = coef < 0 ? "-" : "+";
102 const double abs_coef = std::abs(coef);
103 if (abs_coef == 1.0) {
104 absl::StrAppend(s, " ", op, " ", var_name);
105 } else {
106 absl::StrAppend(s, " ", op, " ", abs_coef, "*", var_name);
107 }
108 }
109}

111void AppendOffset(const double offset, const bool is_first, std::string* s) {
112 if (is_first) {
113 absl::StrAppend(s, offset);
114 } else {
115 if (offset != 0.0) {
116 const std::string op = offset < 0 ? "-" : "+";
117 absl::StrAppend(s, " ", op, " ", std::abs(offset));
118 }
119 }
120}

124std::string LinearExpr::ToString() const {
125 std::vector<const MPVariable*> vars_in_order;
126 for (const auto& var_val_pair : terms_) {
127 vars_in_order.push_back(var_val_pair.first);
128 }
129 std::sort(vars_in_order.begin(), vars_in_order.end(),
130 [](const MPVariable* v, const MPVariable* u) {
131 return v->index() < u->index();
132 });
133 std::string result;
134 bool is_first = true;
135 for (const MPVariable* var : vars_in_order) {
136 // MPSolver gives names to all variables, even if you don't.
137 DCHECK(!var->name().empty());
138 AppendTerm(terms_.at(var), var->name(), is_first, &result);
139 is_first = false;
140 }
141 AppendOffset(offset_, is_first, &result);
142 // TODO(user): support optionally cropping long strings.
143 return result;
144}
