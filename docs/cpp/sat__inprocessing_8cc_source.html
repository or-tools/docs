50void PostsolveClauses::AddClauseWithSpecialLiteral(
51 Literal literal, absl::Span<const Literal> clause) {
52 bool found = false;
53 clauses.emplace_back(clause.begin(), clause.end());
54 for (int i = 0; i < clause.size(); ++i) {
55 if (clause[i] == literal) {
56 found = true;
57 std::swap(clauses.back()[0], clauses.back()[i]);
58 break;
59 }
60 }
61 CHECK(found);
62}

67bool Inprocessing::PresolveLoop(SatPresolveOptions options) {
68 WallTimer wall_timer;
69 wall_timer.Start();
70
71 // Mainly useful for development.
72 double probing_time = 0.0;
73 const bool log_round_info = VLOG_IS_ON(1);
74
75 // We currently do the transformations in a given order and restart each time
76 // we did something to make sure that the earlier step cannot strengthen more.
77 // This might not be the best, but it is really good during development phase
78 // to make sure each individual functions is as incremental and as fast as
79 // possible.
80 const double start_dtime = time_limit_->GetElapsedDeterministicTime();
81 const double stop_dtime = start_dtime + options.deterministic_time_limit;
82 while (!time_limit_->LimitReached() &&
83 time_limit_->GetElapsedDeterministicTime() <= stop_dtime) {
84 CHECK_EQ(sat_solver_->CurrentDecisionLevel(), 0);
85 if (!LevelZeroPropagate()) return false;
86
87 // This one is fast since only newly fixed variables are considered.
88 implication_graph_->RemoveFixedVariables();
89 implication_graph_->RemoveDuplicates();
90
91 // This also prepare the stamping below so that we do that on a DAG and do
92 // not consider potential new implications added by
93 // RemoveFixedAndEquivalentVariables().
94 RETURN_IF_FALSE(DetectEquivalencesAndStamp(options.use_transitive_reduction,
95 log_round_info));
96
97 // TODO(user): This should/could be integrated with the stamping since it
98 // seems better to do just one loop instead of two over all clauses. Because
99 // of memory access. it isn't that clear though.
100 RETURN_IF_FALSE(RemoveFixedAndEquivalentVariables(log_round_info));
101
102 // IMPORTANT: Since we only run this on pure sat problem, we can just
103 // get rid of equivalent variable right away and do not need to keep them
104 // in the implication_graph_ for propagation.
105 //
106 // This is needed for the correctness of the bounded variable elimination.
107 implication_graph_->RemoveAllRedundantVariables(&postsolve_->clauses);
108
109 RETURN_IF_FALSE(stamping_simplifier_->DoOneRound(log_round_info));
110
111 // We wait for the fix-point to be reached before doing the other
112 // simplifications below.
113 if (MoreFixedVariableToClean() || MoreRedundantVariableToClean() ||
114 !implication_graph_->IsDag()) {
115 continue;
116 }
117
118 RETURN_IF_FALSE(SubsumeAndStrenghtenRound(log_round_info));
119 if (MoreFixedVariableToClean() || MoreRedundantVariableToClean() ||
120 !implication_graph_->IsDag()) {
121 continue;
122 }
123
124 // TODO(user): Combine the two? this way we don't create a full literal <->
125 // clause graph twice. It might make sense to reach the BCE fix point which
126 // is unique before each variable elimination.
127 blocked_clause_simplifier_->DoOneRound(log_round_info);
128
129 // TODO(user): this break some binary graph invariant. Fix!
130 RETURN_IF_FALSE(RemoveFixedAndEquivalentVariables(log_round_info));
131 RETURN_IF_FALSE(bounded_variable_elimination_->DoOneRound(log_round_info));
132 RETURN_IF_FALSE(LevelZeroPropagate());
133
134 // Probing.
135 const double saved_wtime = wall_timer.Get();
136 const double time_left =
137 stop_dtime - time_limit_->GetElapsedDeterministicTime();
138 if (time_left <= 0) break;
139 ProbingOptions probing_options;
140 probing_options.log_info = log_round_info;
141 probing_options.deterministic_limit = time_left;
142 probing_options.extract_binary_clauses =
143 options.extract_binary_clauses_in_probing;
144 RETURN_IF_FALSE(FailedLiteralProbingRound(probing_options, model_));
145 probing_time += wall_timer.Get() - saved_wtime;
146
147 if (MoreFixedVariableToClean() || MoreRedundantVariableToClean() ||
148 !implication_graph_->IsDag()) {
149 continue;
150 }
151
152 break;
153 }
154
155 // TODO(user): Maintain the total number of literals in the watched clauses.
156 if (!LevelZeroPropagate()) return false;
157
158 SOLVER_LOG(
159 logger_, "[Pure SAT presolve]", " num_fixed: ", trail_->Index(),
160 " num_redundant: ", implication_graph_->num_redundant_literals() / 2, "/",
161 sat_solver_->NumVariables(),
162 " num_implications: ", implication_graph_->num_implications(),
163 " num_watched_clauses: ", clause_manager_->num_watched_clauses(),
164 " dtime: ", time_limit_->GetElapsedDeterministicTime() - start_dtime, "/",
165 options.deterministic_time_limit, " wtime: ", wall_timer.Get(),
166 " non-probing time: ", (wall_timer.Get() - probing_time));
167 return true;
168}

170bool Inprocessing::InprocessingRound() {
171 DCHECK_EQ(sat_solver_->CurrentDecisionLevel(), 0);
172 WallTimer wall_timer;
173 wall_timer.Start();
174
175 const bool log_info = VLOG_IS_ON(1);
176 const bool log_round_info = VLOG_IS_ON(2);
177 const double start_dtime = time_limit_->GetElapsedDeterministicTime();
178
179 // Mainly useful for development.
180 double probing_time = 0.0;
181
182 // Store the dtime of the first call (first restart) and wait for the next
183 // restart.
184 if (first_inprocessing_call_) {
185 reference_dtime_ = start_dtime;
186 first_inprocessing_call_ = false;
187 return true;
188 }
189
190 // Try to spend a given ratio of time in the inprocessing.
191 const double diff = start_dtime - reference_dtime_;
192 if (total_dtime_ > params_.inprocessing_dtime_ratio() * diff) {
193 return true;
194 }
195
196 // LP Propagation during inprocessing can be really slow, so we temporarily
197 // disable it.
198 //
199 // TODO(user): The LP and incremental structure will still be called though,
200 // which can take some time, try to disable it more cleanly.
201 std::vector<std::pair<LinearProgrammingConstraint*, bool>> saved_state;
202 for (LinearProgrammingConstraint* lp : *all_lp_constraints_) {
203 saved_state.push_back({lp, lp->PropagationIsEnabled()});
204 lp->EnablePropagation(false);
205 }
206 auto cleanup = absl::MakeCleanup([&saved_state]() {
207 for (const auto [lp, old_value] : saved_state) {
208 lp->EnablePropagation(old_value);
209 }
210 });
211
212 // We make sure we do not "pollute" the current saved polarities. We will
213 // restore them at the end.
214 //
215 // TODO(user): We should probably also disable the variable/clauses activity
216 // updates.
217 decision_policy_->MaybeEnablePhaseSaving(/*save_phase=*/false);
218
219 implication_graph_->RemoveDuplicates();
220 RETURN_IF_FALSE(DetectEquivalencesAndStamp(true, log_round_info));
221 RETURN_IF_FALSE(RemoveFixedAndEquivalentVariables(log_round_info));
222 RETURN_IF_FALSE(LevelZeroPropagate());
223
224 // Probing.
225 if (params_.inprocessing_probing_dtime() > 0.0) {
226 const double saved_wtime = wall_timer.Get();
227 ProbingOptions probing_options;
228 probing_options.log_info = log_round_info;
229 probing_options.deterministic_limit = params_.inprocessing_probing_dtime();
230 probing_options.extract_binary_clauses = true;
231 RETURN_IF_FALSE(FailedLiteralProbingRound(probing_options, model_));
232 probing_time += wall_timer.Get() - saved_wtime;
233 }
234
235 RETURN_IF_FALSE(DetectEquivalencesAndStamp(true, log_round_info));
236 RETURN_IF_FALSE(RemoveFixedAndEquivalentVariables(log_round_info));
237 RETURN_IF_FALSE(LevelZeroPropagate());
238
239 RETURN_IF_FALSE(stamping_simplifier_->DoOneRound(log_round_info));
240 RETURN_IF_FALSE(RemoveFixedAndEquivalentVariables(log_round_info));
241 RETURN_IF_FALSE(LevelZeroPropagate());
242
243 // TODO(user): Add a small wrapper function to time this.
244 const auto old_counter = sat_solver_->counters();
245 if (params_.inprocessing_minimization_dtime() > 0.0) {
246 RETURN_IF_FALSE(sat_solver_->MinimizeByPropagation(
247 params_.inprocessing_minimization_dtime()));
248 }
249 const int64_t mini_num_clause =
250 sat_solver_->counters().minimization_num_clauses -
251 old_counter.minimization_num_clauses;
252 const int64_t mini_num_removed =
253 sat_solver_->counters().minimization_num_removed_literals -
254 old_counter.minimization_num_removed_literals;
255
256 RETURN_IF_FALSE(RemoveFixedAndEquivalentVariables(log_round_info));
257 RETURN_IF_FALSE(SubsumeAndStrenghtenRound(log_round_info));
258 RETURN_IF_FALSE(RemoveFixedAndEquivalentVariables(log_round_info));
259
260 // TODO(user): try to enable these? The problem is that we can only remove
261 // variables not used the non-pure SAT part of a model.
262 if (/*DISABLES_CODE*/ (false)) {
263 blocked_clause_simplifier_->DoOneRound(log_round_info);
264 RETURN_IF_FALSE(bounded_variable_elimination_->DoOneRound(log_round_info));
265 }
266 RETURN_IF_FALSE(LevelZeroPropagate());
267
268 sat_solver_->AdvanceDeterministicTime(time_limit_);
269 total_dtime_ += time_limit_->GetElapsedDeterministicTime() - start_dtime;
270 if (log_info) {
271 SOLVER_LOG(
272 logger_, "Inprocessing.", " fixed:", trail_->Index(),
273 " equiv:", implication_graph_->num_redundant_literals() / 2,
274 " bools:", sat_solver_->NumVariables(),
275 " implications:", implication_graph_->num_implications(),
276 " watched:", clause_manager_->num_watched_clauses(),
277 " minimization:", mini_num_clause, "|", mini_num_removed,
278 " dtime:", time_limit_->GetElapsedDeterministicTime() - start_dtime,
279 " wtime:", wall_timer.Get(),
280 " np_wtime:", (wall_timer.Get() - probing_time));
281 }
282
283 DCHECK_EQ(sat_solver_->CurrentDecisionLevel(), 0);
284 decision_policy_->MaybeEnablePhaseSaving(/*save_phase=*/true);
285 return true;
286}

307// It make sense to do the pre-stamping right after the equivalence detection
308// since it needs a DAG and can detect extra failed literal.
309bool Inprocessing::DetectEquivalencesAndStamp(bool use_transitive_reduction,
310 bool log_info) {
311 if (!LevelZeroPropagate()) return false;
312 implication_graph_->RemoveFixedVariables();
313 if (!implication_graph_->IsDag()) {
314 // TODO(user): consider doing the transitive reduction after each SCC.
315 // It might be slow but we could try to make it more incremental to
316 // compensate and it should allow further reduction.
317 if (!implication_graph_->DetectEquivalences(log_info)) return false;
318 if (!LevelZeroPropagate()) return false;
319 if (use_transitive_reduction) {
320 if (!implication_graph_->ComputeTransitiveReduction(log_info)) {
321 return false;
322 }
323 if (!LevelZeroPropagate()) return false;
324 }
325 }
326
327 if (!stamping_simplifier_->ComputeStampsForNextRound(log_info)) return false;
328 return LevelZeroPropagate();
329}

331bool Inprocessing::RemoveFixedAndEquivalentVariables(bool log_info) {
332 // Preconditions.
333 //
334 // TODO(user): The level zero is required because we remove fixed variables
335 // but if we split this into two functions, we could rewrite clause at any
336 // level.
337 CHECK_EQ(sat_solver_->CurrentDecisionLevel(), 0);
338 if (!LevelZeroPropagate()) return false;
339
340 // Test if some work is needed.
341 //
342 // TODO(user): If only new fixed variables are there, we can use a faster
343 // function. We should also merge the code with the deletion code in
344 // sat_solver_.cc, but that require some refactoring of the dependence between
345 // files.
346 const int64_t new_num_redundant_literals =
347 implication_graph_->num_redundant_literals();
348 const int64_t new_num_fixed_variables = trail_->Index();
349 if (last_num_redundant_literals_ == new_num_redundant_literals &&
350 last_num_fixed_variables_ == new_num_fixed_variables) {
351 return true;
352 }
353 last_num_fixed_variables_ = new_num_fixed_variables;
354 last_num_redundant_literals_ = new_num_redundant_literals;
355
356 // Start the round.
357 WallTimer wall_timer;
358 wall_timer.Start();
359
360 int64_t num_removed_literals = 0;
361 int64_t num_inspected_literals = 0;
362
363 // We need this temporary vector for the DRAT proof settings, otherwise
364 // we could just have done an in-place transformation.
365 std::vector<Literal> new_clause;
366
367 // Used to mark clause literals.
368 const int num_literals(sat_solver_->NumVariables() * 2);
369 absl::StrongVector<LiteralIndex, bool> marked(num_literals, false);
370
371 clause_manager_->DeleteRemovedClauses();
372 clause_manager_->DetachAllClauses();
373 for (SatClause* clause : clause_manager_->AllClausesInCreationOrder()) {
374 bool removed = false;
375 bool need_rewrite = false;
376
377 // We first do a loop to see if there is anything to do.
378 for (const Literal l : clause->AsSpan()) {
379 if (assignment_.LiteralIsTrue(l)) {
380 // TODO(user): we should output literal to the proof right away,
381 // currently if we remove clauses before fixing literal the proof is
382 // wrong.
383 if (!clause_manager_->InprocessingFixLiteral(l)) return false;
384 clause_manager_->InprocessingRemoveClause(clause);
385 num_removed_literals += clause->size();
386 removed = true;
387 break;
388 }
389 if (assignment_.LiteralIsFalse(l) || implication_graph_->IsRedundant(l)) {
390 need_rewrite = true;
391 break;
392 }
393 }
394
395 num_inspected_literals += clause->size();
396 if (removed || !need_rewrite) continue;
397 num_inspected_literals += clause->size();
398
399 // Rewrite the clause.
400 new_clause.clear();
401 for (const Literal l : clause->AsSpan()) {
402 const Literal r = implication_graph_->RepresentativeOf(l);
403 if (marked[r] || assignment_.LiteralIsFalse(r)) {
404 continue;
405 }
406 if (marked[r.NegatedIndex()] || assignment_.LiteralIsTrue(r)) {
407 clause_manager_->InprocessingRemoveClause(clause);
408 num_removed_literals += clause->size();
409 removed = true;
410 break;
411 }
412 marked[r] = true;
413 new_clause.push_back(r);
414 }
415
416 // Restore marked.
417 for (const Literal l : new_clause) marked[l] = false;
418 if (removed) continue;
419
420 num_removed_literals += clause->size() - new_clause.size();
421 if (!clause_manager_->InprocessingRewriteClause(clause, new_clause)) {
422 return false;
423 }
424 }
425
426 // TODO(user): find a way to auto-tune that after a run on borg...
427 const double dtime = static_cast<double>(num_inspected_literals) * 1e-8;
428 time_limit_->AdvanceDeterministicTime(dtime);
429 LOG_IF(INFO, log_info) << "Cleanup. num_removed_literals: "
430 << num_removed_literals << " dtime: " << dtime
431 << " wtime: " << wall_timer.Get();
432 return true;
433}

435// TODO(user): Use better work limits, see SAT09.CRAFTED.ramseycube.Q3inK12
436//
437// TODO(user): Be more incremental, each time a clause is added/reduced track
438// which literal are impacted? Also try to do orthogonal reductions from one
439// round to the next.
440bool Inprocessing::SubsumeAndStrenghtenRound(bool log_info) {
441 WallTimer wall_timer;
442 wall_timer.Start();
443
444 int64_t num_subsumed_clauses = 0;
445 int64_t num_removed_literals = 0;
446 int64_t num_inspected_signatures = 0;
447 int64_t num_inspected_literals = 0;
448
449 // We need this temporary vector for the DRAT proof settings, otherwise
450 // we could just have done an in-place transformation.
451 std::vector<Literal> new_clause;
452
453 // This function needs the watcher to be detached as we might remove some
454 // of the watched literals.
455 //
456 // TODO(user): We could do that only if we do some reduction, but this is
457 // quite fast though.
458 clause_manager_->DeleteRemovedClauses();
459 clause_manager_->DetachAllClauses();
460
461 // Process clause by increasing sizes.
462 // TODO(user): probably faster without the size indirection.
463 std::vector<SatClause*> clauses =
464 clause_manager_->AllClausesInCreationOrder();
465 std::stable_sort(
466 clauses.begin(), clauses.end(),
467 [](SatClause* a, SatClause* b) { return a->size() < b->size(); });
468
469 // Used to mark clause literals.
470 const LiteralIndex num_literals(sat_solver_->NumVariables() * 2);
471 SparseBitset<LiteralIndex> marked(num_literals);
472
473 // Clause index in clauses.
474 // TODO(user): Storing signatures here might be faster?
475 absl::StrongVector<LiteralIndex, absl::InlinedVector<int, 6>> one_watcher(
476 num_literals.value());
477
478 // Clause signatures in the same order as clauses.
479 std::vector<uint64_t> signatures(clauses.size());
480
481 std::vector<Literal> candidates_for_removal;
482 for (int clause_index = 0; clause_index < clauses.size(); ++clause_index) {
483 SatClause* clause = clauses[clause_index];
484
485 // TODO(user): Better abort limit. We could also limit the watcher sizes and
486 // never look at really long clauses. Note that for an easier
487 // incrementality, it is better to reach some kind of completion so we know
488 // what new stuff need to be done.
489 if (num_inspected_literals + num_inspected_signatures > 1e9) {
490 break;
491 }
492
493 // Check for subsumption, note that this currently ignore all clauses in the
494 // binary implication graphs. Stamping is doing some of that (and some also
495 // happen during probing), but we could consider only direct implications
496 // here and be a bit more exhaustive than what stamping do with them (at
497 // least for node with many incoming and outgoing implications).
498 //
499 // TODO(user): Do some reduction using binary clauses. Note that only clause
500 // that never propagated since last round need to be checked for binary
501 // subsumption.
502
503 // Compute hash and mark literals.
504 uint64_t signature = 0;
505 marked.SparseClearAll();
506 for (const Literal l : clause->AsSpan()) {
507 marked.Set(l.Index());
508 signature |= (uint64_t{1} << (l.Variable().value() % 64));
509 }
510
511 // Look for clause that subsumes this one. Note that because we inspect
512 // all one watcher lists for the literals of this clause, if a clause is
513 // included inside this one, it must appear in one of these lists.
514 bool removed = false;
515 candidates_for_removal.clear();
516 const uint64_t mask = ~signature;
517 for (const Literal l : clause->AsSpan()) {
518 num_inspected_signatures += one_watcher[l].size();
519 for (const int i : one_watcher[l]) {
520 if ((mask & signatures[i]) != 0) continue;
521
522 bool subsumed = true;
523 bool stengthen = true;
524 LiteralIndex to_remove = kNoLiteralIndex;
525 num_inspected_literals += clauses[i]->size();
526 for (const Literal o : clauses[i]->AsSpan()) {
527 if (!marked[o]) {
528 subsumed = false;
529 if (to_remove == kNoLiteralIndex && marked[o.NegatedIndex()]) {
530 to_remove = o.NegatedIndex();
531 } else {
532 stengthen = false;
533 break;
534 }
535 }
536 }
537 if (subsumed) {
538 ++num_subsumed_clauses;
539 num_removed_literals += clause->size();
540 clause_manager_->InprocessingRemoveClause(clause);
541 removed = true;
542 break;
543 }
544 if (stengthen) {
545 CHECK_NE(kNoLiteralIndex, to_remove);
546 candidates_for_removal.push_back(Literal(to_remove));
547 }
548 }
549 if (removed) break;
550 }
551 if (removed) continue;
552
553 // For strengthenning we also need to check the negative watcher lists.
554 for (const Literal l : clause->AsSpan()) {
555 num_inspected_signatures += one_watcher[l.NegatedIndex()].size();
556 for (const int i : one_watcher[l.NegatedIndex()]) {
557 if ((mask & signatures[i]) != 0) continue;
558
559 bool stengthen = true;
560 num_inspected_literals += clauses[i]->size();
561 for (const Literal o : clauses[i]->AsSpan()) {
562 if (o == l.Negated()) continue;
563 if (!marked[o]) {
564 stengthen = false;
565 break;
566 }
567 }
568 if (stengthen) {
569 candidates_for_removal.push_back(l);
570 }
571 }
572 }
573
574 // Any literal here can be removed, but afterwards the other might not. For
575 // now we just remove the first one.
576 //
577 // TODO(user): remove first and see if other still removable. Alternatively
578 // use a "removed" marker and redo a check for each clause that simplifies
579 // this one? Or just remove the first one, and wait for next round.
580 if (!candidates_for_removal.empty()) {
581 new_clause.clear();
582 for (const Literal l : clause->AsSpan()) {
583 if (l == candidates_for_removal[0]) continue;
584 new_clause.push_back(l);
585 }
586 CHECK_EQ(new_clause.size() + 1, clause->size());
587
588 num_removed_literals += clause->size() - new_clause.size();
589 if (!clause_manager_->InprocessingRewriteClause(clause, new_clause)) {
590 return false;
591 }
592 if (clause->size() == 0) continue;
593
594 // Recompute signature.
595 signature = 0;
596 for (const Literal l : clause->AsSpan()) {
597 signature |= (uint64_t{1} << (l.Variable().value() % 64));
598 }
599 }
600
601 // Register one literal to watch. Any literal works, but we choose the
602 // smallest list.
603 //
604 // TODO(user): No need to add this clause if we know it cannot subsume
605 // any new clause since last round. i.e. unchanged clause that do not
606 // contains any literals of newly added clause do not need to be added
607 // here. We can track two bitset in LiteralWatchers via a register
608 // mechanism:
609 // - literal of newly watched clauses since last clear.
610 // - literal of reduced clauses since last clear.
611 //
612 // Important: we can only use this clause to subsume/strenghten others if
613 // it cannot be deleted later.
614 if (!clause_manager_->IsRemovable(clause)) {
615 int min_size = std::numeric_limits<int32_t>::max();
616 LiteralIndex min_literal = kNoLiteralIndex;
617 for (const Literal l : clause->AsSpan()) {
618 if (one_watcher[l].size() < min_size) {
619 min_size = one_watcher[l].size();
620 min_literal = l.Index();
621 }
622 }
623
624 // TODO(user): We could/should sort the literal in this clause by
625 // using literals that appear in a small number of clauses first so that
626 // we maximize the chance of early abort in the critical loops above.
627 //
628 // TODO(user): We could also move the watched literal first so we always
629 // skip it.

631 one_watcher[min_literal].push_back(clause_index);
632 }

637
638 // TODO(user): tune the deterministic time.
639 const double dtime = static_cast<double>(num_inspected_signatures) * 1e-8 +
640 static_cast<double>(num_inspected_literals) * 5e-9;
641 time_limit_->AdvanceDeterministicTime(dtime);
642 LOG_IF(INFO, log_info) << "Subsume. num_removed_literals: "
643 << num_removed_literals
644 << " num_subsumed: " << num_subsumed_clauses
645 << " dtime: " << dtime
646 << " wtime: " << wall_timer.Get();
647 return true;

708void StampingSimplifier::SampleTreeAndFillParent() {
709 const int size = implication_graph_->literal_size();
710 CHECK(implication_graph_->IsDag()); // so we don't have cycle.
711 parents_.resize(size);
712 for (LiteralIndex i(0); i < size; ++i) {
713 parents_[i] = i; // default.
714 if (implication_graph_->IsRedundant(Literal(i))) continue;
715 if (assignment_.LiteralIsAssigned(Literal(i))) continue;
716
717 // TODO(user): Better algo to not select redundant parent.
718 //
719 // TODO(user): if parents_[x] = y, try not to have parents_[not(y)] = not(x)
720 // because this is not as useful for the simplification power.
721 //
722 // TODO(user): More generally, we could sample a parent while probing so
723 // that we consider all hyper binary implications (in the case we don't add
724 // them to the implication graph already).
725 for (int num_tries = 0; num_tries < 10; ++num_tries) {
726 // We look for a random lit that implies i. For that we take a random
727 // literal in the direct implications of not(i) and reverse it.
728 const LiteralIndex index =
729 implication_graph_->RandomImpliedLiteral(Literal(i).Negated());
730 if (index == kNoLiteralIndex) break;
731
732 const Literal candidate = Literal(index).Negated();
733 if (implication_graph_->IsRedundant(candidate)) continue;
734 if (i == candidate.Index()) continue;
735
736 // We found an interesting parent.
737 parents_[i] = candidate.Index();
738 break;
739 }
740 }

743bool StampingSimplifier::ComputeStamps() {
744 const int size = implication_graph_->literal_size();
745
746 // Compute sizes.
747 sizes_.assign(size, 0);
748 for (LiteralIndex i(0); i < size; ++i) {
749 if (parents_[i] == i) continue; // leaf.
750 sizes_[parents_[i]]++;
751 }
752
753 // Compute starts in the children_ vector for each node.
754 starts_.resize(size + 1); // We use a sentinel.
755 starts_[LiteralIndex(0)] = 0;
756 for (LiteralIndex i(1); i <= size; ++i) {
757 starts_[i] = starts_[i - 1] + sizes_[i - 1];
758 }
759
760 // Fill children. This messes up starts_.
761 children_.resize(size);
762 for (LiteralIndex i(0); i < size; ++i) {
763 if (parents_[i] == i) continue; // leaf.
764 children_[starts_[parents_[i]]++] = i;
765 }
766
767 // Reset starts to correct value.
768 for (LiteralIndex i(0); i < size; ++i) {
769 starts_[i] -= sizes_[i];
770 }
771
772 if (DEBUG_MODE) {
773 CHECK_EQ(starts_[LiteralIndex(0)], 0);
774 for (LiteralIndex i(1); i <= size; ++i) {
775 CHECK_EQ(starts_[i], starts_[i - 1] + sizes_[i - 1]);
776 }
777 }
778
779 // Perform a DFS from each root to compute the stamps.
780 int64_t stamp = 0;
781 first_stamps_.resize(size);
782 last_stamps_.resize(size);
783 marked_.assign(size, false);
784 for (LiteralIndex i(0); i < size; ++i) {
785 if (parents_[i] != i) continue; // Not a root.
786 DCHECK(!marked_[i]);
787 const LiteralIndex tree_root = i;
788 dfs_stack_.push_back(i);
789 while (!dfs_stack_.empty()) {
790 const LiteralIndex top = dfs_stack_.back();
791 if (marked_[top]) {
792 dfs_stack_.pop_back();
793 last_stamps_[top] = stamp++;
794 continue;
795 }
796 first_stamps_[top] = stamp++;
797 marked_[top] = true;
798
799 // Failed literal detection. If the negation of top is in the same
800 // tree, we can fix the LCA of top and its negation to false.
801 if (marked_[Literal(top).NegatedIndex()] &&
802 first_stamps_[Literal(top).NegatedIndex()] >=
803 first_stamps_[tree_root]) {
804 // Find the LCA.
805 const int first_stamp = first_stamps_[Literal(top).NegatedIndex()];
806 LiteralIndex lca = top;
807 while (first_stamps_[lca] > first_stamp) {
808 lca = parents_[lca];
809 }
810 ++num_fixed_;
811 if (!clause_manager_->InprocessingFixLiteral(Literal(lca).Negated())) {
812 return false;
813 }
814 }
815
816 const int end = starts_[top + 1]; // Ok with sentinel.
817 for (int j = starts_[top]; j < end; ++j) {
818 DCHECK_NE(top, children_[j]); // We removed leaf self-loop.
819 DCHECK(!marked_[children_[j]]); // This is a tree.
820 dfs_stack_.push_back(children_[j]);
821 }
822 }
823 }
824 DCHECK_EQ(stamp, 2 * size);
825 return true;
826}

828bool StampingSimplifier::ProcessClauses() {
829 struct Entry {
830 int i; // Index in the clause.
831 bool is_negated; // Correspond to clause[i] or clause[i].Negated();
832 int start; // Note that all start stamps are different.
833 int end;
834 bool operator<(const Entry& o) const { return start < o.start; }
835 };
836 std::vector<int> to_remove;
837 std::vector<Literal> new_clause;
838 std::vector<Entry> entries;
839 clause_manager_->DeleteRemovedClauses();
840 clause_manager_->DetachAllClauses();
841 for (SatClause* clause : clause_manager_->AllClausesInCreationOrder()) {
842 const auto span = clause->AsSpan();
843 if (span.empty()) continue;
844
845 // Note that we might fix literal as we perform the loop here, so we do
846 // need to deal with them.
847 //
848 // For a and b in the clause, if not(a) => b is present, then the clause is
849 // subsumed. If a => b, then a can be removed, and if not(a) => not(b) then
850 // b can be removed. Nothing can be done if a => not(b).
851 entries.clear();
852 for (int i = 0; i < span.size(); ++i) {
853 if (assignment_.LiteralIsTrue(span[i])) {
854 clause_manager_->InprocessingRemoveClause(clause);
855 break;
856 }
857 if (assignment_.LiteralIsFalse(span[i])) continue;
858 entries.push_back(
859 {i, false, first_stamps_[span[i]], last_stamps_[span[i]]});
860 entries.push_back({i, true, first_stamps_[span[i].NegatedIndex()],
861 last_stamps_[span[i].NegatedIndex()]});
862 }
863 if (clause->IsRemoved()) continue;
864
865 // The sort should be dominant.
866 if (!entries.empty()) {
867 const double n = static_cast<double>(entries.size());
868 dtime_ += 1.5e-8 * n * std::log(n);
869 std::sort(entries.begin(), entries.end());
870 }
871
872 Entry top_entry;
873 top_entry.end = -1; // Sentinel.
874 to_remove.clear();
875 for (const Entry& e : entries) {
876 if (e.end < top_entry.end) {
877 // We found an implication: top_entry => this entry.
878 const Literal lhs = top_entry.is_negated ? span[top_entry.i].Negated()
879 : span[top_entry.i];
880 const Literal rhs = e.is_negated ? span[e.i].Negated() : span[e.i];
881 DCHECK(ImplicationIsInTree(lhs, rhs));
882
883 if (top_entry.is_negated != e.is_negated) {
884 // Failed literal?
885 if (top_entry.i == e.i) {
886 ++num_fixed_;
887 if (top_entry.is_negated) {
888 // not(span[i]) => span[i] so span[i] true.
889 // And the clause is satisfied (so we count as as subsumed).
890 if (!clause_manager_->InprocessingFixLiteral(span[top_entry.i])) {
891 return false;
892 }
893 } else {
894 // span[i] => not(span[i]) so span[i] false.
895 if (!clause_manager_->InprocessingFixLiteral(
896 span[top_entry.i].Negated())) {
897 return false;
898 }
899 to_remove.push_back(top_entry.i);
900 continue;
901 }
902 }
903
904 // not(a) => b : subsumption.
905 // a => not(b), we cannot deduce anything, but it might make sense
906 // to see if not(b) implies anything instead of just keeping
907 // top_entry. See TODO below.
908 if (top_entry.is_negated) {
909 num_subsumed_clauses_++;
910 clause_manager_->InprocessingRemoveClause(clause);
911 break;
912 }
913 } else {
914 CHECK_NE(top_entry.i, e.i);
915 if (top_entry.is_negated) {
916 // not(a) => not(b), we can remove b.
917 to_remove.push_back(e.i);
918 } else {
919 // a => b, we can remove a.
920 //
921 // TODO(user): Note that it is okay to still use top_entry, but we
922 // might miss the removal of b if b => c. Also the paper do things
923 // differently. Make sure we don't miss any simplification
924 // opportunites by not changing top_entry. Same in the other
925 // branches.
926 to_remove.push_back(top_entry.i);
927 }
928 }
929 } else {
930 top_entry = e;
931 }
932 }
933
934 if (clause->IsRemoved()) continue;
935
936 // Strengthen the clause.
937 if (!to_remove.empty() || entries.size() < span.size()) {
938 new_clause.clear();
939 gtl::STLSortAndRemoveDuplicates(&to_remove);
940 int to_remove_index = 0;
941 for (int i = 0; i < span.size(); ++i) {
942 if (to_remove_index < to_remove.size() &&
943 i == to_remove[to_remove_index]) {
944 ++to_remove_index;
945 continue;
946 }
947 if (assignment_.LiteralIsTrue(span[i])) {
948 clause_manager_->InprocessingRemoveClause(clause);
949 continue;
950 }
951 if (assignment_.LiteralIsFalse(span[i])) continue;
952 new_clause.push_back(span[i]);
953 }
954 num_removed_literals_ += span.size() - new_clause.size();
955 if (!clause_manager_->InprocessingRewriteClause(clause, new_clause)) {
956 return false;
957 }
958 }
959 }
960 return true;
961}

1029void BlockedClauseSimplifier::ProcessLiteral(Literal current_literal) {
1030 if (assignment_.LiteralIsAssigned(current_literal)) return;
1031 if (implication_graph_->IsRemoved(current_literal)) return;
1032
1033 // We want to check first that this clause will resolve to trivial clause with
1034 // all binary containing not(current_literal). So mark all literal l so that
1035 // current_literal => l.
1036 //
1037 // TODO(user): We do not need to redo that each time we reprocess
1038 // current_literal.
1039 //
1040 // TODO(user): Ignore redundant literals. That might require pushing
1041 // equivalence to the postsolve stack though. Better to simply remove
1042 // these equivalence if we are allowed to and update the postsolve then.
1043 //
1044 // TODO(user): Make this work in the presence of at most ones.
1045 int num_binary = 0;
1046 const std::vector<Literal>& implications =
1047 implication_graph_->DirectImplications(current_literal);
1048 for (const Literal l : implications) {
1049 if (l == current_literal) continue;
1050 ++num_binary;
1051 marked_[l] = true;
1052 }
1053
1054 // TODO(user): We could also mark a small clause containing
1055 // current_literal.Negated(), and make sure we only include in
1056 // clauses_to_process clauses that resolve trivially with that clause.
1057 std::vector<ClauseIndex> clauses_to_process;
1058 for (const ClauseIndex i : literal_to_clauses_[current_literal]) {
1059 if (clauses_[i]->IsRemoved()) continue;
1060
1061 // Blocked with respect to binary clause only? all marked binary should have
1062 // their negation in clause.
1063 //
1064 // TODO(user): Abort if size left is too small.
1065 if (num_binary > 0) {
1066 if (clauses_[i]->size() <= num_binary) continue;
1067 int num_with_negation_marked = 0;
1068 for (const Literal l : clauses_[i]->AsSpan()) {
1069 if (l == current_literal) continue;
1070 if (marked_[l.NegatedIndex()]) {
1071 ++num_with_negation_marked;
1072 }
1073 }
1074 num_inspected_literals_ += clauses_[i]->size();
1075 if (num_with_negation_marked < num_binary) continue;
1076 }
1077 clauses_to_process.push_back(i);
1078 }
1079
1080 // Clear marked.
1081 for (const Literal l : implications) {
1082 marked_[l] = false;
1083 }
1084
1085 // TODO(user): There is a possible optimization: If we mark all literals of
1086 // all the clause to process, we can check that each clause containing
1087 // current_literal.Negated() contains at least one of these literal negated
1088 // other than current_literal. Otherwise none of the clause are blocked.
1089 //
1090 // TODO(user): If a clause cannot be blocked because of another clause, then
1091 // when we call ProcessLiteral(current_literal.Negated()) we can skip some
1092 // inspection.
1093 for (const ClauseIndex i : clauses_to_process) {
1094 const auto c = clauses_[i]->AsSpan();
1095 if (ClauseIsBlocked(current_literal, c)) {
1096 // Reprocess all clauses that have a negated literal in this one as
1097 // some might be blocked now.
1098 //
1099 // TODO(user): Maybe we can remember for which (literal, clause) pair this
1100 // was used as a certificate for "not-blocked" and just reprocess those,
1101 // but it might be memory intensive.
1102 for (const Literal l : c) {
1103 if (!in_queue_[l.NegatedIndex()]) {
1104 in_queue_[l.NegatedIndex()] = true;
1105 queue_.push_back(l.Negated());
1106 }
1107 }
1108
1109 // Add the clause to the postsolving set.
1110 postsolve_->AddClauseWithSpecialLiteral(current_literal, c);
1111
1112 // We can remove a blocked clause.
1113 ++num_blocked_clauses_;
1114 clause_manager_->InprocessingRemoveClause(clauses_[i]);
1115 }
1116 }
1117}

1119// Note that this assume that the binary clauses have already been checked.
1120bool BlockedClauseSimplifier::ClauseIsBlocked(
1121 Literal current_literal, absl::Span<const Literal> clause) {
1122 bool is_blocked = true;
1123 for (const Literal l : clause) marked_[l] = true;
1124
1125 // TODO(user): For faster reprocessing of the same literal, we should move
1126 // all clauses that are used in a non-blocked certificate first in the list.
1127 for (const ClauseIndex i :
1128 literal_to_clauses_[current_literal.NegatedIndex()]) {
1129 if (clauses_[i]->IsRemoved()) continue;
1130 bool some_marked = false;

1132 // TODO(user): we can be faster here by only updating it at the end?
1133 ++num_inspected_literals_;
1134
1135 if (l == current_literal.Negated()) continue;
1136 if (marked_[l.NegatedIndex()]) {
1137 some_marked = true;
1138 break;
1139 }
1140 }
1141 if (!some_marked) {
1142 is_blocked = false;
1143 break;
1144 }
1145 }
1146
1147 for (const Literal l : clause) marked_[l] = false;
1148 return is_blocked;

1151bool BoundedVariableElimination::DoOneRound(bool log_info) {
1152 WallTimer wall_timer;
1153 wall_timer.Start();
1154
1155 dtime_ = 0.0;
1156 num_inspected_literals_ = 0;
1157 num_eliminated_variables_ = 0;
1158 num_literals_diff_ = 0;
1159 num_clauses_diff_ = 0;
1160 num_simplifications_ = 0;
1161 num_blocked_clauses_ = 0;
1162
1163 clauses_.clear();
1164 clause_manager_->DeleteRemovedClauses();
1165 clause_manager_->DetachAllClauses();
1166 for (SatClause* c : clause_manager_->AllClausesInCreationOrder()) {
1167 // We ignore redundant clause. This shouldn't cause any validity issue.
1168 // TODO(user): but we shouldn't keep clauses containing removed literals.
1169 // It is still valid to do so, but it should be less efficient.
1170 if (clause_manager_->IsRemovable(c)) continue;
1171
1172 clauses_.push_back(c);
1173 }
1174 const int num_literals = clause_manager_->literal_size();
1175 const int num_variables = num_literals / 2;
1176
1177 literal_to_clauses_.clear();
1178 literal_to_clauses_.resize(num_literals);
1179 literal_to_num_clauses_.assign(num_literals, 0);
1180 for (ClauseIndex i(0); i < clauses_.size(); ++i) {
1181 for (const Literal l : clauses_[i]->AsSpan()) {
1182 literal_to_clauses_[l].push_back(i);
1183 literal_to_num_clauses_[l]++;
1184 }
1185 num_inspected_literals_ += clauses_[i]->size();
1186 }
1187
1188 const int saved_trail_index = trail_->Index();
1189 propagation_index_ = trail_->Index();
1190
1191 need_to_be_updated_.clear();
1192 in_need_to_be_updated_.resize(num_variables);
1193 queue_.Reserve(num_variables);
1194 for (BooleanVariable v(0); v < num_variables; ++v) {
1195 if (assignment_.VariableIsAssigned(v)) continue;
1196 UpdatePriorityQueue(v);
1197 }
1198
1199 marked_.resize(num_literals);
1200 DCHECK(
1201 std::all_of(marked_.begin(), marked_.end(), [](bool b) { return !b; }));
1202
1203 // TODO(user): add a local dtime limit for the corner case where this take too
1204 // much time. We can adapt the limit depending on how much we want to spend on
1205 // inprocessing.
1206 while (!time_limit_->LimitReached() && !queue_.IsEmpty()) {
1207 const BooleanVariable top = queue_.Top().var;
1208 queue_.Pop();
1209
1210 // Make sure we fix variables first if needed. Note that because new binary
1211 // clause might appear when we fix variables, we need a loop here.
1212 //
1213 // TODO(user): we might also find new equivalent variable l => var => l
1214 // here, but for now we ignore those.
1215 bool is_unsat = false;
1216 if (!Propagate()) return false;
1217 while (implication_graph_->FindFailedLiteralAroundVar(top, &is_unsat)) {
1218 if (!Propagate()) return false;
1219 }
1220 if (is_unsat) return false;
1221
1222 if (!CrossProduct(top)) return false;
1223
1224 for (const BooleanVariable v : need_to_be_updated_) {
1225 in_need_to_be_updated_[v] = false;
1226
1227 // Currently we never re-add top if we just processed it.
1228 if (v != top) UpdatePriorityQueue(v);
1229 }
1230 in_need_to_be_updated_.clear();
1231 need_to_be_updated_.clear();
1232 }
1233
1234 if (!Propagate()) return false;
1235 implication_graph_->CleanupAllRemovedAndFixedVariables();
1236
1237 // Remove all redundant clause containing a removed literal. This avoid to
1238 // re-introduce a removed literal via conflict learning.
1239 for (SatClause* c : clause_manager_->AllClausesInCreationOrder()) {
1240 bool remove = false;
1241 for (const Literal l : c->AsSpan()) {
1242 if (implication_graph_->IsRemoved(l)) {
1243 remove = true;
1244 break;
1245 }
1246 }
1247 if (remove) clause_manager_->InprocessingRemoveClause(c);
1248 }
1249
1250 // Release some memory.
1251 literal_to_clauses_.clear();
1252 literal_to_num_clauses_.clear();
1253
1254 dtime_ += 1e-8 * num_inspected_literals_;
1255 time_limit_->AdvanceDeterministicTime(dtime_);
1256 log_info |= VLOG_IS_ON(1);
1257 LOG_IF(INFO, log_info) << "BVE." << " num_fixed: "

1298 for (; propagation_index_ < trail_->Index(); ++propagation_index_) {
1299 // Make sure we always propagate the binary clauses first.
1300 if (!implication_graph_->Propagate(trail_)) return false;
1301
1302 const Literal l = (*trail_)[propagation_index_];
1303 for (const ClauseIndex index : literal_to_clauses_[l]) {
1304 if (clauses_[index]->IsRemoved()) continue;
1305 num_clauses_diff_--;
1306 num_literals_diff_ -= clauses_[index]->size();
1307 clause_manager_->InprocessingRemoveClause(clauses_[index]);
1308 }
1309 literal_to_clauses_[l].clear();
1310 for (const ClauseIndex index : literal_to_clauses_[l.NegatedIndex()]) {
1311 if (clauses_[index]->IsRemoved()) continue;
1312 if (!RemoveLiteralFromClause(l.Negated(), clauses_[index])) return false;
1313 }
1314 literal_to_clauses_[l.NegatedIndex()].clear();
1315 }
1316 return true;

1388template <bool score_only, bool with_binary_only>
1389bool BoundedVariableElimination::ResolveAllClauseContaining(Literal lit) {
1390 const int clause_weight = parameters_.presolve_bve_clause_weight();
1391
1392 const std::vector<Literal>& implications =
1393 implication_graph_->DirectImplications(lit);
1394 auto& clause_containing_lit = literal_to_clauses_[lit];
1395 for (int i = 0; i < clause_containing_lit.size(); ++i) {
1396 const ClauseIndex clause_index = clause_containing_lit[i];
1397 const auto clause = clauses_[clause_index]->AsSpan();
1398 if (clause.empty()) continue;
1399
1400 if (!score_only) resolvant_.clear();
1401 for (const Literal l : clause) {
1402 if (!score_only && l != lit) resolvant_.push_back(l);
1403 marked_[l] = true;
1404 }
1405 DCHECK(marked_[lit]);
1406 num_inspected_literals_ += clause.size() + implications.size();
1407
1408 // If this is true, then "clause" is subsumed by one of its resolvant and we
1409 // can just remove lit from it. Then it doesn't need to be acounted at all.
1410 bool clause_can_be_simplified = false;
1411 const int64_t saved_score = new_score_;
1412
1413 // Resolution with binary clauses.
1414 for (const Literal l : implications) {
1415 CHECK_NE(l, lit);
1416 if (marked_[l.NegatedIndex()]) continue; // trivial.
1417 if (marked_[l]) {
1418 clause_can_be_simplified = true;
1419 break;
1420 } else {
1421 if (score_only) {
1422 new_score_ += clause_weight + clause.size();
1423 } else {
1424 resolvant_.push_back(l);
1425 AddClause(resolvant_);
1426 resolvant_.pop_back();
1427 }
1428 }
1429 }
1430
1431 // Resolution with non-binary clauses.
1432 if (!with_binary_only && !clause_can_be_simplified) {
1433 auto& clause_containing_not_lit = literal_to_clauses_[lit.NegatedIndex()];
1434 for (int j = 0; j < clause_containing_not_lit.size(); ++j) {
1435 if (score_only && new_score_ > score_threshold_) break;
1436 const ClauseIndex other_index = clause_containing_not_lit[j];
1437 const auto other = clauses_[other_index]->AsSpan();
1438 if (other.empty()) continue;
1439 bool trivial = false;
1440 int extra_size = 0;
1441 for (const Literal l : other) {
1442 // TODO(user): we can optimize this by updating it outside the loop.
1443 ++num_inspected_literals_;
1444 if (l == lit.Negated()) continue;
1445 if (marked_[l.NegatedIndex()]) {
1446 trivial = true;
1447 break;
1448 }
1449 if (!marked_[l]) {
1450 ++extra_size;
1451 if (!score_only) resolvant_.push_back(l);
1452 }
1453 }
1454 if (trivial) {
1455 if (!score_only) resolvant_.resize(resolvant_.size() - extra_size);
1456 continue;
1457 }
1458
1459 // If this is the case, the other clause is subsumed by the resolvant.
1460 // We can just remove not_lit from it and ignore it.
1461 if (score_only && clause.size() + extra_size <= other.size()) {
1462 // TODO(user): We should have an exact equality here, except if
1463 // presolve is off before the clause are added to the sat solver and
1464 // we have duplicate literals. The code should still work but it
1465 // wasn't written with that in mind nor tested like this, so we should
1466 // just enforce the invariant.
1467 if (false) DCHECK_EQ(clause.size() + extra_size, other.size());
1468 ++num_simplifications_;
1469
1470 // Note that we update the threshold since this clause was counted in
1471 // it.
1472 score_threshold_ -= clause_weight + other.size();
1473
1474 if (extra_size == 0) {
1475 // We have a double self-subsumption. We can just remove this
1476 // clause since it will be subsumed by the clause created in the
1477 // "clause_can_be_simplified" case below.
1478 DeleteClause(clauses_[other_index]);
1479 } else {
1480 if (!RemoveLiteralFromClause(lit.Negated(),
1481 clauses_[other_index])) {
1482 return false;
1483 }
1484 std::swap(clause_containing_not_lit[j],
1485 clause_containing_not_lit.back());
1486 clause_containing_not_lit.pop_back();
1487 --j; // Reprocess the new position.
1488 continue;
1489 }
1490 }
1491
1492 if (extra_size == 0) {
1493 clause_can_be_simplified = true;
1494 break;
1495 } else {
1496 if (score_only) {
1497 // Hack. We do not want to create long clauses during BVE.
1498 if (clause.size() - 1 + extra_size > 100) {
1499 new_score_ = score_threshold_ + 1;
1500 break;
1501 }
1502
1503 new_score_ += clause_weight + clause.size() - 1 + extra_size;
1504 } else {
1505 AddClause(resolvant_);
1506 resolvant_.resize(resolvant_.size() - extra_size);
1507 }
1508 }
1509 }
1510 }
1511
1512 // Note that we need to clear marked before aborting.
1513 for (const Literal l : clause) marked_[l] = false;
1514
1515 // In this case, we simplify and remove the clause from here.
1516 if (clause_can_be_simplified) {
1517 ++num_simplifications_;
1518
1519 // Note that we update the threshold as if this was simplified before.
1520 new_score_ = saved_score;
1521 score_threshold_ -= clause_weight + clause.size();
1522
1523 if (!RemoveLiteralFromClause(lit, clauses_[clause_index])) return false;
1524 std::swap(clause_containing_lit[i], clause_containing_lit.back());
1525 clause_containing_lit.pop_back();
1526 --i; // Reprocess the new position.
1527 }
1528
1529 if (score_only && new_score_ > score_threshold_) return true;
1530
1531 // When this happen, then the clause is blocked (i.e. all its resolvant are
1532 // trivial). So even if we do not actually perform the variable elimination,
1533 // we can still remove this clause. Note that we treat the score as if the
1534 // clause was removed before.
1535 //
1536 // Tricky: The detection only work if we didn't abort the computation above,
1537 // so we do that after the score_threshold_ check.
1538 //
1539 // TODO(user): Also detect blocked clause for not(lit)? It is not as cheap
1540 // though and require more code.
1541 if (score_only && !with_binary_only && !clause_can_be_simplified &&
1542 new_score_ == saved_score) {
1543 ++num_blocked_clauses_;
1544 score_threshold_ -= clause_weight + clause.size();
1545 postsolve_->AddClauseWithSpecialLiteral(lit, clause);
1546 DeleteClause(clauses_[clause_index]);
1547 }
1548 }
1549 return true;
1550}

1552bool BoundedVariableElimination::CrossProduct(BooleanVariable var) {
1553 if (assignment_.VariableIsAssigned(var)) return true;
1554
1555 const Literal lit(var, true);
1556 const Literal not_lit(var, false);
1557 DCHECK(!implication_graph_->IsRedundant(lit));
1558 {
1559 const int s1 = NumClausesContaining(lit);
1560 const int s2 = NumClausesContaining(not_lit);
1561 if (s1 == 0 && s2 == 0) return true;
1562 if (s1 > 0 && s2 == 0) {
1563 num_eliminated_variables_++;
1564 if (!clause_manager_->InprocessingFixLiteral(lit)) return false;
1565 DeleteAllClausesContaining(lit);
1566 return true;
1567 }
1568 if (s1 == 0 && s2 > 0) {
1569 num_eliminated_variables_++;
1570 if (!clause_manager_->InprocessingFixLiteral(not_lit)) return false;
1571 DeleteAllClausesContaining(not_lit);
1572 return true;
1573 }
1574
1575 // Heuristic. Abort if the work required to decide if var should be removed
1576 // seems to big.
1577 if (s1 > 1 && s2 > 1 && s1 * s2 > parameters_.presolve_bve_threshold()) {
1578 return true;
1579 }
1580 }
1581
1582 // TODO(user): swap lit and not_lit for speed? it is unclear if we prefer
1583 // to minimize the number of clause containing lit or not_lit though. Also,
1584 // we might want to alternate since we also detect blocked clause containing
1585 // lit, but don't do it for not_lit.
1586
1587 // Compute the current score.
1588 // TODO(user): cleanup the list lazily at the same time?
1589 int64_t score = 0;
1590 const int clause_weight = parameters_.presolve_bve_clause_weight();
1591 score +=
1592 implication_graph_->DirectImplications(lit).size() * (clause_weight + 2);
1593 score += implication_graph_->DirectImplications(not_lit).size() *
1594 (clause_weight + 2);
1595 for (const ClauseIndex i : literal_to_clauses_[lit]) {
1596 const auto c = clauses_[i]->AsSpan();
1597 if (!c.empty()) score += clause_weight + c.size();
1598 }
1599 for (const ClauseIndex i : literal_to_clauses_[not_lit]) {
1600 const auto c = clauses_[i]->AsSpan();
1601 if (!c.empty()) score += clause_weight + c.size();
1602 }
1603
1604 // Compute the new score after BVE.
1605 // Abort as soon as it crosses the threshold.
1606 //
1607 // TODO(user): Experiment with leaving the implications graph as is. This will
1608 // not remove the variable completely, but it seems interesting since after
1609 // equivalent variable removal and failed literal probing, the cross product
1610 // of the implication always add a quadratic number of implication, except if
1611 // the in (or out) degree is zero or one.
1612 score_threshold_ = score;
1613 new_score_ = implication_graph_->NumImplicationOnVariableRemoval(var) *
1614 (clause_weight + 2);
1615 if (new_score_ > score_threshold_) return true;
1616 if (!ResolveAllClauseContaining</*score_only=*/true,
1617 /*with_binary_only=*/true>(not_lit)) {
1618 return false;
1619 }
1620 if (new_score_ > score_threshold_) return true;
1621 if (!ResolveAllClauseContaining</*score_only=*/true,
1622 /*with_binary_only=*/false>(lit)) {
1623 return false;
1624 }
1625 if (new_score_ > score_threshold_) return true;
1626
1627 // Perform BVE.
1628 if (new_score_ > 0) {
1629 if (!ResolveAllClauseContaining</*score_only=*/false,
1630 /*with_binary_only=*/false>(lit)) {
1631 return false;
1632 }
1633 if (!ResolveAllClauseContaining</*score_only=*/false,
1634 /*with_binary_only=*/true>(not_lit)) {
1635 return false;
1636 }
1637 }
1638
1639 ++num_eliminated_variables_;
1640 implication_graph_->RemoveBooleanVariable(var, &postsolve_->clauses);
1641 DeleteAllClausesContaining(lit);
1642 DeleteAllClausesContaining(not_lit);
1643 return true;
1644}
