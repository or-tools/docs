110void SharedStatTables::AddLpStat(absl::string_view name, Model* model) {
111 absl::MutexLock mutex_lock(&mutex_);
112
113 // Sum per component for the lp_table.
114 int64_t num_compo = 0;
115 int64_t num_iter = 0;
116 int64_t num_cut = 0;
117 int64_t num_optimal = 0;
118 int64_t num_dual_feasible = 0;
119 int64_t num_dual_unbounded = 0;
120
121 // Last dimension of the first component for the lp_dim_table_.
122 std::string dimension;
123
124 // Sum per component for the lp_manager_table_.
125 int64_t num_constraints = 0;
126 int64_t num_constraint_updates = 0;
127 int64_t num_simplifications = 0;
128 int64_t num_merged_constraints = 0;
129 int64_t num_shortened_constraints = 0;
130 int64_t num_split_constraints = 0;
131 int64_t num_coeff_strenghtening = 0;
132 int64_t num_cuts = 0;
133 int64_t num_add_cut_calls = 0;
134
135 // For the cut table.
136 absl::btree_map<std::string, int> type_to_num_cuts;
137
138 // For the debug table.
139 int64_t total_num_cut_propagations = 0;
140 int64_t total_num_eq_propagations = 0;
141 int64_t num_adjusts = 0;
142 int64_t num_cut_overflows = 0;
143 int64_t num_bad_cuts = 0;
144 int64_t num_scaling_issues = 0;
145
146 auto* lps = model->GetOrCreate<LinearProgrammingConstraintCollection>();
147 for (const auto* lp : *lps) {
148 const auto& manager = lp->constraint_manager();
149 ++num_compo;
150 num_iter += lp->total_num_simplex_iterations();
151 num_cut += manager.num_cuts();
152 const auto& num_solve_by_status = lp->num_solves_by_status();
153
154 const int optimal_as_int = static_cast<int>(glop::ProblemStatus::OPTIMAL);
155 if (optimal_as_int < num_solve_by_status.size()) {
156 num_optimal += num_solve_by_status[optimal_as_int];
157 }
158
159 const int dual_feasible_as_int =
160 static_cast<int>(glop::ProblemStatus::DUAL_FEASIBLE);
161 if (dual_feasible_as_int < num_solve_by_status.size()) {
162 num_dual_feasible += num_solve_by_status[dual_feasible_as_int];
163 }
164
165 const int dual_unbounded_as_int =
166 static_cast<int>(glop::ProblemStatus::DUAL_UNBOUNDED);
167 if (dual_unbounded_as_int < num_solve_by_status.size()) {
168 num_dual_unbounded += num_solve_by_status[dual_unbounded_as_int];
169 }
170
171 // In case of more than one component, we take the first one.
172 if (dimension.empty()) dimension = lp->DimensionString();
173
174 // Sum for the lp debug table.
175 total_num_cut_propagations += lp->total_num_cut_propagations();
176 total_num_eq_propagations += lp->total_num_eq_propagations();
177 num_adjusts += lp->num_adjusts();
178 num_cut_overflows += lp->num_cut_overflows();
179 num_bad_cuts += lp->num_bad_cuts();
180 num_scaling_issues += lp->num_scaling_issues();
181
182 // Sum for the lp manager table.
183 num_constraints += manager.num_constraints();
184 num_constraint_updates += manager.num_constraint_updates();
185 num_simplifications += manager.num_simplifications();
186 num_merged_constraints += manager.num_merged_constraints();
187 num_shortened_constraints += manager.num_shortened_constraints();
188 num_split_constraints += manager.num_split_constraints();
189 num_coeff_strenghtening += manager.num_coeff_strenghtening();
190 num_cuts += manager.num_cuts();
191 num_add_cut_calls += manager.num_add_cut_calls();
192
193 for (const auto& [cut_name, num] : manager.type_to_num_cuts()) {
194 type_to_num_cuts[cut_name] += num;
195 }
196 }
197 if (num_compo == 0) return;
198
199 lp_table_.push_back(
200 {FormatName(name), FormatCounter(num_compo), FormatCounter(num_iter),
201 FormatCounter(num_cut), FormatCounter(num_optimal),
202 FormatCounter(num_dual_feasible), FormatCounter(num_dual_unbounded)});
203
204 if (!dimension.empty()) {
205 lp_dim_table_.push_back({FormatName(name), dimension});
206 }
207
208 if (!type_to_num_cuts.empty()) {
209 lp_cut_table_.push_back({std::string(name), std::move(type_to_num_cuts)});
210 }
211
212 lp_debug_table_.push_back(
213 {FormatName(name), FormatCounter(total_num_cut_propagations),
214 FormatCounter(total_num_eq_propagations), FormatCounter(num_adjusts),
215 FormatCounter(num_cut_overflows), FormatCounter(num_bad_cuts),
216 FormatCounter(num_scaling_issues)});
217
218 lp_manager_table_.push_back({FormatName(name), FormatCounter(num_constraints),
219 FormatCounter(num_constraint_updates),
220 FormatCounter(num_simplifications),
221 FormatCounter(num_merged_constraints),
222 FormatCounter(num_shortened_constraints),
223 FormatCounter(num_split_constraints),
224 FormatCounter(num_coeff_strenghtening),
225 absl::StrCat(FormatCounter(num_cuts), "/",
226 FormatCounter(num_add_cut_calls))});
227}

229void SharedStatTables::AddLnsStat(absl::string_view name,
230 const NeighborhoodGenerator& generator) {
231 absl::MutexLock mutex_lock(&mutex_);
232 const double fully_solved_proportion =
233 static_cast<double>(generator.num_fully_solved_calls()) /
234 static_cast<double>(std::max(int64_t{1}, generator.num_calls()));
235 lns_table_.push_back(
236 {FormatName(name),
237 absl::StrCat(generator.num_improving_calls(), "/",
238 generator.num_calls()),
239 absl::StrFormat("%2.0f%%", 100 * fully_solved_proportion),
240 absl::StrFormat("%0.2f", generator.difficulty()),
241 absl::StrFormat("%0.2f", generator.deterministic_limit())});
242}

257void SharedStatTables::Display(SolverLogger* logger) {
258 if (!logger->LoggingIsEnabled()) return;
259
260 absl::MutexLock mutex_lock(&mutex_);
261 if (timing_table_.size() > 1) SOLVER_LOG(logger, FormatTable(timing_table_));
262 if (search_table_.size() > 1) SOLVER_LOG(logger, FormatTable(search_table_));
263 if (clauses_table_.size() > 1) {
264 SOLVER_LOG(logger, FormatTable(clauses_table_));
265 }
266
267 if (lp_table_.size() > 1) SOLVER_LOG(logger, FormatTable(lp_table_));
268 if (lp_dim_table_.size() > 1) SOLVER_LOG(logger, FormatTable(lp_dim_table_));
269 if (lp_debug_table_.size() > 1) {
270 SOLVER_LOG(logger, FormatTable(lp_debug_table_));
271 }
272 if (lp_manager_table_.size() > 1) {
273 SOLVER_LOG(logger, FormatTable(lp_manager_table_));
274 }
275
276 // Construct and generate lp cut table.
277 // Note that this one is transposed compared to the normal one.
278 if (!lp_cut_table_.empty()) {
279 // Collect all line names.
280 absl::btree_map<std::string, int> lines;
281 for (const auto& [_, map] : lp_cut_table_) {
282 for (const auto& [type_name, _] : map) {
283 lines[type_name] = 0;
284 }
285 }
286
287 // Create header and compute index.
288 std::vector<std::vector<std::string>> table;
289 int line_index = 1;
290 const int num_cols = lp_cut_table_.size() + 1;
291 table.push_back({"Lp Cut"});
292 table[0].resize(num_cols, "");
293 for (const auto& [type_name, _] : lines) {
294 lines[type_name] = line_index++;
295 table.push_back({absl::StrCat(type_name, ":")});
296 table.back().resize(num_cols, "-");
297 }
298
299 // Fill lines.
300 int col_index = 1;
301 for (const auto& [name, map] : lp_cut_table_) {
302 table[0][col_index] =
303 num_cols > 10 && name.size() > 6 ? name.substr(0, 6) : name;
304 for (const auto& [type_name, count] : map) {
305 table[lines[type_name]][col_index] = FormatCounter(count);
306 }
307 ++col_index;
308 }
309
310 if (table.size() > 1) SOLVER_LOG(logger, FormatTable(table));
311 }
312
313 if (lns_table_.size() > 1) SOLVER_LOG(logger, FormatTable(lns_table_));
314 if (ls_table_.size() > 1) SOLVER_LOG(logger, FormatTable(ls_table_));
315}
