35bool HasPrimalFeasibleSolution(const SolutionProto& solution) {
36 return solution.has_primal_solution() &&
37 solution.primal_solution().feasibility_status() ==
38 SOLUTION_STATUS_FEASIBLE;
39}
40
41bool HasPrimalFeasibleSolution(
42 const google::protobuf::RepeatedPtrField<SolutionProto>& solutions) {
43 // Assumes first solution is primal feasible if there is any primal solution.
44 return !solutions.empty() && HasPrimalFeasibleSolution(solutions.at(0));
45}
46
47bool HasDualFeasibleSolution(const SolutionProto& solution) {
48 return solution.has_dual_solution() &&
49 solution.dual_solution().feasibility_status() ==
50 SOLUTION_STATUS_FEASIBLE;
51}
52
53bool HasDualFeasibleSolution(
54 const google::protobuf::RepeatedPtrField<SolutionProto>& solutions) {
55 for (const auto& solution : solutions) {
56 if (HasDualFeasibleSolution(solution)) {
57 return true;
58 }
59 }
60 return false;
61}

64absl::Status ValidateSolutions(
65 const google::protobuf::RepeatedPtrField<SolutionProto>& solutions,
66 const ModelSolveParametersProto& parameters,
67 const ModelSummary& model_summary) {
68 // Validate individual solutions
69 for (int i = 0; i < solutions.size(); ++i) {
70 RETURN_IF_ERROR(ValidateSolution(solutions[i], parameters, model_summary))
71 << "invalid solutions[" << i << "]";
72 }
73
74 if (solutions.empty()) return absl::OkStatus();
75
76 // Validate solution order.
77 // TODO(b/204457524): check objective ordering when possible.
78 bool previous_primal_feasible = HasPrimalFeasibleSolution(solutions[0]);
79 bool previous_dual_feasible = HasDualFeasibleSolution(solutions[0]);
80 for (int i = 1; i < solutions.size(); ++i) {
81 const bool current_primal_feasible =
82 HasPrimalFeasibleSolution(solutions[i]);
83 const bool current_dual_feasible = HasDualFeasibleSolution(solutions[i]);
84 // Primal-feasible solutions must appear first.
85 if (current_primal_feasible && !previous_primal_feasible) {
86 return absl::InvalidArgumentError(
87 "primal solution ordering not satisfied");
88 }
89 // Dual-feasible solutions must appear first within the groups of
90 // primal-feasible and other solutions. Equivalently, a dual-feasible
91 // solution must be preceded by a dual-feasible solution, except when we
92 // switch from the group of primal-feasible solutions to the group of other
93 // solutions.
94 if (current_dual_feasible && !previous_dual_feasible) {
95 if (!(previous_primal_feasible && !current_primal_feasible)) {
96 return absl::InvalidArgumentError(
97 "dual solution ordering not satisfied");
98 }
99 }
100 previous_primal_feasible = current_primal_feasible;
101 previous_dual_feasible = current_dual_feasible;
102 }
103 return absl::OkStatus();
104}

107absl::Status RequireNoPrimalFeasibleSolution(const SolveResultProto& result) {
108 if (HasPrimalFeasibleSolution(result.solutions())) {
109 return absl::InvalidArgumentError(
110 "expected no primal feasible solution, but one was returned");
111 }

113 return absl::OkStatus();

126absl::Status CheckPrimalSolutionAndStatusConsistency(
127 const TerminationProto& termination,
128 const google::protobuf::RepeatedPtrField<SolutionProto>& solutions) {
129 if (termination.problem_status().primal_status() !=
130 FEASIBILITY_STATUS_FEASIBLE &&
131 HasPrimalFeasibleSolution(solutions)) {
132 return absl::InvalidArgumentError(
133 "primal feasibility status is not FEASIBILITY_STATUS_FEASIBLE, but "
134 "primal feasible solution is returned.");
135 }
136 return absl::OkStatus();
137}

139absl::Status CheckDualSolutionAndStatusConsistency(
140 const TerminationProto& termination,
141 const google::protobuf::RepeatedPtrField<SolutionProto>& solutions) {
142 if (termination.problem_status().dual_status() !=
143 FEASIBILITY_STATUS_FEASIBLE &&
144 HasDualFeasibleSolution(solutions)) {
145 return absl::InvalidArgumentError(
146 "dual feasibility status is not FEASIBILITY_STATUS_FEASIBLE, but "
147 "dual feasible solution is returned.");
148 }
149 return absl::OkStatus();
150}

155absl::Status ValidateSolveStatsTerminationEqualities(
156 const SolveResultProto& solve_result) {
157 const ObjectiveBoundsProto objective_bounds =
158 GetObjectiveBounds(solve_result);
159 const SolveStatsProto solve_stats = solve_result.solve_stats();
160 const ProblemStatusProto problem_status = GetProblemStatus(solve_result);
161 if (problem_status.primal_status() !=
162 solve_stats.problem_status().primal_status()) {
163 return util::InvalidArgumentErrorBuilder()
164 << problem_status.primal_status()
165 << " = termination.problem_status.primal_status != "
166 "solve_stats.problem_status.primal_status = "
167 << solve_stats.problem_status().primal_status();
168 }
169 if (problem_status.dual_status() !=
170 solve_stats.problem_status().dual_status()) {
171 return util::InvalidArgumentErrorBuilder()
172 << problem_status.dual_status()
173 << " = termination.problem_status.dual_status != "
174 "solve_stats.problem_status.dual_status = "
175 << solve_stats.problem_status().dual_status();
176 }
177 if (problem_status.primal_or_dual_infeasible() !=
178 solve_stats.problem_status().primal_or_dual_infeasible()) {
179 return util::InvalidArgumentErrorBuilder()
180 << problem_status.primal_or_dual_infeasible()
181 << " = termination.problem_status.primal_or_dual_infeasible != "
182 "solve_stats.problem_status.primal_or_dual_infeasible = "
183 << solve_stats.problem_status().primal_or_dual_infeasible();
184 }
185 if (objective_bounds.primal_bound() != solve_stats.best_primal_bound()) {
186 return util::InvalidArgumentErrorBuilder()
187 << objective_bounds.primal_bound()
188 << " = termination.objective_bounds.primal_bound != "
189 "solve_stats.best_primal_bound = "
190 << solve_stats.best_primal_bound();
191 }
192 if (objective_bounds.dual_bound() != solve_stats.best_dual_bound()) {
193 return util::InvalidArgumentErrorBuilder()
194 << objective_bounds.dual_bound()
195 << " = termination.objective_bounds.dual_bound != "
196 "solve_stats.best_dual_bound = "
197 << solve_stats.best_dual_bound();
198 }
199 return absl::OkStatus();
200}

204absl::Status ValidateResult(const SolveResultProto& result,
205 const ModelSolveParametersProto& parameters,
206 const ModelSummary& model_summary) {
207 // TODO(b/290091715): Remove once problem_status and objective bounds are
208 // removed from solve_stats and their presence is guaranteed in termination.
209 RETURN_IF_ERROR(ValidateSolveStatsTerminationEqualities(result));
210 // TODO(b/290091715): Replace by
211 // TerminationProto termination = result.termination();
212 // once problem_status and objective bounds are removed from solve_stats and
213 // their presence is guaranteed in termination.
214 TerminationProto termination = result.termination();
215 *termination.mutable_objective_bounds() = GetObjectiveBounds(result);
216 *termination.mutable_problem_status() = GetProblemStatus(result);
217
218 RETURN_IF_ERROR(ValidateTermination(termination, model_summary.maximize));
219 RETURN_IF_ERROR(ValidateSolveStats(result.solve_stats()));
220 RETURN_IF_ERROR(
221 ValidateSolutions(result.solutions(), parameters, model_summary));
222
223 if (termination.reason() == TERMINATION_REASON_OPTIMAL ||
224 termination.reason() == TERMINATION_REASON_FEASIBLE) {
225 RETURN_IF_ERROR(CheckHasPrimalSolution(result))
226 << "inconsistent termination reason "
227 << ProtoEnumToString(termination.reason());
228 }
229 if (termination.reason() == TERMINATION_REASON_NO_SOLUTION_FOUND) {
230 RETURN_IF_ERROR(RequireNoPrimalFeasibleSolution(result))
231 << "inconsistent termination reason "
232 << ProtoEnumToString(termination.reason());
233 }
234
235 RETURN_IF_ERROR(CheckPrimalSolutionAndStatusConsistency(result.termination(),
236 result.solutions()));
237 RETURN_IF_ERROR(CheckDualSolutionAndStatusConsistency(result.termination(),
238 result.solutions()));
239
240 if (result.primal_rays_size() > 0 &&
241 termination.problem_status().dual_status() ==
242 FEASIBILITY_STATUS_FEASIBLE) {
243 return absl::InvalidArgumentError(
244 "termination.problem_status.dual_status = "
245 "FEASIBILITY_STATUS_FEASIBLE, "
246 "but a primal ray is returned");
247 }
248 for (int i = 0; i < result.primal_rays_size(); ++i) {
249 RETURN_IF_ERROR(ValidatePrimalRay(result.primal_rays(i),
250 parameters.variable_values_filter(),
251 model_summary))
252 << "Invalid primal_rays[" << i << "]";
253 }
254 if (result.dual_rays_size() > 0 &&
255 termination.problem_status().primal_status() ==
256 FEASIBILITY_STATUS_FEASIBLE) {
257 return absl::InvalidArgumentError(
258 "termination.problem_status.primal_status = "
259 "FEASIBILITY_STATUS_FEASIBLE, but a dual ray is returned");
260 }
261 for (int i = 0; i < result.dual_rays_size(); ++i) {
262 RETURN_IF_ERROR(
263 ValidateDualRay(result.dual_rays(i), parameters, model_summary))
264 << "Invalid dual_rays[" << i << "]";
265 }
266
267 return absl::OkStatus();
268}
