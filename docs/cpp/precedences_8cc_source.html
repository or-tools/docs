54bool PrecedencesPropagator::Propagate(Trail* trail) { return Propagate(); }

56bool PrecedencesPropagator::Propagate() {
57 while (propagation_trail_index_ < trail_->Index()) {
58 const Literal literal = (*trail_)[propagation_trail_index_++];
59 if (literal.Index() >= literal_to_new_impacted_arcs_.size()) continue;
60
61 // IMPORTANT: Because of the way Untrail() work, we need to add all the
62 // potential arcs before we can abort. It is why we iterate twice here.
63 for (const ArcIndex arc_index :
64 literal_to_new_impacted_arcs_[literal.Index()]) {
65 if (--arc_counts_[arc_index] == 0) {
66 const ArcInfo& arc = arcs_[arc_index];
67 impacted_arcs_[arc.tail_var].push_back(arc_index);
68 }
69 }
70
71 // Iterate again to check for a propagation and indirectly update
72 // modified_vars_.
73 for (const ArcIndex arc_index :
74 literal_to_new_impacted_arcs_[literal.Index()]) {
75 if (arc_counts_[arc_index] > 0) continue;
76 const ArcInfo& arc = arcs_[arc_index];
77 if (integer_trail_->IsCurrentlyIgnored(arc.head_var)) continue;
78 const IntegerValue new_head_lb =
79 integer_trail_->LowerBound(arc.tail_var) + ArcOffset(arc);
80 if (new_head_lb > integer_trail_->LowerBound(arc.head_var)) {
81 if (!EnqueueAndCheck(arc, new_head_lb, trail_)) return false;
82 }
83 }
84 }
85
86 // Do the actual propagation of the IntegerVariable bounds.
87 InitializeBFQueueWithModifiedNodes();
88 if (!BellmanFordTarjan(trail_)) return false;
89
90 // We can only test that no propagation is left if we didn't enqueue new
91 // literal in the presence of optional variables.
92 //
93 // TODO(user): Because of our code to deal with InPropagationLoop(), this is
94 // not always true. Find a cleaner way to DCHECK() while not failing in this
95 // corner case.
96 if (/*DISABLES CODE*/ (false) &&
97 propagation_trail_index_ == trail_->Index()) {
98 DCHECK(NoPropagationLeft(*trail_));
99 }
100
101 // Propagate the presence literals of the arcs that can't be added.
102 PropagateOptionalArcs(trail_);
103
104 // Clean-up modified_vars_ to do as little as possible on the next call.
105 modified_vars_.ClearAndResize(integer_trail_->NumIntegerVariables());
106 return true;
107}

109bool PrecedencesPropagator::PropagateOutgoingArcs(IntegerVariable var) {
110 CHECK_NE(var, kNoIntegerVariable);
111 if (var >= impacted_arcs_.size()) return true;
112 for (const ArcIndex arc_index : impacted_arcs_[var]) {
113 const ArcInfo& arc = arcs_[arc_index];
114 if (integer_trail_->IsCurrentlyIgnored(arc.head_var)) continue;
115 const IntegerValue new_head_lb =
116 integer_trail_->LowerBound(arc.tail_var) + ArcOffset(arc);
117 if (new_head_lb > integer_trail_->LowerBound(arc.head_var)) {
118 if (!EnqueueAndCheck(arc, new_head_lb, trail_)) return false;
119 }
120 }
121 return true;
122}

124void PrecedencesPropagator::Untrail(const Trail& trail, int trail_index) {
125 if (propagation_trail_index_ > trail_index) {
126 // This means that we already propagated all there is to propagate
127 // at the level trail_index, so we can safely clear modified_vars_ in case
128 // it wasn't already done.
129 modified_vars_.ClearAndResize(integer_trail_->NumIntegerVariables());
130 }
131 while (propagation_trail_index_ > trail_index) {
132 const Literal literal = trail[--propagation_trail_index_];
133 if (literal.Index() >= literal_to_new_impacted_arcs_.size()) continue;
134 for (const ArcIndex arc_index :
135 literal_to_new_impacted_arcs_[literal.Index()]) {
136 if (arc_counts_[arc_index]++ == 0) {
137 const ArcInfo& arc = arcs_[arc_index];
138 impacted_arcs_[arc.tail_var].pop_back();
139 }
140 }
141 }
142}

144// Instead of simply sorting the IntegerPrecedences returned by .var,
145// experiments showed that it is faster to regroup all the same .var "by hand"
146// by first computing how many times they appear and then apply the sorting
147// permutation.
148void PrecedencesPropagator::ComputePrecedences(
149 const std::vector<IntegerVariable>& vars,
150 std::vector<IntegerPrecedences>* output) {
151 tmp_sorted_vars_.clear();
152 tmp_precedences_.clear();
153 for (int index = 0; index < vars.size(); ++index) {
154 const IntegerVariable var = vars[index];
155 CHECK_NE(kNoIntegerVariable, var);
156 if (var >= impacted_arcs_.size()) continue;
157 for (const ArcIndex arc_index : impacted_arcs_[var]) {
158 const ArcInfo& arc = arcs_[arc_index];
159 if (integer_trail_->IsCurrentlyIgnored(arc.head_var)) continue;
160
161 IntegerValue offset = arc.offset;
162 if (arc.offset_var != kNoIntegerVariable) {
163 offset += integer_trail_->LowerBound(arc.offset_var);
164 }
165
166 // TODO(user): it seems better to ignore negative min offset as we will
167 // often have relation of the form interval_start >= interval_end -
168 // offset, and such relation are usually not useful. Revisit this in case
169 // we see problems where we can propagate more without this test.
170 if (offset < 0) continue;
171
172 if (var_to_degree_[arc.head_var] == 0) {
173 tmp_sorted_vars_.push_back(
174 {arc.head_var, integer_trail_->LowerBound(arc.head_var)});
175 } else {
176 // This "seen" mechanism is needed because we may have multi-arc and we
177 // don't want any duplicates in the "is_before" relation. Note that it
178 // works because var_to_last_index_ is reset by the var_to_degree_ == 0
179 // case.
180 if (var_to_last_index_[arc.head_var] == index) continue;
181 }
182 var_to_last_index_[arc.head_var] = index;
183 var_to_degree_[arc.head_var]++;
184 tmp_precedences_.push_back(
185 {index, arc.head_var, arc_index.value(), offset});
186 }
187 }
188
189 // This order is a topological order for the precedences relation order
190 // provided that all the offset between the involved IntegerVariable are
191 // positive.
192 //
193 // TODO(user): use an order that is always topological? This is not clear
194 // since it may be slower to compute and not worth it because the order below
195 // is more natural and may work better.
196 std::sort(tmp_sorted_vars_.begin(), tmp_sorted_vars_.end());
197
198 // Permute tmp_precedences_ into the output to put it in the correct order.
199 // For that we transform var_to_degree_ to point to the first position of
200 // each lbvar in the output vector.
201 int start = 0;
202 for (const SortedVar pair : tmp_sorted_vars_) {
203 const int degree = var_to_degree_[pair.var];
204 if (degree > 1) {
205 var_to_degree_[pair.var] = start;
206 start += degree;
207 } else {
208 // Optimization: we remove degree one relations.
209 var_to_degree_[pair.var] = -1;
210 }
211 }
212 output->resize(start);
213 for (const IntegerPrecedences& precedence : tmp_precedences_) {
214 if (var_to_degree_[precedence.var] < 0) continue;
215 (*output)[var_to_degree_[precedence.var]++] = precedence;
216 }
217
218 // Cleanup var_to_degree_, note that we don't need to clean
219 // var_to_last_index_.
220 for (const SortedVar pair : tmp_sorted_vars_) {
221 var_to_degree_[pair.var] = 0;
222 }
223}

225void PrecedencesPropagator::ComputeFullPrecedences(
226 bool call_compute_precedences, const std::vector<IntegerVariable>& vars,
227 std::vector<FullIntegerPrecedence>* output) {
228 output->clear();
229 DCHECK_EQ(trail_->CurrentDecisionLevel(), 0);
230
231 if (call_compute_precedences) {
232 std::vector<PrecedencesPropagator::IntegerPrecedences> before;
233 ComputePrecedences(vars, &before);
234
235 // Convert format.
236 const int size = before.size();
237 for (int i = 0; i < size;) {
238 FullIntegerPrecedence data;
239 data.var = before[i].var;
240 const IntegerVariable var = before[i].var;
241 DCHECK_NE(var, kNoIntegerVariable);
242 for (; i < size && before[i].var == var; ++i) {
243 data.indices.push_back(before[i].index);
244 data.offsets.push_back(before[i].offset);
245 }
246 output->push_back(std::move(data));
247 }
248 return;
249 }
250
251 // Get a topological order of the DAG formed by all the arcs that are present.
252 //
253 // TODO(user): This can fail if we don't have a DAG. We could just skip Bad
254 // edges instead, and have a sub-DAG as an heuristic. Or analyze the arc
255 // weight and make sure cycle are not an issue. We can also start with arcs
256 // with strictly positive weight.
257 //
258 // TODO(user): Only explore the sub-graph reachable from "vars".
259 const int num_nodes = integer_trail_->NumIntegerVariables().value();
260 DenseIntStableTopologicalSorter sorter(num_nodes);
261 for (const auto& arcs : impacted_arcs_) {
262 for (const ArcIndex arc_index : arcs) {
263 const ArcInfo& arc = arcs_[arc_index];
264 if (arc.tail_var == arc.head_var) continue;
265 if (integer_trail_->IsCurrentlyIgnored(arc.head_var)) continue;
266 sorter.AddEdge(arc.tail_var.value(), arc.head_var.value());
267 }
268 }
269 int next;
270 bool graph_has_cycle = false;
271 std::vector<IntegerVariable> topological_order;
272 while (sorter.GetNext(&next, &graph_has_cycle, nullptr)) {
273 topological_order.push_back(IntegerVariable(next));
274 if (graph_has_cycle) return;
275 }
276
277 // Compute all precedences.
278 // We loop over the node in topological order, and we maintain for all
279 // variable we encounter, the list of "to_consider" variables that are before.
280 //
281 // TODO(user): use vector of fixed size.
282 absl::flat_hash_set<IntegerVariable> is_interesting;
283 absl::flat_hash_set<IntegerVariable> to_consider(vars.begin(), vars.end());
284 absl::flat_hash_map<IntegerVariable,
285 absl::flat_hash_map<IntegerVariable, IntegerValue>>
286 vars_before_with_offset;
287 absl::flat_hash_map<IntegerVariable, IntegerValue> tail_map;
288 for (const IntegerVariable tail_var : topological_order) {
289 if (!to_consider.contains(tail_var) &&
290 !vars_before_with_offset.contains(tail_var)) {
291 continue;
292 }
293
294 // Update the relation for the variable that are after.
295 if (tail_var >= impacted_arcs_.size()) continue;
296
297 // We copy the data for tail_var here, because the pointer is not stable.
298 // TODO(user): optimize when needed.
299 tail_map.clear();
300 {
301 const auto it = vars_before_with_offset.find(tail_var);
302 if (it != vars_before_with_offset.end()) {
303 tail_map = it->second;
304 }
305 }
306
307 for (const ArcIndex arc_index : impacted_arcs_[tail_var]) {
308 const ArcInfo& arc = arcs_[arc_index];
309 if (arc.tail_var == arc.head_var) continue;
310 if (integer_trail_->IsCurrentlyIgnored(arc.head_var)) continue;
311 CHECK_EQ(arc.tail_var, tail_var);
312
313 // No need to create an empty entry in this case.
314 if (tail_map.empty() && !to_consider.contains(tail_var)) continue;
315
316 IntegerValue arc_offset = arc.offset;
317 if (arc.offset_var != kNoIntegerVariable) {
318 arc_offset += integer_trail_->LowerBound(arc.offset_var);
319 }
320
321 auto& to_update = vars_before_with_offset[arc.head_var];
322 for (const auto& [var_before, offset] : tail_map) {
323 if (!to_update.contains(var_before)) {
324 to_update[var_before] = arc_offset + offset;
325 } else {
326 to_update[var_before] =
327 std::max(arc_offset + offset, to_update[var_before]);
328 }
329 }
330 if (to_consider.contains(tail_var)) {
331 if (!to_update.contains(tail_var)) {
332 to_update[tail_var] = arc_offset;
333 } else {
334 to_update[tail_var] = std::max(arc_offset, to_update[tail_var]);
335 }
336 }
337
338 // Small filtering heuristic: if we have (before) < tail, and tail < head,
339 // we really do not need to list (before, tail) < head. We only need that
340 // if the list of variable before head contains some variable that are not
341 // already before tail.
342 if (to_update.size() > tail_map.size() + 1) {
343 is_interesting.insert(arc.head_var);
344 } else {
345 is_interesting.erase(arc.head_var);
346 }
347 }
348
349 // Extract the output for tail_var. Because of the topological ordering, the
350 // data for tail_var is already final now.
351 //
352 // TODO(user): Release the memory right away.
353 if (!is_interesting.contains(tail_var)) continue;
354 if (tail_map.size() == 1) continue;
355
356 FullIntegerPrecedence data;
357 data.var = tail_var;
358 IntegerValue min_offset = kMaxIntegerValue;
359 for (int i = 0; i < vars.size(); ++i) {
360 const auto offset_it = tail_map.find(vars[i]);
361 if (offset_it == tail_map.end()) continue;
362 data.indices.push_back(i);
363 data.offsets.push_back(offset_it->second);
364 min_offset = std::min(data.offsets.back(), min_offset);
365 }
366 output->push_back(std::move(data));
367 }
368}

370void PrecedencesPropagator::AddPrecedenceReason(
371 int arc_index, IntegerValue min_offset,
372 std::vector<Literal>* literal_reason,
373 std::vector<IntegerLiteral>* integer_reason) const {
374 const ArcInfo& arc = arcs_[ArcIndex(arc_index)];
375 for (const Literal l : arc.presence_literals) {
376 literal_reason->push_back(l.Negated());
377 }
378 if (arc.offset_var != kNoIntegerVariable) {
379 // Reason for ArcOffset(arc) to be >= min_offset.
380 integer_reason->push_back(IntegerLiteral::GreaterOrEqual(
381 arc.offset_var, min_offset - arc.offset));
382 }
383}

385void PrecedencesPropagator::AdjustSizeFor(IntegerVariable i) {
386 const int index = std::max(i.value(), NegationOf(i).value());
387 if (index >= impacted_arcs_.size()) {
388 // TODO(user): only watch lower bound of the relevant variable instead
389 // of watching everything in [0, max_index_of_variable_used_in_this_class).
390 for (IntegerVariable var(impacted_arcs_.size()); var <= index; ++var) {
391 watcher_->WatchLowerBound(var, watcher_id_);
392 }
393 impacted_arcs_.resize(index + 1);
394 impacted_potential_arcs_.resize(index + 1);
395 var_to_degree_.resize(index + 1);
396 var_to_last_index_.resize(index + 1);
397 }

400void PrecedencesPropagator::AddArc(
401 IntegerVariable tail, IntegerVariable head, IntegerValue offset,
402 IntegerVariable offset_var, absl::Span<const Literal> presence_literals) {
403 DCHECK_EQ(trail_->CurrentDecisionLevel(), 0);
404 AdjustSizeFor(tail);
405 AdjustSizeFor(head);
406 if (offset_var != kNoIntegerVariable) AdjustSizeFor(offset_var);
407
408 // This arc is present iff all the literals here are true.
409 absl::InlinedVector<Literal, 6> enforcement_literals;
410 {
411 for (const Literal l : presence_literals) {
412 enforcement_literals.push_back(l);
413 }
414 if (integer_trail_->IsOptional(tail)) {
415 enforcement_literals.push_back(
416 integer_trail_->IsIgnoredLiteral(tail).Negated());
417 }
418 if (integer_trail_->IsOptional(head)) {
419 enforcement_literals.push_back(
420 integer_trail_->IsIgnoredLiteral(head).Negated());
421 }
422 if (offset_var != kNoIntegerVariable &&
423 integer_trail_->IsOptional(offset_var)) {
424 enforcement_literals.push_back(
425 integer_trail_->IsIgnoredLiteral(offset_var).Negated());
426 }
427 gtl::STLSortAndRemoveDuplicates(&enforcement_literals);
428 int new_size = 0;
429 for (const Literal l : enforcement_literals) {
430 if (trail_->Assignment().LiteralIsTrue(Literal(l))) {
431 continue; // At true, ignore this literal.
432 } else if (trail_->Assignment().LiteralIsFalse(Literal(l))) {
433 return; // At false, ignore completely this arc.
434 }
435 enforcement_literals[new_size++] = l;
436 }
437 enforcement_literals.resize(new_size);
438 }
439
440 if (head == tail) {
441 // A self-arc is either plain SAT or plain UNSAT or it forces something on
442 // the given offset_var or presence_literal_index. In any case it could be
443 // presolved in something more efficient.
444 VLOG(1) << "Self arc! This could be presolved. "
445 << "var:" << tail << " offset:" << offset
446 << " offset_var:" << offset_var
447 << " conditioned_by:" << presence_literals;
448 }
449
450 // Remove the offset_var if it is fixed.
451 // TODO(user): We should also handle the case where tail or head is fixed.
452 if (offset_var != kNoIntegerVariable) {
453 const IntegerValue lb = integer_trail_->LowerBound(offset_var);
454 if (lb == integer_trail_->UpperBound(offset_var)) {
455 offset += lb;
456 offset_var = kNoIntegerVariable;
457 }
458 }
459
460 // Deal first with impacted_potential_arcs_/potential_arcs_.
461 if (!enforcement_literals.empty()) {
462 const OptionalArcIndex arc_index(potential_arcs_.size());
463 potential_arcs_.push_back(
464 {tail, head, offset, offset_var, enforcement_literals});
465 impacted_potential_arcs_[tail].push_back(arc_index);
466 impacted_potential_arcs_[NegationOf(head)].push_back(arc_index);
467 if (offset_var != kNoIntegerVariable) {
468 impacted_potential_arcs_[offset_var].push_back(arc_index);
469 }
470 }
471
472 // Now deal with impacted_arcs_/arcs_.
473 struct InternalArc {
474 IntegerVariable tail_var;
475 IntegerVariable head_var;
476 IntegerVariable offset_var;
477 };
478 std::vector<InternalArc> to_add;
479 if (offset_var == kNoIntegerVariable) {
480 // a + offset <= b and -b + offset <= -a
481 to_add.push_back({tail, head, kNoIntegerVariable});
482 to_add.push_back({NegationOf(head), NegationOf(tail), kNoIntegerVariable});
483 } else {
484 // tail (a) and offset_var (b) are symmetric, so we add:
485 // - a + b + offset <= c
486 to_add.push_back({tail, head, offset_var});
487 to_add.push_back({offset_var, head, tail});
488 // - a - c + offset <= -b
489 to_add.push_back({tail, NegationOf(offset_var), NegationOf(head)});
490 to_add.push_back({NegationOf(head), NegationOf(offset_var), tail});
491 // - b - c + offset <= -a
492 to_add.push_back({offset_var, NegationOf(tail), NegationOf(head)});
493 to_add.push_back({NegationOf(head), NegationOf(tail), offset_var});
494 }
495 for (const InternalArc a : to_add) {
496 // Since we add a new arc, we will need to consider its tail during the next
497 // propagation. Note that the size of modified_vars_ will be automatically
498 // updated when new integer variables are created since we register it with
499 // IntegerTrail in this class constructor.
500 //
501 // TODO(user): Adding arcs and then calling Untrail() before Propagate()
502 // will cause this mecanism to break. Find a more robust implementation.
503 //
504 // TODO(user): In some rare corner case, rescanning the whole list of arc
505 // leaving tail_var can make AddVar() have a quadratic complexity where it
506 // shouldn't. A better solution would be to see if this new arc currently
507 // propagate something, and if it does, just update the lower bound of
508 // a.head_var and let the normal "is modified" mecanism handle any eventual
509 // follow up propagations.
510 modified_vars_.Set(a.tail_var);
511
512 // If a.head_var is optional, we can potentially remove some literal from
513 // enforcement_literals.
514 const ArcIndex arc_index(arcs_.size());
515 arcs_.push_back(
516 {a.tail_var, a.head_var, offset, a.offset_var, enforcement_literals});
517 auto& presence_literals = arcs_.back().presence_literals;
518 if (integer_trail_->IsOptional(a.head_var)) {
519 // TODO(user): More generally, we can remove any literal that is implied
520 // by to_remove.
521 const Literal to_remove =
522 integer_trail_->IsIgnoredLiteral(a.head_var).Negated();
523 const auto it = std::find(presence_literals.begin(),
524 presence_literals.end(), to_remove);
525 if (it != presence_literals.end()) presence_literals.erase(it);
526 }
527
528 if (presence_literals.empty()) {
529 impacted_arcs_[a.tail_var].push_back(arc_index);
530 } else {
531 for (const Literal l : presence_literals) {
532 if (l.Index() >= literal_to_new_impacted_arcs_.size()) {
533 literal_to_new_impacted_arcs_.resize(l.Index().value() + 1);
534 }
535 literal_to_new_impacted_arcs_[l.Index()].push_back(arc_index);
536 }
537 }
538 arc_counts_.push_back(presence_literals.size());
539 }
540}
541
542// TODO(user): On jobshop problems with a lot of tasks per machine (500), this
543// takes up a big chunck of the running time even before we find a solution.
544// This is because, for each lower bound changed, we inspect 500 arcs even
545// though they will never be propagated because the other bound is still at the
546// horizon. Find an even sparser algorithm?
547void PrecedencesPropagator::PropagateOptionalArcs(Trail* trail) {
548 for (const IntegerVariable var : modified_vars_.PositionsSetAtLeastOnce()) {
549 // The variables are not in increasing order, so we need to continue.
550 if (var >= impacted_potential_arcs_.size()) continue;
551
552 // Note that we can currently check the same ArcInfo up to 3 times, one for
553 // each of the arc variables: tail, NegationOf(head) and offset_var.
554 for (const OptionalArcIndex arc_index : impacted_potential_arcs_[var]) {
555 const ArcInfo& arc = potential_arcs_[arc_index];
556 int num_not_true = 0;
557 Literal to_propagate;
558 for (const Literal l : arc.presence_literals) {
559 if (!trail->Assignment().LiteralIsTrue(l)) {
560 ++num_not_true;
561 to_propagate = l;
562 }
563 }
564 if (num_not_true != 1) continue;
565 if (trail->Assignment().LiteralIsFalse(to_propagate)) continue;
566
567 // Test if this arc can be present or not.
568 // Important arc.tail_var can be different from var here.
569 const IntegerValue tail_lb = integer_trail_->LowerBound(arc.tail_var);
570 const IntegerValue head_ub = integer_trail_->UpperBound(arc.head_var);
571 if (tail_lb + ArcOffset(arc) > head_ub) {
572 integer_reason_.clear();
573 integer_reason_.push_back(
574 integer_trail_->LowerBoundAsLiteral(arc.tail_var));
575 integer_reason_.push_back(
576 integer_trail_->UpperBoundAsLiteral(arc.head_var));
577 AppendLowerBoundReasonIfValid(arc.offset_var, *integer_trail_,
578 &integer_reason_);
579 literal_reason_.clear();
580 for (const Literal l : arc.presence_literals) {
581 if (l != to_propagate) literal_reason_.push_back(l.Negated());
582 }
583 integer_trail_->EnqueueLiteral(to_propagate.Negated(), literal_reason_,
584 integer_reason_);
585 }
586 }
587 }

590IntegerValue PrecedencesPropagator::ArcOffset(const ArcInfo& arc) const {
591 return arc.offset + (arc.offset_var == kNoIntegerVariable
592 ? IntegerValue(0)
593 : integer_trail_->LowerBound(arc.offset_var));
594}

596bool PrecedencesPropagator::EnqueueAndCheck(const ArcInfo& arc,
597 IntegerValue new_head_lb,
598 Trail* trail) {
599 DCHECK_GT(new_head_lb, integer_trail_->LowerBound(arc.head_var));
600
601 // Compute the reason for new_head_lb.
602 //
603 // TODO(user): do like for clause and keep the negation of
604 // arc.presence_literals? I think we could change the integer.h API to accept
605 // true literal like for IntegerVariable, it is really confusing currently.
606 literal_reason_.clear();
607 for (const Literal l : arc.presence_literals) {
608 literal_reason_.push_back(l.Negated());
609 }
610
611 integer_reason_.clear();
612 integer_reason_.push_back(integer_trail_->LowerBoundAsLiteral(arc.tail_var));
613 AppendLowerBoundReasonIfValid(arc.offset_var, *integer_trail_,
614 &integer_reason_);
615
616 // The code works without this block since Enqueue() below can already take
617 // care of conflicts. However, it is better to deal with the conflict
618 // ourselves because we can be smarter about the reason this way.
619 //
620 // The reason for a "precedence" conflict is always a linear reason
621 // involving the tail lower_bound, the head upper bound and eventually the
622 // size lower bound. Because of that, we can use the RelaxLinearReason()
623 // code.
624 if (new_head_lb > integer_trail_->UpperBound(arc.head_var)) {
625 const IntegerValue slack =
626 new_head_lb - integer_trail_->UpperBound(arc.head_var) - 1;
627 integer_reason_.push_back(
628 integer_trail_->UpperBoundAsLiteral(arc.head_var));
629 std::vector<IntegerValue> coeffs(integer_reason_.size(), IntegerValue(1));
630 integer_trail_->RelaxLinearReason(slack, coeffs, &integer_reason_);
631
632 if (!integer_trail_->IsOptional(arc.head_var)) {
633 return integer_trail_->ReportConflict(literal_reason_, integer_reason_);
634 } else {
635 CHECK(!integer_trail_->IsCurrentlyIgnored(arc.head_var));
636 const Literal l = integer_trail_->IsIgnoredLiteral(arc.head_var);
637 if (trail->Assignment().LiteralIsFalse(l)) {
638 literal_reason_.push_back(l);
639 return integer_trail_->ReportConflict(literal_reason_, integer_reason_);
640 } else {
641 integer_trail_->EnqueueLiteral(l, literal_reason_, integer_reason_);
642 return true;
643 }
644 }
645 }
646
647 return integer_trail_->Enqueue(
648 IntegerLiteral::GreaterOrEqual(arc.head_var, new_head_lb),
649 literal_reason_, integer_reason_);
650}

702bool PrecedencesPropagator::DisassembleSubtree(
703 int source, int target, std::vector<bool>* can_be_skipped) {
704 // Note that we explore a tree, so we can do it in any order, and the one
705 // below seems to be the fastest.
706 tmp_vector_.clear();
707 tmp_vector_.push_back(source);
708 while (!tmp_vector_.empty()) {
709 const int tail = tmp_vector_.back();
710 tmp_vector_.pop_back();
711 for (const ArcIndex arc_index : impacted_arcs_[IntegerVariable(tail)]) {
712 const ArcInfo& arc = arcs_[arc_index];
713 if (arc.is_marked) {
714 arc.is_marked = false; // mutable.
715 if (arc.head_var.value() == target) return true;
716 DCHECK(!(*can_be_skipped)[arc.head_var.value()]);
717 (*can_be_skipped)[arc.head_var.value()] = true;
718 tmp_vector_.push_back(arc.head_var.value());
719 }
720 }
721 }
722 return false;

725void PrecedencesPropagator::AnalyzePositiveCycle(
726 ArcIndex first_arc, Trail* trail, std::vector<Literal>* must_be_all_true,
727 std::vector<Literal>* literal_reason,
728 std::vector<IntegerLiteral>* integer_reason) {
729 must_be_all_true->clear();
730 literal_reason->clear();
731 integer_reason->clear();
732
733 // Follow bf_parent_arc_of_[] to find the cycle containing first_arc.
734 const IntegerVariable first_arc_head = arcs_[first_arc].head_var;
735 ArcIndex arc_index = first_arc;
736 std::vector<ArcIndex> arc_on_cycle;
737
738 // Just to be safe and avoid an infinite loop we use the fact that the maximum
739 // cycle size on a graph with n nodes is of size n. If we have more in the
740 // code below, it means first_arc is not part of a cycle according to
741 // bf_parent_arc_of_[], which should never happen.
742 const int num_nodes = impacted_arcs_.size();
743 while (arc_on_cycle.size() <= num_nodes) {
744 arc_on_cycle.push_back(arc_index);
745 const ArcInfo& arc = arcs_[arc_index];
746 if (arc.tail_var == first_arc_head) break;
747 arc_index = bf_parent_arc_of_[arc.tail_var.value()];
748 CHECK_NE(arc_index, ArcIndex(-1));
749 }
750 CHECK_NE(arc_on_cycle.size(), num_nodes + 1) << "Infinite loop.";
751
752 // Compute the reason for this cycle.
753 IntegerValue sum(0);
754 for (const ArcIndex arc_index : arc_on_cycle) {
755 const ArcInfo& arc = arcs_[arc_index];
756 sum += ArcOffset(arc);
757 AppendLowerBoundReasonIfValid(arc.offset_var, *integer_trail_,
758 integer_reason);
759 for (const Literal l : arc.presence_literals) {
760 literal_reason->push_back(l.Negated());
761 }
762
763 // If the cycle happens to contain optional variable not yet ignored, then
764 // it is not a conflict anymore, but we can infer that these variable must
765 // all be ignored. This is because since we propagated them even if they
766 // where not present for sure, their presence literal must form a cycle
767 // together (i.e. they are all absent or present at the same time).
768 if (integer_trail_->IsOptional(arc.head_var)) {
769 must_be_all_true->push_back(
770 integer_trail_->IsIgnoredLiteral(arc.head_var));
771 }
772 }
773
774 // TODO(user): what if the sum overflow? this is just a check so I guess
775 // we don't really care, but fix the issue.
776 CHECK_GT(sum, 0);
777}
778
779// Note that in our settings it is important to use an algorithm that tries to
780// minimize the number of integer_trail_->Enqueue() as much as possible.
781//
782// TODO(user): The current algorithm is quite efficient, but there is probably
783// still room for improvements.
784bool PrecedencesPropagator::BellmanFordTarjan(Trail* trail) {
785 const int num_nodes = impacted_arcs_.size();
786
787 // These vector are reset by CleanUpMarkedArcsAndParents() so resize is ok.
788 bf_can_be_skipped_.resize(num_nodes, false);
789 bf_parent_arc_of_.resize(num_nodes, ArcIndex(-1));
790 const auto cleanup =
791 ::absl::MakeCleanup([this]() { CleanUpMarkedArcsAndParents(); });
792
793 // The queue initialization is done by InitializeBFQueueWithModifiedNodes().
794 while (!bf_queue_.empty()) {
795 const int node = bf_queue_.front();
796 bf_queue_.pop_front();
797 bf_in_queue_[node] = false;
798
799 // TODO(user): we don't need bf_can_be_skipped_ since we can detect this
800 // if this node has a parent arc which is not marked. Investigate if it is
801 // faster without the vector<bool>.
802 //
803 // TODO(user): An alternative algorithm is to remove all these nodes from
804 // the queue instead of simply marking them. This should also lead to a
805 // better "relaxation" order of the arcs. It is however a bit more work to
806 // remove them since we need to track their position.
807 if (bf_can_be_skipped_[node]) {
808 DCHECK_NE(bf_parent_arc_of_[node], -1);
809 DCHECK(!arcs_[bf_parent_arc_of_[node]].is_marked);
810 continue;
811 }
812
813 const IntegerValue tail_lb =
814 integer_trail_->LowerBound(IntegerVariable(node));
815 for (const ArcIndex arc_index : impacted_arcs_[IntegerVariable(node)]) {
816 const ArcInfo& arc = arcs_[arc_index];
817 DCHECK_EQ(arc.tail_var, node);
818 const IntegerValue candidate = tail_lb + ArcOffset(arc);
819 if (candidate > integer_trail_->LowerBound(arc.head_var)) {
820 if (integer_trail_->IsCurrentlyIgnored(arc.head_var)) continue;
821 if (!EnqueueAndCheck(arc, candidate, trail)) return false;
822
823 // This is the Tarjan contribution to Bellman-Ford. This code detect
824 // positive cycle, and because it disassemble the subtree while doing
825 // so, the cost is amortized during the algorithm execution. Another
826 // advantages is that it will mark the node explored here as skippable
827 // which will avoid to propagate them too early (knowing that they will
828 // need to be propagated again later).
829 if (DisassembleSubtree(arc.head_var.value(), arc.tail_var.value(),
830 &bf_can_be_skipped_)) {
831 std::vector<Literal> must_be_all_true;
832 AnalyzePositiveCycle(arc_index, trail, &must_be_all_true,
833 &literal_reason_, &integer_reason_);
834 if (must_be_all_true.empty()) {
835 return integer_trail_->ReportConflict(literal_reason_,
836 integer_reason_);
837 } else {
838 gtl::STLSortAndRemoveDuplicates(&must_be_all_true);
839 for (const Literal l : must_be_all_true) {
840 if (trail_->Assignment().LiteralIsFalse(l)) {
841 literal_reason_.push_back(l);
842 return integer_trail_->ReportConflict(literal_reason_,
843 integer_reason_);
844 }
845 }
846 for (const Literal l : must_be_all_true) {
847 if (trail_->Assignment().LiteralIsTrue(l)) continue;
848 integer_trail_->EnqueueLiteral(l, literal_reason_,
849 integer_reason_);
850 }
851
852 // We just marked some optional variable as ignored, no need
853 // to update bf_parent_arc_of_[].
854 continue;
855 }
856 }
857
858 // We need to enforce the invariant that only the arc_index in
859 // bf_parent_arc_of_[] are marked (but not necessarily all of them
860 // since we unmark some in DisassembleSubtree()).
861 if (bf_parent_arc_of_[arc.head_var.value()] != -1) {
862 arcs_[bf_parent_arc_of_[arc.head_var.value()]].is_marked = false;
863 }
864
865 // Tricky: We just enqueued the fact that the lower-bound of head is
866 // candidate. However, because the domain of head may be discrete, it is
867 // possible that the lower-bound of head is now higher than candidate!
868 // If this is the case, we don't update bf_parent_arc_of_[] so that we
869 // don't wrongly detect a positive weight cycle because of this "extra
870 // push".
871 const IntegerValue new_bound = integer_trail_->LowerBound(arc.head_var);
872 if (new_bound == candidate) {
873 bf_parent_arc_of_[arc.head_var.value()] = arc_index;
874 arcs_[arc_index].is_marked = true;
875 } else {
876 // We still unmark any previous dependency, since we have pushed the
877 // value of arc.head_var further.
878 bf_parent_arc_of_[arc.head_var.value()] = -1;
879 }
880
881 // We do not re-enqueue if we are in a propagation loop and new_bound
882 // was not pushed to candidate or higher.
883 bf_can_be_skipped_[arc.head_var.value()] = false;
884 if (!bf_in_queue_[arc.head_var.value()] && new_bound >= candidate) {
885 bf_queue_.push_back(arc.head_var.value());
886 bf_in_queue_[arc.head_var.value()] = true;
887 }
888 }
889 }
890 }
891 return true;
892}

894int PrecedencesPropagator::AddGreaterThanAtLeastOneOfConstraintsFromClause(
895 const absl::Span<const Literal> clause, Model* model) {
896 CHECK_EQ(model->GetOrCreate<Trail>()->CurrentDecisionLevel(), 0);
897 if (clause.size() < 2) return 0;
898
899 // Collect all arcs impacted by this clause.
900 std::vector<ArcInfo> infos;
901 for (const Literal l : clause) {
902 if (l.Index() >= literal_to_new_impacted_arcs_.size()) continue;
903 for (const ArcIndex arc_index : literal_to_new_impacted_arcs_[l.Index()]) {
904 const ArcInfo& arc = arcs_[arc_index];
905 if (arc.presence_literals.size() != 1) continue;
906
907 // TODO(user): Support variable offset.
908 if (arc.offset_var != kNoIntegerVariable) continue;
909 infos.push_back(arc);
910 }
911 }
912 if (infos.size() <= 1) return 0;
913
914 // Stable sort by head_var so that for a same head_var, the entry are sorted
915 // by Literal as they appear in clause.
916 std::stable_sort(infos.begin(), infos.end(),
917 [](const ArcInfo& a, const ArcInfo& b) {
918 return a.head_var < b.head_var;
919 });
920
921 // We process ArcInfo with the same head_var toghether.
922 int num_added_constraints = 0;
923 auto* solver = model->GetOrCreate<SatSolver>();
924 for (int i = 0; i < infos.size();) {
925 const int start = i;
926 const IntegerVariable head_var = infos[start].head_var;
927 for (i++; i < infos.size() && infos[i].head_var == head_var; ++i) {
928 }
929 const absl::Span<ArcInfo> arcs(&infos[start], i - start);
930
931 // Skip single arcs since it will already be fully propagated.
932 if (arcs.size() < 2) continue;
933
934 // Heuristic. Look for full or almost full clauses. We could add
935 // GreaterThanAtLeastOneOf() with more enforcement literals. TODO(user):
936 // experiments.
937 if (arcs.size() + 1 < clause.size()) continue;
938
939 std::vector<IntegerVariable> vars;
940 std::vector<IntegerValue> offsets;
941 std::vector<Literal> selectors;
942 std::vector<Literal> enforcements;
943
944 int j = 0;
945 for (const Literal l : clause) {
946 bool added = false;
947 for (; j < arcs.size() && l == arcs[j].presence_literals.front(); ++j) {
948 added = true;
949 vars.push_back(arcs[j].tail_var);
950 offsets.push_back(arcs[j].offset);
951
952 // Note that duplicate selector are supported.
953 //
954 // TODO(user): If we support variable offset, we should regroup the arcs
955 // into one (tail + offset <= head) though, instead of having too
956 // identical entries.
957 selectors.push_back(l);
958 }
959 if (!added) {
960 enforcements.push_back(l.Negated());
961 }
962 }
963
964 // No point adding a constraint if there is not at least two different
965 // literals in selectors.
966 if (enforcements.size() + 1 == clause.size()) continue;
967
968 ++num_added_constraints;
969 model->Add(GreaterThanAtLeastOneOf(head_var, vars, offsets, selectors,
970 enforcements));
971 if (!solver->FinishPropagation()) return num_added_constraints;
972 }
973 return num_added_constraints;
974}

976int PrecedencesPropagator::
977 AddGreaterThanAtLeastOneOfConstraintsWithClauseAutoDetection(Model* model) {
978 auto* time_limit = model->GetOrCreate<TimeLimit>();
979 auto* solver = model->GetOrCreate<SatSolver>();
980
981 // Fill the set of incoming conditional arcs for each variables.
982 absl::StrongVector<IntegerVariable, std::vector<ArcIndex>> incoming_arcs_;
983 for (ArcIndex arc_index(0); arc_index < arcs_.size(); ++arc_index) {
984 const ArcInfo& arc = arcs_[arc_index];
985
986 // Only keep arc that have a fixed offset and a single presence_literals.
987 if (arc.offset_var != kNoIntegerVariable) continue;
988 if (arc.tail_var == arc.head_var) continue;
989 if (arc.presence_literals.size() != 1) continue;
990
991 if (arc.head_var >= incoming_arcs_.size()) {
992 incoming_arcs_.resize(arc.head_var.value() + 1);
993 }
994 incoming_arcs_[arc.head_var].push_back(arc_index);
995 }
996
997 int num_added_constraints = 0;
998 for (IntegerVariable target(0); target < incoming_arcs_.size(); ++target) {
999 if (incoming_arcs_[target].size() <= 1) continue;
1000 if (time_limit->LimitReached()) return num_added_constraints;
1001
1002 // Detect set of incoming arcs for which at least one must be present.
1003 // TODO(user): Find more than one disjoint set of incoming arcs.
1004 // TODO(user): call MinimizeCoreWithPropagation() on the clause.
1005 solver->Backtrack(0);
1006 if (solver->IsModelUnsat()) return num_added_constraints;
1007 std::vector<Literal> clause;
1008 for (const ArcIndex arc_index : incoming_arcs_[target]) {
1009 const Literal literal = arcs_[arc_index].presence_literals.front();
1010 if (solver->Assignment().LiteralIsFalse(literal)) continue;
1011 const SatSolver::Status status =
1012 solver->EnqueueDecisionAndBacktrackOnConflict(literal.Negated());
1013 if (status == SatSolver::INFEASIBLE) return num_added_constraints;
1014 if (status == SatSolver::ASSUMPTIONS_UNSAT) {
1015 clause = solver->GetLastIncompatibleDecisions();
1016 break;
1017 }
1018 }
1019 solver->Backtrack(0);
1020
1021 if (clause.size() > 1) {
1022 // Extract the set of arc for which at least one must be present.
1023 const absl::btree_set<Literal> clause_set(clause.begin(), clause.end());
1024 std::vector<ArcIndex> arcs_in_clause;
1025 for (const ArcIndex arc_index : incoming_arcs_[target]) {
1026 const Literal literal(arcs_[arc_index].presence_literals.front());
1027 if (clause_set.contains(literal.Negated())) {
1028 arcs_in_clause.push_back(arc_index);
1029 }
1030 }
1031
1032 VLOG(2) << arcs_in_clause.size() << "/" << incoming_arcs_[target].size();
1033
1034 ++num_added_constraints;
1035 std::vector<IntegerVariable> vars;
1036 std::vector<IntegerValue> offsets;
1037 std::vector<Literal> selectors;
1038 for (const ArcIndex a : arcs_in_clause) {
1039 vars.push_back(arcs_[a].tail_var);
1040 offsets.push_back(arcs_[a].offset);
1041 selectors.push_back(Literal(arcs_[a].presence_literals.front()));
1042 }
1043 model->Add(GreaterThanAtLeastOneOf(target, vars, offsets, selectors));
1044 if (!solver->FinishPropagation()) return num_added_constraints;
1045 }
1046 }
1047
1048 return num_added_constraints;
1049}

1051int PrecedencesPropagator::AddGreaterThanAtLeastOneOfConstraints(Model* model) {
1052 VLOG(1) << "Detecting GreaterThanAtLeastOneOf() constraints...";
1053 auto* time_limit = model->GetOrCreate<TimeLimit>();
1054 auto* solver = model->GetOrCreate<SatSolver>();
1055 auto* clauses = model->GetOrCreate<LiteralWatchers>();
1056 int num_added_constraints = 0;
1057
1058 // We have two possible approaches. For now, we prefer the first one except if
1059 // there is too many clauses in the problem.
1060 //
1061 // TODO(user): Do more extensive experiment. Remove the second approach as
1062 // it is more time consuming? or identify when it make sense. Note that the
1063 // first approach also allows to use "incomplete" at least one between arcs.
1064 if (clauses->AllClausesInCreationOrder().size() < 1e6) {
1065 // TODO(user): This does not take into account clause of size 2 since they
1066 // are stored in the BinaryImplicationGraph instead. Some ideas specific
1067 // to size 2:
1068 // - There can be a lot of such clauses, but it might be nice to consider
1069 // them. we need to experiments.
1070 // - The automatic clause detection might be a better approach and it
1071 // could be combined with probing.
1072 for (const SatClause* clause : clauses->AllClausesInCreationOrder()) {
1073 if (time_limit->LimitReached()) return num_added_constraints;
1074 if (solver->IsModelUnsat()) return num_added_constraints;
1075 num_added_constraints += AddGreaterThanAtLeastOneOfConstraintsFromClause(
1076 clause->AsSpan(), model);
1077 }
1078
1079 // It is common that there is only two alternatives to push a variable.
1080 // In this case, our presolve most likely made sure that the two are
1081 // controlled by a single Boolean. This allows to detect this and add the
1082 // appropriate greater than at least one of.
1083 const int num_booleans = solver->NumVariables();
1084 if (num_booleans < 1e6) {
1085 for (int i = 0; i < num_booleans; ++i) {
1086 if (time_limit->LimitReached()) return num_added_constraints;
1087 if (solver->IsModelUnsat()) return num_added_constraints;
1088 num_added_constraints +=
1089 AddGreaterThanAtLeastOneOfConstraintsFromClause(
1090 {Literal(BooleanVariable(i), true),
1091 Literal(BooleanVariable(i), false)},
1092 model);
1093 }
1094 }
1095
1096 } else {
1097 num_added_constraints +=
1098 AddGreaterThanAtLeastOneOfConstraintsWithClauseAutoDetection(model);
1099 }
1100
1101 VLOG(1) << "Added " << num_added_constraints
1102 << " GreaterThanAtLeastOneOf() constraints.";
1103 return num_added_constraints;
1104}
