29// A filter that only keeps some specific key-value pairs of a map.
30//
31// It is used to limit the quantity of data returned in a SolveResult or in a
32// CallbackResult when the models are huge and the user is only interested in
33// the values of a subset of the keys.
34//
35// The keys, the type KeyType, should match the definition of "key types" given
36// in key_types.h.
37//
38// A filter is composed of two sub-filters that act as a veto system: a
39// key-value pair is kept only when it is kept by both filters. Those filters
40// are:
41// - skip_zero_values: when true, only keep pairs if the value is non zero (if
42// the value is boolean, keep only pairs with true value).
43// - filtered_keys: when set, only keep pairs which keys are in the provided
44// list. If this list is empty, no pairs are returned. When unset, keep all
45// pairs.
46//
47// See the MakeSkipAllFilter(), MakeSkipZerosFilter() and MakeKeepKeysFilter()
48// functions below that provide shortcuts for making filters.
49//
50// This class is a factory of SparseVectorFilterProto.
51template <typename KeyType>
52struct MapFilter {
53 // If true, omits the pairs with zero values (the pairs with false value for
54 // bool vectors).
55 //
56 // Default is false, pairs with zero (or false) value are kept.
57 //
58 // Prefer using MakeSkipZerosFilter() when appropriate.
59 bool skip_zero_values = false;
60
61 // The set of keys of pairs to keep. When unset, all pairs are kept
62 // (at least the ones with non-zero values, when skip_zero_values() is true).
63 //
64 // Default is unset, all pairs are kept.
65 //
66 // Example:
67 // MapFilter<Variable> filter = ...;
68 //
69 // // Unset the filter.
70 // filter.filtered_keys.reset();
71 // // alternatively:
72 // filter.filtered_keys = std::nullopt;
73 //
74 // // Set the filter with an empty list of keys (filtering out all pairs).
75 // //
76 // // Note that here `= {}` would NOT work since it would unset the
77 // // filtered_keys optional (i.e. like calling reset()).
78 // filter.filtered_keys.emplace();
79 //
80 // // Set the filter to a fix set of variables.
81 // const Variable x = ...;
82 // const Variable y = ...;
83 // filter.filtered_keys = {x, y};
84 //
85 // // Set the filter from a collection of variables.
86 // std::vector<Variable> decision_vars = {...};
87 // filter.emplace(decision_vars.begin(), decision_vars.end());
88 //
89 // Prefer using MakeSkipAllFilter() or MakeKeepKeysFilter() when appropriate.
90 std::optional<IdSet<KeyType>> filtered_keys;
91
92 // Returns the model of filtered keys. It returns a non-null value if and only
93 // if the filtered_keys is set and non-empty.
94 inline const ModelStorage* storage() const;
95
96 // Returns the proto corresponding to this filter.
97 SparseVectorFilterProto Proto() const;
98};

100// Returns a filter that skips all key-value pairs.
101//
102// This is typically used to disable the dual data in SolveResult when these are
103// ignored by the user.
104//
105// Example:
106// const auto filter = MakeSkipAllFilter<Variable>();
107template <typename KeyType>
108MapFilter<KeyType> MakeSkipAllFilter() {
109 MapFilter<KeyType> ret;
110 ret.filtered_keys.emplace();
111 return ret;
112}
113
114// Returns a filter that skips all key-value pairs with zero values (or false
115// values for bool).
116//
117// Example:
118// const auto filter = MakeSkipZerosFilter<Variable>();
119template <typename KeyType>
120MapFilter<KeyType> MakeSkipZerosFilter() {
121 MapFilter<KeyType> ret;
122 ret.skip_zero_values = true;
123 return ret;
124}
125
126// Returns a filter that keeps the key-value pairs with the given keys.
127//
128// Example:
129// std::vector<Variable> decision_vars = ...;
130// const auto filter = MakeKeepKeysFilter(decision_vars);
131template <typename Collection,
132 typename ValueType = typename Collection::value_type>
133MapFilter<ValueType> MakeKeepKeysFilter(const Collection& keys) {
134 MapFilter<ValueType> ret;
135 ret.filtered_keys.emplace(keys.begin(), keys.end());
136 return ret;
137}
138
139// Returns a filter that keeps the key-value pairs with the given keys.
140//
141// This overload is necessary since C++ does not automatically deduce the type
142// when using the previous one.
143//
144// Example:
145// const Variable x = ...;
146// const Variable y = ...;
147// const auto filter = MakeKeepKeysFilter({x, y});
148template <typename KeyType>
149MapFilter<KeyType> MakeKeepKeysFilter(std::initializer_list<KeyType> keys) {
150 MapFilter<KeyType> ret;
151 ret.filtered_keys = keys;
152 return ret;
153}
154
156// Inline functions implementations.
158
159template <typename KeyType>
160const ModelStorage* MapFilter<KeyType>::storage() const {
161 return filtered_keys ? filtered_keys->storage() : nullptr;
162}
163
164template <typename KeyType>
165SparseVectorFilterProto MapFilter<KeyType>::Proto() const {
166 SparseVectorFilterProto ret;
167 ret.set_skip_zero_values(skip_zero_values);
168 if (filtered_keys) {
169 ret.set_filter_by_ids(true);
170 const auto filtered_ids = ret.mutable_filtered_ids();
171 filtered_ids->Reserve(filtered_keys->size());
172 for (const auto id : filtered_keys->raw_set()) {
173 filtered_ids->Add(id.value());
174 }
175 // Iteration on the set is random but we want the proto to be stable.
176 std::sort(filtered_ids->begin(), filtered_ids->end());
177 }
178 return ret;
179}
