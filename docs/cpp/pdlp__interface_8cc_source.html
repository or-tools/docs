108MPSolver::ResultStatus PdlpInterface::Solve(const MPSolverParameters& param) {
109 // Reset extraction as this interface is not incremental yet.
110 Reset();
111 interrupt_solver_ = false;
112 ExtractModel();
113
114 SetParameters(param);
115 if (quiet_) {
116 parameters_.set_verbosity_level(0);
117 } else {
118 parameters_.set_verbosity_level(3);
119 }
120
121 solver_->SetSolverSpecificParametersAsString(
122 solver_->solver_specific_parameter_string_);
123
124 // Time limit.
125 if (solver_->time_limit()) {
126 VLOG(1) << "Setting time limit = " << solver_->time_limit() << " ms.";
127 parameters_.mutable_termination_criteria()->set_time_sec_limit(
128 static_cast<double>(solver_->time_limit()) / 1000.0);
129 }
130
131 // Mark variables and constraints as extracted.
132 for (int i = 0; i < solver_->variables_.size(); ++i) {
133 set_variable_as_extracted(i, true);
134 }
135 for (int i = 0; i < solver_->constraints_.size(); ++i) {
136 set_constraint_as_extracted(i, true);
137 }
138
139 MPModelProto model_proto;
140 solver_->ExportModelToProto(&model_proto);
141 MPModelRequest request;
142 *request.mutable_model() = std::move(model_proto);
143 if (!google::protobuf::TextFormat::PrintToString(
144 parameters_, request.mutable_solver_specific_parameters())) {
145 LOG(QFATAL) << "Error converting parameters to text format: "
146 << parameters_.DebugString();
147 }
148 absl::StatusOr<MPSolutionResponse> response = PdlpSolveProto(
149 request, /*relax_integer_variables=*/true, &interrupt_solver_);
150
151 if (!response.ok()) {
152 LOG(ERROR) << "Unexpected error solving with PDLP: " << response.status();
153 return MPSolver::ABNORMAL;
154 }
155
156 // The solution must be marked as synchronized even when no solution exists.
157 sync_status_ = SOLUTION_SYNCHRONIZED;
158 if (response->status() == MPSOLVER_CANCELLED_BY_USER) {
159 // MPSOLVER_CANCELLED_BY_USER is only for when the solver didn't have time
160 // to return a proper status, and is not convertible to an MPSolver status.
161 result_status_ = MPSolver::NOT_SOLVED;
162 } else {
163 result_status_ = static_cast<MPSolver::ResultStatus>(response->status());
164 }
165 if (response->has_solver_specific_info()) {
166 if (!solve_log_.ParseFromString(response->solver_specific_info())) {
167 LOG(DFATAL)
168 << "Unable to parse PDLP's SolveLog from solver_specific_info";
169 }
170 }
171
172 if (response->status() == MPSOLVER_FEASIBLE ||
173 response->status() == MPSOLVER_OPTIMAL) {
174 const absl::Status result = solver_->LoadSolutionFromProto(*response);
175 if (!result.ok()) {
176 LOG(ERROR) << "LoadSolutionFromProto failed: " << result;
177 }
178 }
179
180 return result_status_;
181}

183std::optional<MPSolutionResponse> PdlpInterface::DirectlySolveProto(
184 const MPModelRequest& request, std::atomic<bool>* interrupt) {
185 absl::StatusOr<MPSolutionResponse> response =
186 PdlpSolveProto(request, /*relax_integer_variables=*/true, interrupt);
187
188 if (!response.ok()) {
189 LOG(ERROR) << "Unexpected error solving with PDLP: " << response.status();
190 MPSolutionResponse error_response;
191 error_response.set_status(MPSolverResponseStatus::MPSOLVER_ABNORMAL);
192 error_response.set_status_str(response.status().ToString());
193 return error_response;
194 }
195
196 return *response;
197}
