62void MinimizeCoreWithPropagation(TimeLimit* limit, SatSolver* solver,
63 std::vector<Literal>* core) {
64 if (solver->ModelIsUnsat()) return;
65 absl::btree_set<LiteralIndex> moved_last;
66 std::vector<Literal> candidate(core->begin(), core->end());
67
68 solver->Backtrack(0);
69 solver->SetAssumptionLevel(0);
70 if (!solver->FinishPropagation()) return;
71 while (!limit->LimitReached()) {
72 // We want each literal in candidate to appear last once in our propagation
73 // order. We want to do that while maximizing the reutilization of the
74 // current assignment prefix, that is minimizing the number of
75 // decision/progagation we need to perform.
76 const int target_level = MoveOneUnprocessedLiteralLast(
77 moved_last, solver->CurrentDecisionLevel(), &candidate);
78 if (target_level == -1) break;
79 solver->Backtrack(target_level);
80 while (!solver->ModelIsUnsat() && !limit->LimitReached() &&
81 solver->CurrentDecisionLevel() < candidate.size()) {
82 const Literal decision = candidate[solver->CurrentDecisionLevel()];
83 if (solver->Assignment().LiteralIsTrue(decision)) {
84 candidate.erase(candidate.begin() + solver->CurrentDecisionLevel());
85 continue;
86 } else if (solver->Assignment().LiteralIsFalse(decision)) {
87 // This is a "weird" API to get the subset of decisions that caused
88 // this literal to be false with reason analysis.
89 solver->EnqueueDecisionAndBacktrackOnConflict(decision);
90 candidate = solver->GetLastIncompatibleDecisions();
91 break;
92 } else {
93 solver->EnqueueDecisionAndBackjumpOnConflict(decision);
94 }
95 }
96 if (candidate.empty() || solver->ModelIsUnsat()) return;
97 moved_last.insert(candidate.back().Index());
98 }
99
100 solver->Backtrack(0);
101 solver->SetAssumptionLevel(0);
102 if (candidate.size() < core->size()) {
103 VLOG(1) << "minimization with propag " << core->size() << " -> "
104 << candidate.size();
105
106 // We want to preserve the order of literal in the response.
107 absl::flat_hash_set<LiteralIndex> set;
108 for (const Literal l : candidate) set.insert(l.Index());
109 int new_size = 0;
110 for (const Literal l : *core) {
111 if (set.contains(l.Index())) {
112 (*core)[new_size++] = l;
113 }
114 }
115 core->resize(new_size);
116 }
117}

119void MinimizeCoreWithSearch(TimeLimit* limit, SatSolver* solver,
120 std::vector<Literal>* core) {
121 if (solver->ModelIsUnsat()) return;
122
123 // TODO(user): tune.
124 if (core->size() > 100 || core->size() == 1) return;
125
126 const bool old_log_state = solver->mutable_logger()->LoggingIsEnabled();
127 solver->mutable_logger()->EnableLogging(false);
128
129 const int old_size = core->size();
130 std::vector<Literal> assumptions;
131 absl::flat_hash_set<LiteralIndex> removed_once;
132 while (true) {
133 if (limit->LimitReached()) break;
134
135 // Find a not yet removed literal to remove.
136 // We prefer to remove high indices since these are more likely to be of
137 // high depth.
138 //
139 // TODO(user): Properly use the node depth instead.
140 int to_remove = -1;
141 for (int i = core->size(); --i >= 0;) {
142 const Literal l = (*core)[i];
143 const auto [_, inserted] = removed_once.insert(l.Index());
144 if (inserted) {
145 to_remove = i;
146 break;
147 }
148 }
149 if (to_remove == -1) break;
150
151 assumptions.clear();
152 for (int i = 0; i < core->size(); ++i) {
153 if (i == to_remove) continue;
154 assumptions.push_back((*core)[i]);
155 }
156
157 const auto status = solver->ResetAndSolveWithGivenAssumptions(
158 assumptions, /*max_number_of_conflicts=*/1000);
159 if (status == SatSolver::ASSUMPTIONS_UNSAT) {
160 *core = solver->GetLastIncompatibleDecisions();
161 }
162 }
163
164 if (core->size() < old_size) {
165 VLOG(1) << "minimization with search " << old_size << " -> "
166 << core->size();
167 }
168
169 solver->ResetToLevelZero();
170 solver->mutable_logger()->EnableLogging(old_log_state);
171}

173bool ProbeLiteral(Literal assumption, SatSolver* solver) {
174 if (solver->ModelIsUnsat()) return false;
175
176 const bool old_log_state = solver->mutable_logger()->LoggingIsEnabled();
177 solver->mutable_logger()->EnableLogging(false);
178
179 // Note that since we only care about Booleans here, even if we have a
180 // feasible solution, it might not be feasible for the full cp_model.
181 //
182 // TODO(user): Still use it if the problem is Boolean only.
183 const auto status = solver->ResetAndSolveWithGivenAssumptions(
184 {assumption}, /*max_number_of_conflicts=*/1'000);
185 solver->ResetToLevelZero();
186 if (status == SatSolver::ASSUMPTIONS_UNSAT) {
187 solver->AddUnitClause(assumption.Negated());
188 solver->Propagate();
189 }
190
191 solver->mutable_logger()->EnableLogging(old_log_state);
192 return solver->Assignment().LiteralIsAssigned(assumption);
193}

196void FilterAssignedLiteral(const VariablesAssignment& assignment,
197 std::vector<Literal>* core) {
198 int new_size = 0;
199 for (const Literal lit : *core) {
200 if (assignment.LiteralIsTrue(lit)) continue;
201 if (assignment.LiteralIsFalse(lit)) {
202 (*core)[0] = lit;
203 core->resize(1);
204 return;
205 }
206 (*core)[new_size++] = lit;
207 }
208 core->resize(new_size);

211SatSolver::Status MinimizeIntegerVariableWithLinearScanAndLazyEncoding(
212 IntegerVariable objective_var,
213 const std::function<void()>& feasible_solution_observer, Model* model) {
214 auto* sat_solver = model->GetOrCreate<SatSolver>();
215 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
216 auto* search = model->GetOrCreate<IntegerSearchHelper>();
217 const SatParameters& parameters = *(model->GetOrCreate<SatParameters>());
218
219 // Simple linear scan algorithm to find the optimal.
220 if (!sat_solver->ResetToLevelZero()) return SatSolver::INFEASIBLE;
221 while (true) {
222 const SatSolver::Status result = search->SolveIntegerProblem();
223 if (result != SatSolver::FEASIBLE) return result;
224
225 // The objective is the current lower bound of the objective_var.
226 const IntegerValue objective = integer_trail->LowerBound(objective_var);
227
228 // We have a solution!
229 if (feasible_solution_observer != nullptr) {
230 feasible_solution_observer();
231 }
232 if (parameters.stop_after_first_solution()) {
233 return SatSolver::LIMIT_REACHED;
234 }
235
236 // Restrict the objective.
237 sat_solver->Backtrack(0);
238 if (!integer_trail->Enqueue(
239 IntegerLiteral::LowerOrEqual(objective_var, objective - 1), {},
240 {})) {
241 return SatSolver::INFEASIBLE;
242 }
243 }

246void RestrictObjectiveDomainWithBinarySearch(
247 IntegerVariable objective_var,
248 const std::function<void()>& feasible_solution_observer, Model* model) {
249 const SatParameters old_params = *model->GetOrCreate<SatParameters>();
250 SatSolver* sat_solver = model->GetOrCreate<SatSolver>();
251 IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
252 IntegerEncoder* integer_encoder = model->GetOrCreate<IntegerEncoder>();
253
254 // Set the requested conflict limit.
255 {
256 SatParameters new_params = old_params;
257 new_params.set_max_number_of_conflicts(
258 old_params.binary_search_num_conflicts());
259 *model->GetOrCreate<SatParameters>() = new_params;
260 }
261
262 // The assumption (objective <= value) for values in
263 // [unknown_min, unknown_max] reached the conflict limit.
264 bool loop = true;
265 IntegerValue unknown_min = integer_trail->UpperBound(objective_var);
266 IntegerValue unknown_max = integer_trail->LowerBound(objective_var);
267 while (loop) {
268 sat_solver->Backtrack(0);
269 const IntegerValue lb = integer_trail->LowerBound(objective_var);
270 const IntegerValue ub = integer_trail->UpperBound(objective_var);
271 unknown_min = std::min(unknown_min, ub);
272 unknown_max = std::max(unknown_max, lb);
273
274 // We first refine the lower bound and then the upper bound.
275 IntegerValue target;
276 if (lb < unknown_min) {
277 target = lb + (unknown_min - lb) / 2;
278 } else if (unknown_max < ub) {
279 target = ub - (ub - unknown_max) / 2;
280 } else {
281 VLOG(1) << "Binary-search, done.";
282 break;
283 }
284 VLOG(1) << "Binary-search, objective: [" << lb << "," << ub << "]"
285 << " tried: [" << unknown_min << "," << unknown_max << "]"
286 << " target: obj<=" << target;
287 SatSolver::Status result;
288 if (target < ub) {
289 const Literal assumption = integer_encoder->GetOrCreateAssociatedLiteral(
290 IntegerLiteral::LowerOrEqual(objective_var, target));
291 result = ResetAndSolveIntegerProblem({assumption}, model);
292 } else {
293 result = ResetAndSolveIntegerProblem({}, model);
294 }
295
296 switch (result) {
297 case SatSolver::INFEASIBLE: {
298 loop = false;
299 break;
300 }
301 case SatSolver::ASSUMPTIONS_UNSAT: {
302 // Update the objective lower bound.
303 sat_solver->Backtrack(0);
304 if (!integer_trail->Enqueue(
305 IntegerLiteral::GreaterOrEqual(objective_var, target + 1), {},
306 {})) {
307 loop = false;
308 }
309 break;
310 }
311 case SatSolver::FEASIBLE: {
312 // The objective is the current lower bound of the objective_var.
313 const IntegerValue objective = integer_trail->LowerBound(objective_var);
314 if (feasible_solution_observer != nullptr) {
315 feasible_solution_observer();
316 }
317
318 // We have a solution, restrict the objective upper bound to only look
319 // for better ones now.
320 sat_solver->Backtrack(0);
321 if (!integer_trail->Enqueue(
322 IntegerLiteral::LowerOrEqual(objective_var, objective - 1), {},
323 {})) {
324 loop = false;
325 }
326 break;
327 }
328 case SatSolver::LIMIT_REACHED: {
329 unknown_min = std::min(target, unknown_min);
330 unknown_max = std::max(target, unknown_max);
331 break;
332 }
333 }
334 }
335
336 sat_solver->Backtrack(0);
337 *model->GetOrCreate<SatParameters>() = old_params;
338}

342// If the given model is unsat under the given assumptions, returns one or more
343// non-overlapping set of assumptions, each set making the problem infeasible on
344// its own (the cores).
345//
346// In presence of weights, we "generalize" the notions of disjoints core using
347// the WCE idea describe in "Weight-Aware Core Extraction in SAT-Based MaxSAT
348// solving" Jeremias Berg And Matti Jarvisalo.
349//
350// The returned status can be either:
351// - ASSUMPTIONS_UNSAT if the set of returned core perfectly cover the given
352// assumptions, in this case, we don't bother trying to find a SAT solution
353// with no assumptions.
354// - FEASIBLE if after finding zero or more core we have a solution.
355// - LIMIT_REACHED if we reached the time-limit before one of the two status
356// above could be decided.
357//
358// TODO(user): There is many way to combine the WCE and stratification
359// heuristics. I didn't had time to properly compare the different approach. See
360// the WCE papers for some ideas, but there is many more ways to try to find a
361// lot of core at once and try to minimize the minimum weight of each of the
362// cores.
363SatSolver::Status FindCores(std::vector<Literal> assumptions,
364 std::vector<IntegerValue> assumption_weights,
365 IntegerValue stratified_threshold, Model* model,
366 std::vector<std::vector<Literal>>* cores) {
367 cores->clear();
368 SatSolver* sat_solver = model->GetOrCreate<SatSolver>();
369 TimeLimit* limit = model->GetOrCreate<TimeLimit>();
370 do {
371 if (limit->LimitReached()) return SatSolver::LIMIT_REACHED;
372
373 const SatSolver::Status result =
374 ResetAndSolveIntegerProblem(assumptions, model);
375 if (result != SatSolver::ASSUMPTIONS_UNSAT) return result;
376 std::vector<Literal> core = sat_solver->GetLastIncompatibleDecisions();
377 if (sat_solver->parameters().minimize_core()) {
378 MinimizeCoreWithPropagation(limit, sat_solver, &core);
379 }
380 if (core.size() == 1) {
381 if (!sat_solver->AddUnitClause(core[0].Negated())) {
382 return SatSolver::INFEASIBLE;
383 }
384 }
385 if (core.empty()) return sat_solver->UnsatStatus();
386 cores->push_back(core);
387 if (!sat_solver->parameters().find_multiple_cores()) break;
388
389 // Recover the original indices of the assumptions that are part of the
390 // core.
391 std::vector<int> indices;
392 {
393 absl::btree_set<Literal> temp(core.begin(), core.end());
394 for (int i = 0; i < assumptions.size(); ++i) {
395 if (temp.contains(assumptions[i])) {
396 indices.push_back(i);
397 }
398 }
399 }
400
401 // Remove min_weight from the weights of all the assumptions in the core.
402 //
403 // TODO(user): push right away the objective bound by that much? This should
404 // be better in a multi-threading context as we can share more quickly the
405 // better bound.
406 IntegerValue min_weight = assumption_weights[indices.front()];
407 for (const int i : indices) {
408 min_weight = std::min(min_weight, assumption_weights[i]);
409 }
410 for (const int i : indices) {
411 assumption_weights[i] -= min_weight;
412 }
413
414 // Remove from assumptions all the one with a new weight smaller than the
415 // current stratification threshold and see if we can find another core.
416 int new_size = 0;
417 for (int i = 0; i < assumptions.size(); ++i) {
418 if (assumption_weights[i] < stratified_threshold) continue;
419 assumptions[new_size] = assumptions[i];
420 assumption_weights[new_size] = assumption_weights[i];
421 ++new_size;
422 }
423 assumptions.resize(new_size);
424 assumption_weights.resize(new_size);
425 } while (!assumptions.empty());
426 return SatSolver::ASSUMPTIONS_UNSAT;
427}

431CoreBasedOptimizer::CoreBasedOptimizer(
432 IntegerVariable objective_var,
433 const std::vector<IntegerVariable>& variables,
434 const std::vector<IntegerValue>& coefficients,
435 std::function<void()> feasible_solution_observer, Model* model)
436 : parameters_(model->GetOrCreate<SatParameters>()),
437 sat_solver_(model->GetOrCreate<SatSolver>()),
438 time_limit_(model->GetOrCreate<TimeLimit>()),
439 implications_(model->GetOrCreate<BinaryImplicationGraph>()),
440 integer_trail_(model->GetOrCreate<IntegerTrail>()),
441 integer_encoder_(model->GetOrCreate<IntegerEncoder>()),
442 model_(model),
443 objective_var_(objective_var),
444 feasible_solution_observer_(std::move(feasible_solution_observer)) {
445 CHECK_EQ(variables.size(), coefficients.size());
446 for (int i = 0; i < variables.size(); ++i) {
447 if (coefficients[i] > 0) {
448 terms_.push_back({variables[i], coefficients[i]});
449 } else if (coefficients[i] < 0) {
450 terms_.push_back({NegationOf(variables[i]), -coefficients[i]});
451 } else {
452 continue; // coefficients[i] == 0
453 }
454 terms_.back().depth = 0;
455 }
456
457 // This is used by the "stratified" approach. We will only consider terms with
458 // a weight not lower than this threshold. The threshold will decrease as the
459 // algorithm progress.
460 stratification_threshold_ = parameters_->max_sat_stratification() ==
461 SatParameters::STRATIFICATION_NONE
462 ? IntegerValue(1)
463 : kMaxIntegerValue;
464}

466bool CoreBasedOptimizer::ProcessSolution() {
467 // We don't assume that objective_var is linked with its linear term, so
468 // we recompute the objective here.
469 IntegerValue objective(0);
470 for (ObjectiveTerm& term : terms_) {
471 const IntegerValue value = integer_trail_->LowerBound(term.var);
472 objective += term.weight * value;
473
474 // Also keep in term.cover_ub the minimum value for term.var that we have
475 // seens amongst all the feasible solutions found so far.
476 term.cover_ub = std::min(term.cover_ub, value);
477 }
478
479 // Test that the current objective value fall in the requested objective
480 // domain, which could potentially have holes.
481 if (!integer_trail_->InitialVariableDomain(objective_var_)
482 .Contains(objective.value())) {
483 return true;
484 }
485
486 if (feasible_solution_observer_ != nullptr) {
487 feasible_solution_observer_();
488 }
489 if (parameters_->stop_after_first_solution()) {
490 stop_ = true;
491 }
492
493 // Constrain objective_var. This has a better result when objective_var is
494 // used in an LP relaxation for instance.
495 sat_solver_->Backtrack(0);
496 sat_solver_->SetAssumptionLevel(0);
497 return integer_trail_->Enqueue(
498 IntegerLiteral::LowerOrEqual(objective_var_, objective - 1), {}, {});
499}

501bool CoreBasedOptimizer::PropagateObjectiveBounds() {
502 // We assumes all terms (modulo stratification) at their lower-bound.
503 bool some_bound_were_tightened = true;
504 while (some_bound_were_tightened) {
505 some_bound_were_tightened = false;
506 if (!sat_solver_->ResetToLevelZero()) return false;
507 if (time_limit_->LimitReached()) return true;
508
509 // Compute implied lb.
510 IntegerValue implied_objective_lb(0);
511 for (ObjectiveTerm& term : terms_) {
512 const IntegerValue var_lb = integer_trail_->LowerBound(term.var);
513 term.old_var_lb = var_lb;
514 implied_objective_lb += term.weight * var_lb.value();
515 }
516
517 // Update the objective lower bound with our current bound.
518 if (implied_objective_lb > integer_trail_->LowerBound(objective_var_)) {
519 if (!integer_trail_->Enqueue(IntegerLiteral::GreaterOrEqual(
520 objective_var_, implied_objective_lb),
521 {}, {})) {
522 return false;
523 }
524
525 some_bound_were_tightened = true;
526 }
527
528 // The gap is used to propagate the upper-bound of all variable that are
529 // in the current objective (Exactly like done in the propagation of a
530 // linear constraint with the slack). When this fix a variable to its
531 // lower bound, it is called "hardening" in the max-sat literature. This
532 // has a really beneficial effect on some weighted max-sat problems like
533 // the haplotyping-pedigrees ones.
534 const IntegerValue gap =
535 integer_trail_->UpperBound(objective_var_) - implied_objective_lb;
536
537 for (const ObjectiveTerm& term : terms_) {
538 if (term.weight == 0) continue;
539 const IntegerValue var_lb = integer_trail_->LowerBound(term.var);
540 const IntegerValue var_ub = integer_trail_->UpperBound(term.var);
541 if (var_lb == var_ub) continue;
542
543 // Hardening. This basically just propagate the implied upper bound on
544 // term.var from the current best solution. Note that the gap is
545 // non-negative and the weight positive here. The test is done in order
546 // to avoid any integer overflow provided (ub - lb) do not overflow, but
547 // this is a precondition in our cp-model.
548 if (gap / term.weight < var_ub - var_lb) {
549 some_bound_were_tightened = true;
550 const IntegerValue new_ub = var_lb + gap / term.weight;
551 DCHECK_LT(new_ub, var_ub);
552 DCHECK(!integer_trail_->IsCurrentlyIgnored(term.var));
553 if (!integer_trail_->Enqueue(
554 IntegerLiteral::LowerOrEqual(term.var, new_ub), {}, {})) {
555 return false;
556 }
557 }
558 }
559 }
560 return true;
561}

563// A basic algorithm is to take the next one, or at least the next one
564// that invalidate the current solution. But to avoid corner cases for
565// problem with a lot of terms all with different objective weights (in
566// which case we will kind of introduce only one assumption per loop
567// which is little), we use an heuristic and take the 90% percentile of
568// the unique weights not yet included.
569//
570// TODO(user): There is many other possible heuristics here, and I
571// didn't have the time to properly compare them.
572void CoreBasedOptimizer::ComputeNextStratificationThreshold() {
573 std::vector<IntegerValue> weights;
574 for (ObjectiveTerm& term : terms_) {
575 if (term.weight >= stratification_threshold_) continue;
576 if (term.weight == 0) continue;
577
578 const IntegerValue var_lb = integer_trail_->LevelZeroLowerBound(term.var);
579 const IntegerValue var_ub = integer_trail_->LevelZeroUpperBound(term.var);
580 if (var_lb == var_ub) continue;
581
582 weights.push_back(term.weight);
583 }
584 if (weights.empty()) {
585 stratification_threshold_ = IntegerValue(0);
586 return;
587 }
588
589 gtl::STLSortAndRemoveDuplicates(&weights);
590 stratification_threshold_ =
591 weights[static_cast<int>(std::floor(0.9 * weights.size()))];
592}

594bool CoreBasedOptimizer::CoverOptimization() {
595 if (!sat_solver_->ResetToLevelZero()) return false;
596
597 // We set a fix deterministic time limit per all sub-solve and skip to the
598 // next core if the sum of the subsolve is also over this limit.
599 constexpr double max_dtime_per_core = 0.5;
600 const double old_time_limit = parameters_->max_deterministic_time();
601 parameters_->set_max_deterministic_time(max_dtime_per_core);
602 auto cleanup = ::absl::MakeCleanup([old_time_limit, this]() {
603 parameters_->set_max_deterministic_time(old_time_limit);
604 });
605
606 for (const ObjectiveTerm& term : terms_) {
607 // We currently skip the initial objective terms as there could be many
608 // of them. TODO(user): provide an option to cover-optimize them? I
609 // fear that this will slow down the solver too much though.
610 if (term.depth == 0) continue;
611
612 // Find out the true lower bound of var. This is called "cover
613 // optimization" in some of the max-SAT literature. It can helps on some
614 // problem families and hurt on others, but the overall impact is
615 // positive.
616 const IntegerVariable var = term.var;
617 IntegerValue best =
618 std::min(term.cover_ub, integer_trail_->UpperBound(var));
619
620 // Note(user): this can happen in some corner case because each time we
621 // find a solution, we constrain the objective to be smaller than it, so
622 // it is possible that a previous best is now infeasible.
623 if (best <= integer_trail_->LowerBound(var)) continue;
624
625 // Compute the global deterministic time for this core cover
626 // optimization.
627 const double deterministic_limit =
628 time_limit_->GetElapsedDeterministicTime() + max_dtime_per_core;
629
630 // Simple linear scan algorithm to find the optimal of var.
631 SatSolver::Status result;
632 while (best > integer_trail_->LowerBound(var)) {
633 const Literal assumption = integer_encoder_->GetOrCreateAssociatedLiteral(
634 IntegerLiteral::LowerOrEqual(var, best - 1));
635 result = ResetAndSolveIntegerProblem({assumption}, model_);
636 if (result != SatSolver::FEASIBLE) break;
637
638 best = integer_trail_->LowerBound(var);
639 VLOG(1) << "cover_opt var:" << var << " domain:["
640 << integer_trail_->LevelZeroLowerBound(var) << "," << best << "]";
641 if (!ProcessSolution()) return false;
642 if (!sat_solver_->ResetToLevelZero()) return false;
643 if (stop_ ||
644 time_limit_->GetElapsedDeterministicTime() > deterministic_limit) {
645 break;
646 }
647 }
648 if (result == SatSolver::INFEASIBLE) return false;
649 if (result == SatSolver::ASSUMPTIONS_UNSAT) {
650 if (!sat_solver_->ResetToLevelZero()) return false;
651
652 // TODO(user): If we improve the lower bound of var, we should check
653 // if our global lower bound reached our current best solution in
654 // order to abort early if the optimal is proved.
655 if (!integer_trail_->Enqueue(IntegerLiteral::GreaterOrEqual(var, best),
656 {}, {})) {
657 return false;
658 }
659 }
660 }
661
662 return PropagateObjectiveBounds();
663}

665SatSolver::Status CoreBasedOptimizer::OptimizeWithSatEncoding(
666 const std::vector<Literal>& literals,
667 const std::vector<IntegerVariable>& vars,
668 const std::vector<Coefficient>& coefficients, Coefficient offset) {
669 // Create one initial nodes per variables with cost.
670 // TODO(user): We could create EncodingNode out of IntegerVariable.
671 //
672 // Note that the nodes order and assumptions extracted from it will be stable.
673 // In particular, new nodes will be appended at the end, which make the solver
674 // more likely to find core involving only the first assumptions. This is
675 // important at the beginning so the solver as a chance to find a lot of
676 // non-overlapping small cores without the need to have dedicated
677 // non-overlapping core finder.
678 // TODO(user): It could still be beneficial to add one. Experiments.
679 ObjectiveEncoder encoder(model_);
680 if (vars.empty()) {
681 // All Booleans.
682 for (int i = 0; i < literals.size(); ++i) {
683 CHECK_GT(coefficients[i], 0);
684 encoder.AddBaseNode(
685 EncodingNode::LiteralNode(literals[i], coefficients[i]));
686 }
687 } else {
688 // Use integer encoding.
689 CHECK_EQ(vars.size(), coefficients.size());
690 for (int i = 0; i < vars.size(); ++i) {
691 CHECK_GT(coefficients[i], 0);
692 const IntegerVariable var = vars[i];
693 const IntegerValue var_lb = integer_trail_->LowerBound(var);
694 const IntegerValue var_ub = integer_trail_->UpperBound(var);
695 if (var_ub - var_lb == 1) {
696 const Literal lit = integer_encoder_->GetOrCreateAssociatedLiteral(
697 IntegerLiteral::GreaterOrEqual(var, var_ub));
698 encoder.AddBaseNode(EncodingNode::LiteralNode(lit, coefficients[i]));
699 } else {
700 // TODO(user): This might not be ideal if there are holes in the domain.
701 // It should work by adding duplicates literal, but we should be able to
702 // be more efficient.
703 const int lb = 0;
704 const int ub = static_cast<int>(var_ub.value() - var_lb.value());
705 encoder.AddBaseNode(EncodingNode::GenericNode(
706 lb, ub,
707 [var, var_lb, this](int x) {
708 return integer_encoder_->GetOrCreateAssociatedLiteral(
709 IntegerLiteral::GreaterOrEqual(var,
710 var_lb + IntegerValue(x + 1)));
711 },
712 coefficients[i]));
713 }
714 }
715 }
716
717 // Initialize the bounds.
718 // This is in term of number of variables not at their minimal value.
719 Coefficient lower_bound(0);
720
721 // This is used by the "stratified" approach.
722 Coefficient stratified_lower_bound(0);
723 if (parameters_->max_sat_stratification() !=
724 SatParameters::STRATIFICATION_NONE) {
725 for (EncodingNode* n : encoder.nodes()) {
726 stratified_lower_bound = std::max(stratified_lower_bound, n->weight());
727 }
728 }
729
730 // Start the algorithm.
731 int max_depth = 0;
732 std::string previous_core_info = "";
733 for (int iter = 0;;) {
734 if (time_limit_->LimitReached()) return SatSolver::LIMIT_REACHED;
735 if (!sat_solver_->ResetToLevelZero()) return SatSolver::INFEASIBLE;
736
737 // Note that the objective_var_ upper bound is the one from the "improving"
738 // problem, so if we have a feasible solution, it will be the best solution
739 // objective value - 1.
740 const Coefficient upper_bound(
741 integer_trail_->UpperBound(objective_var_).value() - offset.value());
742 ReduceNodes(upper_bound, &lower_bound, encoder.mutable_nodes(),
743 sat_solver_);
744 const IntegerValue new_obj_lb(lower_bound.value() + offset.value());
745 if (new_obj_lb > integer_trail_->LowerBound(objective_var_)) {
746 if (!integer_trail_->Enqueue(
747 IntegerLiteral::GreaterOrEqual(objective_var_, new_obj_lb), {},
748 {})) {
749 return SatSolver::INFEASIBLE;
750 }
751
752 // Report the improvement.
753 // Note that we have a callback that will do the same, but doing it
754 // earlier allow us to add more information.
755 const int num_bools = sat_solver_->NumVariables();
756 const int num_fixed = sat_solver_->NumFixedVariables();
757 model_->GetOrCreate<SharedResponseManager>()->UpdateInnerObjectiveBounds(
758 absl::StrFormat("bool_core num_cores:%d [%s] assumptions:%u "
759 "depth:%d fixed_bools:%d/%d",
760 iter, previous_core_info, encoder.nodes().size(),
761 max_depth, num_fixed, num_bools),
762 new_obj_lb, integer_trail_->LevelZeroUpperBound(objective_var_));
763 }
764
765 if (parameters_->cover_optimization() && encoder.nodes().size() > 1) {
766 if (ProbeLiteral(
767 encoder.mutable_nodes()->back()->GetAssumption(sat_solver_),
768 sat_solver_)) {
769 previous_core_info = "cover";
770 continue;
771 }
772 }
773
774 // We adapt the stratified lower bound when the gap is small. All literals
775 // with such weight will be in an at_most_one relationship, which will lead
776 // to a nice encoding if we find a core.
777 const Coefficient gap = upper_bound - lower_bound;
778 if (stratified_lower_bound > (gap + 2) / 2) {
779 stratified_lower_bound = (gap + 2) / 2;
780 }
781 std::vector<Literal> assumptions;
782 while (true) {
783 assumptions = ExtractAssumptions(stratified_lower_bound, encoder.nodes(),
784 sat_solver_);
785 if (!assumptions.empty()) break;
786
787 stratified_lower_bound =
788 MaxNodeWeightSmallerThan(encoder.nodes(), stratified_lower_bound);
789 if (stratified_lower_bound > 0) continue;
790
791 // We do not have any assumptions anymore, but we still need to see
792 // if the problem is feasible or not!
793 break;
794 }
795 VLOG(2) << "[Core] #nodes " << encoder.nodes().size()
796 << " #assumptions:" << assumptions.size()
797 << " stratification:" << stratified_lower_bound << " gap:" << gap;
798
799 // Solve under the assumptions.
800 //
801 // TODO(user): Find multiple core like in the "main" algorithm. This is just
802 // trying to solve with assumptions not involving the newly found core.
803 //
804 // TODO(user): With stratification, sometime we just spend too much time
805 // trying to find a feasible solution/prove infeasibility and we could
806 // instead just use stratification=0 to find easty core and improve lower
807 // bound.
808 const SatSolver::Status result =
809 ResetAndSolveIntegerProblem(assumptions, model_);
810 if (result == SatSolver::FEASIBLE) {
811 if (!ProcessSolution()) return SatSolver::INFEASIBLE;
812 if (stop_) return SatSolver::LIMIT_REACHED;
813
814 // If not all assumptions were taken, continue with a lower stratified
815 // bound. Otherwise we have an optimal solution.
816 stratified_lower_bound =
817 MaxNodeWeightSmallerThan(encoder.nodes(), stratified_lower_bound);
818 if (stratified_lower_bound > 0) continue;
819 return SatSolver::INFEASIBLE;
820 }
821 if (result != SatSolver::ASSUMPTIONS_UNSAT) return result;
822
823 // We have a new core.
824 std::vector<Literal> core = sat_solver_->GetLastIncompatibleDecisions();
825 if (parameters_->minimize_core()) {
826 MinimizeCoreWithPropagation(time_limit_, sat_solver_, &core);
827 MinimizeCoreWithSearch(time_limit_, sat_solver_, &core);
828 }
829 FilterAssignedLiteral(sat_solver_->Assignment(), &core);
830 if (core.empty()) return SatSolver::INFEASIBLE;
831
832 // Compute the min weight of all the nodes in the core.
833 // The lower bound will be increased by that much.
834 const Coefficient min_weight = ComputeCoreMinWeight(encoder.nodes(), core);
835 previous_core_info =
836 absl::StrFormat("size:%u mw:%d", core.size(), min_weight.value());
837
838 // We only count an iter when we found a core.
839 ++iter;
840 if (!encoder.ProcessCore(core, min_weight, gap, &previous_core_info)) {
841 return SatSolver::INFEASIBLE;
842 }
843 max_depth = std::max(max_depth, encoder.nodes().back()->depth());
844 }
845
846 return SatSolver::FEASIBLE; // shouldn't reach here.
847}

849void PresolveBooleanLinearExpression(std::vector<Literal>* literals,
850 std::vector<Coefficient>* coefficients,
851 Coefficient* offset) {
852 // Sorting by literal index regroup duplicate or negated literal together.
853 std::vector<std::pair<LiteralIndex, Coefficient>> pairs;
854 const int size = literals->size();
855 for (int i = 0; i < size; ++i) {
856 pairs.push_back({(*literals)[i].Index(), (*coefficients)[i]});
857 }
858 std::sort(pairs.begin(), pairs.end());
859
860 // Merge terms if needed.
861 int new_size = 0;
862 for (const auto& [index, coeff] : pairs) {
863 if (new_size > 0) {
864 if (pairs[new_size - 1].first == index) {
865 pairs[new_size - 1].second += coeff;
866 continue;
867 } else if (pairs[new_size - 1].first == Literal(index).NegatedIndex()) {
868 // The term is coeff *( 1 - X).
869 pairs[new_size - 1].second -= coeff;
870 *offset += coeff;
871 continue;
872 }
873 }
874 pairs[new_size++] = {index, coeff};
875 }
876 pairs.resize(new_size);
877
878 // Rebuild with positive coeff.
879 literals->clear();
880 coefficients->clear();
881 for (const auto& [index, coeff] : pairs) {
882 if (coeff > 0) {
883 literals->push_back(Literal(index));
884 coefficients->push_back(coeff);
885 } else if (coeff < 0) {
886 // coeff * X = coeff - coeff * (1 - X).
887 *offset += coeff;
888 literals->push_back(Literal(index).Negated());
889 coefficients->push_back(-coeff);
890 }
891 }
892}

894void CoreBasedOptimizer::PresolveObjectiveWithAtMostOne(
895 std::vector<Literal>* literals, std::vector<Coefficient>* coefficients,
896 Coefficient* offset) {
897 // This contains non-negative value. If a literal has negative weight, then
898 // we just put a positive weight on its negation and update the offset.
899 const int num_literals = implications_->literal_size();
900 absl::StrongVector<LiteralIndex, Coefficient> weights(num_literals);
901 absl::StrongVector<LiteralIndex, bool> is_candidate(num_literals);
902
903 // For now, we do not use weight. Note that finding the at most on in the
904 // creation order of the variable make a HUGE difference on the max-sat frb
905 // family.
906 //
907 // TODO(user): We can assign preferences to literals to favor certain at most
908 // one instead of other. For now we don't, so ExpandAtMostOneWithWeight() will
909 // kind of randomize the expansion amongst possible choices.
910 absl::StrongVector<LiteralIndex, double> preferences;
911
912 // Collect all literals with "negative weights", we will try to find at most
913 // one between them.
914 std::vector<Literal> candidates;
915 const int num_terms = literals->size();
916 for (int i = 0; i < num_terms; ++i) {
917 const Literal lit = (*literals)[i];
918 const Coefficient coeff = (*coefficients)[i];
919
920 // For now we know the input only has positive weight, but it is easy to
921 // adapt if needed.
922 CHECK_GT(coeff, 0);
923 weights[lit.Index()] = coeff;
924
925 candidates.push_back(lit.Negated());
926 is_candidate[lit.NegatedIndex()] = true;
927 }
928
929 int num_at_most_ones = 0;
930 Coefficient overall_lb_increase(0);
931
932 std::vector<Literal> at_most_one;
933 std::vector<std::pair<Literal, Coefficient>> new_obj_terms;
934 implications_->ResetWorkDone();
935 for (const Literal root : candidates) {
936 if (weights[root.NegatedIndex()] == 0) continue;
937 if (implications_->WorkDone() > 1e8) continue;
938
939 // We never put weight on both a literal and its negation.
940 CHECK_EQ(weights[root.Index()], 0);
941
942 // Note that for this to be as exhaustive as possible, the probing needs
943 // to have added binary clauses corresponding to lvl0 propagation.
944 at_most_one =
945 implications_->ExpandAtMostOneWithWeight</*use_weight=*/false>(
946 {root}, is_candidate, preferences);
947 if (at_most_one.size() <= 1) continue;
948 ++num_at_most_ones;
949
950 // Change the objective weights. Note that all the literal in the at most
951 // one will not be processed again since the weight of their negation will
952 // be zero after this step.
953 Coefficient max_coeff(0);
954 Coefficient lb_increase(0);
955 for (const Literal lit : at_most_one) {
956 const Coefficient coeff = weights[lit.NegatedIndex()];
957 lb_increase += coeff;
958 max_coeff = std::max(max_coeff, coeff);
959 }
960 lb_increase -= max_coeff;
961
962 *offset += lb_increase;
963 overall_lb_increase += lb_increase;
964
965 for (const Literal lit : at_most_one) {
966 is_candidate[lit.Index()] = false;
967 const Coefficient new_weight = max_coeff - weights[lit.NegatedIndex()];
968 CHECK_EQ(weights[lit.Index()], 0);
969 weights[lit.Index()] = new_weight;
970 weights[lit.NegatedIndex()] = 0;
971 if (new_weight > 0) {
972 // TODO(user): While we autorize this to be in future at most one, it
973 // will not appear in the "literal" list. We might also want to continue
974 // until we reached the fix point.
975 is_candidate[lit.NegatedIndex()] = true;
976 }
977 }
978
979 // Create a new Boolean with weight max_coeff.
980 const Literal new_lit(sat_solver_->NewBooleanVariable(), true);
981 new_obj_terms.push_back({new_lit, max_coeff});
982
983 // The new boolean is true only if all the one in the at most one are false.
984 at_most_one.push_back(new_lit);
985 sat_solver_->AddProblemClause(at_most_one);
986 is_candidate.resize(implications_->literal_size(), false);
987 preferences.resize(implications_->literal_size(), 1.0);
988 }
989
990 if (overall_lb_increase > 0) {
991 // Report new bounds right away with extra information.
992 model_->GetOrCreate<SharedResponseManager>()->UpdateInnerObjectiveBounds(
993 absl::StrFormat("am1_presolve num_literals:%d num_am1:%d "
994 "increase:%lld work_done:%lld",
995 (int)candidates.size(), num_at_most_ones,
996 overall_lb_increase.value(), implications_->WorkDone()),
997 IntegerValue(offset->value()),
998 integer_trail_->LevelZeroUpperBound(objective_var_));
999 }
1000
1001 // Reconstruct the objective.
1002 literals->clear();
1003 coefficients->clear();
1004 for (const Literal root : candidates) {
1005 if (weights[root.Index()] > 0) {
1006 CHECK_EQ(weights[root.NegatedIndex()], 0);
1007 literals->push_back(root);
1008 coefficients->push_back(weights[root.Index()]);
1009 }
1010 if (weights[root.NegatedIndex()] > 0) {
1011 CHECK_EQ(weights[root.Index()], 0);
1012 literals->push_back(root.Negated());
1013 coefficients->push_back(weights[root.NegatedIndex()]);
1014 }
1015 }
1016 for (const auto& [lit, coeff] : new_obj_terms) {
1017 literals->push_back(lit);
1018 coefficients->push_back(coeff);
1019 }
1020}

1022SatSolver::Status CoreBasedOptimizer::Optimize() {
1023 // Hack: If the objective is fully Boolean, we use the
1024 // OptimizeWithSatEncoding() version as it seems to be better.
1025 //
1026 // TODO(user): Try to understand exactly why and merge both code path.
1027 if (!parameters_->interleave_search()) {
1028 Coefficient offset(0);
1029 std::vector<Literal> literals;
1030 std::vector<IntegerVariable> vars;
1031 std::vector<Coefficient> coefficients;
1032 bool all_booleans = true;
1033 IntegerValue range(0);
1034 for (const ObjectiveTerm& term : terms_) {
1035 const IntegerVariable var = term.var;
1036 const IntegerValue coeff = term.weight;
1037 const IntegerValue lb = integer_trail_->LowerBound(var);
1038 const IntegerValue ub = integer_trail_->UpperBound(var);
1039 offset += Coefficient((lb * coeff).value());
1040 if (lb == ub) continue;
1041
1042 vars.push_back(var);
1043 coefficients.push_back(Coefficient(coeff.value()));
1044 if (ub - lb == 1) {
1045 literals.push_back(integer_encoder_->GetOrCreateAssociatedLiteral(
1046 IntegerLiteral::GreaterOrEqual(var, ub)));
1047 } else {
1048 all_booleans = false;
1049 range += ub - lb;
1050 }
1051 }
1052 if (all_booleans) {
1053 // In some corner case, it is possible the GetOrCreateAssociatedLiteral()
1054 // returns identical or negated literal of another term. We don't support
1055 // this below, so we need to make sure this is not the case.
1056 PresolveBooleanLinearExpression(&literals, &coefficients, &offset);
1057
1058 // TODO(user): It might be interesting to redo this kind of presolving
1059 // once high cost booleans have been fixed as we might have more at most
1060 // one between literal in the objective by then.
1061 //
1062 // Or alternatively, we could try this or something like it on the
1063 // literals from the cores as they are found. We should probably make
1064 // sure that if it exist, a core of size two is always added. And for
1065 // such core, we can always try to see if the "at most one" can be
1066 // extended.
1067 PresolveObjectiveWithAtMostOne(&literals, &coefficients, &offset);
1068 return OptimizeWithSatEncoding(literals, {}, coefficients, offset);
1069 }
1070 if (range < 1e8) {
1071 return OptimizeWithSatEncoding({}, vars, coefficients, offset);
1072 }
1073 }
1074
1075 // TODO(user): The core is returned in the same order as the assumptions,
1076 // so we don't really need this map, we could just do a linear scan to
1077 // recover which node are part of the core. This however needs to be properly
1078 // unit tested before usage.
1079 absl::btree_map<LiteralIndex, int> literal_to_term_index;
1080
1081 // Start the algorithm.
1082 stop_ = false;
1083 while (true) {
1084 // TODO(user): This always resets the solver to level zero.
1085 // Because of that we don't resume a solve in "chunk" perfectly. Fix.
1086 if (!PropagateObjectiveBounds()) return SatSolver::INFEASIBLE;
1087 if (time_limit_->LimitReached()) return SatSolver::LIMIT_REACHED;
1088
1089 // Bulk cover optimization.
1090 //
1091 // TODO(user): If the search is aborted during this phase and we solve in
1092 // "chunk", we don't resume perfectly from where it was. Fix.
1093 if (parameters_->cover_optimization()) {
1094 if (!CoverOptimization()) return SatSolver::INFEASIBLE;
1095 if (stop_) return SatSolver::LIMIT_REACHED;
1096 }
1097
1098 // We assumes all terms (modulo stratification) at their lower-bound.
1099 std::vector<int> term_indices;
1100 std::vector<IntegerLiteral> integer_assumptions;
1101 std::vector<IntegerValue> assumption_weights;
1102 IntegerValue objective_offset(0);
1103 bool some_assumptions_were_skipped = false;
1104 for (int i = 0; i < terms_.size(); ++i) {
1105 const ObjectiveTerm term = terms_[i];
1106
1107 // TODO(user): These can be simply removed from the list.
1108 if (term.weight == 0) continue;
1109
1110 // Skip fixed terms.
1111 // We still keep them around for a proper lower-bound computation.
1112 //
1113 // TODO(user): we could keep an objective offset instead.
1114 const IntegerValue var_lb = integer_trail_->LowerBound(term.var);
1115 const IntegerValue var_ub = integer_trail_->UpperBound(term.var);
1116 if (var_lb == var_ub) {
1117 objective_offset += term.weight * var_lb.value();
1118 continue;
1119 }
1120
1121 // Only consider the terms above the threshold.
1122 if (term.weight >= stratification_threshold_) {
1123 integer_assumptions.push_back(
1124 IntegerLiteral::LowerOrEqual(term.var, var_lb));
1125 assumption_weights.push_back(term.weight);
1126 term_indices.push_back(i);
1127 } else {
1128 some_assumptions_were_skipped = true;
1129 }
1130 }
1131
1132 // No assumptions with the current stratification? use the next one.
1133 if (term_indices.empty() && some_assumptions_were_skipped) {
1134 ComputeNextStratificationThreshold();
1135 continue;
1136 }
1137
1138 // If there is only one or two assumptions left, we switch the algorithm.
1139 if (term_indices.size() <= 2 && !some_assumptions_were_skipped) {
1140 VLOG(1) << "Switching to linear scan...";
1141 if (!already_switched_to_linear_scan_) {
1142 already_switched_to_linear_scan_ = true;
1143 std::vector<IntegerVariable> constraint_vars;
1144 std::vector<int64_t> constraint_coeffs;
1145 for (const int index : term_indices) {
1146 constraint_vars.push_back(terms_[index].var);
1147 constraint_coeffs.push_back(terms_[index].weight.value());
1148 }
1149 constraint_vars.push_back(objective_var_);
1150 constraint_coeffs.push_back(-1);
1151 model_->Add(WeightedSumLowerOrEqual(constraint_vars, constraint_coeffs,
1152 -objective_offset.value()));
1153 }
1154
1155 return MinimizeIntegerVariableWithLinearScanAndLazyEncoding(
1156 objective_var_, feasible_solution_observer_, model_);
1157 }
1158
1159 // Display the progress.
1160 if (VLOG_IS_ON(1)) {
1161 int max_depth = 0;
1162 for (const ObjectiveTerm& term : terms_) {
1163 max_depth = std::max(max_depth, term.depth);
1164 }
1165 const int64_t lb = integer_trail_->LowerBound(objective_var_).value();
1166 const int64_t ub = integer_trail_->UpperBound(objective_var_).value();
1167 const int gap =
1168 lb == ub
1169 ? 0
1170 : static_cast<int>(std::ceil(
1171 100.0 * (ub - lb) / std::max(std::abs(ub), std::abs(lb))));
1172 VLOG(1) << absl::StrCat("unscaled_next_obj_range:[", lb, ",", ub,
1173 "]"
1174 " gap:",
1175 gap, "%", " assumptions:", term_indices.size(),
1176 " strat:", stratification_threshold_.value(),
1177 " depth:", max_depth,
1178 " bool: ", sat_solver_->NumVariables());
1179 }
1180
1181 // Convert integer_assumptions to Literals.
1182 std::vector<Literal> assumptions;
1183 literal_to_term_index.clear();
1184 for (int i = 0; i < integer_assumptions.size(); ++i) {
1185 assumptions.push_back(integer_encoder_->GetOrCreateAssociatedLiteral(
1186 integer_assumptions[i]));
1187
1188 // Tricky: In some rare case, it is possible that the same literal
1189 // correspond to more that one assumptions. In this case, we can just
1190 // pick one of them when converting back a core to term indices.
1191 //
1192 // TODO(user): We can probably be smarter about the cost of the
1193 // assumptions though.
1194 literal_to_term_index[assumptions.back().Index()] = term_indices[i];
1195 }
1196
1197 // Solve under the assumptions.
1198 //
1199 // TODO(user): If the "search" is interrupted while computing cores, we
1200 // currently do not resume it flawlessly. We however add any cores we found
1201 // before aborting.
1202 std::vector<std::vector<Literal>> cores;
1203 const SatSolver::Status result =
1204 FindCores(assumptions, assumption_weights, stratification_threshold_,
1205 model_, &cores);
1206 if (result == SatSolver::INFEASIBLE) return SatSolver::INFEASIBLE;
1207 if (result == SatSolver::FEASIBLE) {
1208 if (!ProcessSolution()) return SatSolver::INFEASIBLE;
1209 if (stop_) return SatSolver::LIMIT_REACHED;
1210 if (cores.empty()) {
1211 ComputeNextStratificationThreshold();
1212 if (stratification_threshold_ == 0) return SatSolver::INFEASIBLE;
1213 continue;
1214 }
1215 }
1216
1217 // Process the cores by creating new variables and transferring the minimum
1218 // weight of each core to it.
1219 if (!sat_solver_->ResetToLevelZero()) return SatSolver::INFEASIBLE;
1220 for (const std::vector<Literal>& core : cores) {
1221 // This just increase the lower-bound of the corresponding node.
1222 // TODO(user): Maybe the solver should do it right away.
1223 if (core.size() == 1) {
1224 if (!sat_solver_->AddUnitClause(core[0].Negated())) {
1225 return SatSolver::INFEASIBLE;
1226 }
1227 continue;
1228 }
1229
1230 // Compute the min weight of all the terms in the core. The lower bound
1231 // will be increased by that much because at least one assumption in the
1232 // core must be true. This is also why we can start at 1 for new_var_lb.
1233 bool ignore_this_core = false;
1234 IntegerValue min_weight = kMaxIntegerValue;
1235 IntegerValue max_weight(0);
1236 IntegerValue new_var_lb(1);
1237 IntegerValue new_var_ub(0);
1238 int new_depth = 0;
1239 for (const Literal lit : core) {
1240 const int index = literal_to_term_index.at(lit.Index());
1241
1242 // When this happen, the core is now trivially "minimized" by the new
1243 // bound on this variable, so there is no point in adding it.
1244 if (terms_[index].old_var_lb <
1245 integer_trail_->LowerBound(terms_[index].var)) {
1246 ignore_this_core = true;
1247 break;
1248 }
1249
1250 const IntegerValue weight = terms_[index].weight;
1251 min_weight = std::min(min_weight, weight);
1252 max_weight = std::max(max_weight, weight);
1253 new_depth = std::max(new_depth, terms_[index].depth + 1);
1254 new_var_lb += integer_trail_->LowerBound(terms_[index].var);
1255 new_var_ub += integer_trail_->UpperBound(terms_[index].var);
1256 }
1257 if (ignore_this_core) continue;
1258
1259 VLOG(1) << absl::StrFormat(
1260 "core:%u weight:[%d,%d] domain:[%d,%d] depth:%d", core.size(),
1261 min_weight.value(), max_weight.value(), new_var_lb.value(),
1262 new_var_ub.value(), new_depth);
1263
1264 // We will "transfer" min_weight from all the variables of the core
1265 // to a new variable.
1266 const IntegerVariable new_var =
1267 integer_trail_->AddIntegerVariable(new_var_lb, new_var_ub);
1268 terms_.push_back({new_var, min_weight, new_depth});
1269 terms_.back().cover_ub = new_var_ub;
1270
1271 // Sum variables in the core <= new_var.
1272 {
1273 std::vector<IntegerVariable> constraint_vars;
1274 std::vector<int64_t> constraint_coeffs;
1275 for (const Literal lit : core) {
1276 const int index = literal_to_term_index.at(lit.Index());
1277 terms_[index].weight -= min_weight;
1278 constraint_vars.push_back(terms_[index].var);
1279 constraint_coeffs.push_back(1);
1280 }
1281 constraint_vars.push_back(new_var);
1282 constraint_coeffs.push_back(-1);
1283 model_->Add(
1284 WeightedSumLowerOrEqual(constraint_vars, constraint_coeffs, 0));
1285 }
1286 }
1287
1288 // Abort if we reached the time limit. Note that we still add any cores we
1289 // found in case the solve is split in "chunk".
1290 if (result == SatSolver::LIMIT_REACHED) return result;
1291 }
1292}
