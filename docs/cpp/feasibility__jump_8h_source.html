42// This class lazily caches the results of `compute_jump(var)` which returns a
43// <delta, score> pair.
44// Variables' scores can be manually modified using MutableScores (if the
45// optimal jump is known not to change), or marked for recomputation on the next
46// call to GetJump(var) by calling Recompute.
47class JumpTable {
48 public:
49 explicit JumpTable(
50 absl::AnyInvocable<std::pair<int64_t, double>(int)> compute_jump);
51 void RecomputeAll(int num_variables);
52
53 // Gets the current jump delta and score, recomputing if necessary.
54 std::pair<int64_t, double> GetJump(int var);
55 // If the new optimum value and score is known, users can update it directly.
56 // e.g. after weight rescaling, or after changing a binary variable.
57 void SetJump(int var, int64_t delta, double score);
58 // Recompute the jump for `var` when `GetJump(var)` is next called.
59 void Recompute(int var);
60 // Returns true if the jump score for `var` might be negative.
61 bool PossiblyGood(int var) const;
62
63 // Advanced usage, allows users to read possibly stale deltas for incremental
64 // score updates.
65 absl::Span<const int64_t> Deltas() const {
66 return absl::MakeConstSpan(deltas_);
67 }
68 absl::Span<const double> Scores() const {
69 return absl::MakeConstSpan(scores_);
70 }
71
72 absl::Span<double> MutableScores() { return absl::MakeSpan(scores_); }
73
74 // Note if you have very high weights (e.g. when using decay), the tolerances
75 // in this function are likely too tight.
76 bool JumpIsUpToDate(int var); // For debugging and testing.
77
78 private:
79 absl::AnyInvocable<std::pair<int64_t, double>(int)> compute_jump_;
80
81 // For each variable, we store:
82 // - A jump delta which represents a change in variable value:
83 // new_value = old + delta, which is non-zero except for fixed variables.
84 // - The associated weighted feasibility violation change if we take this
85 // jump.
86 std::vector<int64_t> deltas_;
87 std::vector<double> scores_;
88 std::vector<bool> needs_recomputation_;
89};

91// Implements and heuristic similar to the one described in the paper:
92// "Feasibility Jump: an LP-free Lagrangian MIP heuristic", Bj√∏rnar
93// Luteberget, Giorgio Sartor, 2023, Mathematical Programming Computation.
94//
95// This is basically a Guided local search (GLS) with a nice algo to know what
96// value an integer variable should move to (its jump value). For binary, it
97// can only be swapped, so the situation is easier.
98//
99// TODO(user): If we have more than one of these solver, we might want to share
100// the evaluator memory between them. Right now we basically keep a copy of the
101// model and its transpose for each FeasibilityJumpSolver.

312// This class helps keep track of moves that change more than one variable.
313// Mainly this class keeps track of how to backtrack back to the local minimum
314// as you make a sequence of exploratory moves, so in order to commit a compound
315// move, you just need to call `Clear` instead of Backtracking over the changes.
316class CompoundMoveBuilder {
317 public:
318 CompoundMoveBuilder(LsEvaluator* evaluator, int num_variables)
319 : evaluator_(evaluator), var_on_stack_(num_variables, false) {}
320
321 // Adds an atomic move to the stack.
322 // `var` must not be on the stack (this is DCHECKed).
323 void Push(int var, int64_t prev_value, double score);
324
325 // Sets var, val and score to a move that will revert the most recent atomic
326 // move on the stack, and pops this move from the stack.
327 // Returns false if the stack is empty.
328 bool Backtrack(int* var, int64_t* value, double* score);
329
330 // Removes all moves on the stack.
331 void Clear();
332
333 // Returns the number of variables in the move.
334 int Size() const { return stack_.size(); }
335
336 // Returns true if this var has been set in this move already,
337 bool OnStack(int var) const;
338
339 // Returns the sum of scores of atomic moved pushed to this compound move.
340 double Score() const {
341 return stack_.empty() ? 0.0 : stack_.back().cumulative_score;
342 }
343
344 double BestChildScore() const {
345 return stack_.empty() ? 0.0 : stack_.back().best_child_score;
346 }
347
348 // Returns the number of backtracks to any ancestor of the current leaf.
349 int Discrepancy() const {
350 return stack_.empty() ? 0 : stack_.back().discrepancy;
351 }
352
353 // Returns the number of backtracking moves that have been applied.
354 int NumBacktracks() const { return num_backtracks_; }
355
356 // Returns true if all prev_values on the stack are in the appropriate domain.
357 bool StackValuesInDomains(absl::Span<const Domain> var_domains) const;
358
359 private:
360 struct UnitMove {
361 int var;
362 int64_t prev_value;
363 // Note that this stores the score of reverting to prev_value.
364 double score;
365 // Instead of tracking this on the compound move, we store the partial sums
366 // to avoid numerical issues causing negative scores after backtracking.
367 double cumulative_score;
368
369 // Used to avoid infinite loops, this tracks the best score of any immediate
370 // children (and not deeper descendants) to avoid re-exploring the same
371 // prefix.
372 double best_child_score = 0.0;
373 int discrepancy = 0;
374 };
375 LsEvaluator* evaluator_;
376 std::vector<bool> var_on_stack_;
377 std::vector<UnitMove> stack_;
378
379 int64_t num_backtracks_ = 0;
380};
