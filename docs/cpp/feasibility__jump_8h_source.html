37// Implements and heuristic similar to the one described in the paper:
38// "Feasibility Jump: an LP-free Lagrangian MIP heuristic", Bj√∏rnar
39// Luteberget, Giorgio Sartor, 2023, Mathematical Programming Computation.
40//
41// This is basically a Guided local search (GLS) with a nice algo to know what
42// value an integer variable should move to (its jump value). For binary, it
43// can only be swapped, so the situation is easier.
44//
45// TODO(user): If we have more than one of these solver, we might want to share
46// the evaluator memory between them. Right now we basically keep a copy of the
47// model and its transpose for each FeasibilityJumpSolver.
48class FeasibilityJumpSolver : public SubSolver {
49 public:
50 FeasibilityJumpSolver(const std::string name, SubSolver::SubsolverType type,
51 const LinearModel* linear_model, SatParameters params,
52 ModelSharedTimeLimit* shared_time_limit,
53 SharedResponseManager* shared_response,
54 SharedBoundsManager* shared_bounds,
55 SharedStatistics* shared_stats)
56 : SubSolver(name, type),
57 linear_model_(linear_model),
58 params_(params),
59 shared_time_limit_(shared_time_limit),
60 shared_response_(shared_response),
61 shared_bounds_(shared_bounds),
62 shared_stats_(shared_stats),
63 random_(params_) {}
64
65 // If VLOG_IS_ON(1), it will export a bunch of statistics.
66 ~FeasibilityJumpSolver() override;
67
68 // No synchronization needed for TaskIsAvailable().
69 void Synchronize() final {}
70
71 // Note that this should only returns true if there is a need to delete this
72 // subsolver early to reclaim memory, otherwise we will not properly have the
73 // stats.
74 //
75 // TODO(user): Save the logging stats before deletion.
76 bool IsDone() final {
77 // Tricky: we cannot delete something if there is a task in flight, we will
78 // have to wait.
79 if (task_generated_.load()) return false;
80 if (!model_is_supported_.load()) return true;
81
82 // We are done after the first task is done in the FIRST_SOLUTION mode.
83 return type() == SubSolver::FIRST_SOLUTION &&
84 shared_response_->first_solution_solvers_should_stop()->load();
85 }
86
87 bool TaskIsAvailable() final {
88 if (task_generated_.load()) return false;
89 if (shared_response_->ProblemIsSolved()) return false;
90 if (shared_time_limit_->LimitReached()) return false;
91
92 return (shared_response_->SolutionsRepository().NumSolutions() > 0) ==
93 (type() == SubSolver::INCOMPLETE);
94 }
