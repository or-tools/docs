24// As used below, we define "#variables" = size(VariablesProto.ids).
25message VariablesProto {
26 // Must be nonnegative and strictly increasing. The max(int64) value can't be
27 // used.
28 repeated int64 ids = 1;
29 // Should have length equal to #variables, values in [-inf, inf).
30 repeated double lower_bounds = 2;
31 // Should have length equal to #variables, values in (-inf, inf].
32 repeated double upper_bounds = 3;
33 // Should have length equal to #variables. Value is false for continuous
34 // variables and true for integer variables.
35 repeated bool integers = 4;
36 // If not set, assumed to be all empty strings. Otherwise, should have length
37 // equal to #variables.
38 //
39 // All nonempty names must be distinct. TODO(b/169575522): we may relax this.
40 repeated string names = 5;
41}

43message ObjectiveProto {
44 // false is minimize, true is maximize
45 bool maximize = 1;
46 double offset = 2;
47
48 // ObjectiveProto terms that are linear in the decision variables.
49 //
50 // Requirements:
51 // * linear_coefficients.ids are elements of VariablesProto.ids.
52 // * VariablesProto not specified correspond to zero.
53 // * linear_coefficients.values must all be finite.
54 // * linear_coefficients.values can be zero, but this just wastes space.
55 SparseDoubleVectorProto linear_coefficients = 3;
56
57 // Objective terms that are quadratic in the decision variables.
58 //
59 // Requirements in addition to those on SparseDoubleMatrixProto messages:
60 // * Each element of quadratic_coefficients.row_ids and each element of
61 // quadratic_coefficients.column_ids must be an element of
62 // VariablesProto.ids.
63 // * The matrix must be upper triangular: for each i,
64 // quadratic_coefficients.row_ids[i] <=
65 // quadratic_coefficients.column_ids[i].
66 //
67 // Notes:
68 // * Terms not explicitly stored have zero coefficient.
69 // * Elements of quadratic_coefficients.coefficients can be zero, but this
70 // just wastes space.
71 SparseDoubleMatrixProto quadratic_coefficients = 4;
72
73 // Parent messages may have uniqueness requirements on this field; e.g., see
74 // ModelProto.objectives and AuxiliaryObjectivesUpdatesProto.new_objectives.
75 string name = 5;
76
77 // For multi-objective problems, the priority of this objective relative to
78 // the others (lower is more important). This value must be nonnegative.
79 // Furthermore, each objective priority in the model must be distinct at solve
80 // time. This condition is not validated at the proto level, so models may
81 // temporarily have objectives with the same priority.
82 int64 priority = 6;
83}

102// A single quadratic constraint of the form:
103// lb <= sum{linear_terms} + sum{quadratic_terms} <= ub.
104//
105// If a variable involved in this constraint is deleted, it is treated as if it
106// were set to zero.
107message QuadraticConstraintProto {
108 // Terms that are linear in the decision variables.
109 //
110 // In addition to requirements on SparseDoubleVectorProto messages we require
111 // that:
112 // * linear_terms.ids are elements of VariablesProto.ids.
113 // * linear_terms.values must all be finite and not-NaN.
114 //
115 // Notes:
116 // * Variable ids omitted have a corresponding coefficient of zero.
117 // * linear_terms.values can be zero, but this just wastes space.
118 SparseDoubleVectorProto linear_terms = 1;
119
120 // Terms that are quadratic in the decision variables.
121 //
122 // In addition to requirements on SparseDoubleMatrixProto messages we require
123 // that:
124 // * Each element of quadratic_terms.row_ids and each element of
125 // quadratic_terms.column_ids must be an element of VariablesProto.ids.
126 // * The matrix must be upper triangular: for each i,
127 // quadratic_terms.row_ids[i] <= quadratic_terms.column_ids[i].
128 //
129 // Notes:
130 // * Terms not explicitly stored have zero coefficient.
131 // * Elements of quadratic_terms.coefficients can be zero, but this just
132 // wastes space.
133 SparseDoubleMatrixProto quadratic_terms = 2;
134
135 // Must have value in [-inf, inf), and be less than or equal to `upper_bound`.
136 double lower_bound = 3;
137
138 // Must have value in (-inf, inf], and be greater than or equal to
139 // `lower_bound`.
140 double upper_bound = 4;
141
142 // Parent messages may have uniqueness requirements on this field; e.g., see
143 // ModelProto.quadratic_constraints and
144 // QuadraticConstraintUpdatesProto.new_constraints.
145 string name = 5;
146}

148// Data for representing a single SOS1 or SOS2 constraint.
149//
150// If a variable involved in this constraint is deleted, it is treated as if it
151// were set to zero.
152message SosConstraintProto {
153 // The expressions over which to apply the SOS constraint:
154 // * SOS1: At most one element takes a nonzero value.
155 // * SOS2: At most two elements take nonzero values, and they must be
156 // adjacent in the repeated ordering.
157 repeated LinearExpressionProto expressions = 1;
158
159 // Either empty or of equal length to expressions. If empty, default weights
160 // are 1, 2, ...
161 // If present, the entries must be unique.
162 repeated double weights = 2;
163
164 // Parent messages may have uniqueness requirements on this field; e.g., see
165 // ModelProto.sos1_constraints and SosConstraintUpdatesProto.new_constraints.
166 string name = 3;
167}

169// Data for representing a single indicator constraint of the form:
170// Variable(indicator_id) = (activate_on_zero ? 0 : 1) --> expression sense rhs
171//
172// If a variable involved in this constraint is deleted, it is treated as if it
173// were set to zero. In particular, deleting the indicator variable means that
174// the indicator constraint is vacuous.
175message IndicatorConstraintProto {
176 // An ID corresponding to a binary variable, or unset. If unset, the indicator
177 // constraint is ignored. If set, we require that:
178 // * VariablesProto.integers[indicator_id] = true,
179 // * VariablesProto.lower_bounds[indicator_id] >= 0,
180 // * VariablesProto.upper_bounds[indicator_id] <= 1.
181 // These conditions are not validated by MathOpt, but if not satisfied will
182 // lead to the solver returning an error upon solving.
183 optional int64 indicator_id = 1;
184
185 // If true, then if the indicator variable takes value 0, the implied
186 // constraint must hold. Otherwise, if the indicator variable takes value 1,
187 // then the implied constraint must hold.
188 bool activate_on_zero = 6;
189
190 // Must be a valid linear expression with respect to the containing model:
191 // * All stated conditions on `SparseDoubleVectorProto`,
192 // * All elements of `expression.values` must be finite,
193 // * `expression.ids` are a subset of `VariablesProto.ids`.
194 SparseDoubleVectorProto expression = 2;
195
196 // Must have value in [-inf, inf); cannot be NaN.
197 double lower_bound = 3;
198
199 // Must have value in (-inf, inf]; cannot be NaN.
200 double upper_bound = 4;
201
202 // Parent messages may have uniqueness requirements on this field; e.g., see
203 // `ModelProto.indicator_constraints` and
204 // `IndicatorConstraintUpdatesProto.new_constraints`.
205 string name = 5;
206}

208// An optimization problem. For full details, see go/mathopt-model.
209//
210// MathOpt supports:
211// - Continuous and integer decision variables with optional finite bounds.
212// - Linear and quadratic objectives (single or multiple objectives), either
213// minimized or maximized.
214// - A number of constraints types, including:
215// * Linear constraints
216// * Quadratic constraints
217// * Logical constraints
218// > SOS1 and SOS2 constraints
219// > Indicator constraints
220//
221// By default, constraints are represented in "id-to-data" maps. However, we
222// represent linear constraints in a more efficient "struct-of-arrays" format.
223message ModelProto {
224 string name = 1;
225 VariablesProto variables = 2;
226
227 // The primary objective in the model.
228 ObjectiveProto objective = 3;
229
230 // Auxiliary objectives for use in multi-objective models.
231 //
232 // Map key IDs must be in [0, max(int64)). Each priority, and each nonempty
233 // name, must be unique and also distinct from the primary `objective`.
234 map<int64, ObjectiveProto> auxiliary_objectives = 10;
235
236 LinearConstraintsProto linear_constraints = 4;
237
238 // The variable coefficients for the linear constraints.
239 //
240 // If a variable involved in this constraint is deleted, it is treated as if
241 // it were set to zero.
242 //
243 // Requirements:
244 // * linear_constraint_matrix.row_ids are elements of linear_constraints.ids.
245 // * linear_constraint_matrix.column_ids are elements of variables.ids.
246 // * Matrix entries not specified are zero.
247 // * linear_constraint_matrix.values must all be finite.
248 SparseDoubleMatrixProto linear_constraint_matrix = 5;
249
250 // Mapped constraints (i.e., stored in "constraint ID"-to-"constraint data"
251 // map). For each subsequent submessage, we require that:
252 // * Each key is in [0, max(int64)).
253 // * Each key is unique in its respective map (but not necessarily across
254 // constraint types)
255 // * Each value contains a name field (called `name`), and each nonempty
256 // name must be distinct across all map entries (but not necessarily
257 // across constraint types).
258
259 // Quadratic constraints in the model.
260 map<int64, QuadraticConstraintProto> quadratic_constraints = 6;
261
262 // SOS1 constraints in the model, which constrain that at most one
263 // `expression` can be nonzero. The optional `weights` entries are an
264 // implementation detail used by the solver to (hopefully) converge more
265 // quickly. In more detail, solvers may (or may not) use these weights to
266 // select branching decisions that produce "balanced" children nodes.
267 map<int64, SosConstraintProto> sos1_constraints = 7;
268
269 // SOS2 constraints in the model, which constrain that at most two entries of
270 // `expression` can be nonzero, and they must be adjacent in their ordering.
271 // If no `weights` are provided, this ordering is their linear ordering in the
272 // `expressions` list; if `weights` are presented, the ordering is taken with
273 // respect to these values in increasing order.
274 map<int64, SosConstraintProto> sos2_constraints = 8;
275
276 // Indicator constraints in the model, which enforce that, if a binary
277 // "indicator variable" is set to one, then an "implied constraint" must hold.
278 map<int64, IndicatorConstraintProto> indicator_constraints = 9;
