299absl::Status CheckSolverSpecificOutputEmpty(const SolveResultProto& result) {
300 if (result.solver_specific_output_case() ==
301 SolveResultProto::SOLVER_SPECIFIC_OUTPUT_NOT_SET) {
302 return absl::OkStatus();
303 }
304 return util::InvalidArgumentErrorBuilder()
305 << "cannot set solver specific output twice, was already "
306 << static_cast<int>(result.solver_specific_output_case());
307}

309absl::StatusOr<SolveResultProto> SolveResult::Proto() const {
310 SolveResultProto result;
311 *result.mutable_termination() = termination.Proto();
312 OR_ASSIGN_OR_RETURN3(*result.mutable_solve_stats(), solve_stats.Proto(),
313 _ << "invalid solve_stats");
314 for (const Solution& solution : solutions) {
315 *result.add_solutions() = solution.Proto();
316 }
317 for (const PrimalRay& primal_ray : primal_rays) {
318 *result.add_primal_rays() = primal_ray.Proto();
319 }
320 for (const DualRay& dual_ray : dual_rays) {
321 *result.add_dual_rays() = dual_ray.Proto();
322 }
323 // See yaqs/5107601535926272 on checking if a proto is empty.
324 if (gscip_solver_specific_output.ByteSizeLong() > 0) {
325 *result.mutable_gscip_output() = gscip_solver_specific_output;
326 }
327 return result;
328}

330absl::StatusOr<SolveResult> SolveResult::FromProto(
331 const ModelStorage* model, const SolveResultProto& solve_result_proto) {
332 OR_ASSIGN_OR_RETURN3(auto termination,
333 Termination::FromProto(solve_result_proto.termination()),
334 _ << "invalid termination");
335 SolveResult result(std::move(termination));
336 OR_ASSIGN_OR_RETURN3(result.solve_stats,
337 SolveStats::FromProto(solve_result_proto.solve_stats()),
338 _ << "invalid solve_stats");
339
340 for (int i = 0; i < solve_result_proto.solutions_size(); ++i) {
341 OR_ASSIGN_OR_RETURN3(
342 auto solution,
343 Solution::FromProto(model, solve_result_proto.solutions(i)),
344 _ << "invalid solution at index " << i);
345 result.solutions.push_back(std::move(solution));
346 }
347 for (int i = 0; i < solve_result_proto.primal_rays_size(); ++i) {
348 OR_ASSIGN_OR_RETURN3(
349 auto primal_ray,
350 PrimalRay::FromProto(model, solve_result_proto.primal_rays(i)),
351 _ << "invalid primal ray at index " << i);
352 result.primal_rays.push_back(std::move(primal_ray));
353 }
354 for (int i = 0; i < solve_result_proto.dual_rays_size(); ++i) {
355 OR_ASSIGN_OR_RETURN3(
356 auto dual_ray,
357 DualRay::FromProto(model, solve_result_proto.dual_rays(i)),
358 _ << "invalid dual ray at index " << i);
359 result.dual_rays.push_back(std::move(dual_ray));
360 }
361 switch (solve_result_proto.solver_specific_output_case()) {
362 case SolveResultProto::kGscipOutput:
363 result.gscip_solver_specific_output = solve_result_proto.gscip_output();
364 return result;
365 case SolveResultProto::SOLVER_SPECIFIC_OUTPUT_NOT_SET:
366 return result;
367 }
368 return util::InvalidArgumentErrorBuilder()
369 << "unexpected value of solver_specific_output_case "
370 << solve_result_proto.solver_specific_output_case();
371}
