172absl::Status Termination::ReasonIs(const TerminationReason reason) const {

175 << "expected termination reason '" << reason << "' but got " << *this;

178absl::Status Termination::ReasonIsAnyOf(
179 std::initializer_list<TerminationReason> reasons) const {
180 for (const TerminationReason reason : reasons) {
181 if (this->reason == reason) return absl::OkStatus();
182 }
183 return util::InternalErrorBuilder()
184 << "expected termination reason in {"
185 << absl::StrJoin(reasons, ", ",
186 [](std::string* out, TerminationReason reason) {
187 absl::StrAppend(out, "'");
188 absl::StreamFormatter()(out, reason);
189 absl::StrAppend(out, "'");
190 })
191 << "} but got " << *this;
192}

195 return ReasonIs(TerminationReason::kOptimal);

342absl::StatusOr<SolveResultProto> SolveResult::Proto() const {
343 SolveResultProto result;
344 *result.mutable_termination() = termination.Proto();
345 OR_ASSIGN_OR_RETURN3(*result.mutable_solve_stats(), solve_stats.Proto(),
346 _ << "invalid solve_stats");
347 for (const Solution& solution : solutions) {
348 *result.add_solutions() = solution.Proto();
349 }
350 for (const PrimalRay& primal_ray : primal_rays) {
351 *result.add_primal_rays() = primal_ray.Proto();
352 }
353 for (const DualRay& dual_ray : dual_rays) {
354 *result.add_dual_rays() = dual_ray.Proto();
355 }
356 // See yaqs/5107601535926272 on checking if a proto is empty.
357 if (gscip_solver_specific_output.ByteSizeLong() > 0) {
358 *result.mutable_gscip_output() = gscip_solver_specific_output;
359 }
360 if (pdlp_solver_specific_output.ByteSizeLong() > 0) {
361 *result.mutable_pdlp_output() = pdlp_solver_specific_output;
362 }
363 return result;
364}

366absl::StatusOr<SolveResult> SolveResult::FromProto(
367 const ModelStorage* model, const SolveResultProto& solve_result_proto) {
368 OR_ASSIGN_OR_RETURN3(auto termination,
369 Termination::FromProto(solve_result_proto.termination()),
370 _ << "invalid termination");
371 SolveResult result(std::move(termination));
372 OR_ASSIGN_OR_RETURN3(result.solve_stats,
373 SolveStats::FromProto(solve_result_proto.solve_stats()),
374 _ << "invalid solve_stats");
375
376 for (int i = 0; i < solve_result_proto.solutions_size(); ++i) {
377 OR_ASSIGN_OR_RETURN3(
378 auto solution,
379 Solution::FromProto(model, solve_result_proto.solutions(i)),
380 _ << "invalid solution at index " << i);
381 result.solutions.push_back(std::move(solution));
382 }
383 for (int i = 0; i < solve_result_proto.primal_rays_size(); ++i) {
384 OR_ASSIGN_OR_RETURN3(
385 auto primal_ray,
386 PrimalRay::FromProto(model, solve_result_proto.primal_rays(i)),
387 _ << "invalid primal ray at index " << i);
388 result.primal_rays.push_back(std::move(primal_ray));
389 }
390 for (int i = 0; i < solve_result_proto.dual_rays_size(); ++i) {
391 OR_ASSIGN_OR_RETURN3(
392 auto dual_ray,
393 DualRay::FromProto(model, solve_result_proto.dual_rays(i)),
394 _ << "invalid dual ray at index " << i);
395 result.dual_rays.push_back(std::move(dual_ray));
396 }
397 switch (solve_result_proto.solver_specific_output_case()) {
398 case SolveResultProto::kGscipOutput:
399 result.gscip_solver_specific_output = solve_result_proto.gscip_output();
400 return result;
401 case SolveResultProto::kPdlpOutput:
402 result.pdlp_solver_specific_output = solve_result_proto.pdlp_output();
403 return result;
404 case SolveResultProto::SOLVER_SPECIFIC_OUTPUT_NOT_SET:
405 return result;
406 }
407 return util::InvalidArgumentErrorBuilder()
408 << "unexpected value of solver_specific_output_case "
409 << solve_result_proto.solver_specific_output_case();
410}
