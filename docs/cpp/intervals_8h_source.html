52// This class maintains a set of intervals which correspond to three integer
53// variables (start, end and size). It automatically registers with the
54// PrecedencesPropagator the relation between the bounds of each interval and
55// provides many helper functions to add precedences relation between intervals.
56class IntervalsRepository {
57 public:
58 explicit IntervalsRepository(Model* model)
59 : model_(model),
60 assignment_(model->GetOrCreate<Trail>()->Assignment()),
61 sat_solver_(model->GetOrCreate<SatSolver>()),
62 implications_(model->GetOrCreate<BinaryImplicationGraph>()),
63 integer_trail_(model->GetOrCreate<IntegerTrail>()) {}

65 // This type is neither copyable nor movable.
66 IntervalsRepository(const IntervalsRepository&) = delete;
67 IntervalsRepository& operator=(const IntervalsRepository&) = delete;

69 // Returns the current number of intervals in the repository.
70 // The interval will always be identified by an integer in [0, num_intervals).
71 int NumIntervals() const { return starts_.size(); }
72
73 // Functions to add a new interval to the repository.
74 // If add_linear_relation is true, then we also link start, size and end.
75 //
76 // - If size == kNoIntegerVariable, then the size is fixed to fixed_size.
77 // - If is_present != kNoLiteralIndex, then this is an optional interval.
78 IntervalVariable CreateInterval(IntegerVariable start, IntegerVariable end,
79 IntegerVariable size, IntegerValue fixed_size,
80 LiteralIndex is_present);
81 IntervalVariable CreateInterval(AffineExpression start, AffineExpression end,
82 AffineExpression size,
83 LiteralIndex is_present = kNoLiteralIndex,
84 bool add_linear_relation = false);

86 // Returns whether or not a interval is optional and the associated literal.
87 bool IsOptional(IntervalVariable i) const {
88 return is_present_[i] != kNoLiteralIndex;
89 }
90 Literal PresenceLiteral(IntervalVariable i) const {
91 return Literal(is_present_[i]);
92 }
93 bool IsPresent(IntervalVariable i) const {
94 if (!IsOptional(i)) return true;
95 return assignment_.LiteralIsTrue(PresenceLiteral(i));
96 }
97 bool IsAbsent(IntervalVariable i) const {
98 if (!IsOptional(i)) return false;
99 return assignment_.LiteralIsFalse(PresenceLiteral(i));
100 }

102 // The 3 integer variables associated to a interval.
103 // Fixed size intervals will have a kNoIntegerVariable as size.
104 //
105 // Note: For an optional interval, the start/end variables are propagated
106 // assuming the interval is present. Because of that, these variables can
107 // cross each other or have an empty domain. If any of this happen, then the
108 // PresenceLiteral() of this interval will be propagated to false.
109 AffineExpression Size(IntervalVariable i) const { return sizes_[i]; }
110 AffineExpression Start(IntervalVariable i) const { return starts_[i]; }
111 AffineExpression End(IntervalVariable i) const { return ends_[i]; }
112
113 // Deprecated.
114 IntegerVariable SizeVar(IntervalVariable i) const {
115 if (sizes_[i].var != kNoIntegerVariable) {
116 CHECK_EQ(sizes_[i].coeff, 1);
117 CHECK_EQ(sizes_[i].constant, 0);
118 }
119 return sizes_[i].var;
120 }
121 IntegerVariable StartVar(IntervalVariable i) const {
122 if (starts_[i].var != kNoIntegerVariable) {
123 CHECK_EQ(starts_[i].coeff, 1);
124 CHECK_EQ(starts_[i].constant, 0);
125 }
126 return starts_[i].var;
127 }
128 IntegerVariable EndVar(IntervalVariable i) const {
129 if (ends_[i].var != kNoIntegerVariable) {
130 CHECK_EQ(ends_[i].coeff, 1);
131 CHECK_EQ(ends_[i].constant, 0);
132 }
133 return ends_[i].var;
134 }
135
136 // Return the minimum size of the given IntervalVariable.
137 IntegerValue MinSize(IntervalVariable i) const {
138 return integer_trail_->LowerBound(sizes_[i]);
139 }
140
141 // Return the maximum size of the given IntervalVariable.
142 IntegerValue MaxSize(IntervalVariable i) const {
143 return integer_trail_->UpperBound(sizes_[i]);
144 }

146 // Utility function that returns a vector will all intervals.

148 std::vector<IntervalVariable> result;
149 for (IntervalVariable i(0); i < NumIntervals(); ++i) {

313 // Helpers for the current bounds on the current task time window.
314 // [ (size-min) ... (size-min) ]
315 // ^ ^ ^ ^
316 // start-min end-min start-max end-max
317 //
318 // Note that for tasks with variable durations, we don't necessarily have
319 // duration-min between the XXX-min and XXX-max value.
320 //
321 // Remark: We use cached values for most of these function as this is faster.
322 // In practice, the cache will almost always be up to date, but not in corner
323 // cases where pushing the start of one task will change values for many
324 // others. This is fine as the new values will be picked up as we reach the
325 // propagation fixed point.

336 // In the presence of tasks with a variable size, we do not necessarily
337 // have start_min + size_min = end_min, we can instead have a situation
338 // like:
339 // | |<--- size-min --->|
340 // ^ ^ ^
341 // start-min | end-min
342 // |
343 // We define the "shifted start min" to be the right most time such that
344 // we known that we must have min-size "energy" to the right of it if the
345 // task is present. Using it in our scheduling propagators allows to propagate
346 // more in the presence of tasks with variable size (or optional task
347 // where we also do not necessarily have start_min + size_min = end_min.
348 //
349 // To explain this shifted start min, one must use the AddEnergyAfterReason().
350 IntegerValue ShiftedStartMin(int t) const {
351 return cached_shifted_start_min_[t];
352 }

622 // When defined, the interval will consume this much demand during its whole

792inline void SchedulingConstraintHelper::AddGenericReason(
793 const AffineExpression& a, IntegerValue upper_bound,
794 const AffineExpression& b, const AffineExpression& c) {
795 if (integer_trail_->UpperBound(a) <= upper_bound) {
796 if (a.var != kNoIntegerVariable) {
797 integer_reason_.push_back(a.LowerOrEqual(upper_bound));
798 }
799 return;
800 }
801 CHECK_NE(a.var, kNoIntegerVariable);
802
803 // Here we assume that the upper_bound on a comes from the bound on b + c.
804 const IntegerValue slack = upper_bound - integer_trail_->UpperBound(b) -
805 integer_trail_->UpperBound(c);
806 CHECK_GE(slack, 0);
807 if (b.var == kNoIntegerVariable && c.var == kNoIntegerVariable) return;
808 if (b.var == kNoIntegerVariable) {
809 integer_reason_.push_back(c.LowerOrEqual(upper_bound - b.constant));
810 } else if (c.var == kNoIntegerVariable) {
811 integer_reason_.push_back(b.LowerOrEqual(upper_bound - c.constant));
812 } else {
813 integer_trail_->AppendRelaxedLinearReason(
814 slack, {b.coeff, c.coeff}, {NegationOf(b.var), NegationOf(c.var)},
815 &integer_reason_);
816 }
817}

1020inline std::function<void(Model*)> IntervalWithAlternatives(
1021 IntervalVariable parent, const std::vector<IntervalVariable>& members) {
1022 return [=](Model* model) {
1023 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
1024 auto* intervals = model->GetOrCreate<IntervalsRepository>();
1025
1026 std::vector<Literal> presences;
1027 std::vector<IntegerValue> sizes;
1028
1029 // Create an "exactly one executed" constraint on the alternatives.
1030 std::vector<LiteralWithCoeff> sat_ct;
1031 for (const IntervalVariable member : members) {
1032 CHECK(intervals->IsOptional(member));
1033 const Literal is_present = intervals->PresenceLiteral(member);
1034 sat_ct.push_back({is_present, Coefficient(1)});
1035 model->Add(
1036 Equality(model->Get(StartVar(parent)), model->Get(StartVar(member))));
1037 model->Add(
1038 Equality(model->Get(EndVar(parent)), model->Get(EndVar(member))));
1039
1040 // TODO(user): IsOneOf() only work for members with fixed size.
1041 // Generalize to an "int_var_element" constraint.
1042 CHECK(integer_trail->IsFixed(intervals->Size(member)));
1043 presences.push_back(is_present);
1044 sizes.push_back(intervals->MinSize(member));
1045 }
1046 if (intervals->SizeVar(parent) != kNoIntegerVariable) {
1047 model->Add(IsOneOf(intervals->SizeVar(parent), presences, sizes));
1048 }
1049 model->Add(BooleanLinearConstraint(1, 1, &sat_ct));
1050
1051 // Propagate from the candidate bounds to the parent interval ones.
1052 {
1053 std::vector<IntegerVariable> starts;
1054 starts.reserve(members.size());
1055 for (const IntervalVariable member : members) {
1056 starts.push_back(intervals->StartVar(member));
1057 }
1058 model->Add(
1059 PartialIsOneOfVar(intervals->StartVar(parent), starts, presences));
1060 }
1061 {
1062 std::vector<IntegerVariable> ends;
1063 ends.reserve(members.size());
1064 for (const IntervalVariable member : members) {
1065 ends.push_back(intervals->EndVar(member));
1066 }
1067 model->Add(PartialIsOneOfVar(intervals->EndVar(parent), ends, presences));
1068 }
1069 };
