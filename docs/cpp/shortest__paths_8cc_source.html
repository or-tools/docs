37// Base path container implementation class. Defines virtual functions used to
38// fill the container (in particular from the shortest path computation
39// function).
40class PathContainerImpl {
41 public:
42 PathContainerImpl() {}
43 virtual ~PathContainerImpl() {}
44
45 // Initializes the container on source and destination node vectors (num_nodes
46 // is the total number of nodes in the graph containing sources and nodes).
47 // Called before adding any paths to the container.
48 virtual void Initialize(const std::vector<NodeIndex>& sources,
49 const std::vector<NodeIndex>& destinations,
50 NodeIndex num_nodes) = 0;
51
52 // Called when no more path will be added to the container.
53 virtual void Finalize() {}
54
55 // Returns the distance between node 'from' and node 'to' following the path
56 // out of 'from' and into 'to'. Note that if 'from' == 'to', the distance is
57 // not necessarily 0 if the path out of 'to' and back into 'to' has a distance
58 // greater than 0. If you do require the distance to be 0 in this case, add to
59 // the graph an arc from 'to' to itself with a length of 0.
60 // If nodes are not connected, returns kDisconnectedPathDistance.
61 virtual PathDistance GetDistance(NodeIndex from, NodeIndex to) const = 0;
62
63 // Returns the penultimate node on the path out of node 'from' into node 'to'
64 // (the direct predecessor of node 'to' on the path).
65 // If 'from' == 'to', the penultimate node is 'to' only if the shortest path
66 // from 'to' to itself is composed of the arc ('to, 'to'), which might not be
67 // the case if either this arc doesn't exist or if the length of this arc is
68 // greater than the distance of an alternate path.
69 // If nodes are not connected, returns StarGraph::kNilNode.
70 virtual NodeIndex GetPenultimateNodeInPath(NodeIndex from,
71 NodeIndex to) const = 0;
72
73 // Returns path nodes from node "from" to node "to" in a ordered vector.
74 virtual void GetPath(NodeIndex from, NodeIndex to,
75 std::vector<NodeIndex>* path) const = 0;
76
77 // Adds a path tree rooted at node 'from', and to a set of implicit
78 // destinations:
79 // - predecessor_in_path_tree[node] is the predecessor of node 'node' in the
80 // path from 'from' to 'node', or StarGraph::kNilNode if there is no
81 // predecessor (i.e. if 'node' is not in the path tree);
82 // - distance_to_destination[i] is the distance from 'from' to the i-th
83 // destination (see Initialize()).
84 virtual void StoreSingleSourcePaths(
85 NodeIndex from, const std::vector<NodeIndex>& predecessor_in_path_tree,
86 const std::vector<PathDistance>& distance_to_destination) = 0;
87};

91// Class designed to store the tree of paths from a root node to a set of nodes
92// in a very compact way (over performance).
93// Memory consumption is in O(n) (n being the size of the tree) where node
94// indices are "very" non-contiguous (extremely sparse node indices). It keeps
95// node-sorted arrays of node and parent pairs, which can be accessed in
96// O(log(n)) with a binary search.
97// The creation of the tree is done in O(n*log(n)) time.
98// Note that this class uses temporary memory for each call to Initialize which
99// is only an issue for massive parallel calls; in practice for shortest
100// paths computation, the number of threads calling Initialize is very small
101// compared to the total number of trees created.
102class PathTree {
103 public:
104 PathTree() : nodes_(), parents_() {}
105
106 void Initialize(const std::vector<NodeIndex>& paths,
107 const std::vector<NodeIndex>& destinations);
108
109 // Returns the parent (predecessor) of 'node' in the tree in
110 // O(log(path_tree_size)), where path_tree_size is the size of nodes_.
111 NodeIndex GetParent(NodeIndex node) const;
112
113 // Returns the path from node 'from' to node 'to' in the tree in
114 // O(log(path_tree_size) + path_size), where path_tree_size is the size of
115 // nodes_ and path_size is the size of the resulting path.
116 void GetPath(NodeIndex from, NodeIndex to,
117 std::vector<NodeIndex>* path) const;
118
119 private:
120 std::vector<NodeIndex> nodes_;
121 std::vector<int> parents_;
122};
123
124// Initializes the tree from a non-sparse representation of the path tree
125// represented by 'paths'. The tree is reduced to the subtree in which nodes in
126// 'destinations' are the leafs.
127void PathTree::Initialize(const std::vector<NodeIndex>& paths,
128 const std::vector<NodeIndex>& destinations) {
129 const NodeIndex kNilNode = StarGraph::kNilNode;
130 std::vector<bool> node_explored(paths.size(), false);
131 const int destination_size = destinations.size();
132 typedef std::pair<NodeIndex, NodeIndex> NodeParent;
133 std::vector<NodeParent> tree;
134 for (int i = 0; i < destination_size; ++i) {
135 NodeIndex destination = destinations[i];
136 while (!node_explored[destination]) {
137 node_explored[destination] = true;
138 tree.push_back(std::make_pair(destination, paths[destination]));
139 if (paths[destination] != kNilNode) {
140 destination = paths[destination];
141 }
142 }
143 }
144 std::sort(tree.begin(), tree.end());
145 const int num_nodes = tree.size();
146 {
147 // Using a dense_hash_map gives a huge speedup over hash_map here, mainly
148 // due to better memory management.
149 absl::flat_hash_map<NodeIndex, int> node_indices;
150
151 for (int i = 0; i < num_nodes; ++i) {
152 node_indices[tree[i].first] = i;
153 }
154 parents_.resize(num_nodes, -1);
155 for (int i = 0; i < num_nodes; ++i) {
156 if (tree[i].second == kNilNode) {
157 // dense_hash_map does not allow empty key to be used as a key.
158 parents_[i] = kNilNode;
159 } else {
160 parents_[i] =
161 gtl::FindWithDefault(node_indices, tree[i].second, kNilNode);
162 }
163 }
164 }
165 nodes_.resize(num_nodes, kNilNode);
166 for (int i = 0; i < num_nodes; ++i) {
167 nodes_[i] = tree[i].first;
168 }
169}
170
171NodeIndex PathTree::GetParent(NodeIndex node) const {
172 std::vector<NodeIndex>::const_iterator node_position =
173 std::lower_bound(nodes_.begin(), nodes_.end(), node);
174 if (node_position != nodes_.end() && *node_position == node) {
175 const int parent = parents_[node_position - nodes_.begin()];
176 if (parent != StarGraph::kNilNode) {
177 return nodes_[parent];
178 }
179 }
180 return StarGraph::kNilNode;
181}
182
183void PathTree::GetPath(NodeIndex from, NodeIndex to,
184 std::vector<NodeIndex>* path) const {
185 DCHECK(path != nullptr);
186 path->clear();
187 std::vector<NodeIndex>::const_iterator to_position =
188 std::lower_bound(nodes_.begin(), nodes_.end(), to);
189 if (to_position != nodes_.end() && *to_position == to) {
190 int current_index = to_position - nodes_.begin();
191 NodeIndex current_node = to;
192 while (current_node != from) {
193 path->push_back(current_node);
194 current_index = parents_[current_index];
195 // from and to are not connected
196 if (current_index == StarGraph::kNilNode) {
197 path->clear();
198 return;
199 }
200 current_node = nodes_[current_index];
201 }
202 path->push_back(current_node);
203 std::reverse(path->begin(), path->end());
204 }
205}

207// Path container which only stores distances between path nodes.
208class DistanceContainer : public PathContainerImpl {
209 public:
210 DistanceContainer() : reverse_sources_(), distances_() {}
211
212 // This type is neither copyable nor movable.
213 DistanceContainer(const DistanceContainer&) = delete;
214 DistanceContainer& operator=(const DistanceContainer&) = delete;
215 ~DistanceContainer() override {}
216 void Initialize(const std::vector<NodeIndex>& sources,
217 const std::vector<NodeIndex>& destinations,
218 NodeIndex num_nodes) override {
219 ComputeReverse(sources, num_nodes, &reverse_sources_);
220 ComputeReverse(destinations, num_nodes, &reverse_destinations_);
221 distances_.clear();
222 distances_.resize(sources.size());
223 }
224 PathDistance GetDistance(NodeIndex from, NodeIndex to) const override {
225 return distances_[reverse_sources_[from]][reverse_destinations_[to]];
226 }
227 NodeIndex GetPenultimateNodeInPath(NodeIndex from,
228 NodeIndex to) const override {
229 LOG(FATAL) << "Path not stored.";
230 return StarGraph::kNilNode;
231 }
232 void GetPath(NodeIndex from, NodeIndex to,
233 std::vector<NodeIndex>* path) const override {
234 LOG(FATAL) << "Path not stored.";
235 }
236 void StoreSingleSourcePaths(
237 NodeIndex from, const std::vector<NodeIndex>& predecessor_in_path_tree,
238 const std::vector<PathDistance>& distance_to_destination) override {
239 distances_[reverse_sources_[from]] = distance_to_destination;
240 }
241
242 protected:
243 std::vector<int> reverse_sources_;
244 std::vector<int> reverse_destinations_;
245
246 private:
247 static void ComputeReverse(const std::vector<NodeIndex>& nodes,
248 NodeIndex num_nodes,
249 std::vector<int>* reverse_nodes) {
250 CHECK(reverse_nodes != nullptr);
251 const int kUnassignedIndex = -1;
252 reverse_nodes->clear();
253 reverse_nodes->resize(num_nodes, kUnassignedIndex);
254 for (int i = 0; i < nodes.size(); ++i) {
255 reverse_nodes->at(nodes[i]) = i;
256 }
257 }
258
259 std::vector<std::vector<PathDistance> > distances_;
260};

303// Priority queue node entry in the boundary of the Dijkstra algorithm.
304class NodeEntry {
305 public:
306 NodeEntry()
307 : heap_index_(-1),
308 distance_(0),
309 node_(StarGraph::kNilNode),
310 settled_(false),
311 is_destination_(false) {}
312 bool operator<(const NodeEntry& other) const {
313 return distance_ > other.distance_;
314 }
315 void SetHeapIndex(int h) { heap_index_ = h; }
316 int GetHeapIndex() const { return heap_index_; }
317 void set_distance(PathDistance distance) { distance_ = distance; }
318 PathDistance distance() const { return distance_; }
319 void set_node(NodeIndex node) { node_ = node; }
320 NodeIndex node() const { return node_; }
321 void set_settled(bool settled) { settled_ = settled; }
322 bool settled() const { return settled_; }
323 void set_is_destination(bool is_destination) {
324 is_destination_ = is_destination;
325 }
326 bool is_destination() const { return is_destination_; }
327
328 private:
329 int heap_index_;
330 PathDistance distance_;
331 NodeIndex node_;
332 bool settled_;
333 bool is_destination_;
334};

336// Updates an entry with the given distance if it's shorter, and then inserts it
337// in the priority queue (or updates it if it's there already), if needed.
338// Returns true if the entry was modified, false otherwise.
339bool InsertOrUpdateEntry(PathDistance distance, NodeEntry* entry,
340 AdjustablePriorityQueue<NodeEntry>* priority_queue) {
341 DCHECK(priority_queue != nullptr);
342 DCHECK(entry != nullptr);
343 if (!priority_queue->Contains(entry)) {
344 entry->set_distance(distance);
345 priority_queue->Add(entry);
346 return true;
347 } else if (distance < entry->distance()) {
348 entry->set_distance(distance);
349 priority_queue->NoteChangedPriority(entry);
350 return true;
351 }
352 return false;
353}

355// If one wants to use int64_t for either priority or NodeIndex, one should
356// consider using packed ints (putting the two bools with heap_index, for
357// example) in order to stay at 16 bytes instead of 24.
358static_assert(sizeof(NodeEntry) == 16, "node_entry_class_is_not_well_packed");

360// Computes shortest paths from node 'source' to nodes in 'destinations'
361// using a binary heap-based Dijkstra algorithm.
362// TODO(user): Investigate alternate implementation which wouldn't use
363// AdjustablePriorityQueue.
364template <class GraphType>
365void ComputeOneToManyInternal(const GraphType* const graph,
366 const ZVector<PathDistance>* const arc_lengths,
367 NodeIndex source,
368 const std::vector<NodeIndex>* const destinations,
369 PathContainerImpl* const paths) {
370 CHECK(graph != nullptr);
371 CHECK(arc_lengths != nullptr);
372 CHECK(destinations != nullptr);
373 CHECK(paths != nullptr);
374 const int num_nodes = graph->num_nodes();
375 std::vector<NodeIndex> predecessor(num_nodes, GraphType::kNilNode);
376 AdjustablePriorityQueue<NodeEntry> priority_queue;
377 std::vector<NodeEntry> entries(num_nodes);
378 for (typename GraphType::NodeIterator iterator(*graph); iterator.Ok();
379 iterator.Next()) {
380 entries[iterator.Index()].set_node(iterator.Index());
381 }
382 // Marking destination node. This is an optimization stopping the search
383 // when all destinations have been reached.
384 for (int i = 0; i < destinations->size(); ++i) {
385 entries[(*destinations)[i]].set_is_destination(true);
386 }
387 // In this implementation the distance of a node to itself isn't necessarily
388 // 0.
389 // So we push successors of source in the queue instead of the source
390 // directly which will avoid marking the source.
391 for (typename GraphType::OutgoingArcIterator iterator(*graph, source);
392 iterator.Ok(); iterator.Next()) {
393 const ArcIndex arc = iterator.Index();
394 const NodeIndex next = graph->Head(arc);
395 if (InsertOrUpdateEntry(arc_lengths->Value(arc), &entries[next],
396 &priority_queue)) {
397 predecessor[next] = source;
398 }
399 }
400 int destinations_remaining = destinations->size();
401 while (!priority_queue.IsEmpty()) {
402 NodeEntry* current = priority_queue.Top();
403 const NodeIndex current_node = current->node();
404 priority_queue.Pop();
405 current->set_settled(true);
406 if (current->is_destination()) {
407 destinations_remaining--;
408 if (destinations_remaining == 0) {
409 break;
410 }
411 }
412 const PathDistance current_distance = current->distance();
413 for (typename GraphType::OutgoingArcIterator iterator(*graph, current_node);
414 iterator.Ok(); iterator.Next()) {
415 const ArcIndex arc = iterator.Index();
416 const NodeIndex next = graph->Head(arc);
417 NodeEntry* const entry = &entries[next];
418 if (!entry->settled()) {
419 DCHECK_GE(current_distance, 0);
420 const PathDistance arc_length = arc_lengths->Value(arc);
421 DCHECK_LE(current_distance, kDisconnectedPathDistance - arc_length);
422 if (InsertOrUpdateEntry(current_distance + arc_length, entry,
423 &priority_queue)) {
424 predecessor[next] = current_node;
425 }
426 }
427 }
428 }
429 const int destinations_size = destinations->size();
430 std::vector<PathDistance> distances(destinations_size,
431 kDisconnectedPathDistance);
432 for (int i = 0; i < destinations_size; ++i) {
433 NodeIndex node = destinations->at(i);
434 if (entries[node].settled()) {
435 distances[i] = entries[node].distance();
436 }
437 }
438 paths->StoreSingleSourcePaths(source, predecessor, distances);
439}

441template <class GraphType>
442void ComputeOneToManyInternalOnGraph(
443 const GraphType* const graph,
444 const std::vector<PathDistance>* const arc_lengths,
445 typename GraphType::NodeIndex source,
446 const std::vector<typename GraphType::NodeIndex>* const destinations,
447 PathContainerImpl* const paths) {
448 CHECK(graph != nullptr);
449 CHECK(arc_lengths != nullptr);
450 CHECK(destinations != nullptr);
451 CHECK(paths != nullptr);
452 const int num_nodes = graph->num_nodes();
453 std::vector<typename GraphType::NodeIndex> predecessor(num_nodes, -1);
454 AdjustablePriorityQueue<NodeEntry> priority_queue;
455 std::vector<NodeEntry> entries(num_nodes);
456 for (const typename GraphType::NodeIndex node : graph->AllNodes()) {
457 entries[node].set_node(node);
458 }
459 // Marking destination node. This is an optimization stopping the search
460 // when all destinations have been reached.
461 for (int i = 0; i < destinations->size(); ++i) {
462 entries[(*destinations)[i]].set_is_destination(true);
463 }
464 // In this implementation the distance of a node to itself isn't necessarily
465 // 0.
466 // So we push successors of source in the queue instead of the source
467 // directly which will avoid marking the source.
468 for (const typename GraphType::ArcIndex arc : graph->OutgoingArcs(source)) {
469 const typename GraphType::NodeIndex next = graph->Head(arc);
470 if (InsertOrUpdateEntry(arc_lengths->at(arc), &entries[next],
471 &priority_queue)) {
472 predecessor[next] = source;
473 }
474 }
475 int destinations_remaining = destinations->size();
476 while (!priority_queue.IsEmpty()) {
477 NodeEntry* current = priority_queue.Top();
478 const typename GraphType::NodeIndex current_node = current->node();
479 priority_queue.Pop();
480 current->set_settled(true);
481 if (current->is_destination()) {
482 destinations_remaining--;
483 if (destinations_remaining == 0) {
484 break;
485 }
486 }
487 const PathDistance current_distance = current->distance();
488 for (const typename GraphType::ArcIndex arc :
489 graph->OutgoingArcs(current_node)) {
490 const typename GraphType::NodeIndex next = graph->Head(arc);
491 NodeEntry* const entry = &entries[next];
492 if (!entry->settled()) {
493 DCHECK_GE(current_distance, 0);
494 const PathDistance arc_length = arc_lengths->at(arc);
495 DCHECK_LE(current_distance, kDisconnectedPathDistance - arc_length);
496 if (InsertOrUpdateEntry(current_distance + arc_length, entry,
497 &priority_queue)) {
498 predecessor[next] = current_node;
499 }
500 }
501 }
502 }
503 const int destinations_size = destinations->size();
504 std::vector<PathDistance> distances(destinations_size,
505 kDisconnectedPathDistance);
506 for (int i = 0; i < destinations_size; ++i) {
507 NodeIndex node = destinations->at(i);
508 if (entries[node].settled()) {
509 distances[i] = entries[node].distance();
510 }
511 }
512 paths->StoreSingleSourcePaths(source, predecessor, distances);
513}

554template <class GraphType>
555void ComputeManyToManyShortestPathsWithMultipleThreadsInternal(
556 const GraphType& graph, const ZVector<PathDistance>& arc_lengths,
557 const std::vector<NodeIndex>& sources,
558 const std::vector<NodeIndex>& destinations, int num_threads,
559 PathContainer* const paths) {
560 if (graph.num_nodes() > 0) {
561 CHECK_EQ(graph.num_arcs(),
562 1 + arc_lengths.max_index() - arc_lengths.min_index())
563 << "Number of arcs in graph must match arc length vector size";
564 // Removing duplicate sources to allow mutex-free implementation (and it's
565 // more efficient); same with destinations for efficiency reasons.
566 std::vector<NodeIndex> unique_sources = sources;
567 gtl::STLSortAndRemoveDuplicates(&unique_sources);
568 std::vector<NodeIndex> unique_destinations = destinations;
569 gtl::STLSortAndRemoveDuplicates(&unique_destinations);
570 WallTimer timer;
571 timer.Start();
572 PathContainerImpl* container = paths->GetImplementation();
573 container->Initialize(unique_sources, unique_destinations,
574 graph.num_nodes());
575 {
576 std::unique_ptr<ThreadPool> pool(
577 new ThreadPool("OR_Dijkstra", num_threads));
578 pool->StartWorkers();
579 for (int i = 0; i < unique_sources.size(); ++i) {
580 pool->Schedule(absl::bind_front(&ComputeOneToManyInternal<GraphType>,
581 &graph, &arc_lengths, unique_sources[i],
582 &unique_destinations, container));
583 }
584 }
585 container->Finalize();
586 VLOG(2) << "Elapsed time to compute shortest paths: " << timer.Get() << "s";
587 }
588}

612template <class GraphType>
613void ComputeManyToManyShortestPathsWithMultipleThreadsInternal(
614 const GraphType& graph, const std::vector<PathDistance>& arc_lengths,
615 const std::vector<typename GraphType::NodeIndex>& sources,
616 const std::vector<typename GraphType::NodeIndex>& destinations,
617 int num_threads, PathContainer* const paths) {
618 if (graph.num_nodes() > 0) {
619 CHECK_EQ(graph.num_arcs(), arc_lengths.size())

621 // Removing duplicate sources to allow mutex-free implementation (and it's

623 std::vector<typename GraphType::NodeIndex> unique_sources = sources;
624 gtl::STLSortAndRemoveDuplicates(&unique_sources);
625 std::vector<typename GraphType::NodeIndex> unique_destinations =
626 destinations;
627 gtl::STLSortAndRemoveDuplicates(&unique_destinations);
628 WallTimer timer;
629 timer.Start();
630 PathContainerImpl* container = paths->GetImplementation();
631 container->Initialize(unique_sources, unique_destinations,
632 graph.num_nodes());
633 {
634 std::unique_ptr<ThreadPool> pool(
635 new ThreadPool("OR_Dijkstra", num_threads));
636 pool->StartWorkers();
637 for (int i = 0; i < unique_sources.size(); ++i) {
638 pool->Schedule(absl::bind_front(
639 &ComputeOneToManyInternalOnGraph<GraphType>, &graph, &arc_lengths,
640 unique_sources[i], &unique_destinations, container));
641 }
642 }
643 container->Finalize();
644 VLOG(2) << "Elapsed time to compute shortest paths: " << timer.Get() << "s";
645 }

648template <>
649void ComputeManyToManyShortestPathsWithMultipleThreads(
650 const ListGraph<>& graph, const std::vector<PathDistance>& arc_lengths,
651 const std::vector<ListGraph<>::NodeIndex>& sources,
