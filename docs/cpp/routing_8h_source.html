213class PathsMetadata {
214 public:
215 explicit PathsMetadata(const RoutingIndexManager& manager) {
216 const int num_indices = manager.num_indices();
217 const int num_paths = manager.num_vehicles();
218 path_of_node_.resize(num_indices, -1);
219 is_start_.resize(num_indices, false);
220 is_end_.resize(num_indices, false);
221 start_of_path_.resize(num_paths);
222 end_of_path_.resize(num_paths);
223 for (int v = 0; v < num_paths; ++v) {
224 const int64_t start = manager.GetStartIndex(v);
225 start_of_path_[v] = start;
226 path_of_node_[start] = v;
227 is_start_[start] = true;
228 const int64_t end = manager.GetEndIndex(v);
229 end_of_path_[v] = end;
230 path_of_node_[end] = v;
231 is_end_[end] = true;
232 }
233 }
234
235 bool IsStart(int64_t node) const { return is_start_[node]; }
236 bool IsEnd(int64_t node) const { return is_end_[node]; }
237 int GetPath(int64_t start_or_end_node) const {
238 return path_of_node_[start_or_end_node];
239 }
240 int NumPaths() const { return start_of_path_.size(); }
241 const std::vector<int64_t>& Paths() const { return path_of_node_; }
242 const std::vector<int64_t>& Starts() const { return start_of_path_; }
243 const std::vector<int64_t>& Ends() const { return end_of_path_; }
244
245 private:
246 std::vector<bool> is_start_;
247 std::vector<bool> is_end_;
248 std::vector<int64_t> start_of_path_;
249 std::vector<int64_t> end_of_path_;
250 std::vector<int64_t> path_of_node_;
251};
