24#define TEST_IN_RANGE(name, min, max) \
25 if (params.name() < min || params.name() > max) { \
26 return absl::StrCat("parameter '", #name, "' should be in [", min, ",", \
27 max, "]. Current value is ", params.name()); \
28 }
29
30#define TEST_NON_NEGATIVE(name) \
31 if (params.name() < 0) { \
32 return absl::StrCat("Parameters ", #name, " must be non-negative"); \
33 }
34
35#define TEST_NOT_NAN(name) \
36 if (std::isnan(params.name())) { \
37 return absl::StrCat("parameter '", #name, "' is NaN"); \
38 }
