53IntegerVariable CreateVariableWithTightDomain(
54 absl::Span<const AffineExpression> exprs, Model* model) {
55 IntegerValue min = kMaxIntegerValue;
56 IntegerValue max = kMinIntegerValue;
57 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
58 for (const AffineExpression& e : exprs) {
59 min = std::min(min, integer_trail->LevelZeroLowerBound(e));
60 max = std::max(max, integer_trail->LevelZeroUpperBound(e));
61 }
62 return integer_trail->AddIntegerVariable(min, max);
63}

65// TODO(user): Use the faster variable only version if all expressions reduce
66// to a single variable?
67IntegerVariable CreateVariableEqualToMinOf(
68 absl::Span<const AffineExpression> exprs, Model* model) {
69 std::vector<LinearExpression> converted;
70 for (const AffineExpression& affine : exprs) {
71 LinearExpression e;
72 e.offset = affine.constant;
73 if (affine.var != kNoIntegerVariable) {
74 e.vars.push_back(affine.var);
75 e.coeffs.push_back(affine.coeff);
76 }
77 converted.push_back(e);
78 }
79
80 LinearExpression target;
81 const IntegerVariable var = CreateVariableWithTightDomain(exprs, model);
82 target.vars.push_back(var);
83 target.coeffs.push_back(IntegerValue(1));
84 model->Add(IsEqualToMinOf(target, converted));
85 return var;
86}
87
88IntegerVariable CreateVariableEqualToMaxOf(
89 absl::Span<const AffineExpression> exprs, Model* model) {
90 std::vector<LinearExpression> converted;
91 for (const AffineExpression& affine : exprs) {
92 LinearExpression e;
93 e.offset = affine.constant;
94 if (affine.var != kNoIntegerVariable) {
95 e.vars.push_back(affine.var);
96 e.coeffs.push_back(affine.coeff);
97 }
98 converted.push_back(NegationOf(e));
99 }
100
101 LinearExpression target;
102 const IntegerVariable var = CreateVariableWithTightDomain(exprs, model);
103 target.vars.push_back(NegationOf(var));
104 target.coeffs.push_back(IntegerValue(1));
105 model->Add(IsEqualToMinOf(target, converted));
106 return var;
107}

109// Add a cumulative relaxation. That is, on one dimension, it does not enforce
110// the rectangle aspect, allowing vertical slices to move freely.
111void AddDiffnCumulativeRelationOnX(SchedulingConstraintHelper* x,
112 SchedulingConstraintHelper* y,
113 Model* model) {
114 // TODO(user): Use conditional affine min/max !!
115 const IntegerVariable min_start_var =
116 CreateVariableEqualToMinOf(y->Starts(), model);
117 const IntegerVariable max_end_var =
118 CreateVariableEqualToMaxOf(y->Ends(), model);
119
120 // (max_end - min_start) >= capacity.
121 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
122 const AffineExpression capacity(model->Add(NewIntegerVariable(
123 0, CapSub(integer_trail->UpperBound(max_end_var).value(),
124 integer_trail->LowerBound(min_start_var).value()))));
125 const std::vector<int64_t> coeffs = {-capacity.coeff.value(), -1, 1};
126 model->Add(
127 WeightedSumGreaterOrEqual({capacity.var, min_start_var, max_end_var},
128 coeffs, capacity.constant.value()));
129
130 SchedulingDemandHelper* demands =
131 model->GetOrCreate<IntervalsRepository>()->GetOrCreateDemandHelper(
132 x, y->Sizes());
133
134 // Propagator responsible for applying Timetabling filtering rule. It
135 // increases the minimum of the start variables, decrease the maximum of the
136 // end variables, and increase the minimum of the capacity variable.
137 const SatParameters& params = *model->GetOrCreate<SatParameters>();
138 if (params.use_timetabling_in_no_overlap_2d()) {
139 TimeTablingPerTask* time_tabling =
140 new TimeTablingPerTask(capacity, x, demands, model);
141 time_tabling->RegisterWith(model->GetOrCreate<GenericLiteralWatcher>());
142 model->TakeOwnership(time_tabling);
143 }
144
145 // Propagator responsible for applying the Overload Checking filtering rule.
146 // It increases the minimum of the capacity variable.
147 if (params.use_energetic_reasoning_in_no_overlap_2d()) {
148 AddCumulativeOverloadChecker(capacity, x, demands, model);
149 }
150}

167 ClearAndAddMandatoryOverlapReason(box1, box2, y);
168 x->ImportOtherReasons(*y);
169 return x->ReportConflict();

174void AddNonOverlappingRectangles(const std::vector<IntervalVariable>& x,
175 const std::vector<IntervalVariable>& y,
176 Model* model) {
177 IntervalsRepository* repository = model->GetOrCreate<IntervalsRepository>();
178 SchedulingConstraintHelper* x_helper = repository->GetOrCreateHelper(x);
179 SchedulingConstraintHelper* y_helper = repository->GetOrCreateHelper(y);
180
181 NonOverlappingRectanglesDisjunctivePropagator* constraint =
182 new NonOverlappingRectanglesDisjunctivePropagator(x_helper, y_helper,
183 model);
184 constraint->Register(/*fast_priority=*/3, /*slow_priority=*/4);
185 model->TakeOwnership(constraint);
186
187 RectanglePairwisePropagator* pairwise_propagator =
188 new RectanglePairwisePropagator(x_helper, y_helper, model);
189 GenericLiteralWatcher* const watcher =
190 model->GetOrCreate<GenericLiteralWatcher>();
191 watcher->SetPropagatorPriority(pairwise_propagator->RegisterWith(watcher), 4);
192 model->TakeOwnership(pairwise_propagator);
193
194 const SatParameters& params = *model->GetOrCreate<SatParameters>();
195 const bool add_cumulative_relaxation =
196 params.use_timetabling_in_no_overlap_2d() ||
197 params.use_energetic_reasoning_in_no_overlap_2d();
198
199 if (add_cumulative_relaxation) {
200 // We must first check if the cumulative relaxation is possible.
201 bool some_boxes_are_only_optional_on_x = false;
202 bool some_boxes_are_only_optional_on_y = false;
203 for (int i = 0; i < x.size(); ++i) {
204 if (x_helper->IsOptional(i) && y_helper->IsOptional(i) &&
205 x_helper->PresenceLiteral(i) != y_helper->PresenceLiteral(i)) {
206 // Abort as the task would be conditioned by two literals.
207 return;
208 }
209 if (x_helper->IsOptional(i) && !y_helper->IsOptional(i)) {
210 // We cannot use x_size as the demand of the cumulative based on
211 // the y_intervals.
212 some_boxes_are_only_optional_on_x = true;
213 }
214 if (y_helper->IsOptional(i) && !x_helper->IsOptional(i)) {
215 // We cannot use y_size as the demand of the cumulative based on
216 // the y_intervals.
217 some_boxes_are_only_optional_on_y = true;
218 }
219 }
220 if (!some_boxes_are_only_optional_on_y) {
221 AddDiffnCumulativeRelationOnX(x_helper, y_helper, model);
222 }
223 if (!some_boxes_are_only_optional_on_x) {
224 AddDiffnCumulativeRelationOnX(y_helper, x_helper, model);
225 }
226 }
227
228 if (params.use_area_energetic_reasoning_in_no_overlap_2d()) {
229 NonOverlappingRectanglesEnergyPropagator* energy_constraint =
230 new NonOverlappingRectanglesEnergyPropagator(x_helper, y_helper, model);
231 GenericLiteralWatcher* const watcher =
232 model->GetOrCreate<GenericLiteralWatcher>();
233 watcher->SetPropagatorPriority(energy_constraint->RegisterWith(watcher), 5);
234 model->TakeOwnership(energy_constraint);
235 }
236}

261bool NonOverlappingRectanglesEnergyPropagator::Propagate() {
262 // TODO(user): double-check/revisit the algo for box of variable sizes.
263 const int num_boxes = x_.NumTasks();
264 if (!x_.SynchronizeAndSetTimeDirection(true)) return false;
265 if (!y_.SynchronizeAndSetTimeDirection(true)) return false;
266
267 Rectangle bounding_box = {.x_min = std::numeric_limits<IntegerValue>::max(),
268 .x_max = std::numeric_limits<IntegerValue>::min(),
269 .y_min = std::numeric_limits<IntegerValue>::max(),
270 .y_max = std::numeric_limits<IntegerValue>::min()};
271 std::vector<RectangleInRange> active_box_ranges;
272 active_box_ranges.reserve(num_boxes);
273 for (int box = 0; box < num_boxes; ++box) {
274 if (x_.SizeMin(box) == 0 || y_.SizeMin(box) == 0) continue;
275 if (!x_.IsPresent(box) || !y_.IsPresent(box)) continue;
276
277 bounding_box.x_min = std::min(bounding_box.x_min, x_.StartMin(box));
278 bounding_box.x_max = std::max(bounding_box.x_max, x_.EndMax(box));
279 bounding_box.y_min = std::min(bounding_box.y_min, y_.StartMin(box));
280 bounding_box.y_max = std::max(bounding_box.y_max, y_.EndMax(box));
281
282 active_box_ranges.push_back(RectangleInRange{
283 .box_index = box,
284 .bounding_area = {.x_min = x_.StartMin(box),
285 .x_max = x_.StartMax(box) + x_.SizeMin(box),
286 .y_min = y_.StartMin(box),
287 .y_max = y_.StartMax(box) + y_.SizeMin(box)},
288 .x_size = x_.SizeMin(box),
289 .y_size = y_.SizeMin(box)});
290 }
291
292 if (active_box_ranges.size() < 2) {
293 return true;
294 }
295
296 // Our algo is quadratic, so we don't want to run it on really large problems.
297 if (active_box_ranges.size() > 1000) {
298 return true;
299 }
300
301 if (std::max(bounding_box.SizeX(), bounding_box.SizeY()) *
302 active_box_ranges.size() >
303 std::numeric_limits<int32_t>::max()) {
304 // Avoid integer overflows if the area of the boxes get comparable with
305 // INT64_MAX
306 return true;
307 }
308
309 num_calls_++;
310
311 std::optional<Conflict> best_conflict =
312 FindConflict(std::move(active_box_ranges));
313 if (!best_conflict.has_value()) {
314 return true;
315 }
316 num_conflicts_++;
317
318 // We found a conflict, so we can afford to run the propagator again to
319 // search for a best explanation. This is specially the case since we only
320 // want to re-run it over the items that participate in the conflict, so it is
321 // a much smaller problem.
322 IntegerValue best_explanation_size =
323 best_conflict->opp_result.GetItemsParticipatingOnConflict().size();
324 bool refined = false;
325 while (true) {
326 std::vector<RectangleInRange> items_participating_in_conflict;
327 items_participating_in_conflict.reserve(
328 best_conflict->items_for_opp.size());
329 for (const auto& item :
330 best_conflict->opp_result.GetItemsParticipatingOnConflict()) {
331 items_participating_in_conflict.push_back(
332 best_conflict->items_for_opp[item.index]);
333 }
334 std::optional<Conflict> conflict =
335 FindConflict(items_participating_in_conflict);
336 if (!conflict.has_value()) break;
337 // We prefer an explanation with the least number of boxes.
338 if (conflict->opp_result.GetItemsParticipatingOnConflict().size() >=
339 best_explanation_size) {
340 // The new explanation isn't better than the old one. Stop trying.
341 break;
342 }
343 best_explanation_size =
344 conflict->opp_result.GetItemsParticipatingOnConflict().size();
345 best_conflict = std::move(conflict);
346 refined = true;
347 }
348
349 num_refined_conflicts_ += refined;
350 const std::vector<RectangleInRange> generalized_explanation =
351 GeneralizeExplanation(*best_conflict);
352 if (best_explanation_size == 2) {
353 num_conflicts_two_boxes_++;
354 }
355 BuildAndReportEnergyTooLarge(generalized_explanation);
356 return false;
357}

359std::optional<NonOverlappingRectanglesEnergyPropagator::Conflict>
360NonOverlappingRectanglesEnergyPropagator::FindConflict(
361 std::vector<RectangleInRange> active_box_ranges) {
362 const auto rectangles_with_too_much_energy =
363 FindRectanglesWithEnergyConflictMC(active_box_ranges, *random_, 1.0, 0.8);
364
365 if (rectangles_with_too_much_energy.conflicts.empty() &&
366 rectangles_with_too_much_energy.candidates.empty()) {
367 return std::nullopt;
368 }
369
370 Conflict best_conflict;
371
372 // Sample 10 rectangles (at least five among the ones for which we already
373 // detected an energy overflow), extract an orthogonal packing subproblem for
374 // each and look for conflict. Sampling avoids making this heuristic too
375 // costly.
376 constexpr int kSampleSize = 10;
377 absl::InlinedVector<Rectangle, kSampleSize> sampled_rectangles;
378 std::sample(rectangles_with_too_much_energy.conflicts.begin(),
379 rectangles_with_too_much_energy.conflicts.end(),
380 std::back_inserter(sampled_rectangles), 5, *random_);
381 std::sample(rectangles_with_too_much_energy.candidates.begin(),
382 rectangles_with_too_much_energy.candidates.end(),
383 std::back_inserter(sampled_rectangles),
384 kSampleSize - sampled_rectangles.size(), *random_);
385 std::sort(sampled_rectangles.begin(), sampled_rectangles.end(),
386 [](const Rectangle& a, const Rectangle& b) {
387 const bool larger = std::make_pair(a.SizeX(), a.SizeY()) >
388 std::make_pair(b.SizeX(), b.SizeY());
389 // Double-check the invariant from
390 // FindRectanglesWithEnergyConflictMC() that given two returned
391 // rectangles there is one that is fully inside the other.
392 if (larger) {
393 // Rectangle b is fully contained inside a
394 DCHECK(a.x_min <= b.x_min && a.x_max >= b.x_max &&
395 a.y_min <= b.y_min && a.y_max >= b.y_max);
396 } else {
397 // Rectangle a is fully contained inside b
398 DCHECK(a.x_min >= b.x_min && a.x_max <= b.x_max &&
399 a.y_min >= b.y_min && a.y_max <= b.y_max);
400 }
401 return larger;
402 });
403 std::vector<IntegerValue> sizes_x, sizes_y;
404 sizes_x.reserve(active_box_ranges.size());
405 sizes_y.reserve(active_box_ranges.size());
406 std::vector<RectangleInRange> filtered_items;
407 filtered_items.reserve(active_box_ranges.size());
408 for (const Rectangle& r : sampled_rectangles) {
409 sizes_x.clear();
410 sizes_y.clear();
411 filtered_items.clear();
412 for (int i = 0; i < active_box_ranges.size(); ++i) {
413 const RectangleInRange& box = active_box_ranges[i];
414 const Rectangle intersection = box.GetMinimumIntersection(r);
415 if (intersection.SizeX() > 0 && intersection.SizeY() > 0) {
416 sizes_x.push_back(intersection.SizeX());
417 sizes_y.push_back(intersection.SizeY());
418 filtered_items.push_back(box);
419 }
420 }
421 // This check the feasibility of a related orthogonal packing problem where
422 // our rectangle is the bounding box, and we need to fit inside it a set of
423 // items corresponding to the minimum intersection of the original items
424 // with this bounding box.
425 const auto opp_result = orthogonal_packing_checker_.TestFeasibility(
426 sizes_x, sizes_y, {r.SizeX(), r.SizeY()},
427 OrthogonalPackingOptions{
428 .use_pairwise = true,
429 .use_dff_f0 = true,
430 .use_dff_f2 = true,
431 .dff2_max_number_of_parameters_to_check = 100});
432 if (opp_result.GetResult() == OrthogonalPackingResult::Status::INFEASIBLE &&
433 (best_conflict.opp_result.GetResult() !=
434 OrthogonalPackingResult::Status::INFEASIBLE ||
435 best_conflict.opp_result.GetItemsParticipatingOnConflict().size() >
436 opp_result.GetItemsParticipatingOnConflict().size())) {
437 best_conflict.items_for_opp = filtered_items;
438 best_conflict.opp_result = opp_result;
439 best_conflict.rectangle_with_too_much_energy = r;
440 }
441 // Use the fact that our rectangles are ordered in shrinking order to remove
442 // all items that will never contribute again.
443 filtered_items.swap(active_box_ranges);
444 }
445 if (best_conflict.opp_result.GetResult() ==
446 OrthogonalPackingResult::Status::INFEASIBLE) {
447 return best_conflict;
448 } else {
449 return std::nullopt;
450 }
451}

453std::vector<RectangleInRange>
454NonOverlappingRectanglesEnergyPropagator::GeneralizeExplanation(
455 const Conflict& conflict) {
456 const Rectangle& rectangle = conflict.rectangle_with_too_much_energy;
457 OrthogonalPackingResult relaxed_result = conflict.opp_result;
458
459 // Use the potential slack to have a stronger reason.
460 int used_slack_count = 0;
461 const auto& items = relaxed_result.GetItemsParticipatingOnConflict();
462 for (int i = 0; i < items.size(); ++i) {
463 if (!relaxed_result.HasSlack()) {
464 break;
465 }
466 const RectangleInRange& range = conflict.items_for_opp[items[i].index];
467 const RectangleInRange item_in_zero_level_range = {
468 .bounding_area = {.x_min = x_.LevelZeroStartMin(range.box_index),
469 .x_max = x_.LevelZeroStartMax(range.box_index) +
470 range.x_size,
471 .y_min = y_.LevelZeroStartMin(range.box_index),
472 .y_max = y_.LevelZeroStartMax(range.box_index) +
473 range.y_size},
474 .x_size = range.x_size,
475 .y_size = range.y_size};
476 // There is no point trying to intersect less the item with the rectangle
477 // than it would on zero-level. So do not waste the slack with it.
478 const Rectangle max_overlap =
479 item_in_zero_level_range.GetMinimumIntersection(rectangle);
480 used_slack_count += relaxed_result.TryUseSlackToReduceItemSize(
481 i, OrthogonalPackingResult::Coord::kCoordX, max_overlap.SizeX());
482 used_slack_count += relaxed_result.TryUseSlackToReduceItemSize(
483 i, OrthogonalPackingResult::Coord::kCoordY, max_overlap.SizeY());
484 }
485 num_conflicts_with_slack_ += (used_slack_count > 0);
486 VLOG_EVERY_N_SEC(2, 3)
487 << "Found a conflict on the OPP sub-problem of rectangle: " << rectangle
488 << " using "
489 << conflict.opp_result.GetItemsParticipatingOnConflict().size() << "/"
490 << conflict.items_for_opp.size() << " items.";
491
492 std::vector<RectangleInRange> ranges_for_explanation;
493 ranges_for_explanation.reserve(conflict.items_for_opp.size());
494 std::vector<OrthogonalPackingResult::Item> sorted_items =
495 relaxed_result.GetItemsParticipatingOnConflict();
496 // TODO(user) understand why sorting high-impact items first improves the
497 // benchmarks
498 std::sort(sorted_items.begin(), sorted_items.end(),
499 [](const OrthogonalPackingResult::Item& a,
500 const OrthogonalPackingResult::Item& b) {
501 return a.size_x * a.size_y > b.size_x * b.size_y;
502 });
503 for (const auto& item : sorted_items) {
504 const RectangleInRange& range = conflict.items_for_opp[item.index];
505 ranges_for_explanation.push_back(
506 RectangleInRange::BiggestWithMinIntersection(rectangle, range,
507 item.size_x, item.size_y));
508 }
509 return ranges_for_explanation;
510}

522bool NonOverlappingRectanglesEnergyPropagator::BuildAndReportEnergyTooLarge(
523 const std::vector<RectangleInRange>& ranges) {
524 if (ranges.size() == 2) {
525 num_conflicts_two_boxes_++;
526 return ClearAndAddTwoBoxesConflictReason(ranges[0].box_index,
527 ranges[1].box_index, &x_, &y_);
528 }
529 x_.ClearReason();
530 y_.ClearReason();
531 for (const auto& range : ranges) {
532 const int b = range.box_index;
533
534 x_.AddStartMinReason(b, range.bounding_area.x_min);
535 y_.AddStartMinReason(b, range.bounding_area.y_min);
536
537 x_.AddStartMaxReason(b, range.bounding_area.x_max - range.x_size);
538 y_.AddStartMaxReason(b, range.bounding_area.y_max - range.y_size);
539
540 x_.AddSizeMinReason(b);
541 y_.AddSizeMinReason(b);
542
543 x_.AddPresenceReason(b);
544 y_.AddPresenceReason(b);
545 }
546 x_.ImportOtherReasons(y_);
547 return x_.ReportConflict();
548}

552// We want for different propagation to reuse as much as possible the same
553// line. The idea behind this is to compute the 'canonical' line to use
554// when explaining that boxes overlap on the 'y_dim' dimension. We compute
555// the multiple of the biggest power of two that is common to all boxes.
556IntegerValue FindCanonicalValue(IntegerValue lb, IntegerValue ub) {
557 if (lb == ub) return lb;
558 if (lb <= 0 && ub > 0) return IntegerValue(0);
559 if (lb < 0 && ub <= 0) {
560 return -FindCanonicalValue(-ub, -lb);
561 }
562
563 int64_t mask = 0;
564 IntegerValue candidate = ub;
565 for (int o = 0; o < 62; ++o) {
566 mask = 2 * mask + 1;
567 const IntegerValue masked_ub(ub.value() & ~mask);
568 if (masked_ub >= lb) {
569 candidate = masked_ub;
570 } else {
571 break;
572 }
573 }
574 return candidate;
575}

577void SplitDisjointBoxes(const SchedulingConstraintHelper& x,
578 absl::Span<int> boxes,
579 std::vector<absl::Span<int>>* result) {
580 result->clear();
581 std::sort(boxes.begin(), boxes.end(), [&x](int a, int b) {
582 return x.ShiftedStartMin(a) < x.ShiftedStartMin(b);
583 });
584 int current_start = 0;
585 std::size_t current_length = 1;
586 IntegerValue current_max_end = x.EndMax(boxes[0]);
587
588 for (int b = 1; b < boxes.size(); ++b) {
589 const int box = boxes[b];
590 if (x.ShiftedStartMin(box) < current_max_end) {
591 // Merge.
592 current_length++;
593 current_max_end = std::max(current_max_end, x.EndMax(box));
594 } else {
595 if (current_length > 1) { // Ignore lists of size 1.
596 result->emplace_back(&boxes[current_start], current_length);
597 }
598 current_start = b;
599 current_length = 1;
600 current_max_end = x.EndMax(box);
601 }
602 }
603
604 // Push last span.
605 if (current_length > 1) {
606 result->emplace_back(&boxes[current_start], current_length);
607 }
608}
609
610// This function assumes that the left and right boxes overlap on the second
611// dimension, and that left cannot be after right.
612// It checks and pushes the lower bound of the right box and the upper bound
613// of the left box if need.
614//
615// If y is not null, it import the mandatory reason for the overlap on y in
616// the x helper.
617bool LeftBoxBeforeRightBoxOnFirstDimension(int left, int right,
618 SchedulingConstraintHelper* x,
619 SchedulingConstraintHelper* y) {
620 // left box2 pushes right box2.
621 const IntegerValue left_end_min = x->EndMin(left);
622 if (left_end_min > x->StartMin(right)) {
623 x->ClearReason();
624 x->AddPresenceReason(left);
625 x->AddPresenceReason(right);
626 x->AddReasonForBeingBefore(left, right);
627 x->AddEndMinReason(left, left_end_min);
628 if (y != nullptr) {
629 // left and right must overlap on y.
630 ClearAndAddMandatoryOverlapReason(left, right, y);
631 // Propagate with the complete reason.
632 x->ImportOtherReasons(*y);
633 }
634 RETURN_IF_FALSE(x->IncreaseStartMin(right, left_end_min));
635 }
636
637 // right box2 pushes left box2.
638 const IntegerValue right_start_max = x->StartMax(right);
639 if (right_start_max < x->EndMax(left)) {
640 x->ClearReason();
641 x->AddPresenceReason(left);
642 x->AddPresenceReason(right);
643 x->AddReasonForBeingBefore(left, right);
644 x->AddStartMaxReason(right, right_start_max);
645 if (y != nullptr) {
646 // left and right must overlap on y.
647 ClearAndAddMandatoryOverlapReason(left, right, y);
648 // Propagate with the complete reason.
649 x->ImportOtherReasons(*y);
650 }
651 RETURN_IF_FALSE(x->DecreaseEndMax(left, right_start_max));
652 }
653
654 return true;
655}

697bool NonOverlappingRectanglesDisjunctivePropagator::
698 FindBoxesThatMustOverlapAHorizontalLineAndPropagate(
699 bool fast_propagation, const SchedulingConstraintHelper& x,
700 SchedulingConstraintHelper* y) {
701 // Note that since we only push bounds on x, we cache the value for y just
702 // once.
703 if (!y->SynchronizeAndSetTimeDirection(true)) return false;
704
705 // Compute relevant boxes, the one with a mandatory part of y. Because we will
706 // need to sort it this way, we consider them by increasing start max.
707 indexed_boxes_.clear();
708 const auto temp = y->TaskByDecreasingStartMax();
709 for (int i = temp.size(); --i >= 0;) {
710 const int box = temp[i].task_index;
711 // Ignore absent boxes.
712 if (x.IsAbsent(box) || y->IsAbsent(box)) continue;
713
714 // Ignore boxes where the relevant presence literal is only on the y
715 // dimension, or if both intervals are optionals with different literals.
716 if (x.IsPresent(box) && !y->IsPresent(box)) continue;
717 if (!x.IsPresent(box) && !y->IsPresent(box) &&
718 x.PresenceLiteral(box) != y->PresenceLiteral(box)) {
719 continue;
720 }
721
722 const IntegerValue start_max = temp[i].time;
723 const IntegerValue end_min = y->EndMin(box);
724 if (start_max < end_min) {
725 indexed_boxes_.push_back({box, start_max, end_min});
726 }
727 }
728
729 // Less than 2 boxes, no propagation.
730 if (indexed_boxes_.size() < 2) return true;
731 ConstructOverlappingSets(/*already_sorted=*/true, &indexed_boxes_,
732 &events_overlapping_boxes_);
733
734 // Split lists of boxes into disjoint set of boxes (w.r.t. overlap).
735 boxes_to_propagate_.clear();
736 reduced_overlapping_boxes_.clear();
737 for (int i = 0; i < events_overlapping_boxes_.size(); ++i) {
738 SplitDisjointBoxes(x, absl::MakeSpan(events_overlapping_boxes_[i]),
739 &disjoint_boxes_);
740 for (absl::Span<int> sub_boxes : disjoint_boxes_) {
741 // Boxes are sorted in a stable manner in the Split method.
742 // Note that we do not use reduced_overlapping_boxes_ directly so that
743 // the order of iteration is deterministic.
744 const auto& insertion = reduced_overlapping_boxes_.insert(sub_boxes);
745 if (insertion.second) boxes_to_propagate_.push_back(sub_boxes);
746 }
747 }
748
749 // And finally propagate.
750 //
751 // TODO(user): Sorting of boxes seems influential on the performance. Test.
752 for (const absl::Span<const int> boxes : boxes_to_propagate_) {
753 // The case of two boxes should be taken care of during "fast" propagation,
754 // so we can skip it here.
755 if (!fast_propagation && boxes.size() <= 2) continue;
756
757 x_.ClearOtherHelper();
758 if (!x_.ResetFromSubset(x, boxes)) return false;
759
760 // Collect the common overlapping coordinates of all boxes.
761 IntegerValue lb(std::numeric_limits<int64_t>::min());
762 IntegerValue ub(std::numeric_limits<int64_t>::max());
763 for (const int b : boxes) {
764 lb = std::max(lb, y->StartMax(b));
765 ub = std::min(ub, y->EndMin(b) - 1);
766 }
767 CHECK_LE(lb, ub);
768
769 // We want for different propagation to reuse as much as possible the same
770 // line. The idea behind this is to compute the 'canonical' line to use
771 // when explaining that boxes overlap on the 'y_dim' dimension. We compute
772 // the multiple of the biggest power of two that is common to all boxes.
773 //
774 // TODO(user): We should scan the integer trail to find the oldest
775 // non-empty common interval. Then we can pick the canonical value within
776 // it.
777 const IntegerValue line_to_use_for_reason = FindCanonicalValue(lb, ub);
778
779 // Setup x_dim for propagation.
780 x_.SetOtherHelper(y, boxes, line_to_use_for_reason);
781
782 if (fast_propagation) {
783 if (x_.NumTasks() == 2) {
784 // In that case, we can use simpler algorithms.
785 // Note that this case happens frequently (~30% of all calls to this
786 // method according to our tests).
787 RETURN_IF_FALSE(PropagateOnXWhenOnlyTwoBoxes());
788 } else {
789 RETURN_IF_FALSE(overload_checker_.Propagate());
790 RETURN_IF_FALSE(forward_detectable_precedences_.Propagate());
791 RETURN_IF_FALSE(backward_detectable_precedences_.Propagate());
792 }
793 } else {
794 DCHECK_GT(x_.NumTasks(), 2);
795 RETURN_IF_FALSE(forward_not_last_.Propagate());
796 RETURN_IF_FALSE(backward_not_last_.Propagate());
797 RETURN_IF_FALSE(backward_edge_finding_.Propagate());
798 RETURN_IF_FALSE(forward_edge_finding_.Propagate());
799 }
800 }
801
802 return true;
803}

805bool NonOverlappingRectanglesDisjunctivePropagator::Propagate() {
806 global_x_.SetTimeDirection(true);
807 global_y_.SetTimeDirection(true);
808
809 // Note that the code assumes that this was registered twice in fast and slow
810 // mode. So we will not redo some propagation in slow mode that was already
811 // done by the fast mode.
812 const bool fast_propagation = watcher_->GetCurrentId() == fast_id_;
813 RETURN_IF_FALSE(FindBoxesThatMustOverlapAHorizontalLineAndPropagate(
814 fast_propagation, global_x_, &global_y_));
815
816 // We can actually swap dimensions to propagate vertically.
817 RETURN_IF_FALSE(FindBoxesThatMustOverlapAHorizontalLineAndPropagate(
818 fast_propagation, global_y_, &global_x_));
819
820 return true;
821}

823// Specialized propagation on only two boxes that must intersect with the
824// given y_line_for_reason.
825bool NonOverlappingRectanglesDisjunctivePropagator::
826 PropagateOnXWhenOnlyTwoBoxes() {
827 if (!x_.IsPresent(0) || !x_.IsPresent(1)) return true;
828
829 // For each direction and each order, we test if the boxes can be disjoint.
830 const int state =
831 (x_.EndMin(0) <= x_.StartMax(1)) + 2 * (x_.EndMin(1) <= x_.StartMax(0));
832 switch (state) {
833 case 0: { // Conflict.
834 ClearAndAddMandatoryOverlapReason(0, 1, &x_);
835 // Note that the secondary helper is set on x.
836 return x_.ReportConflict();
837 }
838 case 1: { // b1 is left of b2.
839 return LeftBoxBeforeRightBoxOnFirstDimension(0, 1, &x_, /*y=*/nullptr);
840 }
841 case 2: { // b2 is left of b1.
842 return LeftBoxBeforeRightBoxOnFirstDimension(1, 0, &x_, /*y=*/nullptr);
843 }
844 default: { // Nothing to deduce.
845 return true;
846 }
847 }

872bool RectanglePairwisePropagator::Propagate() {
873 if (!global_x_.SynchronizeAndSetTimeDirection(true)) return false;
874 if (!global_y_.SynchronizeAndSetTimeDirection(true)) return false;
875
876 num_calls_++;
877
878 horizontal_zero_area_boxes_.clear();
879 vertical_zero_area_boxes_.clear();
880 point_zero_area_boxes_.clear();
881 non_zero_area_boxes_.clear();
882 for (int b = 0; b < global_x_.NumTasks(); ++b) {
883 if (!global_x_.IsPresent(b) || !global_y_.IsPresent(b)) continue;
884 const IntegerValue x_size_max = global_x_.SizeMax(b);
885 const IntegerValue y_size_max = global_y_.SizeMax(b);
886 ItemForPairwiseRestriction* box;
887 if (x_size_max == 0) {
888 if (y_size_max == 0) {
889 box = &point_zero_area_boxes_.emplace_back();
890 } else {
891 box = &vertical_zero_area_boxes_.emplace_back();
892 }
893 } else if (y_size_max == 0) {
894 box = &horizontal_zero_area_boxes_.emplace_back();
895 } else {
896 box = &non_zero_area_boxes_.emplace_back();
897 }
898 *box = ItemForPairwiseRestriction{.index = b,
899 .x = {.start_min = global_x_.StartMin(b),
900 .start_max = global_x_.StartMax(b),
901 .end_min = global_x_.EndMin(b),
902 .end_max = global_x_.EndMax(b)},
903 .y = {.start_min = global_y_.StartMin(b),
904 .start_max = global_y_.StartMax(b),
905 .end_min = global_y_.EndMin(b),
906 .end_max = global_y_.EndMax(b)}};
907 }
908
909 std::vector<PairwiseRestriction> restrictions;
910 RETURN_IF_FALSE(FindRestrictionsAndPropagateConflict(non_zero_area_boxes_,
911 &restrictions));
912
913 // Check zero area boxes against non-zero area boxes.
914 RETURN_IF_FALSE(FindRestrictionsAndPropagateConflict(
915 non_zero_area_boxes_, horizontal_zero_area_boxes_, &restrictions));
916 RETURN_IF_FALSE(FindRestrictionsAndPropagateConflict(
917 non_zero_area_boxes_, vertical_zero_area_boxes_, &restrictions));
918 RETURN_IF_FALSE(FindRestrictionsAndPropagateConflict(
919 non_zero_area_boxes_, point_zero_area_boxes_, &restrictions));
920
921 // Check vertical zero area boxes against horizontal zero area boxes.
922 RETURN_IF_FALSE(FindRestrictionsAndPropagateConflict(
923 vertical_zero_area_boxes_, horizontal_zero_area_boxes_, &restrictions));
924
925 for (const PairwiseRestriction& restriction : restrictions) {
926 RETURN_IF_FALSE(PropagateTwoBoxes(restriction));
927 }
928 return true;
929}

968bool RectanglePairwisePropagator::PropagateTwoBoxes(
969 const PairwiseRestriction& restriction) {
970 const int box1 = restriction.first_index;
971 const int box2 = restriction.second_index;
972 switch (restriction.type) {
973 case PairwiseRestriction::PairwiseRestrictionType::CONFLICT:
974 num_pairwise_conflicts_++;
975 return ClearAndAddTwoBoxesConflictReason(box1, box2, &global_x_,
976 &global_y_);
977 case PairwiseRestriction::PairwiseRestrictionType::FIRST_LEFT_OF_SECOND:
978 num_pairwise_propagations_++;
979 return LeftBoxBeforeRightBoxOnFirstDimension(box1, box2, &global_x_,
980 &global_y_);
981 case PairwiseRestriction::PairwiseRestrictionType::FIRST_RIGHT_OF_SECOND:
982 num_pairwise_propagations_++;
983 return LeftBoxBeforeRightBoxOnFirstDimension(box2, box1, &global_x_,
984 &global_y_);
985 case PairwiseRestriction::PairwiseRestrictionType::FIRST_BELOW_SECOND:
986 num_pairwise_propagations_++;
987 return LeftBoxBeforeRightBoxOnFirstDimension(box1, box2, &global_y_,
988 &global_x_);
989 case PairwiseRestriction::PairwiseRestrictionType::FIRST_ABOVE_SECOND:
990 num_pairwise_propagations_++;
991 return LeftBoxBeforeRightBoxOnFirstDimension(box2, box1, &global_y_,
992 &global_x_);
993 }
