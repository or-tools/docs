55// A literal is used to represent a variable or its negation. If it represents
56// the variable it is said to be positive. If it represent its negation, it is
57// said to be negative. We support two representations as an integer.
58//
59// The "signed" encoding of a literal is convenient for input/output and is used
60// in the cnf file format. For a 0-based variable index x, (x + 1) represent the
61// variable x and -(x + 1) represent its negation. The signed value 0 is an
62// undefined literal and this class can never contain it.
63//
64// The "index" encoding of a literal is convenient as an index to an array
65// and is the one used internally for efficiency. It is always positive or zero,
66// and for a 0-based variable index x, (x << 1) encode the variable x and the
67// same number XOR 1 encode its negation.
68class Literal {
69 public:
70 // Not explicit for tests so we can write:
71 // vector<literal> literal = {+1, -3, +4, -9};
72 Literal(int signed_value) // NOLINT
73 : index_(signed_value > 0 ? ((signed_value - 1) << 1)
74 : ((-signed_value - 1) << 1) ^ 1) {
75 CHECK_NE(signed_value, 0);
76 }
77
78 Literal() = default;
79 explicit Literal(LiteralIndex index) : index_(index.value()) {}
80 Literal(BooleanVariable variable, bool is_positive)
81 : index_(is_positive ? (variable.value() << 1)
82 : (variable.value() << 1) ^ 1) {}
83
84 // We want a literal to be implicitly converted to a LiteralIndex().
85 // Before this, we used to have many literal.Index() that didn't add anything.
86 //
87 // TODO(user): LiteralIndex might not even be needed, but because of the
88 // signed value business, it is still safer with it.
89 operator LiteralIndex() const { return Index(); } // NOLINT
90
91 BooleanVariable Variable() const { return BooleanVariable(index_ >> 1); }
92 bool IsPositive() const { return !(index_ & 1); }
93 bool IsNegative() const { return (index_ & 1); }
94
95 LiteralIndex Index() const { return LiteralIndex(index_); }
96 LiteralIndex NegatedIndex() const { return LiteralIndex(index_ ^ 1); }
97
98 int SignedValue() const {
99 return (index_ & 1) ? -((index_ >> 1) + 1) : ((index_ >> 1) + 1);
100 }
101
102 Literal Negated() const { return Literal(NegatedIndex()); }
103
104 std::string DebugString() const {
105 return absl::StrFormat("%+d", SignedValue());
106 }
107
108 bool operator==(Literal other) const { return index_ == other.index_; }
109 bool operator!=(Literal other) const { return index_ != other.index_; }
110 bool operator<(const Literal& other) const { return index_ < other.index_; }
111
112 private:
113 int index_;
114};

142// Holds the current variable assignment of the solver.
143// Each variable can be unassigned or be assigned to true or false.
144class VariablesAssignment {
145 public:
146 VariablesAssignment() = default;
147 explicit VariablesAssignment(int num_variables) { Resize(num_variables); }
148
149 // This type is neither copyable nor movable.
150 VariablesAssignment(const VariablesAssignment&) = delete;
151 VariablesAssignment& operator=(const VariablesAssignment&) = delete;
152 void Resize(int num_variables) {
153 assignment_.Resize(LiteralIndex(num_variables << 1));
154 }
155
156 // Makes the given literal true by assigning its underlying variable to either
157 // true or false depending on the literal sign. This can only be called on an
158 // unassigned variable.
159 void AssignFromTrueLiteral(Literal literal) {
160 DCHECK(!VariableIsAssigned(literal.Variable()));
161 assignment_.Set(literal.Index());
162 }
163
164 // Unassign the variable corresponding to the given literal.
165 // This can only be called on an assigned variable.
166 void UnassignLiteral(Literal literal) {
167 DCHECK(VariableIsAssigned(literal.Variable()));
168 assignment_.ClearTwoBits(literal.Index());
169 }
170
171 // Literal getters. Note that both can be false, in which case the
172 // corresponding variable is not assigned.
173 bool LiteralIsFalse(Literal literal) const {
174 return assignment_.IsSet(literal.NegatedIndex());
175 }
176 bool LiteralIsTrue(Literal literal) const {
177 return assignment_.IsSet(literal.Index());
178 }
179 bool LiteralIsAssigned(Literal literal) const {
180 return assignment_.AreOneOfTwoBitsSet(literal.Index());
181 }
182
183 // Returns true iff the given variable is assigned.
184 bool VariableIsAssigned(BooleanVariable var) const {
185 return assignment_.AreOneOfTwoBitsSet(LiteralIndex(var.value() << 1));
186 }
187
188 // Returns the literal of the given variable that is assigned to true.
189 // That is, depending on the variable, it can be the positive literal or the
190 // negative one. Only call this on an assigned variable.
191 Literal GetTrueLiteralForAssignedVariable(BooleanVariable var) const {
192 DCHECK(VariableIsAssigned(var));
193 return Literal(var, assignment_.IsSet(LiteralIndex(var.value() << 1)));
194 }
195
196 int NumberOfVariables() const { return assignment_.size().value() / 2; }
197
198 private:
199 // The encoding is as follows:
200 // - assignment_.IsSet(literal.Index()) means literal is true.
201 // - assignment_.IsSet(literal.Index() ^ 1]) means literal is false.
202 // - If both are false, then the variable (and the literal) is unassigned.
203 Bitset64<LiteralIndex> assignment_;
204
205 friend class AssignmentView;
206};

208// For "hot" loop, it is better not to reload the Bitset64 pointer on each
209// check.
210class AssignmentView {
211 public:
212 explicit AssignmentView(const VariablesAssignment& assignment)
213 : view_(assignment.assignment_.const_view()) {}
214
215 bool LiteralIsFalse(Literal literal) const {
216 return view_[literal.NegatedIndex()];
217 }
218
219 bool LiteralIsTrue(Literal literal) const { return view_[literal.Index()]; }
220
221 private:
222 Bitset64<LiteralIndex>::ConstView view_;
223};

229// Information about a variable assignment.
230struct AssignmentInfo {
231 // The decision level at which this assignment was made. This starts at 0 and
232 // increases each time the solver takes a search decision.
233 //
234 // TODO(user): We may be able to get rid of that for faster enqueues. Most of
235 // the code only need to know if this is 0 or the highest level, and for the
236 // LBD computation, the literal of the conflict are already ordered by level,
237 // so we could do it fairly efficiently.
238 //
239 // TODO(user): We currently don't support more than 2^28 decision levels. That
240 // should be enough for most practical problem, but we should fail properly if
241 // this limit is reached.
242 uint32_t level : 28;
243
244 // The type of assignment (see AssignmentType below).
245 //
246 // Note(user): We currently don't support more than 16 types of assignment.
247 // This is checked in RegisterPropagator().
248 mutable uint32_t type : 4;
249
250 // The index of this assignment in the trail.
251 int32_t trail_index;
252
253 std::string DebugString() const {
254 return absl::StrFormat("level:%d type:%d trail_index:%d", level, type,
255 trail_index);
256 }
257};
258static_assert(sizeof(AssignmentInfo) == 8,
259 "ERROR_AssignmentInfo_is_not_well_compacted");

263struct AssignmentType {
264 static constexpr int kCachedReason = 0;
265 static constexpr int kUnitReason = 1;
266 static constexpr int kSearchDecision = 2;
267 static constexpr int kSameReasonAs = 3;

269 // Propagator ids starts from there and are created dynamically.
270 static constexpr int kFirstFreePropagationId = 4;
271};

273// The solver trail stores the assignment made by the solver in order.
274// This class is responsible for maintaining the assignment of each variable
275// and the information of each assignment.
276class Trail {
277 public:
278 Trail() {
279 current_info_.trail_index = 0;
280 current_info_.level = 0;
281 }
282
283 // This type is neither copyable nor movable.
284 Trail(const Trail&) = delete;
285 Trail& operator=(const Trail&) = delete;
286
287 void Resize(int num_variables);
288
289 // Registers a propagator. This assigns a unique id to this propagator and
290 // calls SetPropagatorId() on it.
291 void RegisterPropagator(SatPropagator* propagator);
292
293 // Enqueues the assignment that make the given literal true on the trail. This
294 // should only be called on unassigned variables.
295 void SetCurrentPropagatorId(int propagator_id) {
296 current_info_.type = propagator_id;
297 }
298 void FastEnqueue(Literal true_literal) {
299 DCHECK(!assignment_.VariableIsAssigned(true_literal.Variable()));
300 trail_[current_info_.trail_index] = true_literal;
301 info_[true_literal.Variable()] = current_info_;
302 assignment_.AssignFromTrueLiteral(true_literal);
303 ++current_info_.trail_index;
304 }
305 void Enqueue(Literal true_literal, int propagator_id) {
306 SetCurrentPropagatorId(propagator_id);
307 FastEnqueue(true_literal);
308 }
309
310 // Specific Enqueue() version for the search decision.
311 void EnqueueSearchDecision(Literal true_literal) {
312 Enqueue(true_literal, AssignmentType::kSearchDecision);
313 }
314
315 // Specific Enqueue() version for a fixed variable.
316 void EnqueueWithUnitReason(Literal true_literal) {
317 Enqueue(true_literal, AssignmentType::kUnitReason);
318 }
319
320 // Some constraints propagate a lot of literals at once. In these cases, it is
321 // more efficient to have all the propagated literals except the first one
322 // referring to the reason of the first of them.
323 void EnqueueWithSameReasonAs(Literal true_literal,
324 BooleanVariable reference_var) {
325 reference_var_with_same_reason_as_[true_literal.Variable()] = reference_var;
326 Enqueue(true_literal, AssignmentType::kSameReasonAs);
327 }
328
329 // Enqueues the given literal using the current content of
330 // GetEmptyVectorToStoreReason() as the reason. This API is a bit more
331 // leanient and does not require the literal to be unassigned. If it is
332 // already assigned to false, then MutableConflict() will be set appropriately
333 // and this will return false otherwise this will enqueue the literal and
334 // returns true.
335 ABSL_MUST_USE_RESULT bool EnqueueWithStoredReason(Literal true_literal) {
336 if (assignment_.LiteralIsTrue(true_literal)) return true;
337 if (assignment_.LiteralIsFalse(true_literal)) {
338 *MutableConflict() = reasons_repository_[Index()];
339 MutableConflict()->push_back(true_literal);
340 return false;
341 }
342
343 Enqueue(true_literal, AssignmentType::kCachedReason);
344 const BooleanVariable var = true_literal.Variable();
345 reasons_[var] = reasons_repository_[info_[var].trail_index];
346 old_type_[var] = info_[var].type;
347 info_[var].type = AssignmentType::kCachedReason;
348 return true;
349 }
350
351 // Returns the reason why this variable was assigned.
352 //
353 // Note that this shouldn't be called on a variable at level zero, because we
354 // don't cleanup the reason data for these variables but the underlying
355 // clauses may have been deleted.
356 absl::Span<const Literal> Reason(BooleanVariable var) const;
357
358 // Returns the "type" of an assignment (see AssignmentType). Note that this
359 // function never returns kSameReasonAs or kCachedReason, it instead returns
360 // the initial type that caused this assignment. As such, it is different
361 // from Info(var).type and the latter should not be used outside this class.
362 int AssignmentType(BooleanVariable var) const;
363
364 // If a variable was propagated with EnqueueWithSameReasonAs(), returns its
365 // reference variable. Otherwise return the given variable.
366 BooleanVariable ReferenceVarWithSameReason(BooleanVariable var) const;
367
368 // This can be used to get a location at which the reason for the literal
369 // at trail_index on the trail can be stored. This clears the vector before
370 // returning it.
371 std::vector<Literal>* GetEmptyVectorToStoreReason(int trail_index) const {
372 if (trail_index >= reasons_repository_.size()) {
373 reasons_repository_.resize(trail_index + 1);
374 }
375 reasons_repository_[trail_index].clear();
376 return &reasons_repository_[trail_index];
377 }
378
379 // Shortcut for GetEmptyVectorToStoreReason(Index()).
380 std::vector<Literal>* GetEmptyVectorToStoreReason() const {
381 return GetEmptyVectorToStoreReason(Index());
382 }
383
384 // Explicitly overwrite the reason so that the given propagator will be
385 // asked for it. This is currently only used by the BinaryImplicationGraph.
386 void ChangeReason(int trail_index, int propagator_id) {
387 const BooleanVariable var = trail_[trail_index].Variable();
388 info_[var].type = propagator_id;
389 old_type_[var] = propagator_id;
390 }
391
392 // Reverts the trail and underlying assignment to the given target trail
393 // index. Note that we do not touch the assignment info.
394 void Untrail(int target_trail_index) {
395 const int index = Index();
396 num_untrailed_enqueues_ += index - target_trail_index;
397 for (int i = target_trail_index; i < index; ++i) {
398 assignment_.UnassignLiteral(trail_[i]);
399 }
400 current_info_.trail_index = target_trail_index;
401 }
402
403 // Changes the decision level used by the next Enqueue().
404 void SetDecisionLevel(int level) { current_info_.level = level; }
405 int CurrentDecisionLevel() const { return current_info_.level; }
406
407 // Generic interface to set the current failing clause.
408 //
409 // Returns the address of a vector where a client can store the current
410 // conflict. This vector will be returned by the FailingClause() call.
411 std::vector<Literal>* MutableConflict() {
412 failing_sat_clause_ = nullptr;
413 return &conflict_;
414 }
415
416 // Returns the last conflict.
417 absl::Span<const Literal> FailingClause() const {
418 if (DEBUG_MODE && debug_checker_ != nullptr) {
419 CHECK(debug_checker_(conflict_));
420 }
421 return conflict_;
422 }
423
424 // Specific SatClause interface so we can update the conflict clause activity.
425 // Note that MutableConflict() automatically sets this to nullptr, so we can
426 // know whether or not the last conflict was caused by a clause.
427 void SetFailingSatClause(SatClause* clause) { failing_sat_clause_ = clause; }
428 SatClause* FailingSatClause() const { return failing_sat_clause_; }
429
430 // Getters.
431 int NumVariables() const { return trail_.size(); }
432 int64_t NumberOfEnqueues() const { return num_untrailed_enqueues_ + Index(); }
433 int Index() const { return current_info_.trail_index; }
434 // This accessor can return trail_.end(). operator[] cannot. This allows
435 // normal std:vector operations, such as assign(begin, end).
436 std::vector<Literal>::const_iterator IteratorAt(int index) const {
437 return trail_.begin() + index;
438 }
439 const Literal& operator[](int index) const { return trail_[index]; }
440 const VariablesAssignment& Assignment() const { return assignment_; }
441 const AssignmentInfo& Info(BooleanVariable var) const {
442 DCHECK_GE(var, 0);
443 DCHECK_LT(var, info_.size());
444 return info_[var];
445 }
446
447 // Print the current literals on the trail.
448 std::string DebugString() const {
449 std::string result;
450 for (int i = 0; i < current_info_.trail_index; ++i) {
451 if (!result.empty()) result += " ";
452 result += trail_[i].DebugString();
453 }
454 return result;
455 }
456
457 void RegisterDebugChecker(
458 std::function<bool(absl::Span<const Literal> clause)> checker) {
459 debug_checker_ = std::move(checker);
460 }
461
462 private:
463 int64_t num_untrailed_enqueues_ = 0;
464 AssignmentInfo current_info_;
465 VariablesAssignment assignment_;
466 std::vector<Literal> trail_;
467 std::vector<Literal> conflict_;
468 absl::StrongVector<BooleanVariable, AssignmentInfo> info_;
469 SatClause* failing_sat_clause_;
470
471 // Data used by EnqueueWithSameReasonAs().
472 absl::StrongVector<BooleanVariable, BooleanVariable>
473 reference_var_with_same_reason_as_;
474
475 // Reason cache. Mutable since we want the API to be the same whether the
476 // reason are cached or not.
477 //
478 // When a reason is computed for the first time, we change the type of the
479 // variable assignment to kCachedReason so that we know that if it is needed
480 // again the reason can just be retrieved by a direct access to reasons_. The
481 // old type is saved in old_type_ and can be retrieved by
482 // AssignmentType().
483 //
484 // Note(user): Changing the type is not "clean" but it is efficient. The idea
485 // is that it is important to do as little as possible when pushing/popping
486 // literals on the trail. Computing the reason happens a lot less often, so it
487 // is okay to do slightly more work then. Note also, that we don't need to
488 // do anything on "untrail", the kCachedReason type will be overwritten when
489 // the same variable is assigned again.
490 //
491 // TODO(user): An alternative would be to change the sign of the type. This
492 // would remove the need for a separate old_type_ vector, but it requires
493 // more bits for the type filed in AssignmentInfo.
494 //
495 // Note that we use a deque for the reason repository so that if we add
496 // variables, the memory address of the vectors (kept in reasons_) are still
497 // valid.
498 mutable std::deque<std::vector<Literal>> reasons_repository_;
499 mutable absl::StrongVector<BooleanVariable, absl::Span<const Literal>>
500 reasons_;
501 mutable absl::StrongVector<BooleanVariable, int> old_type_;
502
503 // This is used by RegisterPropagator() and Reason().
504 std::vector<SatPropagator*> propagators_;
505
506 std::function<bool(absl::Span<const Literal> clause)> debug_checker_ =
507 nullptr;
508};

510// Base class for all the SAT constraints.
511class SatPropagator {
512 public:
513 explicit SatPropagator(const std::string& name)
514 : name_(name), propagator_id_(-1), propagation_trail_index_(0) {}
515
516 // This type is neither copyable nor movable.
517 SatPropagator(const SatPropagator&) = delete;
518 SatPropagator& operator=(const SatPropagator&) = delete;
519 virtual ~SatPropagator() = default;
520
521 // Sets/Gets this propagator unique id.
522 void SetPropagatorId(int id) { propagator_id_ = id; }
523 int PropagatorId() const { return propagator_id_; }
524
525 // Inspects the trail from propagation_trail_index_ until at least one literal
526 // is propagated. Returns false iff a conflict is detected (in which case
527 // trail->SetFailingClause() must be called).
528 //
529 // This must update propagation_trail_index_ so that all the literals before
530 // it have been propagated. In particular, if nothing was propagated, then
531 // PropagationIsDone() must return true.
532 virtual bool Propagate(Trail* trail) = 0;
533
534 // Reverts the state so that all the literals with a trail index greater or
535 // equal to the given one are not processed for propagation. Note that the
536 // trail current decision level is already reverted before this is called.
537 //
538 // TODO(user): Currently this is called at each Backtrack(), but we could
539 // bundle the calls in case multiple conflict one after the other are detected
540 // even before the Propagate() call of a SatPropagator is called.
541 //
542 // TODO(user): It is not yet 100% the case, but this can be guaranteed to be
543 // called with a trail index that will always be the start of a new decision
544 // level.
545 virtual void Untrail(const Trail& trail, int trail_index) {
546 propagation_trail_index_ = std::min(propagation_trail_index_, trail_index);
547 }
548
549 // Explains why the literal at given trail_index was propagated by returning a
550 // reason for this propagation. This will only be called for literals that are
551 // on the trail and were propagated by this class.
552 //
553 // The interpretation is that because all the literals of a reason were
554 // assigned to false, we could deduce the assignment of the given variable.
555 //
556 // The returned Span has to be valid until the literal is untrailed. A client
557 // can use trail_.GetEmptyVectorToStoreReason() if it doesn't have a memory
558 // location that already contains the reason.
559 virtual absl::Span<const Literal> Reason(const Trail& trail,
560 int /*trail_index*/) const {
561 LOG(FATAL) << "Not implemented.";
562 return {};
563 }
564
565 // Returns true if all the preconditions for Propagate() are satisfied.
566 // This is just meant to be used in a DCHECK.
567 bool PropagatePreconditionsAreSatisfied(const Trail& trail) const;
568
569 // Returns true iff all the trail was inspected by this propagator.
570 bool PropagationIsDone(const Trail& trail) const {
571 return propagation_trail_index_ == trail.Index();
572 }
573
574 // Small optimization: If a propagator does not contain any "constraints"
575 // there is no point calling propagate on it. Before each propagation, the
576 // solver will checks for emptiness, and construct an optimized list of
577 // propagator before looping many time over the list.
578 virtual bool IsEmpty() const { return false; }
579
580 protected:
581 const std::string name_;
582 int propagator_id_;
583 int propagation_trail_index_;
584};

590inline bool SatPropagator::PropagatePreconditionsAreSatisfied(
591 const Trail& trail) const {
592 if (propagation_trail_index_ > trail.Index()) {
593 LOG(INFO) << "Issue in '" << name_ << ":"
594 << " propagation_trail_index_=" << propagation_trail_index_
595 << " trail_.Index()=" << trail.Index();
596 return false;
597 }
598 if (propagation_trail_index_ < trail.Index() &&
599 trail.Info(trail[propagation_trail_index_].Variable()).level !=
600 trail.CurrentDecisionLevel()) {
601 LOG(INFO) << "Issue in '" << name_ << "':" << " propagation_trail_index_="
602 << propagation_trail_index_ << " trail_.Index()=" << trail.Index()
603 << " level_at_propagation_index="
604 << trail.Info(trail[propagation_trail_index_].Variable()).level
605 << " current_decision_level=" << trail.CurrentDecisionLevel();
606 return false;
607 }
608 return true;
609}

632inline BooleanVariable Trail::ReferenceVarWithSameReason(
633 BooleanVariable var) const {
634 DCHECK(Assignment().VariableIsAssigned(var));
635 // Note that we don't use AssignmentType() here.
636 if (info_[var].type == AssignmentType::kSameReasonAs) {
637 var = reference_var_with_same_reason_as_[var];
638 DCHECK(Assignment().VariableIsAssigned(var));
639 DCHECK_NE(info_[var].type, AssignmentType::kSameReasonAs);
640 }
641 return var;
642}
643
644inline int Trail::AssignmentType(BooleanVariable var) const {
645 if (info_[var].type == AssignmentType::kSameReasonAs) {
646 var = reference_var_with_same_reason_as_[var];
647 DCHECK_NE(info_[var].type, AssignmentType::kSameReasonAs);
648 }
649 const int type = info_[var].type;
650 return type != AssignmentType::kCachedReason ? type : old_type_[var];
651}

653inline absl::Span<const Literal> Trail::Reason(BooleanVariable var) const {
654 // Special case for AssignmentType::kSameReasonAs to avoid a recursive call.
655 var = ReferenceVarWithSameReason(var);
656
657 // Fast-track for cached reason.
