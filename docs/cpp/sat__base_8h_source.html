55// A literal is used to represent a variable or its negation. If it represents
56// the variable it is said to be positive. If it represent its negation, it is
57// said to be negative. We support two representations as an integer.
58//
59// The "signed" encoding of a literal is convenient for input/output and is used
60// in the cnf file format. For a 0-based variable index x, (x + 1) represent the
61// variable x and -(x + 1) represent its negation. The signed value 0 is an
62// undefined literal and this class can never contain it.
63//
64// The "index" encoding of a literal is convenient as an index to an array
65// and is the one used internally for efficiency. It is always positive or zero,
66// and for a 0-based variable index x, (x << 1) encode the variable x and the
67// same number XOR 1 encode its negation.
68class Literal {
69 public:
70 // Not explicit for tests so we can write:
71 // vector<literal> literal = {+1, -3, +4, -9};
72 Literal(int signed_value) // NOLINT
73 : index_(signed_value > 0 ? ((signed_value - 1) << 1)
74 : ((-signed_value - 1) << 1) ^ 1) {
75 CHECK_NE(signed_value, 0);
76 }
77
78 Literal() {}
79 explicit Literal(LiteralIndex index) : index_(index.value()) {}
80 Literal(BooleanVariable variable, bool is_positive)
81 : index_(is_positive ? (variable.value() << 1)
82 : (variable.value() << 1) ^ 1) {}
83
84 BooleanVariable Variable() const { return BooleanVariable(index_ >> 1); }
85 bool IsPositive() const { return !(index_ & 1); }
86 bool IsNegative() const { return (index_ & 1); }
87
88 LiteralIndex Index() const { return LiteralIndex(index_); }
89 LiteralIndex NegatedIndex() const { return LiteralIndex(index_ ^ 1); }
90
91 int SignedValue() const {
92 return (index_ & 1) ? -((index_ >> 1) + 1) : ((index_ >> 1) + 1);
93 }
94
95 Literal Negated() const { return Literal(NegatedIndex()); }
96
97 std::string DebugString() const {
98 return absl::StrFormat("%+d", SignedValue());
99 }
100 bool operator==(Literal other) const { return index_ == other.index_; }
101 bool operator!=(Literal other) const { return index_ != other.index_; }
102
103 bool operator<(const Literal& literal) const {
104 return Index() < literal.Index();
105 }
106
107 private:
108 int index_;
109};

132// Holds the current variable assignment of the solver.
133// Each variable can be unassigned or be assigned to true or false.
134class VariablesAssignment {
135 public:
136 VariablesAssignment() {}
137 explicit VariablesAssignment(int num_variables) { Resize(num_variables); }
138 void Resize(int num_variables) {
139 assignment_.Resize(LiteralIndex(num_variables << 1));
140 }
141
142 // Makes the given literal true by assigning its underlying variable to either
143 // true or false depending on the literal sign. This can only be called on an
144 // unassigned variable.
145 void AssignFromTrueLiteral(Literal literal) {
146 DCHECK(!VariableIsAssigned(literal.Variable()));
147 assignment_.Set(literal.Index());
148 }
149
150 // Unassign the variable corresponding to the given literal.
151 // This can only be called on an assigned variable.
152 void UnassignLiteral(Literal literal) {
153 DCHECK(VariableIsAssigned(literal.Variable()));
154 assignment_.ClearTwoBits(literal.Index());
155 }
156
157 // Literal getters. Note that both can be false, in which case the
158 // corresponding variable is not assigned.
159 bool LiteralIsFalse(Literal literal) const {
160 return assignment_.IsSet(literal.NegatedIndex());
161 }
162 bool LiteralIsTrue(Literal literal) const {
163 return assignment_.IsSet(literal.Index());
164 }
165 bool LiteralIsAssigned(Literal literal) const {
166 return assignment_.AreOneOfTwoBitsSet(literal.Index());
167 }
168
169 // Returns true iff the given variable is assigned.
170 bool VariableIsAssigned(BooleanVariable var) const {
171 return assignment_.AreOneOfTwoBitsSet(LiteralIndex(var.value() << 1));
172 }
173
174 // Returns the literal of the given variable that is assigned to true.
175 // That is, depending on the variable, it can be the positive literal or the
176 // negative one. Only call this on an assigned variable.
177 Literal GetTrueLiteralForAssignedVariable(BooleanVariable var) const {
178 DCHECK(VariableIsAssigned(var));
179 return Literal(var, assignment_.IsSet(LiteralIndex(var.value() << 1)));
180 }
181
182 int NumberOfVariables() const { return assignment_.size().value() / 2; }
183
184 private:
185 // The encoding is as follows:
186 // - assignment_.IsSet(literal.Index()) means literal is true.
187 // - assignment_.IsSet(literal.Index() ^ 1]) means literal is false.
188 // - If both are false, then the variable (and the literal) is unassigned.
189 Bitset64<LiteralIndex> assignment_;
190
191 DISALLOW_COPY_AND_ASSIGN(VariablesAssignment);
192};

198// Information about a variable assignment.
199struct AssignmentInfo {
200 // The decision level at which this assignment was made. This starts at 0 and
201 // increases each time the solver takes a search decision.
202 //
203 // TODO(user): We may be able to get rid of that for faster enqueues. Most of
204 // the code only need to know if this is 0 or the highest level, and for the
205 // LBD computation, the literal of the conflict are already ordered by level,
206 // so we could do it fairly efficiently.
207 //
208 // TODO(user): We currently don't support more than 2^28 decision levels. That
209 // should be enough for most practical problem, but we should fail properly if
210 // this limit is reached.
211 uint32_t level : 28;
212
213 // The type of assignment (see AssignmentType below).
214 //
215 // Note(user): We currently don't support more than 16 types of assignment.
216 // This is checked in RegisterPropagator().
217 mutable uint32_t type : 4;
218
219 // The index of this assignment in the trail.
220 int32_t trail_index;
221
222 std::string DebugString() const {
223 return absl::StrFormat("level:%d type:%d trail_index:%d", level, type,
224 trail_index);
225 }
226};
227static_assert(sizeof(AssignmentInfo) == 8,
228 "ERROR_AssignmentInfo_is_not_well_compacted");

232struct AssignmentType {

235 static constexpr int kSearchDecision = 2;
236 static constexpr int kSameReasonAs = 3;

239 static constexpr int kFirstFreePropagationId = 4;

242// The solver trail stores the assignment made by the solver in order.
243// This class is responsible for maintaining the assignment of each variable
244// and the information of each assignment.
245class Trail {
246 public:
247 Trail() {
248 current_info_.trail_index = 0;
249 current_info_.level = 0;
250 }
251
252 void Resize(int num_variables);
253
254 // Registers a propagator. This assigns a unique id to this propagator and
255 // calls SetPropagatorId() on it.
256 void RegisterPropagator(SatPropagator* propagator);
257
258 // Enqueues the assignment that make the given literal true on the trail. This
259 // should only be called on unassigned variables.
260 void Enqueue(Literal true_literal, int propagator_id) {
261 DCHECK(!assignment_.VariableIsAssigned(true_literal.Variable()));
262 trail_[current_info_.trail_index] = true_literal;
263 current_info_.type = propagator_id;
264 info_[true_literal.Variable()] = current_info_;
265 assignment_.AssignFromTrueLiteral(true_literal);
266 ++current_info_.trail_index;
267 }
268
269 // Specific Enqueue() version for the search decision.
270 void EnqueueSearchDecision(Literal true_literal) {
271 Enqueue(true_literal, AssignmentType::kSearchDecision);
272 }
273
274 // Specific Enqueue() version for a fixed variable.
275 void EnqueueWithUnitReason(Literal true_literal) {
276 Enqueue(true_literal, AssignmentType::kUnitReason);
277 }
278
279 // Some constraints propagate a lot of literals at once. In these cases, it is
280 // more efficient to have all the propagated literals except the first one
281 // referring to the reason of the first of them.
282 void EnqueueWithSameReasonAs(Literal true_literal,
283 BooleanVariable reference_var) {
284 reference_var_with_same_reason_as_[true_literal.Variable()] = reference_var;
285 Enqueue(true_literal, AssignmentType::kSameReasonAs);
286 }
287
288 // Enqueues the given literal using the current content of
289 // GetEmptyVectorToStoreReason() as the reason. This API is a bit more
290 // leanient and does not require the literal to be unassigned. If it is
291 // already assigned to false, then MutableConflict() will be set appropriately
292 // and this will return false otherwise this will enqueue the literal and
293 // returns true.
294 ABSL_MUST_USE_RESULT bool EnqueueWithStoredReason(Literal true_literal) {
295 if (assignment_.LiteralIsTrue(true_literal)) return true;
296 if (assignment_.LiteralIsFalse(true_literal)) {
297 *MutableConflict() = reasons_repository_[Index()];
298 MutableConflict()->push_back(true_literal);
299 return false;
300 }
301
302 Enqueue(true_literal, AssignmentType::kCachedReason);
303 const BooleanVariable var = true_literal.Variable();
304 reasons_[var] = reasons_repository_[info_[var].trail_index];
305 old_type_[var] = info_[var].type;
306 info_[var].type = AssignmentType::kCachedReason;
307 return true;
308 }
309
310 // Returns the reason why this variable was assigned.
311 //
312 // Note that this shouldn't be called on a variable at level zero, because we
313 // don't cleanup the reason data for these variables but the underlying
314 // clauses may have been deleted.
315 absl::Span<const Literal> Reason(BooleanVariable var) const;
316
317 // Returns the "type" of an assignment (see AssignmentType). Note that this
318 // function never returns kSameReasonAs or kCachedReason, it instead returns
319 // the initial type that caused this assignment. As such, it is different
320 // from Info(var).type and the latter should not be used outside this class.
321 int AssignmentType(BooleanVariable var) const;
322
323 // If a variable was propagated with EnqueueWithSameReasonAs(), returns its
324 // reference variable. Otherwise return the given variable.
325 BooleanVariable ReferenceVarWithSameReason(BooleanVariable var) const;
326
327 // This can be used to get a location at which the reason for the literal
328 // at trail_index on the trail can be stored. This clears the vector before
329 // returning it.
330 std::vector<Literal>* GetEmptyVectorToStoreReason(int trail_index) const {
331 if (trail_index >= reasons_repository_.size()) {
332 reasons_repository_.resize(trail_index + 1);
333 }
334 reasons_repository_[trail_index].clear();
335 return &reasons_repository_[trail_index];
336 }
337
338 // Shortcut for GetEmptyVectorToStoreReason(Index()).
339 std::vector<Literal>* GetEmptyVectorToStoreReason() const {
340 return GetEmptyVectorToStoreReason(Index());
341 }
342
343 // Explicitly overwrite the reason so that the given propagator will be
344 // asked for it. This is currently only used by the BinaryImplicationGraph.
345 void ChangeReason(int trail_index, int propagator_id) {
346 const BooleanVariable var = trail_[trail_index].Variable();
347 info_[var].type = propagator_id;
348 old_type_[var] = propagator_id;
349 }
350
351 // Reverts the trail and underlying assignment to the given target trail
352 // index. Note that we do not touch the assignment info.
353 void Untrail(int target_trail_index) {
354 const int index = Index();
355 num_untrailed_enqueues_ += index - target_trail_index;
356 for (int i = target_trail_index; i < index; ++i) {
357 assignment_.UnassignLiteral(trail_[i]);
358 }
359 current_info_.trail_index = target_trail_index;
360 }
361 void Dequeue() { Untrail(Index() - 1); }
362
363 // Changes the decision level used by the next Enqueue().
364 void SetDecisionLevel(int level) { current_info_.level = level; }
365 int CurrentDecisionLevel() const { return current_info_.level; }
366
367 // Generic interface to set the current failing clause.
368 //
369 // Returns the address of a vector where a client can store the current
370 // conflict. This vector will be returned by the FailingClause() call.
371 std::vector<Literal>* MutableConflict() {
372 failing_sat_clause_ = nullptr;
373 return &conflict_;
374 }
375
376 // Returns the last conflict.
377 absl::Span<const Literal> FailingClause() const { return conflict_; }
378
379 // Specific SatClause interface so we can update the conflict clause activity.
380 // Note that MutableConflict() automatically sets this to nullptr, so we can
381 // know whether or not the last conflict was caused by a clause.
382 void SetFailingSatClause(SatClause* clause) { failing_sat_clause_ = clause; }
383 SatClause* FailingSatClause() const { return failing_sat_clause_; }
384
385 // Getters.
386 int NumVariables() const { return trail_.size(); }
387 int64_t NumberOfEnqueues() const { return num_untrailed_enqueues_ + Index(); }
388 int Index() const { return current_info_.trail_index; }
389 // This accessor can return trail_.end(). operator[] cannot. This allows
390 // normal std:vector operations, such as assign(begin, end).
391 const std::vector<Literal>::const_iterator IteratorAt(int index) const {
392 return trail_.begin() + index;
393 }
394 const Literal& operator[](int index) const { return trail_[index]; }
395 const VariablesAssignment& Assignment() const { return assignment_; }
396 const AssignmentInfo& Info(BooleanVariable var) const {
397 DCHECK_GE(var, 0);
398 DCHECK_LT(var, info_.size());
399 return info_[var];
400 }
401
402 // Print the current literals on the trail.
403 std::string DebugString() {
404 std::string result;
405 for (int i = 0; i < current_info_.trail_index; ++i) {
406 if (!result.empty()) result += " ";
407 result += trail_[i].DebugString();
408 }
409 return result;
410 }
411
412 private:
413 int64_t num_untrailed_enqueues_ = 0;
414 AssignmentInfo current_info_;
415 VariablesAssignment assignment_;
416 std::vector<Literal> trail_;
417 std::vector<Literal> conflict_;
418 absl::StrongVector<BooleanVariable, AssignmentInfo> info_;
419 SatClause* failing_sat_clause_;
420
421 // Data used by EnqueueWithSameReasonAs().
422 absl::StrongVector<BooleanVariable, BooleanVariable>
423 reference_var_with_same_reason_as_;
424
425 // Reason cache. Mutable since we want the API to be the same whether the
426 // reason are cached or not.
427 //
428 // When a reason is computed for the first time, we change the type of the
429 // variable assignment to kCachedReason so that we know that if it is needed
430 // again the reason can just be retrieved by a direct access to reasons_. The
431 // old type is saved in old_type_ and can be retrieved by
432 // AssignmentType().
433 //
434 // Note(user): Changing the type is not "clean" but it is efficient. The idea
435 // is that it is important to do as little as possible when pushing/popping
436 // literals on the trail. Computing the reason happens a lot less often, so it
437 // is okay to do slightly more work then. Note also, that we don't need to
438 // do anything on "untrail", the kCachedReason type will be overwritten when
439 // the same variable is assigned again.
440 //
441 // TODO(user): An alternative would be to change the sign of the type. This
442 // would remove the need for a separate old_type_ vector, but it requires
443 // more bits for the type filed in AssignmentInfo.
444 //
445 // Note that we use a deque for the reason repository so that if we add
446 // variables, the memory address of the vectors (kept in reasons_) are still
447 // valid.
448 mutable std::deque<std::vector<Literal>> reasons_repository_;
449 mutable absl::StrongVector<BooleanVariable, absl::Span<const Literal>>
450 reasons_;
451 mutable absl::StrongVector<BooleanVariable, int> old_type_;
452
453 // This is used by RegisterPropagator() and Reason().
454 std::vector<SatPropagator*> propagators_;
455
456 DISALLOW_COPY_AND_ASSIGN(Trail);
457};

459// Base class for all the SAT constraints.
460class SatPropagator {
461 public:
462 explicit SatPropagator(const std::string& name)
463 : name_(name), propagator_id_(-1), propagation_trail_index_(0) {}
464 virtual ~SatPropagator() {}
465
466 // Sets/Gets this propagator unique id.
467 void SetPropagatorId(int id) { propagator_id_ = id; }
468 int PropagatorId() const { return propagator_id_; }
469
470 // Inspects the trail from propagation_trail_index_ until at least one literal
471 // is propagated. Returns false iff a conflict is detected (in which case
472 // trail->SetFailingClause() must be called).
473 //
474 // This must update propagation_trail_index_ so that all the literals before
475 // it have been propagated. In particular, if nothing was propagated, then
476 // PropagationIsDone() must return true.
477 virtual bool Propagate(Trail* trail) = 0;
478
479 // Reverts the state so that all the literals with a trail index greater or
480 // equal to the given one are not processed for propagation. Note that the
481 // trail current decision level is already reverted before this is called.
482 //
483 // TODO(user): Currently this is called at each Backtrack(), but we could
484 // bundle the calls in case multiple conflict one after the other are detected
485 // even before the Propagate() call of a SatPropagator is called.
486 //
487 // TODO(user): It is not yet 100% the case, but this can be guaranteed to be
488 // called with a trail index that will always be the start of a new decision
489 // level.
490 virtual void Untrail(const Trail& trail, int trail_index) {
491 propagation_trail_index_ = std::min(propagation_trail_index_, trail_index);
492 }
493
494 // Explains why the literal at given trail_index was propagated by returning a
495 // reason for this propagation. This will only be called for literals that are
496 // on the trail and were propagated by this class.
497 //
498 // The interpretation is that because all the literals of a reason were
499 // assigned to false, we could deduce the assignment of the given variable.
500 //
501 // The returned Span has to be valid until the literal is untrailed. A client
502 // can use trail_.GetEmptyVectorToStoreReason() if it doesn't have a memory
503 // location that already contains the reason.
504 virtual absl::Span<const Literal> Reason(const Trail& trail,
505 int trail_index) const {
506 LOG(FATAL) << "Not implemented.";
507 return {};
508 }
509
510 // Returns true if all the preconditions for Propagate() are satisfied.
511 // This is just meant to be used in a DCHECK.
512 bool PropagatePreconditionsAreSatisfied(const Trail& trail) const;
513
514 // Returns true iff all the trail was inspected by this propagator.
515 bool PropagationIsDone(const Trail& trail) const {
516 return propagation_trail_index_ == trail.Index();
517 }
518
519 // Small optimization: If a propagator does not contain any "constraints"
520 // there is no point calling propagate on it. Before each propagation, the
521 // solver will checks for emptiness, and construct an optimized list of
522 // propagator before looping many time over the list.
523 virtual bool IsEmpty() const { return false; }
524
525 protected:
526 const std::string name_;
527 int propagator_id_;
528 int propagation_trail_index_;
529
530 private:
531 DISALLOW_COPY_AND_ASSIGN(SatPropagator);
532};

538inline bool SatPropagator::PropagatePreconditionsAreSatisfied(
539 const Trail& trail) const {
540 if (propagation_trail_index_ > trail.Index()) {
541 LOG(INFO) << "Issue in '" << name_ << ":"
542 << " propagation_trail_index_=" << propagation_trail_index_
543 << " trail_.Index()=" << trail.Index();
544 return false;
545 }
546 if (propagation_trail_index_ < trail.Index() &&
547 trail.Info(trail[propagation_trail_index_].Variable()).level !=
548 trail.CurrentDecisionLevel()) {
549 LOG(INFO) << "Issue in '" << name_ << "':"
550 << " propagation_trail_index_=" << propagation_trail_index_
551 << " trail_.Index()=" << trail.Index()
552 << " level_at_propagation_index="
553 << trail.Info(trail[propagation_trail_index_].Variable()).level
554 << " current_decision_level=" << trail.CurrentDecisionLevel();
555 return false;
556 }
557 return true;
558}

572inline void Trail::RegisterPropagator(SatPropagator* propagator) {
573 if (propagators_.empty()) {
574 propagators_.resize(AssignmentType::kFirstFreePropagationId);
575 }
576 CHECK_LT(propagators_.size(), 16);
577 propagator->SetPropagatorId(propagators_.size());
578 propagators_.push_back(propagator);

581inline BooleanVariable Trail::ReferenceVarWithSameReason(
582 BooleanVariable var) const {
583 DCHECK(Assignment().VariableIsAssigned(var));
584 // Note that we don't use AssignmentType() here.
585 if (info_[var].type == AssignmentType::kSameReasonAs) {
586 var = reference_var_with_same_reason_as_[var];
587 DCHECK(Assignment().VariableIsAssigned(var));
588 DCHECK_NE(info_[var].type, AssignmentType::kSameReasonAs);
589 }
590 return var;
591}
592
593inline int Trail::AssignmentType(BooleanVariable var) const {
594 if (info_[var].type == AssignmentType::kSameReasonAs) {
595 var = reference_var_with_same_reason_as_[var];
596 DCHECK_NE(info_[var].type, AssignmentType::kSameReasonAs);
597 }
598 const int type = info_[var].type;
599 return type != AssignmentType::kCachedReason ? type : old_type_[var];
600}
601
602inline absl::Span<const Literal> Trail::Reason(BooleanVariable var) const {
603 // Special case for AssignmentType::kSameReasonAs to avoid a recursive call.
604 var = ReferenceVarWithSameReason(var);
605
606 // Fast-track for cached reason.
607 if (info_[var].type == AssignmentType::kCachedReason) return reasons_[var];
608
609 const AssignmentInfo& info = info_[var];
610 if (info.type == AssignmentType::kUnitReason ||
611 info.type == AssignmentType::kSearchDecision) {
612 reasons_[var] = {};
613 } else {
614 DCHECK_LT(info.type, propagators_.size());
615 DCHECK(propagators_[info.type] != nullptr) << info.type;
616 reasons_[var] = propagators_[info.type]->Reason(*this, info.trail_index);
617 }
618 old_type_[var] = info.type;
619 info_[var].type = AssignmentType::kCachedReason;
620 return reasons_[var];
621}
