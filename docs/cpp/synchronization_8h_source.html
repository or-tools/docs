51// Thread-safe. Keeps a set of n unique best solution found so far.
52//
53// TODO(user): Maybe add some criteria to only keep solution with an objective
54// really close to the best solution.
55template <typename ValueType>
56class SharedSolutionRepository {
57 public:
58 explicit SharedSolutionRepository(int num_solutions_to_keep,
59 absl::string_view name = "")
60 : name_(name), num_solutions_to_keep_(num_solutions_to_keep) {}
61
62 // The solution format used by this class.
63 struct Solution {
64 // Solution with lower "rank" will be preferred
65 //
66 // TODO(user): Some LNS code assume that for the SharedSolutionRepository
67 // this rank is actually the unscaled internal minimization objective.
68 // Remove this assumptions by simply recomputing this value since it is not
69 // too costly to do so.
70 int64_t rank = 0;
71
72 std::vector<ValueType> variable_values;
73
74 std::string info;
75
76 // Number of time this was returned by GetRandomBiasedSolution(). We use
77 // this information during the selection process.
78 //
79 // Should be private: only SharedSolutionRepository should modify this.
80 mutable int num_selected = 0;
81
82 bool operator==(const Solution& other) const {
83 return rank == other.rank && variable_values == other.variable_values;
84 }
85 bool operator<(const Solution& other) const {
86 if (rank != other.rank) {
87 return rank < other.rank;
88 }
89 return variable_values < other.variable_values;
90 }
91 };
92
93 // Returns the number of current solution in the pool. This will never
94 // decrease.
95 int NumSolutions() const;
96
97 // Returns the solution #i where i must be smaller than NumSolutions().
98 Solution GetSolution(int index) const;
99
100 // Returns the variable value of variable 'var_index' from solution
101 // 'solution_index' where solution_index must be smaller than NumSolutions()
102 // and 'var_index' must be smaller than number of variables.
103 ValueType GetVariableValueInSolution(int var_index, int solution_index) const;
104
105 // Returns a random solution biased towards good solutions.
106 Solution GetRandomBiasedSolution(absl::BitGenRef random) const;
107
108 // Add a new solution. Note that it will not be added to the pool of solution
109 // right away. One must call Synchronize for this to happen.
110 //
111 // Works in O(num_solutions_to_keep_).
112 void Add(const Solution& solution);
113
114 // Updates the current pool of solution with the one recently added. Note that
115 // we use a stable ordering of solutions, so the final pool will be
116 // independent on the order of the calls to AddSolution() provided that the
117 // set of added solutions is the same.
118 //
119 // Works in O(num_solutions_to_keep_).
120 void Synchronize();
121
122 std::vector<std::string> TableLineStats() const {
123 absl::MutexLock mutex_lock(&mutex_);
124 return {FormatName(name_), FormatCounter(num_added_),
125 FormatCounter(num_queried_), FormatCounter(num_ignored_),
126 FormatCounter(num_synchronization_)};
127 }
128
129 protected:
130 // Helper method for adding the solutions once the mutex is acquired.
131 void AddInternal(const Solution& solution)
132 ABSL_EXCLUSIVE_LOCKS_REQUIRED(mutex_);
133
134 const std::string name_;
135 const int num_solutions_to_keep_;
136
137 mutable absl::Mutex mutex_;
138 int64_t num_added_ ABSL_GUARDED_BY(mutex_) = 0;
139 int64_t num_ignored_ ABSL_GUARDED_BY(mutex_) = 0;
140 mutable int64_t num_queried_ ABSL_GUARDED_BY(mutex_) = 0;
141 int64_t num_synchronization_ ABSL_GUARDED_BY(mutex_) = 0;
142
143 // Our two solutions pools, the current one and the new one that will be
144 // merged into the current one on each Synchronize() calls.
145 mutable std::vector<int> tmp_indices_ ABSL_GUARDED_BY(mutex_);
146 std::vector<Solution> solutions_ ABSL_GUARDED_BY(mutex_);
147 std::vector<Solution> new_solutions_ ABSL_GUARDED_BY(mutex_);
148};

620SharedSolutionRepository<ValueType>::GetRandomBiasedSolution(
621 absl::BitGenRef random) const {
622 absl::MutexLock mutex_lock(&mutex_);
623 ++num_queried_;
624 const int64_t best_rank = solutions_[0].rank;
625
626 // As long as we have solution with the best objective that haven't been
627 // explored too much, we select one uniformly. Otherwise, we select a solution
628 // from the pool uniformly.
629 //
630 // Note(user): Because of the increase of num_selected, this is dependent on
631 // the order of call. It should be fine for "determinism" because we do
632 // generate the task of a batch always in the same order.
633 const int kExplorationThreshold = 100;
634
635 // Select all the best solution with a low enough selection count.
636 tmp_indices_.clear();
637 for (int i = 0; i < solutions_.size(); ++i) {
638 const auto& solution = solutions_[i];
639 if (solution.rank == best_rank &&
640 solution.num_selected <= kExplorationThreshold) {
641 tmp_indices_.push_back(i);
642 }
643 }
644
645 int index = 0;
646 if (tmp_indices_.empty()) {
647 index = absl::Uniform<int>(random, 0, solutions_.size());
648 } else {
649 index = tmp_indices_[absl::Uniform<int>(random, 0, tmp_indices_.size())];
650 }
651 solutions_[index].num_selected++;
652 return solutions_[index];
653}

657 if (num_solutions_to_keep_ <= 0) return;
658 absl::MutexLock mutex_lock(&mutex_);

662template <typename ValueType>
663void SharedSolutionRepository<ValueType>::AddInternal(
664 const Solution& solution) {
665 int worse_solution_index = 0;
666 for (int i = 0; i < new_solutions_.size(); ++i) {
667 // Do not add identical solution.
668 if (new_solutions_[i] == solution) return;
669 if (new_solutions_[worse_solution_index] < new_solutions_[i]) {
670 worse_solution_index = i;
671 }
672 }
673 if (new_solutions_.size() < num_solutions_to_keep_) {
674 ++num_added_;
675 new_solutions_.push_back(solution);
676 } else if (solution < new_solutions_[worse_solution_index]) {
677 ++num_added_;
678 new_solutions_[worse_solution_index] = solution;
679 } else {
680 ++num_ignored_;
681 }
682}

684template <typename ValueType>
685void SharedSolutionRepository<ValueType>::Synchronize() {
686 absl::MutexLock mutex_lock(&mutex_);
687 if (new_solutions_.empty()) return;
688
689 solutions_.insert(solutions_.end(), new_solutions_.begin(),
690 new_solutions_.end());
691 new_solutions_.clear();
692
693 // We use a stable sort to keep the num_selected count for the already
694 // existing solutions.
695 //
696 // TODO(user): Introduce a notion of orthogonality to diversify the pool?
697 gtl::STLStableSortAndRemoveDuplicates(&solutions_);
698 if (solutions_.size() > num_solutions_to_keep_) {
699 solutions_.resize(num_solutions_to_keep_);
700 }
701
702 if (!solutions_.empty()) {
703 VLOG(2) << "Solution pool update:"
704 << " num_solutions=" << solutions_.size()
705 << " min_rank=" << solutions_[0].rank
706 << " max_rank=" << solutions_.back().rank;
707 }
708
709 num_synchronization_++;
710}
