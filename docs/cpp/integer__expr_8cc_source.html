45template <bool use_int128>
46LinearConstraintPropagator<use_int128>::LinearConstraintPropagator(
47 const std::vector<Literal>& enforcement_literals,
48 const std::vector<IntegerVariable>& vars,
49 const std::vector<IntegerValue>& coeffs, IntegerValue upper, Model* model)
50 : upper_bound_(upper),
51 shared_(
52 model->GetOrCreate<LinearConstraintPropagator<use_int128>::Shared>()),
53 size_(vars.size()),
54 vars_(new IntegerVariable[size_]),
55 coeffs_(new IntegerValue[size_]),
56 max_variations_(new IntegerValue[size_]) {
57 // TODO(user): deal with this corner case.
58 CHECK(!vars.empty());
59
60 // Copy data.
61 memcpy(vars_.get(), vars.data(), size_ * sizeof(IntegerVariable));
62 memcpy(coeffs_.get(), coeffs.data(), size_ * sizeof(IntegerValue));
63
64 // Handle negative coefficients.
65 for (int i = 0; i < size_; ++i) {
66 if (coeffs_[i] < 0) {
67 vars_[i] = NegationOf(vars_[i]);
68 coeffs_[i] = -coeffs_[i];
69 }
70 }
71
72 // Literal reason will only be used with the negation of enforcement_literals.
73 // It will stay constant. We also do not store enforcement_literals, but
74 // retrieve them from there.
75 literal_reason_.reserve(enforcement_literals.size());
76 for (const Literal literal : enforcement_literals) {
77 literal_reason_.push_back(literal.Negated());
78 }
79
80 // Initialize the reversible numbers.
81 rev_num_fixed_vars_ = 0;
82 rev_lb_fixed_vars_ = IntegerValue(0);
83}

85template <bool use_int128>
86void LinearConstraintPropagator<use_int128>::FillIntegerReason() {
87 shared_->integer_reason.clear();
88 shared_->reason_coeffs.clear();
89 for (int i = 0; i < size_; ++i) {
90 const IntegerVariable var = vars_[i];
91 if (!shared_->integer_trail->VariableLowerBoundIsFromLevelZero(var)) {
92 shared_->integer_reason.push_back(
93 shared_->integer_trail->LowerBoundAsLiteral(var));
94 shared_->reason_coeffs.push_back(coeffs_[i]);
95 }
96 }

109// NOTE(user): This is only used with int128, so we code only a single version.
110template <bool use_int128>
111std::pair<IntegerValue, IntegerValue>
112LinearConstraintPropagator<use_int128>::ConditionalLb(
113 IntegerLiteral integer_literal, IntegerVariable target_var) const {
114 // The code below is wrong if integer_literal and target_var are the same.
115 // In this case we return the trivial bounds.
116 if (PositiveVariable(integer_literal.var) == PositiveVariable(target_var)) {
117 if (integer_literal.var == target_var) {
118 return {kMinIntegerValue, integer_literal.bound};
119 } else {
120 return {integer_literal.Negated().bound, kMinIntegerValue};
121 }
122 }
123
124 // Recall that all our coefficient are positive.
125 bool literal_var_present = false;
126 bool literal_var_present_positively = false;
127 IntegerValue var_coeff;
128
129 bool target_var_present_negatively = false;
130 absl::int128 target_coeff;
131
132 // Warning: It is important to do the computation like the propagation is
133 // doing it to be sure we don't have overflow, since this is what we check
134 // when creating constraints.
135 absl::int128 lb_128 = 0;
136 for (int i = 0; i < size_; ++i) {
137 const IntegerVariable var = vars_[i];
138 const IntegerValue coeff = coeffs_[i];
139 if (var == NegationOf(target_var)) {
140 target_coeff = absl::int128(coeff.value());
141 target_var_present_negatively = true;
142 }
143
144 const IntegerValue lb = shared_->integer_trail->LowerBound(var);
145 lb_128 += absl::int128(coeff.value()) * absl::int128(lb.value());
146 if (PositiveVariable(var) == PositiveVariable(integer_literal.var)) {
147 var_coeff = coeff;
148 literal_var_present = true;
149 literal_var_present_positively = (var == integer_literal.var);
150 }
151 }
152
153 if (!literal_var_present || !target_var_present_negatively) {
154 return {kMinIntegerValue, kMinIntegerValue};
155 }
156
157 // The upper bound on NegationOf(target_var) are lb(-target) + slack / coeff.
158 // So the lower bound on target_var is ub - slack / coeff.
159 const absl::int128 slack128 = absl::int128(upper_bound_.value()) - lb_128;
160 const IntegerValue target_lb = shared_->integer_trail->LowerBound(target_var);
161 const IntegerValue target_ub = shared_->integer_trail->UpperBound(target_var);
162 if (slack128 <= 0) {
163 // TODO(user): If there is a conflict (negative slack) we can be more
164 // precise.
165 return {target_ub, target_ub};
166 }
167
168 const IntegerValue target_diff = target_ub - target_lb;
169 const IntegerValue delta = CappedCast(slack128 / target_coeff, target_diff);
170
171 // A literal means var >= bound.
172 if (literal_var_present_positively) {
173 // We have var_coeff * var in the expression, the literal is var >= bound.
174 // When it is false, it is not relevant as implied_lb used var >= lb.
175 // When it is true, the diff is bound - lb.
176 const IntegerValue diff =
177 std::max(IntegerValue(0),
178 integer_literal.bound -
179 shared_->integer_trail->LowerBound(integer_literal.var));
180 const absl::int128 tighter_slack =
181 std::max(absl::int128(0), slack128 - absl::int128(var_coeff.value()) *
182 absl::int128(diff.value()));
183 const IntegerValue tighter_delta =
184 CappedCast(tighter_slack / target_coeff, target_diff);
185 return {target_ub - delta, target_ub - tighter_delta};
186 } else {
187 // We have var_coeff * -var in the expression, the literal is var >= bound.
188 // When it is true, it is not relevant as implied_lb used -var >= -ub.
189 // And when it is false it means var < bound, so -var >= -bound + 1
190 const IntegerValue diff =
191 std::max(IntegerValue(0),
192 shared_->integer_trail->UpperBound(integer_literal.var) -
193 integer_literal.bound + 1);
194 const absl::int128 tighter_slack =
195 std::max(absl::int128(0), slack128 - absl::int128(var_coeff.value()) *
196 absl::int128(diff.value()));
197 const IntegerValue tighter_delta =
198 CappedCast(tighter_slack / target_coeff, target_diff);
199 return {target_ub - tighter_delta, target_ub - delta};
200 }

203template <bool use_int128>
204bool LinearConstraintPropagator<use_int128>::Propagate() {
205 // Reified case: If any of the enforcement_literals are false, we ignore the
206 // constraint.
207 int num_unassigned_enforcement_literal = 0;
208 LiteralIndex unique_unnasigned_literal = kNoLiteralIndex;
209 for (const Literal negated_enforcement : literal_reason_) {
210 const Literal literal = negated_enforcement.Negated();
211 if (shared_->assignment.LiteralIsFalse(literal)) return true;
212 if (!shared_->assignment.LiteralIsTrue(literal)) {
213 ++num_unassigned_enforcement_literal;
214 unique_unnasigned_literal = literal.Index();
215 }
216 }
217
218 // Unfortunately, we can't propagate anything if we have more than one
219 // unassigned enforcement literal.
220 if (num_unassigned_enforcement_literal > 1) return true;
221
222 int num_fixed_vars = rev_num_fixed_vars_;
223 IntegerValue lb_fixed_vars = rev_lb_fixed_vars_;
224
225 // Compute the new lower bound and update the reversible structures.
226 absl::int128 lb_128 = 0;
227 IntegerValue lb_unfixed_vars = IntegerValue(0);
228 for (int i = num_fixed_vars; i < size_; ++i) {
229 const IntegerVariable var = vars_[i];
230 const IntegerValue coeff = coeffs_[i];
231 const IntegerValue lb = shared_->integer_trail->LowerBound(var);
232 const IntegerValue ub = shared_->integer_trail->UpperBound(var);
233 if (use_int128) {
234 lb_128 += absl::int128(lb.value()) * absl::int128(coeff.value());
235 continue;
236 }
237
238 if (lb != ub) {
239 max_variations_[i] = (ub - lb) * coeff;
240 lb_unfixed_vars += lb * coeff;
241 } else {
242 // Update the set of fixed variables.
243 std::swap(vars_[i], vars_[num_fixed_vars]);
244 std::swap(coeffs_[i], coeffs_[num_fixed_vars]);
245 std::swap(max_variations_[i], max_variations_[num_fixed_vars]);
246 num_fixed_vars++;
247 lb_fixed_vars += lb * coeff;
248 }
249 }
250 shared_->time_limit->AdvanceDeterministicTime(
251 static_cast<double>(size_ - num_fixed_vars) * 5e-9);
252
253 // Save the current sum of fixed variables.
254 if (is_registered_ && num_fixed_vars != rev_num_fixed_vars_) {
255 CHECK(!use_int128);
256 shared_->rev_integer_value_repository->SaveState(&rev_lb_fixed_vars_);
257 shared_->rev_int_repository->SaveState(&rev_num_fixed_vars_);
258 rev_num_fixed_vars_ = num_fixed_vars;
259 rev_lb_fixed_vars_ = lb_fixed_vars;
260 }
261
262 // If use_int128 is true, the slack or propagation slack can be larger than
263 // this. To detect overflow with capped arithmetic, it is important the slack
264 // used in our algo never exceed this value.
265 const absl::int128 max_slack = std::numeric_limits<int64_t>::max() - 1;
266
267 // Conflict?
268 IntegerValue slack;
269 absl::int128 slack128;
270 if (use_int128) {
271 slack128 = absl::int128(upper_bound_.value()) - lb_128;
272 if (slack128 < 0) {
273 // It is fine if we don't relax as much as possible.
274 // Note that RelaxLinearReason() is overflow safe.
275 slack = static_cast<int64_t>(std::max(-max_slack, slack128));
276 }
277 } else {
278 slack = upper_bound_ - (lb_fixed_vars + lb_unfixed_vars);
279 }
280 if (slack < 0) {
281 FillIntegerReason();
282 shared_->integer_trail->RelaxLinearReason(
283 -slack - 1, shared_->reason_coeffs, &shared_->integer_reason);
284
285 if (num_unassigned_enforcement_literal == 1) {
286 // Propagate the only non-true literal to false.
287 const Literal to_propagate = Literal(unique_unnasigned_literal).Negated();
288 std::vector<Literal> tmp = literal_reason_;
289 tmp.erase(std::find(tmp.begin(), tmp.end(), to_propagate));
290 shared_->integer_trail->EnqueueLiteral(to_propagate, tmp,
291 shared_->integer_reason);
292 return true;
293 }
294 return shared_->integer_trail->ReportConflict(literal_reason_,
295 shared_->integer_reason);
296 }
297
298 // We can only propagate more if all the enforcement literals are true.
299 if (num_unassigned_enforcement_literal > 0) return true;
300
301 // The lower bound of all the variables except one can be used to update the
302 // upper bound of the last one.
303 for (int i = num_fixed_vars; i < size_; ++i) {
304 if (!use_int128 && max_variations_[i] <= slack) continue;
305
306 // TODO(user): If the new ub fall into an hole of the variable, we can
307 // actually relax the reason more by computing a better slack.
308 const IntegerVariable var = vars_[i];
309 const IntegerValue coeff = coeffs_[i];
310 const IntegerValue lb = shared_->integer_trail->LowerBound(var);
311
312 IntegerValue new_ub;
313 IntegerValue propagation_slack;
314 if (use_int128) {
315 const absl::int128 coeff128 = absl::int128(coeff.value());
316 const absl::int128 div128 = slack128 / coeff128;
317 const IntegerValue ub = shared_->integer_trail->UpperBound(var);
318 if (absl::int128(lb.value()) + div128 >= absl::int128(ub.value())) {
319 continue;
320 }
321 new_ub = lb + IntegerValue(static_cast<int64_t>(div128));
322 propagation_slack = static_cast<int64_t>(
323 std::min(max_slack, (div128 + 1) * coeff128 - slack128 - 1));
324 } else {
325 const IntegerValue div = slack / coeff;
326 new_ub = lb + div;
327 propagation_slack = (div + 1) * coeff - slack - 1;
328 }
329 if (!shared_->integer_trail->Enqueue(
330 IntegerLiteral::LowerOrEqual(var, new_ub),
331 /*lazy_reason=*/[this, propagation_slack](
332 IntegerLiteral i_lit, int trail_index,
333 std::vector<Literal>* literal_reason,
334 std::vector<int>* trail_indices_reason) {
335 *literal_reason = literal_reason_;
336 trail_indices_reason->clear();
337 shared_->reason_coeffs.clear();
338 for (int i = 0; i < size_; ++i) {
339 const IntegerVariable var = vars_[i];
340 if (PositiveVariable(var) == PositiveVariable(i_lit.var)) {
341 continue;
342 }
343 const int index =
344 shared_->integer_trail->FindTrailIndexOfVarBefore(
345 var, trail_index);
346 if (index >= 0) {
347 trail_indices_reason->push_back(index);
348 if (propagation_slack > 0) {
349 shared_->reason_coeffs.push_back(coeffs_[i]);
350 }
351 }
352 }
353 if (propagation_slack > 0) {
354 shared_->integer_trail->RelaxLinearReason(
355 propagation_slack, shared_->reason_coeffs,
356 trail_indices_reason);
357 }
358 })) {
359 // TODO(user): this is never supposed to happen since if we didn't have a
360 // conflict above, we should be able to reduce the upper bound. It might
361 // indicate an issue with our Boolean <-> integer encoding.
362 return false;
363 }
364 }
365
366 return true;
367}

369template <bool use_int128>
370bool LinearConstraintPropagator<use_int128>::PropagateAtLevelZero() {
371 // TODO(user): Deal with enforcements. It is just a bit of code to read the
372 // value of the literals at level zero.
373 if (!literal_reason_.empty()) return true;
374
375 // Compute the new lower bound and update the reversible structures.
376 absl::int128 lb_128 = 0;
377 IntegerValue min_activity = IntegerValue(0);
378 for (int i = 0; i < size_; ++i) {
379 const IntegerVariable var = vars_[i];
380 const IntegerValue coeff = coeffs_[i];
381 const IntegerValue lb = shared_->integer_trail->LevelZeroLowerBound(var);
382 if (use_int128) {
383 lb_128 += absl::int128(lb.value()) * absl::int128(coeff.value());
384 } else {
385 const IntegerValue ub = shared_->integer_trail->LevelZeroUpperBound(var);
386 max_variations_[i] = (ub - lb) * coeff;
387 min_activity += lb * coeff;
388 }
389 }
390 shared_->time_limit->AdvanceDeterministicTime(
391 static_cast<double>(size_ * 1e-9));
392
393 // Conflict?
394 IntegerValue slack;
395 absl::int128 slack128;
396 if (use_int128) {
397 slack128 = absl::int128(upper_bound_.value()) - lb_128;
398 if (slack128 < 0) {
399 return shared_->integer_trail->ReportConflict({}, {});
400 }
401 } else {
402 slack = upper_bound_ - min_activity;
403 if (slack < 0) {
404 return shared_->integer_trail->ReportConflict({}, {});
405 }
406 }
407
408 // The lower bound of all the variables except one can be used to update the
409 // upper bound of the last one.
410 for (int i = 0; i < size_; ++i) {
411 if (!use_int128 && max_variations_[i] <= slack) continue;
412
413 const IntegerVariable var = vars_[i];
414 const IntegerValue coeff = coeffs_[i];
415 const IntegerValue lb = shared_->integer_trail->LevelZeroLowerBound(var);
416
417 IntegerValue new_ub;
418 if (use_int128) {
419 const IntegerValue ub = shared_->integer_trail->LevelZeroUpperBound(var);
420 const absl::int128 div128 = slack128 / absl::int128(coeff.value());
421 if (absl::int128(lb.value()) + div128 >= absl::int128(ub.value())) {
422 continue;
423 }
424 new_ub = lb + IntegerValue(static_cast<int64_t>(div128));
425 } else {
426 const IntegerValue div = slack / coeff;
427 new_ub = lb + div;
428 }
429 if (!shared_->integer_trail->Enqueue(
430 IntegerLiteral::LowerOrEqual(var, new_ub), {}, {})) {
431 return false;
432 }
433 }
434
435 return true;
436}

477// TODO(user): We could go even further than just the GCD, and do more
478// arithmetic to tighten the target bounds. See for instance a problem like
479// ej.mps.gz that we don't solve easily, but has just 3 variables! the goal is
480// to minimize X, given 31013 X - 41014 Y - 51015 Z = -31013 (all >=0, Y and Z
481// bounded with high values). I know some MIP solvers have a basic linear
482// diophantine equation support.
483bool LevelZeroEquality::Propagate() {
484 // TODO(user): Once the GCD is not 1, we could at any level make sure the
485 // objective is of the correct form. For now, this only happen in a few
486 // miplib problem that we close quickly, so I didn't add the extra code yet.
487 if (trail_->CurrentDecisionLevel() != 0) return true;
488
489 int64_t gcd = 0;
490 IntegerValue sum(0);
491 for (int i = 0; i < vars_.size(); ++i) {
492 if (integer_trail_->IsFixed(vars_[i])) {
493 sum += coeffs_[i] * integer_trail_->LowerBound(vars_[i]);
494 continue;
495 }
496 gcd = MathUtil::GCD64(gcd, std::abs(coeffs_[i].value()));
497 if (gcd == 1) break;
498 }
499 if (gcd == 0) return true; // All fixed.
500
501 if (gcd > gcd_) {
502 VLOG(1) << "Objective gcd: " << gcd;
503 }
504 CHECK_GE(gcd, gcd_);
505 gcd_ = IntegerValue(gcd);
506
507 const IntegerValue lb = integer_trail_->LowerBound(target_);
508 const IntegerValue lb_remainder = PositiveRemainder(lb - sum, gcd_);
509 if (lb_remainder != 0) {
510 if (!integer_trail_->Enqueue(
511 IntegerLiteral::GreaterOrEqual(target_, lb + gcd_ - lb_remainder),
512 {}, {}))
513 return false;
514 }
515
516 const IntegerValue ub = integer_trail_->UpperBound(target_);
517 const IntegerValue ub_remainder =
518 PositiveRemainder(ub - sum, IntegerValue(gcd));
519 if (ub_remainder != 0) {
520 if (!integer_trail_->Enqueue(
521 IntegerLiteral::LowerOrEqual(target_, ub - ub_remainder), {}, {}))
522 return false;
523 }
524
525 return true;
526}

533bool MinPropagator::Propagate() {
534 if (vars_.empty()) return true;
535
536 // Count the number of interval that are possible candidate for the min.
537 // Only the intervals for which lb > current_min_ub cannot.
538 const IntegerLiteral min_ub_literal =
539 integer_trail_->UpperBoundAsLiteral(min_var_);
540 const IntegerValue current_min_ub = integer_trail_->UpperBound(min_var_);
541 int num_intervals_that_can_be_min = 0;
542 int last_possible_min_interval = 0;
543
544 IntegerValue min = kMaxIntegerValue;
545 for (int i = 0; i < vars_.size(); ++i) {
546 const IntegerValue lb = integer_trail_->LowerBound(vars_[i]);
547 min = std::min(min, lb);
548 if (lb <= current_min_ub) {
549 ++num_intervals_that_can_be_min;
550 last_possible_min_interval = i;
551 }
552 }
553
554 // Propagation a)
555 if (min > integer_trail_->LowerBound(min_var_)) {
556 integer_reason_.clear();
557 for (const IntegerVariable var : vars_) {
558 integer_reason_.push_back(IntegerLiteral::GreaterOrEqual(var, min));
559 }
560 if (!integer_trail_->Enqueue(IntegerLiteral::GreaterOrEqual(min_var_, min),
561 {}, integer_reason_)) {
562 return false;
563 }
564 }
565
566 // Propagation b)
567 if (num_intervals_that_can_be_min == 1) {
568 const IntegerValue ub_of_only_candidate =
569 integer_trail_->UpperBound(vars_[last_possible_min_interval]);
570 if (current_min_ub < ub_of_only_candidate) {
571 integer_reason_.clear();
572
573 // The reason is that all the other interval start after current_min_ub.
574 // And that min_ub has its current value.
575 integer_reason_.push_back(min_ub_literal);
576 for (const IntegerVariable var : vars_) {
577 if (var == vars_[last_possible_min_interval]) continue;
578 integer_reason_.push_back(
579 IntegerLiteral::GreaterOrEqual(var, current_min_ub + 1));
580 }
581 if (!integer_trail_->Enqueue(
582 IntegerLiteral::LowerOrEqual(vars_[last_possible_min_interval],
583 current_min_ub),
584 {}, integer_reason_)) {
585 return false;
586 }
587 }
588 }
589
590 // Conflict.
591 //
592 // TODO(user): Not sure this code is useful since this will be detected
593 // by the fact that the [lb, ub] of the min is empty. It depends on the
594 // propagation order though, but probably the precedences propagator would
595 // propagate before this one. So change this to a CHECK?
596 if (num_intervals_that_can_be_min == 0) {
597 integer_reason_.clear();
598
599 // Almost the same as propagation b).
600 integer_reason_.push_back(min_ub_literal);
601 for (const IntegerVariable var : vars_) {
602 integer_reason_.push_back(
603 IntegerLiteral::GreaterOrEqual(var, current_min_ub + 1));
604 }
605 return integer_trail_->ReportConflict(integer_reason_);
606 }
607
608 return true;
609}

626bool LinMinPropagator::PropagateLinearUpperBound(
627 const std::vector<IntegerVariable>& vars,
628 const std::vector<IntegerValue>& coeffs, const IntegerValue upper_bound) {
629 IntegerValue sum_lb = IntegerValue(0);
630 const int num_vars = vars.size();
631 max_variations_.resize(num_vars);
632 for (int i = 0; i < num_vars; ++i) {
633 const IntegerVariable var = vars[i];
634 const IntegerValue coeff = coeffs[i];
635 // The coefficients are assumed to be positive for this to work properly.
636 DCHECK_GE(coeff, 0);
637 const IntegerValue lb = integer_trail_->LowerBound(var);
638 const IntegerValue ub = integer_trail_->UpperBound(var);
639 max_variations_[i] = (ub - lb) * coeff;
640 sum_lb += lb * coeff;
641 }
642
643 model_->GetOrCreate<TimeLimit>()->AdvanceDeterministicTime(
644 static_cast<double>(num_vars) * 1e-9);
645
646 const IntegerValue slack = upper_bound - sum_lb;
647 if (slack < 0) {
648 // Conflict.
649 local_reason_.clear();
650 reason_coeffs_.clear();
651 for (int i = 0; i < num_vars; ++i) {
652 const IntegerVariable var = vars[i];
653 if (!integer_trail_->VariableLowerBoundIsFromLevelZero(var)) {
654 local_reason_.push_back(integer_trail_->LowerBoundAsLiteral(var));
655 reason_coeffs_.push_back(coeffs[i]);
656 }
657 }
658 integer_trail_->RelaxLinearReason(-slack - 1, reason_coeffs_,
659 &local_reason_);
660 local_reason_.insert(local_reason_.end(),
661 integer_reason_for_unique_candidate_.begin(),
662 integer_reason_for_unique_candidate_.end());
663 return integer_trail_->ReportConflict({}, local_reason_);
664 }
665
666 // The lower bound of all the variables except one can be used to update the
667 // upper bound of the last one.
668 for (int i = 0; i < num_vars; ++i) {
669 if (max_variations_[i] <= slack) continue;
670
671 const IntegerVariable var = vars[i];
672 const IntegerValue coeff = coeffs[i];
673 const IntegerValue div = slack / coeff;
674 const IntegerValue new_ub = integer_trail_->LowerBound(var) + div;
675
676 const IntegerValue propagation_slack = (div + 1) * coeff - slack - 1;
677 if (!integer_trail_->Enqueue(
678 IntegerLiteral::LowerOrEqual(var, new_ub),
679 /*lazy_reason=*/[this, &vars, &coeffs, propagation_slack](
680 IntegerLiteral i_lit, int trail_index,
681 std::vector<Literal>* literal_reason,
682 std::vector<int>* trail_indices_reason) {
683 literal_reason->clear();
684 trail_indices_reason->clear();
685 std::vector<IntegerValue> reason_coeffs;
686 const int size = vars.size();
687 for (int i = 0; i < size; ++i) {
688 const IntegerVariable var = vars[i];
689 if (PositiveVariable(var) == PositiveVariable(i_lit.var)) {
690 continue;
691 }
692 const int index =
693 integer_trail_->FindTrailIndexOfVarBefore(var, trail_index);
694 if (index >= 0) {
695 trail_indices_reason->push_back(index);
696 if (propagation_slack > 0) {
697 reason_coeffs.push_back(coeffs[i]);
698 }
699 }
700 }
701 if (propagation_slack > 0) {
702 integer_trail_->RelaxLinearReason(
703 propagation_slack, reason_coeffs, trail_indices_reason);
704 }
705 // Now add the old integer_reason that triggered this propagation.
706 for (IntegerLiteral reason_lit :
707 integer_reason_for_unique_candidate_) {
708 const int index = integer_trail_->FindTrailIndexOfVarBefore(
709 reason_lit.var, trail_index);
710 if (index >= 0) {
711 trail_indices_reason->push_back(index);
712 }
713 }
714 })) {
715 return false;
716 }
717 }
718 return true;
719}

721bool LinMinPropagator::Propagate() {
722 if (exprs_.empty()) return true;
723
724 // Count the number of interval that are possible candidate for the min.
725 // Only the intervals for which lb > current_min_ub cannot.
726 const IntegerValue current_min_ub = integer_trail_->UpperBound(min_var_);
727 int num_intervals_that_can_be_min = 0;
728 int last_possible_min_interval = 0;
729
730 expr_lbs_.clear();
731 IntegerValue min_of_linear_expression_lb = kMaxIntegerValue;
732 for (int i = 0; i < exprs_.size(); ++i) {
733 const IntegerValue lb = exprs_[i].Min(*integer_trail_);
734 expr_lbs_.push_back(lb);
735 min_of_linear_expression_lb = std::min(min_of_linear_expression_lb, lb);
736 if (lb <= current_min_ub) {
737 ++num_intervals_that_can_be_min;
738 last_possible_min_interval = i;
739 }
740 }
741
742 // Propagation a) lb(min) >= lb(MIN(exprs)) = MIN(lb(exprs));
743
744 // Conflict will be detected by the fact that the [lb, ub] of the min is
745 // empty. In case of conflict, we just need the reason for pushing UB + 1.
746 if (min_of_linear_expression_lb > current_min_ub) {
747 min_of_linear_expression_lb = current_min_ub + 1;
748 }
749 if (min_of_linear_expression_lb > integer_trail_->LowerBound(min_var_)) {
750 local_reason_.clear();
751 for (int i = 0; i < exprs_.size(); ++i) {
752 const IntegerValue slack = expr_lbs_[i] - min_of_linear_expression_lb;
753 integer_trail_->AppendRelaxedLinearReason(slack, exprs_[i].coeffs,
754 exprs_[i].vars, &local_reason_);
755 }
756 if (!integer_trail_->Enqueue(IntegerLiteral::GreaterOrEqual(
757 min_var_, min_of_linear_expression_lb),
758 {}, local_reason_)) {
759 return false;
760 }
761 }
762
763 // Propagation b) ub(min) >= ub(MIN(exprs)) and we can't propagate anything
764 // here unless there is just one possible expression 'e' that can be the min:
765 // for all u != e, lb(u) > ub(min);
766 // In this case, ub(min) >= ub(e).
767 if (num_intervals_that_can_be_min == 1) {
768 const IntegerValue ub_of_only_candidate =
769 exprs_[last_possible_min_interval].Max(*integer_trail_);
770 if (current_min_ub < ub_of_only_candidate) {
771 // For this propagation, we only need to fill the integer reason once at
772 // the lowest level. At higher levels this reason still remains valid.
773 if (rev_unique_candidate_ == 0) {
774 integer_reason_for_unique_candidate_.clear();
775
776 // The reason is that all the other interval start after current_min_ub.
777 // And that min_ub has its current value.
778 integer_reason_for_unique_candidate_.push_back(
779 integer_trail_->UpperBoundAsLiteral(min_var_));
780 for (int i = 0; i < exprs_.size(); ++i) {
781 if (i == last_possible_min_interval) continue;
782 const IntegerValue slack = expr_lbs_[i] - (current_min_ub + 1);
783 integer_trail_->AppendRelaxedLinearReason(
784 slack, exprs_[i].coeffs, exprs_[i].vars,
785 &integer_reason_for_unique_candidate_);
786 }
787 rev_unique_candidate_ = 1;
788 }
789
790 return PropagateLinearUpperBound(
791 exprs_[last_possible_min_interval].vars,
792 exprs_[last_possible_min_interval].coeffs,
793 current_min_ub - exprs_[last_possible_min_interval].offset);
794 }
795 }
796
797 return true;
798}

817ProductPropagator::ProductPropagator(AffineExpression a, AffineExpression b,
818 AffineExpression p,
819 IntegerTrail* integer_trail)
820 : a_(a), b_(b), p_(p), integer_trail_(integer_trail) {}

822// We want all affine expression to be either non-negative or across zero.
823bool ProductPropagator::CanonicalizeCases() {
824 if (integer_trail_->UpperBound(a_) <= 0) {
825 a_ = a_.Negated();
826 p_ = p_.Negated();
827 }
828 if (integer_trail_->UpperBound(b_) <= 0) {
829 b_ = b_.Negated();
830 p_ = p_.Negated();
831 }
832
833 // If both a and b positive, p must be too.
834 if (integer_trail_->LowerBound(a_) >= 0 &&
835 integer_trail_->LowerBound(b_) >= 0) {
836 return integer_trail_->SafeEnqueue(
837 p_.GreaterOrEqual(0), {a_.GreaterOrEqual(0), b_.GreaterOrEqual(0)});
838 }
839
840 // Otherwise, make sure p is non-negative or across zero.
841 if (integer_trail_->UpperBound(p_) <= 0) {
842 if (integer_trail_->LowerBound(a_) < 0) {
843 DCHECK_GT(integer_trail_->UpperBound(a_), 0);
844 a_ = a_.Negated();
845 p_ = p_.Negated();
846 } else {
847 DCHECK_LT(integer_trail_->LowerBound(b_), 0);
848 DCHECK_GT(integer_trail_->UpperBound(b_), 0);
849 b_ = b_.Negated();
850 p_ = p_.Negated();
851 }
852 }
853
854 return true;
855}
856
857// Note that this propagation is exact, except on the domain of p as this
858// involves more complex arithmetic.
859//
860// TODO(user): We could tighten the bounds on p by removing extreme value that
861// do not contains divisor in the domains of a or b. There is an algo in O(
862// smallest domain size between a or b).
863bool ProductPropagator::PropagateWhenAllNonNegative() {
864 {
865 const IntegerValue max_a = integer_trail_->UpperBound(a_);
866 const IntegerValue max_b = integer_trail_->UpperBound(b_);
867 const IntegerValue new_max = CapProdI(max_a, max_b);
868 if (new_max < integer_trail_->UpperBound(p_)) {
869 if (!integer_trail_->SafeEnqueue(
870 p_.LowerOrEqual(new_max),
871 {integer_trail_->UpperBoundAsLiteral(a_),
872 integer_trail_->UpperBoundAsLiteral(b_), a_.GreaterOrEqual(0),
873 b_.GreaterOrEqual(0)})) {
874 return false;
875 }
876 }
877 }
878
879 {
880 const IntegerValue min_a = integer_trail_->LowerBound(a_);
881 const IntegerValue min_b = integer_trail_->LowerBound(b_);
882 const IntegerValue new_min = CapProdI(min_a, min_b);
883
884 // The conflict test is needed because when new_min is large, we could
885 // have an overflow in p_.GreaterOrEqual(new_min);
886 if (new_min > integer_trail_->UpperBound(p_)) {
887 return integer_trail_->ReportConflict(
888 {integer_trail_->UpperBoundAsLiteral(p_),
889 integer_trail_->LowerBoundAsLiteral(a_),
890 integer_trail_->LowerBoundAsLiteral(b_)});
891 }
892 if (new_min > integer_trail_->LowerBound(p_)) {
893 if (!integer_trail_->SafeEnqueue(
894 p_.GreaterOrEqual(new_min),
895 {integer_trail_->LowerBoundAsLiteral(a_),
896 integer_trail_->LowerBoundAsLiteral(b_)})) {
897 return false;
898 }
899 }
900 }
901
902 for (int i = 0; i < 2; ++i) {
903 const AffineExpression a = i == 0 ? a_ : b_;
904 const AffineExpression b = i == 0 ? b_ : a_;
905 const IntegerValue max_a = integer_trail_->UpperBound(a);
906 const IntegerValue min_b = integer_trail_->LowerBound(b);
907 const IntegerValue min_p = integer_trail_->LowerBound(p_);
908 const IntegerValue max_p = integer_trail_->UpperBound(p_);
909 const IntegerValue prod = CapProdI(max_a, min_b);
910 if (prod > max_p) {
911 if (!integer_trail_->SafeEnqueue(a.LowerOrEqual(FloorRatio(max_p, min_b)),
912 {integer_trail_->LowerBoundAsLiteral(b),
913 integer_trail_->UpperBoundAsLiteral(p_),
914 p_.GreaterOrEqual(0)})) {
915 return false;
916 }
917 } else if (prod < min_p && max_a != 0) {
918 if (!integer_trail_->SafeEnqueue(
919 b.GreaterOrEqual(CeilRatio(min_p, max_a)),
920 {integer_trail_->UpperBoundAsLiteral(a),
921 integer_trail_->LowerBoundAsLiteral(p_), a.GreaterOrEqual(0)})) {
922 return false;
923 }
924 }
925 }
926
927 return true;
928}

930// This assumes p > 0, p = a * X, and X can take any value.
931// We can propagate max of a by computing a bound on the min b when positive.
932// The expression b is just used to detect when there is no solution given the
933// upper bound of b.
934bool ProductPropagator::PropagateMaxOnPositiveProduct(AffineExpression a,
935 AffineExpression b,
936 IntegerValue min_p,
937 IntegerValue max_p) {
938 const IntegerValue max_a = integer_trail_->UpperBound(a);
939 if (max_a <= 0) return true;
940 DCHECK_GT(min_p, 0);
941
942 if (max_a >= min_p) {
943 if (max_p < max_a) {
944 if (!integer_trail_->SafeEnqueue(
945 a.LowerOrEqual(max_p),
946 {p_.LowerOrEqual(max_p), p_.GreaterOrEqual(1)})) {
947 return false;
948 }
949 }
950 return true;
951 }
952
953 const IntegerValue min_pos_b = CeilRatio(min_p, max_a);
954 if (min_pos_b > integer_trail_->UpperBound(b)) {
955 if (!integer_trail_->SafeEnqueue(
956 b.LowerOrEqual(0), {integer_trail_->LowerBoundAsLiteral(p_),
957 integer_trail_->UpperBoundAsLiteral(a),
958 integer_trail_->UpperBoundAsLiteral(b)})) {
959 return false;
960 }
961 return true;
962 }
963
964 const IntegerValue new_max_a = FloorRatio(max_p, min_pos_b);
965 if (new_max_a < integer_trail_->UpperBound(a)) {
966 if (!integer_trail_->SafeEnqueue(
967 a.LowerOrEqual(new_max_a),
968 {integer_trail_->LowerBoundAsLiteral(p_),
969 integer_trail_->UpperBoundAsLiteral(a),
970 integer_trail_->UpperBoundAsLiteral(p_)})) {
971 return false;
972 }
973 }
974 return true;
975}

977bool ProductPropagator::Propagate() {
978 if (!CanonicalizeCases()) return false;
979
980 // In the most common case, we use better reasons even though the code
981 // below would propagate the same.
982 const int64_t min_a = integer_trail_->LowerBound(a_).value();
983 const int64_t min_b = integer_trail_->LowerBound(b_).value();
984 if (min_a >= 0 && min_b >= 0) {
985 // This was done by CanonicalizeCases().
986 DCHECK_GE(integer_trail_->LowerBound(p_), 0);
987 return PropagateWhenAllNonNegative();
988 }
989
990 // Lets propagate on p_ first, the max/min is given by one of: max_a * max_b,
991 // max_a * min_b, min_a * max_b, min_a * min_b. This is true, because any
992 // product x * y, depending on the sign, is dominated by one of these.
993 //
994 // TODO(user): In the reasons, including all 4 bounds is always correct, but
995 // we might be able to relax some of them.
996 const IntegerValue max_a = integer_trail_->UpperBound(a_);
997 const IntegerValue max_b = integer_trail_->UpperBound(b_);
998 const IntegerValue p1 = CapProdI(max_a, max_b);
999 const IntegerValue p2 = CapProdI(max_a, min_b);
1000 const IntegerValue p3 = CapProdI(min_a, max_b);
1001 const IntegerValue p4 = CapProdI(min_a, min_b);
1002 const IntegerValue new_max_p = std::max({p1, p2, p3, p4});
1003 if (new_max_p < integer_trail_->UpperBound(p_)) {
1004 if (!integer_trail_->SafeEnqueue(
1005 p_.LowerOrEqual(new_max_p),
1006 {integer_trail_->LowerBoundAsLiteral(a_),
1007 integer_trail_->LowerBoundAsLiteral(b_),
1008 integer_trail_->UpperBoundAsLiteral(a_),
1009 integer_trail_->UpperBoundAsLiteral(b_)})) {
1010 return false;
1011 }
1012 }
1013 const IntegerValue new_min_p = std::min({p1, p2, p3, p4});
1014 if (new_min_p > integer_trail_->LowerBound(p_)) {
1015 if (!integer_trail_->SafeEnqueue(
1016 p_.GreaterOrEqual(new_min_p),
1017 {integer_trail_->LowerBoundAsLiteral(a_),
1018 integer_trail_->LowerBoundAsLiteral(b_),
1019 integer_trail_->UpperBoundAsLiteral(a_),
1020 integer_trail_->UpperBoundAsLiteral(b_)})) {
1021 return false;
1022 }
1023 }
1024
1025 // Lets propagate on a and b.
1026 const IntegerValue min_p = integer_trail_->LowerBound(p_);
1027 const IntegerValue max_p = integer_trail_->UpperBound(p_);
1028
1029 // We need a bit more propagation to avoid bad cases below.
1030 const bool zero_is_possible = min_p <= 0;
1031 if (!zero_is_possible) {
1032 if (integer_trail_->LowerBound(a_) == 0) {
1033 if (!integer_trail_->SafeEnqueue(
1034 a_.GreaterOrEqual(1),
1035 {p_.GreaterOrEqual(1), a_.GreaterOrEqual(0)})) {
1036 return false;
1037 }
1038 }
1039 if (integer_trail_->LowerBound(b_) == 0) {
1040 if (!integer_trail_->SafeEnqueue(
1041 b_.GreaterOrEqual(1),
1042 {p_.GreaterOrEqual(1), b_.GreaterOrEqual(0)})) {
1043 return false;
1044 }
1045 }
1046 if (integer_trail_->LowerBound(a_) >= 0 &&
1047 integer_trail_->LowerBound(b_) <= 0) {
1048 return integer_trail_->SafeEnqueue(
1049 b_.GreaterOrEqual(1), {a_.GreaterOrEqual(0), p_.GreaterOrEqual(1)});
1050 }
1051 if (integer_trail_->LowerBound(b_) >= 0 &&
1052 integer_trail_->LowerBound(a_) <= 0) {
1053 return integer_trail_->SafeEnqueue(
1054 a_.GreaterOrEqual(1), {b_.GreaterOrEqual(0), p_.GreaterOrEqual(1)});
1055 }
1056 }
1057
1058 for (int i = 0; i < 2; ++i) {
1059 // p = a * b, what is the min/max of a?
1060 const AffineExpression a = i == 0 ? a_ : b_;
1061 const AffineExpression b = i == 0 ? b_ : a_;
1062 const IntegerValue max_b = integer_trail_->UpperBound(b);
1063 const IntegerValue min_b = integer_trail_->LowerBound(b);
1064
1065 // If the domain of b contain zero, we can't propagate anything on a.
1066 // Because of CanonicalizeCases(), we just deal with min_b > 0 here.
1067 if (zero_is_possible && min_b <= 0) continue;
1068
1069 // Here both a and b are across zero, but zero is not possible.
1070 if (min_b < 0 && max_b > 0) {
1071 CHECK_GT(min_p, 0); // Because zero is not possible.
1072
1073 // If a is not across zero, we will deal with this on the next
1074 // Propagate() call.
1075 if (!PropagateMaxOnPositiveProduct(a, b, min_p, max_p)) {
1076 return false;
1077 }
1078 if (!PropagateMaxOnPositiveProduct(a.Negated(), b.Negated(), min_p,
1079 max_p)) {
1080 return false;
1081 }
1082 continue;
1083 }
1084
1085 // This shouldn't happen here.
1086 // If it does, we should reach the fixed point on the next iteration.
1087 if (min_b <= 0) continue;
1088 if (min_p >= 0) {
1089 return integer_trail_->SafeEnqueue(
1090 a.GreaterOrEqual(0), {p_.GreaterOrEqual(0), b.GreaterOrEqual(1)});
1091 }
1092 if (max_p <= 0) {
1093 return integer_trail_->SafeEnqueue(
1094 a.LowerOrEqual(0), {p_.LowerOrEqual(0), b.GreaterOrEqual(1)});
1095 }
1096
1097 // So min_b > 0 and p is across zero: min_p < 0 and max_p > 0.
1098 const IntegerValue new_max_a = FloorRatio(max_p, min_b);
1099 if (new_max_a < integer_trail_->UpperBound(a)) {
1100 if (!integer_trail_->SafeEnqueue(
1101 a.LowerOrEqual(new_max_a),
1102 {integer_trail_->UpperBoundAsLiteral(p_),
1103 integer_trail_->LowerBoundAsLiteral(b)})) {
1104 return false;
1105 }
1106 }
1107 const IntegerValue new_min_a = CeilRatio(min_p, min_b);
1108 if (new_min_a > integer_trail_->LowerBound(a)) {
1109 if (!integer_trail_->SafeEnqueue(
1110 a.GreaterOrEqual(new_min_a),
1111 {integer_trail_->LowerBoundAsLiteral(p_),
1112 integer_trail_->LowerBoundAsLiteral(b)})) {
1113 return false;
1114 }
1115 }
1116 }
1117
1118 return true;
1119}

1322bool DivisionPropagator::PropagatePositiveDomains(AffineExpression num,
1323 AffineExpression denom,
1324 AffineExpression div) {

1326 const IntegerValue max_num = integer_trail_->UpperBound(num);
1327 const IntegerValue min_denom = integer_trail_->LowerBound(denom);
1328 const IntegerValue max_denom = integer_trail_->UpperBound(denom);
1329 const IntegerValue min_div = integer_trail_->LowerBound(div);
1330 const IntegerValue max_div = integer_trail_->UpperBound(div);
1331
1332 const IntegerValue new_min_div = min_num / max_denom;
1333 if (min_div < new_min_div) {
1334 if (!integer_trail_->SafeEnqueue(
1335 div.GreaterOrEqual(new_min_div),
1336 {num.GreaterOrEqual(min_num), denom.LowerOrEqual(max_denom),
1337 denom.GreaterOrEqual(1)})) {
1338 return false;
1339 }
1340 }
1341
1342 // We start from num / denom >= min_div.
1343 // num >= min_div * denom.
1344 // num >= min_div * min_denom.
1345 const IntegerValue new_min_num = CapProdI(min_denom, min_div);
1346 if (min_num < new_min_num) {
1347 if (!integer_trail_->SafeEnqueue(
1348 num.GreaterOrEqual(new_min_num),
1349 {denom.GreaterOrEqual(min_denom), div.GreaterOrEqual(min_div)})) {
1350 return false;
1351 }
1352 }

1354 // We start with num / denom >= min_div.
1355 // So num >= min_div * denom
1356 // If min_div == 0 we can't deduce anything.
1357 // Otherwise, denom <= num / min_div and denom <= max_num / min_div.
1358 if (min_div > 0) {
1359 const IntegerValue new_max_denom = max_num / min_div;
1360 if (max_denom > new_max_denom) {
1361 if (!integer_trail_->SafeEnqueue(
1362 denom.LowerOrEqual(new_max_denom),
1363 {num.LowerOrEqual(max_num), num.GreaterOrEqual(0),
1364 div.GreaterOrEqual(min_div), denom.GreaterOrEqual(1)})) {
1365 return false;
1366 }
1367 }
1368 }
1369
1370 // denom >= CeilRatio(num + 1, max_div + 1)
1371 // >= CeilRatio(min_num + 1, max_div + 1).
1372 const IntegerValue new_min_denom = CeilRatio(min_num + 1, max_div + 1);
1373 if (min_denom < new_min_denom) {
1374 if (!integer_trail_->SafeEnqueue(
1375 denom.GreaterOrEqual(new_min_denom),
1376 {num.GreaterOrEqual(min_num), div.LowerOrEqual(max_div),
1377 div.GreaterOrEqual(0), denom.GreaterOrEqual(1)})) {

1379 }
1380 }
