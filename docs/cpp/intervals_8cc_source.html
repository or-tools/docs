55IntervalVariable IntervalsRepository::CreateInterval(AffineExpression start,
56 AffineExpression end,
57 AffineExpression size,
58 LiteralIndex is_present,
59 bool add_linear_relation) {
60 // Create the interval.
61 const IntervalVariable i(starts_.size());
62 starts_.push_back(start);
63 ends_.push_back(end);
64 sizes_.push_back(size);
65 is_present_.push_back(is_present);
66
67 std::vector<Literal> enforcement_literals;
68 if (is_present != kNoLiteralIndex) {
69 enforcement_literals.push_back(Literal(is_present));
70 }
71
72 if (add_linear_relation) {
73 LinearConstraintBuilder builder(model_, IntegerValue(0), IntegerValue(0));
74 builder.AddTerm(Start(i), IntegerValue(1));
75 builder.AddTerm(Size(i), IntegerValue(1));
76 builder.AddTerm(End(i), IntegerValue(-1));
77 LoadConditionalLinearConstraint(enforcement_literals, builder.Build(),
78 model_);
79 }
80
81 return i;
82}

84void IntervalsRepository::CreateDisjunctivePrecedenceLiteral(
85 IntervalVariable a, IntervalVariable b) {
86 if (disjunctive_precedences_.contains({a, b})) return;
87
88 std::vector<Literal> enforcement_literals;
89 if (IsOptional(a)) enforcement_literals.push_back(PresenceLiteral(a));
90 if (IsOptional(b)) enforcement_literals.push_back(PresenceLiteral(b));
91 if (sat_solver_->CurrentDecisionLevel() == 0) {
92 int new_size = 0;
93 for (const Literal l : enforcement_literals) {
94 // We can ignore always absent interval, and skip the literal of the
95 // interval that are now always present.
96 if (assignment_.LiteralIsTrue(l)) continue;
97 if (assignment_.LiteralIsFalse(l)) return;
98 enforcement_literals[new_size++] = l;
99 }
100 enforcement_literals.resize(new_size);
101 }
102
103 const AffineExpression start_a = Start(a);
104 const AffineExpression end_a = End(a);
105 const AffineExpression start_b = Start(b);
106 const AffineExpression end_b = End(b);
107
108 // task_a is always before task_b ?
109 if (integer_trail_->UpperBound(start_a) < integer_trail_->LowerBound(end_b)) {
110 AddConditionalAffinePrecedence(enforcement_literals, end_a, start_b,
111 model_);
112 return;
113 }
114
115 // task_b is always before task_a ?
116 if (integer_trail_->UpperBound(start_b) < integer_trail_->LowerBound(end_a)) {
117 AddConditionalAffinePrecedence(enforcement_literals, end_b, start_a,
118 model_);
119 return;
120 }
121
122 // Create a new literal.
123 const BooleanVariable boolean_var = sat_solver_->NewBooleanVariable();
124 const Literal a_before_b = Literal(boolean_var, true);
125 disjunctive_precedences_.insert({{a, b}, a_before_b});
126 disjunctive_precedences_.insert({{b, a}, a_before_b.Negated()});
127
128 // Also insert it in precedences.
129 // TODO(user): also add the reverse like start_b + 1 <= end_a if negated?
130 precedences_.insert({{end_a, start_b}, a_before_b});
131 precedences_.insert({{end_b, start_a}, a_before_b.Negated()});
132
133 enforcement_literals.push_back(a_before_b);
134 AddConditionalAffinePrecedence(enforcement_literals, end_a, start_b, model_);
135 enforcement_literals.pop_back();
136
137 enforcement_literals.push_back(a_before_b.Negated());
138 AddConditionalAffinePrecedence(enforcement_literals, end_b, start_a, model_);
139 enforcement_literals.pop_back();
140
141 // Force the value of boolean_var in case the precedence is not active. This
142 // avoids duplicate solutions when enumerating all possible solutions.
143 for (const Literal l : enforcement_literals) {
144 implications_->AddBinaryClause(l, a_before_b);
145 }
146}

148bool IntervalsRepository::CreatePrecedenceLiteral(IntervalVariable a,
149 IntervalVariable b) {
150 const AffineExpression x = End(a);
151 const AffineExpression y = Start(b);
152 if (precedences_.contains({x, y})) return false;
153
154 // We want l => x <= y and not(l) => x > y <=> y + 1 <= x
155 // Do not create l if the relation is always true or false.
156 if (integer_trail_->UpperBound(x) <= integer_trail_->LowerBound(y)) {
157 return false;
158 }
159 if (integer_trail_->LowerBound(x) > integer_trail_->UpperBound(y)) {
160 return false;
161 }
162
163 // Create a new literal.
164 const BooleanVariable boolean_var = sat_solver_->NewBooleanVariable();
165 const Literal x_before_y = Literal(boolean_var, true);
166
167 // TODO(user): Also add {{y_plus_one, x}, x_before_y.Negated()} ?
168 precedences_.insert({{x, y}, x_before_y});
169
170 AffineExpression y_plus_one = y;
171 y_plus_one.constant += 1;
172 AddConditionalAffinePrecedence({x_before_y}, x, y, model_);
173 AddConditionalAffinePrecedence({x_before_y.Negated()}, y_plus_one, x, model_);
174 return true;
175}

227SchedulingConstraintHelper::SchedulingConstraintHelper(
228 const std::vector<IntervalVariable>& tasks, Model* model)
229 : trail_(model->GetOrCreate<Trail>()),
230 integer_trail_(model->GetOrCreate<IntegerTrail>()),
231 precedence_relations_(model->GetOrCreate<PrecedenceRelations>()),
232 precedences_(model->GetOrCreate<PrecedencesPropagator>()),
233 interval_variables_(tasks),
234 capacity_(tasks.size()),
235 cached_size_min_(new IntegerValue[capacity_]),
236 cached_start_min_(new IntegerValue[capacity_]),
237 cached_end_min_(new IntegerValue[capacity_]),
238 cached_negated_start_max_(new IntegerValue[capacity_]),
239 cached_negated_end_max_(new IntegerValue[capacity_]),
240 cached_shifted_start_min_(new IntegerValue[capacity_]),
241 cached_negated_shifted_end_max_(new IntegerValue[capacity_]) {
242 starts_.clear();
243 ends_.clear();
244 minus_ends_.clear();
245 minus_starts_.clear();
246 sizes_.clear();
247 reason_for_presence_.clear();
248
249 auto* repository = model->GetOrCreate<IntervalsRepository>();
250 for (const IntervalVariable i : tasks) {
251 if (repository->IsOptional(i)) {
252 reason_for_presence_.push_back(repository->PresenceLiteral(i).Index());
253 } else {
254 reason_for_presence_.push_back(kNoLiteralIndex);
255 }
256 sizes_.push_back(repository->Size(i));
257 starts_.push_back(repository->Start(i));
258 ends_.push_back(repository->End(i));
259 minus_starts_.push_back(repository->Start(i).Negated());
260 minus_ends_.push_back(repository->End(i).Negated());
261 }
262
263 RegisterWith(model->GetOrCreate<GenericLiteralWatcher>());
264 InitSortedVectors();
265 if (!SynchronizeAndSetTimeDirection(true)) {
266 model->GetOrCreate<SatSolver>()->NotifyThatModelIsUnsat();
267 }
268}

327bool SchedulingConstraintHelper::UpdateCachedValues(int t) {
328 recompute_cache_[t] = false;
329 if (IsAbsent(t)) return true;
330
331 IntegerValue smin = integer_trail_->LowerBound(starts_[t]);
332 IntegerValue smax = integer_trail_->UpperBound(starts_[t]);
333 IntegerValue emin = integer_trail_->LowerBound(ends_[t]);
334 IntegerValue emax = integer_trail_->UpperBound(ends_[t]);
335
336 // We take the max for the corner case where the size of an optional interval
337 // is used elsewhere and has a domain with negative value.
338 //
339 // TODO(user): maybe we should just disallow size with a negative domain, but
340 // is is harder to enforce if we have a linear expression for size.
341 IntegerValue dmin =
342 std::max(IntegerValue(0), integer_trail_->LowerBound(sizes_[t]));
343 IntegerValue dmax = integer_trail_->UpperBound(sizes_[t]);
344
345 // Detect first if we have a conflict using the relation start + size = end.
346 if (dmax < 0) {
347 AddSizeMaxReason(t, dmax);
348 return PushTaskAbsence(t);
349 }
350 if (smin + dmin - emax > 0) {
351 ClearReason();
352 AddStartMinReason(t, smin);
353 AddSizeMinReason(t, dmin);
354 AddEndMaxReason(t, emax);
355 return PushTaskAbsence(t);
356 }
357 if (smax + dmax - emin < 0) {
358 ClearReason();
359 AddStartMaxReason(t, smax);
360 AddSizeMaxReason(t, dmax);
361 AddEndMinReason(t, emin);
362 return PushTaskAbsence(t);
363 }
364
365 // Sometimes, for optional interval with non-optional bounds, this propagation
366 // give tighter bounds. We always consider the value assuming
367 // the interval is present.
368 //
369 // Note that this is also useful in case not everything was propagated. Note
370 // also that since there is no conflict, we reach the fix point in one pass.
371 smin = std::max(smin, emin - dmax);
372 smax = std::min(smax, emax - dmin);
373 dmin = std::max(dmin, emin - smax);
374 emin = std::max(emin, smin + dmin);
375 emax = std::min(emax, smax + dmax);
376
377 if (emin != cached_end_min_[t]) {
378 recompute_energy_profile_ = true;
379 }
380
381 cached_start_min_[t] = smin;
382 cached_end_min_[t] = emin;
383 cached_negated_start_max_[t] = -smax;
384 cached_negated_end_max_[t] = -emax;
385 cached_size_min_[t] = dmin;
386
387 // Note that we use the cached value here for EndMin()/StartMax().
388 const IntegerValue new_shifted_start_min = emin - dmin;
389 if (new_shifted_start_min != cached_shifted_start_min_[t]) {
390 recompute_energy_profile_ = true;
391 recompute_shifted_start_min_ = true;
392 cached_shifted_start_min_[t] = new_shifted_start_min;
393 }
394 const IntegerValue new_negated_shifted_end_max = -(smax + dmin);
395 if (new_negated_shifted_end_max != cached_negated_shifted_end_max_[t]) {
396 recompute_negated_shifted_end_max_ = true;
397 cached_negated_shifted_end_max_[t] = new_negated_shifted_end_max;
398 }
399 return true;
400}

402bool SchedulingConstraintHelper::ResetFromSubset(
403 const SchedulingConstraintHelper& other, absl::Span<const int> tasks) {
404 current_time_direction_ = other.current_time_direction_;
405
406 const int num_tasks = tasks.size();
407 interval_variables_.resize(num_tasks);
408 starts_.resize(num_tasks);
409 ends_.resize(num_tasks);
410 minus_ends_.resize(num_tasks);
411 minus_starts_.resize(num_tasks);
412 sizes_.resize(num_tasks);
413 reason_for_presence_.resize(num_tasks);
414 for (int i = 0; i < num_tasks; ++i) {
415 const int t = tasks[i];
416 interval_variables_[i] = other.interval_variables_[t];
417 starts_[i] = other.starts_[t];
418 ends_[i] = other.ends_[t];
419 minus_ends_[i] = other.minus_ends_[t];
420 minus_starts_[i] = other.minus_starts_[t];
421 sizes_[i] = other.sizes_[t];
422 reason_for_presence_[i] = other.reason_for_presence_[t];
423 }
424
425 InitSortedVectors();
426 return SynchronizeAndSetTimeDirection(true);
427}

429void SchedulingConstraintHelper::InitSortedVectors() {
430 const int num_tasks = starts_.size();
431
432 recompute_all_cache_ = true;
433 recompute_cache_.resize(num_tasks, true);
434
435 // Make sure all the cached_* arrays can hold enough data.
436 CHECK_LE(num_tasks, capacity_);
437
438 task_by_increasing_start_min_.resize(num_tasks);
439 task_by_increasing_end_min_.resize(num_tasks);
440 task_by_decreasing_start_max_.resize(num_tasks);
441 task_by_decreasing_end_max_.resize(num_tasks);
442 task_by_increasing_shifted_start_min_.resize(num_tasks);
443 task_by_negated_shifted_end_max_.resize(num_tasks);
444 for (int t = 0; t < num_tasks; ++t) {
445 task_by_increasing_start_min_[t].task_index = t;
446 task_by_increasing_end_min_[t].task_index = t;
447 task_by_decreasing_start_max_[t].task_index = t;
448 task_by_decreasing_end_max_[t].task_index = t;
449 task_by_increasing_shifted_start_min_[t].task_index = t;
450 task_by_negated_shifted_end_max_[t].task_index = t;
451 }
452
453 recompute_energy_profile_ = true;
454 recompute_shifted_start_min_ = true;
455 recompute_negated_shifted_end_max_ = true;
456}

458void SchedulingConstraintHelper::SetTimeDirection(bool is_forward) {
459 if (current_time_direction_ != is_forward) {
460 current_time_direction_ = is_forward;
461
462 std::swap(starts_, minus_ends_);
463 std::swap(ends_, minus_starts_);
464
465 std::swap(task_by_increasing_start_min_, task_by_decreasing_end_max_);
466 std::swap(task_by_increasing_end_min_, task_by_decreasing_start_max_);
467 std::swap(task_by_increasing_shifted_start_min_,
468 task_by_negated_shifted_end_max_);
469
470 recompute_energy_profile_ = true;
471 std::swap(cached_start_min_, cached_negated_end_max_);
472 std::swap(cached_end_min_, cached_negated_start_max_);
473 std::swap(cached_shifted_start_min_, cached_negated_shifted_end_max_);
474 std::swap(recompute_shifted_start_min_, recompute_negated_shifted_end_max_);
475 }

478bool SchedulingConstraintHelper::SynchronizeAndSetTimeDirection(
479 bool is_forward) {
480 SetTimeDirection(is_forward);
481 if (recompute_all_cache_) {
482 for (int t = 0; t < recompute_cache_.size(); ++t) {
483 if (!UpdateCachedValues(t)) return false;
484 }
485 } else {
486 for (int t = 0; t < recompute_cache_.size(); ++t) {
487 if (recompute_cache_[t]) {
488 if (!UpdateCachedValues(t)) return false;
489 }
490 }
491 }
492 recompute_all_cache_ = false;
493 return true;
494}

496// TODO(user): be more precise when we know a and b are in disjunction.
497// we really just need start_b > start_a, or even >= if duration is non-zero.
498IntegerValue SchedulingConstraintHelper::GetCurrentMinDistanceBetweenTasks(
499 int a, int b, bool add_reason_if_after) {
500 const AffineExpression before = ends_[a];
501 const AffineExpression after = starts_[b];
502 if (before.var == kNoIntegerVariable) return kMinIntegerValue;
503 if (after.var == kNoIntegerVariable) return kMinIntegerValue;
504
505 const IntegerValue needed = before.constant - after.constant;
506 const IntegerValue static_known =
507 precedence_relations_->GetOffset(before.var, after.var);
508
509 const std::pair<Literal, IntegerValue> dynamic_known =
510 precedences_->GetConditionalOffset(before.var, after.var);
511
512 const IntegerValue best = std::max(static_known, dynamic_known.second);
513 if (best == kMinIntegerValue) return kMinIntegerValue;
514
515 if (add_reason_if_after && dynamic_known.second > static_known &&
516 dynamic_known.second >= needed) {
517 literal_reason_.push_back(dynamic_known.first.Negated());
518 }
519 return best - needed;
520}

522void SchedulingConstraintHelper::AddLevelZeroPrecedence(int a, int b) {
523 CHECK_EQ(trail_->CurrentDecisionLevel(), 0);
524 if (!IsPresent(a)) return;
525 if (!IsPresent(b)) return;
526 const AffineExpression before = ends_[a];
527 const AffineExpression after = starts_[b];
528 if (after.coeff != 1) return;
529 if (before.coeff != 1) return;
530 if (after.var == kNoIntegerVariable) return;
531 if (before.var == kNoIntegerVariable) return;
532 const IntegerValue offset = before.constant - after.constant;
533 precedence_relations_->Add(before.var, after.var, offset);
534 if (precedences_->AddPrecedenceWithOffsetIfNew(before.var, after.var,
535 offset)) {
536 VLOG(2) << "new relation " << TaskDebugString(a)
537 << " <= " << TaskDebugString(b);
538 }

613 if (energy_profile_.empty()) {
614 const int num_tasks = NumTasks();
615 for (int t = 0; t < num_tasks; ++t) {
616 energy_profile_.push_back(
617 {cached_shifted_start_min_[t], t, /*is_first=*/true});
618 energy_profile_.push_back({cached_end_min_[t], t, /*is_first=*/false});
619 }
620 } else {
621 if (!recompute_energy_profile_) return energy_profile_;
622 for (ProfileEvent& ref : energy_profile_) {
623 const int t = ref.task;
624 if (ref.is_first) {
625 ref.time = cached_shifted_start_min_[t];
626 } else {
627 ref.time = cached_end_min_[t];
628 }
629 }
630 }
631 IncrementalSort(energy_profile_.begin(), energy_profile_.end());
632 recompute_energy_profile_ = false;
633 return energy_profile_;
634}

636// Produces a relaxed reason for StartMax(before) < EndMin(after).
637void SchedulingConstraintHelper::AddReasonForBeingBefore(int before,
638 int after) {
639 AddOtherReason(before);
640 AddOtherReason(after);
641
642 // The reason will be a linear expression greater than a value. Note that all
643 // coeff must be positive, and we will use the variable lower bound.
644 std::vector<IntegerVariable> vars;
645 std::vector<IntegerValue> coeffs;
646
647 // Reason for StartMax(before).
648 const IntegerValue smax_before = StartMax(before);
649 if (smax_before >= integer_trail_->UpperBound(starts_[before])) {
650 if (starts_[before].var != kNoIntegerVariable) {
651 vars.push_back(NegationOf(starts_[before].var));
652 coeffs.push_back(starts_[before].coeff);
653 }
654 } else {
655 if (ends_[before].var != kNoIntegerVariable) {
656 vars.push_back(NegationOf(ends_[before].var));
657 coeffs.push_back(ends_[before].coeff);
658 }
659 if (sizes_[before].var != kNoIntegerVariable) {
660 vars.push_back(sizes_[before].var);
661 coeffs.push_back(sizes_[before].coeff);
662 }
663 }
664
665 // Reason for EndMin(after);
666 const IntegerValue emin_after = EndMin(after);
667 if (emin_after <= integer_trail_->LowerBound(ends_[after])) {
668 if (ends_[after].var != kNoIntegerVariable) {
669 vars.push_back(ends_[after].var);
670 coeffs.push_back(ends_[after].coeff);
671 }
672 } else {
673 if (starts_[after].var != kNoIntegerVariable) {
674 vars.push_back(starts_[after].var);
675 coeffs.push_back(starts_[after].coeff);
676 }
677 if (sizes_[after].var != kNoIntegerVariable) {
678 vars.push_back(sizes_[after].var);
679 coeffs.push_back(sizes_[after].coeff);
680 }
681 }
682
683 DCHECK_LT(smax_before, emin_after);
684 const IntegerValue slack = emin_after - smax_before - 1;
685 integer_trail_->AppendRelaxedLinearReason(slack, coeffs, vars,
686 &integer_reason_);
687}

840IntegerValue ComputeEnergyMinInWindow(
841 IntegerValue start_min, IntegerValue start_max, IntegerValue end_min,
842 IntegerValue end_max, IntegerValue size_min, IntegerValue demand_min,
843 absl::Span<const LiteralValueValue> filtered_energy,
844 IntegerValue window_start, IntegerValue window_end) {
845 if (window_end <= window_start) return IntegerValue(0);
846
847 // Returns zero if the interval do not necessarily overlap.
848 if (end_min <= window_start) return IntegerValue(0);
849 if (start_max >= window_end) return IntegerValue(0);
850 const IntegerValue window_size = window_end - window_start;
851 const IntegerValue simple_energy_min =
852 demand_min * std::min({end_min - window_start, window_end - start_max,
853 size_min, window_size});
854 if (filtered_energy.empty()) return simple_energy_min;
855
856 IntegerValue result = kMaxIntegerValue;
857 for (const auto [lit, fixed_size, fixed_demand] : filtered_energy) {
858 const IntegerValue alt_end_min = std::max(end_min, start_min + fixed_size);
859 const IntegerValue alt_start_max =
860 std::min(start_max, end_max - fixed_size);
861 const IntegerValue energy_min =
862 fixed_demand *
863 std::min({alt_end_min - window_start, window_end - alt_start_max,
864 fixed_size, window_size});
865 result = std::min(result, energy_min);
866 }
867 if (result == kMaxIntegerValue) return simple_energy_min;
868 return std::max(simple_energy_min, result);
869}

952 const int num_tasks = cached_energies_min_.size();
953 const bool is_at_level_zero = sat_solver_->CurrentDecisionLevel() == 0;
954 for (int t = 0; t < num_tasks; ++t) {
955 // Try to reduce the size of the decomposed energy vector.
956 if (is_at_level_zero) {
957 int new_size = 0;
958 for (int i = 0; i < decomposed_energies_[t].size(); ++i) {
959 if (assignment_.LiteralIsFalse(decomposed_energies_[t][i].literal)) {
960 continue;
961 }

963 }
964 decomposed_energies_[t].resize(new_size);
965 }
966
967 cached_energies_min_[t] = std::max(
968 {SimpleEnergyMin(t), LinearEnergyMin(t), DecomposedEnergyMin(t)});
969 CHECK_NE(cached_energies_min_[t], kMinIntegerValue);
970 energy_is_quadratic_[t] =
971 decomposed_energies_[t].empty() && !demands_.empty() &&
972 !integer_trail_->IsFixed(demands_[t]) && !helper_->SizeIsFixed(t);
973 cached_energies_max_[t] = std::min(
974 {SimpleEnergyMax(t), LinearEnergyMax(t), DecomposedEnergyMax(t)});
975 CHECK_NE(cached_energies_max_[t], kMaxIntegerValue);

993bool SchedulingDemandHelper::DecreaseEnergyMax(int t, IntegerValue value) {
994 if (value < EnergyMin(t)) {
995 if (helper_->IsOptional(t)) {
996 return helper_->PushTaskAbsence(t);
997 } else {
998 return helper_->ReportConflict();
999 }
1000 } else if (!decomposed_energies_[t].empty()) {

1002 if (fixed_size * fixed_demand > value) {
1003 if (assignment_.LiteralIsTrue(lit)) return helper_->ReportConflict();
1004 if (assignment_.LiteralIsFalse(lit)) continue;
1005 if (!helper_->PushLiteral(lit.Negated())) return false;
1006 }
1007 }
1008 } else if (linearized_energies_[t].has_value() &&
1009 linearized_energies_[t]->vars.size() == 1) {
1010 const LinearExpression& e = linearized_energies_[t].value();
1011 const AffineExpression affine_energy(e.vars[0], e.coeffs[0], e.offset);
1012 const IntegerLiteral deduction = affine_energy.LowerOrEqual(value);
1013 if (!helper_->PushIntegerLiteralIfTaskPresent(t, deduction)) {
1014 return false;
1015 }
1016 } else {
1017 // TODO(user): Propagate if possible.
1018 VLOG(3) << "Cumulative energy missed propagation";
1019 }
1020 return true;
1021}

1036 const int old_size = reason->size();

1038 if (assignment_.LiteralIsTrue(lit)) {

1043 assignment_.LiteralIsFalse(lit)) {
1044 reason->push_back(lit);
1045 }
1046 }
1047 } else if (SimpleEnergyMin(t) >= value) {
1048 AddDemandMinReason(t);
1049 helper_->AddSizeMinReason(t);
1050 } else {
1051 DCHECK_GE(LinearEnergyMin(t), value);
1052 for (const IntegerVariable var : linearized_energies_[t]->vars) {
1053 helper_->MutableIntegerReason()->push_back(
1054 integer_trail_->LowerBoundAsLiteral(var));
1055 }

1124// here.
1125void SchedulingDemandHelper::AddEnergyMinInWindowReason(
1126 int t, IntegerValue window_start, IntegerValue window_end) {
1127 const IntegerValue actual_energy_min =
1128 EnergyMinInWindow(t, window_start, window_end);
1129 if (actual_energy_min == 0) return;
1130
1131 // Return simple reason right away if there is no decomposition or the simple
1132 // energy is enough.
1133 const IntegerValue start_max = helper_->StartMax(t);
1134 const IntegerValue end_min = helper_->EndMin(t);
1135 const IntegerValue min_overlap =
1136 helper_->GetMinOverlap(t, window_start, window_end);
1137 const IntegerValue simple_energy_min = DemandMin(t) * min_overlap;
1138 if (simple_energy_min == actual_energy_min) {
1139 AddDemandMinReason(t);
1140 helper_->AddSizeMinReason(t);
1141 helper_->AddStartMaxReason(t, start_max);
1142 helper_->AddEndMinReason(t, end_min);
1143 return;
1144 }
1145
1146 // TODO(user): only include the one we need?
1147 const IntegerValue start_min = helper_->StartMin(t);
1148 const IntegerValue end_max = helper_->EndMax(t);
1149 DCHECK(!decomposed_energies_[t].empty());
1150 helper_->AddStartMinReason(t, start_min);
1151 helper_->AddStartMaxReason(t, start_max);
1152 helper_->AddEndMinReason(t, end_min);
1153 helper_->AddEndMaxReason(t, end_max);
1154
1155 auto* literal_reason = helper_->MutableLiteralReason();
1156 const int old_size = literal_reason->size();
1157
1158 DCHECK(!decomposed_energies_[t].empty());
1159 for (const auto [lit, fixed_size, fixed_demand] : decomposed_energies_[t]) {
1160 // Should be the same in most cases.
1161 if (assignment_.LiteralIsTrue(lit)) {
1162 literal_reason->resize(old_size);
1163 literal_reason->push_back(lit.Negated());
1164 return;
1165 }
1166 if (assignment_.LiteralIsFalse(lit)) {
1167 const IntegerValue alt_em = std::max(end_min, start_min + fixed_size);
1168 const IntegerValue alt_sm = std::min(start_max, end_max - fixed_size);
1169 const IntegerValue energy_min =
1170 fixed_demand *
1171 std::min({alt_em - window_start, window_end - alt_sm, fixed_size});
1172 if (energy_min >= actual_energy_min) continue;
1173 literal_reason->push_back(lit);
1174 }
1175 }
