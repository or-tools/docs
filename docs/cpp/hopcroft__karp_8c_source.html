42void hopcroft_karp_state_init(int u_size, int v_size, int cpu,
43 struct hopcroft_karp_state* hk_state)
44{
45 int bytes;
46 int* memory;
47
48 hk_state->graph.u_size = u_size;
49 hk_state->graph.v_size = v_size;
50
51 /*
52 * Memory Required
53 *
54 * hk_state->pair_u: u_size * sizeof(int)
55 * hk_state->pair_v: v_size * sizeof(int)
56 * hk_state->distance: (1 + u_size) * sizeof(int)
57 * hk_state->queue_data: u_size * sizeof(int)
58 * hk_state->u_stack: u_size * sizeof(int)
59 * hk_state->v_stack: u_size * sizeof(int)
60 * hk_state->graph.adjacent: u_size * v_size * sizeof(char)
61 */
62 bytes = u_size * sizeof(int) + v_size * sizeof(int) +
63 (1 + u_size) * sizeof(int) + u_size * sizeof(int) +
64 u_size * sizeof(int) + u_size * sizeof(int) +
65 u_size * v_size * sizeof(char);
66
67 memory = (int*)ALLOC(bytes, cpu);
68 if (WARN_ON_ONCE(memory == NULL)) {
69 return;
70 }
71
72 hk_state->pair_u = memory;
73 memory += u_size;
74
75 hk_state->pair_v = memory;
76 memory += v_size;
77
78 hk_state->distance = memory;
79 memory += (1 + u_size);
80
81 hk_state->queue_data = memory;
82 memory += u_size;
83
84 hk_state->u_stack = memory;
85 memory += u_size;
86
87 hk_state->v_stack = memory;
88 memory += u_size;
89
90 hk_state->graph.adjacent = (char*)memory;
91}

105/*
106 * Breadth-first search: Separates the vertices of the graph into layers.
107 */
108static int breadth_first_search(struct hopcroft_karp_state* hk_state)
109{
110 int u;
111 int queue_size = 0, queue_front = 0, queue_index = 0;
112
113 /*
114 * The free vertices of U, i.e. those that are paired with the NIL node,
115 * are used as the starting vertices and form the first layer of the
116 * partition.
117 */
118 for (u = 0; u < hk_state->graph.u_size; ++u) {
119 if (hk_state->pair_u[u] == HK_NIL_NODE) {
120 hk_state->distance[1 + u] = 0;
121 /* Enqueue u */
122 hk_state->queue_data[queue_size] = u;
123 queue_size++;
124 } else {
125 hk_state->distance[1 + u] = INT_MAX;
126 }
127 }
128
129 /* NIL node */
130 hk_state->distance[0] = INT_MAX;
131
132 /*
133 * Traversal: Alternate between unmatched edges starting from free U nodes
134 * and matched edges starting from paired V nodes. The search terminates
135 * at the first level k where one or more free vertices in V are reached.
136 */
137 while (queue_size > 0) {
138 /* Dequeue u */
139 u = hk_state->queue_data[queue_front];
140 queue_size--;
141 queue_front++;
142 queue_front = queue_front < hk_state->graph.u_size ? queue_front : 0;
143
144 if (hk_state->distance[1 + u] < hk_state->distance[0]) {
145 int v;
146 int u_row = u * hk_state->graph.v_size;
147 for (v = 0; v < hk_state->graph.v_size; ++v) {
148 if (hk_state->graph.adjacent[u_row + v]) {
149 if (hk_state->distance[1 + hk_state->pair_v[v]] == INT_MAX) {
150 hk_state->distance[1 + hk_state->pair_v[v]] =
151 SATURATED_INC(hk_state->distance[1 + u]);
152 /* Enqueue pair_v[v] */
153 queue_index = queue_front + queue_size;
154 queue_index = queue_index < hk_state->graph.u_size ?
155 queue_index :
156 queue_index - hk_state->graph.u_size;
157 hk_state->queue_data[queue_index] = hk_state->pair_v[v];
158 queue_size++;
159 }
160 }
161 }
162 }
163 }
164
165 return hk_state->distance[0] != INT_MAX;
166}

168/*
169 * Depth-first search: Computes a maximal set of vertex disjoint shortest
170 * augmenting paths of length k starting from the free
171 * vertices of V computed by BFS.
172 * (Iterative implementation)
173 */
174static int depth_first_search(int u_node, struct hopcroft_karp_state* hk_state)
175{
176 int stack_length = 1, u_row, u;
177 int* v_ptr;
178
179 /* If a free vertex of U is reached, an augmenting path was found. */
180 if (u_node == HK_NIL_NODE) {
181 return 1;
182 }
183
184 if (WARN_ON_ONCE(hk_state->graph.u_size <= 0)) {
185 return -1;
186 }
187
188 /* For each 'u' node start examining from the first node in the V set */
189 memset(hk_state->v_stack, 0, hk_state->graph.u_size * sizeof(int));
190 hk_state->u_stack[0] = u_node;
191
192 while (stack_length > 0) {
193 /* u = top of stack */
194 u = hk_state->u_stack[stack_length - 1];
195 u_row = u * hk_state->graph.v_size;
196 v_ptr = &(hk_state->v_stack[stack_length - 1]);
197
198 /* For this 'u' node scan the 'v' nodes in the V set */
199 for (; *v_ptr < hk_state->graph.v_size; (*v_ptr)++) {
200 /* If this 'v' is connected to the current 'u' node */
201 if (hk_state->graph.adjacent[u_row + *v_ptr]) {
202 /* If it is not more than one hop away */
203 if (hk_state->distance[1 + hk_state->pair_v[*v_ptr]]
204 == SATURATED_INC(hk_state->distance[1 + u])) {
205 /* Is this 'v' node paired with a NIL node? */
206 if (hk_state->pair_v[*v_ptr] == HK_NIL_NODE) {
207 /*
208 * A free vertex of U is reached and an augmenting path
209 * was found. Pop the entire stack and return true.
210 */
211 while (stack_length > 0) {
212 stack_length--;
213 /* Adjust pairs */
214 hk_state->pair_v[hk_state->v_stack[stack_length]] =
215 hk_state->u_stack[stack_length];
216 hk_state->pair_u[hk_state->u_stack[stack_length]] =
217 hk_state->v_stack[stack_length];
218 }
219 return 1;
220 } else {
221 /*
222 * In case we need further investigation push
223 * the new 'u' node discovered, in the stack.
224 */
225 hk_state->v_stack[stack_length] = 0;
226 hk_state->u_stack[stack_length] =
227 hk_state->pair_v[*v_ptr];
228 stack_length++;
229 }
230 break;
231 }
232 }
233 }
234
235 if (*v_ptr >= hk_state->graph.v_size) {
236 /*
237 * The 'u' node's neighbors in V set were fully scanned and
238 * nothing interesting was found: pop and continue with the rest.
239 */
240 hk_state->distance[1 + u] = INT_MAX;
241 stack_length--;
242 }
243 }
244
245 return 0;
246}

248int hopcroft_karp_matching(struct hopcroft_karp_state* hk_state, int* pairs)
249{
250 int u, v, matching = 0;
251
252 for (u = 0; u < hk_state->graph.u_size; ++u) {
253 hk_state->pair_u[u] = HK_NIL_NODE;
254 }
255 for (v = 0; v < hk_state->graph.v_size; ++v) {
256 hk_state->pair_v[v] = HK_NIL_NODE;
257 }
258
259 while (breadth_first_search(hk_state)) {
260 for (u = 0; u < hk_state->graph.u_size; ++u) {
261 if (hk_state->pair_u[u] == HK_NIL_NODE) {
262 if (depth_first_search(u, hk_state)) {
263 ++matching;
264 }
265 }
266 }
267 }
268
269 /* Result */
270 if (pairs) {
271 for (u = 0; u < hk_state->graph.u_size; ++u) {
272 pairs[u] = hk_state->pair_u[u];
273 }
274 }
275
276 return matching;
277}
