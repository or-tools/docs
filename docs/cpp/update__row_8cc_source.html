61void UpdateRow::ComputeUnitRowLeftInverse(RowIndex leaving_row) {
62 if (left_inverse_computed_for_ == leaving_row) return;
63 left_inverse_computed_for_ = leaving_row;
64 SCOPED_TIME_STAT(&stats_);
65
66 basis_factorization_.LeftSolveForUnitRow(RowToColIndex(leaving_row),
67 &unit_row_left_inverse_);
68
69 // TODO(user): Refactorize if the estimated accuracy is above a threshold.
70 IF_STATS_ENABLED(stats_.unit_row_left_inverse_accuracy.Add(
71 matrix_.ColumnScalarProduct(basis_[leaving_row],
72 unit_row_left_inverse_.values) -
73 1.0));
74 IF_STATS_ENABLED(stats_.unit_row_left_inverse_density.Add(
75 Density(unit_row_left_inverse_.values)));

78void UpdateRow::ComputeUpdateRow(RowIndex leaving_row) {
79 if (update_row_computed_for_ == leaving_row) return;
80 update_row_computed_for_ = leaving_row;
81 ComputeUnitRowLeftInverse(leaving_row);
82 SCOPED_TIME_STAT(&stats_);
83
84 if (parameters_.use_transposed_matrix()) {
85 // Number of entries that ComputeUpdatesRowWise() will need to look at.
86 EntryIndex num_row_wise_entries(0);
87
88 // Because we are about to do an expensive matrix-vector product, we make
89 // sure we drop small entries in the vector for the row-wise algorithm. We
90 // also computes its non-zeros to simplify the code below.
91 //
92 // TODO(user): So far we didn't generalize the use of drop tolerances
93 // everywhere in the solver, so we make sure to not modify
94 // unit_row_left_inverse_ that is also used elsewhere. However, because of
95 // that, we will not get the exact same result depending on the algortihm
96 // used below because the ComputeUpdatesColumnWise() will still use these
97 // small entries (no complexity changes).
98 const Fractional drop_tolerance = parameters_.drop_tolerance();
99 unit_row_left_inverse_filtered_non_zeros_.clear();
100 const auto view = transposed_matrix_.view();
101 if (unit_row_left_inverse_.non_zeros.empty()) {
102 const ColIndex size = unit_row_left_inverse_.values.size();
103 for (ColIndex col(0); col < size; ++col) {
104 if (std::abs(unit_row_left_inverse_.values[col]) > drop_tolerance) {
105 unit_row_left_inverse_filtered_non_zeros_.push_back(col);
106 num_row_wise_entries += view.ColumnNumEntries(col);
107 }
108 }
109 } else {
110 for (const auto e : unit_row_left_inverse_) {
111 if (std::abs(e.coefficient()) > drop_tolerance) {
112 unit_row_left_inverse_filtered_non_zeros_.push_back(e.column());
113 num_row_wise_entries += view.ColumnNumEntries(e.column());
114 }
115 }
116 }
117
118 // The case of size 1 happens often enough to deserve special code.
119 //
120 // TODO(user): The impact is not as high as I hopped though, so not too
121 // important.
122 if (unit_row_left_inverse_filtered_non_zeros_.size() == 1) {
123 ComputeUpdatesForSingleRow(
124 unit_row_left_inverse_filtered_non_zeros_.front());
125 num_operations_ += num_row_wise_entries.value();
126 IF_STATS_ENABLED(stats_.update_row_density.Add(
127 static_cast<double>(num_non_zeros_) /
128 static_cast<double>(matrix_.num_cols().value())));
129 return;
130 }
131
132 // Number of entries that ComputeUpdatesColumnWise() will need to look at.
133 const EntryIndex num_col_wise_entries =
134 variables_info_.GetNumEntriesInRelevantColumns();
135
136 // Note that the thresholds were chosen (more or less) from the result of
137 // the microbenchmark tests of this file in September 2013.
138 // TODO(user): automate the computation of these constants at run-time?
139 const double row_wise = static_cast<double>(num_row_wise_entries.value());
140 if (row_wise < 0.5 * static_cast<double>(num_col_wise_entries.value())) {
141 if (row_wise < 1.1 * static_cast<double>(matrix_.num_cols().value())) {
142 ComputeUpdatesRowWiseHypersparse();
143
144 // We use a multiplicative factor because these entries are often widely
145 // spread in memory. There is also some overhead to each fp operations.
146 num_operations_ +=
147 5 * num_row_wise_entries.value() + matrix_.num_cols().value() / 64;
148 } else {
149 ComputeUpdatesRowWise();
150 num_operations_ +=
151 num_row_wise_entries.value() + matrix_.num_rows().value();
152 }
153 } else {
154 ComputeUpdatesColumnWise();
155 num_operations_ +=
156 num_col_wise_entries.value() + matrix_.num_cols().value();
157 }
158 } else {
159 ComputeUpdatesColumnWise();
160 num_operations_ +=
161 variables_info_.GetNumEntriesInRelevantColumns().value() +
162 matrix_.num_cols().value();
163 }
164 IF_STATS_ENABLED(stats_.update_row_density.Add(
165 static_cast<double>(num_non_zeros_) /
166 static_cast<double>(matrix_.num_cols().value())));
167}

169void UpdateRow::ComputeUpdateRowForBenchmark(const DenseRow& lhs,
170 const std::string& algorithm) {
171 unit_row_left_inverse_.values = lhs;
172 ComputeNonZeros(lhs, &unit_row_left_inverse_filtered_non_zeros_);
173 if (algorithm == "column") {
174 ComputeUpdatesColumnWise();
175 } else if (algorithm == "row") {
176 ComputeUpdatesRowWise();
177 } else if (algorithm == "row_hypersparse") {
178 ComputeUpdatesRowWiseHypersparse();
179 } else {
180 LOG(DFATAL) << "Unknown algorithm in ComputeUpdateRowForBenchmark(): '"
181 << algorithm << "'";
182 }
183}

195// This is optimized for the case when the total number of entries is about
196// the same as, or greater than, the number of columns.
197void UpdateRow::ComputeUpdatesRowWise() {
198 SCOPED_TIME_STAT(&stats_);
199 coefficient_.AssignToZero(matrix_.num_cols());
200 const auto output_coeffs = coefficient_.view();
201 const auto view = transposed_matrix_.view();
202 for (const ColIndex col : unit_row_left_inverse_filtered_non_zeros_) {
203 const Fractional multiplier = unit_row_left_inverse_[col];
204 for (const EntryIndex i : view.Column(col)) {
205 const ColIndex pos = RowToColIndex(view.EntryRow(i));
206 output_coeffs[pos] += multiplier * view.EntryCoefficient(i);
207 }
208 }
209
210 non_zero_position_list_.resize(matrix_.num_cols().value());
211 auto* non_zeros = non_zero_position_list_.data();
212 const Fractional drop_tolerance = parameters_.drop_tolerance();
213 for (const ColIndex col : variables_info_.GetIsRelevantBitRow()) {
214 if (std::abs(output_coeffs[col]) > drop_tolerance) {
215 *non_zeros++ = col;
216 }
217 }
218 num_non_zeros_ = non_zeros - non_zero_position_list_.data();
219}

221// This is optimized for the case when the total number of entries is smaller
222// than the number of columns.
223void UpdateRow::ComputeUpdatesRowWiseHypersparse() {
224 SCOPED_TIME_STAT(&stats_);
225 const ColIndex num_cols = matrix_.num_cols();
226 non_zero_position_set_.ClearAndResize(num_cols);
227 coefficient_.resize(num_cols, 0.0);
228
229 const auto output_coeffs = coefficient_.view();
230 const auto view = transposed_matrix_.view();
231 const auto nz_set = non_zero_position_set_.const_view();
232 for (const ColIndex col : unit_row_left_inverse_filtered_non_zeros_) {
233 const Fractional multiplier = unit_row_left_inverse_[col];
234 for (const EntryIndex i : view.Column(col)) {
235 const ColIndex pos = RowToColIndex(view.EntryRow(i));
236 const Fractional v = multiplier * view.EntryCoefficient(i);
237 if (!nz_set[pos]) {
238 // Note that we could create the non_zero_position_list_ here, but we
239 // prefer to keep the non-zero positions sorted, so using the bitset is
240 // a good alternative. Of course if the solution is really really
241 // sparse, then sorting non_zero_position_list_ will be faster.
242 output_coeffs[pos] = v;
243 non_zero_position_set_.Set(pos);
244 } else {
245 output_coeffs[pos] += v;
246 }
247 }
248 }
249
250 // Only keep in non_zero_position_set_ the relevant positions.
251 non_zero_position_set_.Intersection(variables_info_.GetIsRelevantBitRow());
252 non_zero_position_list_.resize(matrix_.num_cols().value());
253 auto* non_zeros = non_zero_position_list_.data();
254 const Fractional drop_tolerance = parameters_.drop_tolerance();
255 for (const ColIndex col : non_zero_position_set_) {
256 // TODO(user): Since the solution is really sparse, maybe storing the
257 // non-zero coefficients contiguously in a vector is better than keeping
258 // them as they are. Note however that we will iterate only twice on the
259 // update row coefficients during an iteration.
260 if (std::abs(output_coeffs[col]) > drop_tolerance) {
261 *non_zeros++ = col;
262 }
263 }
264 num_non_zeros_ = non_zeros - non_zero_position_list_.data();
265}

267void UpdateRow::ComputeUpdatesForSingleRow(ColIndex row_as_col) {
268 coefficient_.resize(matrix_.num_cols(), 0.0);
269 non_zero_position_list_.resize(matrix_.num_cols().value());
270 auto* non_zeros = non_zero_position_list_.data();
271
272 const DenseBitRow& is_relevant = variables_info_.GetIsRelevantBitRow();
273 const Fractional drop_tolerance = parameters_.drop_tolerance();
274 const Fractional multiplier = unit_row_left_inverse_[row_as_col];
275 const auto output_coeffs = coefficient_.view();
276 const auto view = transposed_matrix_.view();
277 for (const EntryIndex i : view.Column(row_as_col)) {
278 const ColIndex pos = RowToColIndex(view.EntryRow(i));
279 if (!is_relevant[pos]) continue;
280
281 const Fractional v = multiplier * view.EntryCoefficient(i);
282 if (std::abs(v) > drop_tolerance) {
283 output_coeffs[pos] = v;
284 *non_zeros++ = pos;
285 }
286 }
287 num_non_zeros_ = non_zeros - non_zero_position_list_.data();
288}

290void UpdateRow::ComputeUpdatesColumnWise() {
291 SCOPED_TIME_STAT(&stats_);
292
293 coefficient_.resize(matrix_.num_cols(), 0.0);
294 non_zero_position_list_.resize(matrix_.num_cols().value());
295 auto* non_zeros = non_zero_position_list_.data();
296
297 const Fractional drop_tolerance = parameters_.drop_tolerance();
298 const auto output_coeffs = coefficient_.view();
299 const auto view = matrix_.view();
300 const auto unit_row_left_inverse = unit_row_left_inverse_.values.const_view();
301 for (const ColIndex col : variables_info_.GetIsRelevantBitRow()) {
302 // Coefficient of the column right inverse on the 'leaving_row'.
303 const Fractional coeff =
304 view.ColumnScalarProduct(col, unit_row_left_inverse);
305
306 // Nothing to do if 'coeff' is (almost) zero which does happen due to
307 // sparsity. Note that it shouldn't be too bad to use a non-zero drop
308 // tolerance here because even if we introduce some precision issues, the
309 // quantities updated by this update row will eventually be recomputed.
310 if (std::abs(coeff) > drop_tolerance) {
311 *non_zeros++ = col;
312 output_coeffs[col] = coeff;
313 }
314 }
315 num_non_zeros_ = non_zeros - non_zero_position_list_.data();
316}

318// Note that we use the same algo as ComputeUpdatesColumnWise() here. The
319// others version might be faster, but this is called at most once per solve, so
320// it shouldn't be too bad.
321void UpdateRow::ComputeFullUpdateRow(RowIndex leaving_row,
322 DenseRow* output) const {
323 CHECK_EQ(leaving_row, left_inverse_computed_for_);
324
325 const ColIndex num_cols = matrix_.num_cols();
326 output->AssignToZero(num_cols);
327
328 // Fills the only position at one in the basic columns.
329 (*output)[basis_[leaving_row]] = 1.0;
330
331 // Fills the non-basic column.
332 const Fractional drop_tolerance = parameters_.drop_tolerance();
333 const auto view = matrix_.view();
334 const auto unit_row_left_inverse = unit_row_left_inverse_.values.const_view();
335 for (const ColIndex col : variables_info_.GetNotBasicBitRow()) {
336 const Fractional coeff =
337 view.ColumnScalarProduct(col, unit_row_left_inverse);
338 if (std::abs(coeff) > drop_tolerance) {
339 (*output)[col] = coeff;
340 }
341 }
342}
