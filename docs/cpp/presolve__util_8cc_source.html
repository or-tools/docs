65void DomainDeductions::AddDeduction(int literal_ref, int var, Domain domain) {
66 CHECK_GE(var, 0);
67 const Index index = IndexFromLiteral(literal_ref);
68 if (index >= something_changed_.size()) {
69 something_changed_.Resize(index + 1);
70 enforcement_to_vars_.resize(index.value() + 1);
71 }
72 if (var >= tmp_num_occurrences_.size()) {
73 tmp_num_occurrences_.resize(var + 1, 0);
74 }
75 const auto insert = deductions_.insert({{index, var}, domain});
76 if (insert.second) {
77 // New element.
78 something_changed_.Set(index);
79 enforcement_to_vars_[index].push_back(var);
80 } else {
81 // Existing element.
82 const Domain& old_domain = insert.first->second;
83 if (!old_domain.IsIncludedIn(domain)) {
84 insert.first->second = domain.IntersectionWith(old_domain);
85 something_changed_.Set(index);
86 }
87 }
88}

90Domain DomainDeductions::ImpliedDomain(int literal_ref, int var) const {
91 CHECK_GE(var, 0);
92 const Index index = IndexFromLiteral(literal_ref);
93 const auto it = deductions_.find({index, var});
94 if (it == deductions_.end()) return Domain::AllValues();
95 return it->second;
96}

98std::vector<std::pair<int, Domain>> DomainDeductions::ProcessClause(
99 absl::Span<const int> clause) {
100 std::vector<std::pair<int, Domain>> result;
101
102 // We only need to process this clause if something changed since last time.
103 bool abort = true;
104 for (const int ref : clause) {
105 const Index index = IndexFromLiteral(ref);
106 if (index >= something_changed_.size()) return result;
107 if (something_changed_[index]) {
108 abort = false;
109 }
110 }
111 if (abort) return result;
112
113 // Count for each variable, how many times it appears in the deductions lists.
114 std::vector<int> to_process;
115 std::vector<int> to_clean;
116 for (const int ref : clause) {
117 const Index index = IndexFromLiteral(ref);
118 for (const int var : enforcement_to_vars_[index]) {
119 if (tmp_num_occurrences_[var] == 0) {
120 to_clean.push_back(var);
121 }
122 tmp_num_occurrences_[var]++;
123 if (tmp_num_occurrences_[var] == clause.size()) {
124 to_process.push_back(var);
125 }
126 }
127 }
128
129 // Clear the counts.
130 for (const int var : to_clean) {
131 tmp_num_occurrences_[var] = 0;
132 }
133
134 // Compute the domain unions.
135 std::vector<Domain> domains(to_process.size());
136 for (const int ref : clause) {
137 const Index index = IndexFromLiteral(ref);
138 for (int i = 0; i < to_process.size(); ++i) {
139 domains[i] = domains[i].UnionWith(deductions_.at({index, to_process[i]}));
140 }
141 }
142
143 for (int i = 0; i < to_process.size(); ++i) {
144 result.push_back({to_process[i], std::move(domains[i])});
145 }
146 return result;
147}

151// Helper method for variable substitution.
152// Sorts and merges the terms in 'terms'.
153// Returns false on overflow.
154bool SortAndMergeTerms(std::vector<std::pair<int, int64_t>>* terms) {
155 std::sort(terms->begin(), terms->end());
156
157 int new_size = 0;
158 int current_var = 0;
159 int64_t current_coeff = 0;
160 for (const auto& entry : *terms) {
161 DCHECK(RefIsPositive(entry.first));
162 if (entry.first == current_var) {
163 current_coeff = CapAdd(current_coeff, entry.second);
164 if (AtMinOrMaxInt64(current_coeff)) return false;
165 } else {
166 if (current_coeff != 0) {
167 (*terms)[new_size++] = {current_var, current_coeff};
168 }
169 current_var = entry.first;
170 current_coeff = entry.second;
171 }
172 }
173 if (current_coeff != 0) {
174 (*terms)[new_size++] = {current_var, current_coeff};
175 }
176 terms->resize(new_size);
177 return true;
178}

182bool AddLinearConstraintMultiple(int64_t factor, const ConstraintProto& to_add,
183 ConstraintProto* to_modify) {
184 if (factor == 0) return true;
185 DCHECK_EQ(to_add.constraint_case(), ConstraintProto::kLinear);
186 DCHECK_EQ(to_modify->constraint_case(), ConstraintProto::kLinear);
187
188 // Copy to_modify terms.
189 std::vector<std::pair<int, int64_t>> terms;
190 LinearConstraintProto* out = to_modify->mutable_linear();
191 const int initial_size = out->vars().size();
192 for (int i = 0; i < initial_size; ++i) {
193 const int var = out->vars(i);
194 const int64_t coeff = out->coeffs(i);
195 if (!RefIsPositive(var)) return false;
196 terms.push_back({var, coeff});
197 }
198
199 // Add factor * to_add and check first kind of overflow.
200 const int to_add_size = to_add.linear().vars().size();
201 for (int i = 0; i < to_add_size; ++i) {
202 const int var = to_add.linear().vars(i);
203 const int64_t coeff = to_add.linear().coeffs(i);
204 if (!RefIsPositive(var)) return false;
205 terms.push_back({var, CapProd(coeff, factor)});
206 if (AtMinOrMaxInt64(terms.back().second)) return false;
207 }
208
209 // Merge terms, return false if we get an overflow here.
210 if (!SortAndMergeTerms(&terms)) return false;
211
212 // Copy terms.
213 out->clear_vars();
214 out->clear_coeffs();
215 for (const auto [var, coeff] : terms) {
216 out->add_vars(var);
217 out->add_coeffs(coeff);
218 }
219
220 // Write new rhs. We want to be exact during the multiplication. Note that in
221 // practice this domain is fixed, so this will always be the case.
222 bool exact = false;
223 Domain offset = ReadDomainFromProto(to_add.linear());
224 offset = offset.MultiplicationBy(factor, &exact);
225 CHECK(exact);
226
227 const Domain rhs = ReadDomainFromProto(*out);
228 FillDomainInProto(rhs.AdditionWith(offset), out);
229 return true;
230}

232bool SubstituteVariable(int var, int64_t var_coeff_in_definition,
233 const ConstraintProto& definition,
234 ConstraintProto* ct) {
235 CHECK(RefIsPositive(var));
236
237 // Get the coefficient of var in the constraint.
238 // We assume positive reference here (it should always be the case now).
239 // If we don't find var, we abort.
240 int64_t var_coeff = 0;
241 const int initial_ct_size = ct->linear().vars().size();
242 for (int i = 0; i < initial_ct_size; ++i) {
243 const int ref = ct->linear().vars(i);
244 if (!RefIsPositive(ref)) return false;
245 if (ref == var) {
246 // If var appear multiple time, we add all its coefficients.
247 var_coeff += ct->linear().coeffs(i);
248 }
249 }
250 if (var_coeff == 0) return false;
251
252 CHECK_EQ(std::abs(var_coeff_in_definition), 1);
253 const int64_t factor = var_coeff_in_definition > 0 ? -var_coeff : var_coeff;
254 return AddLinearConstraintMultiple(factor, definition, ct);
255}

262void ActivityBoundHelper::AddAtMostOne(absl::Span<const int> amo) {
263 int num_skipped = 0;
264 const int complexity_limit = 50;
265 for (const int literal : amo) {
266 const Index i = IndexFromLiteral(literal);
267 if (i >= amo_indices_.size()) amo_indices_.resize(i + 1);
268 if (amo_indices_[i].size() >= complexity_limit) ++num_skipped;
269 }
270 if (num_skipped + 1 >= amo.size()) return;
271
272 // Add it.
273 const int unique_index = num_at_most_ones_++;
274 for (const int literal : amo) {
275 const Index i = IndexFromLiteral(literal);
276 if (amo_indices_[i].size() < complexity_limit) {
277 amo_indices_[i].push_back(unique_index);
278 }
279 }
280}

283void ActivityBoundHelper::AddAllAtMostOnes(const CpModelProto& proto) {
284 for (const ConstraintProto& ct : proto.constraints()) {
285 const auto type = ct.constraint_case();
286 if (type == ConstraintProto::kAtMostOne) {
287 AddAtMostOne(ct.at_most_one().literals());
288 } else if (type == ConstraintProto::kExactlyOne) {
289 AddAtMostOne(ct.exactly_one().literals());
290 } else if (type == ConstraintProto::kBoolAnd) {
291 if (ct.enforcement_literal().size() == 1) {
292 const int a = ct.enforcement_literal(0);
293 for (const int b : ct.bool_and().literals()) {
294 // a => b same as amo(a, not(b)).
295 AddAtMostOne({a, NegatedRef(b)});
296 }
297 }
298 }

302int64_t ActivityBoundHelper::ComputeActivity(
303 bool compute_min, absl::Span<const std::pair<int, int64_t>> terms,
304 std::vector<std::array<int64_t, 2>>* conditional) {
305 tmp_terms_.clear();
306 tmp_terms_.reserve(terms.size());
307 int64_t offset = 0;
308 for (auto [lit, coeff] : terms) {
309 if (compute_min) coeff = -coeff; // Negate.
310 if (coeff >= 0) {
311 tmp_terms_.push_back({lit, coeff});
312 } else {
313 // l is the same as 1 - (1 - l)
314 tmp_terms_.push_back({NegatedRef(lit), -coeff});
315 offset += coeff;
316 }
317 }
318 const int64_t internal_result =
319 ComputeMaxActivityInternal(tmp_terms_, conditional);
320
321 // Correct everything.
322 if (conditional != nullptr) {
323 const int num_terms = terms.size();
324 for (int i = 0; i < num_terms; ++i) {
325 if (tmp_terms_[i].first != terms[i].first) {
326 // The true/false meaning is swapped
327 std::swap((*conditional)[i][0], (*conditional)[i][1]);
328 }
329 (*conditional)[i][0] += offset;
330 (*conditional)[i][1] += offset;
331 if (compute_min) {
332 (*conditional)[i][0] = -(*conditional)[i][0];
333 (*conditional)[i][1] = -(*conditional)[i][1];
334 }
335 }
336 }
337 if (compute_min) return -(offset + internal_result);
338 return offset + internal_result;
339}

341// Use trivial heuristic for now:
342// - Sort by decreasing coeff.
343// - If belong to a chosen part, use it.
344// - If not, choose biggest part left. TODO(user): compute sum of coeff in part?
345void ActivityBoundHelper::PartitionIntoAmo(
346 absl::Span<const std::pair<int, int64_t>> terms) {
347 amo_sums_.clear();
348
349 const int num_terms = terms.size();
350 to_sort_.clear();
351 to_sort_.reserve(num_terms);
352 for (int i = 0; i < num_terms; ++i) {
353 const Index index = IndexFromLiteral(terms[i].first);
354 const int64_t coeff = terms[i].second;
355 if (index < amo_indices_.size()) {
356 for (const int a : amo_indices_[index]) {
357 amo_sums_[a] += coeff;
358 }
359 }
360 to_sort_.push_back({terms[i].second, i});
361 }
362 std::sort(to_sort_.begin(), to_sort_.end(), std::greater<>());
363
364 int num_parts = 0;
365 partition_.resize(num_terms);
366 used_amo_to_dense_index_.clear();
367 for (int i = 0; i < num_terms; ++i) {
368 const int original_i = to_sort_[i].second;
369 const Index index = IndexFromLiteral(terms[original_i].first);
370 const int64_t coeff = terms[original_i].second;
371 int best = -1;
372 int64_t best_sum = 0;
373 bool done = false;
374 if (index < amo_indices_.size()) {
375 for (const int a : amo_indices_[index]) {
376 const auto it = used_amo_to_dense_index_.find(a);
377 if (it != used_amo_to_dense_index_.end()) {
378 partition_[original_i] = it->second;
379 done = true;
380 break;
381 }
382
383 const int64_t sum_left = amo_sums_[a];
384 amo_sums_[a] -= coeff;
385 if (sum_left > best_sum) {
386 best_sum = sum_left;
387 best = a;
388 }
389 }
390 }
391 if (done) continue;
392
393 // New element.
394 if (best == -1) {
395 partition_[original_i] = num_parts++;
396 } else {
397 used_amo_to_dense_index_[best] = num_parts;
398 partition_[original_i] = num_parts;
399 ++num_parts;
400 }
401 }
402 for (const int p : partition_) CHECK_LT(p, num_parts);
403 CHECK_LE(num_parts, num_terms);
404}

406// Similar algo as above for this simpler case.
407std::vector<absl::Span<const int>>
408ActivityBoundHelper::PartitionLiteralsIntoAmo(absl::Span<const int> literals) {
409 amo_sums_.clear();
410 for (const int ref : literals) {
411 const Index index = IndexFromLiteral(ref);
412 if (index < amo_indices_.size()) {
413 for (const int a : amo_indices_[index]) {
414 amo_sums_[a] += 1;
415 }
416 }
417 }
418
419 int num_parts = 0;
420 const int num_literals = literals.size();
421 partition_.resize(num_literals);
422 used_amo_to_dense_index_.clear();
423 for (int i = 0; i < literals.size(); ++i) {
424 const Index index = IndexFromLiteral(literals[i]);
425 int best = -1;
426 int64_t best_sum = 0;
427 bool done = false;
428 if (index < amo_indices_.size()) {
429 for (const int a : amo_indices_[index]) {
430 const auto it = used_amo_to_dense_index_.find(a);
431 if (it != used_amo_to_dense_index_.end()) {
432 partition_[i] = it->second;
433 done = true;
434 break;
435 }
436
437 const int64_t sum_left = amo_sums_[a];
438 amo_sums_[a] -= 1;
439 if (sum_left > best_sum) {
440 best_sum = sum_left;
441 best = a;
442 }
443 }
444 }
445 if (done) continue;
446
447 // New element.
448 if (best != -1) {
449 used_amo_to_dense_index_[best] = num_parts;
450 }
451 partition_[i] = num_parts++;
452 }
453
454 // We have the partition, lets construct the spans now.
455 part_to_literals_.ResetFromFlatMapping(partition_, literals);
456 DCHECK_EQ(part_to_literals_.size(), num_parts);
457 return part_to_literals_.AsVectorOfSpan();
458}

460bool ActivityBoundHelper::IsAmo(absl::Span<const int> literals) {
461 amo_sums_.clear();
462 for (int i = 0; i < literals.size(); ++i) {
463 bool has_max_size = false;
464 const Index index = IndexFromLiteral(literals[i]);
465 if (index >= amo_indices_.size()) return false;
466 for (const int a : amo_indices_[index]) {
467 if (amo_sums_[a]++ == i) has_max_size = true;
468 }
469 if (!has_max_size) return false;
470 }
471 return true;
472}

474int64_t ActivityBoundHelper::ComputeMaxActivityInternal(
475 absl::Span<const std::pair<int, int64_t>> terms,
476 std::vector<std::array<int64_t, 2>>* conditional) {
477 PartitionIntoAmo(terms);
478
479 // Compute the max coefficient in each partition.
480 const int num_terms = terms.size();
481 max_by_partition_.assign(num_terms, 0);
482 second_max_by_partition_.assign(num_terms, 0);
483 for (int i = 0; i < num_terms; ++i) {
484 const int p = partition_[i];
485 const int64_t coeff = terms[i].second;
486 if (coeff >= max_by_partition_[p]) {
487 second_max_by_partition_[p] = max_by_partition_[p];
488 max_by_partition_[p] = coeff;
489 } else if (coeff > second_max_by_partition_[p]) {
490 second_max_by_partition_[p] = coeff;
491 }
492 }
493
494 // Once we have this, we can compute bound.
495 int64_t max_activity = 0;
496 for (int p = 0; p < partition_.size(); ++p) {
497 max_activity += max_by_partition_[p];
498 }
499 if (conditional != nullptr) {
500 conditional->resize(num_terms);
501 for (int i = 0; i < num_terms; ++i) {
502 const int64_t coeff = terms[i].second;
503 const int p = partition_[i];
504 const int64_t max_used = max_by_partition_[p];
505
506 // We have two cases depending if coeff was the maximum in its part or
507 // not.
508 if (coeff == max_used) {
509 // Use the second max.
510 (*conditional)[i][0] =
511 max_activity - max_used + second_max_by_partition_[p];
512 (*conditional)[i][1] = max_activity;
513 } else {
514 // The max is still there, no change at 0 but change for 1.
515 (*conditional)[i][0] = max_activity;
516 (*conditional)[i][1] = max_activity - max_used + coeff;
517 }
518 }
519 }
520 return max_activity;
521}

532bool ActivityBoundHelper::PresolveEnforcement(
533 absl::Span<const int> refs, ConstraintProto* ct,
534 absl::flat_hash_set<int>* literals_at_true) {
535 if (ct->enforcement_literal().empty()) return true;
536
537 literals_at_true->clear();
538 triggered_amo_.clear();
539 int new_size = 0;
540 for (int i = 0; i < ct->enforcement_literal().size(); ++i) {
541 const int ref = ct->enforcement_literal(i);
542 if (literals_at_true->contains(ref)) continue; // Duplicate.
543 if (literals_at_true->contains(NegatedRef(ref))) return false; // False.
544 literals_at_true->insert(ref);
545
546 // If a previous enforcement literal implies this one, we can skip it.
547 //
548 // Tricky: We need to do that before appending the amo containing ref in
549 // case an amo contains both ref and not(ref).
550 // TODO(user): Ideally these amo should not be added to this class.
551 if (AppearInTriggeredAmo(NegatedRef(ref))) continue;
552
553 const Index index = IndexFromLiteral(ref);
554 if (index < amo_indices_.size()) {
555 for (const int a : amo_indices_[index]) {
556 // If some other literal is at one in this amo, literal must be false,
557 // and so the constraint cannot be enforced.
558 const auto [_, inserted] = triggered_amo_.insert(a);
559 if (!inserted) return false;
560 }
561 }
562
563 // Keep this enforcement.
564 ct->set_enforcement_literal(new_size++, ref);
565 }
566 ct->mutable_enforcement_literal()->Truncate(new_size);
567
568 for (const int ref : refs) {
569 // Skip already fixed.
570 if (literals_at_true->contains(ref)) continue;
571 if (literals_at_true->contains(NegatedRef(ref))) continue;
572 for (const int to_test : {ref, NegatedRef(ref)}) {
573 const Index index = IndexFromLiteral(to_test);
574 if (index < amo_indices_.size()) {
575 for (const int a : amo_indices_[index]) {
576 if (triggered_amo_.contains(a)) {
577 // If some other literal is at one in this amo,
578 // literal must be false.
579 if (literals_at_true->contains(to_test)) return false;
580 literals_at_true->insert(NegatedRef(to_test));
581 break;
582 }
583 }
584 }
585 }
586 }
587
588 return true;
589}

591int ActivityBoundHelper::RemoveEnforcementThatMakesConstraintTrivial(
592 absl::Span<const std::pair<int, int64_t>> boolean_terms,
593 const Domain& other_terms, const Domain& rhs, ConstraintProto* ct) {
594 tmp_set_.clear();
595 for (const int enf_lit : ct->enforcement_literal()) {
596 const Index negated_index = IndexFromLiteral(NegatedRef(enf_lit));
597 if (negated_index >= amo_indices_.size()) continue;
598 if (amo_indices_[negated_index].empty()) continue;
599
600 triggered_amo_.clear();
601 triggered_amo_.insert(amo_indices_[negated_index].begin(),
602 amo_indices_[negated_index].end());
603
604 // Compute min_max activity when enf_lit is false.
605 int64_t min_activity = 0;
606 int64_t max_activity = 0;
607 for (const auto [ref, coeff] : boolean_terms) {
608 // This is not supposed to happen after PresolveEnforcement(), so we
609 // just abort in this case.
610 if (ref == enf_lit || ref == NegatedRef(enf_lit)) break;
611
612 const bool is_true = AppearInTriggeredAmo(NegatedRef(ref));
613 const bool is_false = AppearInTriggeredAmo(ref);
614
615 // Similarly, this is not supposed to happen after PresolveEnforcement().
616 if (is_true && is_false) break;
617
618 if (is_false) continue;
619 if (is_true) {
620 min_activity += coeff;
621 max_activity += coeff;
622 continue;
623 }
624 if (coeff > 0) {
625 max_activity += coeff;
626 } else {
627 min_activity += coeff;
628 }
629 }
630
631 if (Domain(min_activity, max_activity)
632 .AdditionWith(other_terms)
633 .IsIncludedIn(rhs)) {
634 tmp_set_.insert(enf_lit);
635 }
636 }
637
638 if (!tmp_set_.empty()) {
639 int new_size = 0;
640 for (const int ref : ct->enforcement_literal()) {
641 if (tmp_set_.contains(ref)) continue;
642 ct->set_enforcement_literal(new_size++, ref);
643 }
644 const int old_size = ct->enforcement_literal().size();
645 ct->mutable_enforcement_literal()->Truncate(new_size);
646 return old_size - new_size;
647 }
648 return 0;
649}

651void ClauseWithOneMissingHasher::RegisterClause(int c,
652 absl::Span<const int> clause) {
653 uint64_t hash = 0;
654 for (const int ref : clause) {
655 const Index index = IndexFromLiteral(ref);
656 while (index >= literal_to_hash_.size()) {
657 // We use random value for a literal hash.
658 literal_to_hash_.push_back(absl::Uniform<uint64_t>(random_));
659 }
660 hash ^= literal_to_hash_[index];
661 }
662
663 if (c >= clause_to_hash_.size()) clause_to_hash_.resize(c + 1, 0);
664 clause_to_hash_[c] = hash;

681bool FindSingleLinearDifference(const LinearConstraintProto& lin1,
682 const LinearConstraintProto& lin2, int* var1,
683 int64_t* coeff1, int* var2, int64_t* coeff2) {
684 const int size = lin1.vars().size();
685 CHECK_EQ(size, lin2.vars().size());
686 *coeff1 = 0;
687 *coeff2 = 0;
688 int i = 0;
689 int j = 0;
690 while (i < size || j < size) {
691 // Note that we can't have both undefined or the loop would have exited.
692 const int v1 = i < size ? lin1.vars(i) : std::numeric_limits<int>::max();
693 const int v2 = j < size ? lin2.vars(j) : std::numeric_limits<int>::max();
694
695 // Same term, continue.
696 if (v1 == v2 && lin1.coeffs(i) == lin2.coeffs(j)) {
697 ++i;
698 ++j;
699 continue;
700 }
701
702 // We have a diff.
703 // term i not in lin2.
704 if (v1 < v2) {
705 if (*coeff1 != 0) return false; // Returns if second diff.
706 *var1 = v1;
707 *coeff1 = lin1.coeffs(i);
708 ++i;
709 continue;
710 }
711
712 // term j not in lin1.
713 if (v1 > v2) {
714 if (*coeff2 != 0) return false; // Returns if second diff.
715 *var2 = v2;
716 *coeff2 = lin2.coeffs(j);
717 ++j;
718 continue;
719 }
720
721 // Coeff differ. Returns if we had a diff previously.
722 if (*coeff1 != 0 || *coeff2 != 0) return false;
723 *var1 = v1;
724 *var2 = v2;
725 *coeff1 = lin1.coeffs(i);
726 *coeff2 = lin2.coeffs(j);
727 ++i;
728 ++j;
729 }
730
731 return *coeff1 != 0 && *coeff2 != 0;
732}
