39class OrthogonalPackingResult {
40 public:
41 explicit OrthogonalPackingResult() : result_(Status::UNKNOWN) {}
42 enum class Status {
43 INFEASIBLE,
44 FEASIBLE,
45 UNKNOWN,
46 };
47
48 Status GetResult() const { return result_; }
49
50 struct Item {
51 // Index of the item on the original sizes_x/sizes_y input.
52 int index;
53 // New size for item of index `i` which is smaller or equal to the initial
54 // size. The subproblem remain infeasible if every item is shrinked to its
55 // new size.
56 IntegerValue size_x;
57 IntegerValue size_y;
58 };
59 const std::vector<Item>& GetItemsParticipatingOnConflict() const {
60 return items_participating_on_conflict_;
61 }
62
63 bool HasSlack() const { return slack_ > IntegerValue(0); }
64
65 enum class Coord { kCoordX, kCoordY };
66 // Use an eventual slack to reduce the size of item corresponding to the
67 // `i`-th element on GetItemsParticipatingOnConflict(). It will not use any
68 // slack to reduce it beyond lower_bound. This is a no-op if HasSlack() is
69 // false.
70 bool TryUseSlackToReduceItemSize(int i, Coord coord,
71 IntegerValue lower_bound = 0);
72
73 // If *this is identical or not easily comparable to other, returns false.
74 bool IsBetterThan(const OrthogonalPackingResult& other) const {
75 if (result_ == Status::UNKNOWN && other.result_ == Status::UNKNOWN) {
76 return false;
77 }
78 if (other.result_ == Status::UNKNOWN) {
79 return true;
80 }
81 if (result_ == Status::UNKNOWN) {
82 return false;
83 }
84 if (other.result_ == Status::FEASIBLE) {
85 CHECK(result_ != Status::INFEASIBLE);
86 return result_ == Status::FEASIBLE;
87 }
88
89 // other.result_ == Status::INFEASIBLE
90 CHECK(result_ == Status::INFEASIBLE);
91 if (other.items_participating_on_conflict_.size() <
92 items_participating_on_conflict_.size()) {
93 return false;
94 }
95 if (other.items_participating_on_conflict_.size() >
96 items_participating_on_conflict_.size()) {
97 return true;
98 }
99
100 IntegerValue total_area_this = 0;
101 IntegerValue total_area_other = 0;
102 for (int i = 0; i < items_participating_on_conflict_.size(); ++i) {
103 total_area_this += items_participating_on_conflict_[i].size_x *
104 items_participating_on_conflict_[i].size_y;
105 total_area_other += other.items_participating_on_conflict_[i].size_x *
106 other.items_participating_on_conflict_[i].size_y;
107 }
108 return total_area_this - slack_ < total_area_other - other.slack_;
109 }
110
111 private:
112 friend class OrthogonalPackingInfeasibilityDetector;
113
114 explicit OrthogonalPackingResult(Status result) : result_(result) {}
115
116 enum class ConflictType {
117 NO_CONFLICT,
118 TRIVIAL,
119 PAIRWISE,
120 DFF_F0,
121 DFF_F2,
122 };
123
124 Status result_;
125 ConflictType conflict_type_ = ConflictType::NO_CONFLICT;
126 IntegerValue slack_ = 0;
127 std::vector<Item> items_participating_on_conflict_;
128};

198// If we have a container of size `C` and a parameter `k` taking values in
199// [0, C/2], the Dual Feasible Function often named `f_0^k(x)` is equivalent to
200// the operation of removing all values of size less than `k`, and symmetrically
201// increasing to `C` the size of the large values. It is defined as:
202//
203// / C, if x > C - k,
204// f_0^k(x) = | x, if k <= x <= C - k,
205// \ 0, if x < k.
206//
207// This is a Maximal DFF. See for example [1] for some discussion about it.
208//
209// [1] Clautiaux, François, Cláudio Alves, and José Valério de Carvalho. "A
210// survey of dual-feasible and superadditive functions." Annals of Operations
211// Research 179 (2010): 317-342.
212class DualFeasibleFunctionF0 {
213 public:
214 DualFeasibleFunctionF0(IntegerValue max_x, IntegerValue k)
215 : k_(k), max_x_(max_x) {
216 DCHECK_GE(k_, 0);
217 DCHECK_LE(2 * k_, max_x_);
218 }
219
220 // `x` must be in [0, max_x].
221 IntegerValue operator()(IntegerValue x) const {
222 DCHECK_GE(x, 0);
223 DCHECK_LE(x, max_x_);
224 if (x > max_x_ - k_) {
225 return max_x_;
226 } else if (x < k_) {
227 return 0;
228 } else {
229 return x;
230 }
231 }
232
233 // Return the lowest integer y so that Dff(x) >= y.
234 // y must be in [0, Dff(max_x)].
235 IntegerValue LowestInverse(IntegerValue x) const {
236 DCHECK_GE(x, 0);
237 DCHECK_LE(x, max_x_);
238 if (x > max_x_ - k_) {
239 return max_x_ - k_ + 1;
240 } else if (x == 0) {
241 return 0;
242 } else if (x < k_) {
243 return k_;
244 } else {
245 return x;
246 }
247 }
248
249 private:
250 const IntegerValue k_;
251 const IntegerValue max_x_;
252};

254// Dual Feasible Function based on rounding. Called f_2 on [1].
255//
256// The `f_2^k(x)` function for an integer x in [0, C] and a parameter `k` taking
257// values in [0, C/2] is defined as:
258//
259// / 2 * [ floor(C / k) - floor( (C - x) / k) ], if x > C / 2,
260// f_2^k(x) = | floor(C / k), if k = C / 2,
261// \ floor(x / k), if x < C / 2.
262//
263// This function is a Maximal Dual Feasible Function. Ie., it satisfies:
264// - f_2 is nondecreasing,
265// - f_2 is superadditive, i.e., f_2(x) + f_2(y) <= f_2(x + y),
266// - f_2 is symmetric, i.e., f_2(x) + f_2(C - x) = f_2(C),
267// - f_2(0) = 0.
268//
269// [1] Carlier, Jacques, François Clautiaux, and Aziz Moukrim. "New reduction
270// procedures and lower bounds for the two-dimensional bin packing problem with
271// fixed orientation." Computers & Operations Research 34.8 (2007): 2223-2250.
272class RoundingDualFeasibleFunction {
273 public:
274 // `max_x` must fit in a uint16_t and `k` in [0, max_x/2].
275 RoundingDualFeasibleFunction(IntegerValue max_x, IntegerValue k)
276 : div_(k.value()),
277 max_x_(max_x),
278 c_k_(div_.DivideByDivisor(max_x_.value())),
279 k_(k) {
280 DCHECK_GT(k, 0);
281 DCHECK_LE(2 * k, max_x_);
282 DCHECK_LE(max_x_, std::numeric_limits<uint16_t>::max());
283 }
284
285 // `x` must be in [0, max_x].
286 IntegerValue operator()(IntegerValue x) const {
287 DCHECK_GE(x, 0);
288 DCHECK_LE(x, max_x_);
289
290 if (2 * x > max_x_) {
291 return 2 * (c_k_ - div_.DivideByDivisor(max_x_.value() - x.value()));
292 } else if (2 * x == max_x_) {
293 return c_k_;
294 } else {
295 return 2 * div_.DivideByDivisor(x.value());
296 }
297 }
298
299 // Return the lowest integer y so that Dff(x) >= y.
300 // y must be in [0, Dff(max_x)].
301 IntegerValue LowestInverse(IntegerValue y) const;
302
303 private:
304 const QuickSmallDivision div_;
305 const IntegerValue max_x_;
306 const IntegerValue c_k_;
307 const IntegerValue k_;
308};

310// Same as above for k = 2^log2_k.
311class RoundingDualFeasibleFunctionPowerOfTwo {
312 public:
313 RoundingDualFeasibleFunctionPowerOfTwo(IntegerValue max_x,
314 IntegerValue log2_k)
315 : log2_k_(log2_k), max_x_(max_x), c_k_(max_x_ >> log2_k_) {
316 DCHECK_GE(log2_k_, 0);
317 DCHECK_LT(log2_k_, 63);
318 DCHECK_LE(2 * (1 << log2_k), max_x_);
319 DCHECK_LE(max_x_, std::numeric_limits<int64_t>::max() / 2);
320 }
321
322 IntegerValue operator()(IntegerValue x) const {
323 DCHECK_GE(x, 0);
324 DCHECK_LE(x, max_x_);
325
326 if (2 * x > max_x_) {
327 return 2 * (c_k_ - ((max_x_ - x) >> log2_k_));
328 } else if (2 * x == max_x_) {
329 return c_k_;
330 } else {
331 return 2 * (x >> log2_k_);
332 }
333 }
334
335 // Return the lowest integer y so that Dff(x) >= y.
336 // y must be in [0, Dff(max_x)].
337 IntegerValue LowestInverse(IntegerValue y) const;
338
339 private:
340 const IntegerValue log2_k_;
341 const IntegerValue max_x_;
342 const IntegerValue c_k_;
343};

345// Using our definition for the inverse, composition produces a valid
346// DFF with a valid inverse. This class defines f2(f0(x)).
347class DFFComposedF2F0 {
348 public:
349 DFFComposedF2F0(IntegerValue max_x, IntegerValue k_f0, IntegerValue k_f2)
350 : f0_(max_x, k_f0), f2_(max_x, k_f2) {}
351
352 IntegerValue operator()(IntegerValue x) const { return f2_(f0_(x)); }
353
354 IntegerValue LowestInverse(IntegerValue x) const {
355 return f0_.LowestInverse(f2_.LowestInverse(x));
356 }

358 private:
359 const DualFeasibleFunctionF0 f0_;
360 const RoundingDualFeasibleFunction f2_;
361};
