39absl::Status CheckIdsRangeAndStrictlyIncreasing2(
40 absl::Span<const int64_t> ids) {
41 int64_t previous{-1};
42 for (int i = 0; i < ids.size(); previous = ids[i], ++i) {
43 if (ids[i] < 0 || ids[i] == std::numeric_limits<int64_t>::max()) {
44 return util::InvalidArgumentErrorBuilder()
45 << "Expected ids to be nonnegative and not max(int64_t) but at "
46 "index "
47 << i << " found id: " << ids[i];
48 }
49 if (ids[i] <= previous) {
50 return util::InvalidArgumentErrorBuilder()
51 << "Expected ids to be strictly increasing, but at index " << i
52 << " found id: " << ids[i] << " and at previous index " << i - 1
53 << " found id: " << ids[i - 1];
54 }
55 }
56 return absl::OkStatus();
57}

71IdNameBiMap& IdNameBiMap::operator=(const IdNameBiMap& other) {
72 if (&other == this) {
73 return *this;
74 }
75 next_free_id_ = other.next_free_id_;
76 id_to_name_ = other.id_to_name_;
77 if (!other.nonempty_name_to_id_.has_value()) {
78 nonempty_name_to_id_ = std::nullopt;
79 } else {
80 nonempty_name_to_id_.emplace();
81 for (const auto& [id, name] : id_to_name_) {
82 if (!name.empty()) {
83 const auto [it, success] =
84 nonempty_name_to_id_->insert({absl::string_view(name), id});
85 CHECK(success); // CHECK is OK, other cannot have duplicate names and
86 // non nullopt nonempty_name_to_id_.
87 }
88 }
89 }
90
91 return *this;
92}

94absl::Status IdNameBiMap::BulkUpdate(
95 absl::Span<const int64_t> deleted_ids, absl::Span<const int64_t> new_ids,
96 const absl::Span<const std::string* const> names) {
97 RETURN_IF_ERROR(internal::CheckIdsRangeAndStrictlyIncreasing2(deleted_ids))
98 << "invalid deleted ids";
99 RETURN_IF_ERROR(internal::CheckIdsRangeAndStrictlyIncreasing2(new_ids))
100 << "invalid new ids";
101 if (!names.empty() && names.size() != new_ids.size()) {
102 return util::InvalidArgumentErrorBuilder()
103 << "names had size " << names.size()
104 << " but should either be empty of have size matching new_ids which "
105 "has size "
106 << new_ids.size();
107 }
108 for (const int64_t id : deleted_ids) {
109 RETURN_IF_ERROR(Erase(id));
110 }
111 for (int i = 0; i < new_ids.size(); ++i) {
112 RETURN_IF_ERROR(
113 Insert(new_ids[i], names.empty() ? std::string{} : *names[i]));
114 }
115 return absl::OkStatus();
116}

125absl::StatusOr<ModelSummary> ModelSummary::Create(const ModelProto& model,
126 const bool check_names) {
127 ModelSummary summary(check_names);
128 RETURN_IF_ERROR(summary.variables.BulkUpdate({}, model.variables().ids(),
129 model.variables().names()))
130 << "Model.variables are invalid";
131 RETURN_IF_ERROR(summary.linear_constraints.BulkUpdate(
132 {}, model.linear_constraints().ids(), model.linear_constraints().names()))
133 << "Model.linear_constraints are invalid";
134 RETURN_IF_ERROR(internal::UpdateBiMapFromMappedConstraints(
135 {}, model.quadratic_constraints(), summary.quadratic_constraints))
136 << "Model.quadratic_constraints are invalid";
137 RETURN_IF_ERROR(internal::UpdateBiMapFromMappedConstraints(
138 {}, model.sos1_constraints(), summary.sos1_constraints))
139 << "Model.sos1_constraints are invalid";
140 RETURN_IF_ERROR(internal::UpdateBiMapFromMappedConstraints(
141 {}, model.sos2_constraints(), summary.sos2_constraints))
142 << "Model.sos2_constraints are invalid";
143 return summary;
144}

146absl::Status ModelSummary::Update(const ModelUpdateProto& model_update) {
147 RETURN_IF_ERROR(variables.BulkUpdate(model_update.deleted_variable_ids(),
148 model_update.new_variables().ids(),
149 model_update.new_variables().names()))
150 << "invalid variables";
151 RETURN_IF_ERROR(linear_constraints.BulkUpdate(
152 model_update.deleted_linear_constraint_ids(),
153 model_update.new_linear_constraints().ids(),
154 model_update.new_linear_constraints().names()))
155 << "invalid linear constraints";
156 RETURN_IF_ERROR(internal::UpdateBiMapFromMappedConstraints(
157 model_update.quadratic_constraint_updates().deleted_constraint_ids(),
158 model_update.quadratic_constraint_updates().new_constraints(),
159 quadratic_constraints))
160 << "invalid quadratic constraints";
161 RETURN_IF_ERROR(internal::UpdateBiMapFromMappedConstraints(
162 model_update.sos1_constraint_updates().deleted_constraint_ids(),
163 model_update.sos1_constraint_updates().new_constraints(),
164 sos1_constraints))
165 << "invalid sos1 constraints";
166 RETURN_IF_ERROR(internal::UpdateBiMapFromMappedConstraints(
167 model_update.sos2_constraint_updates().deleted_constraint_ids(),
168 model_update.sos2_constraint_updates().new_constraints(),
169 sos2_constraints))
170 << "invalid sos2 constraints";
171 return absl::OkStatus();
172}
