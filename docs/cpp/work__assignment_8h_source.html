48class ProtoLiteral {
49 public:
50 ProtoLiteral() = default;
51 ProtoLiteral(int var, IntegerValue lb) : proto_var_(var), lb_(lb) {}
52 ProtoLiteral Negated() const {
53 return ProtoLiteral(NegatedRef(proto_var_), -lb_ + 1);
54 }
55 bool operator==(const ProtoLiteral& other) const {
56 return proto_var_ == other.proto_var_ && lb_ == other.lb_;
57 }
58 bool operator!=(const ProtoLiteral& other) const { return !(*this == other); }
59 Literal Decode(CpModelMapping*, IntegerEncoder*) const;
60 static std::optional<ProtoLiteral> Encode(Literal, CpModelMapping*,
61 IntegerEncoder*);
62
63 private:
64 IntegerLiteral DecodeInteger(CpModelMapping*) const;
65 static std::optional<ProtoLiteral> EncodeInteger(IntegerLiteral,
66 CpModelMapping*);
67
68 int proto_var_ = std::numeric_limits<int>::max();
69 IntegerValue lb_ = kMaxIntegerValue;
70};

72// ProtoTrail acts as an intermediate datastructure that can be synced
73// with the shared tree and the local Trail as appropriate.
74// It's intended that you sync a ProtoTrail with the tree on restart or when
75// a subtree is closed, and with the local trail after propagation.
76// Specifically it stores objective lower bounds, and literals that have been
77// branched on and later proven to be implied by the prior decisions (i.e. they
78// can be enqueued at this level).
79// TODO(user): It'd be good to store an earlier level at which
80// implications may be propagated.
81class ProtoTrail {
82 public:
83 // Adds a new assigned level to the trail.
84 void PushLevel(const ProtoLiteral& decision, IntegerValue objective_lb,
85 int node_id);
86
87 // Asserts that the decision at `level` is implied by earlier decisions.
88 void SetLevelImplied(int level);
89
90 // Clear the trail, removing all levels.
91 void Clear();
92
93 // Set a lower bound on the objective at level.
94 void SetObjectiveLb(int level, IntegerValue objective_lb);
95
96 // Returns the maximum decision level stored in the trail.
97 int MaxLevel() const { return decision_indexes_.size(); }
98
99 // Returns the decision assigned at `level`.
100 ProtoLiteral Decision(int level) const {
101 CHECK_GE(level, 1);
102 CHECK_LE(level, decision_indexes_.size());
103 return literals_[decision_indexes_[level - 1]];
104 }
105
106 // Returns the node ids for decisions and implications at `level`.
107 absl::Span<const int> NodeIds(int level) const;
108
109 // Returns literals which may be propagated at `level`, this does not include
110 // the decision.
111 absl::Span<const ProtoLiteral> Implications(int level) const;
112
113 IntegerValue ObjectiveLb(int level) const {
114 CHECK_GE(level, 1);
115 return level_to_objective_lbs_[level - 1];
116 }
117
118 absl::Span<const ProtoLiteral> Literals() const { return literals_; }
119
120 private:
121 // Parallel vectors encoding the literals and node ids on the trail.
122 std::vector<ProtoLiteral> literals_;
123 std::vector<int> node_ids_;
124
125 // The index in the literals_/node_ids_ vectors for the start of each level.
126 std::vector<int> decision_indexes_;
127
128 // The objective lower bound of each level.
129 std::vector<IntegerValue> level_to_objective_lbs_;
130};

132// Experimental thread-safe class for managing work assignments between workers.
133// This API is intended to investigate Graeme Gange & Peter Stuckey's proposal
134// "Scalable Parallelization of Learning Solvers".
135// The core idea of this implementation is that workers can maintain several
136// ProtoTrails, and periodically replace the "worst" one.
137// With 1 assignment per worker, this leads to something very similar to
138// Embarassingly Parallel Search.
139class SharedTreeManager {
140 public:
141 explicit SharedTreeManager(Model* model);
142 SharedTreeManager(const SharedTreeManager&) = delete;
143
144 int NumWorkers() const { return num_workers_; }
145 int NumNodes() const ABSL_LOCKS_EXCLUDED(mu_);
146
147 // Returns the number of splits each worker should propose this restart.
148 int SplitsToGeneratePerWorker() const;
149
150 // Syncs the state of path with the shared search tree.
151 // Clears `path` and returns false if the assigned subtree is closed or a
152 // restart has invalidated the path.
153 bool SyncTree(ProtoTrail& path) ABSL_LOCKS_EXCLUDED(mu_);
154
155 // Assigns a path prefix that the worker should explore.
156 void ReplaceTree(ProtoTrail& path);
157
158 // Asserts that the subtree in path up to `level` contains no improving
159 // solutions. Clears path.
160 void CloseTree(ProtoTrail& path, int level);
161
162 // Called by workers in order to split the shared tree.
163 // `path` may or may not be extended by one level, branching on `decision`.
164 void ProposeSplit(ProtoTrail& path, ProtoLiteral decision);
165
166 void Restart() {
167 absl::MutexLock l(&mu_);
168 RestartLockHeld();
169 }
170
171 private:
172 struct Node {
173 ProtoLiteral literal;
174 IntegerValue objective_lb = kMinIntegerValue;
175 Node* parent = nullptr;
176 std::array<Node*, 2> children = {nullptr, nullptr};
177 // A node's id is related to its index in `nodes_`, see `node_id_offset_`.
178 int id;
179 bool closed = false;
180 bool implied = false;
181 };
182 bool IsValid(const ProtoTrail& path) const ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
183 Node* GetSibling(Node* node) ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
184 void Split(std::vector<std::pair<Node*, int>>& nodes, ProtoLiteral lit)
185 ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
186 Node* MakeSubtree(Node* parent, ProtoLiteral literal)
187 ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
188 void ProcessNodeChanges() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
189 std::vector<std::pair<Node*, int>> GetAssignedNodes(const ProtoTrail& path)
190 ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
191 void AssignLeaf(ProtoTrail& path, Node* leaf)
192 ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
193 void RestartLockHeld() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
194 std::string ShortStatus() const ABSL_EXCLUSIVE_LOCKS_REQUIRED(mu_);
195
196 mutable absl::Mutex mu_;
197 const SatParameters& params_;
198 const int num_workers_;
199 SharedResponseManager* const shared_response_manager_;
200
201 // Stores the node id of the root, this is used to handle global restarts.
202 int node_id_offset_ ABSL_GUARDED_BY(mu_) = 0;
203
204 // Stores the nodes in the search tree.
205 std::deque<Node> nodes_ ABSL_GUARDED_BY(mu_);
206 std::vector<Node*> unassigned_leaves_ ABSL_GUARDED_BY(mu_);
207
208 // How many splits we should generate now to keep the desired number of
209 // leaves.
210 int num_splits_wanted_;
211
212 // We limit the total nodes generated per restart to cap the RAM usage and
213 // communication overhead. If we exceed this, we will restart the shared tree.
214 const int max_nodes_;
215 int num_leaves_assigned_ ABSL_GUARDED_BY(mu_) = 0;
216
217 // Temporary vectors used to maintain the state of the tree when nodes are
218 // closed and/or children are updated.
219 std::vector<Node*> to_close_ ABSL_GUARDED_BY(mu_);
220 std::vector<Node*> to_update_ ABSL_GUARDED_BY(mu_);
221
222 int64_t num_restarts_ ABSL_GUARDED_BY(mu_) = 0;
223 int64_t num_syncs_since_restart_ ABSL_GUARDED_BY(mu_) = 0;
224};
