37// A sparse symmetric double valued matrix over VariableIds.
38//
39// Note that the matrix is sparse in both:
40// * The IDs of the rows/columns (both VariableIds), stored as flat_hash_map.
41// * The entries with nonzero value.
42//
43// Getting/setting/clearing entries are O(1) operations. Getting a row of the
44// matrix runs in O(size of the row) if nothing has been deleted, and getting
45// all the rows runs in O(number of nonzero entries), even with deletions
46// (with deletions, accessing a particular row with many deletions may be slow).
47//
48// Implementation: The entries are stored in a
49// flat_hash_map<pair<VariableId, VariableId>, double> `values_` where for each
50// key, key.first <= key.second. Additionally, we maintain a
51// flat_hash_map<VariableId, vector<VariableId>> `related_variables_` that says
52// for each variable, which variables they have a nonzero term with. When a
53// coefficient is set to zero or a variable is deleted, we do not immediately
54// delete the data from values_ or related_variables_, we simply set the
55// coefficient to zero in values_. We track how many zeros are in values_, and
56// when more than some constant fraction of all entries are zero (see
57// kZerosCleanup in cc file), we clean up related_variables_ and values_ to
58// remove all the zeros. Iteration over the rows or total entries of the matrix
59// must check for zeros in values_ and skip these terms.
60//
61// Memory use:
62// * 3*8 bytes per nonzero plus hash capacity overhead for values_.
63// * 2*8 bytes per nonzero plus vector capacity overhead for
64// related_variables_.
65// * ~5*8 bytes per variable participating in any quadratic term; one heap
66// allocation per such variable.
67class SparseSymmetricMatrix {
68 public:
69 // Setting `value` to zero removes the value from the matrix.
70 // Returns true if `value` is different from the existing value in the matrix.
71 inline bool set(VariableId first, VariableId second, double value);
72
73 // Zero is returned if the value is not present.
74 inline double get(VariableId first, VariableId second) const;
75
76 // Zeros out all coefficients for this variable.
77 void Delete(VariableId variable);
78
79 // Returns the variables that have nonzero entries with `variable`.
80 //
81 // The return order is deterministic but not defined.
82 // TODO(b/233630053): expose an iterator based API to avoid making a copy.
83 std::vector<VariableId> RelatedVariables(VariableId variable) const;
84
85 // Returns the variable value pairs (x, c) where `variable` and x have nonzero
86 // coefficient c.
87 //
88 // The return order is deterministic but not defined.
89 // TODO(b/233630053): expose an iterator based API to avoid making a copy.
90 std::vector<std::pair<VariableId, double>> Terms(VariableId variable) const;
91
92 // Returns (x, y, c) tuples where variables x and y have nonzero coefficient
93 // c, and x <= y.
94 //
95 // The return order is non-deterministic and not defined.
96 // TODO(b/233630053): expose an iterator based API to avoid making a copy.
97 std::vector<std::tuple<VariableId, VariableId, double>> Terms() const;
98
99 // Removes all terms from the matrix.
100 void Clear();
101
102 // The number of (var, var) keys with nonzero value. Note that (x, y) and
103 // (y, x) are the same key.
104 int64_t nonzeros() const { return nonzeros_; }
105
106 // Visible for testing, do not depend on this.
107 int64_t compactions() const { return compactions_; }
108
109 // TODO(b/233630053): do not expose values_ directly, instead offer a way to
110 // iterate over all the nonzero entries.
111 // Warning: this map will contain zeros.
112 const absl::flat_hash_map<std::pair<VariableId, VariableId>, double>& values()
113 const {
114 return values_;
115 }
116
117 SparseDoubleMatrixProto Proto() const;
118
119 template <typename VarContainer>
120 inline SparseDoubleMatrixProto Update(
121 const VarContainer& variables, VariableId checkpoint, VariableId next_var,
122 const absl::flat_hash_set<std::pair<VariableId, VariableId>>& dirty)
123 const;
124
125 private:
126 inline std::pair<VariableId, VariableId> make_key(VariableId first,
127 VariableId second) const;
128 void CompactIfNeeded();
129
130 // The keys of values_ have key.first <= key.second.
131 absl::flat_hash_map<std::pair<VariableId, VariableId>, double> values_;
132 absl::flat_hash_map<VariableId, std::vector<VariableId>> related_variables_;
133
134 // The number of nonzero elements in values_.
135 int64_t nonzeros_ = 0;
136 int64_t compactions_ = 0;
137};

143std::pair<VariableId, VariableId> SparseSymmetricMatrix::make_key(
144 const VariableId first, const VariableId second) const {
145 return {std::min(first, second), std::max(first, second)};
146}

148bool SparseSymmetricMatrix::set(const VariableId first, const VariableId second,
149 const double value) {
150 const std::pair<VariableId, VariableId> key = make_key(first, second);
151 auto map_iter = values_.find(key);
152
153 if (map_iter == values_.end()) {
154 if (value == 0.0) {
155 return false;
156 }
157 related_variables_[first].push_back(second);
158 if (first != second) {
159 related_variables_[second].push_back(first);
160 }
161 values_[key] = value;
162 nonzeros_++;
163 return true;
164 } else {
165 if (map_iter->second == value) {
166 return false;
167 }
168 const double old_value = map_iter->second;
169 map_iter->second = value;
170 if (value == 0.0) {
171 nonzeros_--;
172 CompactIfNeeded();
173 } else if (old_value == 0.0) {
174 nonzeros_++;
175 }
176 return true;
177 }
178}

180double SparseSymmetricMatrix::get(const VariableId first,
181 const VariableId second) const {
182 return gtl::FindWithDefault(values_, make_key(first, second));
183}

185template <typename VarContainer>
186SparseDoubleMatrixProto SparseSymmetricMatrix::Update(
187 const VarContainer& variables, const VariableId checkpoint,
188 const VariableId next_var,
189 const absl::flat_hash_set<std::pair<VariableId, VariableId>>& dirty) const {
190 std::vector<std::pair<VariableId, VariableId>> updates;
191 for (const std::pair<VariableId, VariableId> pair : dirty) {
192 // If either variable has been deleted, don't add it.
193 if (variables.contains(pair.first) && variables.contains(pair.second)) {
194 updates.push_back(pair);
195 }
196 }
197
198 for (const VariableId v : MakeStrongIntRange(checkpoint, next_var)) {
199 if (related_variables_.contains(v)) {
200 // TODO(b/233630053): do not allocate here.
201 for (const VariableId other : RelatedVariables(v)) {
202 if (v <= other) {
203 updates.push_back({v, other});
204 } else if (other < checkpoint) {
205 updates.push_back({other, v});
206 }
207 }
208 }
209 }
210 absl::c_sort(updates);
211 SparseDoubleMatrixProto result;
212 for (const auto [row, col] : updates) {
213 result.add_row_ids(row.value());
214 result.add_column_ids(col.value());
215 result.add_coefficients(get(row, col));
216 }
217 return result;
218}
