66std::string ExtractName(const std::string& full_filename) {
67 std::string filename = std::string(file::Basename(full_filename));
68 // The order is important as '.pb.txt.gz' is a common suffix.
69 TryToRemoveSuffix("gz", &filename);
70 TryToRemoveSuffix("txt", &filename);
71 TryToRemoveSuffix("pb", &filename);
72 TryToRemoveSuffix("pbtxt", &filename);
73 TryToRemoveSuffix("proto", &filename);
74 TryToRemoveSuffix("prototxt", &filename);
75 TryToRemoveSuffix("textproto", &filename);
76 TryToRemoveSuffix("bin", &filename);
77 return filename;
78}

80bool LoadProblem(const std::string& filename, CpModelProto* cp_model) {
81 if (absl::EndsWith(filename, ".opb") ||
82 absl::EndsWith(filename, ".opb.bz2")) {
83 OpbReader reader;
84 LinearBooleanProblem problem;
85 if (!reader.Load(filename, &problem)) {
86 LOG(FATAL) << "Cannot load file '" << filename << "'.";
87 }
88 *cp_model = BooleanProblemToCpModelproto(problem);
89 } else if (absl::EndsWith(filename, ".cnf") ||
90 absl::EndsWith(filename, ".cnf.xz") ||
91 absl::EndsWith(filename, ".cnf.gz") ||
92 absl::EndsWith(filename, ".wcnf") ||
93 absl::EndsWith(filename, ".wcnf.xz") ||
94 absl::EndsWith(filename, ".wcnf.gz")) {
95 SatCnfReader reader(absl::GetFlag(FLAGS_wcnf_use_strong_slack));
96 if (!reader.Load(filename, cp_model)) {
97 LOG(FATAL) << "Cannot load file '" << filename << "'.";
98 }
99 } else {
100 LOG(INFO) << "Reading a CpModelProto.";
101 *cp_model = ReadFileToProtoOrDie<CpModelProto>(filename);
102 }
103 if (cp_model->name().empty()) {
104 cp_model->set_name(ExtractName(filename));
105 }
106 return true;
107}

109int Run() {
110 SatParameters parameters;
111 if (absl::GetFlag(FLAGS_input).empty()) {
112 LOG(FATAL) << "Please supply a data file with --input=";
113 }
114
115 // Parse the --params flag.
116 parameters.set_log_search_progress(true);
117 if (!absl::GetFlag(FLAGS_params).empty()) {
118 CHECK(google::protobuf::TextFormat::MergeFromString(
119 absl::GetFlag(FLAGS_params), &parameters))
120 << absl::GetFlag(FLAGS_params);
121 }
122
123 // Read the problem.
124 google::protobuf::Arena arena;
125 CpModelProto* cp_model =
126 google::protobuf::Arena::CreateMessage<CpModelProto>(&arena);
127 if (!LoadProblem(absl::GetFlag(FLAGS_input), cp_model)) {
128 CpSolverResponse response;
129 response.set_status(CpSolverStatus::MODEL_INVALID);
130 return EXIT_SUCCESS;
131 }
132
133 Model model;
134 model.Add(NewSatParameters(parameters));
135 const CpSolverResponse response = SolveCpModel(*cp_model, &model);
136
137 if (!absl::GetFlag(FLAGS_output).empty()) {
138 if (absl::EndsWith(absl::GetFlag(FLAGS_output), "txt")) {
139 CHECK_OK(file::SetTextProto(absl::GetFlag(FLAGS_output), response,
140 file::Defaults()));
141 } else {
142 CHECK_OK(file::SetBinaryProto(absl::GetFlag(FLAGS_output), response,
143 file::Defaults()));
144 }
145 }
146
147 // The SAT competition requires a particular exit code and since we don't
148 // really use it for any other purpose, we comply.
149 if (response.status() == CpSolverStatus::OPTIMAL) return 10;
150 if (response.status() == CpSolverStatus::FEASIBLE) return 10;
151 if (response.status() == CpSolverStatus::INFEASIBLE) return 20;
152 return EXIT_SUCCESS;
153}
