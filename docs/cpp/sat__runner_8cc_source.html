76std::string ExtractName(absl::string_view full_filename) {
77 std::string filename = std::string(file::Basename(full_filename));
78 // The order is important as '.pb.txt.gz' is a common suffix.
79 TryToRemoveSuffix("gz", &filename);
80 TryToRemoveSuffix("txt", &filename);
81 TryToRemoveSuffix("pb", &filename);
82 TryToRemoveSuffix("pbtxt", &filename);
83 TryToRemoveSuffix("proto", &filename);
84 TryToRemoveSuffix("prototxt", &filename);
85 TryToRemoveSuffix("textproto", &filename);
86 TryToRemoveSuffix("bin", &filename);
87 return filename;
88}

90bool LoadProblem(const std::string& filename, absl::string_view hint_file,
91 CpModelProto* cp_model) {
92 if (absl::EndsWith(filename, ".opb") ||
93 absl::EndsWith(filename, ".opb.bz2")) {
94 OpbReader reader;
95 LinearBooleanProblem problem;
96 if (!reader.Load(filename, &problem)) {
97 LOG(FATAL) << "Cannot load file '" << filename << "'.";
98 }
99 *cp_model = BooleanProblemToCpModelproto(problem);
100 } else if (absl::EndsWith(filename, ".cnf") ||
101 absl::EndsWith(filename, ".cnf.xz") ||
102 absl::EndsWith(filename, ".cnf.gz") ||
103 absl::EndsWith(filename, ".wcnf") ||
104 absl::EndsWith(filename, ".wcnf.xz") ||
105 absl::EndsWith(filename, ".wcnf.gz")) {
106 SatCnfReader reader(absl::GetFlag(FLAGS_wcnf_use_strong_slack));
107 if (!reader.Load(filename, cp_model)) {
108 LOG(FATAL) << "Cannot load file '" << filename << "'.";
109 }
110 } else {
111 LOG(INFO) << "Reading a CpModelProto.";
112 CHECK_OK(ReadFileToProto(filename, cp_model));
113 }
114
115 // Read the hint file.
116 if (!hint_file.empty()) {
117 CpSolverResponse response;
118 LOG(INFO) << "Reading a CpSolverResponse.";
119 CHECK_OK(ReadFileToProto(hint_file, &response));
120 CHECK_EQ(response.solution_size(), cp_model->variables_size())
121 << "The hint proto is not compatible with the model proto";
122
123 cp_model->clear_solution_hint();
124 for (int i = 0; i < cp_model->variables_size(); ++i) {
125 cp_model->mutable_solution_hint()->add_vars(i);
126 cp_model->mutable_solution_hint()->add_values(response.solution(i));
127 }
128 }
129
130 // Set the name if not present.
131 if (cp_model->name().empty()) {
132 cp_model->set_name(ExtractName(filename));
133 }
134
135 return true;
136}

138int Run() {
139 SatParameters parameters;
140 if (absl::GetFlag(FLAGS_input).empty()) {
141 LOG(FATAL) << "Please supply a data file with --input=";
142 }
143
144 // Parse the --params flag.
145 parameters.set_log_search_progress(true);
146 if (!absl::GetFlag(FLAGS_params).empty()) {
147 CHECK(google::protobuf::TextFormat::MergeFromString(
148 absl::GetFlag(FLAGS_params), &parameters))
149 << absl::GetFlag(FLAGS_params);
150 }
151
152 // Read the problem.
153 google::protobuf::Arena arena;
154 CpModelProto* cp_model =
155 google::protobuf::Arena::Create<CpModelProto>(&arena);
156 if (!LoadProblem(absl::GetFlag(FLAGS_input), absl::GetFlag(FLAGS_hint_file),
157 cp_model)) {
158 CpSolverResponse response;
159 response.set_status(CpSolverStatus::MODEL_INVALID);
160 return EXIT_SUCCESS;
161 }
162
163 Model model;
164 model.Add(NewSatParameters(parameters));
165 if (absl::GetFlag(FLAGS_fingerprint_intermediate_solutions)) {
166 // Let's add a solution callback that will display the fingerprint of all
167 // solutions.
168 model.Add(NewFeasibleSolutionLogCallback([](const CpSolverResponse& r) {
169 return absl::StrFormat(
170 "fingerprint: %#x",
171 FingerprintRepeatedField(r.solution(), kDefaultFingerprintSeed));
172 }));
173 }
174 const CpSolverResponse response = SolveCpModel(*cp_model, &model);
175
176 if (!absl::GetFlag(FLAGS_output).empty()) {
177 if (absl::EndsWith(absl::GetFlag(FLAGS_output), "txt")) {
178 CHECK_OK(file::SetTextProto(absl::GetFlag(FLAGS_output), response,
179 file::Defaults()));
180 } else {
181 CHECK_OK(file::SetBinaryProto(absl::GetFlag(FLAGS_output), response,
182 file::Defaults()));
183 }
184 }
185
186 // The SAT competition requires a particular exit code and since we don't
187 // really use it for any other purpose, we comply.
188 if (response.status() == CpSolverStatus::OPTIMAL) return 10;
189 if (response.status() == CpSolverStatus::FEASIBLE) return 10;
190 if (response.status() == CpSolverStatus::INFEASIBLE) return 20;
191 return EXIT_SUCCESS;
192}
