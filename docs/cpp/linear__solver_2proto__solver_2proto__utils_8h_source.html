32// Returns a string that should be used in MPModelRequest's
33// solver_specific_parameters field to encode the glop parameters.
34//
35// The returned string's content depends on the version of the proto library
36// that is linked in the binary.
37//
38// By default it will contain the textual representation of the input proto.
39// But when the proto-lite is used, it will contain the binary stream of the
40// proto instead since it is not possible to build the textual representation in
41// that case.
42//
43// This function will test if the proto-lite is used and expect a binary stream
44// when it is the case. So in order for your code to be portable, you should
45// always use this function to set the specific parameters.
46//
47// Proto-lite disables some features of protobufs and messages inherit from
48// MessageLite directly instead of inheriting from Message (which is itself a
49// specialization of MessageLite).
50// See https://protobuf.dev/reference/cpp/cpp-generated/#message for details.
51template <typename P>
52std::string EncodeParametersAsString(const P& parameters) {
53 if constexpr (!std::is_base_of<Message, P>::value) {
54 // Here we use SerializeToString() instead of SerializeAsString() since the
55 // later ignores errors and returns an empty string instead (which can be a
56 // valid value when no fields are set).
57 std::string bytes;
58 CHECK(parameters.SerializeToString(&bytes));
59 return bytes;
60 }
61
62 return ProtobufShortDebugString(parameters);
63}
