449 class PairEntry {
450 public:
451 PairEntry(int pickup_to_insert, int pickup_insert_after,
452 int delivery_to_insert, int delivery_insert_after, int vehicle,
453 int64_t bucket)
454 : value_(std::numeric_limits<int64_t>::max()),
455 heap_index_(-1),
456 pickup_to_insert_(pickup_to_insert),
457 pickup_insert_after_(pickup_insert_after),
458 delivery_to_insert_(delivery_to_insert),
459 delivery_insert_after_(delivery_insert_after),
460 vehicle_(vehicle),
461 bucket_(bucket) {}
462 // Note: for compatibility reasons, comparator follows tie-breaking rules
463 // used in the first version of GlobalCheapestInsertion.
464 bool operator<(const PairEntry& other) const {
465 // We give higher priority to insertions from lower buckets.
466 if (bucket_ != other.bucket_) {
467 return bucket_ > other.bucket_;
468 }
469 // We then compare by value, then we favor insertions (vehicle != -1).
470 // The rest of the tie-breaking is done with std::tie.
471 if (value_ != other.value_) {
472 return value_ > other.value_;
473 }
474 if ((vehicle_ == -1) ^ (other.vehicle_ == -1)) {
475 return vehicle_ == -1;
476 }
477 return std::tie(pickup_insert_after_, pickup_to_insert_,
478 delivery_insert_after_, delivery_to_insert_, vehicle_) >
479 std::tie(other.pickup_insert_after_, other.pickup_to_insert_,
480 other.delivery_insert_after_, other.delivery_to_insert_,
481 other.vehicle_);
482 }
483 void SetHeapIndex(int h) { heap_index_ = h; }
484 int GetHeapIndex() const { return heap_index_; }
485 void set_value(int64_t value) { value_ = value; }
486 int pickup_to_insert() const { return pickup_to_insert_; }
487 int pickup_insert_after() const { return pickup_insert_after_; }
488 void set_pickup_insert_after(int pickup_insert_after) {
489 pickup_insert_after_ = pickup_insert_after;
490 }
491 int delivery_to_insert() const { return delivery_to_insert_; }
492 int delivery_insert_after() const { return delivery_insert_after_; }
493 int vehicle() const { return vehicle_; }
494 void set_vehicle(int vehicle) { vehicle_ = vehicle; }
495
496 private:
497 int64_t value_;
498 int heap_index_;
499 int pickup_to_insert_;
500 int pickup_insert_after_;

502 int delivery_insert_after_;
503 int vehicle_;

831// Holds sequences of insertions.
832// A sequence of insertions must be in the same path, each insertion must
833// take place either after the previously inserted node or further down the
834// path, never before.
835class InsertionSequenceContainer {
836 private:
837 // InsertionSequenceContainer holds all insertion sequences in the same vector
838 // contiguously, each insertion sequence is defined by a pair of bounds.
839 // Using Insertion* directly to delimit bounds would cause out-of-memory reads
840 // when the underlying vector<Insertion> is extended and reallocated,
841 // so this stores integer bounds in InsertionBounds to delimit sequences,
842 // and InsertionSequenceIterator translates those bounds to
843 // Insertion*-based ranges (InsertionSequence) on-the-fly when iterating over
844 // all sequences.
845 struct InsertionBounds {
846 size_t begin;
847 size_t end;
848 int vehicle;
849 int64_t cost;
850 bool operator<(const InsertionBounds& other) const {
851 return std::tie(cost, vehicle, begin) <
852 std::tie(other.cost, other.vehicle, other.begin);
853 }
854 size_t Size() const { return end - begin; }
855 };
856
857 public:
858 struct Insertion {
859 int pred;
860 int node;
861 bool operator==(const Insertion& other) const {
862 return pred == other.pred && node == other.node;
863 }
864 };
