47void CustomFifoQueue::IncreaseSize(int n) {
48 CHECK_GE(n, pos_.size());
49 CHECK_LE(left_, right_);
50 pos_.resize(n, -1);
51 tmp_positions_.resize(n, 0);
52
53 // We need 1 more space since we can add at most n element.
54 queue_.resize(n + 1, 0);
55}
56
57int CustomFifoQueue::Pop() {
58 DCHECK(!empty());
59 const int id = queue_[left_];
60 pos_[id] = -1;
61 ++left_;
62 if (left_ == queue_.size()) left_ = 0;
63 return id;
64}
65
66void CustomFifoQueue::Push(int id) {
67 DCHECK_GE(id, 0);
68 DCHECK_LE(id, pos_.size());
69 DCHECK(!Contains(id));
70 pos_[id] = right_;
71 queue_[right_] = id;
72 ++right_;
73 if (right_ == queue_.size()) right_ = 0;
74}

76void CustomFifoQueue::Reorder(absl::Span<const int> order) {
77 if (order.size() <= 1) return;
78
79 const int capacity = queue_.size();
80 const int size = left_ < right_ ? right_ - left_ : left_ + capacity - right_;
81 if (order.size() > size / 8) {
82 return ReorderDense(order);
83 }
84
85 int index = 0;
86 for (const int id : order) {
87 const int p = pos_[id];
88 DCHECK_GE(p, 0);
89 tmp_positions_[index++] = p >= left_ ? p : p + capacity;
90 }
91 std::sort(&tmp_positions_[0], &tmp_positions_[index]);
92 DCHECK(std::unique(&tmp_positions_[0], &tmp_positions_[index]) ==
93 &tmp_positions_[index]);
94
95 index = 0;
96 for (const int id : order) {
97 int p = tmp_positions_[index++];
98 if (p >= capacity) p -= capacity;
99 pos_[id] = p;
100 queue_[p] = id;
101 }
102}

104void CustomFifoQueue::ReorderDense(absl::Span<const int> order) {
105 for (const int id : order) {
106 DCHECK_GE(pos_[id], 0);
107 queue_[pos_[id]] = -1;
108 }
109 int order_index = 0;
110 if (left_ <= right_) {
111 for (int i = left_; i < right_; ++i) {
112 if (queue_[i] == -1) {
113 queue_[i] = order[order_index++];
114 pos_[queue_[i]] = i;
115 }
116 }
117 } else {
118 const int size = queue_.size();
119 for (int i = left_; i < size; ++i) {
120 if (queue_[i] == -1) {
121 queue_[i] = order[order_index++];
122 pos_[queue_[i]] = i;
123 }
124 }
125 for (int i = 0; i < left_; ++i) {
126 if (queue_[i] == -1) {
127 queue_[i] = order[order_index++];
128 pos_[queue_[i]] = i;
129 }
130 }
131 }
132 DCHECK_EQ(order_index, order.size());
133}

135void CustomFifoQueue::SortByPos(absl::Span<int> elements) {
136 std::sort(elements.begin(), elements.end(),
137 [this](const int id1, const int id2) {
138 const int p1 = pos_[id1];
139 const int p2 = pos_[id2];
140 if (p1 >= left_) {
141 if (p2 >= left_) return p1 < p2;
142 return true;
143 } else {
144 // p1 < left_.
145 if (p2 < left_) return p1 < p2;
146 return false;
147 }
148 });
149}

183bool EnforcementPropagator::Propagate(Trail* /*trail*/) {
184 rev_int_repository_->SaveStateWithStamp(&rev_stack_size_, &rev_stamp_);
185 while (propagation_trail_index_ < trail_.Index()) {
186 const Literal literal = trail_[propagation_trail_index_++];
187 if (literal.Index() >= static_cast<int>(watcher_.size())) continue;
188
189 int new_size = 0;
190 auto& watch_list = watcher_[literal.Index()];
191 for (const EnforcementId id : watch_list) {
192 const LiteralIndex index = ProcessIdOnTrue(literal, id);
193 if (index == kNoLiteralIndex) {
194 // We keep the same watcher.
195 watch_list[new_size++] = id;
196 } else {
197 // Change the watcher.
198 CHECK_NE(index, literal.Index());
199 watcher_[index].push_back(id);
200 }
201 }
202 watch_list.resize(new_size);
203
204 // We also mark some constraint false.
205 for (const EnforcementId id : watcher_[literal.NegatedIndex()]) {
206 ChangeStatus(id, EnforcementStatus::IS_FALSE);
207 }
208 }
209 rev_stack_size_ = static_cast<int>(untrail_stack_.size());
210 return true;
211}

213void EnforcementPropagator::Untrail(const Trail& /*trail*/, int trail_index) {
214 // Simply revert the status change.
215 const int size = static_cast<int>(untrail_stack_.size());
216 for (int i = size - 1; i >= rev_stack_size_; --i) {
217 const auto [id, status] = untrail_stack_[i];
218 statuses_[id] = status;
219 if (callbacks_[id] != nullptr) callbacks_[id](status);
220 }
221 untrail_stack_.resize(rev_stack_size_);
222 propagation_trail_index_ = trail_index;
223}

225// Adds a new constraint to the class and returns the constraint id.
226//
227// Note that we accept empty enforcement list so that client code can be used
228// regardless of the presence of enforcement or not. A negative id means the
229// constraint is never enforced, and should be ignored.
230EnforcementId EnforcementPropagator::Register(
231 absl::Span<const Literal> enforcement,
232 std::function<void(EnforcementStatus)> callback) {
233 CHECK_EQ(trail_.CurrentDecisionLevel(), 0);
234 int num_true = 0;
235 int num_false = 0;
236 temp_literals_.clear();
237 for (const Literal l : enforcement) {
238 // Make sure we always have enough room for the literal and its negation.
239 const int size = std::max(l.Index().value(), l.NegatedIndex().value()) + 1;
240 if (size > static_cast<int>(watcher_.size())) {
241 watcher_.resize(size);
242 }
243 if (assignment_.LiteralIsTrue(l)) {
244 ++num_true;
245 continue;
246 }
247 if (assignment_.LiteralIsFalse(l)) {
248 ++num_false;
249 continue;
250 }
251 temp_literals_.push_back(l);
252 }
253 gtl::STLSortAndRemoveDuplicates(&temp_literals_);
254
255 // Return special indices if never/always enforced.
256 if (num_false > 0) {
257 if (callback != nullptr) callback(EnforcementStatus::IS_FALSE);
258 return EnforcementId(-1);
259 }
260 if (num_true == enforcement.size()) {
261 if (callback != nullptr) callback(EnforcementStatus::IS_ENFORCED);
262 return EnforcementId(-1);
263 }
264
265 const EnforcementId id(static_cast<int>(callbacks_.size()));
266 callbacks_.push_back(std::move(callback));
267
268 CHECK(!temp_literals_.empty());
269 buffer_.insert(buffer_.end(), temp_literals_.begin(), temp_literals_.end());
270 starts_.push_back(buffer_.size()); // Sentinel.
271 statuses_.push_back(EnforcementStatus::CANNOT_PROPAGATE);
272
273 if (temp_literals_.size() == 1) {
274 watcher_[temp_literals_[0].Index()].push_back(id);
275 ChangeStatus(id, EnforcementStatus::CAN_PROPAGATE);
276 } else {
277 watcher_[temp_literals_[0].Index()].push_back(id);
278 watcher_[temp_literals_[1].Index()].push_back(id);
279 }
280 return id;
281}

291// Try to propagate when the enforced constraint is not satisfiable.
292// This is currently in O(enforcement_size);
293bool EnforcementPropagator::PropagateWhenFalse(
294 EnforcementId id, absl::Span<const Literal> literal_reason,
295 absl::Span<const IntegerLiteral> integer_reason) {
296 temp_reason_.clear();
297 LiteralIndex unique_unassigned = kNoLiteralIndex;
298 for (const Literal l : GetSpan(id)) {
299 if (assignment_.LiteralIsFalse(l)) return true;
300 if (assignment_.LiteralIsTrue(l)) {
301 temp_reason_.push_back(l.Negated());
302 continue;
303 }
304 if (unique_unassigned != kNoLiteralIndex) return true;
305 unique_unassigned = l.Index();
306 }
307
308 temp_reason_.insert(temp_reason_.end(), literal_reason.begin(),
309 literal_reason.end());
310 if (unique_unassigned == kNoLiteralIndex) {
311 return integer_trail_->ReportConflict(temp_reason_, integer_reason);
312 }
313
314 integer_trail_->EnqueueLiteral(Literal(unique_unassigned).Negated(),
315 temp_reason_, integer_reason);
316 return true;
317}

320 if (id < 0) return {};

336LiteralIndex EnforcementPropagator::ProcessIdOnTrue(Literal watched,
337 EnforcementId id) {
338 const EnforcementStatus status = statuses_[id];
339 if (status == EnforcementStatus::IS_FALSE) return kNoLiteralIndex;
340
341 const auto span = GetSpan(id);
342 if (span.size() == 1) {
343 CHECK_EQ(status, EnforcementStatus::CAN_PROPAGATE);
344 ChangeStatus(id, EnforcementStatus::IS_ENFORCED);
345 return kNoLiteralIndex;
346 }
347
348 const int watched_pos = (span[0] == watched) ? 0 : 1;
349 CHECK_EQ(span[watched_pos], watched);
350 if (assignment_.LiteralIsFalse(span[watched_pos ^ 1])) {
351 ChangeStatus(id, EnforcementStatus::IS_FALSE);
352 return kNoLiteralIndex;
353 }
354
355 for (int i = 2; i < span.size(); ++i) {
356 const Literal l = span[i];
357 if (assignment_.LiteralIsFalse(l)) {
358 ChangeStatus(id, EnforcementStatus::IS_FALSE);
359 return kNoLiteralIndex;
360 }
361 if (!assignment_.LiteralIsAssigned(l)) {
362 // Replace the watched literal. Note that if the other watched literal is
363 // true, it should be processed afterwards. We do not change the status
364 std::swap(span[watched_pos], span[i]);
365 return span[watched_pos].Index();
366 }
367 }
368
369 // All literal with index > 1 are true. Two case.
370 if (assignment_.LiteralIsTrue(span[watched_pos ^ 1])) {
371 // All literals are true.
372 ChangeStatus(id, EnforcementStatus::IS_ENFORCED);
373 return kNoLiteralIndex;
374 } else {
375 // The other watched literal is the last unassigned
376 CHECK_EQ(status, EnforcementStatus::CANNOT_PROPAGATE);
377 ChangeStatus(id, EnforcementStatus::CAN_PROPAGATE);
378 return kNoLiteralIndex;
379 }
380}

382void EnforcementPropagator::ChangeStatus(EnforcementId id,
383 EnforcementStatus new_status) {
384 const EnforcementStatus old_status = statuses_[id];
385 if (old_status == new_status) return;
386 if (trail_.CurrentDecisionLevel() != 0) {
387 untrail_stack_.push_back({id, old_status});
388 }
389 statuses_[id] = new_status;
390 if (callbacks_[id] != nullptr) callbacks_[id](new_status);
391}

433void LinearPropagator::SetLevel(int level) {
434 if (level < previous_level_) {
435 // If the solver backtracked at any point, we invalidate all our queue
436 // and propagated_by information.
437 ClearPropagatedBy();
438 while (!propagation_queue_.empty()) {
439 in_queue_[propagation_queue_.Pop()] = false;
440 }
441 for (int i = rev_at_false_size_; i < in_queue_and_at_false_.size(); ++i) {
442 in_queue_[in_queue_and_at_false_[i]] = false;
443 }
444 in_queue_and_at_false_.resize(rev_at_false_size_);
445 } else if (level > previous_level_) {
446 rev_at_false_size_ = in_queue_and_at_false_.size();
447 rev_int_repository_->SaveState(&rev_at_false_size_);
448 }
449 previous_level_ = level;
450
451 // Tricky: if we aborted the current propagation because we pushed a Boolean,
452 // by default, the GenericLiteralWatcher will only call Propagate() again if
453 // one of the watched variable changed. With this, it is guaranteed to call
454 // it again if it wasn't in the queue already.
455 if (!propagation_queue_.empty()) {
456 watcher_->CallOnNextPropagate(watcher_id_);
457 }
458}

460bool LinearPropagator::Propagate() {
461 id_scanned_at_least_once_.ClearAndResize(in_queue_.size());
462
463 // Initial addition.
464 // We will clear modified_vars_ on exit.
465 for (const IntegerVariable var : modified_vars_.PositionsSetAtLeastOnce()) {
466 if (var >= var_to_constraint_ids_.size()) continue;
467 SetPropagatedBy(var, -1);
468 AddWatchedToQueue(var);
469 }
470
471 // TODO(user): Abort this propagator as soon as a Boolean is propagated ? so
472 // that we always finish the Boolean propagation first. This can happen when
473 // we push a bound that has associated Booleans. The idea is to resume from
474 // our current state when we are called again. Note however that we have to
475 // clear the propagated_by_ info has other propagator might have pushed the
476 // same variable further.
477 //
478 // Empty FIFO queue.
479 const int saved_index = trail_->Index();
480 while (!propagation_queue_.empty()) {
481 const int id = propagation_queue_.Pop();
482 in_queue_[id] = false;
483 if (!PropagateOneConstraint(id)) {
484 modified_vars_.ClearAndResize(integer_trail_->NumIntegerVariables());
485 return false;
486 }
487
488 if (trail_->Index() > saved_index) {
489 ++num_bool_aborts_;
490 break;
491 }
492 }
493
494 // Clean-up modified_vars_ to do as little as possible on the next call.
495 modified_vars_.ClearAndResize(integer_trail_->NumIntegerVariables());
496 return true;
497}

499// Adds a new constraint to the propagator.
500void LinearPropagator::AddConstraint(
501 absl::Span<const Literal> enforcement_literals,
502 absl::Span<const IntegerVariable> vars,
503 absl::Span<const IntegerValue> coeffs, IntegerValue upper_bound) {
504 if (vars.empty()) return;
505 for (const Literal l : enforcement_literals) {
506 if (trail_->Assignment().LiteralIsFalse(l)) return;
507 }
508
509 // Make sure max_variations_ is of correct size.
510 // Note that we also have a hard limit of 1 << 29 on the size.
511 CHECK_LT(vars.size(), 1 << 29);
512 if (vars.size() > max_variations_.size()) {
513 max_variations_.resize(vars.size(), 0);
514 buffer_of_ones_.resize(vars.size(), IntegerValue(1));
515 }
516
517 // Initialize constraint data.
518 CHECK_EQ(vars.size(), coeffs.size());
519 const int id = infos_.size();
520 {
521 ConstraintInfo info;
522 info.all_coeffs_are_one = false;
523 info.start = variables_buffer_.size();
524 info.initial_size = vars.size();
525 info.rev_rhs = upper_bound;
526 info.rev_size = vars.size();
527 infos_.push_back(std::move(info));
528 }
529
530 id_to_propagation_count_.push_back(0);
531 variables_buffer_.insert(variables_buffer_.end(), vars.begin(), vars.end());
532 coeffs_buffer_.insert(coeffs_buffer_.end(), coeffs.begin(), coeffs.end());
533 CanonicalizeConstraint(id);
534
535 bool all_at_one = true;
536 for (const IntegerValue coeff : GetCoeffs(infos_.back())) {
537 if (coeff != 1) {
538 all_at_one = false;
539 break;
540 }
541 }
542 if (all_at_one) {
543 // TODO(user): we still waste the space in coeffs_buffer_ so that the
544 // start are aligned with the variables_buffer_.
545 infos_.back().all_coeffs_are_one = true;
546 }
547
548 // Initialize watchers.
549 // Initialy we want everything to be propagated at least once.
550 in_queue_.push_back(false);
551 propagation_queue_.IncreaseSize(in_queue_.size());
552
553 if (!enforcement_literals.empty()) {
554 infos_.back().enf_status = EnforcementStatus::CANNOT_PROPAGATE;
555 infos_.back().enf_id = enforcement_propagator_->Register(
556 enforcement_literals, [this, id](EnforcementStatus status) {
557 infos_[id].enf_status = status;
558 // TODO(user): With some care, when we cannot propagate or the
559 // constraint is not enforced, we could live in_queue_[] at true but
560 // not put the constraint in the queue.
561 if (status == EnforcementStatus::CAN_PROPAGATE ||
562 status == EnforcementStatus::IS_ENFORCED) {
563 AddToQueueIfNeeded(id);
564 watcher_->CallOnNextPropagate(watcher_id_);
565 }
566 });
567 } else {
568 AddToQueueIfNeeded(id);
569 infos_.back().enf_id = -1;
570 infos_.back().enf_status = EnforcementStatus::IS_ENFORCED;
571 }
572
573 for (const IntegerVariable var : GetVariables(infos_[id])) {
574 // Transposed graph to know which constraint to wake up.
575 if (var >= var_to_constraint_ids_.size()) {
576 // We need both the var entry and its negation to be allocated.
577 const int size = std::max(var, NegationOf(var)).value() + 1;
578 var_to_constraint_ids_.resize(size);
579 propagated_by_.resize(size, -1);
580 propagated_by_was_set_.Resize(IntegerVariable(size));
581 is_watched_.resize(size, false);
582 }
583
584 // TODO(user): Shall we decide on some ordering here? maybe big coeff first
585 // so that we get the largest change in slack? the idea being to propagate
586 // large change first in case of cycles.
587 var_to_constraint_ids_[var].push_back(id);
588
589 // We need to be registered to the watcher so Propagate() is called at
590 // the proper priority. But then we rely on modified_vars_.
591 if (!is_watched_[var]) {
592 is_watched_[var] = true;
593 watcher_->WatchLowerBound(var, watcher_id_);
594 }
595 }
596}

611void LinearPropagator::CanonicalizeConstraint(int id) {
612 const ConstraintInfo& info = infos_[id];
613 auto coeffs = GetCoeffs(info);
614 auto vars = GetVariables(info);
615 for (int i = 0; i < vars.size(); ++i) {
616 if (coeffs[i] < 0) {
617 coeffs[i] = -coeffs[i];
618 vars[i] = NegationOf(vars[i]);
619 }
620 }
621}

623// TODO(user): template everything for the case info.all_coeffs_are_one ?
624bool LinearPropagator::PropagateOneConstraint(int id) {
625 // This is here for development purpose, it is a bit too slow to check by
626 // default though, even VLOG_IS_ON(1) so we disable it.
627 if (/* DISABLES CODE */ (false)) {
628 ++num_scanned_;
629 if (id_scanned_at_least_once_[id]) {
630 ++num_extra_scans_;
631 } else {
632 id_scanned_at_least_once_.Set(id);
633 }
634 }
635
636 // Skip constraint not enforced or that cannot propagate if false.
637 ConstraintInfo& info = infos_[id];
638 if (info.enf_status == EnforcementStatus::IS_FALSE ||
639 info.enf_status == EnforcementStatus::CANNOT_PROPAGATE) {
640 DCHECK(!in_queue_[id]);
641 if (info.enf_status == EnforcementStatus::IS_FALSE) {
642 // We mark this constraint as in the queue but will never inspect it
643 // again until we backtrack over this time.
644 in_queue_[id] = true;
645 in_queue_and_at_false_.push_back(id);
646 }
647 ++num_ignored_;
648 return true;
649 }
650
651 // Compute the slack and max_variations_ of each variables.
652 // We also filter out fixed variables in a reversible way.
653 IntegerValue implied_lb(0);
654 auto vars = GetVariables(info);
655 auto coeffs = GetCoeffs(info);
656 IntegerValue max_variation(0);
657 bool first_change = true;
658 time_limit_->AdvanceDeterministicTime(static_cast<double>(info.rev_size) *
659 1e-9);
660 for (int i = 0; i < info.rev_size;) {
661 const IntegerVariable var = vars[i];
662 const IntegerValue coeff = coeffs[i];
663 const IntegerValue lb = integer_trail_->LowerBound(var);
664 const IntegerValue ub = integer_trail_->UpperBound(var);
665 if (lb == ub) {
666 if (first_change) {
667 // Note that we can save at most one state per fixed var. Also at
668 // level zero we don't save anything.
669 rev_int_repository_->SaveState(&info.rev_size);
670 rev_integer_value_repository_->SaveState(&info.rev_rhs);
671 first_change = false;
672 }
673 info.rev_size--;
674 std::swap(vars[i], vars[info.rev_size]);
675 std::swap(coeffs[i], coeffs[info.rev_size]);
676 info.rev_rhs -= coeff * lb;
677 } else {
678 implied_lb += coeff * lb;
679 max_variations_[i] = (ub - lb) * coeff;
680 max_variation = std::max(max_variation, max_variations_[i]);
681 ++i;
682 }
683 }
684 const IntegerValue slack = info.rev_rhs - implied_lb;
685
686 // Negative slack means the constraint is false.
687 if (max_variation <= slack) return true;
688 if (slack < 0) {
689 // Fill integer reason.
690 integer_reason_.clear();
691 reason_coeffs_.clear();
692 for (int i = 0; i < info.initial_size; ++i) {
693 const IntegerVariable var = vars[i];
694 if (!integer_trail_->VariableLowerBoundIsFromLevelZero(var)) {
695 integer_reason_.push_back(integer_trail_->LowerBoundAsLiteral(var));
696 reason_coeffs_.push_back(coeffs[i]);
697 }
698 }
699
700 // Relax it.
701 integer_trail_->RelaxLinearReason(-slack - 1, reason_coeffs_,
702 &integer_reason_);
703 ++num_enforcement_pushes_;
704 return enforcement_propagator_->PropagateWhenFalse(info.enf_id, {},
705 integer_reason_);
706 }
707
708 // We can only propagate more if all the enforcement literals are true.
709 if (info.enf_status != EnforcementStatus::IS_ENFORCED) return true;
710
711 // The lower bound of all the variables except one can be used to update the
712 // upper bound of the last one.
713 int num_pushed = 0;
714 for (int i = 0; i < info.rev_size; ++i) {
715 if (max_variations_[i] <= slack) continue;
716
717 // TODO(user): If the new ub fall into an hole of the variable, we can
718 // actually relax the reason more by computing a better slack.
719 ++num_pushes_;
720 const IntegerVariable var = vars[i];
721 const IntegerValue coeff = coeffs[i];
722 const IntegerValue div = slack / coeff;
723 const IntegerValue new_ub = integer_trail_->LowerBound(var) + div;
724 const IntegerValue propagation_slack = (div + 1) * coeff - slack - 1;
725 if (!integer_trail_->Enqueue(
726 IntegerLiteral::LowerOrEqual(var, new_ub),
727 /*lazy_reason=*/[this, info, propagation_slack](
728 IntegerLiteral i_lit, int trail_index,
729 std::vector<Literal>* literal_reason,
730 std::vector<int>* trail_indices_reason) {
731 literal_reason->clear();
732 trail_indices_reason->clear();
733 enforcement_propagator_->AddEnforcementReason(info.enf_id,
734 literal_reason);
735 reason_coeffs_.clear();
736
737 auto coeffs = GetCoeffs(info);
738 auto vars = GetVariables(info);
739 for (int i = 0; i < info.initial_size; ++i) {
740 const IntegerVariable var = vars[i];
741 if (PositiveVariable(var) == PositiveVariable(i_lit.var)) {
742 continue;
743 }
744 const int index =
745 integer_trail_->FindTrailIndexOfVarBefore(var, trail_index);
746 if (index >= 0) {
747 trail_indices_reason->push_back(index);
748 if (propagation_slack > 0) {
749 reason_coeffs_.push_back(coeffs[i]);
750 }
751 }
752 }
753 if (propagation_slack > 0) {
754 integer_trail_->RelaxLinearReason(
755 propagation_slack, reason_coeffs_, trail_indices_reason);
756 }
757 })) {
758 return false;
759 }
760
761 // Add to the queue all touched constraint.
762 const IntegerValue actual_ub = integer_trail_->UpperBound(var);
763 const IntegerVariable next_var = NegationOf(var);
764 if (actual_ub < new_ub) {
765 // Was pushed further due to hole. We clear it.
766 SetPropagatedBy(next_var, -1);
767 AddWatchedToQueue(next_var);
768 } else if (actual_ub == new_ub) {
769 SetPropagatedBy(next_var, id);
770 AddWatchedToQueue(next_var);
771
772 // We reorder them first.
773 std::swap(vars[i], vars[num_pushed]);
774 std::swap(coeffs[i], coeffs[num_pushed]);
775 ++num_pushed;
776 }
777
778 // Explore the subtree and detect cycles greedily.
779 // Also postpone some propagation.
780 if (num_pushed > 0) {
781 if (!DisassembleSubtree(id, num_pushed)) {
782 return false;
783 }
784 }
785 }
786
787 return true;
788}

791 std::string result;

812bool LinearPropagator::ReportConflictingCycle() {
813 // Often, all coefficients of the variable involved in the cycle are the same
814 // and if we sum all constraint, we get an infeasible one. If this is the
815 // case, we simplify the reason.
816 //
817 // TODO(user): We could relax if the coefficient of the sum do not overflow.
818 // TODO(user): Sum constraints with eventual factor in more cases.
819 {
820 literal_reason_.clear();
821 integer_reason_.clear();
822 absl::int128 rhs_sum = 0;
823 absl::flat_hash_map<IntegerVariable, absl::int128> map_sum;
824 for (const auto [id, next_var] : disassemble_branch_) {
825 const ConstraintInfo& info = infos_[id];
826 enforcement_propagator_->AddEnforcementReason(info.enf_id,
827 &literal_reason_);
828 auto coeffs = GetCoeffs(info);
829 auto vars = GetVariables(info);
830 IntegerValue rhs_correction(0);
831 for (int i = 0; i < info.initial_size; ++i) {
832 if (i >= info.rev_size) {
833 rhs_correction += coeffs[i] * integer_trail_->LowerBound(vars[i]);
834 }
835 if (VariableIsPositive(vars[i])) {
836 map_sum[vars[i]] += coeffs[i].value();
837 } else {
838 map_sum[PositiveVariable(vars[i])] -= coeffs[i].value();
839 }
840 }
841 rhs_sum += (info.rev_rhs + rhs_correction).value();
842 }
843
844 // We shouldn't have overflow since each component do not overflow an
845 // int64_t and we sum a small amount of them.
846 absl::int128 implied_lb = 0;
847 for (const auto [var, coeff] : map_sum) {
848 if (coeff > 0) {
849 if (!integer_trail_->VariableLowerBoundIsFromLevelZero(var)) {
850 integer_reason_.push_back(integer_trail_->LowerBoundAsLiteral(var));
851 }
852 implied_lb +=
853 coeff * absl::int128{integer_trail_->LowerBound(var).value()};
854 } else if (coeff < 0) {
855 if (!integer_trail_->VariableLowerBoundIsFromLevelZero(
856 NegationOf(var))) {
857 integer_reason_.push_back(integer_trail_->UpperBoundAsLiteral(var));
858 }
859 implied_lb +=
860 coeff * absl::int128{integer_trail_->UpperBound(var).value()};
861 }
862 }
863 if (implied_lb > rhs_sum) {
864 // We sort for determinism.
865 std::sort(integer_reason_.begin(), integer_reason_.end(),
866 [](const IntegerLiteral& a, const IntegerLiteral& b) {
867 return a.var < b.var;
868 });
869
870 // Relax the linear reason if everything fit on an int64_t.
871 const absl::int128 limit{std::numeric_limits<int64_t>::max()};
872 const absl::int128 slack = implied_lb - rhs_sum;
873 if (slack > 1) {
874 reason_coeffs_.clear();
875 bool abort = false;
876 for (const IntegerLiteral i_lit : integer_reason_) {
877 absl::int128 c = map_sum.at(PositiveVariable(i_lit.var));
878 if (c < 0) c = -c; // No std::abs() for int128.
879 if (c >= limit) {
880 abort = true;
881 break;
882 }
883 reason_coeffs_.push_back(static_cast<int64_t>(c));
884 }
885 if (!abort) {
886 const IntegerValue slack64(
887 static_cast<int64_t>(std::min(limit, slack)));
888 integer_trail_->RelaxLinearReason(slack64 - 1, reason_coeffs_,
889 &integer_reason_);
890 }
891 }
892
893 ++num_simple_cycles_;
894 VLOG(2) << "Simplified " << integer_reason_.size() << " slack "
895 << implied_lb - rhs_sum;
896 return integer_trail_->ReportConflict(literal_reason_, integer_reason_);
897 }
898 }
899
900 // For the complex reason, we just use the bound of every variable.
901 // We do some basic simplification for the variable involved in the cycle.
902 //
903 // TODO(user): Can we simplify more?
904 VLOG(2) << "Cycle";
905 literal_reason_.clear();
906 integer_reason_.clear();
907 IntegerVariable previous_var = kNoIntegerVariable;
908 for (const auto [id, next_var] : disassemble_branch_) {
909 const ConstraintInfo& info = infos_[id];
910 enforcement_propagator_->AddEnforcementReason(info.enf_id,
911 &literal_reason_);
912 for (const IntegerVariable var : GetVariables(infos_[id])) {
913 // The lower bound of this variable is implied by the previous constraint,
914 // so we do not need to include it.
915 if (var == previous_var) continue;
916
917 // We do not need the lower bound of var to propagate its upper bound.
918 if (var == NegationOf(next_var)) continue;
919
920 if (!integer_trail_->VariableLowerBoundIsFromLevelZero(var)) {
921 integer_reason_.push_back(integer_trail_->LowerBoundAsLiteral(var));
922 }
923 }
924 previous_var = next_var;
925
926 VLOG(2) << next_var << " [" << integer_trail_->LowerBound(next_var) << ","
927 << integer_trail_->UpperBound(next_var)
928 << "] : " << ConstraintDebugString(id);
929 }
930 ++num_complex_cycles_;
931 return integer_trail_->ReportConflict(literal_reason_, integer_reason_);
932}
933
934// Note that if there is a loop in the propagated_by_ graph, it must be
935// from root_id -> root_var, because each time we add an edge, we do
936// disassemble.
937//
938// TODO(user): If one of the var coeff is > previous slack we push an id again,
939// we can stop early with a conflict by propagating the ids in sequence.
940bool LinearPropagator::DisassembleSubtree(int root_id, int num_pushed) {
941 disassemble_to_reorder_.ClearAndResize(in_queue_.size());
942 disassemble_reverse_topo_order_.clear();
943
944 // The variable was just pushed, we explore the set of variable that will
945 // be pushed further due to this push. Basically, if a constraint propagated
946 // before and its slack will reduce due to the push, then any previously
947 // propagated variable with a coefficient NOT GREATER than the one of the
948 // variable reducing the slack will be pushed further.
949 disassemble_queue_.clear();
950 disassemble_branch_.clear();
951 {
952 const ConstraintInfo& info = infos_[root_id];
953 auto vars = GetVariables(info);
954 for (int i = 0; i < num_pushed; ++i) {
955 disassemble_queue_.push_back({root_id, NegationOf(vars[i])});
956 }
957 }
958
959 // Note that all var should be unique since there is only one propagated_by_
960 // for each one. And each time we explore an id, we disassemble the tree.
961 while (!disassemble_queue_.empty()) {
962 const auto [prev_id, var] = disassemble_queue_.back();
963 if (!disassemble_branch_.empty() &&
964 disassemble_branch_.back().first == prev_id &&
965 disassemble_branch_.back().second == var) {
966 disassemble_branch_.pop_back();
967 disassemble_reverse_topo_order_.push_back(prev_id);
968 disassemble_queue_.pop_back();
969 continue;
970 }
971
972 disassemble_branch_.push_back({prev_id, var});
973 time_limit_->AdvanceDeterministicTime(
974 static_cast<double>(var_to_constraint_ids_[var].size()) * 1e-9);
975 for (const int id : var_to_constraint_ids_[var]) {
976 if (prev_id == root_id) {
977 // Root id was just propagated, so there is no need to reorder what
978 // it pushes.
979 DCHECK_NE(id, root_id);
980 if (disassemble_to_reorder_[id]) continue;
981 disassemble_to_reorder_.Set(id);
982 } else if (id == root_id) {
983 // TODO(user): Check previous slack vs var coeff?
984 // TODO(user): Make sure there are none or detect cycle not going back
985 // to the root.
986 CHECK(!disassemble_branch_.empty());
987
988 // This is a corner case in which there is actually no cycle.
989 const IntegerVariable root_var = disassemble_branch_[0].second;
990 CHECK_EQ(disassemble_branch_[0].first, root_id);
991 CHECK_NE(var, root_var);
992 if (var == NegationOf(root_var)) continue;
993
994 // Tricky: We have a cycle here only if coeff of var >= root_coeff.
995 // If there is no cycle, we will just finish the branch here.
996 //
997 // TODO(user): Can we be more precise? if one coeff is big, the
998 // variation in slack might be big enough to push a variable twice and
999 // thus push a lower coeff.
1000 const ConstraintInfo& info = infos_[id];
1001 auto coeffs = GetCoeffs(info);
1002 auto vars = GetVariables(info);
1003 IntegerValue root_coeff(0);
1004 IntegerValue var_coeff(0);
1005 for (int i = 0; i < info.initial_size; ++i) {
1006 if (vars[i] == var) var_coeff = coeffs[i];
1007 if (vars[i] == NegationOf(root_var)) root_coeff = coeffs[i];
1008 }
1009 CHECK_NE(root_coeff, 0);
1010 CHECK_NE(var_coeff, 0);
1011 if (var_coeff >= root_coeff) {
1012 return ReportConflictingCycle();
1013 } else {
1014 // We don't want to continue the search from root_id.
1015 continue;
1016 }
1017 }
1018
1019 if (id_to_propagation_count_[id] == 0) continue; // Didn't push.
1020 disassemble_to_reorder_.Set(id);
1021
1022 // The constraint pushed some variable. Identify which ones will be pushed
1023 // further. Disassemble the whole info since we are about to propagate
1024 // this constraint again. Any pushed variable must be before the rev_size.
1025 const ConstraintInfo& info = infos_[id];
1026 auto coeffs = GetCoeffs(info);
1027 auto vars = GetVariables(info);
1028 IntegerValue var_coeff(0);
1029 disassemble_candidates_.clear();
1030 ++num_explored_in_disassemble_;
1031 time_limit_->AdvanceDeterministicTime(static_cast<double>(info.rev_size) *
1032 1e-9);
1033 for (int i = 0; i < info.rev_size; ++i) {
1034 if (vars[i] == var) {
1035 var_coeff = coeffs[i];
1036 continue;
1037 }
1038 const IntegerVariable next_var = NegationOf(vars[i]);
1039 if (propagated_by_[next_var] == id) {
1040 disassemble_candidates_.push_back({next_var, coeffs[i]});
1041
1042 // We will propagate var again later, so clear all this for now.
1043 propagated_by_[next_var] = -1;
1044 id_to_propagation_count_[id]--;
1045 }
1046 }
1047 for (const auto [next_var, coeff] : disassemble_candidates_) {
1048 if (coeff <= var_coeff) {
1049 // We are guaranteed to push next_var only if var_coeff will move
1050 // the slack enough.
1051 //
1052 // TODO(user): Keep current delta in term of the DFS so we detect
1053 // cycle and depedendences in more cases.
1054 disassemble_queue_.push_back({id, next_var});
1055 }
1056 }
1057 }
1058 }
1059
1060 CHECK(!disassemble_to_reorder_[root_id]);
1061 tmp_to_reorder_.clear();
1062 std::reverse(disassemble_reverse_topo_order_.begin(),
1063 disassemble_reverse_topo_order_.end()); // !! not unique
1064 for (const int id : disassemble_reverse_topo_order_) {
1065 if (!disassemble_to_reorder_[id]) continue;
1066 disassemble_to_reorder_.Clear(id);
1067 AddToQueueIfNeeded(id);
1068 if (!propagation_queue_.Contains(id)) continue;
1069 tmp_to_reorder_.push_back(id);
1070 }
1071
1072 // TODO(user): Reordering can be sloe since require sort and can touch many
1073 // entries. Investigate alternatives. We could probably optimize this a bit
1074 // more.
1075 if (tmp_to_reorder_.empty()) return true;
1076 const int important_size = static_cast<int>(tmp_to_reorder_.size());
1077
1078 for (const int id : disassemble_to_reorder_.PositionsSetAtLeastOnce()) {
1079 if (!disassemble_to_reorder_[id]) continue;
1080 disassemble_to_reorder_.Clear(id);
1081 if (!propagation_queue_.Contains(id)) continue;
1082 tmp_to_reorder_.push_back(id);
1083 }
1084 disassemble_to_reorder_.NotifyAllClear();
1085
1086 // We try to keep the same order as before for the elements not in the
1087 // topological order.
1088 propagation_queue_.SortByPos(
1089 absl::MakeSpan(&tmp_to_reorder_[important_size],
1090 tmp_to_reorder_.size() - important_size));
1091
1092 num_reordered_ += tmp_to_reorder_.size();
1093 propagation_queue_.Reorder(tmp_to_reorder_);
1094 return true;
1095}
