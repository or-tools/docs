32bool GurobiIsCorrectlyInstalled() {
33 absl::StatusOr<GRBenv*> status = GetGurobiEnv();
34 if (!status.ok() || status.value() == nullptr) {
35 LOG(WARNING) << status.status();
36 return false;
37 }
38
39 GRBfreeenv(status.value());
40
41 return true;
42}

345std::vector<std::string> GurobiDynamicLibraryPotentialPaths() {
346 std::vector<std::string> potential_paths;
347 const std::vector<std::string> kGurobiVersions = {
348 "1100", "1003", "1002", "1001", "1000", "952", "951", "950",
349 "911", "910", "903", "902", "811", "801", "752"};
350 potential_paths.reserve(kGurobiVersions.size() * 3);
351
352 // Look for libraries pointed by GUROBI_HOME first.
353 const char* gurobi_home_from_env = getenv("GUROBI_HOME");
354 if (gurobi_home_from_env != nullptr) {
355 for (const std::string& version : kGurobiVersions) {
356 const std::string lib = version.substr(0, version.size() - 1);
357#if defined(_MSC_VER) // Windows
358 potential_paths.push_back(
359 absl::StrCat(gurobi_home_from_env, "\\bin\\gurobi", lib, ".dll"));
360#elif defined(__APPLE__) // OS X
361 potential_paths.push_back(
362 absl::StrCat(gurobi_home_from_env, "/lib/libgurobi", lib, ".dylib"));
363#elif defined(__GNUC__) // Linux
364 potential_paths.push_back(
365 absl::StrCat(gurobi_home_from_env, "/lib/libgurobi", lib, ".so"));
366 potential_paths.push_back(
367 absl::StrCat(gurobi_home_from_env, "/lib64/libgurobi", lib, ".so"));
368#else
369 LOG(ERROR) << "OS Not recognized by gurobi/environment.cc."
370 << " You won't be able to use Gurobi.";
371#endif
372 }
373 }
374
375 // Search for canonical places.
376 for (const std::string& version : kGurobiVersions) {
377 const std::string lib = version.substr(0, version.size() - 1);
378#if defined(_MSC_VER) // Windows
379 potential_paths.push_back(absl::StrCat("C:\\Program Files\\gurobi", version,
380 "\\win64\\bin\\gurobi", lib,
381 ".dll"));
382 potential_paths.push_back(absl::StrCat(
383 "C:\\gurobi", version, "\\win64\\bin\\gurobi", lib, ".dll"));
384 potential_paths.push_back(absl::StrCat("gurobi", lib, ".dll"));
385#elif defined(__APPLE__) // OS X
386 potential_paths.push_back(absl::StrCat(
387 "/Library/gurobi", version, "/mac64/lib/libgurobi", lib, ".dylib"));
388 potential_paths.push_back(absl::StrCat("/Library/gurobi", version,
389 "/macos_universal2/lib/libgurobi",
390 lib, ".dylib"));
391#elif defined(__GNUC__) // Linux
392 potential_paths.push_back(absl::StrCat(
393 "/opt/gurobi", version, "/linux64/lib/libgurobi", lib, ".so"));
394 potential_paths.push_back(absl::StrCat(
395 "/opt/gurobi", version, "/linux64/lib64/libgurobi", lib, ".so"));
396 potential_paths.push_back(
397 absl::StrCat("/opt/gurobi/linux64/lib/libgurobi", lib, ".so"));
398 potential_paths.push_back(
399 absl::StrCat("/opt/gurobi/linux64/lib64/libgurobi", lib, ".so"));
400#else
401 LOG(ERROR) << "OS Not recognized by gurobi/environment.cc."
402 << " You won't be able to use Gurobi.";
403#endif
404 }
405
406#if defined(__GNUC__) // path in linux64 gurobi/optimizer docker image.
407 for (const std::string& version : {"11.0.0", "10.0.3", "10.0.2", "10.0.1",
408 "10.0.0", "9.5.2", "9.5.1", "9.5.0"}) {
409 potential_paths.push_back(
410 absl::StrCat("/opt/gurobi/linux64/lib/libgurobi.so.", version));
411 }
412#endif
413 return potential_paths;
414}

416absl::Status LoadGurobiDynamicLibrary(
417 std::vector<std::string> potential_paths) {
418 static std::once_flag gurobi_loading_done;
419 static absl::Status gurobi_load_status;
420 static DynamicLibrary gurobi_library;
421 static absl::Mutex mutex;
422
423 absl::MutexLock lock(&mutex);
424
425 std::call_once(gurobi_loading_done, [&potential_paths]() {
426 const std::vector<std::string> canonical_paths =
427 GurobiDynamicLibraryPotentialPaths();
428 potential_paths.insert(potential_paths.end(), canonical_paths.begin(),
429 canonical_paths.end());
430 for (const std::string& path : potential_paths) {
431 if (gurobi_library.TryToLoad(path)) {
432 LOG(INFO) << "Found the Gurobi library in '" << path << ".";
433 break;
434 }
435 }
436
437 if (gurobi_library.LibraryIsLoaded()) {
438 LoadGurobiFunctions(&gurobi_library);
439 gurobi_load_status = absl::OkStatus();
440 } else {
441 gurobi_load_status = absl::NotFoundError(absl::StrCat(
442 "Could not find the Gurobi shared library. Looked in: [",
443 absl::StrJoin(potential_paths, "', '"),
444 "]. If you know where it"
445 " is, pass the full path to 'LoadGurobiDynamicLibrary()'."));
446 }
447 });
448 return gurobi_load_status;
449}
450
451absl::StatusOr<GRBenv*> GetGurobiEnv() {
452 RETURN_IF_ERROR(LoadGurobiDynamicLibrary({}));
453
454 GRBenv* env = nullptr;
455
456 if (GRBloadenv(&env, nullptr) != 0 || env == nullptr) {
457 return absl::FailedPreconditionError(
458 absl::StrCat("Found the Gurobi shared library, but could not create "
459 "Gurobi environment: is Gurobi licensed on this machine?",
460 GRBgeterrormsg(env)));
461 }
462 return env;
463}
