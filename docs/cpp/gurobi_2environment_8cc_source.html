33bool GurobiIsCorrectlyInstalled() {
34 absl::StatusOr<GRBenv*> status = GetGurobiEnv();
35 if (!status.ok() || status.value() == nullptr) {
36 LOG(WARNING) << status.status();
37 return false;
38 }
39
40 GRBfreeenv(status.value());
41
42 return true;
43}

351std::vector<std::string> GurobiDynamicLibraryPotentialPaths() {
352 std::vector<std::string> potential_paths;
353 const std::vector<std::string> kGurobiVersions = {
354 "1201", "1200", "1103", "1102", "1101", "1100", "1003",
355 "1002", "1001", "1000", "952", "951", "950", "911",
356 "910", "903", "902", "811", "801", "752"};
357 potential_paths.reserve(kGurobiVersions.size() * 3);
358
359 // Look for libraries pointed by GUROBI_HOME first.
360 const char* gurobi_home_from_env = getenv("GUROBI_HOME");
361 if (gurobi_home_from_env != nullptr) {
362 for (const std::string& version : kGurobiVersions) {
363 const std::string lib = version.substr(0, version.size() - 1);
364#if defined(_MSC_VER) // Windows
365 potential_paths.push_back(
366 absl::StrCat(gurobi_home_from_env, "\\bin\\gurobi", lib, ".dll"));
367#elif defined(__APPLE__) // OS X
368 potential_paths.push_back(
369 absl::StrCat(gurobi_home_from_env, "/lib/libgurobi", lib, ".dylib"));
370#elif defined(__GNUC__) // Linux
371 potential_paths.push_back(
372 absl::StrCat(gurobi_home_from_env, "/lib/libgurobi", lib, ".so"));
373 potential_paths.push_back(
374 absl::StrCat(gurobi_home_from_env, "/lib64/libgurobi", lib, ".so"));
375#else
376 LOG(ERROR) << "OS Not recognized by gurobi/environment.cc."
377 << " You won't be able to use Gurobi.";
378#endif
379 }
380 }
381
382 // Search for canonical places.
383 for (const std::string& version : kGurobiVersions) {
384 const std::string lib = version.substr(0, version.size() - 1);
385#if defined(_MSC_VER) // Windows
386 potential_paths.push_back(absl::StrCat("C:\\Program Files\\gurobi", version,
387 "\\win64\\bin\\gurobi", lib,
388 ".dll"));
389 potential_paths.push_back(absl::StrCat(
390 "C:\\gurobi", version, "\\win64\\bin\\gurobi", lib, ".dll"));
391 potential_paths.push_back(absl::StrCat("gurobi", lib, ".dll"));
392#elif defined(__APPLE__) // OS X
393 potential_paths.push_back(absl::StrCat(
394 "/Library/gurobi", version, "/mac64/lib/libgurobi", lib, ".dylib"));
395 potential_paths.push_back(absl::StrCat("/Library/gurobi", version,
396 "/macos_universal2/lib/libgurobi",
397 lib, ".dylib"));
398#elif defined(__GNUC__) // Linux
399 potential_paths.push_back(absl::StrCat(
400 "/opt/gurobi", version, "/linux64/lib/libgurobi", lib, ".so"));
401 potential_paths.push_back(absl::StrCat(
402 "/opt/gurobi", version, "/linux64/lib64/libgurobi", lib, ".so"));
403 potential_paths.push_back(
404 absl::StrCat("/opt/gurobi/linux64/lib/libgurobi", lib, ".so"));
405 potential_paths.push_back(
406 absl::StrCat("/opt/gurobi/linux64/lib64/libgurobi", lib, ".so"));
407#else
408 LOG(ERROR) << "OS Not recognized by gurobi/environment.cc."
409 << " You won't be able to use Gurobi.";
410#endif
411 }
412
413#if defined(__GNUC__) // path in linux64 gurobi/optimizer docker image.
414 for (const std::string& version :
415 {"12.0.1", "12.0.0", "11.0.3", "11.0.2", "11.0.1", "11.0.0", "10.0.3",
416 "10.0.2", "10.0.1", "10.0.0", "9.5.2", "9.5.1", "9.5.0"}) {
417 potential_paths.push_back(
418 absl::StrCat("/opt/gurobi/linux64/lib/libgurobi.so.", version));
419 }
420#endif
421 return potential_paths;
422}

424absl::Status LoadGurobiDynamicLibrary(
425 std::vector<std::string> potential_paths) {
426 static std::once_flag gurobi_loading_done;
427 static absl::Status gurobi_load_status;
428 static DynamicLibrary gurobi_library;
429 static absl::Mutex mutex;
430
431 absl::MutexLock lock(&mutex);
432
433 std::call_once(gurobi_loading_done, [&potential_paths]() {
434 const std::vector<std::string> canonical_paths =
435 GurobiDynamicLibraryPotentialPaths();
436 potential_paths.insert(potential_paths.end(), canonical_paths.begin(),
437 canonical_paths.end());
438 for (const std::string& path : potential_paths) {
439 if (gurobi_library.TryToLoad(path)) {
440 LOG(INFO) << "Found the Gurobi library in '" << path << ".";
441 break;
442 }
443 }
444
445 if (gurobi_library.LibraryIsLoaded()) {
446 LoadGurobiFunctions(&gurobi_library);
447 gurobi_load_status = absl::OkStatus();
448 } else {
449 gurobi_load_status = absl::NotFoundError(absl::StrCat(
450 "Could not find the Gurobi shared library. Looked in: [",
451 absl::StrJoin(potential_paths, "', '"),
452 "]. If you know where it"
453 " is, pass the full path to 'LoadGurobiDynamicLibrary()'."));
454 }
455 });
456 return gurobi_load_status;
457}
