73CallbackData::CallbackData(const ModelStorage* storage,
74 const CallbackDataProto& proto)
75 // iOS 11 does not support .value() hence we use operator* here and CHECK
76 // below that we have a value.
77 : event(*EnumFromProto(proto.event())),
78 presolve_stats(proto.presolve_stats()),
79 simplex_stats(proto.simplex_stats()),
80 barrier_stats(proto.barrier_stats()),
81 mip_stats(proto.mip_stats()) {
82 CHECK(EnumFromProto(proto.event()).has_value());
83 if (proto.has_primal_solution_vector()) {
84 solution = VariableValuesFromProto(storage, proto.primal_solution_vector())
85 .value();
86 }
87 auto maybe_time = util_time::DecodeGoogleApiProto(proto.runtime());
88 CHECK_OK(maybe_time.status());
89 runtime = *maybe_time;
90}

105CallbackRegistrationProto CallbackRegistration::Proto() const {
106 CallbackRegistrationProto result;
107 for (const CallbackEvent event : events) {
108 result.add_request_registration(EnumToProto(event));
109 }
110 std::sort(result.mutable_request_registration()->begin(),
111 result.mutable_request_registration()->end());
112 *result.mutable_mip_solution_filter() = mip_solution_filter.Proto();
113 *result.mutable_mip_node_filter() = mip_node_filter.Proto();
114 result.set_add_lazy_constraints(add_lazy_constraints);
115 result.set_add_cuts(add_cuts);
116 return result;
117}

119absl::Status CallbackResult::CheckModelStorage(
120 const ModelStorage* const expected_storage) const {
121 for (const GeneratedLinearConstraint& constraint : new_constraints) {
122 RETURN_IF_ERROR(
123 internal::CheckModelStorage(/*storage=*/constraint.storage(),
124 /*expected_storage=*/expected_storage))
125 << "invalid new_constraints";
126 }
127 for (const VariableMap<double>& solution : suggested_solutions) {
128 RETURN_IF_ERROR(internal::CheckModelStorage(
129 /*storage=*/solution.storage(), /*expected_storage=*/expected_storage))
130 << "invalid suggested_solutions";
131 }
132 return absl::OkStatus();
133}
134
135CallbackResultProto CallbackResult::Proto() const {
136 CallbackResultProto result;
137 result.set_terminate(terminate);
138 for (const VariableMap<double>& solution : suggested_solutions) {
139 *result.add_suggested_solutions() = VariableValuesToProto(solution);
140 }
141 for (const GeneratedLinearConstraint& constraint : new_constraints) {
142 CallbackResultProto::GeneratedLinearConstraint* constraint_proto =
143 result.add_cuts();
144 constraint_proto->set_is_lazy(constraint.is_lazy);
145 constraint_proto->set_lower_bound(
146 constraint.linear_constraint.lower_bound_minus_offset());
147 constraint_proto->set_upper_bound(
148 constraint.linear_constraint.upper_bound_minus_offset());
149 *constraint_proto->mutable_linear_expression() =
150 VariableValuesToProto(constraint.linear_constraint.expression.terms());
151 }
152 return result;
153}
