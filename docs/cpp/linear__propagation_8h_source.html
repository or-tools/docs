46// A FIFO queue that allows some form of reordering of its element.
47class CustomFifoQueue {
48 public:
49 CustomFifoQueue() = default;
50
51 // Note that this requires the queue to be empty or to never have been poped
52 // before.
53 void IncreaseSize(int n);
54
55 int Pop();
56 void Push(int id);
57
58 bool empty() const { return left_ == right_; }
59 bool Contains(int id) const { return pos_[id] != -1; }
60
61 // Reorder the given element to match their given order. They must all be in
62 // the queue.
63 void Reorder(absl::Span<const int> order);
64 void ReorderDense(absl::Span<const int> order);
65
66 // Sorts the given elements by their current position from the top.
67 // Elements should all be in the queue.
68 void SortByPos(absl::Span<int> elements);
69
70 private:
71 // The queue is stored in [left_, right_) with eventual wrap around % size.
72 // The positions of each element is in pos_[element] and never changes during
73 // normal operation. A position of -1 means that the element is not in the
74 // queue.
75 std::vector<int> pos_;
76 std::vector<int> queue_;
77 int left_ = 0;
78 int right_ = 0;
79
80 std::vector<int> tmp_positions_;
81 std::vector<int> tmp_order_;
82};

84// An enforced constraint can be in one of these 4 states.
85// Note that we rely on the integer encoding to take 2 bits for optimization.
86enum EnforcementStatus {
87 // One enforcement literal is false.
88 IS_FALSE = 0,
89 // More than two literals are unassigned.
90 CANNOT_PROPAGATE = 1,
91 // All enforcement literals are true but one.
92 CAN_PROPAGATE = 2,
93 // All enforcement literals are true.
94 IS_ENFORCED = 3,
95};

169// This is meant to supersede both IntegerSumLE and the PrecedencePropagator.
170//
171// TODO(user): This is a work in progress and is currently incomplete:
172// - Lack more incremental support for faster propag.
173// - Lack detection and propagation of at least one of these linear is true
174// which can be used to propagate more bound if a variable appear in all these
175// constraint.
176class LinearPropagator : public PropagatorInterface, ReversibleInterface {
177 public:

188
189 private:
190 // We try to pack the struct as much as possible. Using a maximum size of
191 // 1 << 29 should be okay since we split long constraint anyway. Technically
192 // we could use int16_t or even int8_t if we wanted, but we just need to make
193 // sure we do split ALL constraints, not just the one from the initial model.
194 //
195 // TODO(user): We could also move some less often used fields out. like
196 // initial size and enf_id that are only needed when we push something.
197 struct ConstraintInfo {
198 unsigned int enf_status : 2;
199 bool all_coeffs_are_one : 1;
200 unsigned int initial_size : 29; // Const. The size including all terms.
201
202 EnforcementId enf_id; // Const. The id in enforcement_propagator_.
203 int start; // Const. The start of the constraint in the buffers.
204 int rev_size; // The size of the non-fixed terms.
205 IntegerValue rev_rhs; // The current rhs, updated on fixed terms.
206 };
