48// Neighborhood returned by Neighborhood generators.
49struct Neighborhood {
50 // True if neighborhood generator was able to generate a neighborhood.
51 bool is_generated = false;
52
53 // True if an optimal solution to the neighborhood is also an optimal solution
54 // to the original model.
55 bool is_reduced = false;
56
57 // True if this neighborhood was just obtained by fixing some variables.
58 bool is_simple = false;
59
60 // Specification of the delta between the initial model and the lns fragment.
61 // The delta will contains all variables from the initial model, potentially
62 // with updated domains.
63 // It can contains new variables and new constraints, and solution hinting.
64 CpModelProto delta;
65 std::vector<int> constraints_to_ignore;
66
67 // Neighborhood Id. Used to identify the neighborhood by a generator.
68 // Currently only used by WeightedRandomRelaxationNeighborhoodGenerator.
69 // TODO(user): Make sure that the id is unique for each generated
70 // neighborhood for each generator.
71 int64_t id = 0;
72
73 // Used for identifying the source of the neighborhood if it is generated
74 // using solution repositories.
75 std::string source_info = "";
76
77 // Statistic, only filled when is_simple is true.
78 int num_relaxed_variables = 0;
79 int num_relaxed_variables_in_objective = 0;
80
81 // Only filled when is_simple is true. If we solve the fragment to optimality,
82 // then we can just fix the variable listed here to that optimal solution.
83 //
84 // This can happen if the neighborhood fully cover some part that are
85 // completely independent from the rest of the model. Like for instance an
86 // unused but not yet fixed variable.
87 //
88 // WARNING: all such variables should be fixed at once in a lock-like manner,
89 // because they can be multiple optimal solutions on these variables.
90 std::vector<int> variables_that_can_be_fixed_to_local_optimum;
91};

308// Base class for a CpModelProto neighborhood generator.
309class NeighborhoodGenerator {
310 public:
311 NeighborhoodGenerator(const std::string& name,
312 NeighborhoodGeneratorHelper const* helper)
313 : name_(name), helper_(*helper), difficulty_(0.5) {}
314 virtual ~NeighborhoodGenerator() {}
315
316 // Generates a "local" subproblem for the given seed.
317 //
318 // The difficulty will be in [0, 1] and is related to the asked neighborhood
319 // size (and thus local problem difficulty). A difficulty of 0.0 means empty
320 // neighborhood and a difficulty of 1.0 means the full problem. The algorithm
321 // should try to generate a neighborhood according to this difficulty which
322 // will be dynamically adjusted depending on whether or not we can solve the
323 // subproblem in a given time limit.
324 //
325 // The given initial_solution should contain a feasible solution to the
326 // initial CpModelProto given to this class. Any solution to the returned
327 // CPModelProto should also be valid solution to the same initial model.
328 //
329 // This function should be thread-safe.
330 virtual Neighborhood Generate(const CpSolverResponse& initial_solution,
331 double difficulty, absl::BitGenRef random) = 0;
332
333 // Returns true if the neighborhood generator can generate a neighborhood.
334 virtual bool ReadyToGenerate() const;
335
336 // Returns true if the neighborhood generator generates relaxation of the
337 // given problem.
338 virtual bool IsRelaxationGenerator() const { return false; }
339
340 // Uses UCB1 algorithm to compute the score (Multi armed bandit problem).
341 // Details are at
342 // https://lilianweng.github.io/lil-log/2018/01/23/the-multi-armed-bandit-problem-and-its-solutions.html.
343 // 'total_num_calls' should be the sum of calls across all generators part of
344 // the multi armed bandit problem.
345 // If the generator is called less than 10 times then the method returns
346 // infinity as score in order to get more data about the generator
347 // performance.
348 double GetUCBScore(int64_t total_num_calls) const;
349
350 // Adds solve data about one "solved" neighborhood.
351 struct SolveData {
352 // Neighborhood Id. Used to identify the neighborhood by a generator.
353 // Currently only used by WeightedRandomRelaxationNeighborhoodGenerator.
354 int64_t neighborhood_id = 0;
355
356 // The status of the sub-solve.
357 CpSolverStatus status = CpSolverStatus::UNKNOWN;
358
359 // The difficulty when this neighborhood was generated.
360 double difficulty = 0.0;
361
362 // The determinitic time limit given to the solver for this neighborhood.
363 double deterministic_limit = 0.0;
364
365 // The time it took to solve this neighborhood.
366 double deterministic_time = 0.0;
367
368 // Objective information. These only refer to the "internal" objective
369 // without scaling or offset so we are exact and it is always in the
370 // minimization direction.
371 // - The initial best objective is the one of the best known solution at the
372 // time the neighborhood was generated.
373 // - The base objective is the one of the base solution from which this
374 // neighborhood was generated.
375 // - The new objective is the objective of the best solution found by
376 // solving the neighborhood.
377 IntegerValue initial_best_objective = IntegerValue(0);
378 IntegerValue base_objective = IntegerValue(0);
379 IntegerValue new_objective = IntegerValue(0);
380
381 // Bounds data is only used by relaxation neighborhoods.
382 IntegerValue initial_best_objective_bound = IntegerValue(0);
383 IntegerValue new_objective_bound = IntegerValue(0);
384
385 // This is just used to construct a deterministic order for the updates.
386 bool operator<(const SolveData& o) const {
387 return std::tie(status, difficulty, deterministic_limit,
388 deterministic_time, initial_best_objective,
389 base_objective, new_objective,
390 initial_best_objective_bound, new_objective_bound,
391 neighborhood_id) <
392 std::tie(o.status, o.difficulty, o.deterministic_limit,
393 o.deterministic_time, o.initial_best_objective,
394 o.base_objective, o.new_objective,
395 o.initial_best_objective_bound, o.new_objective_bound,
396 o.neighborhood_id);
397 }
398 };
399 void AddSolveData(SolveData data) {
400 absl::MutexLock mutex_lock(&generator_mutex_);
401 solve_data_.push_back(data);
402 }
403
404 // Process all the recently added solve data and update this generator
405 // score and difficulty.
406 void Synchronize();
407
408 // Returns a short description of the generator.
409 std::string name() const { return name_; }
410
411 // Number of times this generator was called.
412 int64_t num_calls() const {
413 absl::MutexLock mutex_lock(&generator_mutex_);
414 return num_calls_;
415 }
416
417 // Number of time the neighborhood was fully solved (OPTIMAL/INFEASIBLE).
418 int64_t num_fully_solved_calls() const {
419 absl::MutexLock mutex_lock(&generator_mutex_);
420 return num_fully_solved_calls_;
421 }
422
423 // The current difficulty of this generator
424 double difficulty() const {
425 absl::MutexLock mutex_lock(&generator_mutex_);
426 return difficulty_.value();
427 }
428
429 // The current time limit that the sub-solve should use on this generator.
430 double deterministic_limit() const {
431 absl::MutexLock mutex_lock(&generator_mutex_);
432 return deterministic_limit_;
433 }
434
435 // The sum of the deterministic time spent in this generator.
436 double deterministic_time() const {
437 absl::MutexLock mutex_lock(&generator_mutex_);
438 return deterministic_time_;
439 }
440
441 protected:
442 // Triggered with each call to Synchronize() for each recently added
443 // SolveData. This is meant to be used for processing feedbacks by specific
444 // neighborhood generators to adjust the neighborhood generation process.
445 virtual void AdditionalProcessingOnSynchronize(const SolveData& solve_data) {}
446
447 const std::string name_;
448 const NeighborhoodGeneratorHelper& helper_;
449 mutable absl::Mutex generator_mutex_;
450
451 private:
452 std::vector<SolveData> solve_data_;
453
454 // Current parameters to be used when generating/solving a neighborhood with
455 // this generator. Only updated on Synchronize().
456 AdaptiveParameterValue difficulty_;
457 double deterministic_limit_ = 0.1;
458
459 // Current statistics of the last solved neighborhood.
460 // Only updated on Synchronize().
461 int64_t num_calls_ = 0;
462 int64_t num_fully_solved_calls_ = 0;
463 int64_t num_consecutive_non_improving_calls_ = 0;
464 double deterministic_time_ = 0.0;
465 double current_average_ = 0.0;
466};

646// Generates a neighborhood by fixing the variables to solutions reported in
647// various repositories. This is inspired from RINS published in "Exploring
648// relaxation induced neighborhoods to improve MIP solutions" 2004 by E. Danna
649// et.
650//
651// If incomplete_solutions is provided, this generates a neighborhood by fixing
652// the variable values to a solution in the SharedIncompleteSolutionManager and
653// ignores the other repositories.
654//
655// Otherwise, if response_manager is not provided, this generates a neighborhood
656// using only the linear/general relaxation values. The domain of the variables
