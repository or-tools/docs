188// Makes the string fit in one line by cutting it in the middle if necessary.
189std::string Summarize(absl::string_view input) {
190 if (input.size() < 105) return std::string(input);
191 const int half = 50;
192 return absl::StrCat(input.substr(0, half), " ... ",
193 input.substr(input.size() - half, half));
194}

209}

217std::string CpModelStats(const CpModelProto& model_proto) {
218 // Note that we only store pointer to "constant" string literals. This is
219 // slightly faster and take less space for model with millions of constraints.
220 absl::flat_hash_map<char const*, int> name_to_num_constraints;
221 absl::flat_hash_map<char const*, int> name_to_num_reified;
222 absl::flat_hash_map<char const*, int> name_to_num_multi_reified;
223 absl::flat_hash_map<char const*, int> name_to_num_literals;
224 absl::flat_hash_map<char const*, int> name_to_num_terms;
225 absl::flat_hash_map<char const*, int> name_to_num_complex_domain;
226 absl::flat_hash_map<char const*, int> name_to_num_expressions;
227
228 int no_overlap_2d_num_rectangles = 0;
229 int no_overlap_2d_num_optional_rectangles = 0;
230 int no_overlap_2d_num_linear_areas = 0;
231 int no_overlap_2d_num_quadratic_areas = 0;
232 int no_overlap_2d_num_fixed_rectangles = 0;
233
234 int cumulative_num_intervals = 0;
235 int cumulative_num_optional_intervals = 0;
236 int cumulative_num_variable_sizes = 0;
237 int cumulative_num_variable_demands = 0;
238 int cumulative_num_fixed_intervals = 0;
239
240 int no_overlap_num_intervals = 0;
241 int no_overlap_num_optional_intervals = 0;
242 int no_overlap_num_variable_sizes = 0;
243 int no_overlap_num_fixed_intervals = 0;
244
245 int num_fixed_intervals = 0;
246
247 for (const ConstraintProto& ct : model_proto.constraints()) {
248 // We split the linear constraints into 3 buckets has it gives more insight
249 // on the type of problem we are facing.
250 char const* name;
251 if (ct.constraint_case() == ConstraintProto::ConstraintCase::kLinear) {
252 if (ct.linear().vars_size() == 0) name = "kLinear0";
253 if (ct.linear().vars_size() == 1) name = "kLinear1";
254 if (ct.linear().vars_size() == 2) name = "kLinear2";
255 if (ct.linear().vars_size() == 3) name = "kLinear3";
256 if (ct.linear().vars_size() > 3) name = "kLinearN";
257 } else {
258 name = ConstraintCaseName(ct.constraint_case()).data();
259 }
260
261 name_to_num_constraints[name]++;
262 if (!ct.enforcement_literal().empty()) {
263 name_to_num_reified[name]++;
264 if (ct.enforcement_literal().size() > 1) {
265 name_to_num_multi_reified[name]++;
266 }
267 }
268
269 auto variable_is_fixed = [&model_proto](int ref) {
270 const IntegerVariableProto& proto =
271 model_proto.variables(PositiveRef(ref));
272 return proto.domain_size() == 2 && proto.domain(0) == proto.domain(1);
273 };
274
275 auto expression_is_fixed =
276 [&variable_is_fixed](const LinearExpressionProto& expr) {
277 for (const int ref : expr.vars()) {
278 if (!variable_is_fixed(ref)) {
279 return false;
280 }
281 }
282 return true;
283 };
284
285 auto interval_has_fixed_size = [&model_proto, &expression_is_fixed](int c) {
286 return expression_is_fixed(model_proto.constraints(c).interval().size());
287 };
288
289 auto constraint_is_optional = [&model_proto](int i) {
290 return !model_proto.constraints(i).enforcement_literal().empty();
291 };
292
293 auto interval_is_fixed = [&variable_is_fixed,
294 expression_is_fixed](const ConstraintProto& ct) {
295 if (ct.constraint_case() != ConstraintProto::ConstraintCase::kInterval) {
296 return false;
297 }
298 for (const int lit : ct.enforcement_literal()) {
299 if (!variable_is_fixed(lit)) return false;
300 }
301 return (expression_is_fixed(ct.interval().start()) &&
302 expression_is_fixed(ct.interval().size()) &&
303 expression_is_fixed(ct.interval().end()));
304 };
305
306 // For pure Boolean constraints, we also display the total number of literal
307 // involved as this gives a good idea of the problem size.
308 if (ct.constraint_case() == ConstraintProto::ConstraintCase::kBoolOr) {
309 name_to_num_literals[name] += ct.bool_or().literals().size();
310 } else if (ct.constraint_case() ==
311 ConstraintProto::ConstraintCase::kBoolAnd) {
312 name_to_num_literals[name] +=
313 ct.enforcement_literal().size() + ct.bool_and().literals().size();
314 } else if (ct.constraint_case() ==
315 ConstraintProto::ConstraintCase::kAtMostOne) {
316 name_to_num_literals[name] += ct.at_most_one().literals().size();
317 } else if (ct.constraint_case() ==
318 ConstraintProto::ConstraintCase::kExactlyOne) {
319 name_to_num_literals[name] += ct.exactly_one().literals().size();
320 } else if (ct.constraint_case() ==
321 ConstraintProto::ConstraintCase::kLinMax) {
322 name_to_num_expressions[name] += ct.lin_max().exprs().size();
323 } else if (ct.constraint_case() ==
324 ConstraintProto::ConstraintCase::kInterval) {
325 if (interval_is_fixed(ct)) num_fixed_intervals++;
326 } else if (ct.constraint_case() ==
327 ConstraintProto::ConstraintCase::kNoOverlap2D) {
328 const int num_boxes = ct.no_overlap_2d().x_intervals_size();
329 no_overlap_2d_num_rectangles += num_boxes;
330 for (int i = 0; i < num_boxes; ++i) {
331 const int x_interval = ct.no_overlap_2d().x_intervals(i);
332 const int y_interval = ct.no_overlap_2d().y_intervals(i);
333 if (constraint_is_optional(x_interval) ||
334 constraint_is_optional(y_interval)) {
335 no_overlap_2d_num_optional_rectangles++;
336 }
337 const int num_fixed = interval_has_fixed_size(x_interval) +
338 interval_has_fixed_size(y_interval);
339 if (num_fixed == 0) {
340 no_overlap_2d_num_quadratic_areas++;
341 } else if (num_fixed == 1) {
342 no_overlap_2d_num_linear_areas++;
343 }
344 if (interval_is_fixed(model_proto.constraints(x_interval)) &&
345 interval_is_fixed(model_proto.constraints(y_interval))) {
346 no_overlap_2d_num_fixed_rectangles++;
347 }
348 }
349 } else if (ct.constraint_case() ==
350 ConstraintProto::ConstraintCase::kNoOverlap) {
351 const int num_intervals = ct.no_overlap().intervals_size();
352 no_overlap_num_intervals += num_intervals;
353 for (int i = 0; i < num_intervals; ++i) {
354 const int interval = ct.no_overlap().intervals(i);
355 if (constraint_is_optional(interval)) {
356 no_overlap_num_optional_intervals++;
357 }
358 if (!interval_has_fixed_size(interval)) {
359 no_overlap_num_variable_sizes++;
360 }
361 if (interval_is_fixed(model_proto.constraints(interval))) {
362 no_overlap_num_fixed_intervals++;
363 }
364 }
365 } else if (ct.constraint_case() ==
366 ConstraintProto::ConstraintCase::kCumulative) {
367 const int num_intervals = ct.cumulative().intervals_size();
368 cumulative_num_intervals += num_intervals;
369 for (int i = 0; i < num_intervals; ++i) {
370 const int interval = ct.cumulative().intervals(i);
371 if (constraint_is_optional(interval)) {
372 cumulative_num_optional_intervals++;
373 }
374 if (!interval_has_fixed_size(interval)) {
375 cumulative_num_variable_sizes++;
376 }
377 if (!expression_is_fixed(ct.cumulative().demands(i))) {
378 cumulative_num_variable_demands++;
379 }
380 if (interval_is_fixed(model_proto.constraints(interval))) {
381 cumulative_num_fixed_intervals++;
382 }
383 }
384 }
385
386 if (ct.constraint_case() == ConstraintProto::ConstraintCase::kLinear &&
387 ct.linear().vars_size() > 3) {
388 name_to_num_terms[name] += ct.linear().vars_size();
389 }
390 if (ct.constraint_case() == ConstraintProto::ConstraintCase::kLinear &&
391 ct.linear().vars_size() > 1 && ct.linear().domain().size() > 2) {
392 name_to_num_complex_domain[name]++;
393 }
394 }
395
396 int num_constants = 0;
397 absl::btree_set<int64_t> constant_values;
398 absl::btree_map<Domain, int> num_vars_per_domains;
399 for (const IntegerVariableProto& var : model_proto.variables()) {
400 if (var.domain_size() == 2 && var.domain(0) == var.domain(1)) {
401 ++num_constants;
402 constant_values.insert(var.domain(0));
403 } else {
404 num_vars_per_domains[ReadDomainFromProto(var)]++;
405 }
406 }
407
408 std::string result;
409 const std::string model_fingerprint_str =
410 (absl::GetFlag(FLAGS_cp_model_fingerprint_model))
411 ? absl::StrFormat(" (model_fingerprint: %#x)",
412 FingerprintModel(model_proto))
413 : "";
414
415 if (model_proto.has_objective() ||
416 model_proto.has_floating_point_objective()) {
417 absl::StrAppend(&result, "optimization model '", model_proto.name(),
418 "':", model_fingerprint_str, "\n");
419 } else {
420 absl::StrAppend(&result, "satisfaction model '", model_proto.name(),
421 "':", model_fingerprint_str, "\n");
422 }
423
424 for (const DecisionStrategyProto& strategy : model_proto.search_strategy()) {
425 absl::StrAppend(
426 &result, "Search strategy: on ", strategy.variables_size(),
427 " variables, ",
428 ProtoEnumToString<DecisionStrategyProto::VariableSelectionStrategy>(
429 strategy.variable_selection_strategy()),
430 ", ",
431 ProtoEnumToString<DecisionStrategyProto::DomainReductionStrategy>(
432 strategy.domain_reduction_strategy()),
433 "\n");
434 }
435
436 auto count_variables_by_type =
437 [&model_proto](const google::protobuf::RepeatedField<int>& vars,
438 int* num_booleans, int* num_integers) {
439 for (const int ref : vars) {
440 const auto& var_proto = model_proto.variables(PositiveRef(ref));
441 if (var_proto.domain_size() == 2 && var_proto.domain(0) == 0 &&
442 var_proto.domain(1) == 1) {
443 (*num_booleans)++;
444 }
445 }
446 *num_integers = vars.size() - *num_booleans;
447 };
448
449 {
450 int num_boolean_variables_in_objective = 0;
451 int num_integer_variables_in_objective = 0;
452 if (model_proto.has_objective()) {
453 count_variables_by_type(model_proto.objective().vars(),
454 &num_boolean_variables_in_objective,
455 &num_integer_variables_in_objective);
456 }
457 if (model_proto.has_floating_point_objective()) {
458 count_variables_by_type(model_proto.floating_point_objective().vars(),
459 &num_boolean_variables_in_objective,
460 &num_integer_variables_in_objective);
461 }
462
463 std::vector<std::string> obj_vars_strings;
464 if (num_boolean_variables_in_objective > 0) {
465 obj_vars_strings.push_back(absl::StrCat(
466 "#bools: ", FormatCounter(num_boolean_variables_in_objective)));
467 }
468 if (num_integer_variables_in_objective > 0) {
469 obj_vars_strings.push_back(absl::StrCat(
470 "#ints: ", FormatCounter(num_integer_variables_in_objective)));
471 }
472
473 const std::string objective_string =
474 model_proto.has_objective()
475 ? absl::StrCat(" (", absl::StrJoin(obj_vars_strings, " "),
476 " in objective)")
477 : (model_proto.has_floating_point_objective()
478 ? absl::StrCat(" (", absl::StrJoin(obj_vars_strings, " "),
479 " in floating point objective)")
480 : "");
481 absl::StrAppend(&result,
482 "#Variables: ", FormatCounter(model_proto.variables_size()),
483 objective_string, "\n");
484 }
485 if (num_vars_per_domains.contains(Domain(0, 1))) {
486 // We always list Boolean first.
487 const int num_bools = num_vars_per_domains[Domain(0, 1)];
488 const std::string temp =
489 absl::StrCat(" - ", FormatCounter(num_bools), " Booleans in ",
490 Domain(0, 1).ToString(), "\n");
491 absl::StrAppend(&result, Summarize(temp));
492 num_vars_per_domains.erase(Domain(0, 1));
493 }
494 if (num_vars_per_domains.size() < 100) {
495 for (const auto& entry : num_vars_per_domains) {
496 const std::string temp =
497 absl::StrCat(" - ", FormatCounter(entry.second), " in ",
498 entry.first.ToString(), "\n");
499 absl::StrAppend(&result, Summarize(temp));
500 }
501 } else {
502 int64_t max_complexity = 0;
503 int64_t min = std::numeric_limits<int64_t>::max();
504 int64_t max = std::numeric_limits<int64_t>::min();
505 for (const auto& entry : num_vars_per_domains) {
506 min = std::min(min, entry.first.Min());
507 max = std::max(max, entry.first.Max());
508 max_complexity = std::max(
509 max_complexity, static_cast<int64_t>(entry.first.NumIntervals()));
510 }
511 absl::StrAppend(&result, " - ", FormatCounter(num_vars_per_domains.size()),
512 " different domains in [", min, ",", max,
513 "] with a largest complexity of ", max_complexity, ".\n");
514 }
515
516 if (num_constants > 0) {
517 const std::string temp =
518 absl::StrCat(" - ", num_constants, " constants in {",
519 absl::StrJoin(constant_values, ","), "} \n");
520 absl::StrAppend(&result, Summarize(temp));
521 }
522
523 std::vector<std::string> constraints;
524 constraints.reserve(name_to_num_constraints.size());
525 for (const auto& [c_name, count] : name_to_num_constraints) {
526 const std::string name(c_name);
527 constraints.push_back(absl::StrCat("#", name, ": ", FormatCounter(count)));
528 if (name_to_num_reified.contains(c_name)) {
529 if (name_to_num_multi_reified.contains(c_name)) {
530 absl::StrAppend(
531 &constraints.back(),
532 " (#enforced: ", FormatCounter(name_to_num_reified[c_name]),
533 " #multi: ", FormatCounter(name_to_num_multi_reified[c_name]), ")");
534 } else {
535 absl::StrAppend(&constraints.back(), " (#enforced: ",
536 FormatCounter(name_to_num_reified[c_name]), ")");
537 }
538 }
539 if (name_to_num_literals.contains(c_name)) {
540 absl::StrAppend(&constraints.back(), " (#literals: ",
541 FormatCounter(name_to_num_literals[c_name]), ")");
542 }
543 if (name_to_num_terms.contains(c_name)) {
544 absl::StrAppend(&constraints.back(),
545 " (#terms: ", FormatCounter(name_to_num_terms[c_name]),
546 ")");
547 }
548 if (name_to_num_expressions.contains(c_name)) {
549 absl::StrAppend(&constraints.back(), " (#expressions: ",
550 FormatCounter(name_to_num_expressions[c_name]), ")");
551 }
552 if (name_to_num_complex_domain.contains(c_name)) {
553 absl::StrAppend(&constraints.back(), " (#complex_domain: ",
554 FormatCounter(name_to_num_complex_domain[c_name]), ")");
555 }
556 if (name == "kInterval" && num_fixed_intervals > 0) {
557 absl::StrAppend(&constraints.back(), " (#fixed: ", num_fixed_intervals,
558 ")");
559 } else if (name == "kNoOverlap2D") {
560 absl::StrAppend(&constraints.back(),
561 " (#rectangles: ", no_overlap_2d_num_rectangles);
562 if (no_overlap_2d_num_optional_rectangles > 0) {

695// This both copy the "main" DebugSolution to a local_model and also cache
696// the value of the integer variables in that solution.
697void InitializeDebugSolution(const CpModelProto& model_proto, Model* model) {
698 auto* shared_response = model->Get<SharedResponseManager>();
699 if (shared_response == nullptr) return;
700 if (shared_response->DebugSolution().empty()) return;
701
702 // Copy the proto values.
703 DebugSolution& debug_sol = *model->GetOrCreate<DebugSolution>();
704 debug_sol.proto_values = shared_response->DebugSolution();
705
706 // Fill the values by integer variable.
707 const int num_integers =
708 model->GetOrCreate<IntegerTrail>()->NumIntegerVariables().value();
709 debug_sol.ivar_has_value.assign(num_integers, false);
710 debug_sol.ivar_values.assign(num_integers, 0);
711
712 const auto& mapping = *model->GetOrCreate<CpModelMapping>();
713 for (int i = 0; i < debug_sol.proto_values.size(); ++i) {
714 if (!mapping.IsInteger(i)) continue;
715 const IntegerVariable var = mapping.Integer(i);
716 debug_sol.ivar_has_value[var] = true;
717 debug_sol.ivar_has_value[NegationOf(var)] = true;
718 debug_sol.ivar_values[var] = debug_sol.proto_values[i];
719 debug_sol.ivar_values[NegationOf(var)] = -debug_sol.proto_values[i];
720 }
721
722 // The objective variable is usually not part of the proto, but it is still
723 // nice to have it, so we recompute it here.
724 auto* objective_def = model->Get<ObjectiveDefinition>();
725 if (objective_def != nullptr) {
726 const IntegerVariable objective_var = objective_def->objective_var;
727 const int64_t objective_value =
728 ComputeInnerObjective(model_proto.objective(), debug_sol.proto_values);
729 debug_sol.ivar_has_value[objective_var] = true;
730 debug_sol.ivar_has_value[NegationOf(objective_var)] = true;
731 debug_sol.ivar_values[objective_var] = objective_value;
732 debug_sol.ivar_values[NegationOf(objective_var)] = -objective_value;
733 }
734
735 // We also register a DEBUG callback to check our reasons.
736 auto* encoder = model->GetOrCreate<IntegerEncoder>();
737 const auto checker = [mapping, encoder, debug_sol, model](
738 absl::Span<const Literal> clause,
739 absl::Span<const IntegerLiteral> integers) {
740 bool is_satisfied = false;
741 int num_bools = 0;
742 int num_ints = 0;
743 std::vector<std::tuple<Literal, IntegerLiteral, int>> to_print;
744 for (const Literal l : clause) {
745 // First case, this Boolean is mapped.
746 {
747 const int proto_var =
748 mapping.GetProtoVariableFromBooleanVariable(l.Variable());
749 if (proto_var != -1) {
750 to_print.push_back({l, IntegerLiteral(), proto_var});
751 if (debug_sol.proto_values[proto_var] == (l.IsPositive() ? 1 : 0)) {
752 is_satisfied = true;
753 break;
754 }
755 ++num_bools;
756 continue;
757 }
758 }
759
760 // Second case, it is associated to IntVar >= value.
761 // We can use any of them, so if one is false, we use this one.
762 bool all_true = true;
763 for (const IntegerLiteral associated : encoder->GetIntegerLiterals(l)) {
764 const int proto_var = mapping.GetProtoVariableFromIntegerVariable(
765 PositiveVariable(associated.var));
766 if (proto_var == -1) break;
767 int64_t value = debug_sol.proto_values[proto_var];
768 to_print.push_back({l, associated, proto_var});
769
770 if (!VariableIsPositive(associated.var)) value = -value;
771 if (value < associated.bound) {
772 ++num_ints;
773 all_true = false;
774 break;
775 }
776 }
777 if (all_true) {
778 is_satisfied = true;
779 break;
780 }
781 }
782 for (const IntegerLiteral i_lit : integers) {
783 const int proto_var = mapping.GetProtoVariableFromIntegerVariable(
784 PositiveVariable(i_lit.var));
785 if (proto_var == -1) {
786 is_satisfied = true;
787 break;
788 }
789
790 int64_t value = debug_sol.proto_values[proto_var];
791 to_print.push_back({Literal(kNoLiteralIndex), i_lit, proto_var});
792
793 if (!VariableIsPositive(i_lit.var)) value = -value;
794 // Note the sign is inversed, we cannot have all literal false and all
795 // integer literal true.
796 if (value >= i_lit.bound) {
797 is_satisfied = true;
798 break;
799 }
800 }
801 if (!is_satisfied) {
802 LOG(INFO) << "Reason clause is not satisfied by loaded solution:";
803 LOG(INFO) << "Worker '" << model->Name() << "', level="
804 << model->GetOrCreate<SatSolver>()->CurrentDecisionLevel();
805 LOG(INFO) << "literals (neg): " << clause;
806 LOG(INFO) << "integer literals: " << integers;
807 for (const auto [l, i_lit, proto_var] : to_print) {
808 LOG(INFO) << l << " " << i_lit << " var=" << proto_var
809 << " value_in_sol=" << debug_sol.proto_values[proto_var];
810 }
811 }
812 return is_satisfied;
813 };
814 const auto lit_checker = [checker](absl::Span<const Literal> clause) {
815 return checker(clause, {});
816 };
817
818 model->GetOrCreate<Trail>()->RegisterDebugChecker(lit_checker);
819 model->GetOrCreate<IntegerTrail>()->RegisterDebugChecker(checker);
820}

851 }

857IntegerVariable GetOrCreateVariableWithTightBound(
858 const std::vector<std::pair<IntegerVariable, int64_t>>& terms,
859 Model* model) {
860 if (terms.empty()) return model->Add(ConstantIntegerVariable(0));
861 if (terms.size() == 1 && terms.front().second == 1) {
862 return terms.front().first;
863 }
864 if (terms.size() == 1 && terms.front().second == -1) {
865 return NegationOf(terms.front().first);
866 }
867
868 int64_t sum_min = 0;
869 int64_t sum_max = 0;
870 for (const std::pair<IntegerVariable, int64_t>& var_coeff : terms) {
871 const int64_t min_domain = model->Get(LowerBound(var_coeff.first));
872 const int64_t max_domain = model->Get(UpperBound(var_coeff.first));
873 const int64_t coeff = var_coeff.second;
874 const int64_t prod1 = min_domain * coeff;
875 const int64_t prod2 = max_domain * coeff;
876 sum_min += std::min(prod1, prod2);
877 sum_max += std::max(prod1, prod2);
878 }
879 return model->Add(NewIntegerVariable(sum_min, sum_max));
880}
881
882IntegerVariable GetOrCreateVariableLinkedToSumOf(
883 const std::vector<std::pair<IntegerVariable, int64_t>>& terms,
884 bool use_equality, Model* model) {
885 if (terms.empty()) return model->Add(ConstantIntegerVariable(0));
886 if (terms.size() == 1 && terms.front().second == 1) {
887 return terms.front().first;
888 }
889 if (terms.size() == 1 && terms.front().second == -1) {
890 return NegationOf(terms.front().first);
891 }
892
893 // Add var == terms or var >= terms if use_equality = false.
894 const IntegerVariable new_var =
895 GetOrCreateVariableWithTightBound(terms, model);
896
897 // TODO(user): use the same format, i.e. LinearExpression in both code!
898 std::vector<IntegerVariable> vars;
899 std::vector<int64_t> coeffs;
900 for (const auto [var, coeff] : terms) {
901 vars.push_back(var);
902 coeffs.push_back(coeff);
903 }
904 vars.push_back(new_var);
905 coeffs.push_back(-1);
906
907 // We want == 0 or <= 0 if use_equality = false.
908 const bool lb_required = use_equality;
909 const bool ub_required = true;
910
911 // Split if linear is large.
912 if (vars.size() > model->GetOrCreate<SatParameters>()->linear_split_size()) {
913 SplitAndLoadIntermediateConstraints(lb_required, ub_required, &vars,
914 &coeffs, model);
915 }
916
917 // Load the top-level constraint.
918 if (lb_required) {
919 model->Add(WeightedSumGreaterOrEqual(vars, coeffs, 0));
920 }
921 if (ub_required) {
922 model->Add(WeightedSumLowerOrEqual(vars, coeffs, 0));
923 }
924
925 return new_var;
926}

930// Adds one LinearProgrammingConstraint per connected component of the model.
931IntegerVariable AddLPConstraints(bool objective_need_to_be_tight,
932 const CpModelProto& model_proto, Model* m) {
933 const LinearRelaxation relaxation = ComputeLinearRelaxation(model_proto, m);
934
935 // The bipartite graph of LP constraints might be disconnected:
936 // make a partition of the variables into connected components.
937 // Constraint nodes are indexed by [0..num_lp_constraints),
938 // variable nodes by [num_lp_constraints..num_lp_constraints+num_variables).
939 //
940 // TODO(user): look into biconnected components.
941 const int num_lp_constraints =
942 static_cast<int>(relaxation.linear_constraints.size());
943 const int num_lp_cut_generators =
944 static_cast<int>(relaxation.cut_generators.size());
945 const int num_integer_variables =
946 m->GetOrCreate<IntegerTrail>()->NumIntegerVariables().value();
947
948 DenseConnectedComponentsFinder components;
949 components.SetNumberOfNodes(num_lp_constraints + num_lp_cut_generators +
950 num_integer_variables);
951 auto get_constraint_index = [](int ct_index) { return ct_index; };
952 auto get_cut_generator_index = [num_lp_constraints](int cut_index) {
953 return num_lp_constraints + cut_index;
954 };
955 auto get_var_index = [num_lp_constraints,
956 num_lp_cut_generators](IntegerVariable var) {
957 return num_lp_constraints + num_lp_cut_generators +
958 PositiveVariable(var).value();
959 };
960 for (int i = 0; i < num_lp_constraints; i++) {
961 for (const IntegerVariable var : relaxation.linear_constraints[i].vars) {
962 components.AddEdge(get_constraint_index(i), get_var_index(var));
963 }
964 }
965 for (int i = 0; i < num_lp_cut_generators; ++i) {
966 for (const IntegerVariable var : relaxation.cut_generators[i].vars) {
967 components.AddEdge(get_cut_generator_index(i), get_var_index(var));
968 }
969 }
970
971 const int num_components = components.GetNumberOfComponents();
972 std::vector<int> component_sizes(num_components, 0);
973 const std::vector<int> index_to_component = components.GetComponentIds();
974 for (int i = 0; i < num_lp_constraints; i++) {
975 ++component_sizes[index_to_component[get_constraint_index(i)]];
976 }
977 for (int i = 0; i < num_lp_cut_generators; i++) {
978 ++component_sizes[index_to_component[get_cut_generator_index(i)]];
979 }
980
981 // TODO(user): Optimize memory layout.
982 std::vector<std::vector<IntegerVariable>> component_to_var(num_components);
983 for (IntegerVariable var(0); var < num_integer_variables; var += 2) {
984 DCHECK(VariableIsPositive(var));
985 component_to_var[index_to_component[get_var_index(var)]].push_back(var);
986 }
987
988 // Make sure any constraint that touch the objective is not discarded even
989 // if it is the only one in its component. This is important to propagate
990 // as much as possible the objective bound by using any bounds the LP give
991 // us on one of its components. This is critical on the zephyrus problems for
992 // instance.
993 auto* mapping = m->GetOrCreate<CpModelMapping>();
994 for (int i = 0; i < model_proto.objective().coeffs_size(); ++i) {
995 const IntegerVariable var =
996 mapping->Integer(model_proto.objective().vars(i));
997 ++component_sizes[index_to_component[get_var_index(var)]];
998 }
999
1000 // Dispatch every constraint to its LinearProgrammingConstraint.
1001 std::vector<LinearProgrammingConstraint*> lp_constraints(num_components,
1002 nullptr);
1003 std::vector<std::vector<LinearConstraint>> component_to_constraints(
1004 num_components);
1005 for (int i = 0; i < num_lp_constraints; i++) {
1006 const int c = index_to_component[get_constraint_index(i)];
1007 if (component_sizes[c] <= 1) continue;
1008 component_to_constraints[c].push_back(relaxation.linear_constraints[i]);
1009 if (lp_constraints[c] == nullptr) {
1010 lp_constraints[c] =
1011 new LinearProgrammingConstraint(m, component_to_var[c]);
1012 m->TakeOwnership(lp_constraints[c]);
1013 }
1014 // Load the constraint.
1015 lp_constraints[c]->AddLinearConstraint(relaxation.linear_constraints[i]);
1016 }
1017
1018 // Dispatch every cut generator to its LinearProgrammingConstraint.
1019 for (int i = 0; i < num_lp_cut_generators; i++) {
1020 const int c = index_to_component[get_cut_generator_index(i)];
1021 if (lp_constraints[c] == nullptr) {
1022 lp_constraints[c] =
1023 new LinearProgrammingConstraint(m, component_to_var[c]);
1024 m->TakeOwnership(lp_constraints[c]);
1025 }
1026 lp_constraints[c]->AddCutGenerator(std::move(relaxation.cut_generators[i]));
1027 }
1028
1029 // Add the objective.
1030 std::vector<std::vector<std::pair<IntegerVariable, int64_t>>>
1031 component_to_cp_terms(num_components);
1032 std::vector<std::pair<IntegerVariable, int64_t>> top_level_cp_terms;
1033 int num_components_containing_objective = 0;
1034 if (model_proto.has_objective()) {
1035 // First pass: set objective coefficients on the lp constraints, and store
1036 // the cp terms in one vector per component.
1037 for (int i = 0; i < model_proto.objective().coeffs_size(); ++i) {
1038 const IntegerVariable var =
1039 mapping->Integer(model_proto.objective().vars(i));
1040 const int64_t coeff = model_proto.objective().coeffs(i);
1041 const int c = index_to_component[get_var_index(var)];
1042 if (lp_constraints[c] != nullptr) {
1043 lp_constraints[c]->SetObjectiveCoefficient(var, IntegerValue(coeff));
1044 component_to_cp_terms[c].push_back(std::make_pair(var, coeff));
1045 } else {
1046 // Component is too small. We still need to store the objective term.
1047 top_level_cp_terms.push_back(std::make_pair(var, coeff));
1048 }
1049 }
1050 // Second pass: Build the cp sub-objectives per component.
1051 for (int c = 0; c < num_components; ++c) {
1052 if (component_to_cp_terms[c].empty()) continue;
1053 const IntegerVariable sub_obj_var = GetOrCreateVariableLinkedToSumOf(
1054 component_to_cp_terms[c], objective_need_to_be_tight, m);
1055 top_level_cp_terms.push_back(std::make_pair(sub_obj_var, 1));
1056 lp_constraints[c]->SetMainObjectiveVariable(sub_obj_var);
1057 num_components_containing_objective++;
1058 }
1059 }
1060
1061 const IntegerVariable main_objective_var =
1062 model_proto.has_objective()
1063 ? GetOrCreateVariableLinkedToSumOf(top_level_cp_terms,
1064 objective_need_to_be_tight, m)
1065 : kNoIntegerVariable;
1066
1067 // Register LP constraints. Note that this needs to be done after all the
1068 // constraints have been added.
1069 for (LinearProgrammingConstraint* lp_constraint : lp_constraints) {
1070 if (lp_constraint == nullptr) continue;
1071 lp_constraint->RegisterWith(m);
1072 VLOG(3) << "LP constraint: " << lp_constraint->DimensionString() << ".";
1073 }
1074
1075 VLOG(3) << top_level_cp_terms.size()
1076 << " terms in the main objective linear equation ("
1077 << num_components_containing_objective << " from LP constraints).";
1078 return main_objective_var;
1079}

1141 absl::flat_hash_set<int> visited_variables;
1142
1143 auto broadcast_level_zero_bounds =
1144 [=](const std::vector<IntegerVariable>& modified_vars) mutable {
1145 // Inspect the modified IntegerVariables.
1146 for (const IntegerVariable& var : modified_vars) {
1147 const IntegerVariable positive_var = PositiveVariable(var);
1148 const int model_var =
1149 mapping->GetProtoVariableFromIntegerVariable(positive_var);
1150
1151 if (model_var == -1) continue;
1152 const auto [_, inserted] = visited_variables.insert(model_var);
1153 if (!inserted) continue;
1154
1155 const int64_t new_lb =
1156 integer_trail->LevelZeroLowerBound(positive_var).value();
1157 const int64_t new_ub =
1158 integer_trail->LevelZeroUpperBound(positive_var).value();
1159
1160 // TODO(user): We could imagine an API based on atomic<int64_t>
1161 // that could preemptively check if this new bounds are improving.
1162 model_variables.push_back(model_var);
1163 new_lower_bounds.push_back(new_lb);
1164 new_upper_bounds.push_back(new_ub);
1165 }
1166
1167 // Inspect the newly modified Booleans.
1168 for (; saved_trail_index < trail->Index(); ++saved_trail_index) {
1169 const Literal fixed_literal = (*trail)[saved_trail_index];
1170 const int model_var = mapping->GetProtoVariableFromBooleanVariable(
1171 fixed_literal.Variable());
1172
1173 if (model_var == -1) continue;
1174 const auto [_, inserted] = visited_variables.insert(model_var);
1175 if (!inserted) continue;
1176
1177 model_variables.push_back(model_var);
1178 if (fixed_literal.IsPositive()) {
1179 new_lower_bounds.push_back(1);
1180 new_upper_bounds.push_back(1);
1181 } else {
1182 new_lower_bounds.push_back(0);
1183 new_upper_bounds.push_back(0);
1184 }
1185 }
1186
1187 if (!model_variables.empty()) {
1188 shared_bounds_manager->ReportPotentialNewBounds(
1189 model->Name(), model_variables, new_lower_bounds,
1190 new_upper_bounds);
1191
1192 // Clear for next call.
1193 model_variables.clear();
1194 new_lower_bounds.clear();
1195 new_upper_bounds.clear();
1196 visited_variables.clear();
1197
1198 // If we are not in interleave_search we synchronize right away.
1199 if (!model->Get<SatParameters>()->interleave_search()) {
1200 shared_bounds_manager->Synchronize();
1201 }
1202 }
1203 };
1204
1205 // The callback will just be called on NEWLY modified var. So initially,
1206 // we do want to read all variables.
1207 //
1208 // TODO(user): Find a better way? It seems nicer to register this before
1209 // any variable is modified. But then we don't want to call it each time
1210 // we reach level zero during probing. It should be better to only call
1211 // it when a new variable has been fixed.
1212 const IntegerVariable num_vars =
1213 model->GetOrCreate<IntegerTrail>()->NumIntegerVariables();
1214 std::vector<IntegerVariable> all_variables;
1215 all_variables.reserve(num_vars.value());
1216 for (IntegerVariable var(0); var < num_vars; ++var) {
1217 all_variables.push_back(var);
1218 }
1219 broadcast_level_zero_bounds(all_variables);
1220
1221 model->GetOrCreate<GenericLiteralWatcher>()
1222 ->RegisterLevelZeroModifiedVariablesCallback(broadcast_level_zero_bounds);
1223}

1225// Registers a callback to import new variables bounds stored in the
1226// shared_bounds_manager. These bounds are imported at level 0 of the search
1227// in the linear scan minimize function.
1228void RegisterVariableBoundsLevelZeroImport(
1229 const CpModelProto& model_proto, SharedBoundsManager* shared_bounds_manager,
1230 Model* model) {
1231 CHECK(shared_bounds_manager != nullptr);
1232 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
1233 CpModelMapping* const mapping = model->GetOrCreate<CpModelMapping>();
1234 const int id = shared_bounds_manager->RegisterNewId();
1235
1236 const auto& import_level_zero_bounds = [&model_proto, shared_bounds_manager,
1237 model, integer_trail, id, mapping]() {
1238 std::vector<int> model_variables;
1239 std::vector<int64_t> new_lower_bounds;
1240 std::vector<int64_t> new_upper_bounds;
1241 shared_bounds_manager->GetChangedBounds(
1242 id, &model_variables, &new_lower_bounds, &new_upper_bounds);
1243 bool new_bounds_have_been_imported = false;
1244 for (int i = 0; i < model_variables.size(); ++i) {
1245 const int model_var = model_variables[i];
1246 // This can happen if a boolean variables is forced to have an
1247 // integer view in one thread, and not in another thread.
1248 if (!mapping->IsInteger(model_var)) continue;
1249 const IntegerVariable var = mapping->Integer(model_var);
1250 const IntegerValue new_lb(new_lower_bounds[i]);
1251 const IntegerValue new_ub(new_upper_bounds[i]);
1252 const IntegerValue old_lb = integer_trail->LowerBound(var);
1253 const IntegerValue old_ub = integer_trail->UpperBound(var);
1254 const bool changed_lb = new_lb > old_lb;
1255 const bool changed_ub = new_ub < old_ub;
1256 if (!changed_lb && !changed_ub) continue;
1257
1258 new_bounds_have_been_imported = true;
1259 if (VLOG_IS_ON(3)) {
1260 const IntegerVariableProto& var_proto =
1261 model_proto.variables(model_var);
1262 const std::string& var_name =
1263 var_proto.name().empty()
1264 ? absl::StrCat("anonymous_var(", model_var, ")")
1265 : var_proto.name();
1266 LOG(INFO) << " '" << model->Name() << "' imports new bounds for "
1267 << var_name << ": from [" << old_lb << ", " << old_ub
1268 << "] to [" << new_lb << ", " << new_ub << "]";
1269 }
1270
1271 if (changed_lb &&
1272 !integer_trail->Enqueue(IntegerLiteral::GreaterOrEqual(var, new_lb),
1273 {}, {})) {
1274 return false;
1275 }
1276 if (changed_ub &&
1277 !integer_trail->Enqueue(IntegerLiteral::LowerOrEqual(var, new_ub), {},
1278 {})) {
1279 return false;
1280 }
1281 }
1282 if (new_bounds_have_been_imported &&
1283 !model->GetOrCreate<SatSolver>()->FinishPropagation()) {
1284 return false;
1285 }
1286 return true;
1287 };
1288 model->GetOrCreate<LevelZeroCallbackHelper>()->callbacks.push_back(
1289 import_level_zero_bounds);
1290}

1426void LoadBaseModel(const CpModelProto& model_proto, Model* model) {
1427 auto* shared_response_manager = model->GetOrCreate<SharedResponseManager>();
1428 CHECK(shared_response_manager != nullptr);
1429 auto* sat_solver = model->GetOrCreate<SatSolver>();
1430
1431 // Simple function for the few places where we do "return unsat()".
1432 const auto unsat = [shared_response_manager, sat_solver, model] {
1433 sat_solver->NotifyThatModelIsUnsat();
1434 shared_response_manager->NotifyThatImprovingProblemIsInfeasible(
1435 absl::StrCat(model->Name(), " [loading]"));
1436 };
1437
1438 // We will add them all at once after model_proto is loaded.
1439 model->GetOrCreate<IntegerEncoder>()->DisableImplicationBetweenLiteral();
1440
1441 auto* mapping = model->GetOrCreate<CpModelMapping>();
1442 const SatParameters& parameters = *(model->GetOrCreate<SatParameters>());
1443 const bool view_all_booleans_as_integers =
1444 (parameters.linearization_level() >= 2) ||
1445 (parameters.search_branching() == SatParameters::FIXED_SEARCH &&
1446 model_proto.search_strategy().empty()) ||
1447 parameters.optimize_with_max_hs();
1448 LoadVariables(model_proto, view_all_booleans_as_integers, model);
1449 DetectOptionalVariables(model_proto, model);
1450
1451 // TODO(user): The core algo and symmetries seems to be problematic in some
1452 // cases. See for instance: neos-691058.mps.gz. This is probably because as
1453 // we modify the model, our symmetry might be wrong? investigate.
1454 //
1455 // TODO(user): More generally, we cannot load the symmetry if we create
1456 // new Booleans and constraints that link them to some Booleans of the model.
1457 // Creating Booleans related to integer variable is fine since we only deal
1458 // with Boolean only symmetry here. It is why we disable this when we have
1459 // linear relaxation as some of them create new constraints.
1460 if (!parameters.optimize_with_core() && parameters.symmetry_level() > 1 &&
1461 !parameters.enumerate_all_solutions() &&
1462 parameters.linearization_level() == 0) {
1463 LoadBooleanSymmetries(model_proto, model);
1464 }
1465
1466 ExtractEncoding(model_proto, model);
1467 PropagateEncodingFromEquivalenceRelations(model_proto, model);
1468
1469 // Check the model is still feasible before continuing.
1470 if (sat_solver->ModelIsUnsat()) return unsat();
1471
1472 // Fully encode variables as needed by the search strategy.
1473 AddFullEncodingFromSearchBranching(model_proto, model);
1474
1475 // Reserve space for the precedence relations.
1476 model->GetOrCreate<PrecedenceRelations>()->Resize(
1477 model->GetOrCreate<IntegerTrail>()->NumIntegerVariables().value());
1478
1479 // Load the constraints.
1480 int num_ignored_constraints = 0;
1481 absl::flat_hash_set<ConstraintProto::ConstraintCase> unsupported_types;
1482 for (const ConstraintProto& ct : model_proto.constraints()) {
1483 if (mapping->ConstraintIsAlreadyLoaded(&ct)) {
1484 ++num_ignored_constraints;
1485 continue;
1486 }
1487
1488 if (!LoadConstraint(ct, model)) {
1489 unsupported_types.insert(ct.constraint_case());
1490 continue;
1491 }
1492
1493 // We propagate after each new Boolean constraint but not the integer
1494 // ones. So we call FinishPropagation() manually here.
1495 //
1496 // Note that we only do that in debug mode as this can be really slow on
1497 // certain types of problems with millions of constraints.
1498 if (DEBUG_MODE) {
1499 if (sat_solver->FinishPropagation()) {
1500 Trail* trail = model->GetOrCreate<Trail>();
1501 const int old_num_fixed = trail->Index();
1502 if (trail->Index() > old_num_fixed) {
1503 VLOG(3) << "Constraint fixed " << trail->Index() - old_num_fixed
1504 << " Boolean variable(s): " << ProtobufDebugString(ct);
1505 }
1506 }
1507 }
1508 if (sat_solver->ModelIsUnsat()) {
1509 VLOG(2) << "UNSAT during extraction (after adding '"
1510 << ConstraintCaseName(ct.constraint_case()) << "'). "
1511 << ProtobufDebugString(ct);
1512 break;
1513 }
1514 }
1515 if (num_ignored_constraints > 0) {
1516 VLOG(3) << num_ignored_constraints << " constraints were skipped.";
1517 }
1518 if (!unsupported_types.empty()) {
1519 VLOG(1) << "There is unsupported constraints types in this model: ";
1520 std::vector<absl::string_view> names;
1521 for (const ConstraintProto::ConstraintCase type : unsupported_types) {
1522 names.push_back(ConstraintCaseName(type));
1523 }
1524 std::sort(names.begin(), names.end());
1525 for (const absl::string_view name : names) {
1526 VLOG(1) << " - " << name;
1527 }
1528 return unsat();
1529 }
1530
1531 model->GetOrCreate<IntegerEncoder>()
1532 ->AddAllImplicationsBetweenAssociatedLiterals();
1533 if (!sat_solver->FinishPropagation()) return unsat();
1534
1535 model->GetOrCreate<ProductDetector>()->ProcessImplicationGraph(
1536 model->GetOrCreate<BinaryImplicationGraph>());
1537}

1539void LoadFeasibilityPump(const CpModelProto& model_proto, Model* model) {
1540 LoadBaseModel(model_proto, model);
1541
1542 auto* mapping = model->GetOrCreate<CpModelMapping>();
1543 const SatParameters& parameters = *(model->GetOrCreate<SatParameters>());
1544 if (parameters.linearization_level() == 0) return;
1545
1546 // Add linear constraints to Feasibility Pump.
1547 const LinearRelaxation relaxation =
1548 ComputeLinearRelaxation(model_proto, model);
1549 const int num_lp_constraints =
1550 static_cast<int>(relaxation.linear_constraints.size());
1551 if (num_lp_constraints == 0) return;
1552 auto* feasibility_pump = model->GetOrCreate<FeasibilityPump>();
1553 for (int i = 0; i < num_lp_constraints; i++) {
1554 feasibility_pump->AddLinearConstraint(relaxation.linear_constraints[i]);
1555 }
1556
1557 if (model_proto.has_objective()) {
1558 for (int i = 0; i < model_proto.objective().coeffs_size(); ++i) {
1559 const IntegerVariable var =
1560 mapping->Integer(model_proto.objective().vars(i));
1561 const int64_t coeff = model_proto.objective().coeffs(i);
1562 feasibility_pump->SetObjectiveCoefficient(var, IntegerValue(coeff));
1563 }
1564 }
1565}

1567// Loads a CpModelProto inside the given model.
1568// This should only be called once on a given 'Model' class.
1569//
1570// TODO(user): move to cp_model_loader.h/.cc
1571void LoadCpModel(const CpModelProto& model_proto, Model* model) {
1572 LoadBaseModel(model_proto, model);
1573
1574 // We want to load the debug solution before the initial propag.
1575 // But at this point the objective is not loaded yet, so we will not have
1576 // a value for the objective integer variable, so we do it again later.
1577 InitializeDebugSolution(model_proto, model);
1578
1579 // Simple function for the few places where we do "return unsat()".
1580 auto* sat_solver = model->GetOrCreate<SatSolver>();
1581 auto* shared_response_manager = model->GetOrCreate<SharedResponseManager>();
1582 const auto unsat = [shared_response_manager, sat_solver, model] {
1583 sat_solver->NotifyThatModelIsUnsat();
1584 shared_response_manager->NotifyThatImprovingProblemIsInfeasible(
1585 absl::StrCat(model->Name(), " [loading]"));
1586 };
1587
1588 auto* mapping = model->GetOrCreate<CpModelMapping>();
1589 const SatParameters& parameters = *(model->GetOrCreate<SatParameters>());
1590
1591 // Auto detect "at least one of" constraints in the PrecedencesPropagator.
1592 // Note that we do that before we finish loading the problem (objective and
1593 // LP relaxation), because propagation will be faster at this point and it
1594 // should be enough for the purpose of this auto-detection.
1595 if (model->Mutable<PrecedencesPropagator>() != nullptr &&
1596 parameters.auto_detect_greater_than_at_least_one_of()) {
1597 model->Mutable<PrecedencesPropagator>()
1598 ->AddGreaterThanAtLeastOneOfConstraints(model);
1599 if (!sat_solver->FinishPropagation()) return unsat();
1600 }
1601
1602 // Note that this is already done in the presolve, but it is important to redo
1603 // it here to collect literal => integer >= bound constraints that are used in
1604 // many places. Without it, we don't detect them if they depends on long chain
1605 // of implications.
1606 //
1607 // TODO(user): We don't have a good deterministic time on all constraints,
1608 // so this might take more time than wanted.
1609 if (parameters.cp_model_probing_level() > 1) {
1610 Prober* prober = model->GetOrCreate<Prober>();
1611 prober->ProbeBooleanVariables(/*deterministic_time_limit=*/1.0);
1612 if (!model->GetOrCreate<BinaryImplicationGraph>()
1613 ->ComputeTransitiveReduction()) {
1614 return unsat();
1615 }
1616 }
1617 if (sat_solver->ModelIsUnsat()) return unsat();
1618
1619 // Note that it is important to do that after the probing.
1620 ExtractElementEncoding(model_proto, model);
1621
1622 // Compute decomposed energies on demands helper.
1623 IntervalsRepository* repository = model->Mutable<IntervalsRepository>();
1624 if (repository != nullptr) {
1625 repository->InitAllDecomposedEnergies();
1626 }
1627
1628 // We need to know beforehand if the objective var can just be >= terms or
1629 // needs to be == terms.
1630 bool objective_need_to_be_tight = false;
1631 if (model_proto.has_objective() &&
1632 !model_proto.objective().domain().empty()) {
1633 int64_t min_value = 0;
1634 int64_t max_value = 0;
1635 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
1636 const CpObjectiveProto& obj = model_proto.objective();
1637 for (int i = 0; i < obj.vars_size(); ++i) {
1638 const int64_t coeff = obj.coeffs(i);
1639 const IntegerVariable var = mapping->Integer(obj.vars(i));
1640 if (coeff > 0) {
1641 min_value += coeff * integer_trail->LowerBound(var).value();
1642 max_value += coeff * integer_trail->UpperBound(var).value();
1643 } else {
1644 min_value += coeff * integer_trail->UpperBound(var).value();
1645 max_value += coeff * integer_trail->LowerBound(var).value();
1646 }
1647 }
1648 const Domain user_domain = ReadDomainFromProto(model_proto.objective());
1649 const Domain automatic_domain = Domain(min_value, max_value);
1650 objective_need_to_be_tight = !automatic_domain.IsIncludedIn(user_domain);
1651 }
1652
1653 // Create an objective variable and its associated linear constraint if
1654 // needed.
1655 IntegerVariable objective_var = kNoIntegerVariable;
1656 if (parameters.linearization_level() > 0) {
1657 // Linearize some part of the problem and register LP constraint(s).
1658 objective_var =
1659 AddLPConstraints(objective_need_to_be_tight, model_proto, model);
1660 } else if (model_proto.has_objective()) {
1661 const CpObjectiveProto& obj = model_proto.objective();
1662 std::vector<std::pair<IntegerVariable, int64_t>> terms;
1663 terms.reserve(obj.vars_size());
1664 for (int i = 0; i < obj.vars_size(); ++i) {
1665 terms.push_back(
1666 std::make_pair(mapping->Integer(obj.vars(i)), obj.coeffs(i)));
1667 }
1668 if (parameters.optimize_with_core() && !objective_need_to_be_tight) {
1669 objective_var = GetOrCreateVariableWithTightBound(terms, model);
1670 } else {
1671 objective_var = GetOrCreateVariableLinkedToSumOf(
1672 terms, objective_need_to_be_tight, model);
1673 }
1674 }
1675
1676 // Create the objective definition inside the Model so that it can be accessed
1677 // by the heuristics than needs it.
1678 if (objective_var != kNoIntegerVariable) {
1679 const CpObjectiveProto& objective_proto = model_proto.objective();
1680 auto* objective_definition = model->GetOrCreate<ObjectiveDefinition>();
1681
1682 objective_definition->scaling_factor = objective_proto.scaling_factor();
1683 if (objective_definition->scaling_factor == 0.0) {
1684 objective_definition->scaling_factor = 1.0;
1685 }
1686 objective_definition->offset = objective_proto.offset();
1687 objective_definition->objective_var = objective_var;
1688
1689 const int size = objective_proto.vars_size();
1690 objective_definition->vars.resize(size);
1691 objective_definition->coeffs.resize(size);
1692 for (int i = 0; i < objective_proto.vars_size(); ++i) {
1693 // Note that if there is no mapping, then the variable will be
1694 // kNoIntegerVariable.
1695 objective_definition->vars[i] = mapping->Integer(objective_proto.vars(i));
1696 objective_definition->coeffs[i] = IntegerValue(objective_proto.coeffs(i));
1697
1698 // Fill the objective heuristics data.
1699 const int ref = objective_proto.vars(i);
1700 if (mapping->IsInteger(ref)) {
1701 const IntegerVariable var = mapping->Integer(objective_proto.vars(i));
1702 objective_definition->objective_impacting_variables.insert(
1703 objective_proto.coeffs(i) > 0 ? var : NegationOf(var));
1704 }
1705 }
1706
1707 // Register an objective special propagator.
1708 model->TakeOwnership(
1709 new LevelZeroEquality(objective_var, objective_definition->vars,
1710 objective_definition->coeffs, model));
1711 }
1712
1713 // Intersect the objective domain with the given one if any.
1714 if (!model_proto.objective().domain().empty()) {
1715 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
1716 const Domain user_domain = ReadDomainFromProto(model_proto.objective());
1717 const Domain automatic_domain =
1718 integer_trail->InitialVariableDomain(objective_var);
1719 VLOG(3) << "Objective offset:" << model_proto.objective().offset()
1720 << " scaling_factor:" << model_proto.objective().scaling_factor();
1721 VLOG(3) << "Automatic internal objective domain: " << automatic_domain;
1722 VLOG(3) << "User specified internal objective domain: " << user_domain;
1723 CHECK_NE(objective_var, kNoIntegerVariable);
1724 if (!integer_trail->UpdateInitialDomain(objective_var, user_domain)) {
1725 VLOG(2) << "UNSAT due to the objective domain.";
1726 return unsat();
1727 }
1728 }
1729
1730 // Note that we do one last propagation at level zero once all the
1731 // constraints were added.
1732 SOLVER_LOG(model->GetOrCreate<SolverLogger>(),
1733 "Initial num_bool: ", sat_solver->NumVariables());
1734 if (!sat_solver->FinishPropagation()) return unsat();
1735
1736 if (model_proto.has_objective()) {
1737 // Report the initial objective variable bounds.
1738 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
1739 shared_response_manager->UpdateInnerObjectiveBounds(
1740 absl::StrCat(model->Name(), " initial_propagation"),
1741 integer_trail->LowerBound(objective_var),
1742 integer_trail->UpperBound(objective_var));
1743
1744 // Watch improved objective best bounds.
1745 RegisterObjectiveBestBoundExport(objective_var, shared_response_manager,
1746 model);
1747
1748 // Import objective bounds.
1749 // TODO(user): Support objective bounds import in LNS and Core based
1750 // search.
1751 if (model->GetOrCreate<SatParameters>()->share_objective_bounds()) {
1752 RegisterObjectiveBoundsImport(shared_response_manager, model);
1753 }
1754 }
1755
1756 // Cache the links between model vars, IntegerVariables and lp constraints.
1757 // TODO(user): Cache this only if it is actually used.
1758 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
1759 auto* lp_dispatcher = model->GetOrCreate<LinearProgrammingDispatcher>();
1760 auto* lp_vars = model->GetOrCreate<LPVariables>();
1761 IntegerVariable size = integer_trail->NumIntegerVariables();
1762 for (IntegerVariable positive_var(0); positive_var < size;
1763 positive_var += 2) {
1764 LPVariable lp_var;
1765 lp_var.positive_var = positive_var;
1766 lp_var.model_var =
1767 mapping->GetProtoVariableFromIntegerVariable(positive_var);
1768 const auto& it = lp_dispatcher->find(positive_var);
1769 lp_var.lp = it != lp_dispatcher->end() ? it->second : nullptr;
1770
1771 if (lp_var.model_var >= 0) {
1772 lp_vars->vars.push_back(lp_var);
1773 lp_vars->model_vars_size =
1774 std::max(lp_vars->model_vars_size, lp_var.model_var + 1);
1775 }
1776 }
1777
1778 // Initialize the fixed_search strategy.
1779 auto* search_heuristics = model->GetOrCreate<SearchHeuristics>();
1780 if (parameters.search_branching() == SatParameters::PARTIAL_FIXED_SEARCH) {
1781 search_heuristics->user_search =
1782 ConstructUserSearchStrategy(model_proto, model);
1783 }
1784 search_heuristics->fixed_search = ConstructFixedSearchStrategy(
1785 model_proto, mapping->GetVariableMapping(), objective_var, model);
1786 if (VLOG_IS_ON(3)) {
1787 search_heuristics->fixed_search =
1788 InstrumentSearchStrategy(model_proto, mapping->GetVariableMapping(),
1789 search_heuristics->fixed_search, model);
1790 }
1791
1792 // Initialize the "follow hint" strategy.
1793 std::vector<BooleanOrIntegerVariable> vars;
1794 std::vector<IntegerValue> values;
1795 for (int i = 0; i < model_proto.solution_hint().vars_size(); ++i) {
1796 const int ref = model_proto.solution_hint().vars(i);
1797 CHECK(RefIsPositive(ref));
1798 BooleanOrIntegerVariable var;
1799 if (mapping->IsBoolean(ref)) {
1800 var.bool_var = mapping->Literal(ref).Variable();
1801 } else {
1802 var.int_var = mapping->Integer(ref);
1803 }
1804 vars.push_back(var);
1805 values.push_back(IntegerValue(model_proto.solution_hint().values(i)));
1806 }
1807 search_heuristics->hint_search = FollowHint(vars, values, model);
1808
1809 // Create the CoreBasedOptimizer class if needed.
1810 if (parameters.optimize_with_core()) {
1811 // TODO(user): Remove code duplication with the solution_observer in
1812 // SolveLoadedCpModel().
1813 const auto solution_observer = [&model_proto, model,
1814 shared_response_manager,
1815 best_obj_ub = kMaxIntegerValue]() mutable {
1816 const std::vector<int64_t> solution =
1817 GetSolutionValues(model_proto, *model);
1818 const IntegerValue obj_ub =
1819 ComputeInnerObjective(model_proto.objective(), solution);
1820 if (obj_ub < best_obj_ub) {
1821 best_obj_ub = obj_ub;
1822 shared_response_manager->NewSolution(solution, model->Name(), model);
1823 }
1824 };
1825
1826 const auto& objective = *model->GetOrCreate<ObjectiveDefinition>();
1827 if (parameters.optimize_with_max_hs()) {
1828 HittingSetOptimizer* max_hs = new HittingSetOptimizer(
1829 model_proto, objective, solution_observer, model);
1830 model->Register<HittingSetOptimizer>(max_hs);
1831 model->TakeOwnership(max_hs);
1832 } else {
1833 CoreBasedOptimizer* core =
1834 new CoreBasedOptimizer(objective_var, objective.vars,
1835 objective.coeffs, solution_observer, model);
1836 model->Register<CoreBasedOptimizer>(core);
1837 model->TakeOwnership(core);
1838 }
1839 }
1840
1841 InitializeDebugSolution(model_proto, model);
1842}

1844// Solves an already loaded cp_model_proto.
1845// The final CpSolverResponse must be read from the shared_response_manager.
1846//
1847// TODO(user): This should be transformed so that it can be called many times
1848// and resume from the last search state as if it wasn't interrupted. That would
1849// allow use to easily interleave different heuristics in the same thread.
1850void SolveLoadedCpModel(const CpModelProto& model_proto, Model* model) {
1851 auto* shared_response_manager = model->GetOrCreate<SharedResponseManager>();
1852 if (shared_response_manager->ProblemIsSolved()) return;
1853
1854 const SatParameters& parameters = *model->GetOrCreate<SatParameters>();
1855 if (parameters.stop_after_root_propagation()) return;
1856
1857 auto solution_observer = [&model_proto, model, shared_response_manager,
1858 best_obj_ub = kMaxIntegerValue]() mutable {
1859 const std::vector<int64_t> solution =
1860 GetSolutionValues(model_proto, *model);
1861 if (model_proto.has_objective()) {
1862 const IntegerValue obj_ub =
1863 ComputeInnerObjective(model_proto.objective(), solution);
1864 if (obj_ub < best_obj_ub) {
1865 best_obj_ub = obj_ub;
1866 shared_response_manager->NewSolution(solution, model->Name(), model);
1867 }
1868 } else {
1869 shared_response_manager->NewSolution(solution, model->Name(), model);
1870 }
1871 };
1872
1873 // Reconfigure search heuristic if it was changed.
1874 ConfigureSearchHeuristics(model);
1875
1876 const auto& mapping = *model->GetOrCreate<CpModelMapping>();
1877 SatSolver::Status status;
1878
1879 if (parameters.use_probing_search()) {
1880 ContinuousProber prober(model_proto, model);
1881 while (true) {
1882 status = prober.Probe();
1883 if (status == SatSolver::INFEASIBLE) {
1884 shared_response_manager->NotifyThatImprovingProblemIsInfeasible(
1885 model->Name());
1886 break;
1887 }
1888 if (status == SatSolver::FEASIBLE) {
1889 solution_observer();
1890 } else {
1891 break;
1892 }
1893 }
1894 } else if (!model_proto.has_objective()) {
1895 while (true) {
1896 if (parameters.use_shared_tree_search()) {
1897 auto* subtree_worker = model->GetOrCreate<SharedTreeWorker>();
1898 status = subtree_worker->Search(solution_observer);
1899 } else {
1900 status = ResetAndSolveIntegerProblem(
1901 mapping.Literals(model_proto.assumptions()), model);
1902 }
1903 if (status != SatSolver::Status::FEASIBLE) break;
1904 solution_observer();
1905 if (!parameters.enumerate_all_solutions()) break;
1906 model->Add(ExcludeCurrentSolutionWithoutIgnoredVariableAndBacktrack());
1907 }
1908 if (status == SatSolver::INFEASIBLE) {
1909 shared_response_manager->NotifyThatImprovingProblemIsInfeasible(
1910 model->Name());
1911 }
1912 if (status == SatSolver::ASSUMPTIONS_UNSAT) {
1913 shared_response_manager->NotifyThatImprovingProblemIsInfeasible(
1914 model->Name());
1915
1916 // Extract a good subset of assumptions and add it to the response.
1917 auto* time_limit = model->GetOrCreate<TimeLimit>();
1918 auto* sat_solver = model->GetOrCreate<SatSolver>();
1919 std::vector<Literal> core = sat_solver->GetLastIncompatibleDecisions();
1920 MinimizeCoreWithPropagation(time_limit, sat_solver, &core);
1921 std::vector<int> core_in_proto_format;
1922 for (const Literal l : core) {
1923 core_in_proto_format.push_back(
1924 mapping.GetProtoVariableFromBooleanVariable(l.Variable()));
1925 if (!l.IsPositive()) {
1926 core_in_proto_format.back() = NegatedRef(core_in_proto_format.back());
1927 }
1928 }
1929 shared_response_manager->AddUnsatCore(core_in_proto_format);
1930 }
1931 } else {
1932 // Optimization problem.
1933 const auto& objective = *model->GetOrCreate<ObjectiveDefinition>();
1934 const IntegerVariable objective_var = objective.objective_var;
1935 CHECK_NE(objective_var, kNoIntegerVariable);
1936
1937 if (parameters.optimize_with_lb_tree_search()) {
1938 auto* search = model->GetOrCreate<LbTreeSearch>();
1939 status = search->Search(solution_observer);
1940 } else if (parameters.optimize_with_core()) {
1941 // TODO(user): This doesn't work with splitting in chunk for now. It
1942 // shouldn't be too hard to fix.
1943 if (parameters.optimize_with_max_hs()) {
1944 status = model->Mutable<HittingSetOptimizer>()->Optimize();
1945 } else {
1946 status = model->Mutable<CoreBasedOptimizer>()->Optimize();
1947 }
1948 } else if (parameters.use_shared_tree_search()) {
1949 auto* subtree_worker = model->GetOrCreate<SharedTreeWorker>();
1950 status = subtree_worker->Search(solution_observer);
1951 } else {
1952 // TODO(user): This parameter breaks the splitting in chunk of a Solve().
1953 // It should probably be moved into another SubSolver altogether.
1954 if (parameters.binary_search_num_conflicts() >= 0) {
1955 RestrictObjectiveDomainWithBinarySearch(objective_var,
1956 solution_observer, model);
1957 }
1958 status = MinimizeIntegerVariableWithLinearScanAndLazyEncoding(
1959 objective_var, solution_observer, model);
1960 }
1961
1962 // The search is done in both case.
1963 //
1964 // TODO(user): Remove the weird translation INFEASIBLE->FEASIBLE in the
1965 // function above?
1966 if (status == SatSolver::INFEASIBLE || status == SatSolver::FEASIBLE) {
1967 shared_response_manager->NotifyThatImprovingProblemIsInfeasible(
1968 model->Name());
1969 }
1970 }

1973// Try to find a solution by following the hint and using a low conflict limit.
1974// The CpModelProto must already be loaded in the Model.
1975void QuickSolveWithHint(const CpModelProto& model_proto, Model* model) {
1976 if (!model_proto.has_solution_hint()) return;
1977
1978 auto* shared_response_manager = model->GetOrCreate<SharedResponseManager>();
1979 if (shared_response_manager->ProblemIsSolved()) return;
1980
1981 // Temporarily change the parameters.
1982 auto* parameters = model->GetOrCreate<SatParameters>();
1983
1984 // If the model was loaded with "optimize_with_core" then the objective
1985 // variable is not linked to its linear expression. Because of that, we can
1986 // return a solution that does not satisfy the objective domain.
1987 //
1988 // TODO(user): This is fixable, but then do we need the hint when optimizing
1989 // with core?
1990 if (parameters->optimize_with_core()) return;
1991
1992 const SatParameters saved_params = *parameters;
1993 parameters->set_max_number_of_conflicts(parameters->hint_conflict_limit());
1994 parameters->set_search_branching(SatParameters::HINT_SEARCH);
1995 parameters->set_optimize_with_core(false);
1996 auto cleanup = ::absl::MakeCleanup(
1997 [parameters, saved_params]() { *parameters = saved_params; });
1998
1999 // Solve decision problem.
2000 ConfigureSearchHeuristics(model);
2001 const auto& mapping = *model->GetOrCreate<CpModelMapping>();
2002 const SatSolver::Status status = ResetAndSolveIntegerProblem(
2003 mapping.Literals(model_proto.assumptions()), model);
2004
2005 const std::string& solution_info = model->Name();
2006 if (status == SatSolver::Status::FEASIBLE) {
2007 const std::vector<int64_t> solution =
2008 GetSolutionValues(model_proto, *model);
2009 shared_response_manager->NewSolution(
2010 solution, absl::StrCat(solution_info, " [hint]"), model);
2011
2012 if (!model_proto.has_objective()) {
2013 if (parameters->enumerate_all_solutions()) {
2014 model->Add(ExcludeCurrentSolutionWithoutIgnoredVariableAndBacktrack());
2015 }
2016 } else {
2017 // Restrict the objective.
2018 const IntegerVariable objective_var =
2019 model->GetOrCreate<ObjectiveDefinition>()->objective_var;
2020 model->GetOrCreate<SatSolver>()->Backtrack(0);
2021 IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
2022 if (!integer_trail->Enqueue(
2023 IntegerLiteral::LowerOrEqual(
2024 objective_var,
2025 shared_response_manager->GetInnerObjectiveUpperBound()),
2026 {}, {})) {
2027 shared_response_manager->NotifyThatImprovingProblemIsInfeasible(
2028 absl::StrCat(solution_info, " [hint]"));
2029 }
2030 }
2031 return;
2032 }
2033
2034 // This code is here to debug bad presolve during LNS that corrupt the hint.
2035 // Note that sometime the deterministic limit is hit before the hint can be
2036 // completed, so we don't report that has an error.
2037 //
2038 // Tricky: We can only test that if we don't already have a feasible solution
2039 // like we do if the hint is complete.
2040 if (parameters->debug_crash_on_bad_hint() &&
2041 shared_response_manager->SolutionsRepository().NumSolutions() == 0 &&
2042 !model->GetOrCreate<TimeLimit>()->LimitReached() &&
2043 status != SatSolver::Status::FEASIBLE) {
2044 LOG(FATAL) << "QuickSolveWithHint() didn't find a feasible solution."
2045 << " The model name is '" << model_proto.name() << "'."
2046 << " Status: " << status << ".";
2047 }
2048
2049 if (status == SatSolver::INFEASIBLE) {
2050 shared_response_manager->NotifyThatImprovingProblemIsInfeasible(
2051 absl::StrCat(solution_info, " [hint]"));
2052 return;
2053 }
2054}

2277 case ConstraintProto::ConstraintCase::kBoolAnd: {

2279 for (const int ref : ct.bool_and().literals()) {
2280 drat_proof_handler->AddProblemClause({get_literal(ref)});
2281 }
2282 } else {
2283 // a => b
2284 const Literal not_a =
2285 get_literal(ct.enforcement_literal(0)).Negated();
2286 for (const int ref : ct.bool_and().literals()) {
2287 drat_proof_handler->AddProblemClause({not_a, get_literal(ref)});
2288 }
2289 }
2290 break;
2291 }
2292 case ConstraintProto::ConstraintCase::kBoolOr:
2293 temp.clear();
2294 for (const int ref : ct.bool_or().literals()) {
2295 temp.push_back(get_literal(ref));
2296 }
2297 for (const int ref : ct.enforcement_literal()) {
2298 temp.push_back(get_literal(ref).Negated());
2299 }
2300 drat_proof_handler->AddProblemClause(temp);
2301 break;
2302 default:
2303 LOG(FATAL) << "Not supported";
2304 }
2305 }
2306 }
2307
2308 for (const ConstraintProto& ct : model_proto.constraints()) {
2309 switch (ct.constraint_case()) {
2310 case ConstraintProto::ConstraintCase::kBoolAnd: {
2311 if (ct.enforcement_literal_size() == 0) {
2312 for (const int ref : ct.bool_and().literals()) {
2313 const Literal b = get_literal(ref);
2314 solver->AddUnitClause(b);
2315 }
2316 } else {
2317 // a => b
2318 const Literal not_a =
2319 get_literal(ct.enforcement_literal(0)).Negated();
2320 for (const int ref : ct.bool_and().literals()) {
2321 const Literal b = get_literal(ref);
2322 solver->AddProblemClause({not_a, b}, /*is_safe=*/false);
2323 }
2324 }
2325 break;
2326 }
2327 case ConstraintProto::ConstraintCase::kBoolOr:
2328 temp.clear();
2329 for (const int ref : ct.bool_or().literals()) {
2330 temp.push_back(get_literal(ref));
2331 }
2332 for (const int ref : ct.enforcement_literal()) {
2333 temp.push_back(get_literal(ref).Negated());
2334 }
2335 solver->AddProblemClause(temp, /*is_safe=*/false);
2336 break;
2337 default:
2338 LOG(FATAL) << "Not supported";
2339 }
2340 }
2341
2342 // Deal with fixed variables.
2343 for (int ref = 0; ref < num_variables; ++ref) {
2344 const Domain domain = ReadDomainFromProto(model_proto.variables(ref));
2345 if (domain.Min() == domain.Max()) {
2346 const Literal ref_literal =
2347 domain.Min() == 0 ? get_literal(ref).Negated() : get_literal(ref);
2348 solver->AddUnitClause(ref_literal);
2349 }
2350 }
2351
2352 SatSolver::Status status;
2353 CpSolverResponse response;
2354 if (parameters.cp_model_presolve()) {
2355 std::vector<bool> solution;
2356 status = SolveWithPresolve(&solver, model->GetOrCreate<TimeLimit>(),
2357 &solution, drat_proof_handler.get(), logger);
2358 if (status == SatSolver::FEASIBLE) {
2359 response.clear_solution();
2360 for (int ref = 0; ref < num_variables; ++ref) {
2361 response.add_solution(solution[ref]);
2362 }
2363 }
2364 } else {
2365 status = solver->SolveWithTimeLimit(model->GetOrCreate<TimeLimit>());
2366 if (status == SatSolver::FEASIBLE) {
2367 response.clear_solution();
2368 for (int ref = 0; ref < num_variables; ++ref) {
2369 response.add_solution(
2370 solver->Assignment().LiteralIsTrue(get_literal(ref)) ? 1 : 0);
2371 }
2372 }
2373 }
2374
2375 // Tricky: the model local time limit is updated by the new functions, but
2376 // the old ones update time_limit directly.
2377 model->GetOrCreate<TimeLimit>()->AdvanceDeterministicTime(
2378 solver->model()->GetOrCreate<TimeLimit>()->GetElapsedDeterministicTime());
2379
2380 switch (status) {
2381 case SatSolver::LIMIT_REACHED: {
2382 response.set_status(CpSolverStatus::UNKNOWN);
2383 break;
2384 }
2385 case SatSolver::FEASIBLE: {
2386 CHECK(SolutionIsFeasible(
2387 model_proto, std::vector<int64_t>(response.solution().begin(),
2388 response.solution().end())));
2389 response.set_status(CpSolverStatus::OPTIMAL);
2390 break;
2391 }
2392 case SatSolver::INFEASIBLE: {
2393 response.set_status(CpSolverStatus::INFEASIBLE);
2394 break;
2395 }
2396 default:
2397 LOG(FATAL) << "Unexpected SatSolver::Status " << status;
2398 }
2399 response.set_num_booleans(solver->NumVariables());
2400 response.set_num_branches(solver->num_branches());
2401 response.set_num_conflicts(solver->num_failures());
2402 response.set_num_binary_propagations(solver->num_propagations());
2403 response.set_num_integer_propagations(0);
2404 response.set_wall_time(wall_timer->Get());
2405 response.set_deterministic_time(
2406 model->Get<TimeLimit>()->GetElapsedDeterministicTime());
2407
2408 if (status == SatSolver::INFEASIBLE && drat_proof_handler != nullptr) {
2409 WallTimer drat_timer;
2410 drat_timer.Start();
2411 DratChecker::Status drat_status = drat_proof_handler->Check(
2412 absl::GetFlag(FLAGS_max_drat_time_in_seconds));
2413 switch (drat_status) {
2414 case DratChecker::UNKNOWN:
2415 LOG(INFO) << "DRAT status: UNKNOWN";
2416 break;
2417 case DratChecker::VALID:
2418 LOG(INFO) << "DRAT status: VALID";
2419 break;
2420 case DratChecker::INVALID:
2421 LOG(ERROR) << "DRAT status: INVALID";
2422 break;
2423 default:
2424 // Should not happen.
2425 break;
2426 }
2427 LOG(INFO) << "DRAT wall time: " << drat_timer.Get();
2428 } else if (drat_proof_handler != nullptr) {
2429 // Always log a DRAT status to make it easier to extract it from a multirun
2430 // result with awk.
2431 LOG(INFO) << "DRAT status: NA";
2432 LOG(INFO) << "DRAT wall time: NA";
2433 LOG(INFO) << "DRAT user time: NA";
2434 }
2435 return response;
2436}

3952void TestSolutionHintForFeasibility(const CpModelProto& model_proto,
3953 SolverLogger* logger,
3954 SharedResponseManager* manager = nullptr) {
3955 if (!model_proto.has_solution_hint()) return;
3956
3957 // TODO(user): If the hint specifies all non-fixed variables we could also
3958 // do the check.
3959 if (model_proto.solution_hint().vars_size() != model_proto.variables_size()) {
3960 return;
3961 }
3962
3963 std::vector<int64_t> solution(model_proto.variables_size(), 0);
3964 for (int i = 0; i < model_proto.solution_hint().vars_size(); ++i) {
3965 const int ref = model_proto.solution_hint().vars(i);
3966 const int64_t value = model_proto.solution_hint().values(i);
3967 solution[PositiveRef(ref)] = RefIsPositive(ref) ? value : -value;
3968 }
3969 if (SolutionIsFeasible(model_proto, solution)) {
3970 if (manager != nullptr) {
3971 // Add it to the pool right away! Note that we already have a log in this
3972 // case, so we don't log anything more.
3973 manager->NewSolution(solution, "complete_hint", nullptr);
3974 } else {
3975 SOLVER_LOG(logger, "The solution hint is complete and is feasible.");
3976 }
3977 } else {
3978 // TODO(user): Change the code to make the solution checker more
3979 // informative by returning a message instead of just VLOGing it.
3980 SOLVER_LOG(logger,
3981 "The solution hint is complete, but it is infeasible! we "
3982 "will try to repair it.");

4256 &logger](CpSolverResponse* response) {
4257 if (response->solution().empty()) return;

4259 // Compute the true objective of the best returned solution.
4260 const auto& float_obj = model_proto.floating_point_objective();

4263 for (int i = 0; i < num_terms; ++i) {
4264 value += float_obj.coeffs(i) *
4265 static_cast<double>(response->solution(float_obj.vars(i)));

4280 !integer_obj.scaling_was_exact()) {

4283 model_proto, integer_obj, integer_lb);
4284 SOLVER_LOG(logger, "[Scaling] scaled_objective_bound: ",
4285 response->best_objective_bound(),
4286 " corrected_bound: ", lb,
4287 " delta: ", response->best_objective_bound() - lb);

4290 // min/max with the objective value.

4292 response->set_best_objective_bound(
4293 std::max(lb, response->objective_value()));
4294 } else {
4295 response->set_best_objective_bound(
4296 std::min(lb, response->objective_value()));
4297 }
4298 }
4299

4304 response->best_objective_bound());

4307 "[Scaling] Warning: OPTIMAL was reported, yet the "
4308 "objective gap (",
4309 gap, ") is greater than requested absolute limit (",
4310 params.absolute_gap_limit(), ").");
4311 }
4312 }
4313 });

4321 logger,
4322 "Warning: solving with assumptions was requested in a non-fully "

4334 shared_response_manager->AddFinalResponsePostprocessor(
4335 [&model_proto](CpSolverResponse* response) {

4339 *response->mutable_sufficient_assumptions_for_infeasibility() =

4341 });

4351 status_response.add_sufficient_assumptions_for_infeasibility(ref);

4354 return shared_response_manager->GetResponse();
4355 }
4356 }
4357 }

4361 const CpSolverStatus presolve_status =
4362 PresolveCpModel(context.get(), &postsolve_mapping);

4366 status_response.set_status(presolve_status);

4368 shared_response_manager->AppendResponseToBeMerged(status_response);

4372 SOLVER_LOG(logger, "");
4373 SOLVER_LOG(logger, "Presolved ", CpModelStats(*new_cp_model_proto));
4374
4375 if (params.cp_model_presolve()) {
4376 shared_response_manager->AddSolutionPostprocessor(
4377 [&model_proto, &params, mapping_proto, &model,
4378 &postsolve_mapping](std::vector<int64_t>* solution) {
4379 AddPostsolveClauses(postsolve_mapping, model, mapping_proto);
4380 PostsolveResponseWrapper(params, model_proto.variables_size(),
4381 *mapping_proto, postsolve_mapping, solution);
4382 });

4384 [&model_proto, &params, mapping_proto,
4385 &postsolve_mapping](CpSolverResponse* response) {
4386 // Map back the sufficient assumptions for infeasibility.
4387 for (int& ref :
4388 *(response
4389 ->mutable_sufficient_assumptions_for_infeasibility())) {

4391 ? postsolve_mapping[ref]
4392 : NegatedRef(postsolve_mapping[PositiveRef(ref)]);

4395 CHECK(SolutionIsFeasible(
4396 model_proto,
4397 std::vector<int64_t>(response->solution().begin(),
4398 response->solution().end()),
4399 mapping_proto, &postsolve_mapping))
4400 << "postsolved solution";
4401 }

4404 // presolve.

4406 model_proto.variables().size()) {
4407 for (int i = 0; i < model_proto.variables().size(); ++i) {
4408 *response->add_tightened_variables() =
4409 mapping_proto->variables(i);
4410 }
4411 }
4412 }
4413 });
4414 } else {
4415 shared_response_manager->AddFinalResponsePostprocessor(
4416 [&model_proto](CpSolverResponse* response) {
4417 if (!response->solution().empty()) {
4418 CHECK(SolutionIsFeasible(
4419 model_proto, std::vector<int64_t>(response->solution().begin(),
4420 response->solution().end())));
4421 }
4422 });
4423 shared_response_manager->AddResponsePostprocessor(
4424 [&model_proto, &params](CpSolverResponse* response) {
4425 // Truncate the solution in case model expansion added more variables.
4426 const int initial_size = model_proto.variables_size();

4428 response->mutable_solution()->Truncate(initial_size);
4429 if (DEBUG_MODE ||
4430 absl::GetFlag(FLAGS_cp_model_check_intermediate_solutions)) {
4431 CHECK(SolutionIsFeasible(
4432 model_proto,
4433 std::vector<int64_t>(response->solution().begin(),
4434 response->solution().end())));
4435 }
4436 }
4437 if (params.fill_tightened_domains_in_response()) {
4438 *response->mutable_tightened_variables() = model_proto.variables();
4439 }
4440 });
4441 }

4444 context.reset(nullptr);

4447 if (!observers.empty()) {

4449 [&observers](const CpSolverResponse& response) {
4450 for (const auto& observer : observers) {
4451 observer(response);
4452 }
4453 });
4454 }

4457 if (params.stop_after_first_solution()) {

4459 [shared_time_limit](const CpSolverResponse&) {
4460 shared_time_limit->Stop();
4461 });

4547 // We ignore the multithreading parameter in this case.
4548#else // __PORTABLE_PLATFORM__
4549 if (params.num_workers() > 1 || params.interleave_search() ||

4552#endif // __PORTABLE_PLATFORM__
4553 } else if (!model->GetOrCreate<TimeLimit>()->LimitReached()) {
4554 SOLVER_LOG(logger, "");
4555 SOLVER_LOG(logger, absl::StrFormat("Starting to load the model at %.2fs",
4556 wall_timer->Get()));

4558

4560 // parallel case. When this model will be destroyed, we will collect some
4561 // stats that are used to debug/improve internal algorithm.
4562 Model local_model;
4563 local_model.Register<SolverLogger>(logger);
4564 local_model.Register<TimeLimit>(model->GetOrCreate<TimeLimit>());
4565 local_model.Register<SatParameters>(model->GetOrCreate<SatParameters>());
4566 local_model.Register<SharedStatistics>(
4567 model->GetOrCreate<SharedStatistics>());
4568 local_model.Register<SharedResponseManager>(shared_response_manager);
4569
4570 LoadCpModel(*new_cp_model_proto, &local_model);
4571
4572 SOLVER_LOG(logger, "");
4573 SOLVER_LOG(logger, absl::StrFormat("Starting sequential search at %.2fs",
4574 wall_timer->Get()));
4575 if (params.repair_hint()) {
4576 MinimizeL1DistanceWithHint(*new_cp_model_proto, &local_model);
4577 } else {
4578 QuickSolveWithHint(*new_cp_model_proto, &local_model);
4579 }
4580 SolveLoadedCpModel(*new_cp_model_proto, &local_model);
4581
4582 // Export statistics.
4583 CpSolverResponse status_response;
4584 FillSolveStatsInResponse(&local_model, &status_response);
4585 shared_response_manager->AppendResponseToBeMerged(status_response);
4586
4587 // Sequential logging of LP statistics.
4588 if (logger->LoggingIsEnabled()) {
4589 const auto& lps =
4590 *local_model.GetOrCreate<LinearProgrammingConstraintCollection>();
4591 if (!lps.empty()) {
4592 SOLVER_LOG(logger, "");
4593 for (const auto* lp : lps) {
4594 SOLVER_LOG(logger, lp->Statistics());
4595 }
4596 }
4597 }
4598 }
4599
4600 // Extra logging if needed.
4601 if (logger->LoggingIsEnabled()) {
4602 model->GetOrCreate<SharedStatistics>()->Log(logger);
4603 }
4604 return shared_response_manager->GetResponse();
4605}
