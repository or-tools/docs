56void ImpliedBounds::Add(Literal literal, IntegerLiteral integer_literal) {
57 if (!parameters_.use_implied_bounds()) return;
58 const IntegerVariable var = integer_literal.var;
59
60 // Update our local level-zero bound.
61 if (var >= level_zero_lower_bounds_.size()) {
62 level_zero_lower_bounds_.resize(var.value() + 1, kMinIntegerValue);
63 new_level_zero_bounds_.Resize(var + 1);
64 }
65 level_zero_lower_bounds_[var] = std::max(
66 level_zero_lower_bounds_[var], integer_trail_->LevelZeroLowerBound(var));
67
68 // Ignore any Add() with a bound worse than the level zero one.
69 // TODO(user): Check that this never happen? it shouldn't.
70 if (integer_literal.bound <= level_zero_lower_bounds_[var]) {
71 return;
72 }
73
74 // We skip any IntegerLiteral referring to a variable with only two
75 // consecutive possible values. This is because, once shifted this will
76 // already be a variable in [0, 1] so we shouldn't gain much by substituing
77 // it.
78 if (integer_trail_->LevelZeroLowerBound(var) + 1 >=
79 integer_trail_->LevelZeroUpperBound(var)) {
80 return;
81 }
82
83 // Add or update the current bound.
84 const auto key = std::make_pair(literal.Index(), var);
85 auto insert_result = bounds_.insert({key, integer_literal.bound});
86 if (!insert_result.second) {
87 if (insert_result.first->second < integer_literal.bound) {
88 insert_result.first->second = integer_literal.bound;
89 } else {
90 // No new info.
91 return;
92 }
93 }
94
95 // Checks if the variable is now fixed.
96 if (integer_trail_->LevelZeroUpperBound(var) == integer_literal.bound) {
97 AddLiteralImpliesVarEqValue(literal, var, integer_literal.bound);
98 } else {
99 const auto it =
100 bounds_.find(std::make_pair(literal.Index(), NegationOf(var)));
101 if (it != bounds_.end() && it->second == -integer_literal.bound) {
102 AddLiteralImpliesVarEqValue(literal, var, integer_literal.bound);
103 }
104 }
105
106 // Check if we have any deduction. Since at least one of (literal,
107 // literal.Negated()) must be true, we can take the min bound as valid at
108 // level zero.
109 //
110 // TODO(user): Like in probing, we can also create hole in the domain if there
111 // is some implied bounds for (literal.NegatedIndex, NegagtionOf(var)) that
112 // crosses integer_literal.bound.
113 const auto it = bounds_.find(std::make_pair(literal.NegatedIndex(), var));
114 if (it != bounds_.end()) {
115 if (it->second <= level_zero_lower_bounds_[var]) {
116 // The other bounds is worse than the new level-zero bound which can
117 // happen because of lazy update, so here we just remove it.
118 bounds_.erase(it);
119 } else {
120 const IntegerValue deduction =
121 std::min(integer_literal.bound, it->second);
122 DCHECK_GT(deduction, level_zero_lower_bounds_[var]);
123 DCHECK_GT(deduction, integer_trail_->LevelZeroLowerBound(var));
124
125 // TODO(user): support Enqueueing level zero fact at a positive level.
126 // That is, do not loose the info on backtrack. This should be doable. It
127 // is also why we return a bool in case of conflict when pushing
128 // deduction.
129 ++num_deductions_;
130 level_zero_lower_bounds_[var] = deduction;
131 new_level_zero_bounds_.Set(var);
132
133 VLOG(2) << "Deduction old: "
134 << IntegerLiteral::GreaterOrEqual(
135 var, integer_trail_->LevelZeroLowerBound(var))
136 << " new: " << IntegerLiteral::GreaterOrEqual(var, deduction);
137
138 // The entries that are equal to the min no longer need to be stored once
139 // the level zero bound is enqueued.
140 if (it->second == deduction) {
141 bounds_.erase(it);
142 }
143 if (integer_literal.bound == deduction) {
144 bounds_.erase(std::make_pair(literal.Index(), var));
145
146 // No need to update var_to_bounds_ in this case.
147 return;
148 }
149 }
150 }
151
152 // While the code above deal correctly with optionality, we cannot just
153 // register a literal => bound for an optional variable, because the equation
154 // might end up in the LP which do not handle them correctly.
155 //
156 // TODO(user): Maybe we can handle this case somehow, as long as every
157 // constraint using this bound is protected by the variable optional literal.
158 // Alternativelly we could disable optional variable when we are at
159 // linearization level 2.
160 if (integer_trail_->IsOptional(var)) return;
161
162 // If we have a new implied bound and the literal has a view, add it to
163 // var_to_bounds_. Note that we might add more than one entry with the same
164 // literal_view, and we will later need to lazily clean the vector up.
165 if (integer_encoder_->GetLiteralView(literal) != kNoIntegerVariable) {
166 if (var_to_bounds_.size() <= var) {
167 var_to_bounds_.resize(var.value() + 1);
168 has_implied_bounds_.Resize(var + 1);
169 }
170 ++num_enqueued_in_var_to_bounds_;
171 has_implied_bounds_.Set(var);
172 var_to_bounds_[var].push_back({integer_encoder_->GetLiteralView(literal),
173 integer_literal.bound, true});
174 } else if (integer_encoder_->GetLiteralView(literal.Negated()) !=
175 kNoIntegerVariable) {
176 if (var_to_bounds_.size() <= var) {
177 var_to_bounds_.resize(var.value() + 1);
178 has_implied_bounds_.Resize(var + 1);
179 }
180 ++num_enqueued_in_var_to_bounds_;
181 has_implied_bounds_.Set(var);
182 var_to_bounds_[var].push_back(
183 {integer_encoder_->GetLiteralView(literal.Negated()),
184 integer_literal.bound, false});
185 }
186}

188const std::vector<ImpliedBoundEntry>& ImpliedBounds::GetImpliedBounds(
189 IntegerVariable var) {
190 if (var >= var_to_bounds_.size()) return empty_implied_bounds_;
191
192 // Lazily remove obsolete entries from the vector.
193 //
194 // TODO(user): Check no duplicate and remove old entry if the enforcement
195 // is tighter.
196 int new_size = 0;
197 std::vector<ImpliedBoundEntry>& ref = var_to_bounds_[var];
198 const IntegerValue level_zero_lb = std::max(
199 level_zero_lower_bounds_[var], integer_trail_->LevelZeroLowerBound(var));
200 level_zero_lower_bounds_[var] = level_zero_lb;
201 for (const ImpliedBoundEntry& entry : ref) {
202 if (entry.lower_bound <= level_zero_lb) continue;
203 ref[new_size++] = entry;
204 }
205 ref.resize(new_size);
206
207 return ref;
208}

252bool ImpliedBounds::EnqueueNewDeductions() {
253 CHECK_EQ(sat_solver_->CurrentDecisionLevel(), 0);
254 for (const IntegerVariable var :
255 new_level_zero_bounds_.PositionsSetAtLeastOnce()) {
256 if (!integer_trail_->Enqueue(
257 IntegerLiteral::GreaterOrEqual(var, level_zero_lower_bounds_[var]),
258 {}, {})) {
259 return false;
260 }
261 }
262 new_level_zero_bounds_.SparseClearAll();
263 return sat_solver_->FinishPropagation();
264}

275// If a variable has a size of 2, it is most likely reduced to an affine
276// expression pointing to a variable with domain [0,1] or [-1,0].
277// If the original variable has been removed from the model, then there are no
278// implied values from any exactly_one constraint to its domain.
279// If we are lucky, one of the literal of the exactly_one constraints, and its
280// negation are used to encode the Boolean variable of the affine.
281//
282// This may fail if exactly_one(l0, l1, l2, l3); l0 and l1 imply x = 0,
283// l2 and l3 imply x = 1. In that case, one must look at the binary
284// implications to find the missing link.
285//
286// TODO(user): Consider removing this once we are more complete in our implied
287// bounds repository. Because if we can reconcile an encoding, then any of the
288// literal in the at most one should imply a value on the boolean view use in
289// the size2 affine.
290std::vector<LiteralValueValue> TryToReconcileEncodings(
291 const AffineExpression& size2_affine, const AffineExpression& affine,
292 const std::vector<ValueLiteralPair>& affine_var_encoding,
293 bool put_affine_left_in_result, Model* model) {
294 IntegerEncoder* integer_encoder = model->GetOrCreate<IntegerEncoder>();
295 IntegerVariable binary = size2_affine.var;
296 std::vector<LiteralValueValue> terms;
297 if (!integer_encoder->VariableIsFullyEncoded(binary)) return terms;
298 const std::vector<ValueLiteralPair>& size2_enc =
299 integer_encoder->FullDomainEncoding(binary);
300
301 // TODO(user): I am not sure how this can happen since size2_affine is
302 // supposed to be non-fixed. Maybe we miss some propag. Investigate.
303 if (size2_enc.size() != 2) return terms;
304
305 Literal lit0 = size2_enc[0].literal;
306 IntegerValue value0 = size2_affine.ValueAt(size2_enc[0].value);
307 Literal lit1 = size2_enc[1].literal;
308 IntegerValue value1 = size2_affine.ValueAt(size2_enc[1].value);
309
310 for (const auto& [unused, candidate_literal] : affine_var_encoding) {
311 if (candidate_literal == lit1) {
312 std::swap(lit0, lit1);
313 std::swap(value0, value1);
314 }
315 if (candidate_literal != lit0) continue;
316
317 // Build the decomposition.
318 for (const auto& [value, literal] : affine_var_encoding) {
319 const IntegerValue size_2_value = literal == lit0 ? value0 : value1;
320 const IntegerValue affine_value = affine.ValueAt(value);
321 if (put_affine_left_in_result) {
322 terms.push_back({literal, affine_value, size_2_value});
323 } else {
324 terms.push_back({literal, size_2_value, affine_value});
325 }
326 }
327 break;
328 }
329
330 return terms;
331}

333// Specialized case of encoding reconciliation when both variables have a domain
334// of size of 2.
335std::vector<LiteralValueValue> TryToReconcileSize2Encodings(
336 const AffineExpression& left, const AffineExpression& right, Model* model) {
337 IntegerEncoder* integer_encoder = model->GetOrCreate<IntegerEncoder>();
338 std::vector<LiteralValueValue> terms;
339 if (!integer_encoder->VariableIsFullyEncoded(left.var) ||
340 !integer_encoder->VariableIsFullyEncoded(right.var)) {
341 return terms;
342 }
343 const std::vector<ValueLiteralPair>& left_enc =
344 integer_encoder->FullDomainEncoding(left.var);
345 const std::vector<ValueLiteralPair>& right_enc =
346 integer_encoder->FullDomainEncoding(right.var);
347 if (left_enc.size() != 2 || right_enc.size() != 2) {
348 VLOG(2) << "encodings are not fully propagated";
349 return terms;
350 }
351
352 const Literal left_lit0 = left_enc[0].literal;
353 const IntegerValue left_value0 = left.ValueAt(left_enc[0].value);
354 const Literal left_lit1 = left_enc[1].literal;
355 const IntegerValue left_value1 = left.ValueAt(left_enc[1].value);
356
357 const Literal right_lit0 = right_enc[0].literal;
358 const IntegerValue right_value0 = right.ValueAt(right_enc[0].value);
359 const Literal right_lit1 = right_enc[1].literal;
360 const IntegerValue right_value1 = right.ValueAt(right_enc[1].value);
361
362 if (left_lit0 == right_lit0 || left_lit0 == right_lit1.Negated()) {
363 terms.push_back({left_lit0, left_value0, right_value0});
364 terms.push_back({left_lit0.Negated(), left_value1, right_value1});
365 } else if (left_lit0 == right_lit1 || left_lit0 == right_lit0.Negated()) {
366 terms.push_back({left_lit0, left_value0, right_value1});
367 terms.push_back({left_lit0.Negated(), left_value1, right_value0});
368 } else if (left_lit1 == right_lit1 || left_lit1 == right_lit0.Negated()) {
369 terms.push_back({left_lit1.Negated(), left_value0, right_value0});
370 terms.push_back({left_lit1, left_value1, right_value1});
371 } else if (left_lit1 == right_lit0 || left_lit1 == right_lit1.Negated()) {
372 terms.push_back({left_lit1.Negated(), left_value0, right_value1});
373 terms.push_back({left_lit1, left_value1, right_value0});
374 } else {
375 VLOG(3) << "Complex size 2 encoding case, need to scan exactly_ones";
376 }
377
378 return terms;
379}

381std::vector<LiteralValueValue> TryToDecomposeProduct(
382 const AffineExpression& left, const AffineExpression& right, Model* model) {
383 IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
384 if (integer_trail->IsFixed(left) || integer_trail->IsFixed(right)) return {};
385
386 // Fill in the encodings for the left variable.
387 ImpliedBounds* implied_bounds = model->GetOrCreate<ImpliedBounds>();
388 const absl::flat_hash_map<int, std::vector<ValueLiteralPair>>&
389 left_encodings = implied_bounds->GetElementEncodings(left.var);
390
391 // Fill in the encodings for the right variable.
392 const absl::flat_hash_map<int, std::vector<ValueLiteralPair>>&
393 right_encodings = implied_bounds->GetElementEncodings(right.var);
394
395 std::vector<int> compatible_keys;
396 for (const auto& [index, encoding] : left_encodings) {
397 if (right_encodings.contains(index)) {
398 compatible_keys.push_back(index);
399 }
400 }
401
402 if (compatible_keys.empty()) {
403 if (integer_trail->InitialVariableDomain(left.var).Size() == 2) {
404 for (const auto& [index, right_encoding] : right_encodings) {
405 const std::vector<LiteralValueValue> result =
406 TryToReconcileEncodings(left, right, right_encoding,
407 /*put_affine_left_in_result=*/false, model);
408 if (!result.empty()) {
409 return result;
410 }
411 }
412 }
413 if (integer_trail->InitialVariableDomain(right.var).Size() == 2) {
414 for (const auto& [index, left_encoding] : left_encodings) {
415 const std::vector<LiteralValueValue> result =
416 TryToReconcileEncodings(right, left, left_encoding,
417 /*put_affine_left_in_result=*/true, model);
418 if (!result.empty()) {
419 return result;
420 }
421 }
422 }
423 if (integer_trail->InitialVariableDomain(left.var).Size() == 2 &&
424 integer_trail->InitialVariableDomain(right.var).Size() == 2) {
425 const std::vector<LiteralValueValue> result =
426 TryToReconcileSize2Encodings(left, right, model);
427 if (!result.empty()) {
428 return result;
429 }
430 }
431 return {};
432 }
433
434 if (compatible_keys.size() > 1) {
435 VLOG(3) << "More than one exactly_one involved in the encoding of the two "
436 "variables";
437 }
438
439 // Select the compatible encoding with the minimum index.
440 const int min_index =
441 *std::min_element(compatible_keys.begin(), compatible_keys.end());
442 // By construction, encodings follow the order of literals in the exactly_one
443 // constraint.
444 const std::vector<ValueLiteralPair>& left_encoding =
445 left_encodings.at(min_index);
446 const std::vector<ValueLiteralPair>& right_encoding =
447 right_encodings.at(min_index);
448 DCHECK_EQ(left_encoding.size(), right_encoding.size());
449
450 // Build decomposition of the product.
451 std::vector<LiteralValueValue> terms;
452 for (int i = 0; i < left_encoding.size(); ++i) {
453 const Literal literal = left_encoding[i].literal;
454 DCHECK_EQ(literal, right_encoding[i].literal);
455 terms.push_back({literal, left.ValueAt(left_encoding[i].value),
456 right.ValueAt(right_encoding[i].value)});
457 }
458
459 return terms;
460}

462// TODO(user): Experiment with x * x where constants = 0, x is
463// fully encoded, and the domain is small.
464bool DetectLinearEncodingOfProducts(const AffineExpression& left,
465 const AffineExpression& right, Model* model,
466 LinearConstraintBuilder* builder) {
467 DCHECK(builder != nullptr);
468 builder->Clear();
469
470 IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
471 if (integer_trail->IsFixed(left)) {
472 if (integer_trail->IsFixed(right)) {
473 builder->AddConstant(integer_trail->FixedValue(left) *
474 integer_trail->FixedValue(right));
475 return true;
476 }
477 builder->AddTerm(right, integer_trail->FixedValue(left));
478 return true;
479 }
480
481 if (integer_trail->IsFixed(right)) {
482 builder->AddTerm(left, integer_trail->FixedValue(right));
483 return true;
484 }
485
486 // Linearization is possible if both left and right have the same Boolean
487 // variable.
488 if (PositiveVariable(left.var) == PositiveVariable(right.var) &&
489 integer_trail->LowerBound(PositiveVariable(left.var)) == 0 &&
490 integer_trail->UpperBound(PositiveVariable(left.var)) == 1) {
491 const IntegerValue left_coeff =
492 VariableIsPositive(left.var) ? left.coeff : -left.coeff;
493 const IntegerValue right_coeff =
494 VariableIsPositive(right.var) ? right.coeff : -right.coeff;
495 builder->AddTerm(PositiveVariable(left.var),
496 left_coeff * right_coeff + left.constant * right_coeff +
497 left_coeff * right.constant);
498 builder->AddConstant(left.constant * right.constant);
499 return true;
500 }
501
502 const std::vector<LiteralValueValue> product =
503 TryToDecomposeProduct(left, right, model);
504 if (product.empty()) return false;
505
506 IntegerValue min_coefficient = kMaxIntegerValue;
507 for (const LiteralValueValue& term : product) {
508 min_coefficient =
509 std::min(min_coefficient, term.left_value * term.right_value);
510 }
511
512 for (const LiteralValueValue& term : product) {
513 const IntegerValue coefficient =
514 term.left_value * term.right_value - min_coefficient;
515 if (coefficient == 0) continue;
516 if (!builder->AddLiteralTerm(term.literal, coefficient)) {
517 return false;
518 }
519 }
520 builder->AddConstant(min_coefficient);
521 return true;
522}

552void ProductDetector::ProcessTernaryClause(
553 absl::Span<const Literal> ternary_clause) {
554 if (!enabled_) return;
555 if (ternary_clause.size() != 3) return;
556 ++num_processed_ternary_;
557 candidates_[GetKey(ternary_clause[0].Index(), ternary_clause[1].Index())]
558 .push_back(ternary_clause[2].Index());
559 candidates_[GetKey(ternary_clause[0].Index(), ternary_clause[2].Index())]
560 .push_back(ternary_clause[1].Index());
561 candidates_[GetKey(ternary_clause[1].Index(), ternary_clause[2].Index())]
562 .push_back(ternary_clause[0].Index());
563
564 // We mark the literal of the ternary clause as seen.
565 // Only a => b with a seen need to be looked at.
566 for (const Literal l : ternary_clause) {
567 if (l.Index() >= seen_.size()) seen_.resize(l.Index() + 1);
568 seen_[l.Index()] = true;
569 }
570}

585// TODO(user): As product are discovered, we could remove entries from our
586// hash maps!
587void ProductDetector::ProcessBinaryClause(
588 absl::Span<const Literal> binary_clause) {
589 if (!enabled_) return;
590 if (binary_clause.size() != 2) return;
591 ++num_processed_binary_;
592 const std::array<LiteralIndex, 2> key =
593 GetKey(binary_clause[0].NegatedIndex(), binary_clause[1].NegatedIndex());
594 std::array<LiteralIndex, 3> ternary;
595 for (const LiteralIndex l : candidates_[key]) {
596 ternary[0] = key[0];
597 ternary[1] = key[1];
598 ternary[2] = l;
599 std::sort(ternary.begin(), ternary.end());
600 const int l_index = ternary[0] == l ? 0 : ternary[1] == l ? 1 : 2;
601 std::bitset<3>& bs = detector_[ternary];
602 if (bs[l_index]) continue;
603 bs[l_index] = true;
604 if (bs[0] && bs[1] && l_index != 2) {
605 ProcessNewProduct(ternary[2], Literal(ternary[0]).NegatedIndex(),
606 Literal(ternary[1]).NegatedIndex());
607 }
608 if (bs[0] && bs[2] && l_index != 1) {
609 ProcessNewProduct(ternary[1], Literal(ternary[0]).NegatedIndex(),
610 Literal(ternary[2]).NegatedIndex());
611 }
612 if (bs[1] && bs[2] && l_index != 0) {
613 ProcessNewProduct(ternary[0], Literal(ternary[1]).NegatedIndex(),
614 Literal(ternary[2]).NegatedIndex());
615 }
616 }
617}

619void ProductDetector::ProcessImplicationGraph(BinaryImplicationGraph* graph) {
620 if (!enabled_) return;
621 for (LiteralIndex a(0); a < seen_.size(); ++a) {
622 if (!seen_[a]) continue;
623 if (trail_->Assignment().LiteralIsAssigned(Literal(a))) continue;
624 const Literal not_a = Literal(a).Negated();
625 for (const Literal b : graph->DirectImplications(Literal(a))) {
626 ProcessBinaryClause({not_a, b}); // a => b;
627 }
628 }

631void ProductDetector::ProcessTrailAtLevelOne() {
632 if (!enabled_) return;
633 if (trail_->CurrentDecisionLevel() != 1) return;
634 ++num_trail_updates_;
635
636 const SatSolver::Decision decision = sat_solver_->Decisions()[0];
637 if (decision.literal.Index() >= seen_.size() ||
638 !seen_[decision.literal.Index()]) {
639 return;
640 }
641 const Literal not_a = decision.literal.Negated();
642 const int current_index = trail_->Index();
643 for (int i = decision.trail_index + 1; i < current_index; ++i) {
644 const Literal b = (*trail_)[i];
645 ProcessBinaryClause({not_a, b});
646 }

662void ProductDetector::ProcessNewProduct(LiteralIndex p, LiteralIndex a,
663 LiteralIndex b) {
664 // If many literal correspond to the same product, we just keep one.
665 ++num_products_;
666 products_[GetKey(a, b)] = p;
667
668 // This is used by ProductIsLinearizable().
669 has_product_.insert(
670 GetKey(Literal(a).IsPositive() ? a : Literal(a).NegatedIndex(),
671 Literal(b).IsPositive() ? b : Literal(b).NegatedIndex()));

674bool ProductDetector::ProductIsLinearizable(IntegerVariable a,
675 IntegerVariable b) const {
676 if (a == b) return true;
677 if (a == NegationOf(b)) return true;
678
679 // Otherwise, we need both a and b to be expressible as linear expression
680 // involving Booleans whose product is also expressible.
681 if (integer_trail_->InitialVariableDomain(a).Size() != 2) return false;
682 if (integer_trail_->InitialVariableDomain(b).Size() != 2) return false;
683
684 const LiteralIndex la =
685 integer_encoder_->GetAssociatedLiteral(IntegerLiteral::GreaterOrEqual(
686 a, integer_trail_->LevelZeroUpperBound(a)));
687 if (la == kNoLiteralIndex) return false;
688
689 const LiteralIndex lb =
690 integer_encoder_->GetAssociatedLiteral(IntegerLiteral::GreaterOrEqual(
691 b, integer_trail_->LevelZeroUpperBound(b)));
692 if (lb == kNoLiteralIndex) return false;
693
694 // Any product involving la/not(la) * lb/not(lb) can be used.
695 return has_product_.contains(
696 GetKey(Literal(la).IsPositive() ? la : Literal(la).NegatedIndex(),
697 Literal(lb).IsPositive() ? lb : Literal(lb).NegatedIndex()));
698}

719void ProductDetector::ProcessConditionalEquality(Literal l, IntegerVariable x,
720 IntegerVariable y) {
721 ++num_conditional_equalities_;
722 if (x == y) return;
723
724 // We process both possibilities (product = x or product = y).
725 for (int i = 0; i < 2; ++i) {
726 if (!VariableIsPositive(x)) {
727 x = NegationOf(x);
728 y = NegationOf(y);
729 }
730 bool seen = false;
731
732 // TODO(user): Linear scan can be bad if b => X = many other variables.
733 // Hopefully this will not be common.
734 std::vector<IntegerVariable>& others =
735 conditional_equalities_[{l.Index(), x}];
736 for (const IntegerVariable o : others) {
737 if (o == y) {
738 seen = true;
739 break;
740 }
741 }
742
743 if (!seen) {
744 others.push_back(y);
745 if (conditional_zeros_.contains({l.NegatedIndex(), x})) {
746 ProcessNewProduct(/*p=*/x, l, y);
747 }
748 }
749 std::swap(x, y);
750 }
751}

753void ProductDetector::ProcessConditionalZero(Literal l, IntegerVariable p) {
754 ++num_conditional_zeros_;
755 p = PositiveVariable(p);
756 auto [_, inserted] = conditional_zeros_.insert({l.Index(), p});
757 if (inserted) {
758 const auto it = conditional_equalities_.find({l.NegatedIndex(), p});
759 if (it != conditional_equalities_.end()) {
760 for (const IntegerVariable x : it->second) {
761 ProcessNewProduct(p, l.Negated(), x);
762 }
763 }
764 }
