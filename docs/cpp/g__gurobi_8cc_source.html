35constexpr int kGrbOk = 0;
36
37struct UserCallbackData {
38 Gurobi::Callback user_cb;
39 absl::Status status = absl::OkStatus();
40 Gurobi* gurobi = nullptr;
41};
42
43int GurobiCallback(GRBmodel* const model, void* const cbdata, const int where,
44 void* const usrdata) {
45 CHECK(usrdata != nullptr);
46 CHECK(model != nullptr);
47 auto user_cb_data = static_cast<UserCallbackData*>(usrdata);
48 CHECK_EQ(model, user_cb_data->gurobi->model());
49 // NOTE: if a previous callback failed, we never run the callback again.
50 if (!user_cb_data->status.ok()) {
51 return GRB_ERROR_CALLBACK;
52 }
53 const Gurobi::CallbackContext context(user_cb_data->gurobi, cbdata, where);
54 user_cb_data->status = user_cb_data->user_cb(context);
55 if (!user_cb_data->status.ok()) {
56 user_cb_data->gurobi->Terminate();
57 return GRB_ERROR_CALLBACK;
58 }
59 return kGrbOk;
60}

70absl::StatusOr<GRBenvUniquePtr> GurobiNewPrimaryEnv(
71 const std::optional<GurobiIsvKey>& isv_key) {
72 GRBenv* naked_primary_env = nullptr;
73 int err;
74 std::string_view init_env_method;
75 if (isv_key.has_value()) {
76 err = GRBisqp(&naked_primary_env, /*logfilename=*/
77 nullptr, isv_key->name.c_str(),
78 isv_key->application_name.c_str(), isv_key->expiration,
79 isv_key->key.c_str());
80 init_env_method = "GRBisqp()";
81 } else {
82 err = GRBloadenv(&naked_primary_env, /*logfilename=*/nullptr);
83 init_env_method = "GRBloadenv()";
84 }
85 if (err != kGrbOk) {
86 // Surprisingly, even when Gurobi fails to load the environment, it still
87 // creates one. Here we make sure to free it properly.
88 //
89 // We can also use it with GRBgeterrormsg() to get the associated error
90 // message that goes with the error and the contains additional data like
91 // the user, the host and the hostid.
92 const GRBenvUniquePtr primary_env(naked_primary_env);
93 return util::InvalidArgumentErrorBuilder()
94 << "failed to create Gurobi primary environment, " << init_env_method
95 << " returned the error (" << err
96 << "): " << GRBgeterrormsg(primary_env.get());
97 }
98 return GRBenvUniquePtr(naked_primary_env);
99}

116Gurobi::Gurobi(GRBenvUniquePtr optional_owned_primary_env,
117 GRBmodel* const model, GRBenv* const model_env)
118 : owned_primary_env_(std::move(optional_owned_primary_env)),
119 gurobi_model_(ABSL_DIE_IF_NULL(model)),
120 model_env_(ABSL_DIE_IF_NULL(model_env)) {}
121
122absl::StatusOr<std::unique_ptr<Gurobi>> Gurobi::New(
123 GRBenvUniquePtr optional_owned_primary_env, GRBenv* const primary_env) {
124 CHECK(primary_env != nullptr);
125 GRBmodel* model = nullptr;
126 const int err = GRBnewmodel(primary_env, &model,
127 /*Pname=*/nullptr,
128 /*numvars=*/0,
129 /*obj=*/nullptr, /*lb=*/nullptr,
130 /*ub=*/nullptr, /*vtype=*/nullptr,
131 /*varnames=*/nullptr);
132 if (err != kGrbOk) {
133 return util::InvalidArgumentErrorBuilder()
134 << "Error creating gurobi model on GRBnewmodel(), error code: "
135 << err << " message: " << GRBgeterrormsg(primary_env);
136 }
137 CHECK(model != nullptr);
138 GRBenv* const model_env = GRBgetenv(model);
139
140 if (VLOG_IS_ON(3)) {
141 int gurobi_major, gurobi_minor, gurobi_technical;
142 GRBversion(&gurobi_major, &gurobi_minor, &gurobi_technical);
143 VLOG(3) << absl::StrFormat(
144 "Successfully created model for Gurobi v%d.%d.%d (%s)", gurobi_major,
145 gurobi_minor, gurobi_technical, GRBplatform());
146 }
147 return absl::WrapUnique(
148 new Gurobi(std::move(optional_owned_primary_env), model, model_env));
149}

178absl::Status Gurobi::AddVars(const absl::Span<const int> vbegin,
179 const absl::Span<const int> vind,
180 const absl::Span<const double> vval,
181 const absl::Span<const double> obj,
182 const absl::Span<const double> lb,
183 const absl::Span<const double> ub,
184 const absl::Span<const char> vtype,
185 const absl::Span<const std::string> names) {
186 CHECK_EQ(vind.size(), vval.size());
187 const int num_vars = static_cast<int>(lb.size());
188 CHECK_EQ(ub.size(), num_vars);
189 CHECK_EQ(vtype.size(), num_vars);
190 double* c_obj = nullptr;
191 if (!obj.empty()) {
192 CHECK_EQ(obj.size(), num_vars);
193 c_obj = const_cast<double*>(obj.data());
194 }
195 if (!vbegin.empty()) {
196 CHECK_EQ(vbegin.size(), num_vars);
197 }
198 char** c_names = nullptr;
199 std::vector<char*> c_names_data;
200 if (!names.empty()) {
201 CHECK_EQ(num_vars, names.size());
202 for (const std::string& name : names) {
203 c_names_data.push_back(const_cast<char*>(name.c_str()));
204 }
205 c_names = c_names_data.data();
206 }
207 return ToStatus(GRBaddvars(/*model=*/gurobi_model_, /*numvars=*/num_vars,
208 /*numnz=*/vind.size(),
209 /*vbeg=*/const_cast<int*>(vbegin.data()),
210 /*vind=*/const_cast<int*>(vind.data()),
211 /*vval=*/const_cast<double*>(vval.data()),
212 /*obj=*/c_obj,
213 /*lb=*/const_cast<double*>(lb.data()),
214 /*ub=*/const_cast<double*>(ub.data()),
215 /*vtype=*/const_cast<char*>(vtype.data()),
216 /*varnames=*/c_names));
217}

329absl::StatusOr<Gurobi::SparseMat> Gurobi::GetVars(const int first_var,
330 const int num_vars) {
331 SparseMat result;
332 ASSIGN_OR_RETURN(const int nnz, GetNnz(first_var, num_vars));
333 result.begins.resize(num_vars);
334 result.inds.resize(nnz);
335 result.vals.resize(nnz);
336 int read_nnz = 0;
337 RETURN_IF_ERROR(ToStatus(
338 GRBgetvars(gurobi_model_, &read_nnz, result.begins.data(),
339 result.inds.data(), result.vals.data(), first_var, num_vars)));
340 CHECK_EQ(read_nnz, nnz);
341 return result;
342}

348absl::Status Gurobi::Optimize(Callback cb) {
349 bool needs_cb_cleanup = false;
350 UserCallbackData user_cb_data;
351 if (cb != nullptr) {
352 user_cb_data.user_cb = std::move(cb);
353 user_cb_data.gurobi = this;
354 RETURN_IF_ERROR(ToStatus(
355 GRBsetcallbackfunc(gurobi_model_, GurobiCallback, &user_cb_data)));
356 needs_cb_cleanup = true;
357 }
358
359 // Failsafe to try and clear the callback if there is another error. We cannot
360 // raise an error in a destructor, we can only log it.
361 auto callback_cleanup = absl::MakeCleanup([&]() {
362 if (needs_cb_cleanup) {
363 int error = GRBsetcallbackfunc(gurobi_model_, nullptr, nullptr);
364 if (error != kGrbOk) {
365 LOG(ERROR) << "Error cleaning up callback";
366 }
367 }
368 });
369 absl::Status solve_status = ToStatus(GRBoptimize(gurobi_model_));
370 RETURN_IF_ERROR(user_cb_data.status) << "Error in Optimize callback.";
371 RETURN_IF_ERROR(solve_status);
372 if (needs_cb_cleanup) {
373 needs_cb_cleanup = false;
374 RETURN_IF_ERROR(
375 ToStatus(GRBsetcallbackfunc(gurobi_model_, nullptr, nullptr)));
376 }
377 return absl::OkStatus();
378}
