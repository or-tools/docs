37constexpr int kGrbOk = 0;
38
39struct UserCallbackData {
40 Gurobi::Callback user_cb;
41 absl::Status status = absl::OkStatus();
42 Gurobi* gurobi = nullptr;
43};
44
45int GurobiCallback(GRBmodel* const model, void* const cbdata, const int where,
46 void* const usrdata) {
47 CHECK(usrdata != nullptr);
48 CHECK(model != nullptr);
49 auto user_cb_data = static_cast<UserCallbackData*>(usrdata);
50 CHECK_EQ(model, user_cb_data->gurobi->model());
51 // NOTE: if a previous callback failed, we never run the callback again.
52 if (!user_cb_data->status.ok()) {
53 return GRB_ERROR_CALLBACK;
54 }
55 const Gurobi::CallbackContext context(user_cb_data->gurobi, cbdata, where);
56 user_cb_data->status = user_cb_data->user_cb(context);
57 if (!user_cb_data->status.ok()) {
58 user_cb_data->gurobi->Terminate();
59 return GRB_ERROR_CALLBACK;
60 }
61 return kGrbOk;
62}

72absl::StatusOr<GRBenvUniquePtr> GurobiNewPrimaryEnv(
73 const std::optional<GurobiIsvKey>& isv_key) {
74 GRBenv* naked_primary_env = nullptr;
75 int err;
76 std::string_view init_env_method;
77 if (isv_key.has_value()) {
78 err = GRBisqp(&naked_primary_env, /*logfilename=*/
79 nullptr, isv_key->name.c_str(),
80 isv_key->application_name.c_str(), isv_key->expiration,
81 isv_key->key.c_str());
82 init_env_method = "GRBisqp()";
83 } else {
84 err = GRBloadenv(&naked_primary_env, /*logfilename=*/nullptr);
85 init_env_method = "GRBloadenv()";
86 }
87 if (err != kGrbOk) {
88 // Surprisingly, even when Gurobi fails to load the environment, it still
89 // creates one. Here we make sure to free it properly.
90 //
91 // We can also use it with GRBgeterrormsg() to get the associated error
92 // message that goes with the error and the contains additional data like
93 // the user, the host and the hostid.
94 const GRBenvUniquePtr primary_env(naked_primary_env);
95 return util::InvalidArgumentErrorBuilder()
96 << "failed to create Gurobi primary environment, " << init_env_method
97 << " returned the error (" << err
98 << "): " << GRBgeterrormsg(primary_env.get());
99 }
100 return GRBenvUniquePtr(naked_primary_env);
101}

109absl::StatusOr<std::unique_ptr<Gurobi>> Gurobi::New(

111 if (primary_env == nullptr) {
112 ASSIGN_OR_RETURN(primary_env, GurobiNewPrimaryEnv());

118Gurobi::Gurobi(GRBenvUniquePtr optional_owned_primary_env,
119 GRBmodel* const model, GRBenv* const model_env)
120 : owned_primary_env_(std::move(optional_owned_primary_env)),
121 gurobi_model_(ABSL_DIE_IF_NULL(model)),
122 model_env_(ABSL_DIE_IF_NULL(model_env)) {}
123
124absl::StatusOr<std::unique_ptr<Gurobi>> Gurobi::New(
125 GRBenvUniquePtr optional_owned_primary_env, GRBenv* const primary_env) {
126 CHECK(primary_env != nullptr);
127 GRBmodel* model = nullptr;
128 const int err = GRBnewmodel(primary_env, &model,
129 /*Pname=*/nullptr,
130 /*numvars=*/0,
131 /*obj=*/nullptr, /*lb=*/nullptr,
132 /*ub=*/nullptr, /*vtype=*/nullptr,
133 /*varnames=*/nullptr);
134 if (err != kGrbOk) {
135 return util::InvalidArgumentErrorBuilder()
136 << "Error creating gurobi model on GRBnewmodel(), error code: "
137 << err << " message: " << GRBgeterrormsg(primary_env);
138 }
139 CHECK(model != nullptr);
140 GRBenv* const model_env = GRBgetenv(model);
141
142 if (VLOG_IS_ON(3)) {
143 int gurobi_major, gurobi_minor, gurobi_technical;
144 GRBversion(&gurobi_major, &gurobi_minor, &gurobi_technical);
145 VLOG(3) << absl::StrFormat(
146 "Successfully created model for Gurobi v%d.%d.%d (%s)", gurobi_major,
147 gurobi_minor, gurobi_technical, GRBplatform());
148 }
149 return absl::WrapUnique(
150 new Gurobi(std::move(optional_owned_primary_env), model, model_env));
151}

155 if (err != kGrbOk) {
156 LOG(ERROR) << "Error freeing gurobi model, code: " << err
157 << ", message: " << GRBgeterrormsg(model_env_);
158 }

201absl::Status Gurobi::AddVars(const absl::Span<const int> vbegin,
202 const absl::Span<const int> vind,
203 const absl::Span<const double> vval,
204 const absl::Span<const double> obj,
205 const absl::Span<const double> lb,
206 const absl::Span<const double> ub,
207 const absl::Span<const char> vtype,
208 const absl::Span<const std::string> names) {
209 CHECK_EQ(vind.size(), vval.size());
210 const int num_vars = static_cast<int>(lb.size());
211 CHECK_EQ(ub.size(), num_vars);
212 CHECK_EQ(vtype.size(), num_vars);
213 double* c_obj = nullptr;
214 if (!obj.empty()) {
215 CHECK_EQ(obj.size(), num_vars);
216 c_obj = const_cast<double*>(obj.data());
217 }
218 if (!vbegin.empty()) {
219 CHECK_EQ(vbegin.size(), num_vars);
220 }
221 char** c_names = nullptr;
222 std::vector<char*> c_names_data;
223 if (!names.empty()) {
224 CHECK_EQ(num_vars, names.size());
225 for (const std::string& name : names) {
226 c_names_data.push_back(const_cast<char*>(name.c_str()));
227 }
228 c_names = c_names_data.data();
229 }
230 return ToStatus(GRBaddvars(/*model=*/gurobi_model_, /*numvars=*/num_vars,
231 /*numnz=*/vind.size(),
232 /*vbeg=*/const_cast<int*>(vbegin.data()),
233 /*vind=*/const_cast<int*>(vind.data()),
234 /*vval=*/const_cast<double*>(vval.data()),
235 /*obj=*/c_obj,
236 /*lb=*/const_cast<double*>(lb.data()),
237 /*ub=*/const_cast<double*>(ub.data()),
238 /*vtype=*/const_cast<char*>(vtype.data()),
239 /*varnames=*/c_names));
240}

448 CHECK_EQ(read_nnz, nnz);
449 return result;

456absl::Status Gurobi::Optimize(Callback cb) {
457 bool needs_cb_cleanup = false;
458 UserCallbackData user_cb_data;
459 if (cb != nullptr) {
460 user_cb_data.user_cb = std::move(cb);
461 user_cb_data.gurobi = this;
462 RETURN_IF_ERROR(ToStatus(
463 GRBsetcallbackfunc(gurobi_model_, GurobiCallback, &user_cb_data)));
464 needs_cb_cleanup = true;
465 }
466
467 // Failsafe to try and clear the callback if there is another error. We cannot
468 // raise an error in a destructor, we can only log it.
469 auto callback_cleanup = absl::MakeCleanup([&]() {
470 if (needs_cb_cleanup) {
471 int error = GRBsetcallbackfunc(gurobi_model_, nullptr, nullptr);
472 if (error != kGrbOk) {
473 LOG(ERROR) << "Error cleaning up callback";
474 }
475 }
476 });
477 absl::Status solve_status = ToStatus(GRBoptimize(gurobi_model_));
478 RETURN_IF_ERROR(user_cb_data.status) << "Error in Optimize callback.";
479 RETURN_IF_ERROR(solve_status);
480 if (needs_cb_cleanup) {
481 needs_cb_cleanup = false;
482 RETURN_IF_ERROR(
483 ToStatus(GRBsetcallbackfunc(gurobi_model_, nullptr, nullptr)));
484 }
485 return absl::OkStatus();
486}
