76std::string FormatTable(std::vector<std::vector<std::string>>& table,
77 int spacing) {
78 if (table.size() > 1) {
79 // We order by name.
80 std::sort(table.begin() + 1, table.end());
81 }
82
83 std::vector<int> widths;
84 for (const std::vector<std::string>& line : table) {
85 if (line.size() > widths.size()) widths.resize(line.size(), spacing);
86 for (int j = 0; j < line.size(); ++j) {
87 widths[j] = std::max<int>(widths[j], line[j].size() + spacing);
88 }
89 }
90 std::string output;
91 for (int i = 0; i < table.size(); ++i) {
92 for (int j = 0; j < table[i].size(); ++j) {
93 if (i == 0 && j == 0) {
94 // We currently only left align the table name.
95 absl::StrAppend(&output, LeftAlign(table[i][j], widths[j]));
96 } else {
97 absl::StrAppend(&output, RightAlign(table[i][j], widths[j]));
98 }
99 }
100 absl::StrAppend(&output, "\n");
101 }
102 return output;
103}

145// Using the extended Euclidian algo, we find a and b such that
146// a x + b m = gcd(x, m)
147// https://en.wikipedia.org/wiki/Extended_Euclidean_algorithm
148int64_t ModularInverse(int64_t x, int64_t m) {
149 DCHECK_GE(x, 0);
150 DCHECK_LT(x, m);
151
152 int64_t r[2] = {m, x};
153 int64_t t[2] = {0, 1};
154 int64_t q;
155
156 // We only keep the last two terms of the sequences with the "^1" trick:
157 //
158 // q = r[i-2] / r[i-1]
159 // r[i] = r[i-2] % r[i-1]
160 // t[i] = t[i-2] - t[i-1] * q
161 //
162 // We always have:
163 // - gcd(r[i], r[i - 1]) = gcd(r[i - 1], r[i - 2])
164 // - x * t[i] + m * t[i - 1] = r[i]
165 int i = 0;
166 for (; r[i ^ 1] != 0; i ^= 1) {
167 QuotientAndRemainder(r[i], r[i ^ 1], q, r[i]);
168 t[i] -= t[i ^ 1] * q;
169 }
170
171 // If the gcd is not one, there is no inverse, we returns 0.
172 if (r[i] != 1) return 0;
173
174 // Correct the result so that it is in [0, m). Note that abs(t[i]) is known to
175 // be less than or equal to x / 2, and we have thorough unit-tests.
176 if (t[i] < 0) t[i] += m;
177
178 return t[i];
179}

181int64_t PositiveMod(int64_t x, int64_t m) {
182 const int64_t r = x % m;
183 return r < 0 ? r + m : r;
184}

186int64_t ProductWithModularInverse(int64_t coeff, int64_t mod, int64_t rhs) {
187 DCHECK_NE(coeff, 0);
188 DCHECK_NE(mod, 0);
189
190 mod = std::abs(mod);
191 if (rhs == 0 || mod == 1) return 0;
192 DCHECK_EQ(std::gcd(std::abs(coeff), mod), 1);
193
194 // Make both in [0, mod).
195 coeff = PositiveMod(coeff, mod);
196 rhs = PositiveMod(rhs, mod);
197
198 // From X * coeff % mod = rhs
199 // We deduce that X % mod = rhs * inverse % mod
200 const int64_t inverse = ModularInverse(coeff, mod);
201 CHECK_NE(inverse, 0);
202
203 // We make the operation in 128 bits to be sure not to have any overflow here.
204 const absl::int128 p = absl::int128{inverse} * absl::int128{rhs};
205 return static_cast<int64_t>(p % absl::int128{mod});
206}

208bool SolveDiophantineEquationOfSizeTwo(int64_t& a, int64_t& b, int64_t& cte,
209 int64_t& x0, int64_t& y0) {
210 CHECK_NE(a, 0);
211 CHECK_NE(b, 0);
212 CHECK_NE(a, std::numeric_limits<int64_t>::min());
213 CHECK_NE(b, std::numeric_limits<int64_t>::min());
214
215 const int64_t gcd = std::gcd(std::abs(a), std::abs(b));
216 if (cte % gcd != 0) return false;
217 a /= gcd;
218 b /= gcd;
219 cte /= gcd;
220
221 // The simple case where (0, 0) is a solution.
222 if (cte == 0) {
223 x0 = y0 = 0;
224 return true;
225 }
226
227 // We solve a * X + b * Y = cte
228 // We take a valid x0 in [0, b) by considering the equation mod b.
229 x0 = ProductWithModularInverse(a, b, cte);
230
231 // We choose x0 of the same sign as cte.
232 if (cte < 0 && x0 != 0) x0 -= std::abs(b);
233
234 // By plugging X = x0 + b * Z
235 // We have a * (x0 + b * Z) + b * Y = cte
236 // so a * b * Z + b * Y = cte - a * x0;
237 // and y0 = (cte - a * x0) / b (with an exact division by construction).
238 const absl::int128 t = absl::int128{cte} - absl::int128{a} * absl::int128{x0};
239 DCHECK_EQ(t % absl::int128{b}, absl::int128{0});
240
241 // Overflow-wise, there is two cases for cte > 0:
242 // - a * x0 <= cte, in this case y0 will not overflow (<= cte).
243 // - a * x0 > cte, in this case y0 will be in (-a, 0].
244 const absl::int128 r = t / absl::int128{b};
245 DCHECK_LE(r, absl::int128{std::numeric_limits<int64_t>::max()});
246 DCHECK_GE(r, absl::int128{std::numeric_limits<int64_t>::min()});
247
248 y0 = static_cast<int64_t>(r);
249 return true;
250}

252// TODO(user): Find better implementation? In pratice passing via double is
253// almost always correct, but the CapProd() might be a bit slow. However this
254// is only called when we do propagate something.
255int64_t FloorSquareRoot(int64_t a) {
256 int64_t result =
257 static_cast<int64_t>(std::floor(std::sqrt(static_cast<double>(a))));
258 while (CapProd(result, result) > a) --result;
259 while (CapProd(result + 1, result + 1) <= a) ++result;
260 return result;
261}
262
263// TODO(user): Find better implementation?
264int64_t CeilSquareRoot(int64_t a) {
265 int64_t result =
266 static_cast<int64_t>(std::ceil(std::sqrt(static_cast<double>(a))));
267 while (CapProd(result, result) < a) ++result;
268 while ((result - 1) * (result - 1) >= a) --result;
269 return result;
270}

272int64_t ClosestMultiple(int64_t value, int64_t base) {
273 if (value < 0) return -ClosestMultiple(-value, base);
274 int64_t result = value / base * base;
275 if (value - result > base / 2) result += base;
276 return result;
277}

279bool LinearInequalityCanBeReducedWithClosestMultiple(
280 int64_t base, absl::Span<const int64_t> coeffs,
281 absl::Span<const int64_t> lbs, absl::Span<const int64_t> ubs, int64_t rhs,
282 int64_t* new_rhs) {
283 // Precompute some bounds for the equation base * X + error <= rhs.
284 int64_t max_activity = 0;
285 int64_t max_x = 0;
286 int64_t min_error = 0;
287 const int num_terms = coeffs.size();
288 if (num_terms == 0) return false;
289 for (int i = 0; i < num_terms; ++i) {
290 const int64_t coeff = coeffs[i];
291 CHECK_GT(coeff, 0);
292 const int64_t closest = ClosestMultiple(coeff, base);
293 max_activity += coeff * ubs[i];
294 max_x += closest / base * ubs[i];
295
296 const int64_t error = coeff - closest;
297 if (error >= 0) {
298 min_error += error * lbs[i];
299 } else {
300 min_error += error * ubs[i];
301 }
302 }
303
304 if (max_activity <= rhs) {
305 // The constraint is trivially true.
306 *new_rhs = max_x;
307 return true;
308 }
309
310 // This is the max error assuming that activity > rhs.
311 int64_t max_error_if_invalid = 0;
312 const int64_t slack = max_activity - rhs - 1;
313 for (int i = 0; i < num_terms; ++i) {
314 const int64_t coeff = coeffs[i];
315 const int64_t closest = ClosestMultiple(coeff, base);
316 const int64_t error = coeff - closest;
317 if (error >= 0) {
318 max_error_if_invalid += error * ubs[i];
319 } else {
320 const int64_t lb = std::max(lbs[i], ubs[i] - slack / coeff);
321 max_error_if_invalid += error * lb;
322 }
323 }
324
325 // We have old solution valid =>
326 // base * X + error <= rhs
327 // base * X <= rhs - error
328 // base * X <= rhs - min_error
329 // X <= new_rhs
330 *new_rhs = std::min(max_x, MathUtil::FloorOfRatio(rhs - min_error, base));
331
332 // And we have old solution invalid =>
333 // base * X + error >= rhs + 1
334 // base * X >= rhs + 1 - max_error_if_invalid
335 // X >= infeasibility_bound
336 const int64_t infeasibility_bound =
337 MathUtil::CeilOfRatio(rhs + 1 - max_error_if_invalid, base);
338
339 // If the two bounds can be separated, we have an equivalence !
340 return *new_rhs < infeasibility_bound;
341}

343int MoveOneUnprocessedLiteralLast(
344 const absl::btree_set<LiteralIndex>& processed, int relevant_prefix_size,
345 std::vector<Literal>* literals) {
346 if (literals->empty()) return -1;
347 if (!processed.contains(literals->back().Index())) {
348 return std::min<int>(relevant_prefix_size, literals->size());
349 }
350
351 // To get O(n log n) size of suffixes, we will first process the last n/2
352 // literals, we then move all of them first and process the n/2 literals left.
353 // We use the same algorithm recursively. The sum of the suffixes' size S(n)
354 // is thus S(n/2) + n + S(n/2). That gives us the correct complexity. The code
355 // below simulates one step of this algorithm and is made to be "robust" when
356 // from one call to the next, some literals have been removed (but the order
357 // of literals is preserved).
358 int num_processed = 0;
359 int num_not_processed = 0;
360 int target_prefix_size = literals->size() - 1;
361 for (int i = literals->size() - 1; i >= 0; i--) {
362 if (processed.contains((*literals)[i].Index())) {
363 ++num_processed;
364 } else {
365 ++num_not_processed;
366 target_prefix_size = i;
367 }
368 if (num_not_processed >= num_processed) break;
369 }
370 if (num_not_processed == 0) return -1;
371 target_prefix_size = std::min(target_prefix_size, relevant_prefix_size);
372
373 // Once a prefix size has been decided, it is always better to
374 // enqueue the literal already processed first.
375 std::stable_partition(
376 literals->begin() + target_prefix_size, literals->end(),
377 [&processed](Literal l) { return processed.contains(l.Index()); });
378 return target_prefix_size;
379}

381int WeightedPick(absl::Span<const double> input, absl::BitGenRef random) {
382 DCHECK(!input.empty());
383
384 double total_weight = 0;
385 for (const double w : input) {
386 total_weight += w;
387 }
388
389 const double weight_point = absl::Uniform(random, 0.0f, total_weight);
390 double total_weight2 = 0;
391 for (int i = 0; i < input.size(); ++i) {
392 total_weight2 += input[i];
393 if (total_weight2 > weight_point) {
394 return i;
395 }
396 }
397 return input.size() - 1;
398}

424double Percentile::GetPercentile(double percent) {
425 CHECK_GT(records_.size(), 0);
426 CHECK_LE(percent, 100.0);
427 CHECK_GE(percent, 0.0);
428 std::vector<double> sorted_records(records_.begin(), records_.end());
429 std::sort(sorted_records.begin(), sorted_records.end());
430 const int num_records = sorted_records.size();
431
432 const double percentile_rank =
433 static_cast<double>(num_records) * percent / 100.0 - 0.5;
434 if (percentile_rank <= 0) {
435 return sorted_records.front();
436 } else if (percentile_rank >= num_records - 1) {
437 return sorted_records.back();
438 }
439 // Interpolate.
440 DCHECK_GE(num_records, 2);
441 DCHECK_LT(percentile_rank, num_records - 1);
442 const int lower_rank = static_cast<int>(std::floor(percentile_rank));
443 DCHECK_LT(lower_rank, num_records - 1);
444 return sorted_records[lower_rank] +
445 (percentile_rank - lower_rank) *
446 (sorted_records[lower_rank + 1] - sorted_records[lower_rank]);
447}

449void CompressTuples(absl::Span<const int64_t> domain_sizes,
450 std::vector<std::vector<int64_t>>* tuples) {
451 if (tuples->empty()) return;
452
453 // Remove duplicates if any.
454 gtl::STLSortAndRemoveDuplicates(tuples);
455
456 const int num_vars = (*tuples)[0].size();
457
458 std::vector<int> to_remove;
459 std::vector<int64_t> tuple_minus_var_i(num_vars - 1);
460 for (int i = 0; i < num_vars; ++i) {
461 const int domain_size = domain_sizes[i];
462 if (domain_size == 1) continue;
463 absl::flat_hash_map<const std::vector<int64_t>, std::vector<int>>
464 masked_tuples_to_indices;
465 for (int t = 0; t < tuples->size(); ++t) {
466 int out = 0;
467 for (int j = 0; j < num_vars; ++j) {
468 if (i == j) continue;
469 tuple_minus_var_i[out++] = (*tuples)[t][j];
470 }
471 masked_tuples_to_indices[tuple_minus_var_i].push_back(t);
472 }
473 to_remove.clear();
474 for (const auto& it : masked_tuples_to_indices) {
475 if (it.second.size() != domain_size) continue;
476 (*tuples)[it.second.front()][i] = kTableAnyValue;
477 to_remove.insert(to_remove.end(), it.second.begin() + 1, it.second.end());
478 }
479 std::sort(to_remove.begin(), to_remove.end(), std::greater<int>());
480 for (const int t : to_remove) {
481 (*tuples)[t] = tuples->back();
482 tuples->pop_back();
483 }
484 }
485}

487void MaxBoundedSubsetSum::Reset(int64_t bound) {
488 DCHECK_GE(bound, 0);
489 gcd_ = 0;
490 sums_ = {0};
491 expanded_sums_.clear();
492 current_max_ = 0;
493 bound_ = bound;
494}

496void MaxBoundedSubsetSum::Add(int64_t value) {
497 if (value == 0) return;
498 if (value > bound_) return;
499 gcd_ = std::gcd(gcd_, value);
500 AddChoicesInternal({value});
501}

503void MaxBoundedSubsetSum::AddChoices(absl::Span<const int64_t> choices) {
504 if (DEBUG_MODE) {
505 for (const int64_t c : choices) {
506 DCHECK_GE(c, 0);
507 }
508 }
509
510 // The max is already reachable or we aborted.
511 if (current_max_ == bound_) return;
512
513 // Filter out zero and values greater than bound_.
514 filtered_values_.clear();
515 for (const int64_t c : choices) {
516 if (c == 0 || c > bound_) continue;
517 filtered_values_.push_back(c);
518 gcd_ = std::gcd(gcd_, c);
519 }
520 if (filtered_values_.empty()) return;
521
522 // So we can abort early in the AddChoicesInternal() inner loops.
523 std::sort(filtered_values_.begin(), filtered_values_.end());
524 AddChoicesInternal(filtered_values_);
525}

527void MaxBoundedSubsetSum::AddMultiples(int64_t coeff, int64_t max_value) {
528 DCHECK_GE(coeff, 0);
529 DCHECK_GE(max_value, 0);
530
531 if (coeff == 0 || max_value == 0) return;
532 if (coeff > bound_) return;
533 if (current_max_ == bound_) return;
534 gcd_ = std::gcd(gcd_, coeff);
535
536 const int64_t num_values = std::min(max_value, FloorOfRatio(bound_, coeff));
537 if (num_values > 10) {
538 // We only keep GCD in this case.
539 sums_.clear();
540 expanded_sums_.clear();
541 current_max_ = FloorOfRatio(bound_, gcd_) * gcd_;
542 return;
543 }
544
545 filtered_values_.clear();
546 for (int multiple = 1; multiple <= num_values; ++multiple) {
547 const int64_t v = multiple * coeff;
548 if (v == bound_) {
549 current_max_ = bound_;
550 return;
551 }
552 filtered_values_.push_back(v);
553 }
554 AddChoicesInternal(filtered_values_);
555}

557void MaxBoundedSubsetSum::AddChoicesInternal(absl::Span<const int64_t> values) {
558 // Mode 1: vector of all possible sums (with duplicates).
559 if (!sums_.empty() && sums_.size() <= max_complexity_per_add_) {
560 const int old_size = sums_.size();
561 for (int i = 0; i < old_size; ++i) {
562 for (const int64_t value : values) {
563 const int64_t s = sums_[i] + value;
564 if (s > bound_) break;
565
566 sums_.push_back(s);
567 current_max_ = std::max(current_max_, s);
568 if (current_max_ == bound_) return; // Abort
569 }
570 }
571 return;
572 }
573
574 // Mode 2: bitset of all possible sums.
575 if (bound_ <= max_complexity_per_add_) {
576 if (!sums_.empty()) {
577 expanded_sums_.assign(bound_ + 1, false);
578 for (const int64_t s : sums_) {
579 expanded_sums_[s] = true;
580 }
581 sums_.clear();
582 }
583
584 // The reverse order is important to not add the current value twice.
585 if (!expanded_sums_.empty()) {
586 for (int64_t i = bound_ - 1; i >= 0; --i) {
587 if (!expanded_sums_[i]) continue;
588 for (const int64_t value : values) {
589 if (i + value > bound_) break;
590
591 expanded_sums_[i + value] = true;
592 current_max_ = std::max(current_max_, i + value);
593 if (current_max_ == bound_) return; // Abort
594 }
595 }
596 return;
597 }
598 }
599
600 // Fall back to gcd_.
601 DCHECK_NE(gcd_, 0);
602 if (gcd_ == 1) {
603 current_max_ = bound_;
604 } else {
605 current_max_ = FloorOfRatio(bound_, gcd_) * gcd_;
606 }
607}

609int64_t MaxBoundedSubsetSum::MaxIfAdded(int64_t candidate) const {
610 if (candidate > bound_ || current_max_ == bound_) return current_max_;
611
612 int64_t current_max = current_max_;
613 // Mode 1: vector of all possible sums (with duplicates).
614 if (!sums_.empty()) {
615 for (const int64_t v : sums_) {
616 if (v + candidate > bound_) continue;
617 if (v + candidate > current_max) {
618 current_max = v + candidate;
619 if (current_max == bound_) return current_max;
620 }
621 }
622 return current_max;
623 }
624
625 // Mode 2: bitset of all possible sums.
626 if (!expanded_sums_.empty()) {
627 const int64_t min_useful = std::max<int64_t>(0, current_max_ - candidate);
628 const int64_t max_useful = bound_ - candidate;
629 for (int64_t v = max_useful; v >= min_useful; --v) {
630 if (expanded_sums_[v]) return v + candidate;
631 }
632 }
633 return current_max_;
634}

636BasicKnapsackSolver::Result BasicKnapsackSolver::Solve(
637 const std::vector<Domain>& domains, const std::vector<int64_t>& coeffs,
638 const std::vector<int64_t>& costs, const Domain& rhs) {
639 const int num_vars = domains.size();
640 if (num_vars == 0) return {};
641
642 int64_t min_activity = 0;
643 int64_t max_domain_size = 0;
644 for (int i = 0; i < num_vars; ++i) {
645 max_domain_size = std::max(max_domain_size, domains[i].Size());
646 if (coeffs[i] > 0) {
647 min_activity += coeffs[i] * domains[i].Min();
648 } else {
649 min_activity += coeffs[i] * domains[i].Max();
650 }
651 }
652
653 // The complexity of our DP will depends on the number of "activity" values
654 // that need to be considered.
655 //
656 // TODO(user): We can also solve efficiently if max_activity - rhs.Min() is
657 // small. Implement.
658 const int64_t num_values = rhs.Max() - min_activity + 1;
659 if (num_values < 0) {
660 // Problem is clearly infeasible, we can report the result right away.
661 Result result;
662 result.solved = true;
663 result.infeasible = true;
664 return result;
665 }
666
667 // Abort if complexity too large.
668 const int64_t max_work_per_variable = std::min(num_values, max_domain_size);
669 if (rhs.Max() - min_activity > 1e6) return {};
670 if (num_vars * num_values * max_work_per_variable > 1e8) return {};
671
672 // Canonicalize to positive coeffs and non-negative variables.
673 domains_.clear();
674 coeffs_.clear();
675 costs_.clear();
676 for (int i = 0; i < num_vars; ++i) {
677 if (coeffs[i] > 0) {
678 domains_.push_back(domains[i].AdditionWith(Domain(-domains[i].Min())));
679 coeffs_.push_back(coeffs[i]);
680 costs_.push_back(costs[i]);
681 } else {
682 domains_.push_back(
683 domains[i].Negation().AdditionWith(Domain(domains[i].Max())));
684 coeffs_.push_back(-coeffs[i]);
685 costs_.push_back(-costs[i]);
686 }
687 }
688
689 Result result =
690 InternalSolve(num_values, rhs.AdditionWith(Domain(-min_activity)));
691 if (result.solved && !result.infeasible) {
692 // Transform solution back.
693 for (int i = 0; i < num_vars; ++i) {
694 if (coeffs[i] > 0) {
695 result.solution[i] += domains[i].Min();
696 } else {
697 result.solution[i] = domains[i].Max() - result.solution[i];
698 }
699 }
700 }
701 return result;
702}
703
704BasicKnapsackSolver::Result BasicKnapsackSolver::InternalSolve(
705 int64_t num_values, const Domain& rhs) {
706 const int num_vars = domains_.size();
707
708 // The set of DP states that we will fill.
709 var_activity_states_.assign(num_vars, std::vector<State>(num_values));
710
711 // Initialize with first variable.
712 for (const int64_t v : domains_[0].Values()) {
713 const int64_t value = v * coeffs_[0];
714 CHECK_GE(value, 0);
715 if (value >= num_values) break;
716 var_activity_states_[0][value].cost = v * costs_[0];
717 var_activity_states_[0][value].value = v;
718 }
719
720 // Fill rest of the DP states.
721 for (int i = 1; i < num_vars; ++i) {
722 const std::vector<State>& prev = var_activity_states_[i - 1];
723 std::vector<State>& current = var_activity_states_[i];
724 for (int prev_value = 0; prev_value < num_values; ++prev_value) {
725 if (prev[prev_value].cost == std::numeric_limits<int64_t>::max()) {
726 continue;
727 }
728 for (const int64_t v : domains_[i].Values()) {
729 const int64_t value = prev_value + v * coeffs_[i];
730 CHECK_GE(value, 0);
731 if (value >= num_values) break;
732 const int64_t new_cost = prev[prev_value].cost + v * costs_[i];
733 if (new_cost < current[value].cost) {
734 current[value].cost = new_cost;
735 current[value].value = v;
736 }
737 }
738 }
739 }
740
741 Result result;
742 result.solved = true;
743
744 int64_t best_cost = std::numeric_limits<int64_t>::max();
745 int64_t best_activity;
746 for (int v = 0; v < num_values; ++v) {
747 // TODO(user): optimize this?
748 if (!rhs.Contains(v)) continue;
749 if (var_activity_states_.back()[v].cost < best_cost) {
750 best_cost = var_activity_states_.back()[v].cost;
751 best_activity = v;
752 }
753 }
754
755 if (best_cost == std::numeric_limits<int64_t>::max()) {
756 result.infeasible = true;
757 return result;
758 }
759
760 // Recover the values.
761 result.solution.resize(num_vars);
762 int64_t current_activity = best_activity;
763 for (int i = num_vars - 1; i >= 0; --i) {
764 const int64_t var_value = var_activity_states_[i][current_activity].value;
765 result.solution[i] = var_value;
766 current_activity -= coeffs_[i] * var_value;
767 }
768
769 return result;
770}

774// We will call FullyCompressTuplesRecursive() for a set of prefixes of the
775// original tuples, each having the same suffix (in reversed_suffix).
776//
777// For such set, we will compress it on the last variable of the prefixes. We
778// will then for each unique compressed set of value of that variable, call
779// a new FullyCompressTuplesRecursive() on the corresponding subset.
780void FullyCompressTuplesRecursive(
781 absl::Span<const int64_t> domain_sizes,
782 absl::Span<std::vector<int64_t>> tuples,
783 std::vector<absl::InlinedVector<int64_t, 2>>* reversed_suffix,
784 std::vector<std::vector<absl::InlinedVector<int64_t, 2>>>* output) {
785 struct TempData {
786 absl::InlinedVector<int64_t, 2> values;
787 int index;
788
789 bool operator<(const TempData& other) const {
790 return values < other.values;
791 }
792 };
793 std::vector<TempData> temp_data;
794
795 CHECK(!tuples.empty());
796 CHECK(!tuples[0].empty());
797 const int64_t domain_size = domain_sizes[tuples[0].size() - 1];
798
799 // Sort tuples and regroup common prefix in temp_data.
800 std::sort(tuples.begin(), tuples.end());
801 for (int i = 0; i < tuples.size();) {
802 const int start = i;
803 temp_data.push_back({{tuples[start].back()}, start});
804 tuples[start].pop_back();
805 for (++i; i < tuples.size(); ++i) {
806 const int64_t v = tuples[i].back();
807 tuples[i].pop_back();
808 if (tuples[i] == tuples[start]) {
809 temp_data.back().values.push_back(v);
810 } else {
811 tuples[i].push_back(v);
812 break;
813 }
814 }
815
816 // If one of the value is the special value kTableAnyValue, we convert
817 // it to the "empty means any" format.
818 for (const int64_t v : temp_data.back().values) {
819 if (v == kTableAnyValue) {
820 temp_data.back().values.clear();
821 break;
822 }
823 }
824 gtl::STLSortAndRemoveDuplicates(&temp_data.back().values);
825
826 // If values cover the whole domain, we clear the vector. This allows to
827 // use less space and avoid creating uneeded clauses.
828 if (temp_data.back().values.size() == domain_size) {
829 temp_data.back().values.clear();
830 }
831 }
832
833 if (temp_data.size() == 1) {
834 output->push_back({});
835 for (const int64_t v : tuples[temp_data[0].index]) {
836 if (v == kTableAnyValue) {
837 output->back().push_back({});
838 } else {
839 output->back().push_back({v});
840 }
841 }
842 output->back().push_back(temp_data[0].values);
843 for (int i = reversed_suffix->size(); --i >= 0;) {
844 output->back().push_back((*reversed_suffix)[i]);
845 }
846 return;
847 }
848
849 // Sort temp_data and make recursive call for all tuples that share the
850 // same suffix.
851 std::sort(temp_data.begin(), temp_data.end());
852 std::vector<std::vector<int64_t>> temp_tuples;
853 for (int i = 0; i < temp_data.size();) {
854 reversed_suffix->push_back(temp_data[i].values);
855 const int start = i;
856 temp_tuples.clear();
857 for (; i < temp_data.size(); i++) {
858 if (temp_data[start].values != temp_data[i].values) break;
859 temp_tuples.push_back(tuples[temp_data[i].index]);
860 }
861 FullyCompressTuplesRecursive(domain_sizes, absl::MakeSpan(temp_tuples),
862 reversed_suffix, output);
863 reversed_suffix->pop_back();
864 }
865}
866
867} // namespace
868
869// TODO(user): We can probably reuse the tuples memory always and never create
870// new one. We should also be able to code an iterative version of this. Note
871// however that the recursion level is bounded by the number of coluns which
872// should be small.
873std::vector<std::vector<absl::InlinedVector<int64_t, 2>>> FullyCompressTuples(
874 absl::Span<const int64_t> domain_sizes,
875 std::vector<std::vector<int64_t>>* tuples) {
876 std::vector<absl::InlinedVector<int64_t, 2>> reversed_suffix;
877 std::vector<std::vector<absl::InlinedVector<int64_t, 2>>> output;
878 FullyCompressTuplesRecursive(domain_sizes, absl::MakeSpan(*tuples),
879 &reversed_suffix, &output);
880 return output;
881}

885class CliqueDecomposition {
886 public:
887 CliqueDecomposition(const std::vector<std::vector<int>>& graph,
888 absl::BitGenRef random, std::vector<int>* buffer)
889 : graph_(graph), random_(random), buffer_(buffer) {
890 const int n = graph.size();
891 permutation_.resize(n);
892 buffer_->resize(n);
893
894 // TODO(user): Start by sorting by decreasing size of adjacency list?
895 for (int i = 0; i < n; ++i) permutation_[i] = i;
896 }
897
898 // This works in O(m). All possible decomposition are reachable, depending on
899 // the initial permutation.
900 //
901 // TODO(user): It can be made faster within the same complexity though.
902 void DecomposeGreedily() {
903 decomposition_.clear();
904 candidates_.clear();
905
906 const int n = graph_.size();
907 taken_.assign(n, false);
908 temp_.assign(n, false);
909
910 int buffer_index = 0;
911 for (const int i : permutation_) {
912 if (taken_[i]) continue;
913
914 // Save start of amo and output i.
915 const int start = buffer_index;
916 taken_[i] = true;
917 (*buffer_)[buffer_index++] = i;
918
919 candidates_.clear();
920 for (const int c : graph_[i]) {
921 if (!taken_[c]) candidates_.push_back(c);
922 }
923 while (!candidates_.empty()) {
924 // Extend at most one with lowest possible permutation index.
925 int next = candidates_.front();
926 for (const int n : candidates_) {
927 if (permutation_[n] < permutation_[next]) next = n;
928 }
929
930 // Add to current amo.
931 taken_[next] = true;
932 (*buffer_)[buffer_index++] = next;
933
934 // Filter candidates.
935 // TODO(user): With a sorted graph_, same complexity, but likely faster.
936 for (const int head : graph_[next]) temp_[head] = true;
937 int new_size = 0;
938 for (const int c : candidates_) {
939 if (taken_[c]) continue;
940 if (!temp_[c]) continue;
941 candidates_[new_size++] = c;
942 }
943 candidates_.resize(new_size);
944 for (const int head : graph_[next]) temp_[head] = false;
945 }
946
947 // Output amo.
948 decomposition_.push_back(
949 absl::MakeSpan(buffer_->data() + start, buffer_index - start));
950 }
951 DCHECK_EQ(buffer_index, n);
952 }
953
954 // We follow the heuristic in the paper: "Partitioning Networks into Cliques:
955 // A Randomized Heuristic Approach", David Chaluppa, 2014.
956 //
957 // Note that by keeping the local order of each clique, we cannot make the
958 // order "worse". And each new call to DecomposeGreedily() can only reduce
959 // the number of clique in the cover.
960 void ChangeOrder() {
961 if (absl::Bernoulli(random_, 0.5)) {
962 std::reverse(decomposition_.begin(), decomposition_.end());
963 } else {
964 std::shuffle(decomposition_.begin(), decomposition_.end(), random_);
965 }
966
967 int out_index = 0;
968 for (const absl::Span<const int> clique : decomposition_) {
969 for (const int i : clique) {
970 permutation_[out_index++] = i;
971 }
972 }
973 }
974
975 const std::vector<absl::Span<int>>& decomposition() const {
976 return decomposition_;
977 }
978
979 private:
980 const std::vector<std::vector<int>>& graph_;
981 absl::BitGenRef random_;
982 std::vector<int>* buffer_;
983
984 std::vector<absl::Span<int>> decomposition_;
985 std::vector<int> candidates_;
986 std::vector<int> permutation_;
987 std::vector<bool> taken_;
988 std::vector<bool> temp_;
989};

991} // namespace

993std::vector<absl::Span<int>> AtMostOneDecomposition(
994 const std::vector<std::vector<int>>& graph, absl::BitGenRef random,
995 std::vector<int>* buffer) {
996 CliqueDecomposition decomposer(graph, random, buffer);
997 for (int pass = 0; pass < 10; ++pass) {
998 decomposer.DecomposeGreedily();
999 if (decomposer.decomposition().size() == 1) break;
1000 decomposer.ChangeOrder();
1001 }
1002 return decomposer.decomposition();
1003}
