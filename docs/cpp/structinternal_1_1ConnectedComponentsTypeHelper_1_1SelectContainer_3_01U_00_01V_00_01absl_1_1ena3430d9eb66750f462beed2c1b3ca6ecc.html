template<typename T, typename CompareOrHashT, typename Eq>

struct internal::ConnectedComponentsTypeHelper< T, CompareOrHashT, Eq >::SelectContainer< U, V, absl::enable_if_t< std::is_integral< decltype(std::declval< const U & >()(std::declval< const T & >()))>::value &&std::is_same_v< V, void > > >

Specialization for when U is a hash functor and Eq is void (no custom equality). The expression inside decltype is basically saying that "H(x)" is well-formed, where H is an instance of U and x is an instance of T, and is a value of integral type. That is, we are "duck-typing" on whether U looks like a hash functor.
