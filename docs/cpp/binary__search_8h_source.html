31// EXAMPLE:
32// // Finds the value x in [0,Pi/2] such that cos(x)=2sin(x):
33// BinarySearch<double>(/*x_true=*/0.0, /*x_false=*/M_PI/2,
34// [](double x) { return cos(x) >= 2*sin(x); });
35//
36// Note that x_true > x_false is supported: it works either way.
37//
38// Ideally, f is a monotonic boolean function, such that:
39// - f(x_true) = true
40// - f(x_false) = false
41// - there exists X such that f(x)=true for all x between x_true and X, and
42// f(x)=false for all x between X and x_false.
43//
44// In those conditions, this returns that value X (note that f(X) is true).
45// See below for the NON-MONOTONIC case.
46//
47// Also note that 'Point' may be floating-point types: the function will still
48// converge when the midpoint can't be distinguished from one of the limits,
49// which will always happen.
50// You can use other types than numerical types, too. absl::Duration is
51// naturally supported.
52//
53// OVERFLOWS and NON-NUMERICAL TYPES: If your points may be subject to overflow
54// (e.g. ((kint32max-1) + (kint32max))/2 will overflow an int32_t), or they
55// don't support doing (x+y)/2, you can specialize BinarySearchMidpoint() to
56// suit your needs. See the examples in the unit test.
57//
58// NON-MONOTONIC FUNCTIONS: If f isn't monotonic, the binary search will still
59// run with it typical complexity, and finish. The Point X that it returns
60// will be a "local" inflexion point, meaning that the smallest possible move
61// of that point X to a point X' (in the x_true->x_false direction deduced from
62// the arguments) would make f(X') return false. EXAMPLES:
63// - If Point=int32_t, then the returned X verifies f(X) = true and f(X') =
64// false
65// with X' = X + (x_true < x_false ? 1 : -1).
66// - If Point=double, ditto with X' = X * (1 + (x_true < x_false ? ε : -ε)),
67// where ε = std::numeric_limits<double>::epsilon().
68//
69// Note also that even if f() is non-deterministic, i.e. f(X) can sometimes
70// return true and sometimes false for the same X, then the binary search will
71// still finish, but it's hard to say anything about the returned X.
72template <class Point>
73Point BinarySearch(Point x_true, Point x_false, std::function<bool(Point)> f);

82// Returns the minimum of a convex function on a discrete set of sorted points.
83// It is an error to call this with an empty set of points.
84//
85// We assume the function is "unimodal" with potentially more than one minimum.
86// That is strictly decreasing, then have a minimum that can have many points,
87// and then is strictly increasing. In this case if we have two points with
88// exactly the same value, one of the minimum is always between the two. We will
89// only return one of the minimum.
90//
91// Note that if we allow for non strictly decreasing/increasing, then you have
92// corner cases where one need to check all points to find the minimum. For
93// instance, if the function is constant except at one point where it is lower.
94//
95// The usual algorithm to optimize such a function is a ternary search. However
96// here we assume calls to f() are expensive, and we try to minimize those. So
97// we use a slightly different algorithm than:
98// https://en.wikipedia.org/wiki/Ternary_search
99//
100// TODO(user): Some relevant optimizations:
101// - Abort early if we know a lower bound on the min.
102// - Seed with a starting point if we know one.
103// - We technically do not need the points to be sorted and can use
104// linear-time median computation to speed this up.
105//
106// TODO(user): replace std::function by absl::FunctionRef here and in
107// BinarySearch().
108template <class Point, class Value>
109std::pair<Point, Value> ConvexMinimum(absl::Span<const Point> sorted_points,
110 std::function<Value(Point)> f);

153template <typename T>
154bool AreNumbersOfSameSign(const T& x, const T& y) {
155 if constexpr (std::is_same_v<T, absl::int128>) {
156 return (x < 0) == (y < 0);
157 } else if constexpr (std::is_same_v<T, absl::uint128> ||
158 std::is_unsigned_v<T>) {
159 return true;
160 } else if constexpr (std::is_signed_v<T>) {
161 return std::signbit(x) == std::signbit(y);
162 }
163 return false;
164}

166template <typename IntT>
167constexpr bool UsesTwosComplement() {
168 if constexpr (std::is_integral_v<IntT>) {
169 if constexpr (std::is_unsigned_v<IntT>) {
170 return true;
171 } else {
172 return (IntT{-1} & IntT{3}) == IntT{3};
173 }
174 }
175 return false;
176}

180template <class Point>
181Point BinarySearchMidpoint(Point x, Point y) {
182 Point midpoint;
183 if constexpr (internal::UsesTwosComplement<Point>()) {
184 // For integers using two's complement (all compilers in practice up to
185 // c++17, and all compilers in theory starting from c++20), we can use a
186 // trick from Hacker's delight. See e.g.
187 // https://lemire.me/blog/2022/12/06/fast-midpoint-between-two-integers-without-overflow/
188 midpoint = (x | y) - ((x ^ y) >> 1);
189 } else {
190 midpoint = !internal::AreNumbersOfSameSign(x, y)
191 ? (x + y) / 2
192 :
193 // For integers of the same sign, avoid overflows with a
194 // simple trick.
195 x < y ? x + (y - x) / 2
196 : y + (x - y) / 2;
197 }
198 if (DEBUG_MODE &&
199 !internal::IsNanGeneric(midpoint)) { // Check for overflows.
200 if (x < y) {
201 DCHECK_LE(midpoint, y) << "Overflow?";
202 DCHECK_GE(midpoint, x) << "Overflow?";
203 } else {
204 DCHECK_GE(midpoint, y) << "Overflow?";
205 DCHECK_LE(midpoint, x) << "Overflow?";
206 }
207 }
208 return midpoint;
209}

211template <class Point>
212Point BinarySearch(Point x_true, Point x_false, std::function<bool(Point)> f) {
213 DCHECK(f(x_true)) << x_true;
214 DCHECK(!f(x_false)) << x_false;
215 int num_iterations = 0;
216 constexpr int kMaxNumIterations = 1000000;
217 while (true) {
218 // NOTE(user): If your "Point" type doesn't support the + or the /2
219 // operations, we could imagine using the same trick as IsNanGeneric() to
220 // make BinarySearch() work for your use case.
221 const Point x = BinarySearchMidpoint(x_true, x_false);
222 if (internal::IsNanGeneric(x) || x == x_true || x == x_false) return x_true;
223 if (++num_iterations > kMaxNumIterations) {
224 LOG(DFATAL)
225 << "The binary search seems to loop forever! This indicates that your"
226 " input types don't converge when repeatedly taking the midpoint";
227 return x_true;
228 }
229 if (f(x)) {
230 x_true = x;
231 } else {
232 x_false = x;
233 }
234 }
235}

237template <class Point, class Value>
238std::pair<Point, Value> RangeConvexMinimum(Point begin, Point end,
239 absl::FunctionRef<Value(Point)> f) {
240 DCHECK_LT(begin, end);
241 const Value size = end - begin;
242 if (size == 1) {
243 return {begin, f(begin)};
244 }
245
246 // Starts by splitting interval in two with two queries and getting some info.
247 // Note the current min will be outside the interval.
248 std::pair<Point, Value> current_min;
249 {
250 DCHECK_GE(size, 2);
251 const Point mid = begin + (end - begin) / 2;
252 DCHECK_GT(mid, begin);
253 const Value v = f(mid);
254 const Point before_mid = mid - 1;
255 const Value before_v = f(before_mid);
256 if (before_v == v) return {before_mid, before_v};
257 if (before_v < v) {
258 // Note that we exclude before_mid from the range.
259 current_min = {before_mid, before_v};
260 end = before_mid;
261 } else {
262 current_min = {mid, v};
263 begin = mid + 1;
264 }
265 }
266 if (begin >= end) return current_min;
267 return RangeConvexMinimum<Point, Value>(current_min, begin, end, f);
268}

270template <class Point, class Value>
271std::pair<Point, Value> RangeConvexMinimum(std::pair<Point, Value> current_min,
272 Point begin, Point end,
273 absl::FunctionRef<Value(Point)> f) {
274 DCHECK_LT(begin, end);
275 while ((end - begin) > 1) {
276 DCHECK(current_min.first < begin || current_min.first >= end);
277 bool current_is_after_end = current_min.first >= end;
278 const Point mid = begin + (end - begin) / 2;
279 const Value v = f(mid);
280 if (v >= current_min.second) {
281 // If the midpoint is no better than our current minimum, then the
282 // global min must lie between our midpoint and our current min.
283 if (current_is_after_end) {
284 begin = mid + 1;
285 } else {
286 end = mid;
287 }
288 } else {
289 // v < current_min.second, we cannot decide, so we use a second value
290 // close to v like in the initial step.
291 DCHECK_GT(mid, begin);
292 const Point before_mid = mid - 1;
293 const Value before_v = f(before_mid);
294 if (before_v == v) return {before_mid, before_v};
295 if (before_v < v) {
296 current_min = {before_mid, before_v};
297 current_is_after_end = true;
298 end = before_mid;
299 } else {
300 current_is_after_end = false;
301 current_min = {mid, v};
302 begin = mid + 1;
303 }
304 }
305 }
306
307 if (end - begin == 1) {
308 const Value v = f(begin);
309 if (v <= current_min.second) return {begin, v};
310 }
311 return current_min;
312}

314template <class Point, class Value>
315std::pair<Point, Value> ConvexMinimum(absl::Span<const Point> sorted_points,
316 std::function<Value(Point)> f) {
317 auto index_f = [&](int index) -> Value { return f(sorted_points[index]); };
318 const auto& [index, v] =
319 RangeConvexMinimum<int64_t, Value>(0, sorted_points.size(), index_f);
320 return {sorted_points[index], v};
321}

323template <class Point, class Value>
324std::pair<Point, Value> ConvexMinimum(bool is_to_the_right,
325 std::pair<Point, Value> current_min,
326 absl::Span<const Point> sorted_points,
327 std::function<Value(Point)> f) {
328 auto index_f = [&](int index) -> Value { return f(sorted_points[index]); };
329 std::pair<int, Value> index_current_min = std::make_pair(
330 is_to_the_right ? sorted_points.size() : -1, current_min.second);
331 const auto& [index, v] = RangeConvexMinimum<int64_t, Value>(
332 index_current_min, 0, sorted_points.size(), index_f);
333 if (index == index_current_min.first) return current_min;
334 return {sorted_points[index], v};
335}
