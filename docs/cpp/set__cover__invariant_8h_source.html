30// SetCoverInvariant does the bookkeeping for a solution to the
31// SetCoverModel passed as argument.
32// The state of a SetCoverInvariant instance is uniquely defined by a
33// SubsetBoolVector representing whether a subset is selected in the solution
34// or not.
35//
36// See https://cs.brown.edu/research/pubs/pdfs/1999/Michel-1999-LML.pdf
37// for an explanation of the terminology.
38//
39// A SetCoverInvariant is (relatively) small:
40// is_selected_: a partial solution, vector of Booleans of size #subsets.
41// From this, the following can be computed:
42// coverage_: the number of times an elememt is covered;
43// num_free_elements_: the number of elements of a subset still uncovered;
44// num_coverage_le_1_elements_: the number of elements of a subset that
45// are covered 1 time or less (not overcovered) in the current solution;
46// is_removable_, whether a subset can be removed from the solution.
47// is_removable_[subset] == (num_coverage_le_1_elements_[subet] == 0).
48class SetCoverInvariant {
49 public:
50 // Constructs an empty weighted set covering solver state.
51 // The model may not change after the invariant was built.
52 explicit SetCoverInvariant(SetCoverModel* m) : model_(m) { Initialize(); }
53
54 // Initializes the solver once the data is set. The model cannot be changed
55 // afterwards.
56 void Initialize();
57
58 void Clear() {
59 is_selected_.assign(model_->num_subsets(), false);
60 RecomputeInvariant();
61 }
62
63 // Recomputes all the invariants for the current solution.
64 void RecomputeInvariant();
65
66 // Returns the weighted set covering model to which the state applies.
67 SetCoverModel* model() const { return model_; }
68
69 // Returns the cost of current solution.
70 Cost cost() const { return cost_; }
71
72 // Returns the number of uncovered elements.
73 ElementIndex num_uncovered_elements() const {
74 return num_uncovered_elements_;
75 }
76
77 // Returns the subset assignment vector.
78 const SubsetBoolVector& is_selected() const { return is_selected_; }
79
80 // Returns vector containing the number of elements in each subset that are
81 // not covered in the current solution.
82 const SubsetToElementVector& num_free_elements() const {
83 return num_free_elements_;
84 }
85
86 // Returns the vector of numbers of free or exactly covered elements for
87 // each subset.
88 const SubsetToElementVector& num_coverage_le_1_elements() const {
89 return num_coverage_le_1_elements_;
90 }
91
92 // Returns vector containing number of subsets covering each element.
93 const ElementToSubsetVector& coverage() const { return coverage_; }
94
95 // Returns vector of Booleans telling whether each subset can be removed from
96 // the solution.
97 const SubsetBoolVector& is_removable() const { return is_removable_; }
98
99 // Stores the solution and recomputes the data in the invariant.
100 void LoadSolution(const SubsetBoolVector& c);
101
102 // Returns true if the data stored in the invariant is consistent.
103 bool CheckConsistency() const;
104
105 // Toggles is_selected_[subset] to value, and incrementally updates the
106 // invariant.
107 // Returns a vector of subsets impacted by the change, in case they need
108 // to be reconsidered in a solution geneator or a local search algorithm.
109 // Calls UnsafeToggle, with the added checks:
110 // If value is true, DCHECKs that subset is removable.
111 // If value is true, DCHECKs that marginal impact of subset is removable.
112 std::vector<SubsetIndex> Toggle(SubsetIndex subset, bool value);
113
114 // Same as Toggle, with less DCHECKS.
115 // Useful for some meta-heuristics that allow to go through infeasible
116 // solutions.
117 // Only checks that value is different from is_selected_[subset].
118 std::vector<SubsetIndex> UnsafeToggle(SubsetIndex subset, bool value);
119
120 std::vector<SubsetIndex> UnsafeUse(SubsetIndex subset);
121
122 std::vector<SubsetIndex> UnsafeRemove(SubsetIndex subset);
123
124 // Returns the current solution as a proto.
125 SetCoverSolutionResponse ExportSolutionAsProto() const;
126
127 // Imports the solution from a proto.
128 void ImportSolutionFromProto(const SetCoverSolutionResponse& message);
129
130 private:
131 // Computes the cost and the coverage vector for the given choices.
132 std::tuple<Cost, ElementToSubsetVector> ComputeCostAndCoverage(
133 const SubsetBoolVector& choices) const;
134
135 // Computes the implied data for the given coverage cvrg:
136 // The implied data consists of:
137 std::tuple<ElementIndex, // Number of uncovered elements.
138 SubsetToElementVector, // Number of free elements / subset.
139 SubsetToElementVector, // Number of elem covered 0 or 1 times /
140 // subset.
141 SubsetBoolVector> // Removability for each subset.
142 ComputeImpliedData(const ElementToSubsetVector& cvrg) const;
143
144 // Internal UnsafeToggle where value is a constant for the template.
145 template <bool value>
146 std::vector<SubsetIndex> UnsafeToggleInternal(SubsetIndex subset);
147
148 // The weighted set covering model on which the solver is run.
149 SetCoverModel* model_;
150
151 // Current cost.
152 Cost cost_;
153
154 // The number of uncovered (or free) elements in the current solution.
155 ElementIndex num_uncovered_elements_;
156
157 // Current assignment.
158 SubsetBoolVector is_selected_;
159
160 // A vector that for each subset gives the number of free elements, i.e.
161 // elements whose coverage is 0.
162 // problem.
163 SubsetToElementVector num_free_elements_;
164
165 // Counts the number of free or exactly covered elements, i.e. whose coverage
166 // is 0 or 1.
167 SubsetToElementVector num_coverage_le_1_elements_;
168
169 // The coverage of an element is the number of used subsets which contains
170 // the said element.
171 ElementToSubsetVector coverage_;
172
173 // True if the subset is redundant, i.e. can be removed from the solution
174 // without making it infeasible.
175 SubsetBoolVector is_removable_;
176};
