30// SetCoverInvariant does the bookkeeping for a solution to the
31// SetCoverModel passed as argument.
32// The state of a SetCoverInvariant instance is uniquely defined by a
33// SubsetBoolVector representing whether a subset is selected in the solution
34// or not.
35//
36// See https://cs.brown.edu/research/pubs/pdfs/1999/Michel-1999-LML.pdf
37// for an explanation of the terminology.
38//
39// A SetCoverInvariant is (relatively) small:
40// is_selected_, a partial solution, vector of Booleans of size #subsets.
41// From this, the following can be computed:
42// coverage_, the number of times an elememt is covered;
43// marginal_impacts_, the number of elements of a subset still uncovered;
44// is_removable_, whether a subset can be removed from the solution.
45// Note that is_removable_[subset] implies is_selected_[subset], and thus
46// (is_removable_[subset] <= is_selected_[subset]) == true.
47class SetCoverInvariant {
48 public:
49 // Constructs an empty weighted set covering solver state.
50 // The model may not change after the ledger was built.
51 explicit SetCoverInvariant(SetCoverModel* m) : model_(m) { Initialize(); }
52
53 // Initializes the solver once the data is set. The model cannot be changed
54 // afterwards.
55 void Initialize();
56
57 // Recomputes all the invariants for the current solution.
58 void MakeDataConsistent() {
59 cost_ = ComputeCost(is_selected_);
60 coverage_ = ComputeCoverage(is_selected_);
61 is_removable_ = ComputeIsRemovable(coverage_);
62 marginal_impacts_ = ComputeMarginalImpacts(coverage_);
63 num_elements_covered_ = ComputeNumElementsCovered(coverage_);
64 }
65
66 // Returns the weighted set covering model to which the state applies.
67 SetCoverModel* model() const { return model_; }
68
69 // Returns the cost of current solution.
70 Cost cost() const { return cost_; }
71
72 // Returns the subset assignment vector.
73 const SubsetBoolVector& is_selected() const { return is_selected_; }
74
75 // Returns vector containing the number of elements in each subset that are
76 // not covered in the current solution.
77 const SubsetToElementVector& marginal_impacts() const {
78 return marginal_impacts_;
79 }
80
81 // Returns vector containing number of subsets covering each element.
82 const ElementToSubsetVector& coverage() const { return coverage_; }
83
84 // Returns vector of Booleans telling whether each subset can be removed from
85 // the solution.
86 const SubsetBoolVector& is_removable() const { return is_removable_; }
87
88 // Returns the number of elements covered.
89 ElementIndex num_elements_covered() const { return num_elements_covered_; }
90
91 // Stores the solution and recomputes the data in the ledger.
92 void LoadSolution(const SubsetBoolVector& c);
93
94 // Returns true if the data stored in the ledger is consistent.
95 bool CheckConsistency() const;
96
97 // Computes is_removable_ from scratch for every subset.
98 // TODO(user): reconsider exposing this.
99 void RecomputeIsRemovable() { is_removable_ = ComputeIsRemovable(coverage_); }
100
101 // Returns the subsets that share at least one element with subset.
102 // TODO(user): is it worth to precompute this?
103 std::vector<SubsetIndex> ComputeImpactedSubsets(SubsetIndex subset) const;
104
105 // Toggles is_selected_[subset] to value, and incrementally updates the
106 // ledger.
107 // Returns a vector of subsets impacted by the change, in case they need
108 // to be reconsidered in a solution geneator or a local search algorithm.
109 // Calls UnsafeToggle, with the added checks:
110 // If value is true, DCHECKs that subset is removable.
111 // If value is true, DCHECKs that marginal impact of subset is removable.
112 std::vector<SubsetIndex> Toggle(SubsetIndex subset, bool value);
113
114 // Same as Toggle, with less DCHECKS.
115 // Useful for some meta-heuristics that allow to go through infeasible
116 // solutions.
117 // Only checks that value is different from is_selected_[subset].
118 std::vector<SubsetIndex> UnsafeToggle(SubsetIndex subset, bool value);
119
120 // Update coverage_ for subset when setting is_selected_[subset] to value.
121 void UpdateCoverage(SubsetIndex subset, bool value);
122
123 // Returns true if the elements selected in the current solution cover all
124 // the elements of the set.
125 bool CheckSolution() const;
126
127 // Checks that coverage_ and marginal_impacts_ are consistent with choices.
128 bool CheckCoverageAndMarginalImpacts(const SubsetBoolVector& choices) const;
129
130 // Returns the subsets that are unused that could be used to cover the still
131 // uncovered subsets.
132 std::vector<SubsetIndex> ComputeSettableSubsets() const;
133
134 std::vector<SubsetIndex> ComputeResettableSubsets() const;
135
136 // Returns the current solution as a proto.
137 SetCoverSolutionResponse ExportSolutionAsProto() const;
138
139 // Imports the solution from a proto.
140 void ImportSolutionFromProto(const SetCoverSolutionResponse& message);
141
142 private:
143 // Recomputes the cost from scratch from c.
144 Cost ComputeCost(const SubsetBoolVector& c) const;
145
146 // Computes is_removable based on a coverage cvrg.
147 SubsetBoolVector ComputeIsRemovable(const ElementToSubsetVector& cvrg) const;
148
149 // Computes marginal impacts based on a coverage cvrg.
150 SubsetToElementVector ComputeMarginalImpacts(
151 const ElementToSubsetVector& cvrg) const;
152
153 // Updates marginal_impacts_ for each subset in impacted_subsets.
154 void UpdateMarginalImpacts(absl::Span<const SubsetIndex> impacted_subsets);
155
156 // Computes the number of elements covered based on coverage vector 'cvrg'.
157 ElementIndex ComputeNumElementsCovered(
158 const ElementToSubsetVector& cvrg) const;
159
160 // Returns true if subset can be removed from the solution, i.e. it is
161 // redundant to cover all the elements.
162 // This function is used to check that is_removable[subset] is consistent.
163 bool ComputeIsRemovable(SubsetIndex subset) const;
164
165 // Updates is_removable_ for each subset in impacted_subsets.
166 void UpdateIsRemovable(absl::Span<const SubsetIndex> impacted_subsets);
167
168 // Returns the number of elements currently covered by subset.
169 ElementToSubsetVector ComputeSingleSubsetCoverage(SubsetIndex subset) const;
170
171 // Returns a vector containing the number of subsets covering each element.
172 ElementToSubsetVector ComputeCoverage(const SubsetBoolVector& choices) const;
173
174 // Checks that the value of coverage_ is correct by recomputing and comparing.
175 bool CheckSingleSubsetCoverage(SubsetIndex subset) const;
176
177 // Checks that coverage_ is consistent with choices.
178 bool CheckCoverageAgainstSolution(const SubsetBoolVector& choices) const;
179
180 // Returns true if is_removable_ is consistent.
181 bool CheckIsRemovable() const;
182
183 // The weighted set covering model on which the solver is run.
184 SetCoverModel* model_;
185
186 // Current cost.
187 Cost cost_;
188
189 // The number of elements covered in the current solution.
190 ElementIndex num_elements_covered_;
191
192 // Current assignment.
193 SubsetBoolVector is_selected_;
194
195 // The marginal impact of a subset is the number of elements in that subset
196 // that are not covered in the current solution.
197 SubsetToElementVector marginal_impacts_;
198
199 // The coverage of an element is the number of used subsets which contains
200 // the said element.
201 ElementToSubsetVector coverage_;
202
203 // True if the subset can be removed from the solution without making it
204 // infeasible.
205 SubsetBoolVector is_removable_;
206};
