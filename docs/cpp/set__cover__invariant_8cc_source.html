26// Note: in many of the member functions, variables have "crypterse" names
27// to avoid confusing them with member data. For example mrgnl_impcts is used
28// to avoid confusion with marginal_impacts_.
29void SetCoverInvariant::Initialize() {
30 DCHECK(model_->ComputeFeasibility());
31 model_->CreateSparseRowView();
32
33 const SubsetIndex num_subsets(model_->num_subsets());
34 is_selected_.assign(num_subsets, false);
35 is_removable_.assign(num_subsets, false);
36 marginal_impacts_.assign(num_subsets, ElementIndex(0));
37
38 const SparseColumnView& columns = model_->columns();
39 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
40 marginal_impacts_[subset] = columns[subset].size().value();
41 }
42 const ElementIndex num_elements(model_->num_elements());
43 coverage_.assign(num_elements, SubsetIndex(0));
44 cost_ = 0.0;
45 num_elements_covered_ = ElementIndex(0);
46}

59bool SetCoverInvariant::CheckSolution() const {
60 bool is_ok = true;
61
62 const ElementToSubsetVector cvrg = ComputeCoverage(is_selected_);
63 const ElementIndex num_elements(model_->num_elements());
64 for (ElementIndex element(0); element < num_elements; ++element) {
65 if (cvrg[element] == 0) {
66 LOG(ERROR) << "Recomputed coverage_ for element " << element << " = 0";
67 is_ok = false;
68 }
69 }
70
71 const Cost recomputed_cost = ComputeCost(is_selected_);
72 if (cost_ != recomputed_cost) {
73 LOG(ERROR) << "Cost = " << cost_
74 << ", while recomputed cost_ = " << recomputed_cost;
75 is_ok = false;
76 }
77 return is_ok;
78}

80bool SetCoverInvariant::CheckCoverageAgainstSolution(
81 const SubsetBoolVector& choices) const {
82 const SubsetIndex num_subsets(model_->num_subsets());
83 DCHECK_EQ(num_subsets, choices.size());
84 const ElementToSubsetVector cvrg = ComputeCoverage(choices);
85 bool is_ok = true;
86 const ElementIndex num_elements(model_->num_elements());
87 for (ElementIndex element(0); element < num_elements; ++element) {
88 if (coverage_[element] != cvrg[element]) {
89 LOG(ERROR) << "Recomputed coverage_ for element " << element << " = "
90 << cvrg[element]
91 << ", while updated coverage_ = " << coverage_[element];
92 is_ok = false;
93 }
94 }
95 return is_ok;
96}

98bool SetCoverInvariant::CheckCoverageAndMarginalImpacts(
99 const SubsetBoolVector& choices) const {
100 const SubsetIndex num_subsets(model_->num_subsets());
101 DCHECK_EQ(num_subsets, choices.size());
102 const ElementToSubsetVector cvrg = ComputeCoverage(choices);
103 bool is_ok = CheckCoverageAgainstSolution(choices);
104 const SubsetToElementVector mrgnl_impcts = ComputeMarginalImpacts(cvrg);
105 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
106 if (marginal_impacts_[subset] != mrgnl_impcts[subset]) {
107 LOG(ERROR) << "Recomputed marginal impact for subset " << subset << " = "
108 << mrgnl_impcts[subset] << ", while updated marginal impact = "
109 << marginal_impacts_[subset];
110 is_ok = false;
111 }
112 }
113 return is_ok;
114}

116// Used only once, for testing. TODO(user): Merge with
117// CheckCoverageAndMarginalImpacts.
118SubsetToElementVector SetCoverInvariant::ComputeMarginalImpacts(
119 const ElementToSubsetVector& cvrg) const {
120 const ElementIndex num_elements(model_->num_elements());
121 DCHECK_EQ(num_elements, cvrg.size());
122 const SparseColumnView& columns = model_->columns();
123 const SubsetIndex num_subsets(model_->num_subsets());
124 SubsetToElementVector mrgnl_impcts(num_subsets, ElementIndex(0));
125 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
126 for (ElementIndex element : columns[subset]) {
127 if (cvrg[element] == 0) {
128 ++mrgnl_impcts[subset];
129 }
130 }
131 DCHECK_LE(mrgnl_impcts[subset], columns[subset].size().value());
132 DCHECK_GE(mrgnl_impcts[subset], 0);
133 }
134 return mrgnl_impcts;
135}

137Cost SetCoverInvariant::ComputeCost(const SubsetBoolVector& c) const {
138 DCHECK_EQ(c.size(), model_->num_subsets());
139 Cost recomputed_cost = 0;
140 const SubsetCostVector& subset_costs = model_->subset_costs();
141 for (SubsetIndex subset(0); bool b : c) {
142 if (b) {
143 recomputed_cost += subset_costs[subset];
144 }
145 ++subset;
146 }
147 return recomputed_cost;
148}

162ElementToSubsetVector SetCoverInvariant::ComputeCoverage(
163 const SubsetBoolVector& choices) const {
164 const ElementIndex num_elements(model_->num_elements());
165 const SparseRowView& rows = model_->rows();
166 // Use "crypterse" naming to avoid confusing with coverage_.
167 ElementToSubsetVector cvrg(num_elements, SubsetIndex(0));
168 for (ElementIndex element(0); element < num_elements; ++element) {
169 for (SubsetIndex subset : rows[element]) {
170 if (choices[subset]) {
171 ++cvrg[element];
172 }
173 }
174 DCHECK_LE(cvrg[element], rows[element].size().value());
175 DCHECK_GE(cvrg[element], 0);
176 }
177 return cvrg;
178}

191ElementToSubsetVector SetCoverInvariant::ComputeSingleSubsetCoverage(
192 SubsetIndex subset) const {
193 const SparseColumnView& columns = model_->columns();
194 const ElementIndex num_elements(model_->num_elements());
195 // Use "crypterse" naming to avoid confusing with cvrg.
196 ElementToSubsetVector cvrg(num_elements, SubsetIndex(0));
197 const SparseRowView& rows = model_->rows();
198 for (const ElementIndex element : columns[subset]) {
199 for (SubsetIndex subset : rows[element]) {
200 if (is_selected_[subset]) {
201 ++cvrg[element];

203 }
204 DCHECK_LE(cvrg[element], rows[element].size().value());
205 DCHECK_GE(cvrg[element], 0);
206 }
207 return cvrg;

210std::vector<SubsetIndex> SetCoverInvariant::Toggle(SubsetIndex subset,
211 bool value) {
212 // Note: "if p then q" is also "not(p) or q", or p <= q (p LE q).
213 // If selected, then is_removable, to make sure we still have a solution.
214 DCHECK(is_selected_[subset] <= is_removable_[subset]);
215 // If value, then marginal_impact > 0, to not increase the cost.
216 DCHECK((value <= (marginal_impacts_[subset] > 0)));
217 return UnsafeToggle(subset, value);
218}

220std::vector<SubsetIndex> SetCoverInvariant::UnsafeToggle(SubsetIndex subset,
221 bool value) {
222 // We allow to deselect a non-removable subset, but at least the
223 // Toggle should be a real change.
224 DCHECK_NE(is_selected_[subset], value);
225 // If selected, then marginal_impact == 0.
226 DCHECK(is_selected_[subset] <= (marginal_impacts_[subset] == 0));
227 DVLOG(1) << (value ? "S" : "Des") << "electing subset " << subset;
228 const SubsetCostVector& subset_costs = model_->subset_costs();
229 cost_ += value ? subset_costs[subset] : -subset_costs[subset];
230 is_selected_[subset] = value;
231 UpdateCoverage(subset, value);
232 const std::vector<SubsetIndex> impacted_subsets =
233 ComputeImpactedSubsets(subset);
234 UpdateIsRemovable(impacted_subsets);
235 UpdateMarginalImpacts(impacted_subsets);
236 DCHECK((is_selected_[subset] <= (marginal_impacts_[subset] == 0)));
237 return impacted_subsets;
238}

240void SetCoverInvariant::UpdateCoverage(SubsetIndex subset, bool value) {
241 const SparseColumnView& columns = model_->columns();
242 const SparseRowView& rows = model_->rows();
243 const int delta = value ? 1 : -1;
244 for (const ElementIndex element : columns[subset]) {
245 DVLOG(2) << "Coverage of element " << element << " changed from "
246 << coverage_[element] << " to " << coverage_[element] + delta;
247 coverage_[element] += delta;
248 DCHECK_GE(coverage_[element], 0);
249 DCHECK_LE(coverage_[element], rows[element].size().value());
250 if (coverage_[element] == 1) {
251 ++num_elements_covered_;
252 } else if (coverage_[element] == 0) {
253 --num_elements_covered_;
254 }
255 }
256 DCHECK(CheckSingleSubsetCoverage(subset));
257}

259// Compute the impact of the change in the assignment for each subset
260// containing element. Be careful to add the elements only once.
261std::vector<SubsetIndex> SetCoverInvariant::ComputeImpactedSubsets(
262 SubsetIndex subset) const {
263 const SparseColumnView& columns = model_->columns();
264 const SparseRowView& rows = model_->rows();
265 SubsetBoolVector subset_seen(columns.size(), false);
266 std::vector<SubsetIndex> impacted_subsets;
267 impacted_subsets.reserve(columns.size().value());
268 for (const ElementIndex element : columns[subset]) {
269 for (const SubsetIndex subset : rows[element]) {
270 if (!subset_seen[subset]) {
271 subset_seen[subset] = true;
272 impacted_subsets.push_back(subset);
273 }
274 }
275 }
276 DCHECK_LE(impacted_subsets.size(), model_->num_subsets());
277 // Testing has shown there is no gain in sorting impacted_subsets.
278 return impacted_subsets;
279}

281bool SetCoverInvariant::ComputeIsRemovable(SubsetIndex subset) const {
282 DCHECK(CheckSingleSubsetCoverage(subset));
283 const SparseColumnView& columns = model_->columns();
284 for (const ElementIndex element : columns[subset]) {
285 if (coverage_[element] <= 1) {
286 return false;
287 }
288 }
289 return true;
290}

299SubsetBoolVector SetCoverInvariant::ComputeIsRemovable(
300 const ElementToSubsetVector& cvrg) const {
301 DCHECK(CheckCoverageAgainstSolution(is_selected_));
302 const SubsetIndex num_subsets(model_->num_subsets());
303 SubsetBoolVector is_rmvble(num_subsets, true);
304 const SparseRowView& rows = model_->rows();
305 for (ElementIndex element(0); element < rows.size(); ++element) {
306 if (cvrg[element] <= 1) {
307 for (const SubsetIndex subset : rows[element]) {
308 is_rmvble[subset] = false;
309 }
310 }
311 }
312 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
313 DCHECK_EQ(is_rmvble[subset], ComputeIsRemovable(subset));
314 }
315 return is_rmvble;
316}

320 const SubsetIndex num_subsets(model_->num_subsets());
321 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
322 DCHECK_EQ(is_rmvble[subset], ComputeIsRemovable(subset));

327void SetCoverInvariant::UpdateMarginalImpacts(
328 absl::Span<const SubsetIndex> impacted_subsets) {
329 const SparseColumnView& columns = model_->columns();
330 for (const SubsetIndex subset : impacted_subsets) {
331 ElementIndex impact(0);
332 for (const ElementIndex element : columns[subset]) {
333 if (coverage_[element] == 0) {
334 ++impact;
335 }
336 }
337 DVLOG(2) << "Changing impact of subset " << subset << " from "
338 << marginal_impacts_[subset] << " to " << impact;
339 marginal_impacts_[subset] = impact;
340 DCHECK_LE(marginal_impacts_[subset], columns[subset].size().value());
341 DCHECK_GE(marginal_impacts_[subset], 0);
342 }
343 DCHECK(CheckCoverageAndMarginalImpacts(is_selected_));
344}

346std::vector<SubsetIndex> SetCoverInvariant::ComputeSettableSubsets() const {
347 SubsetBoolVector subset_seen(model_->num_subsets(), false);
348 std::vector<SubsetIndex> focus;
349 focus.reserve(model_->num_subsets().value());
350 const SparseRowView& rows = model_->rows();
351 for (ElementIndex element(0); element < rows.size(); ++element) {
352 if (coverage_[element] >= 1) continue;
353 for (const SubsetIndex subset : rows[element]) {
354 if (!is_selected_[subset]) continue;
355 if (subset_seen[subset]) continue;
356 subset_seen[subset] = true;
357 focus.push_back(subset);
358 }
359 }
360 DCHECK_LE(focus.size(), model_->num_subsets());
361 // Testing has shown there is no gain in sorting focus.
362 return focus;
363}

384SetCoverSolutionResponse SetCoverInvariant::ExportSolutionAsProto() const {
385 SetCoverSolutionResponse message;
386 message.set_num_subsets(is_selected_.size().value());
387 Cost lower_bound = std::numeric_limits<Cost>::max();
388 for (SubsetIndex subset(0); subset < model_->num_subsets(); ++subset) {
389 if (is_selected_[subset]) {
390 message.add_subset(subset.value());
391 }
392 lower_bound = std::min(model_->subset_costs()[subset], lower_bound);
393 }
394 message.set_cost(cost_);
395 message.set_cost_lower_bound(lower_bound);
396 return message;
397}

399void SetCoverInvariant::ImportSolutionFromProto(
400 const SetCoverSolutionResponse& message) {
401 is_selected_.resize(SubsetIndex(message.num_subsets()), false);
402 for (auto s : message.subset()) {
403 is_selected_[SubsetIndex(s)] = true;
404 }
405 MakeDataConsistent();
406 Cost cost = message.cost();
407 CHECK_EQ(cost, cost_);
408}
