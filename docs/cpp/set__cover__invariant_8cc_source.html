26// Note: in many of the member functions, variables have "crypterse" names
27// to avoid confusing them with member data. For example mrgnl_impcts is used
28// to avoid confusion with num_free_elements_.
29void SetCoverInvariant::Initialize() {
30 DCHECK(model_->ComputeFeasibility());
31 model_->CreateSparseRowView();
32
33 cost_ = 0.0;
34
35 const SubsetIndex num_subsets(model_->num_subsets());
36 is_selected_.assign(num_subsets, false);
37 is_removable_.assign(num_subsets, false);
38 num_free_elements_.assign(num_subsets, ElementIndex(0));
39 num_coverage_le_1_elements_.assign(num_subsets, ElementIndex(0));
40
41 const SparseColumnView& columns = model_->columns();
42 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
43 num_free_elements_[subset] = columns[subset].size().value();
44 num_coverage_le_1_elements_[subset] = columns[subset].size().value();
45 }
46
47 const ElementIndex num_elements(model_->num_elements());
48 coverage_.assign(num_elements, SubsetIndex(0));
49
50 num_uncovered_elements_ = num_elements;
51}

53bool SetCoverInvariant::CheckConsistency() const {
54 auto [cst, cvrg] = ComputeCostAndCoverage(is_selected_);
55 CHECK_EQ(cost_, cst);
56 for (ElementIndex element(0); element < model_->num_elements(); ++element) {
57 CHECK_EQ(cvrg[element], coverage_[element]);
58 }
59 auto [num_uncvrd_elts, mrgnl_impcts, num_01_elts, is_rdndnt] =
60 ComputeImpliedData(cvrg);
61 const SparseColumnView& columns = model_->columns();
62 for (SubsetIndex subset(0); subset < columns.size(); ++subset) {
63 CHECK_EQ(mrgnl_impcts[subset], num_free_elements_[subset]);
64 CHECK_EQ(is_rdndnt[subset], is_removable_[subset]);
65 CHECK_EQ(is_rdndnt[subset], num_01_elts[subset] == 0);
66 }
67 return true;
68}

82std::tuple<Cost, ElementToSubsetVector>
83SetCoverInvariant::ComputeCostAndCoverage(
84 const SubsetBoolVector& choices) const {
85 Cost cst = 0.0;
86 ElementToSubsetVector cvrg(model_->num_elements(), SubsetIndex(0));
87 const SparseColumnView& columns = model_->columns();
88 // Initialize coverage, update cost, and compute the coverage for
89 // all the elements covered by the selected subsets.
90 const SubsetCostVector& subset_costs = model_->subset_costs();
91 for (SubsetIndex subset(0); bool b : choices) {
92 if (b) {
93 cst += subset_costs[subset];
94 for (ElementIndex element : columns[subset]) {
95 ++cvrg[element];
96 }
97 }
98 ++subset;
99 }
100 return {cst, cvrg};
101}

103std::tuple<ElementIndex, // Number of uncovered elements
104 SubsetToElementVector, // Number of free elements for each subset
105 SubsetToElementVector, // Number of elem covered 0 or 1 times.
106 SubsetBoolVector> // Removability for each subset
107SetCoverInvariant::ComputeImpliedData(const ElementToSubsetVector& cvrg) const {
108 const ElementIndex num_elements(model_->num_elements());
109 ElementIndex num_uncvrd_elts(num_elements);
110
111 const SubsetIndex num_subsets(model_->num_subsets());
112 SubsetToElementVector num_ovrcvrd_elts(num_subsets, ElementIndex(0));
113 SubsetToElementVector num_free_elts(num_subsets, ElementIndex(0));
114 SubsetToElementVector num_cvrg_le_1_elts(num_subsets, ElementIndex(0));
115 SubsetBoolVector is_rdndnt(num_subsets, false);
116
117 const SparseColumnView& columns = model_->columns();
118
119 // Initialize number of free elements and number of elements covered 0 or 1.
120 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
121 num_free_elts[subset] = columns[subset].size().value();
122 num_cvrg_le_1_elts[subset] = columns[subset].size().value();
123 }
124
125 // Update num_ovrcvrd_elts[subset] and num_elts_cvrd.
126 // When all elements in subset are overcovered, is_removable_[subset] is true.
127 const SparseRowView& rows = model_->rows();
128
129 for (ElementIndex element(0); element < num_elements; ++element) {
130 if (cvrg[element] == 1) {
131 --num_uncvrd_elts; // Just need to update this ...
132 for (SubsetIndex subset : rows[element]) {
133 --num_free_elts[subset];
134 }
135 } else {
136 if (cvrg[element] >= 2) {
137 --num_uncvrd_elts; // Same as when cvrg[element] == 1
138 for (SubsetIndex subset : rows[element]) {
139 --num_free_elts[subset]; // Same as when cvrg[element] == 1
140 --num_cvrg_le_1_elts[subset];
141 if (num_cvrg_le_1_elts[subset] == 0) {
142 is_rdndnt[subset] = true;
143 }
144 }
145 }
146 }
147 }
148 return {num_uncvrd_elts, num_free_elts, num_cvrg_le_1_elts, is_rdndnt};
149}

182std::vector<SubsetIndex> SetCoverInvariant::UnsafeUse(SubsetIndex subset) {
183 DCHECK(!is_selected_[subset]);
184 DVLOG(1) << "Selecting subset " << subset;
185 is_selected_[subset] = true;
186 const SubsetCostVector& subset_costs = model_->subset_costs();
187 cost_ += subset_costs[subset];
188
189 const SparseColumnView& columns = model_->columns();
190
191 SubsetBoolVector subset_seen(columns.size(), false);
192 std::vector<SubsetIndex> impacted_subsets;
193 impacted_subsets.reserve(columns.size().value());
194
195 // Initialize subset_seen so that `subset` not be in impacted_subsets.
196 subset_seen[subset] = true;
197
198 const SparseRowView& rows = model_->rows();
199 for (const ElementIndex element : columns[subset]) {
200 // Update coverage.
201 ++coverage_[element];
202 if (coverage_[element] == 1) {
203 // `element` is newly covered.
204 --num_uncovered_elements_;
205 for (const SubsetIndex impacted_subset : rows[element]) {
206 CollectSubsets(impacted_subset, &impacted_subsets, &subset_seen);
207 --num_free_elements_[impacted_subset];
208 }
209 } else if (coverage_[element] == 2) {
210 // `element` is newly overcovered.
211 for (const SubsetIndex impacted_subset : rows[element]) {
212 CollectSubsets(impacted_subset, &impacted_subsets, &subset_seen);
213 --num_coverage_le_1_elements_[impacted_subset];
214 if (num_coverage_le_1_elements_[impacted_subset] == 0) {
215 // All the elements in impacted_subset are now overcovered, so it
216 // is removable. Note that this happens only when the last element
217 // of impacted_subset becomes overcovered.
218 is_removable_[impacted_subset] = true;
219 }
220 }
221 }
222 }
223 DCHECK(CheckConsistency());
224 return impacted_subsets;
225}

227std::vector<SubsetIndex> SetCoverInvariant::UnsafeRemove(SubsetIndex subset) {
228 DCHECK(is_selected_[subset]);
229 // If already selected, then num_free_elements == 0.
230 DCHECK_EQ(num_free_elements_[subset], 0);
231 DVLOG(1) << "Deselecting subset " << subset;
232 is_selected_[subset] = false;
233 const SubsetCostVector& subset_costs = model_->subset_costs();
234 cost_ -= subset_costs[subset];
235
236 const SparseColumnView& columns = model_->columns();
237
238 SubsetBoolVector subset_seen(columns.size(), false);
239 std::vector<SubsetIndex> impacted_subsets;
240 impacted_subsets.reserve(columns.size().value());
241
242 // Initialize subset_seen so that `subset` not be in impacted_subsets.
243 subset_seen[subset] = true;
244
245 const SparseRowView& rows = model_->rows();
246 for (const ElementIndex element : columns[subset]) {
247 // Update coverage.
248 --coverage_[element];
249 if (coverage_[element] == 0) {
250 // `element` is no longer covered.
251 ++num_uncovered_elements_;
252 for (const SubsetIndex impacted_subset : rows[element]) {
253 CollectSubsets(impacted_subset, &impacted_subsets, &subset_seen);
254 ++num_free_elements_[impacted_subset];
255 }
256 } else if (coverage_[element] == 1) {
257 // `element` is no longer overcovered.
258 for (const SubsetIndex impacted_subset : rows[element]) {
259 CollectSubsets(impacted_subset, &impacted_subsets, &subset_seen);
260 ++num_coverage_le_1_elements_[impacted_subset];
261 if (num_coverage_le_1_elements_[impacted_subset] == 1) {
262 // There is one element of impacted_subset which is not overcovered.
263 // impacted_subset has just become non-removable.
264 is_removable_[impacted_subset] = false;
265 }
266 }
267 }
268 }
269 DCHECK(CheckConsistency());
270 return impacted_subsets;
271}

273SetCoverSolutionResponse SetCoverInvariant::ExportSolutionAsProto() const {
274 SetCoverSolutionResponse message;
275 message.set_num_subsets(is_selected_.size().value());
276 Cost lower_bound = std::numeric_limits<Cost>::max();
277 for (SubsetIndex subset(0); subset < model_->num_subsets(); ++subset) {
278 if (is_selected_[subset]) {
279 message.add_subset(subset.value());
280 }
281 lower_bound = std::min(model_->subset_costs()[subset], lower_bound);
282 }
283 message.set_cost(cost_);
284 message.set_cost_lower_bound(lower_bound);
285 return message;
286}

288void SetCoverInvariant::ImportSolutionFromProto(
289 const SetCoverSolutionResponse& message) {
290 is_selected_.resize(SubsetIndex(message.num_subsets()), false);
291 for (auto s : message.subset()) {
292 is_selected_[SubsetIndex(s)] = true;
293 }
294 RecomputeInvariant();
295 Cost cost = message.cost();
296 CHECK_EQ(cost, cost_);
297}
