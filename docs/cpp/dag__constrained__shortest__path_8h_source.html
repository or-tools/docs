31// This library provides APIs to compute the constrained shortest path (CSP) on
32// a given directed acyclic graph (DAG) with resources on each arc. A CSP is a
33// shortest path on a DAG which does not exceed a set of maximum resources
34// consumption. The algorithm is exponential and has no guarantee to finish.
35//
36// In the DAG, multiple arcs between the same pair of nodes is allowed. However,
37// self-loop arcs are not allowed.
38//
39// Note that we use the length formalism here, but the arc lengths can represent
40// any numeric physical quantity. A shortest path will just be a path minimizing
41// this quantity where the length/resources of a path is the sum of the
42// length/resources of its arcs. An arc length can be negative, or +inf
43// (indicating that it should not be used). An arc length cannot be -inf or nan.
44//
45// Resources on each arc must be non-negative and cannot be +inf or nan.
46
47// -----------------------------------------------------------------------------
48// Basic API.
49// -----------------------------------------------------------------------------
50
51// `tail` and `head` should both be in [0, num_nodes)
52// If the length is +inf, then the arc is not used.
53struct ArcWithLengthAndResources {
54 int from = 0;
55 int to = 0;
56 double length = 0.0;
57 std::vector<double> resources;
58};

61// to the destination. Dies if `arcs_with_length_and_resources` has a cycle.

63 int num_nodes,

71template <class GraphType>
72concept DagGraphType = requires(GraphType graph) {
73 { typename GraphType::NodeIndex{} };
74 { typename GraphType::ArcIndex{} };
75 { graph.num_nodes() } -> std::same_as<typename GraphType::NodeIndex>;
76 { graph.num_arcs() } -> std::same_as<typename GraphType::ArcIndex>;
77 { graph.OutgoingArcs(typename GraphType::NodeIndex{}) };
78 {
79 graph.Tail(typename GraphType::ArcIndex{})
80 } -> std::same_as<typename GraphType::NodeIndex>;
81 {
82 graph.Head(typename GraphType::ArcIndex{})
83 } -> std::same_as<typename GraphType::NodeIndex>;
84};

87// A wrapper that holds the memory needed to run many constrained shortest path
88// computations efficiently on the given DAG (on which resources do not change).
89// `GraphType` can use one of the interfaces defined in `util/graph/graph.h`.
90template <class GraphType>
91#if __cplusplus >= 202002L
92 requires DagGraphType<GraphType>
93#endif
94class ConstrainedShortestPathsOnDagWrapper {
95 public:
96 using NodeIndex = typename GraphType::NodeIndex;
97 using ArcIndex = typename GraphType::ArcIndex;
98
99 // IMPORTANT: All arguments must outlive the class.
100 //
101 // The vectors of `arc_lengths` and `arc_resources[i]` (for all resource i)
102 // *must* be of size `graph.num_arcs()` and indexed the same way as in
103 // `graph`. The vector `arc_resources` and `max_resources` *must* be of same
104 // size.
105 //
106 // You *must* provide a topological order. You can use
107 // `util::graph::FastTopologicalSort(graph)` to compute one if you don't
108 // already have one. An invalid topological order results in an upper bound
109 // for all shortest path computations. For maximum performance, you can
110 // further reindex the nodes under the topological order so that the memory
111 // access pattern is generally forward instead of random. For example, if the
112 // topological order for a graph with 4 nodes is [2,1,0,3], you can re-label
113 // the nodes 2, 1, and 0 to 0, 1, and 2 (and updates arcs accordingly).
114 //
115 // Validity of arcs and topological order are DCHECKed.
116 //
117 // If the number of labels in memory exceeds `max_num_created_labels` at any
118 // point in the algorithm, it returns the best path found so far, most
119 // particularly the empty path if none were found.
120 //
121 // SUBTLE: You can modify the graph, the arc lengths and resources, the
122 // topological order, sources, destinations or the maximum resource between
123 // calls to the `RunConstrainedShortestPathOnDag()` function. That's fine.
124 // Doing so will obviously invalidate the result API of the last constrained
125 // shortest path run, which could return an upper bound, junk, or crash.
126 ConstrainedShortestPathsOnDagWrapper(
127 const GraphType* graph, const std::vector<double>* arc_lengths,
128 const std::vector<std::vector<double>>* arc_resources,
129 const std::vector<NodeIndex>* topological_order,
130 const std::vector<NodeIndex>* sources,
131 const std::vector<NodeIndex>* destinations,
132 const std::vector<double>* max_resources,
133 int max_num_created_labels = 1e9);
134
135 // Returns {+inf, {}, {}} if there is no constrained path of finite length
136 // from one node in `sources` to one node in `destinations`.
137 PathWithLength RunConstrainedShortestPathOnDag();
138
139 private:
140 // Returns the list of all the arcs of the shortest path from one node in
141 // `sources` ending by the arc from a given `label_index` if and only if
142 // `label_index` is between 0 and `labels_from_sources_.size() - 1`.
143 std::vector<ArcIndex> BestArcPathEndingWith(int label_index) const;
144 // Returns the list of all the nodes implied by a given `arc_path`.
145 std::vector<NodeIndex> NodePathImpliedBy(
146 const std::vector<ArcIndex>& arc_path) const;
147
148 const GraphType* const graph_;
149 const std::vector<double>* const arc_lengths_;
150 const std::vector<std::vector<double>>* const arc_resources_;
151 const std::vector<NodeIndex>* const topological_order_;
152 const std::vector<NodeIndex>* const sources_;
153 const std::vector<NodeIndex>* const destinations_;
154 const std::vector<double>* const max_resources_;
155 int max_num_created_labels_;
156
157 std::vector<bool> node_is_source_;
158 std::vector<bool> node_is_destination_;
159 // Using the fact that the graph is a DAG, we can disregard any node that
160 // comes after the last destination (based on the topological order).
161 std::vector<bool> node_is_after_last_destination_;
162
163 // Data for reverse graph.
164 GraphType reverse_graph_;
165 std::vector<ArcIndex> reverse_inverse_arc_permutation_;
166
167 // Data about the last call of the RunConstrainedShortestPathOnDag()
168 // function. A Label includes the cumulative length, resources and the
169 // previous arc used in the path to get to this node.
170 struct Label {
171 double length;
172 // TODO(b/315786885): Optimize resources in Label struct.
173 std::vector<double> resources;
174 ArcIndex incoming_arc;
175 };
176 // A label is present in `labels_from_sources_` if and only if it is feasible
177 // with respect to all resources.
178 std::vector<Label> labels_from_sources_;
179 std::vector<int> node_first_label_;
180 std::vector<int> node_num_labels_;
181};

193ConstrainedShortestPathsOnDagWrapper<GraphType>::
194 ConstrainedShortestPathsOnDagWrapper(
195 const GraphType* graph, const std::vector<double>* arc_lengths,
196 const std::vector<std::vector<double>>* arc_resources,
197 const std::vector<NodeIndex>* topological_order,
198 const std::vector<NodeIndex>* sources,
199 const std::vector<NodeIndex>* destinations,
200 const std::vector<double>* max_resources, int max_num_created_labels)
201 : graph_(graph),
202 arc_lengths_(arc_lengths),
203 arc_resources_(arc_resources),
204 topological_order_(topological_order),
205 sources_(sources),
206 destinations_(destinations),
207 max_resources_(max_resources),
208 max_num_created_labels_(max_num_created_labels) {
209 CHECK(graph_ != nullptr);
210 CHECK(arc_lengths_ != nullptr);
211 CHECK(arc_resources_ != nullptr);
212 CHECK(topological_order_ != nullptr);
213 CHECK(sources_ != nullptr);
214 CHECK(destinations_ != nullptr);
215 CHECK(max_resources_ != nullptr);
216 if (DEBUG_MODE) {
217 CHECK_EQ(arc_lengths_->size(), graph_->num_arcs());
218 CHECK_EQ(arc_resources_->size(), max_resources_->size());
219 for (const std::vector<double>& arcs_resource : *arc_resources_) {
220 CHECK_EQ(arcs_resource.size(), graph_->num_arcs());
221 for (const double arc_resource : arcs_resource) {
222 CHECK(arc_resource >= 0 &&
223 arc_resource != std::numeric_limits<double>::infinity() &&
224 !std::isnan(arc_resource))
225 << absl::StrFormat("resource cannot be negative nor +inf nor NaN");
226 }
227 }
228 for (const double arc_length : *arc_lengths_) {
229 CHECK(arc_length != -std::numeric_limits<double>::infinity() &&
230 !std::isnan(arc_length))
231 << absl::StrFormat("length cannot be -inf nor NaN");
232 }
233 CHECK_OK(TopologicalOrderIsValid(*graph_, *topological_order_))
234 << "Invalid topological order";
235 for (const double max_resource : *max_resources_) {
236 CHECK(max_resource >= 0 &&
237 max_resource != std::numeric_limits<double>::infinity() &&
238 !std::isnan(max_resource))
239 << absl::StrFormat(
240 "max_resource cannot be negative not +inf nor NaN");
241 }
242 }
243
244 node_is_source_.resize(graph_->num_nodes());
245 for (const NodeIndex source : *sources_) {
246 node_is_source_[source] = true;
247 }
248 node_is_destination_.resize(graph_->num_nodes());
249 for (const NodeIndex destination : *destinations_) {
250 node_is_destination_[destination] = true;
251 }
252 node_is_after_last_destination_.resize(graph_->num_nodes());
253 bool last_destination_found = false;
254 for (int i = graph_->num_nodes() - 1; i >= 0; --i) {
255 const NodeIndex node = (*topological_order_)[i];
256 if (node_is_destination_[node]) {
257 last_destination_found = true;
258 }
259 node_is_after_last_destination_[node] = !last_destination_found;
260 }
261
262 // Reverse graph.
263 reverse_graph_ = GraphType(graph_->num_nodes(), graph_->num_arcs());
264 for (ArcIndex arc_index = 0; arc_index < graph_->num_arcs(); ++arc_index) {
265 reverse_graph_.AddArc(graph_->Head(arc_index), graph_->Tail(arc_index));
266 }
267 std::vector<ArcIndex> permutation;
268 reverse_graph_.Build(&permutation);
269 reverse_inverse_arc_permutation_ = GetInversePermutation(permutation);
270
271 // Memory allocation is done here and only once in order to avoid reallocation
272 // at each call of `RunConstrainedShortestPathOnDag()` for better performance.
273 node_first_label_.resize(graph_->num_nodes());
274 node_num_labels_.resize(graph_->num_nodes());
275}

277template <class GraphType>
278#if __cplusplus >= 202002L
279 requires DagGraphType<GraphType>
280#endif
281PathWithLength ConstrainedShortestPathsOnDagWrapper<
282 GraphType>::RunConstrainedShortestPathOnDag() {
283 // Caching the vector addresses allow to not fetch it on each access.
284 const absl::Span<const double> arc_lengths = *arc_lengths_;
285 const absl::Span<const double> max_resources = *max_resources_;
286
287 // Clear all labels from previous run.
288 labels_from_sources_.clear();
289
290 double best_length = std::numeric_limits<double>::infinity();
291 int best_label_index = -1;
292 for (const NodeIndex to : *topological_order_) {
293 node_first_label_[to] = labels_from_sources_.size();
294 int& num_labels = node_num_labels_[to];
295 num_labels = 0;
296 if (node_is_after_last_destination_[to]) {
297 break;
298 }
299 if (node_is_source_[to]) {
300 labels_from_sources_.push_back(
301 {.length = 0,
302 .resources = std::vector<double>(max_resources.size()),
303 .incoming_arc = -1});
304 ++num_labels;
305 if (labels_from_sources_.size() >= max_num_created_labels_) {
306 const std::vector<ArcIndex> arc_path =
307 BestArcPathEndingWith(best_label_index);
308 return {.length = best_length,
309 .arc_path = arc_path,
310 .node_path = NodePathImpliedBy(arc_path)};
311 }
312 }
313 for (const ArcIndex reverse_arc_index : reverse_graph_.OutgoingArcs(to)) {
314 const NodeIndex from = reverse_graph_.Head(reverse_arc_index);
315 const ArcIndex arc_index =
316 reverse_inverse_arc_permutation_.empty()
317 ? reverse_arc_index
318 : reverse_inverse_arc_permutation_[reverse_arc_index];
319 const double arc_length = arc_lengths[arc_index];
320 DCHECK(arc_length != -std::numeric_limits<double>::infinity());
321 if (arc_length == std::numeric_limits<double>::infinity()) {
322 continue;
323 }
324 for (int i = node_first_label_[from];
325 i < node_first_label_[from] + node_num_labels_[from]; ++i) {
326 const Label& label_from = labels_from_sources_[i];
327 bool path_is_feasible = true;
328 for (int r = 0; r < max_resources.size(); ++r) {
329 DCHECK_GE((*arc_resources_)[r][arc_index], 0.0);
330 // TODO(b/314756645): Include lower bound on the resources to
331 // increase pruning.
332 if (label_from.resources[r] + (*arc_resources_)[r][arc_index] >
333 max_resources[r]) {
334 path_is_feasible = false;
335 break;
336 }
337 }
338 if (!path_is_feasible) {
339 continue;
340 }
341 Label label_to = label_from;
342 label_to.length += arc_length;
343 for (int r = 0; r < max_resources.size(); ++r) {
344 label_to.resources[r] += (*arc_resources_)[r][arc_index];
345 }
346 label_to.incoming_arc = arc_index;
347 labels_from_sources_.push_back(label_to);
348 ++num_labels;
349 if (node_is_destination_[to]) {
350 if (best_length > label_to.length) {
351 best_length = label_to.length;
352 best_label_index = labels_from_sources_.size() - 1;
353 }
354 }
355 if (labels_from_sources_.size() >= max_num_created_labels_) {
356 const std::vector<ArcIndex> arc_path =
357 BestArcPathEndingWith(best_label_index);
358 return {.length = best_length,
359 .arc_path = arc_path,
360 .node_path = NodePathImpliedBy(arc_path)};
361 }
362 }
363 }
364 }
365 const std::vector<ArcIndex> arc_path =
366 BestArcPathEndingWith(best_label_index);
367 return {.length = best_length,
368 .arc_path = arc_path,
369 .node_path = NodePathImpliedBy(arc_path)};
370}

372template <typename GraphType>
373#if __cplusplus >= 202002L
374 requires DagGraphType<GraphType>
375#endif
376std::vector<typename GraphType::ArcIndex>
377ConstrainedShortestPathsOnDagWrapper<GraphType>::BestArcPathEndingWith(
378 const int label_index) const {
379 if (label_index < 0 || label_index >= labels_from_sources_.size()) {
380 return {};
381 }
382 int current_label_index = label_index;
383 std::vector<ArcIndex> arc_path;
384 while (labels_from_sources_[current_label_index].incoming_arc != -1) {
385 const Label& label = labels_from_sources_[current_label_index];
386 const NodeIndex node = graph_->Tail(label.incoming_arc);
387 arc_path.push_back(label.incoming_arc);
388 for (int i = node_first_label_[node];
389 i < node_first_label_[node] + node_num_labels_[node]; ++i) {
390 // Since all labels of `labels_from_sources_` are feasible, we can pick
391 // any previous label which satisfies the length equality (irrespective of
392 // the resources it consumes).
393 if (labels_from_sources_[i].length +
394 (*arc_lengths_)[label.incoming_arc] ==
395 label.length) {
396 current_label_index = i;
397 break;
398 }
399 }
400 }
401 absl::c_reverse(arc_path);
402 return arc_path;
403}

409std::vector<typename GraphType::NodeIndex>
410ConstrainedShortestPathsOnDagWrapper<GraphType>::NodePathImpliedBy(
411 const std::vector<ArcIndex>& arc_path) const {
412 if (arc_path.empty()) {
413 return {};
414 }
415 std::vector<NodeIndex> node_path;
416 node_path.reserve(arc_path.size() + 1);
417 for (const ArcIndex arc_index : arc_path) {
418 node_path.push_back(graph_->Tail(arc_index));
419 }
420 node_path.push_back(graph_->Head(arc_path.back()));
421 return node_path;
422}
