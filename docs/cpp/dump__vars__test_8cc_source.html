120TEST(DumpVars, LazyEvaluation) {
121 {
122 int n = 0;
123 auto F = [&]() { return ++n; };
124 auto vars = DUMP_VARS(F());
125 EXPECT_EQ(0, n);
126 EXPECT_EQ("F() = 1", ToString(vars));
127 EXPECT_EQ(1, n);
128 EXPECT_EQ("F() = 2", ToString(vars));
129 EXPECT_EQ(2, n);
130 EXPECT_EQ("F() = 3", vars.str());
131 EXPECT_EQ(3, n);
132 EXPECT_EQ("F() = 4", vars.str());
133 EXPECT_EQ(4, n);
134 EXPECT_EQ("5 = 5", vars.as("5").str());
135 EXPECT_EQ(5, n);
136 }
137 {
138 int n = 0;
139 auto F = [&]() { return ++n; };

147 EXPECT_EQ(3, n);
