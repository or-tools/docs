173// Implements "stamping" as described in "Efficient CNF Simplification based on
174// Binary Implication Graphs", Marijn Heule, Matti Jarvisalo and Armin Biere.
175//
176// This sample the implications graph with a spanning tree, and then simplify
177// all clauses (subsumption / strengthening) using the implications encoded in
178// this tree. So this allows to consider chain of implications instead of just
179// direct ones, but depending on the problem, only a small fraction of the
180// implication graph will be captured by the tree.
181//
182// Note that we randomize the spanning tree at each call. This can benefit by
183// having the implication graph be transitively reduced before.
184class StampingSimplifier {
185 public:
186 explicit StampingSimplifier(Model* model)
187 : assignment_(model->GetOrCreate<Trail>()->Assignment()),
188 implication_graph_(model->GetOrCreate<BinaryImplicationGraph>()),
189 clause_manager_(model->GetOrCreate<LiteralWatchers>()),
190 random_(model->GetOrCreate<ModelRandomGenerator>()),
191 time_limit_(model->GetOrCreate<TimeLimit>()) {}
192
193 // This is "fast" (linear scan + sort of all clauses) so we always complete
194 // the full round.

196 // TODO(user): To save one scan over all the clauses, we could do the fixed
197 // and equivalence variable cleaning here too.
198 bool DoOneRound(bool log_info);
199
200 // When we compute stamps, we might detect fixed variable (via failed literal
201 // probing in the implication graph). So it might make sense to do that until
202 // we have dealt with all fixed literals before calling DoOneRound().
203 bool ComputeStampsForNextRound(bool log_info);
204
205 // Visible for testing.
206 void SampleTreeAndFillParent();
207
208 // Using a DFS visiting order, we can answer reachability query in O(1) on a
209 // tree, this is well known. ComputeStamps() also detect failed literal in
210 // the tree and fix them. It can return false on UNSAT.
211 bool ComputeStamps();
212 bool ImplicationIsInTree(Literal a, Literal b) const {
213 return first_stamps_[a.Index()] < first_stamps_[b.Index()] &&
214 last_stamps_[b.Index()] < last_stamps_[a.Index()];
215 }

231 int64_t num_subsumed_clauses_ = 0;
232 int64_t num_removed_literals_ = 0;
233 int64_t num_fixed_ = 0;

236 absl::StrongVector<LiteralIndex, LiteralIndex> parents_;
237
238 // Adjacency list representation of the parents_ tree.
239 absl::StrongVector<LiteralIndex, int> sizes_;
240 absl::StrongVector<LiteralIndex, int> starts_;

243 // Temporary data for the DFS.

246
247 // First/Last visited index in a DFS of the tree above.
248 absl::StrongVector<LiteralIndex, int> first_stamps_;
249 absl::StrongVector<LiteralIndex, int> last_stamps_;

252// A clause c is "blocked" by a literal l if all clauses containing the
253// negation of l resolve to trivial clause with c. Blocked clause can be
254// simply removed from the problem. At postsolve, if a blocked clause is not
255// satisfied, then l can simply be set to true without breaking any of the
256// clause containing not(l).
257//
258// See the paper "Blocked Clause Elimination", Matti Jarvisalo, Armin Biere,
259// and Marijn Heule.
260//
261// TODO(user): This requires that l only appear in clauses and not in the
262// integer part of CP-SAT.
