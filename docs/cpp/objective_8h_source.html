36constexpr absl::string_view kDeletedObjectiveDefaultDescription =
37 "[objective deleted from model]";

39// A value type that references an objective (either primary or auxiliary) from
40// ModelStorage. Usually this type is passed by copy.
41class Objective {
42 public:
43 // The type used for ids.
44 using IdType = AuxiliaryObjectiveId;
45
46 // Returns an object that refers to the primary objective of the model.
47 inline static Objective Primary(const ModelStorage* storage);
48 // Returns an object that refers to an auxiliary objective of the model.
49 inline static Objective Auxiliary(const ModelStorage* storage,
50 AuxiliaryObjectiveId id);
51
52 // Returns the raw integer ID associated with the objective: nullopt for the
53 // primary objective, a nonnegative int64_t for an auxiliary objective.
54 inline std::optional<int64_t> id() const;
55 // Returns the strong int ID associated with the objective: nullopt for the
56 // primary objective, an AuxiliaryObjectiveId for an auxiliary objective.
57 inline ObjectiveId typed_id() const;
58 // Returns a const-pointer to the underlying storage object for the model.
59 inline const ModelStorage* storage() const;
60
61 // Returns true if the ID corresponds to the primary objective, and false if
62 // it is an auxiliary objective.
63 inline bool is_primary() const;
64
65 // Returns true if the objective is the maximization sense.
66 inline bool maximize() const;
67
68 // Returns the priority (lower is more important) of the objective.
69 inline int64_t priority() const;
70
71 // Returns the name of the objective.
72 inline absl::string_view name() const;
73
74 // Returns the constant offset of the objective.
75 inline double offset() const;
76
77 // Returns the linear coefficient for the variable in the model.
78 inline double coefficient(Variable variable) const;
79 // Returns the quadratic coefficient for the pair of variables in the model.
80 inline double coefficient(Variable first_variable,
81 Variable second_variable) const;
82
83 // Returns true if the variable has a nonzero linear coefficient in the model.
84 inline bool is_coefficient_nonzero(Variable variable) const;
85 // Returns true if the pair of variables has a nonzero quadratic coefficient
86 // in the model.
87 inline bool is_coefficient_nonzero(Variable first_variable,
88 Variable second_variable) const;
89
90 // Returns a representation of the objective as a LinearExpression.
91 // NOTE: This will CHECK fail if the objective has quadratic terms.
92 LinearExpression AsLinearExpression() const;
93 // Returns a representation of the objective as a QuadraticExpression.
94 QuadraticExpression AsQuadraticExpression() const;
95
96 // Returns a detailed string description of the contents of the objective
97 // (not its name, use `<<` for that instead).
98 std::string ToString() const;
99
100 friend inline bool operator==(const Objective& lhs, const Objective& rhs);
101 friend inline bool operator!=(const Objective& lhs, const Objective& rhs);
102 template <typename H>
103 friend H AbslHashValue(H h, const Objective& objective);
104 friend std::ostream& operator<<(std::ostream& ostr,
105 const Objective& objective);
106
107 private:
108 inline Objective(const ModelStorage* storage, ObjectiveId id);
109
110 const ModelStorage* storage_;
111 ObjectiveId id_;
112};

125std::optional<int64_t> Objective::id() const {
126 if (is_primary()) {
127 return std::nullopt;
128 }
129 return id_->value();
130}
131
132ObjectiveId Objective::typed_id() const { return id_; }
133
134const ModelStorage* Objective::storage() const { return storage_; }
135
136bool Objective::is_primary() const { return id_ == kPrimaryObjectiveId; }
137
138int64_t Objective::priority() const {
139 return storage_->objective_priority(id_);
140}
141
142bool Objective::maximize() const { return storage_->is_maximize(id_); }
143
144absl::string_view Objective::name() const {
145 if (is_primary() || storage_->has_auxiliary_objective(*id_)) {
146 return storage_->objective_name(id_);
147 }
148 return kDeletedObjectiveDefaultDescription;
149}
150
151double Objective::offset() const { return storage_->objective_offset(id_); }
152
153double Objective::coefficient(const Variable variable) const {
154 CHECK_EQ(variable.storage(), storage_)
155 << internal::kObjectsFromOtherModelStorage;
156 return storage_->linear_objective_coefficient(id_, variable.typed_id());
157}
158
159double Objective::coefficient(const Variable first_variable,
160 const Variable second_variable) const {
161 CHECK_EQ(first_variable.storage(), storage_)
162 << internal::kObjectsFromOtherModelStorage;
163 CHECK_EQ(second_variable.storage(), storage_)
164 << internal::kObjectsFromOtherModelStorage;
165 return storage_->quadratic_objective_coefficient(
166 id_, first_variable.typed_id(), second_variable.typed_id());
167}
168
169bool Objective::is_coefficient_nonzero(const Variable variable) const {
170 CHECK_EQ(variable.storage(), storage_)
171 << internal::kObjectsFromOtherModelStorage;
172 return storage_->is_linear_objective_coefficient_nonzero(id_,
173 variable.typed_id());
174}
175
176bool Objective::is_coefficient_nonzero(const Variable first_variable,
177 const Variable second_variable) const {
178 CHECK_EQ(first_variable.storage(), storage_)
179 << internal::kObjectsFromOtherModelStorage;
180 CHECK_EQ(second_variable.storage(), storage_)
181 << internal::kObjectsFromOtherModelStorage;
182 return storage_->is_quadratic_objective_coefficient_nonzero(
183 id_, first_variable.typed_id(), second_variable.typed_id());
184}
185
186bool operator==(const Objective& lhs, const Objective& rhs) {
187 return lhs.id_ == rhs.id_ && lhs.storage_ == rhs.storage_;
188}
189
190bool operator!=(const Objective& lhs, const Objective& rhs) {
191 return !(lhs == rhs);
192}
193
194template <typename H>
195H AbslHashValue(H h, const Objective& objective) {
196 return H::combine(std::move(h), objective.id_, objective.storage_);
197}

199Objective::Objective(const ModelStorage* const storage, const ObjectiveId id)
200 : storage_(storage), id_(id) {}

206Objective Objective::Auxiliary(const ModelStorage* const storage,
207 const AuxiliaryObjectiveId id) {
208 return Objective(storage, id);
209}
