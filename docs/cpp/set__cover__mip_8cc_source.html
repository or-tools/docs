34bool SetCoverMip::NextSolution(absl::Span<const SubsetIndex> focus) {
35 SetCoverModel* model = inv_->model();
36 const SubsetIndex num_subsets(model->num_subsets());
37 const ElementIndex num_elements(model->num_elements());
38 SubsetBoolVector choices = inv_->is_selected();
39 MPSolver::OptimizationProblemType problem_type;
40 switch (mip_solver_) {
41 case SetCoverMipSolver::SCIP:
42 problem_type = MPSolver::SCIP_MIXED_INTEGER_PROGRAMMING;
43 break;
44 case SetCoverMipSolver::GUROBI:
45 problem_type = MPSolver::GUROBI_MIXED_INTEGER_PROGRAMMING;
46 break;
47 case SetCoverMipSolver::SAT:
48 problem_type = MPSolver::SAT_INTEGER_PROGRAMMING;
49 break;
50 default:
51 LOG(WARNING) << "Unknown solver value, defaulting to SCIP";
52 problem_type = MPSolver::SCIP_MIXED_INTEGER_PROGRAMMING;
53 }
54 // We are using MPSolver, which is deprecated, because MathOpt does not
55 // provide an interface without using protobufs.
56 // We construct a restricted MIP, omitting all the parts of the problem
57 // that are already fixed in the invariant. The goal is to not spend time
58 // sending data, and having the MIP solver re-discover fixed variables.
59 MPSolver solver("set cover mip", problem_type);
60 solver.SuppressOutput();
61 MPObjective* const objective = solver.MutableObjective();
62 objective->SetMinimization();
63
64 StrictVector<ElementIndex, MPConstraint*> constraints(num_elements, nullptr);
65 StrictVector<SubsetIndex, MPVariable*> vars(num_subsets, nullptr);
66 for (const SubsetIndex subset : focus) {
67 vars[subset] = solver.MakeBoolVar("");
68 objective->SetCoefficient(vars[subset], model->subset_costs()[subset]);
69 for (ElementIndex element : model->columns()[subset]) {
70 if (inv_->coverage()[element] > 0) continue;
71 if (constraints[element] == nullptr) {
72 constexpr double kInfinity = std::numeric_limits<double>::infinity();
73 constraints[element] = solver.MakeRowConstraint(1.0, kInfinity);
74 }
75 constraints[element]->SetCoefficient(vars[subset], 1);
76 }
77 }
78 // set_time_limit takes milliseconds as a unit.
79 solver.set_time_limit(static_cast<int64_t>(time_limit_in_seconds_ * 1000));
80
81 // Call the solver.
82 const MPSolver::ResultStatus solve_status = solver.Solve();
83 switch (solve_status) {
84 case MPSolver::OPTIMAL:
85 break;
86 case MPSolver::FEASIBLE:
87 break;
88 case MPSolver::INFEASIBLE:
89 LOG(ERROR) << "Did not find solution. Problem is infeasible.";
90 break;
91 case MPSolver::UNBOUNDED:
92 LOG(ERROR) << "Did not find solution. Problem is unbounded.";
93 break;
94 default:
95 LOG(ERROR) << "Solving resulted in an error.";
96 return false;
97 }
98 for (const SubsetIndex subset : focus) {
99 choices[subset] = (vars[subset]->solution_value() > 0.9);
100 }
101 inv_->LoadSolution(choices);
102 return true;
103}
