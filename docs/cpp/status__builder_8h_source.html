26class StatusBuilder {

31 explicit StatusBuilder(const absl::Status status)
32 : base_status_(std::move(status)) {}
33
34 operator absl::Status() const { // NOLINT
35 const std::string annotation = ss_.str();
36 if (annotation.empty()) {
37 return base_status_;
38 }
39 if (base_status_.message().empty()) {
40 return absl::Status(base_status_.code(), annotation);
41 }
42 const std::string annotated_message =
43 absl::StrCat(base_status_.message(), "; ", annotation);
44 return absl::Status(base_status_.code(), annotated_message);
45 }

47 template <class T>
48 StatusBuilder& operator<<(const T& t) {
49 ss_ << t;
50 return *this;
51 }

53 StatusBuilder& SetAppend() { return *this; }

55 private:
56 const absl::Status base_status_;
57 std::ostringstream ss_;
58};
