38// A domain represents the possible values of a variable, and its type
39// (which carries display information, i.e. a Boolean will be displayed
40// differently than an integer with domain {0, 1}).
41// It can be:
42// - an explicit list of all possible values, in which case is_interval is
43// false. If the list is empty, then the domain is empty.
44// - an interval, in which case is_interval is true and values.size() == 2,
45// and the interval is [values[0], values[1]].
46// - all integers, in which case values is empty, and is_interval is true.
47// Note that semi-infinite intervals aren't supported.
48// - a Boolean domain({ 0, 1 } with Boolean display tag).
49// TODO(user): Rework domains, all int64_t should be kintmin..kint64max.
50// It is a bit tricky though as we must take care of overflows.
51// If is_a_set is true, then this domain has a set semantics. For a set
52// variable, any subset of the initial set of values is a valid assignment,
53// instead of exactly one value.
54struct Domain {
55 // The values will be sorted and duplicate values will be removed.
56 static Domain IntegerList(std::vector<int64_t> values);
57 static Domain AllInt64();
58 static Domain IntegerValue(int64_t value);
59 static Domain Interval(int64_t included_min, int64_t included_max);
60 static Domain Boolean();
61 static Domain SetOfIntegerList(std::vector<int64_t> values);
62 static Domain SetOfAllInt64();
63 static Domain SetOfIntegerValue(int64_t value);
64 static Domain SetOfInterval(int64_t included_min, int64_t included_max);
65 static Domain SetOfBoolean();
66 static Domain EmptyDomain();
67 static Domain AllFloats();
68 static Domain FloatValue(double value);
69 static Domain FloatInterval(double lb, double ub);
70 // TODO(user): Do we need SetOfFloats() ?
71
72 bool HasOneValue() const;
73 bool empty() const;
74
75 // Returns the min of the domain.
76 int64_t Min() const;
77
78 // Returns the max of the domain.
79 int64_t Max() const;
80
81 // Returns the value of the domain. HasOneValue() must return true.
82 int64_t Value() const;
83
84 // Returns true if the domain is [kint64min..kint64max]
85 bool IsAllInt64() const;
86
87 // Various inclusion tests on a domain.
88 bool Contains(int64_t value) const;
89 bool OverlapsIntList(const std::vector<int64_t>& vec) const;
90 bool OverlapsIntInterval(int64_t lb, int64_t ub) const;
91 bool OverlapsDomain(const Domain& other) const;
92
93 // All the following modifiers change the internal representation
94 // list to interval or interval to list.
95 bool IntersectWithSingleton(int64_t value);
96 bool IntersectWithDomain(const Domain& domain);
97 bool IntersectWithInterval(int64_t interval_min, int64_t interval_max);
98 bool IntersectWithListOfIntegers(absl::Span<const int64_t> integers);
99 bool IntersectWithFloatDomain(const Domain& domain);
100
101 // Returns true iff the value did belong to the domain, and was removed.
102 // Try to remove the value. It returns true if it was actually removed.
103 // If the value is inside a large interval, then it will not be removed.
104 bool RemoveValue(int64_t value);
105 // Sets the empty float domain. Returns true.
106 bool SetEmptyFloatDomain();
107 std::string DebugString() const;
108
109 // These should never be modified from outside the class.
110 std::vector<int64_t> values;
111 bool is_interval = false;
112 bool display_as_boolean = false;
113 // Indicates if the domain was created as a set domain.
114 bool is_a_set = false;
115 // Float domain.
116 bool is_float = false;
117 std::vector<double> float_values;
118};

120// An int var is a name with a domain of possible values, along with
121// some tags. Typically, a Variable is on the heap, and owned by the
122// global Model object.
123struct Variable {
124 // This method tries to unify two variables. This can happen during the
125 // parsing of the model or during presolve. This is possible if at least one
126 // of the two variable is not the target of a constraint. (otherwise it
127 // returns false).
128 // The semantic of the merge is the following:
129 // - the resulting domain is the intersection of the two domains.
130 // - if one variable is not temporary, the result is not temporary.
131 // - if one variable is temporary, the name is the name of the other
132 // variable. If both variables are temporary or both variables are not
133 // temporary, the name is chosen arbitrarily between the two names.
134 bool Merge(absl::string_view other_name, const Domain& other_domain,
135 bool other_temporary);
136
137 std::string DebugString() const;
138
139 std::string name;
140 Domain domain;
141 // Indicates if the variable is a temporary variable created when flattening
142 // the model. For instance, if you write x == y * z + y, then it will be
143 // expanded into y * z == t and x = t + y. And t will be a temporary variable.
144 bool temporary : 1;
145 // Indicates if the variable should be created at all. A temporary variable
146 // can be unreachable in the active model if nobody uses it. In that case,
147 // there is no need to create it.
148 bool active : 1;
149
150 private:
151 friend class Model;
152
153 Variable(absl::string_view name_, const Domain& domain_, bool temporary_);
154};

220// A constraint has a type, some arguments, and a few tags. Typically, a
221// Constraint is on the heap, and owned by the global Model object.
222struct Constraint {
223 Constraint(absl::string_view t, std::vector<Argument> args,
224 bool strong_propag)
225 : type(t),
226 arguments(std::move(args)),
227 strong_propagation(strong_propag),
228 active(true),
229 presolve_propagation_done(false) {}
230
231 std::string DebugString() const;
232
233 // Helpers to be used during presolve.
234 void MarkAsInactive();
235 // Helper method to remove one argument.
236 void RemoveArg(int arg_pos);
237 // Set as a False constraint.
238 void SetAsFalse();
239
240 // The flatzinc type of the constraint (i.e. "int_eq" for integer equality)
241 // stored as a string.
242 std::string type;
243 std::vector<Argument> arguments;
244 // Is true if the constraint should use the strongest level of propagation.
245 // This is a hint in the model. For instance, in the AllDifferent constraint,
246 // there are different algorithms to propagate with different pruning/speed
247 // ratios. When strong_propagation is true, one should use, if possible, the
248 // algorithm with the strongest pruning.
249 bool strong_propagation : 1;
250 // Indicates if the constraint is active. Presolve can make it inactive by
251 // propagating it, or by regrouping it. Once a constraint is inactive, it is
252 // logically removed from the model, it is not extracted, and it is ignored by
253 // presolve.
254 bool active : 1;
255
256 // Indicates if presolve has finished propagating this constraint.
257 bool presolve_propagation_done : 1;
258};
