29// Initial solution hint for warm starting a solver. This can be a full solution
30// (all values specified) or a partial solution (only some values specified). In
31// addition, a full solution does not need to be feasible. The solver may try to
32// complete a partial solution or to repair a full solution that is infeasible.
33//
34// Most MIP solvers can accept a hint, and typically only use variable_values
35// (which can be incomplete). Some LP solvers (e.g. PDLP) can accept a hint as
36// well, and typically need both a complete primal and dual solution. For
37// simplex based solvers, providing an initial basis is generally preferred when
38// both options are supported.
39message SolutionHintProto {
40 // A possibly partial assignment of values to the primal variables of the
41 // problem. The solver-independent requirements for this sub-message are:
42 // * variable_values.ids are elements of VariablesProto.ids.
43 // * variable_values.values must all be finite.
44 SparseDoubleVectorProto variable_values = 1;
45
46 // A (potentially partial) assignment of values to the linear constraints of
47 // the problem.
48 //
49 // Requirements:
50 // * dual_values.ids are elements of LinearConstraintsProto.ids.
51 // * dual_values.values must all be finite.
52 SparseDoubleVectorProto dual_values = 2;
53}

56// Parameters to control a single solve that that are specific to the input
57// model (see SolveParametersProto for model independent parameters).
58message ModelSolveParametersProto {
59 // Filter that is applied to all returned sparse containers keyed by variables
60 // in PrimalSolutionProto and PrimalRayProto
61 // (PrimalSolutionProto.variable_values, PrimalRayProto.variable_values).
62 //
63 // Requirements:
64 // * filtered_ids are elements of VariablesProto.ids.
65 SparseVectorFilterProto variable_values_filter = 1;
66
67 // Filter that is applied to all returned sparse containers keyed by linear
68 // constraints in DualSolutionProto and DualRay
69 // (DualSolutionProto.dual_values, DualRay.dual_values).
70 //
71 // Requirements:
72 // * filtered_ids are elements of LinearConstraints.ids.
73 SparseVectorFilterProto dual_values_filter = 2;
74
75 // Filter that is applied to all returned sparse containers keyed by variables
76 // in DualSolutionProto and DualRay (DualSolutionProto.reduced_costs,
77 // DualRay.reduced_costs).
78 //
79 // Requirements:
80 // * filtered_ids are elements of VariablesProto.ids.
81 SparseVectorFilterProto reduced_costs_filter = 3;
82
83 // Optional initial basis for warm starting simplex LP solvers. If set, it is
84 // expected to be valid according to `ValidateBasis` in
85 // `validators/solution_validator.h` for the current `ModelSummary`.
86 BasisProto initial_basis = 4;
87
88 // TODO(b/183616124): Add multiple solution start support for Gurobi/GSCIP and
89 // add associated tests.
90
91 // Optional solution hints. If set, they are expected to be valid according to
92 // the message description above or equivalently according to
93 // `ValidateSolutionHint` in `validators/model_parameters_validator.cc` for
94 // the current `ModelSummary`.
95 repeated SolutionHintProto solution_hints = 5;
96
97 // Optional branching priorities. Variables with higher values will be
98 // branched on first. Variables for which priorities are not set get the
99 // solver's default priority (usually zero).
100 //
101 // Requirements:
102 // * branching_priorities.values must be finite.
103 // * branching_priorities.ids must be elements of VariablesProto.ids.
104 SparseInt32VectorProto branching_priorities = 6;
105}
