26// Problem feasibility status as claimed by the solver (solver is not required
27// to return a certificate for the claim).

29 // Guard value representing no status.
30 FEASIBILITY_STATUS_UNSPECIFIED = 0;
31
32 // Solver does not claim a status.
33 FEASIBILITY_STATUS_UNDETERMINED = 1;
34
35 // Solver claims the problem is feasible.
36 FEASIBILITY_STATUS_FEASIBLE = 2;
37
38 // Solver claims the problem is infeasible.
39 FEASIBILITY_STATUS_INFEASIBLE = 3;
40}

42// Feasibility status of the primal problem and its dual (or the dual of a
43// continuous relaxation) as claimed by the solver. The solver is not required
44// to return a certificate for the claim (e.g. the solver may claim primal
45// feasibility without returning a primal feasible solutuion). This combined
46// status gives a comprehensive description of a solver's claims about
47// feasibility and unboundedness of the solved problem. For instance,
48//
49// * a feasible status for primal and dual problems indicates the primal is
50// feasible and bounded and likely has an optimal solution (guaranteed for
51// problems without non-linear constraints).
52// * a primal feasible and a dual infeasible status indicates the primal
53// problem is unbounded (i.e. has arbitrarily good solutions).
54//
55// Note that a dual infeasible status by itself (i.e. accompanied by an
56// undetermined primal status) does not imply the primal problem is unbounded as
57// we could have both problems be infeasible. Also, while a primal and dual
58// feasible status may imply the existence of an optimal solution, it does not
59// guarantee the solver has actually found such optimal solution.
60message ProblemStatusProto {
61 // Status for the primal problem.
62 FeasibilityStatusProto primal_status = 1;
63
64 // Status for the dual problem (or for the dual of a continuous relaxation).
65 FeasibilityStatusProto dual_status = 2;
66
67 // If true, the solver claims the primal or dual problem is infeasible, but
68 // it does not know which (or if both are infeasible). Can be true only when
69 // primal_problem_status = dual_problem_status = kUndetermined. This extra
70 // information is often needed when preprocessing determines there is no
71 // optimal solution to the problem (but can't determine if it is due to
72 // infeasibility, unboundedness, or both).
73 bool primal_or_dual_infeasible = 3;
74}

76message SolveStatsProto {
77 // Elapsed wall clock time as measured by math_opt, roughly the time inside
78 // Solver::Solve(). Note: this does not include work done building the model.
79 google.protobuf.Duration solve_time = 1;
80
81 // Solver claims the optimal value is equal or better (smaller for
82 // minimization and larger for maximization) than best_primal_bound:
83 // * best_primal_bound is trivial (+inf for minimization and -inf
84 // maximization) when the solver does not claim to have such bound. This
85 // may happen for some solvers (e.g. PDLP, typically continuous solvers)
86 // even when returning optimal (solver could terminate with slightly
87 // infeasible primal solutions).
88 // * best_primal_bound can be closer to the optimal value than the objective
89 // of the best primal feasible solution. In particular, best_primal_bound
90 // may be non-trivial even when no primal feasible solutions are returned.
91 double best_primal_bound = 2;
92
93 // Solver claims the optimal value is equal or worse (larger for
94 // minimization and smaller for maximization) than best_dual_bound:
95 // * best_dual_bound is always better (smaller for minimization and larger
96 // for maximization) than best_primal_bound.
97 // * best_dual_bound is trivial (-inf for minimization and +inf
98 // maximization) when the solver does not claim to have such bound.
99 // Similarly to best_primal_bound, this may happen for some solvers even
100 // when returning optimal. MIP solvers will typically report a bound even
101 // if it is imprecise.
102 // * for continuous problems best_dual_bound can be closer to the optimal
103 // value than the objective of the best dual feasible solution. For MIP
104 // one of the first non-trivial values for best_dual_bound is often the
105 // optimal value of the LP relaxation of the MIP.
106 double best_dual_bound = 3;
107
108 // Feasibility statuses for primal and dual problems.
109 ProblemStatusProto problem_status = 4;
110
111 int64 simplex_iterations = 5;
112
113 int64 barrier_iterations = 6;
114
115 int64 first_order_iterations = 8;
116
117 int64 node_count = 7;

122// The reason a call to Solve() terminates.
123enum TerminationReasonProto {
124 TERMINATION_REASON_UNSPECIFIED = 0;
125
126 // A provably optimal solution (up to numerical tolerances) has been found.
127 TERMINATION_REASON_OPTIMAL = 1;
128
129 // The primal problem has no feasible solutions.
130 TERMINATION_REASON_INFEASIBLE = 2;
131
132 // The primal problem is feasible and arbitrarily good solutions can be
133 // found along a primal ray.
134 TERMINATION_REASON_UNBOUNDED = 3;
135
136 // The primal problem is either infeasible or unbounded. More details on the
137 // problem status may be available in solve_stats.problem_status. Note that
138 // Gurobi's unbounded status may be mapped here as explained in
139 // go/mathopt-solver-specific#gurobi-inf-or-unb.
140 TERMINATION_REASON_INFEASIBLE_OR_UNBOUNDED = 4;
141
142 // The problem was solved to one of the criteria above (Optimal, Infeasible,
143 // Unbounded, or InfeasibleOrUnbounded), but one or more tolerances was not
144 // met. Some primal/dual solutions/rays be present, but either they will be
145 // slightly infeasible, or (if the problem was nearly optimal) their may be
146 // a gap between the best solution objective and best objective bound.
147 //
148 // Users can still query primal/dual solutions/rays and solution stats, but
149 // they are responsible for dealing with the numerical imprecision.
150 TERMINATION_REASON_IMPRECISE = 5;
151
152 // The optimizer reached some kind of limit and a primal feasible solution
153 // is returned. See SolveResultProto.limit_detail for detailed description of
154 // the kind of limit that was reached.
155 TERMINATION_REASON_FEASIBLE = 9;
156
157 // The optimizer reached some kind of limit and it did not find a primal
158 // feasible solution. See SolveResultProto.limit_detail for detailed
159 // description of the kind of limit that was reached.
160 TERMINATION_REASON_NO_SOLUTION_FOUND = 6;
161
162 // The algorithm stopped because it encountered unrecoverable numerical
163 // error. No solution information is available.
164 TERMINATION_REASON_NUMERICAL_ERROR = 7;
165
166 // The algorithm stopped because of an error not covered by one of the
167 // statuses defined above. No solution information is available.
168 TERMINATION_REASON_OTHER_ERROR = 8;
169}

171// When a Solve() stops early with TerminationReasonProto FEASIBLE or
172// NO_SOLUTION_FOUND, the specific limit that was hit.
173enum LimitProto {
174 // Used as a null value when we terminated not from a limit (e.g.
175 // TERMINATION_REASON_OPTIMAL).
176 LIMIT_UNSPECIFIED = 0;
177
178 // The underlying solver does not expose which limit was reached.
179 LIMIT_UNDETERMINED = 1;
180
181 // An iterative algorithm stopped after conducting the maximum number of
182 // iterations (e.g. simplex or barrier iterations).
183 LIMIT_ITERATION = 2;
184
185 // The algorithm stopped after a user-specified computation time.
186 LIMIT_TIME = 3;
187
188 // A branch-and-bound algorithm stopped because it explored a maximum number
189 // of nodes in the branch-and-bound tree.
190 LIMIT_NODE = 4;
191
192 // The algorithm stopped because it found the required number of solutions.
193 // This is often used in MIPs to get the solver to return the first feasible
194 // solution it encounters.
195 LIMIT_SOLUTION = 5;
196
197 // The algorithm stopped because it ran out of memory.
198 LIMIT_MEMORY = 6;
199
200 // The solver was run with a cutoff (e.g. SolveParameters.cutoff_limit was
201 // set) on the objective, indicating that the user did not want any solution
202 // worse than the cutoff, and the solver concluded there were no solutions at
203 // least as good as the cutoff. Typically no further solution information is
204 // provided.
205 LIMIT_CUTOFF = 12;
206
207 // The algorithm stopped because it either found a solution or a bound better
208 // than a limit set by the user (see SolveParameters.objective_limit and
209 // SolveParameters.best_bound_limit).
210 LIMIT_OBJECTIVE = 7;
211
212 // The algorithm stopped because the norm of an iterate became too large.
213 LIMIT_NORM = 8;
214
215 // The algorithm stopped because of an interrupt signal or a user interrupt
216 // request.
217 LIMIT_INTERRUPTED = 9;
218
219 // The algorithm stopped because it was unable to continue making progress
220 // towards the solution.
221 LIMIT_SLOW_PROGRESS = 10;
222
223 // The algorithm stopped due to a limit not covered by one of the above. Note
224 // that LIMIT_UNDETERMINED is used when the reason cannot be determined, and
225 // LIMIT_OTHER is used when the reason is known but does not fit into any of
226 // the above alternatives.
227 //
228 // TerminationProto.detail may contain additional information about the limit.
229 LIMIT_OTHER = 11;
230}

248// The contract of when primal/dual solutions/rays is complex, see
249// termination_reasons.md for a complete description.
250//
251// Until an exact contract is finalized, it is safest to simply check if a
252// solution/ray is present rather than relying on the termination reason.
253message SolveResultProto {
254 // The reason the solver stopped.
255 TerminationProto termination = 2;
256
257 // Basic solutions use, as of Nov 2021:
258 // * All convex optimization solvers (LP, convex QP) return only one
259 // solution as a primal dual pair.
260 // * Only MI(Q)P solvers return more than one solution. MIP solvers do not
261 // return any dual information, or primal infeasible solutions. Solutions
262 // are returned in order of best primal objective first. Gurobi solves
263 // nonconvex QP (integer or continuous) as MIQP.
264
265 // The general contract for the order of solutions that future solvers should
266 // implement is to order by:
267 // 1. The solutions with a primal feasible solution, ordered by best primal
268 // objective first.
269 // 2. The solutions with a dual feasible solution, ordered by best dual
270 // objective (unknown dual objective is worst)
271 // 3. All remaining solutions can be returned in any order.
272 repeated SolutionProto solutions = 3;
273
274 // Directions of unbounded primal improvement, or equivalently, dual
275 // infeasibility certificates. Typically provided for TerminationReasonProtos
276 // UNBOUNDED and DUAL_INFEASIBLE
277 repeated PrimalRayProto primal_rays = 4;
278
279 // Directions of unbounded dual improvement, or equivalently, primal
280 // infeasibility certificates. Typically provided for TerminationReasonProto
281 // INFEASIBLE.
282 repeated DualRayProto dual_rays = 5;
283
284 // Statistics on the solve process, e.g. running time, iterations.
285 SolveStatsProto solve_stats = 6;
