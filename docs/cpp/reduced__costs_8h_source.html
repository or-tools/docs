36// Maintains the reduced costs of the non-basic variables and some related
37// quantities.
38//
39// Terminology:
40// - To each non-basic column 'a' of A, we can associate an "edge" in the
41// kernel of A equal to 1.0 on the index of 'a' and '-B^{-1}.a' on the basic
42// variables.
43// - 'B^{-1}.a' is called the "right inverse" of 'a'.
44// - The reduced cost of a column is equal to the scalar product of this
45// column's edge with the cost vector (objective_), and corresponds to the
46// variation in the objective function when we add this edge to the current
47// solution.
48// - The dual values are the "left inverse" of the basic objective by B.
49// That is 'basic_objective_.B^{-1}'
50// - The reduced cost of a column is also equal to the scalar product of this
51// column with the vector of the dual values.
52class ReducedCosts {
53 public:
54 // Takes references to the linear program data we need.
55 ReducedCosts(const CompactSparseMatrix& matrix_, const DenseRow& objective,
56 const RowToColMapping& basis,
57 const VariablesInfo& variables_info,
58 const BasisFactorization& basis_factorization,
59 absl::BitGenRef random);

66 // next call to GetReducedCosts().
67 bool NeedsBasisRefactorization() const;
68
69 // Checks the precision of the entering variable choice now that the direction
70 // is computed. Returns its precise version. This will also trigger a
71 // reduced cost recomputation if it was deemed too imprecise.
72 Fractional TestEnteringReducedCostPrecision(ColIndex entering_col,
73 const ScatteredColumn& direction);
74
75 // Computes the current dual residual and infeasibility. Note that these
76 // functions are not really fast (many scalar products will be computed) and
77 // shouldn't be called at each iteration.
78 //
79 // These function will compute the reduced costs if needed.
80 // ComputeMaximumDualResidual() also needs ComputeBasicObjectiveLeftInverse()
81 // and do not depends on reduced costs.
82 Fractional ComputeMaximumDualResidual();
83 Fractional ComputeMaximumDualInfeasibility();
84 Fractional ComputeSumOfDualInfeasibilities();
85
86 // Same as ComputeMaximumDualInfeasibility() but ignore boxed variables.
87 // Because we can always switch bounds of boxed variables, if this is under
88 // the dual tolerance, then we can easily have a dual feasible solution and do
89 // not need to run a dual phase I algorithm.
90 Fractional ComputeMaximumDualInfeasibilityOnNonBoxedVariables();
91
92 // Updates any internal data BEFORE the given simplex pivot is applied to B.
93 // Note that no updates are needed in case of a bound flip.
94 // The arguments are in order:
95 // - The index of the entering non-basic column of A.
96 // - The index in B of the leaving basic variable.
97 // - The 'direction', i.e. the right inverse of the entering column.
98 void UpdateBeforeBasisPivot(ColIndex entering_col, RowIndex leaving_row,
99 const ScatteredColumn& direction,
100 UpdateRow* update_row);
101
102 // Sets the cost of the given non-basic variable to zero and updates its
103 // reduced cost. Note that changing the cost of a non-basic variable only
104 // impacts its reduced cost and not the one of any other variables.
105 // The current_cost pointer must be equal to the address of objective[col]
106 // where objective is the DenseRow passed at construction.
107 void SetNonBasicVariableCostToZero(ColIndex col, Fractional* current_cost);
108
109 // Sets the pricing parameters. This does not change the pricing rule.
110 void SetParameters(const GlopParameters& parameters);
111
112 // Returns true if the current reduced costs are computed with maximum
113 // precision.
114 bool AreReducedCostsPrecise() { return are_reduced_costs_precise_; }
115
116 // Returns true if the current reduced costs where just recomputed or will be
117 // on the next call to GetReducedCosts().
118 bool AreReducedCostsRecomputed() {
119 return recompute_reduced_costs_ || are_reduced_costs_recomputed_;
120 }
121
122 // Makes sure the next time the reduced cost are needed, they will be
123 // recomputed with maximum precision (i.e. from scratch with a basis
124 // refactorization first).
125 void MakeReducedCostsPrecise();
126
127 // Randomly perturb the costs. Both Koberstein and Huangfu recommend doing
128 // that before the dual simplex starts in their Phd thesis.
129 //
130 // The perturbation follows what is explained in Huangfu Q (2013) "High
131 // performance simplex solver", Ph.D, dissertation, University of Edinburgh,
132 // section 3.2.3, page 58.
133 void PerturbCosts();
134
135 // Shifts the cost of the given non-basic column such that its current reduced
136 // cost becomes 0.0. Actually, this shifts the cost a bit more according to
137 // the positive_direction parameter.
138 //
139 // This is explained in Koberstein's thesis (section 6.2.2.3) and helps on
140 // degenerate problems. As of july 2013, this allowed to pass dano3mip and
141 // dbic1 without cycling forever. Note that contrary to what is explained in
142 // the thesis, we do not shift any other variable costs. If any becomes
143 // infeasible, it will be selected and shifted in subsequent iterations.
144 void ShiftCostIfNeeded(bool increasing_rc_is_needed, ColIndex col);
145
146 // Returns true if ShiftCostIfNeeded() was applied since the last
147 // ClearAndRemoveCostShifts().
148 bool HasCostShift() const { return has_cost_shift_; }
149
150 // Returns true if this step direction make the given column even more
151 // infeasible. This is just used for reporting stats.
152 bool StepIsDualDegenerate(bool increasing_rc_is_needed, ColIndex col);
153
154 // Removes any cost shift and cost perturbation. This also lazily forces a
155 // recomputation of all the derived quantities. This effectively resets the
156 // class to its initial state.
157 void ClearAndRemoveCostShifts();
158
159 // Invalidates all internal structure that depends on the objective function.
160 void ResetForNewObjective();
161
162 // Invalidates the data that depends on the order of the column in basis_.
163 void UpdateDataOnBasisPermutation();
164
165 // Returns the current reduced costs. If AreReducedCostsPrecise() is true,
166 // then for basic columns, this gives the error between 'c_B' and 'y.B' and
167 // for non-basic columns, this is the classic reduced cost. If it is false,
168 // then this is defined only for the columns in
169 // variables_info_.GetIsRelevantBitRow().
170 const DenseRow& GetReducedCosts();
171
172 // Same as GetReducedCosts() but trigger a recomputation if not already done
173 // to have access to the reduced costs on all positions, not just the relevant
174 // one.
175 const DenseRow& GetFullReducedCosts();
176
177 // Returns the dual values associated to the current basis.
178 const DenseColumn& GetDualValues();
179
180 // Stats related functions.
181 std::string StatString() const { return stats_.StatString(); }
182
183 // Returns the current dual feasibility tolerance.
184 Fractional GetDualFeasibilityTolerance() const {
185 return dual_feasibility_tolerance_;
186 }
187
188 // Does basic checking of an entering candidate.
189 bool IsValidPrimalEnteringCandidate(ColIndex col) const;
190
191 // Visible for testing.
192 const DenseRow& GetCostPerturbations() const { return cost_perturbations_; }
193
194 // The deterministic time used by this class.
195 double DeterministicTime() const { return deterministic_time_; }
196
197 // Registers a boolean that will be set to true each time the reduced costs
198 // are or will be recomputed. This allows anyone that depends on this to know
199 // that it cannot just assume an incremental changes and needs to updates its
200 // data. Important: UpdateBeforeBasisPivot() will not trigger this.
201 void AddRecomputationWatcher(bool* watcher) { watchers_.push_back(watcher); }
202
203 private:
204 // Statistics about this class.
205 struct Stats : public StatsGroup {
206 Stats()
207 : StatsGroup("ReducedCosts"),
208 basic_objective_left_inverse_density(
209 "basic_objective_left_inverse_density", this),
210 reduced_costs_accuracy("reduced_costs_accuracy", this),

217 // All these Compute() functions fill the corresponding DenseRow using
218 // the current problem data.
219 void ComputeBasicObjective();
220 void ComputeReducedCosts();
221 void ComputeBasicObjectiveLeftInverse();
222
223 // Updates reduced_costs_ according to the given pivot. This adds a multiple
224 // of the vector equal to 1.0 on the leaving column and given by
225 // ComputeUpdateRow() on the non-basic columns. The multiple is such that the
226 // new leaving reduced cost is zero.
227 void UpdateReducedCosts(ColIndex entering_col, ColIndex leaving_col,
228 RowIndex leaving_row, Fractional pivot,
229 UpdateRow* update_row);
230
231 // Updates basic_objective_ according to the given pivot.
232 void UpdateBasicObjective(ColIndex entering_col, RowIndex leaving_row);
233
234 // All places that do 'recompute_reduced_costs_ = true' must go through here.
235 void SetRecomputeReducedCostsAndNotifyWatchers();
236
237 // Problem data that should be updated from outside.
238 const CompactSparseMatrix& matrix_;
239 const DenseRow& objective_;
240 const RowToColMapping& basis_;
241 const VariablesInfo& variables_info_;
242 const BasisFactorization& basis_factorization_;
243 absl::BitGenRef random_;
244
245 // Internal data.
246 GlopParameters parameters_;
247 mutable Stats stats_;
248
249 // Booleans to control what happens on the next ChooseEnteringColumn() call.
250 bool must_refactorize_basis_;
251 bool recompute_basic_objective_left_inverse_;
252 bool recompute_basic_objective_;
253 bool recompute_reduced_costs_;
254
255 // Indicates if we have computed the reduced costs with a good precision.
256 bool are_reduced_costs_precise_;
257 bool are_reduced_costs_recomputed_;
258
259 bool has_cost_shift_ = false;
260
261 // Values of the objective on the columns of the basis. The order is given by
262 // the basis_ mapping. It is usually denoted as 'c_B' in the literature .
263 DenseRow basic_objective_;
264
265 // Perturbations to the objective function. This may be introduced to
266 // counter degenerecency. It will be removed at the end of the algorithm.
267 DenseRow cost_perturbations_;
268
269 // Reduced costs of the relevant columns of A.
270 DenseRow reduced_costs_;
271
272 // Left inverse by B of the basic_objective_. This is known as 'y' or 'pi' in
273 // the literature. Its scalar product with a column 'a' of A gives the value
274 // of the scalar product of the basic objective with the right inverse of 'a'.
275 //
276 // TODO(user): using the unit_row_left_inverse_, we can update the
277 // basic_objective_left_inverse_ at each iteration, this is not needed for the
278 // algorithm, but may gives us a good idea of the current precision of our
279 // estimates. It is also faster to compute the unit_row_left_inverse_ because
280 // of sparsity.
281 ScatteredRow basic_objective_left_inverse_;
282
283 // This is usually parameters_.dual_feasibility_tolerance() except when the
284 // dual residual error |y.B - c_B| is higher than it and we have to increase
285 // the tolerance.
286 Fractional dual_feasibility_tolerance_;
287
288 // Boolean(s) to set to false when the reduced cost are changed outside of the
289 // UpdateBeforeBasisPivot() function.
290 std::vector<bool*> watchers_;
291
292 double deterministic_time_ = 0.0;
293};
