34// This library provides a few APIs to compute the shortest path on a given
35// directed acyclic graph (DAG).
36//
37// In the DAG, multiple arcs between the same pair of nodes is allowed. However,
38// self-loop arcs are not allowed.
39//
40// Note that we use the length formalism here, but the arc lengths can represent
41// any numeric physical quantity. A shortest path will just be a path minimizing
42// this quantity where the length of a path is the sum of the length of its
43// arcs. An arc length can be negative, or +inf (indicating that it should not
44// be used). An arc length cannot be -inf or nan.
45
46// -----------------------------------------------------------------------------
47// Basic API.
48// -----------------------------------------------------------------------------
49
50// `tail` and `head` should both be in [0, num_nodes)
51// If the length is +inf, then the arc should not be used.
52struct ArcWithLength {
53 int tail = 0;
54 int head = 0;
55 double length = 0.0;
56};
57
58struct PathWithLength {
59 double length = 0.0;
60 // The returned arc indices points into the `arcs_with_length` passed to the
61 // function below.
62 std::vector<int> arc_path;
63 std::vector<int> node_path; // includes the source node.
64};
65
66// Returns {+inf, {}} if there is no path of finite length from the source to
67// the destination. Dies if `arcs_with_length` has a cycle.
68PathWithLength ShortestPathsOnDag(
69 int num_nodes, absl::Span<const ArcWithLength> arcs_with_length, int source,
70 int destination);

72// -----------------------------------------------------------------------------
73// Advanced API.
74// -----------------------------------------------------------------------------
75
76// A wrapper that holds the memory needed to run many shortest path computations
77// efficiently on the given DAG. `GraphType` *must* implement the following
78// types and functions: `GraphType::NodeIndex`, `GraphType::ArcIndex`,
79// `num_nodes()`, `num_arcs()`, `OutgoingArcs(tail)`, `Tail(arc)` and
80// `Head(arc)`. You can use on of the interfaces defined in
81// `util/graph/graph.h`.
82template <class GraphType>
83class ShortestPathsOnDagWrapper {
84 public:
85 using NodeIndex = typename GraphType::NodeIndex;
86 using ArcIndex = typename GraphType::ArcIndex;
87
88 // IMPORTANT: All arguments must outlive the class.
89 //
90 // The vector of `arc_lengths` *must* be of size `graph.num_arcs()` and
91 // indexed the same way as in `graph`.
92 //
93 // You *must* provide a topological order. You can use
94 // `util::graph::FastTopologicalSort(graph)` to compute one if you don't
95 // already have one. An invalid topological order results in an upper bound
96 // for all shortest path computations. For maximum performance, you can
97 // further reindex the nodes under the topological order so that the memory
98 // access pattern is generally forward instead of random. For example, if the
99 // topological order for a graph with 4 nodes is [2,1,0,3], you can re-label
100 // the nodes 2, 1, and 0 to 0, 1, and 2 (and updates arcs accordingly).
101 //
102 // Validity of arcs and topological order are CHECKed if compiled in DEBUG
103 // mode.
104 //
105 // SUBTLE: You can modify the graph, the arc lengths or the topological order
106 // between calls to the `RunShortestPathOnDag()` function. That's fine. Doing
107 // so will obviously invalidate the result API of the last shortest path run,
108 // which could return an upper bound, junk, or crash.
109 ShortestPathsOnDagWrapper(const GraphType* graph,
110 const std::vector<double>* arc_lengths,
111 absl::Span<const NodeIndex> topological_order);
112
113 // Computes the shortest path to all reachable nodes from the given sources.
114 // This must be called before any of the query functions below.
115 void RunShortestPathOnDag(absl::Span<const NodeIndex> sources);
116
117 // Returns true if `node` is reachable from at least one source, i.e., the
118 // length from at least one source is finite.
119 bool IsReachable(NodeIndex node) const {
120 return length_from_sources_[node] < std::numeric_limits<double>::infinity();
121 }
122 const std::vector<NodeIndex>& reached_nodes() const { return reached_nodes_; }
123
124 // Returns the length of the shortest path from `node`'s source to `node`.
125 double LengthTo(NodeIndex node) const { return length_from_sources_[node]; }
126 std::vector<double> LengthTo() const { return length_from_sources_; }
127
128 // Returns the list of all the arcs in the shortest path from `node`'s
129 // source to `node`. CHECKs if the node is reachable.
130 std::vector<ArcIndex> ArcPathTo(NodeIndex node) const;
131
132 // Returns the list of all the nodes in the shortest path from `node`'s
133 // source to `node` (including the source). CHECKs if the node is reachable.
134 std::vector<NodeIndex> NodePathTo(NodeIndex node) const;
135
136 // Accessors to the underlying graph and arc lengths.
137 const GraphType& graph() const { return *graph_; }
138 const std::vector<double>& arc_lengths() const { return *arc_lengths_; }
139
140 private:
141 const GraphType* const graph_;
142 const std::vector<double>* const arc_lengths_;
143 absl::Span<const NodeIndex> const topological_order_;
144
145 // Data about the last call of the RunShortestPathOnDag() function.
146 std::vector<double> length_from_sources_;
147 std::vector<ArcIndex> incoming_shortest_path_arc_;
148 std::vector<NodeIndex> reached_nodes_;
149};

160template <class GraphType>
161ShortestPathsOnDagWrapper<GraphType>::ShortestPathsOnDagWrapper(
162 const GraphType* graph, const std::vector<double>* arc_lengths,
163 absl::Span<const NodeIndex> topological_order)
164 : graph_(graph),
165 arc_lengths_(arc_lengths),
166 topological_order_(topological_order) {
167 CHECK(graph_ != nullptr);
168 CHECK(arc_lengths_ != nullptr);
169 if (DEBUG_MODE) {
170 CHECK_EQ(arc_lengths_->size(), graph_->num_arcs());
171 for (const double arc_length : *arc_lengths_) {
172 CHECK(arc_length != -std::numeric_limits<double>::infinity() &&
173 !std::isnan(arc_length))
174 << absl::StrFormat("length cannot be -inf nor NaN");
175 }
176 CHECK_OK(TopologicalOrderIsValid(*graph_, topological_order_))
177 << "Invalid topological order";
178 }
179
180 // Memory allocation is done here and only once in order to avoid reallocation
181 // at each call of `RunShortestPathOnDag()` for better performance.
182 length_from_sources_.resize(graph_->num_nodes(),
183 std::numeric_limits<double>::infinity());
184 incoming_shortest_path_arc_.resize(graph_->num_nodes(), -1);
185 reached_nodes_.reserve(graph_->num_nodes());
186}
187
188template <class GraphType>
189void ShortestPathsOnDagWrapper<GraphType>::RunShortestPathOnDag(
190 absl::Span<const NodeIndex> sources) {
191 // Caching the vector addresses allow to not fetch it on each access.
192 const absl::Span<double> length_from_sources =
193 absl::MakeSpan(length_from_sources_);
194 const absl::Span<const double> arc_lengths = *arc_lengths_;
195
196 // Note that we do not re-assign `incoming_shortest_path_arc_` at every call
197 // for better performance, so it only makes sense for nodes that are reachable
198 // from at least one source, the other ones will contain junk.
199 for (const NodeIndex node : reached_nodes_) {
200 length_from_sources[node] = std::numeric_limits<double>::infinity();
201 }
202 DCHECK(std::all_of(
203 length_from_sources.begin(), length_from_sources.end(),
204 [](double l) { return l == std::numeric_limits<double>::infinity(); }));
205 reached_nodes_.clear();
206
207 for (const NodeIndex source : sources) {
208 length_from_sources[source] = 0.0;
209 }
210
211 for (const NodeIndex tail : topological_order_) {
212 const double length_to_tail = length_from_sources[tail];
213 // Stop exploring a node as soon as its length to all sources is +inf.
214 if (length_to_tail == std::numeric_limits<double>::infinity()) {
215 continue;
216 }
217 reached_nodes_.push_back(tail);
218 for (const ArcIndex arc : graph_->OutgoingArcs(tail)) {
219 const NodeIndex head = graph_->Head(arc);
220 DCHECK(arc_lengths[arc] != -std::numeric_limits<double>::infinity());
221 const double length_to_head = arc_lengths[arc] + length_to_tail;
222 if (length_to_head < length_from_sources[head]) {
223 length_from_sources[head] = length_to_head;
224 incoming_shortest_path_arc_[head] = arc;
225 }
226 }
227 }
228}

230// TODO(user): If `ArcPathTo` and/or `NodePathTo` functions become
231// bottlenecks:
232// (1) have the class preallocate a buffer of size `num_nodes`
233// (2) assign into an index rather than with push_back
234// (3) return by absl::Span (or return a copy) with known size.
235template <typename GraphType>
236std::vector<typename GraphType::ArcIndex>
237ShortestPathsOnDagWrapper<GraphType>::ArcPathTo(NodeIndex node) const {
238 CHECK(IsReachable(node));
239 std::vector<ArcIndex> arc_path;
240 ArcIndex current_arc = incoming_shortest_path_arc_[node];
241 while (current_arc != -1) {
242 arc_path.push_back(current_arc);
243 current_arc = incoming_shortest_path_arc_[graph_->Tail(current_arc)];
244 }
245 absl::c_reverse(arc_path);
246 return arc_path;
247}

249template <typename GraphType>
250std::vector<typename GraphType::NodeIndex>
251ShortestPathsOnDagWrapper<GraphType>::NodePathTo(NodeIndex node) const {
252 CHECK(IsReachable(node));
253 std::vector<NodeIndex> node_path;
254 NodeIndex current_node = node;
255 node_path.push_back(current_node);
256 ArcIndex current_arc = incoming_shortest_path_arc_[current_node];
257 while (current_arc != -1) {
258 current_node = graph_->Tail(current_arc);
259 node_path.push_back(current_node);
260 current_arc = incoming_shortest_path_arc_[current_node];
261 }
262 absl::c_reverse(node_path);
263 return node_path;
264}

266template <typename GraphType>
267absl::Status TopologicalOrderIsValid(
268 const GraphType& graph,
269 absl::Span<const typename GraphType::NodeIndex> topological_order) {
270 using NodeIndex = typename GraphType::NodeIndex;
271 using ArcIndex = typename GraphType::ArcIndex;
272 const NodeIndex num_nodes = graph.num_nodes();
273 if (topological_order.size() != num_nodes) {
274 return absl::InvalidArgumentError(absl::StrFormat(
275 "topological_order.size() = %i, != graph.num_nodes() = %i",
276 topological_order.size(), num_nodes));
277 }
278 std::vector<NodeIndex> inverse_topology(num_nodes, -1);
279 for (NodeIndex node = 0; node < topological_order.size(); ++node) {
280 if (inverse_topology[topological_order[node]] >= 0) {
281 return absl::InvalidArgumentError(
282 absl::StrFormat("node % i appears twice in topological order",
283 topological_order[node]));
284 }
285 inverse_topology[topological_order[node]] = node;
286 }
287 for (NodeIndex tail = 0; tail < num_nodes; ++tail) {
288 for (const ArcIndex arc : graph.OutgoingArcs(tail)) {
289 const NodeIndex head = graph.Head(arc);
290 if (inverse_topology[tail] >= inverse_topology[head]) {
291 return absl::InvalidArgumentError(absl::StrFormat(
292 "arc (%i, %i) is inconsistent with topological order", tail, head));
293 }
294 }
295 }
296 return absl::OkStatus();
297}
