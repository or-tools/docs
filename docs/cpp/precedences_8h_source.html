38// This class implement a propagator on simple inequalities between integer
39// variables of the form (i1 + offset <= i2). The offset can be constant or
40// given by the value of a third integer variable. Offsets can also be negative.
41//
42// The algorithm works by mapping the problem onto a graph where the edges carry
43// the offset and the nodes correspond to one of the two bounds of an integer
44// variable (lower_bound or -upper_bound). It then find the fixed point using an
45// incremental variant of the Bellman-Ford(-Tarjan) algorithm.
46//
47// This is also known as an "integer difference logic theory" in the SMT world.
48// Another word is "separation logic".
49//
50// TODO(user): We could easily generalize the code to support any relation of
51// the form a*X + b*Y + c*Z >= rhs (or <=). Do that since this class should be
52// a lot faster at propagating small linear inequality than the generic
53// propagator and the overhead of supporting coefficient should not be too bad.
54class PrecedencesPropagator : public SatPropagator, PropagatorInterface {
55 public:
56 explicit PrecedencesPropagator(Model* model)
57 : SatPropagator("PrecedencesPropagator"),
58 trail_(model->GetOrCreate<Trail>()),
59 integer_trail_(model->GetOrCreate<IntegerTrail>()),
60 watcher_(model->GetOrCreate<GenericLiteralWatcher>()),
61 watcher_id_(watcher_->Register(this)) {
62 model->GetOrCreate<SatSolver>()->AddPropagator(this);
63 integer_trail_->RegisterWatcher(&modified_vars_);
64 watcher_->SetPropagatorPriority(watcher_id_, 0);
65 }

100 // Finds all the IntegerVariable that are "after" at least two of the
101 // IntegerVariable in vars. Returns a vector of these precedences relation
102 // sorted by IntegerPrecedences.var so that it is efficient to find all the
103 // IntegerVariable "before" another one.
104 //
105 // Note that we only consider direct precedences here. Given our usage, it may
106 // be better to compute the full reachability in the precedence graph, but in
107 // pratice that may be too slow.
108 //
109 // Note that the IntegerVariable in the vector are also returned in
110 // topological order for a more efficient propagation in
111 // DisjunctivePrecedences::Propagate() where this is used.
112 //
113 // Important: For identical vars, the entry are sorted by index.
114 struct IntegerPrecedences {
115 int index; // position in vars.
116 IntegerVariable var; // An IntegerVariable that is >= to vars[index].
117 int arc_index; // Used by AddPrecedenceReason().
118 IntegerValue offset; // we have: vars[index] + offset <= var
119 };
120 void ComputePrecedences(const std::vector<IntegerVariable>& vars,
121 std::vector<IntegerPrecedences>* output);
122 void AddPrecedenceReason(int arc_index, IntegerValue min_offset,
123 std::vector<Literal>* literal_reason,
124 std::vector<IntegerLiteral>* integer_reason) const;

126 // Similar to ComputePrecedences() but this uses a "slow algorithm". It is not
127 // meant to be called often and explore the full precedences graph.
128 //
129 // If call_compute_precedence is true, this just wrap ComputePrecedences() and
130 // convert its output to this function format, which is less efficient but
131 // more convenient to use.
132 //
133 // Important: This currently assumes the full precedence graph form a DAG.
134 // Otherwise it will just fail and returns no precedence. By contrast
135 // ComputePrecedences() still work.
136 //
137 // TODO(user): Put some work limit in place, as this can be slow. Complexity
138 // is in O(vars.size()) * num_arcs.
139 //
140 // TODO(user): Since we don't need ALL precedences, we could just work on a
141 // sub-DAG of the full precedence graph instead of aborting.
142 //
143 // TODO(user): Many relations can be redundant. Filter them.
144 //
145 // Returns a bunch of precedences relations:
146 // An IntegerVariable >= to vars[indices[i]] + offset[i], for i in indices.
147 struct FullIntegerPrecedence {
148 IntegerVariable var;
149 std::vector<int> indices;
150 std::vector<IntegerValue> offsets;
151 };
152 void ComputeFullPrecedences(bool call_compute_precedences,
153 const std::vector<IntegerVariable>& vars,
154 std::vector<FullIntegerPrecedence>* output);

219 // The core algorithm implementation is split in these functions. One must
220 // first call InitializeBFQueueWithModifiedNodes() that will push all the
221 // IntegerVariable whose lower bound has been modified since the last call.
222 // Then, BellmanFordTarjan() will take care of all the propagation and returns
223 // false in case of conflict. Internally, it uses DisassembleSubtree() which
224 // is the Tarjan variant to detect a possible positive cycle. Before exiting,
225 // it will call CleanUpMarkedArcsAndParents().
226 //
227 // The Tarjan version of the Bellam-Ford algorithm is really nice in our
228 // context because it was really easy to make it incremental. Moreover, it
229 // supports batch increment!
230 //
231 // This implementation is kind of unique because of our context and the fact
232 // that it is incremental, but a good reference is "Negative-cycle detection
233 // algorithms", Boris V. Cherkassky, Andrew V. Goldberg, 1996,
234 // http://people.cs.nctu.edu.tw/~tjshen/doc/ne.pdf
235 void InitializeBFQueueWithModifiedNodes();
236 bool BellmanFordTarjan(Trail* trail);
237 bool DisassembleSubtree(int source, int target,
238 std::vector<bool>* can_be_skipped);
239 void AnalyzePositiveCycle(ArcIndex first_arc, Trail* trail,
240 std::vector<Literal>* must_be_all_true,
241 std::vector<Literal>* literal_reason,
242 std::vector<IntegerLiteral>* integer_reason);
243 void CleanUpMarkedArcsAndParents();
244
245 // Loops over all the arcs and verify that there is no propagation left.
246 // This is only meant to be used in a DCHECK() and is not optimized.
247 bool NoPropagationLeft(const Trail& trail) const;

257 // is done, and automatically updated by the integer_trail_ with all the

259 SparseBitset<IntegerVariable> modified_vars_;

261 // An arc needs to be inspected for propagation (i.e. is impacted) if its
262 // tail_var changed. If an arc has 3 variables (tail, offset, head), it will
263 // appear as 6 different entries in the arcs_ vector, one for each variable
264 // and its negation, each time with a different tail.
265 //
266 // TODO(user): rearranging the index so that the arc of the same node are
267 // consecutive like in StaticGraph should have a big performance impact.
268 //
269 // TODO(user): We do not need to store ArcInfo.tail_var here.
270 absl::StrongVector<IntegerVariable, absl::InlinedVector<ArcIndex, 6>>
271 impacted_arcs_;
272 absl::StrongVector<ArcIndex, ArcInfo> arcs_;
273
274 // This is similar to impacted_arcs_/arcs_ but it is only used to propagate
275 // one of the presence literals when the arc cannot be present. An arc needs
276 // to appear only once in potential_arcs_, but it will be referenced by
277 // all its variable in impacted_potential_arcs_.
278 absl::StrongVector<IntegerVariable, absl::InlinedVector<OptionalArcIndex, 6>>
279 impacted_potential_arcs_;
280 absl::StrongVector<OptionalArcIndex, ArcInfo> potential_arcs_;
281
282 // Temporary vectors used by ComputePrecedences().
283 absl::StrongVector<IntegerVariable, int> var_to_degree_;
284 absl::StrongVector<IntegerVariable, int> var_to_last_index_;
285 struct SortedVar {
286 IntegerVariable var;
287 IntegerValue lower_bound;
288 bool operator<(const SortedVar& other) const {
289 return lower_bound < other.lower_bound;
290 }
291 };
292 std::vector<SortedVar> tmp_sorted_vars_;
293 std::vector<IntegerPrecedences> tmp_precedences_;
294
295 // Each time a literal becomes true, this list the set of arcs for which we
296 // need to decrement their count. When an arc count reach zero, it must be
297 // added to the set of impacted_arcs_. Note that counts never becomes
298 // negative.
299 //
300 // TODO(user): Try a one-watcher approach instead. Note that in most cases

302 absl::StrongVector<LiteralIndex, absl::InlinedVector<ArcIndex, 6>>

304 absl::StrongVector<ArcIndex, int> arc_counts_;

306 // Temp vectors to hold the reason of an assignment.

308 std::vector<IntegerLiteral> integer_reason_;
309
310 // Temp vectors for the Bellman-Ford algorithm. The graph in which this

314 std::vector<bool> bf_in_queue_;
315 std::vector<bool> bf_can_be_skipped_;
316 std::vector<ArcIndex> bf_parent_arc_of_;

432// a == b.
433inline std::function<void(Model*)> Equality(IntegerVariable a,
434 IntegerVariable b) {
435 return [=](Model* model) {
436 model->Add(LowerOrEqual(a, b));
437 model->Add(LowerOrEqual(b, a));
438 };
439}

441// a + offset == b.
442inline std::function<void(Model*)> EqualityWithOffset(IntegerVariable a,
443 IntegerVariable b,

447 model->Add(LowerOrEqualWithOffset(b, a, -offset));

467// literals => (a <= b).

469 IntegerVariable a, IntegerVariable b, absl::Span<const Literal> literals) {
470 return [=](Model* model) {
471 PrecedencesPropagator* p = model->GetOrCreate<PrecedencesPropagator>();

476
477// is_le <=> (a + offset <= b).
478inline std::function<void(Model*)> ReifiedLowerOrEqualWithOffset(
479 IntegerVariable a, IntegerVariable b, int64_t offset, Literal is_le) {
480 return [=](Model* model) {
481 PrecedencesPropagator* p = model->GetOrCreate<PrecedencesPropagator>();
482 p->AddConditionalPrecedenceWithOffset(a, b, IntegerValue(offset), is_le);
483
484 // The negation of (a + offset <= b) is (a + offset > b) which can be
485 // rewritten as (b + 1 - offset <= a).
486 p->AddConditionalPrecedenceWithOffset(b, a, IntegerValue(1 - offset),
487 is_le.Negated());
488 };
489}
490
491// is_eq <=> (a == b).
492inline std::function<void(Model*)> ReifiedEquality(IntegerVariable a,
493 IntegerVariable b,
494 Literal is_eq) {
495 return [=](Model* model) {
496 // We creates two extra Boolean variables in this case.
497 //
498 // TODO(user): Avoid creating them if we already have some literal that
499 // have the same meaning. For instance if a client also wanted to know if
500 // a <= b, he would have called ReifiedLowerOrEqualWithOffset() directly.
501 const Literal is_le = Literal(model->Add(NewBooleanVariable()), true);
502 const Literal is_ge = Literal(model->Add(NewBooleanVariable()), true);
503 model->Add(ReifiedBoolAnd({is_le, is_ge}, is_eq));
504 model->Add(ReifiedLowerOrEqualWithOffset(a, b, 0, is_le));
505 model->Add(ReifiedLowerOrEqualWithOffset(b, a, 0, is_ge));
506 };
507}
508
509// is_eq <=> (a + offset == b).
510inline std::function<void(Model*)> ReifiedEqualityWithOffset(IntegerVariable a,
511 IntegerVariable b,
512 int64_t offset,
513 Literal is_eq) {
514 return [=](Model* model) {
515 // We creates two extra Boolean variables in this case.
516 //
517 // TODO(user): Avoid creating them if we already have some literal that
518 // have the same meaning. For instance if a client also wanted to know if
519 // a <= b, he would have called ReifiedLowerOrEqualWithOffset() directly.
520 const Literal is_le = Literal(model->Add(NewBooleanVariable()), true);
521 const Literal is_ge = Literal(model->Add(NewBooleanVariable()), true);
522 model->Add(ReifiedBoolAnd({is_le, is_ge}, is_eq));
523 model->Add(ReifiedLowerOrEqualWithOffset(a, b, offset, is_le));
524 model->Add(ReifiedLowerOrEqualWithOffset(b, a, -offset, is_ge));
525 };
526}
527
528// a != b.
529inline std::function<void(Model*)> NotEqual(IntegerVariable a,
530 IntegerVariable b) {
531 return [=](Model* model) {
532 // We have two options (is_gt or is_lt) and one must be true.
533 const Literal is_lt = Literal(model->Add(NewBooleanVariable()), true);
534 const Literal is_gt = is_lt.Negated();
535 model->Add(ConditionalLowerOrEqualWithOffset(a, b, 1, is_lt));
536 model->Add(ConditionalLowerOrEqualWithOffset(b, a, 1, is_gt));
537 };
