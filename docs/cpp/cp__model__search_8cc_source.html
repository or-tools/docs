51CpModelView::CpModelView(Model* model)
52 : mapping_(*model->GetOrCreate<CpModelMapping>()),
53 boolean_assignment_(model->GetOrCreate<Trail>()->Assignment()),
54 integer_trail_(*model->GetOrCreate<IntegerTrail>()),
55 integer_encoder_(*model->GetOrCreate<IntegerEncoder>()) {}

57int CpModelView::NumVariables() const { return mapping_.NumProtoVariables(); }

59bool CpModelView::IsFixed(int var) const {
60 if (mapping_.IsBoolean(var)) {
61 return boolean_assignment_.VariableIsAssigned(
62 mapping_.Literal(var).Variable());
63 } else if (mapping_.IsInteger(var)) {
64 return integer_trail_.IsFixed(mapping_.Integer(var));
65 }
66 return true; // Default.
67}
68
69int64_t CpModelView::Min(int var) const {
70 if (mapping_.IsBoolean(var)) {
71 const Literal l = mapping_.Literal(var);
72 return boolean_assignment_.LiteralIsTrue(l) ? 1 : 0;
73 } else if (mapping_.IsInteger(var)) {
74 return integer_trail_.LowerBound(mapping_.Integer(var)).value();
75 }
76 return 0; // Default.
77}
78
79int64_t CpModelView::Max(int var) const {
80 if (mapping_.IsBoolean(var)) {
81 const Literal l = mapping_.Literal(var);
82 return boolean_assignment_.LiteralIsFalse(l) ? 0 : 1;
83 } else if (mapping_.IsInteger(var)) {
84 return integer_trail_.UpperBound(mapping_.Integer(var)).value();
85 }
86 return 0; // Default.
87}

89BooleanOrIntegerLiteral CpModelView::GreaterOrEqual(int var,
90 int64_t value) const {
91 DCHECK(!IsFixed(var));
92 BooleanOrIntegerLiteral result;
93 if (mapping_.IsBoolean(var)) {
94 DCHECK(value == 0 || value == 1);
95 if (value == 1) {
96 result.boolean_literal_index = mapping_.Literal(var).Index();
97 }
98 } else if (mapping_.IsInteger(var)) {
99 result.integer_literal = IntegerLiteral::GreaterOrEqual(
100 mapping_.Integer(var), IntegerValue(value));
101 }
102 return result;
103}
104
105BooleanOrIntegerLiteral CpModelView::LowerOrEqual(int var,
106 int64_t value) const {
107 DCHECK(!IsFixed(var));
108 BooleanOrIntegerLiteral result;
109 if (mapping_.IsBoolean(var)) {
110 DCHECK(value == 0 || value == 1);
111 if (value == 0) {
112 result.boolean_literal_index = mapping_.Literal(var).NegatedIndex();
113 }
114 } else if (mapping_.IsInteger(var)) {
115 result.integer_literal = IntegerLiteral::LowerOrEqual(mapping_.Integer(var),
116 IntegerValue(value));
117 }
118 return result;
119}
120
121BooleanOrIntegerLiteral CpModelView::MedianValue(int var) const {
122 DCHECK(!IsFixed(var));
123 BooleanOrIntegerLiteral result;
124 if (mapping_.IsBoolean(var)) {
125 result.boolean_literal_index = mapping_.Literal(var).NegatedIndex();
126 } else if (mapping_.IsInteger(var)) {
127 const IntegerVariable variable = mapping_.Integer(var);
128 const std::vector<ValueLiteralPair> encoding =
129 integer_encoder_.FullDomainEncoding(variable);
130
131 // 5 values -> returns the second.
132 // 4 values -> returns the second too.
133 // Array is 0 based.
134 const int target = (static_cast<int>(encoding.size()) + 1) / 2 - 1;
135 result.boolean_literal_index = encoding[target].literal.Index();
136 }
137 return result;
138}

149bool ModelHasSchedulingConstraints(const CpModelProto& cp_model_proto) {
150 for (const ConstraintProto& ct : cp_model_proto.constraints()) {
151 if (ct.constraint_case() == ConstraintProto::kNoOverlap) return true;
152 if (ct.constraint_case() == ConstraintProto::kCumulative) return true;
153 }
154 return false;

169// We want a random tie breaking among variables with equivalent values.
170struct NoisyInteger {
171 int64_t value;
172 double noise;
173
174 bool operator<=(const NoisyInteger& other) const {
175 return value < other.value ||
176 (value == other.value && noise <= other.noise);
177 }
178 bool operator>(const NoisyInteger& other) const {
179 return value > other.value || (value == other.value && noise > other.noise);
180 }
181};

185std::function<BooleanOrIntegerLiteral()> ConstructUserSearchStrategy(
186 const CpModelProto& cp_model_proto, Model* model) {
187 if (cp_model_proto.search_strategy().empty()) return nullptr;
188
189 std::vector<DecisionStrategyProto> strategies;
190 for (const DecisionStrategyProto& proto : cp_model_proto.search_strategy()) {
191 strategies.push_back(proto);
192 }
193 const auto& view = *model->GetOrCreate<CpModelView>();
194 const auto& parameters = *model->GetOrCreate<SatParameters>();
195 auto* random = model->GetOrCreate<ModelRandomGenerator>();
196
197 // Note that we copy strategies to keep the return function validity
198 // independently of the life of the passed vector.
199 return [&view, &parameters, random, strategies]() {
200 for (const DecisionStrategyProto& strategy : strategies) {
201 int candidate_ref = -1;
202 int64_t candidate_value = std::numeric_limits<int64_t>::max();
203
204 // TODO(user): Improve the complexity if this becomes an issue which
205 // may be the case if we do a fixed_search.
206
207 // To store equivalent variables in randomized search.
208 const bool randomize_decision =
209 parameters.search_random_variable_pool_size() > 1;
210 TopN<int, NoisyInteger> top_variables(
211 randomize_decision ? parameters.search_random_variable_pool_size()
212 : 1);
213
214 for (const LinearExpressionProto& expr : strategy.exprs()) {
215 const int var = expr.vars(0);
216 if (view.IsFixed(var)) continue;
217
218 int64_t coeff = expr.coeffs(0);
219 int64_t offset = expr.offset();
220 int64_t lb = view.Min(var);
221 int64_t ub = view.Max(var);
222 int ref = var;
223 if (coeff < 0) {
224 lb = -view.Max(var);
225 ub = -view.Min(var);
226 coeff = -coeff;
227 ref = NegatedRef(var);
228 }
229
230 int64_t value(0);
231 switch (strategy.variable_selection_strategy()) {
232 case DecisionStrategyProto::CHOOSE_FIRST:
233 break;
234 case DecisionStrategyProto::CHOOSE_LOWEST_MIN:
235 value = coeff * lb + offset;
236 break;
237 case DecisionStrategyProto::CHOOSE_HIGHEST_MAX:
238 value = -(coeff * ub + offset);
239 break;
240 case DecisionStrategyProto::CHOOSE_MIN_DOMAIN_SIZE:
241 value = coeff * (ub - lb + 1);
242 break;
243 case DecisionStrategyProto::CHOOSE_MAX_DOMAIN_SIZE:
244 value = -coeff * (ub - lb + 1);
245 break;
246 default:
247 LOG(FATAL) << "Unknown VariableSelectionStrategy "
248 << strategy.variable_selection_strategy();
249 }
250
251 if (randomize_decision) {
252 // We need to use -value as we want the minimum valued variables.
253 // We add a random noise to get improve the entropy.
254 const double noise = absl::Uniform(*random, 0., 1.0);
255 top_variables.Add(ref, {-value, noise});
256 candidate_value = std::min(candidate_value, value);
257 } else if (value < candidate_value) {
258 candidate_ref = ref;
259 candidate_value = value;
260 }
261
262 // We can stop scanning if the variable selection strategy is to use the
263 // first unbound variable and no randomization is needed.
264 if (strategy.variable_selection_strategy() ==
265 DecisionStrategyProto::CHOOSE_FIRST &&
266 !randomize_decision) {
267 break;
268 }
269 }
270
271 // Check if one active variable has been found.
272 if (candidate_value == std::numeric_limits<int64_t>::max()) continue;
273
274 // Pick the winner when decisions are randomized.
275 if (randomize_decision) {
276 const std::vector<int>& candidates = top_variables.UnorderedElements();
277 candidate_ref = candidates[absl::Uniform(
278 *random, 0, static_cast<int>(candidates.size()))];
279 }
280
281 DecisionStrategyProto::DomainReductionStrategy selection =
282 strategy.domain_reduction_strategy();
283 if (!RefIsPositive(candidate_ref)) {
284 switch (selection) {
285 case DecisionStrategyProto::SELECT_MIN_VALUE:
286 selection = DecisionStrategyProto::SELECT_MAX_VALUE;
287 break;
288 case DecisionStrategyProto::SELECT_MAX_VALUE:
289 selection = DecisionStrategyProto::SELECT_MIN_VALUE;
290 break;
291 case DecisionStrategyProto::SELECT_LOWER_HALF:
292 selection = DecisionStrategyProto::SELECT_UPPER_HALF;
293 break;
294 case DecisionStrategyProto::SELECT_UPPER_HALF:
295 selection = DecisionStrategyProto::SELECT_LOWER_HALF;
296 break;
297 default:
298 break;
299 }
300 }
301
302 const int var = PositiveRef(candidate_ref);
303 const int64_t lb = view.Min(var);
304 const int64_t ub = view.Max(var);
305 switch (selection) {
306 case DecisionStrategyProto::SELECT_MIN_VALUE:
307 return view.LowerOrEqual(var, lb);
308 case DecisionStrategyProto::SELECT_MAX_VALUE:
309 return view.GreaterOrEqual(var, ub);
310 case DecisionStrategyProto::SELECT_LOWER_HALF:
311 return view.LowerOrEqual(var, lb + (ub - lb) / 2);
312 case DecisionStrategyProto::SELECT_UPPER_HALF:
313 return view.GreaterOrEqual(var, ub - (ub - lb) / 2);
314 case DecisionStrategyProto::SELECT_MEDIAN_VALUE:
315 return view.MedianValue(var);
316 default:
317 LOG(FATAL) << "Unknown DomainReductionStrategy "
318 << strategy.domain_reduction_strategy();
319 }
320 }
321 return BooleanOrIntegerLiteral();
322 };
323}

326std::function<BooleanOrIntegerLiteral()> ConstructHeuristicSearchStrategy(
327 const CpModelProto& cp_model_proto, Model* model) {
328 if (ModelHasSchedulingConstraints(cp_model_proto)) {
329 std::vector<std::function<BooleanOrIntegerLiteral()>> heuristics;
330 const auto& params = *model->GetOrCreate<SatParameters>();
331 bool possible_new_constraints = false;
332 if (params.use_dynamic_precedence_in_disjunctive()) {
333 possible_new_constraints = true;
334 heuristics.push_back(DisjunctivePrecedenceSearchHeuristic(model));
335 }
336 if (params.use_dynamic_precedence_in_cumulative()) {
337 possible_new_constraints = true;
338 heuristics.push_back(CumulativePrecedenceSearchHeuristic(model));
339 }
340
341 // Tricky: we need to create this at level zero in case there are no linear
342 // constraint in the model at the beginning.
343 //
344 // TODO(user): Alternatively, support creation of SatPropagator at positive
345 // level.
346 if (possible_new_constraints && params.new_linear_propagation()) {
347 model->GetOrCreate<LinearPropagator>();
348 }
349
350 heuristics.push_back(SchedulingSearchHeuristic(model));
351 return SequentialSearch(std::move(heuristics));
352 }
353 return PseudoCost(model);
354}

356std::function<BooleanOrIntegerLiteral()>
357ConstructIntegerCompletionSearchStrategy(
358 const std::vector<IntegerVariable>& variable_mapping,
359 IntegerVariable objective_var, Model* model) {
360 const auto& params = *model->GetOrCreate<SatParameters>();
361 if (!params.instantiate_all_variables()) {
362 return []() { return BooleanOrIntegerLiteral(); };
363 }
364
365 std::vector<IntegerVariable> decisions;
366 for (const IntegerVariable var : variable_mapping) {
367 if (var == kNoIntegerVariable) continue;
368
369 // Make sure we try to fix the objective to its lowest value first.
370 // TODO(user): we could also fix terms of the objective in the right
371 // direction.
372 if (var == NegationOf(objective_var)) {
373 decisions.push_back(objective_var);
374 } else {
375 decisions.push_back(var);
376 }
377 }
378 return FirstUnassignedVarAtItsMinHeuristic(decisions, model);
379}

381// Constructs a search strategy that follow the hint from the model.
382std::function<BooleanOrIntegerLiteral()> ConstructHintSearchStrategy(
383 const CpModelProto& cp_model_proto, CpModelMapping* mapping, Model* model) {
384 std::vector<BooleanOrIntegerVariable> vars;
385 std::vector<IntegerValue> values;
386 for (int i = 0; i < cp_model_proto.solution_hint().vars_size(); ++i) {
387 const int ref = cp_model_proto.solution_hint().vars(i);
388 CHECK(RefIsPositive(ref));
389 BooleanOrIntegerVariable var;
390 if (mapping->IsBoolean(ref)) {
391 var.bool_var = mapping->Literal(ref).Variable();
392 } else {
393 var.int_var = mapping->Integer(ref);
394 }
395 vars.push_back(var);
396 values.push_back(IntegerValue(cp_model_proto.solution_hint().values(i)));
397 }
398 return FollowHint(vars, values, model);
399}

401std::function<BooleanOrIntegerLiteral()> ConstructFixedSearchStrategy(
402 std::function<BooleanOrIntegerLiteral()> user_search,
403 std::function<BooleanOrIntegerLiteral()> heuristic_search,
404 std::function<BooleanOrIntegerLiteral()> integer_completion) {
405 // We start by the user specified heuristic.
406 std::vector<std::function<BooleanOrIntegerLiteral()>> heuristics;
407 if (user_search != nullptr) {
408 heuristics.push_back(user_search);
409 }
410 if (heuristic_search != nullptr) {
411 heuristics.push_back(heuristic_search);
412 }
413 if (integer_completion != nullptr) {
414 heuristics.push_back(integer_completion);
415 }
416
417 return SequentialSearch(heuristics);
418}

420std::function<BooleanOrIntegerLiteral()> InstrumentSearchStrategy(
421 const CpModelProto& cp_model_proto,
422 const std::vector<IntegerVariable>& variable_mapping,
423 const std::function<BooleanOrIntegerLiteral()>& instrumented_strategy,
424 Model* model) {
425 std::vector<int> ref_to_display;
426 for (int i = 0; i < cp_model_proto.variables_size(); ++i) {
427 if (variable_mapping[i] == kNoIntegerVariable) continue;
428 if (cp_model_proto.variables(i).name().empty()) continue;
429 ref_to_display.push_back(i);
430 }
431 std::sort(ref_to_display.begin(), ref_to_display.end(), [&](int i, int j) {
432 return cp_model_proto.variables(i).name() <
433 cp_model_proto.variables(j).name();
434 });
435
436 std::vector<std::pair<int64_t, int64_t>> old_domains(variable_mapping.size());
437 return [instrumented_strategy, model, variable_mapping, cp_model_proto,
438 old_domains, ref_to_display]() mutable {
439 const BooleanOrIntegerLiteral decision = instrumented_strategy();
440 if (!decision.HasValue()) return decision;
441
442 if (decision.boolean_literal_index != kNoLiteralIndex) {
443 const Literal l = Literal(decision.boolean_literal_index);
444 LOG(INFO) << "Boolean decision " << l;
445 const auto& encoder = model->Get<IntegerEncoder>();
446 for (const IntegerLiteral i_lit : encoder->GetIntegerLiterals(l)) {
447 LOG(INFO) << " - associated with " << i_lit;
448 }
449 for (const auto [var, value] : encoder->GetEqualityLiterals(l)) {
450 LOG(INFO) << " - associated with " << var << " == " << value;
451 }
452 } else {
453 LOG(INFO) << "Integer decision " << decision.integer_literal;
454 }
455 const int level = model->Get<Trail>()->CurrentDecisionLevel();
456 std::string to_display =
457 absl::StrCat("Diff since last call, level=", level, "\n");
458 IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
459 for (const int ref : ref_to_display) {
460 const IntegerVariable var = variable_mapping[ref];
461 const std::pair<int64_t, int64_t> new_domain(
462 integer_trail->LowerBound(var).value(),
463 integer_trail->UpperBound(var).value());
464 if (new_domain != old_domains[ref]) {
465 absl::StrAppend(&to_display, cp_model_proto.variables(ref).name(), " [",
466 old_domains[ref].first, ",", old_domains[ref].second,
467 "] -> [", new_domain.first, ",", new_domain.second,
468 "]\n");
469 old_domains[ref] = new_domain;
470 }
471 }
472 LOG(INFO) << to_display;
473 return decision;
474 };
475}

477// This generates a valid random seed (base_seed + delta) without overflow.
478// We assume |delta| is small.
479int ValidSumSeed(int base_seed, int delta) {
480 CHECK_GE(delta, 0);
481 int64_t result = int64_t{base_seed} + int64_t{delta};
482 const int64_t int32max = int64_t{std::numeric_limits<int>::max()};
483 while (result > int32max) {
484 result -= int32max;
485 }
486 return static_cast<int>(result);
487}

648 {
649 SatParameters new_params = base_params;
650 new_params.set_linearization_level(2);
651 new_params.set_search_branching(SatParameters::LP_SEARCH);
652 if (base_params.use_dual_scheduling_heuristics()) {
653 AddDualSchedulingHeuristics(new_params);
654 }
655 strategies["reduced_costs"] = new_params;
656 }
657
658 {
659 // Note: no dual scheduling heuristics.
660 SatParameters new_params = base_params;
661 new_params.set_linearization_level(2);
662 new_params.set_search_branching(SatParameters::PSEUDO_COST_SEARCH);
663 new_params.set_exploit_best_solution(true);
664 strategies["pseudo_costs"] = new_params;
665 }
666
667 // Less encoding.
668 {
669 SatParameters new_params = base_params;
670 new_params.set_boolean_encoding_level(0);
671 strategies["less_encoding"] = new_params;
672 }
673
674 // Base parameters for shared tree worker.
675 {
676 SatParameters new_params = base_params;
677 new_params.set_search_branching(SatParameters::AUTOMATIC_SEARCH);
678 strategies["shared_tree"] = new_params;
679 }
680
681 // Base parameters for LNS worker.
682 {
683 SatParameters new_params = base_params;
684 new_params.set_stop_after_first_solution(false);
685 new_params.set_cp_model_presolve(true);
686 new_params.set_cp_model_probing_level(0);
687 new_params.set_symmetry_level(0);
688 new_params.set_find_big_linear_overlap(false);
689 new_params.set_log_search_progress(false);
690 new_params.set_solution_pool_size(1); // Keep the best solution found.
691 strategies["lns"] = new_params;
692 }
693
694 // Add user defined ones.
695 // Note that this might be merged to our default ones.
696 for (const SatParameters& params : base_params.subsolver_params()) {
697 auto it = strategies.find(params.name());
698 if (it != strategies.end()) {
699 it->second.MergeFrom(params);
700 } else {
701 // Merge the named parameters with the base parameters to create the new
702 // parameters.
703 SatParameters new_params = base_params;
704 new_params.MergeFrom(params);
705 strategies[params.name()] = new_params;
706 }
707 }
708
709 return strategies;
710}

712// Note: in flatzinc setting, we know we always have a fixed search defined.
713//
714// Things to try:
715// - Specialize for purely boolean problems
716// - Disable linearization_level options for non linear problems
717// - Fast restart in randomized search
718// - Different propatation levels for scheduling constraints
719std::vector<SatParameters> GetDiverseSetOfParameters(
720 const SatParameters& base_params, const CpModelProto& cp_model) {
721 // Defines a set of named strategies so it is easier to read in one place
722 // the one that are used. See below.
723 const auto strategies = GetNamedParameters(base_params);
724
725 // We only use a "fixed search" worker if some strategy is specified or
726 // if we have a scheduling model.
727 //
728 // TODO(user): For scheduling, this is important to find good first solution
729 // but afterwards it is not really great and should probably be replaced by a
730 // LNS worker.
731 const bool use_fixed_strategy = !cp_model.search_strategy().empty() ||
732 ModelHasSchedulingConstraints(cp_model);
733
734 // Our current set of strategies
735 //
736 // TODO(user): Avoid launching two strategies if they are the same,
737 // like if there is no lp, or everything is already linearized at level 1.
738 std::vector<std::string> names;
739
740 // Starts by adding user specified ones.
741 for (const std::string& name : base_params.extra_subsolvers()) {
742 names.push_back(name);
743 }
744
745 // We use the default if empty.
746 if (base_params.subsolvers().empty()) {
747 // Note that the order is important as the list can be truncated.
748 names.push_back("default_lp");
749 names.push_back("fixed");
750 names.push_back("core");
751 names.push_back("no_lp");
752 names.push_back("max_lp");
753 names.push_back("quick_restart");
754 names.push_back("reduced_costs");
755 names.push_back("quick_restart_no_lp");
756 names.push_back("pseudo_costs");
757 names.push_back("lb_tree_search");
758 names.push_back("probing");
759 names.push_back("objective_lb_search");
760 names.push_back("objective_shaving_search_no_lp");
761 names.push_back("objective_shaving_search_max_lp");
762 names.push_back("probing_max_lp");
763 names.push_back("objective_lb_search_no_lp");
764 names.push_back("objective_lb_search_max_lp");
765
766#if !defined(__PORTABLE_PLATFORM__) && defined(USE_SCIP)
767 if (absl::GetFlag(FLAGS_cp_model_use_max_hs)) names.push_back("max_hs");
768#endif // !defined(__PORTABLE_PLATFORM__) && defined(USE_SCIP)
769 } else {
770 for (const std::string& name : base_params.subsolvers()) {
771 // Hack for flatzinc. At the time of parameter setting, the objective is
772 // not expanded. So we do not know if core is applicable or not.
773 if (name == "core_or_no_lp") {
774 if (!cp_model.has_objective() ||
775 cp_model.objective().vars_size() <= 1) {
776 names.push_back("no_lp");
777 } else {
778 names.push_back("core");
779 }
780 } else {
781 names.push_back(name);
782 }
783 }
784 }
785
786 // Remove the names that should be ignored.
787 absl::flat_hash_set<std::string> to_ignore;
788 for (const std::string& name : base_params.ignore_subsolvers()) {
789 to_ignore.insert(name);
790 }
791 int new_size = 0;
792 for (const std::string& name : names) {
793 if (to_ignore.contains(name)) continue;
794 names[new_size++] = name;
795 }
796 names.resize(new_size);
797
798 // Creates the diverse set of parameters with names and seed.
799 std::vector<SatParameters> result;
800 for (const std::string& name : names) {
801 SatParameters params = strategies.at(name);
802
803 // Do some filtering.
804 if (!use_fixed_strategy &&
805 params.search_branching() == SatParameters::FIXED_SEARCH) {
806 continue;
807 }
808
809 // TODO(user): Enable probing_search in deterministic mode.
810 // Currently it timeouts on small problems as the deterministic time limit
811 // never hits the sharding limit.
812 if (params.use_probing_search() && params.interleave_search()) continue;
813
814 // TODO(user): Enable shaving search in interleave mode.
815 // Currently it do not respect ^C, and has no per chunk time limit.
816 if (params.use_objective_shaving_search() && params.interleave_search()) {
817 continue;
818 }
819
820 // In the corner case of empty variable, lets not schedule the probing as
821 // it currently just loop forever instead of returning right away.
822 if (params.use_probing_search() && cp_model.variables().empty()) continue;
823
824 if (cp_model.has_objective() && !cp_model.objective().vars().empty()) {
825 // Disable core search if there is only 1 term in the objective.
826 if (cp_model.objective().vars().size() == 1 &&
827 params.optimize_with_core()) {
828 continue;
829 }
830
831 if (name == "less_encoding") continue;
832
833 // Disable subsolvers that do not implement the deterministic mode.
834 //
835 // TODO(user): Enable lb_tree_search in deterministic mode.
836 if (params.interleave_search() &&
837 (params.optimize_with_lb_tree_search() ||
838 params.use_objective_lb_search())) {
839 continue;
840 }
841 } else {
842 // Remove subsolvers that require an objective.
843 if (params.optimize_with_lb_tree_search()) continue;
844 if (params.optimize_with_core()) continue;
845 if (params.use_objective_lb_search()) continue;
846 if (params.use_objective_shaving_search()) continue;
847 if (params.search_branching() == SatParameters::LP_SEARCH) continue;
848 if (params.search_branching() == SatParameters::PSEUDO_COST_SEARCH) {
849 continue;
850 }
851 }
852
853 // Add this strategy.
854 //
855 // TODO(user): Find a better randomization for the seed so that changing
856 // random_seed() has more impact?
857 params.set_name(name);
858 params.set_random_seed(ValidSumSeed(base_params.random_seed(),
859 static_cast<int>(result.size()) + 1));
860 result.push_back(params);
861 }
862
863 // In interleaved mode, we run all of them
864 // TODO(user): Actually make sure the gap num_workers <-> num_heuristics is
865 // contained.
866 if (base_params.interleave_search()) return result;
867
868 const int num_non_shared_workers = std::max(
869 0, base_params.num_workers() - base_params.shared_tree_num_workers());
870
871 if (cp_model.has_objective() && !cp_model.objective().vars().empty()) {
872 const auto heuristic_num_workers = [](int num_workers) {
873 DCHECK_GE(num_workers, 0);
874 if (num_workers == 1) return 1;
875 if (num_workers <= 4) return num_workers - 1;
876 if (num_workers <= 8) return num_workers - 2;
877 if (num_workers <= 16) return num_workers - (num_workers / 4 + 1);
878 return num_workers - (num_workers / 2 - 3);
879 };
880 const int target = std::min<int>(
881 heuristic_num_workers(num_non_shared_workers), result.size());
882
883 // If there is an objective, the extra workers will use LNS.
884 // Make sure we have at least min_num_lns_workers() of them.
885 if (result.size() > target) result.resize(target);
886 } else { // No objective.
887 // If strategies that do not require a full worker are present, leave a
888 // few workers for them.
889 const bool need_extra_workers =
890 (base_params.use_rins_lns() || base_params.use_feasibility_pump());
891 // Currently, we have 8 SAT search heuristics. So
892 const int num_extra_workers =
893 num_non_shared_workers <= 4 ? 0 : 1 + need_extra_workers;
894 const int target = std::min<int>(num_non_shared_workers - num_extra_workers,
895 result.size());
896 if (result.size() > target) result.resize(target);
897 }
898 return result;
899}

901std::vector<SatParameters> GetFirstSolutionParams(
902 const SatParameters& base_params, const CpModelProto& /*cp_model*/,
903 int num_params_to_generate) {
904 std::vector<SatParameters> result;
905 if (num_params_to_generate <= 0) return result;
906 int num_random = 0;
907 int num_random_qr = 0;
908 while (result.size() < num_params_to_generate) {
909 SatParameters new_params = base_params;
910 const int base_seed = base_params.random_seed();
911 if (num_random <= num_random_qr) { // Random search.
912 new_params.set_search_branching(SatParameters::RANDOMIZED_SEARCH);
913 new_params.set_search_random_variable_pool_size(5);
914 new_params.set_random_seed(ValidSumSeed(base_seed, 2 * num_random + 1));
915 if (num_random % 2 == 1) {
916 new_params.set_name("random_no_lp");
917 new_params.set_linearization_level(0);
918 } else {
919 new_params.set_name("random");
920 }
921 num_random++;
922 } else { // Random quick restart.
923 new_params.set_search_branching(
924 SatParameters::PORTFOLIO_WITH_QUICK_RESTART_SEARCH);
925 new_params.set_search_random_variable_pool_size(5);
926 new_params.set_random_seed(ValidSumSeed(base_seed, 2 * num_random_qr));
927 if (num_random_qr % 2 == 1) {
928 new_params.set_name("random_quick_restart_no_lp");
929 new_params.set_linearization_level(0);
930 } else {
931 new_params.set_name("random_quick_restart");
932 }
933 num_random_qr++;
934 }
935 result.push_back(new_params);
936 }
937 return result;
938}

940std::vector<SatParameters> GetWorkSharingParams(
941 const SatParameters& base_params, const CpModelProto& cp_model,
942 int num_params_to_generate) {
943 std::vector<SatParameters> result;
944 // TODO(user): We could support assumptions, it's just not implemented.
945 if (!cp_model.assumptions().empty()) return result;
946 if (num_params_to_generate <= 0) return result;
947
948 const auto strategies = GetNamedParameters(base_params);
949 const SatParameters& shared_tree_base_params = strategies.at("shared_tree");
950 int num_workers = 0;
951 while (result.size() < num_params_to_generate) {
952 SatParameters new_params = shared_tree_base_params;
953 const int base_seed = base_params.random_seed();
954 new_params.set_random_seed(ValidSumSeed(base_seed, 2 * num_workers + 1));
955 // We force this parameter as it could have been forgotten when set
956 // manually.
957 new_params.set_use_shared_tree_search(true);
958
959 // These settings don't make sense with shared tree search, turn them off as
960 // they can break things.
961 new_params.set_optimize_with_core(false);
962 new_params.set_optimize_with_lb_tree_search(false);
963 new_params.set_optimize_with_max_hs(false);
964
965 absl::string_view lp_tags[] = {"no", "default", "max"};
966 new_params.set_name(absl::StrCat(
967 "shared_", lp_tags[std::min(new_params.linearization_level(), 2)],
968 "_lp_", num_workers));
969 num_workers++;
970 result.push_back(new_params);
971 }
972
973 return result;
974}
