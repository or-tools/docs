45bool MakeRelocateNeighborsOperator::MakeNeighbor() {
46 const auto do_move = [this](int64_t before_chain, int64_t destination) {
47 int64_t chain_end = Next(before_chain);
48 if (IsPathEnd(chain_end)) return false;
49 if (chain_end == destination) return false;
50 const int64_t max_arc_value = arc_evaluator_(destination, chain_end);
51 int64_t next = Next(chain_end);
52 while (!IsPathEnd(next) &&
53 arc_evaluator_(chain_end, next) <= max_arc_value) {
54 // We return false here to avoid symmetric moves. The rationale is that
55 // if destination is part of the same group as the chain, we probably want
56 // to extend the chain to contain it, which means finding another
57 // destination further down the path.
58 // TODO(user): Add a parameter to either return false or break here,
59 // depending if we want to permutate nodes within the same chain.
60 if (next == destination) return false;
61 chain_end = next;
62 next = Next(chain_end);
63 }
64 return MoveChainAndRepair(before_chain, chain_end, destination);
65 };
66 if (HasNeighbors()) {
67 const int64_t node = GetNeighborForBaseNode(0);
68 if (IsInactive(node)) return false;
69 return do_move(/*before_chain=*/Prev(node),
70 /*destination=*/BaseNode(0));
71 } else {
72 return do_move(/*before_chain=*/BaseNode(0),
73 /*destination=*/BaseNode(1));
74 }
75}

77bool MakeRelocateNeighborsOperator::MoveChainAndRepair(int64_t before_chain,
78 int64_t chain_end,
79 int64_t destination) {
80 if (MoveChain(before_chain, chain_end, destination)) {
81 if (!IsPathStart(destination)) {
82 int64_t current = Prev(destination);
83 int64_t last = chain_end;
84 if (current == last) { // chain was just before destination
85 current = before_chain;
86 }
87 while (last >= 0 && !IsPathStart(current) && current != last) {
88 last = Reposition(current, last);
89 current = Prev(current);
90 }
91 }
92 return true;
93 }
94 return false;
95}

97int64_t MakeRelocateNeighborsOperator::Reposition(int64_t before_to_move,
98 int64_t up_to) {
99 const int64_t kNoChange = -1;
100 const int64_t to_move = Next(before_to_move);
101 int64_t next = Next(to_move);
102 if (Var(to_move)->Contains(next)) {
103 return kNoChange;
104 }
105 int64_t prev = next;
106 next = Next(next);
107 while (prev != up_to) {
108 if (Var(prev)->Contains(to_move) && Var(to_move)->Contains(next)) {
109 MoveChain(before_to_move, to_move, prev);
110 return up_to;
111 }
112 prev = next;
113 next = Next(next);
114 }
115 if (Var(prev)->Contains(to_move)) {
116 MoveChain(before_to_move, to_move, prev);
117 return to_move;
118 }
119 return kNoChange;
120}

133 path_predecessor_(vars.size(), -1),
134 touched_(vars.size()) {
135 for (int i = 0; i < alternative_sets_.size(); ++i) {

138 }

142bool SwapActiveToShortestPathOperator::MakeNeighbor() {
143 const int64_t before_chain = BaseNode(0);
144 if (to_alternative_set_[before_chain] != -1) return false;
145 int64_t next = Next(before_chain);
146 std::vector<int> alternatives;
147 while (!IsPathEnd(next) && to_alternative_set_[next] != -1 &&
148 alternative_sets_[to_alternative_set_[next]].size() > 1) {
149 alternatives.push_back(to_alternative_set_[next]);
150 next = Next(next);
151 }
152 if (alternatives.empty()) return false;
153 const int sink = next;
154 next = OldNext(before_chain);
155 bool swap_done = false;
156 for (int64_t node : GetShortestPath(before_chain, sink, alternatives)) {
157 if (node != next) {
158 SwapActiveAndInactive(next, node);
159 swap_done = true;
160 }
161 next = OldNext(next);
162 }
163 return swap_done;
164}

166const std::vector<int64_t>& SwapActiveToShortestPathOperator::GetShortestPath(
167 int source, int sink, const std::vector<int>& alternative_chain) {
168 path_.clear();
169 if (alternative_chain.empty()) return path_;
170 // Initializing values at the first "layer" after the source (from source to
171 // all alternatives at rank 0).
172 const std::vector<int64_t>& first_alternative_set =
173 alternative_sets_[alternative_chain[0]];
174 std::vector<int64_t> prev_values;
175 prev_values.reserve(first_alternative_set.size());
176 for (int alternative_node : first_alternative_set) {
177 prev_values.push_back(arc_evaluator_(source, alternative_node));
178 }
179 // Updating values "layer" by "layer" (each one is fully connected to the
180 // previous one).
181 std::vector<int64_t> current_values;
182 for (int rank = 1; rank < alternative_chain.size(); ++rank) {
183 const std::vector<int64_t>& current_alternative_set =
184 alternative_sets_[alternative_chain[rank]];
185 current_values.clear();
186 current_values.reserve(current_alternative_set.size());
187 const std::vector<int64_t>& prev_alternative_set =
188 alternative_sets_[alternative_chain[rank - 1]];
189 for (int alternative_node : current_alternative_set) {
190 int64_t min_value = kint64max;
191 int predecessor = -1;
192 for (int prev_alternative = 0;
193 prev_alternative < prev_alternative_set.size(); ++prev_alternative) {
194 const int64_t new_value =
195 CapAdd(prev_values[prev_alternative],
196 arc_evaluator_(prev_alternative_set[prev_alternative],
197 alternative_node));
198 if (new_value <= min_value) {
199 min_value = new_value;
200 predecessor = prev_alternative_set[prev_alternative];
201 }
202 }
203 current_values.push_back(min_value);
204 path_predecessor_[alternative_node] = predecessor;
205 }
206 prev_values.swap(current_values);
207 }
208 // Get the predecessor in the shortest path to sink in the last layer.
209 int64_t min_value = kint64max;
210 int predecessor = -1;
211 const std::vector<int64_t>& last_alternative_set =
212 alternative_sets_[alternative_chain.back()];
213 for (int alternative = 0; alternative < last_alternative_set.size();
214 ++alternative) {
215 const int64_t new_value =
216 CapAdd(prev_values[alternative],
217 arc_evaluator_(last_alternative_set[alternative], sink));
218 if (new_value <= min_value) {
219 min_value = new_value;
220 predecessor = last_alternative_set[alternative];
221 }
222 }
223 if (predecessor == -1) return path_;
224 // Build the path from predecessors on the shortest path.
225 path_.resize(alternative_chain.size(), predecessor);
226 touched_.SparseClearAll();
227 touched_.Set(predecessor);
228 for (int rank = alternative_chain.size() - 2; rank >= 0; --rank) {
229 path_[rank] = path_predecessor_[path_[rank + 1]];
230 if (touched_[path_[rank]]) {
231 path_.clear();
232 return path_;
233 }
234 touched_.Set(path_[rank]);
235 }
236 return path_;
237}

251bool MakePairActiveOperator::MakeOneNeighbor() {
252 while (inactive_pair_ < pairs_.size()) {
253 if (PathOperator::MakeOneNeighbor()) return true;
254 ResetPosition();
255 const auto& [pickup_alternatives, delivery_alternatives] =
256 pairs_[inactive_pair_];
257 if (inactive_pair_first_index_ < pickup_alternatives.size() - 1) {
258 ++inactive_pair_first_index_;
259 } else if (inactive_pair_second_index_ < delivery_alternatives.size() - 1) {
260 inactive_pair_first_index_ = 0;
261 ++inactive_pair_second_index_;
262 } else {
263 inactive_pair_ = FindNextInactivePair(inactive_pair_ + 1);
264 inactive_pair_first_index_ = 0;
265 inactive_pair_second_index_ = 0;
266 }
267 }
268 return false;
269}

271bool MakePairActiveOperator::MakeNeighbor() {
272 DCHECK_EQ(StartNode(0), StartNode(1));
273 // Inserting the second node of the pair before the first one which ensures
274 // that the only solutions where both nodes are next to each other have the
275 // first node before the second (the move is not symmetric and doing it this
276 // way ensures that a potential precedence constraint between the nodes of the
277 // pair is not violated).
278 const auto& [pickup_alternatives, delivery_alternatives] =
279 pairs_[inactive_pair_];
280 return MakeActive(delivery_alternatives[inactive_pair_second_index_],
281 BaseNode(1)) &&
282 MakeActive(pickup_alternatives[inactive_pair_first_index_],
283 BaseNode(0));
284}

350bool PairRelocateOperator::MakeNeighbor() {
351 DCHECK_EQ(StartNode(1), StartNode(2));
352 const int64_t first_pair_node = BaseNode(kPairFirstNode);
353 if (IsPathStart(first_pair_node)) {
354 return false;
355 }
356 int64_t first_prev = Prev(first_pair_node);
357 const int second_pair_node = GetActiveAlternativeSibling(first_pair_node);
358 if (second_pair_node < 0 || IsPathEnd(second_pair_node) ||
359 IsPathStart(second_pair_node)) {
360 return false;
361 }
362 const int64_t second_prev = Prev(second_pair_node);
363
364 const int64_t first_node_destination = BaseNode(kPairFirstNodeDestination);
365 if (first_node_destination == second_pair_node) {
366 // The second_pair_node -> first_pair_node link is forbidden.
367 return false;
368 }
369
370 const int64_t second_node_destination = BaseNode(kPairSecondNodeDestination);
371 if (second_prev == first_pair_node && first_node_destination == first_prev &&
372 second_node_destination == first_prev) {
373 // If the current sequence is first_prev -> first_pair_node ->
374 // second_pair_node, and both 1st and 2nd are moved both to prev, the result
375 // of the move will be first_prev -> first_pair_node -> second_pair_node,
376 // which is no move.
377 return false;
378 }
379
380 // Relocation is successful if both moves are feasible and at least one of the
381 // nodes moves.
382 if (second_pair_node == second_node_destination ||
383 first_pair_node == first_node_destination) {
384 return false;
385 }
386 const bool moved_second_pair_node =
387 MoveChain(second_prev, second_pair_node, second_node_destination);
388 // Explicitly calling Prev as second_pair_node might have been moved before
389 // first_pair_node.
390 const bool moved_first_pair_node =
391 MoveChain(Prev(first_pair_node), first_pair_node, first_node_destination);
392 // Swapping alternatives in.
393 SwapActiveAndInactive(second_pair_node,
394 BaseSiblingAlternativeNode(kPairFirstNode));
395 SwapActiveAndInactive(first_pair_node, BaseAlternativeNode(kPairFirstNode));
396 return moved_first_pair_node || moved_second_pair_node;
397}

465bool LightPairRelocateOperator::MakeNeighbor() {
466 const auto do_move = [this](int64_t node, int64_t destination,
467 bool destination_is_lifo) {
468 if (IsPathStart(node) || IsPathEnd(node) || IsInactive(node)) return false;
469 const int64_t prev = Prev(node);
470 if (IsPathEnd(node)) return false;
471 const int64_t sibling = GetActiveAlternativeSibling(node);
472 if (sibling == -1 || destination == sibling) return false;
473
474 // Note: MoveChain will return false if it is a no-op (moving the chain to
475 // its current position). However we want to accept the move if at least
476 // node or sibling gets moved to a new position. Therefore we want to be
477 // sure both MoveChains are called and at least one succeeds.
478
479 // Special case handling relocating the first node of a pair "before" the
480 // first node of another pair. Limiting this to relocating after the start
481 // of the path as other moves will be mostly equivalent to relocating
482 // "after".
483 // TODO(user): extend to relocating before the start of sub-tours (when
484 // all pairs have been matched).
485 if (IsPathStart(destination)) {
486 const bool ok = MoveChain(prev, node, destination);
487 const int64_t destination_sibling =
488 GetActiveAlternativeSibling(Next(node));
489 if (destination_sibling == -1) {
490 // Not inserting before a pair node: insert sibling after node.
491 return MoveChain(Prev(sibling), sibling, node) || ok;
492 } else {
493 // Depending on the lifo status of the path, insert sibling before or
494 // after destination_sibling since node is being inserted before
495 // next(destination).
496 if (!destination_is_lifo) {
497 if (Prev(destination_sibling) == sibling) return ok;
498 return MoveChain(Prev(sibling), sibling, Prev(destination_sibling)) ||
499 ok;
500 } else {
501 return MoveChain(Prev(sibling), sibling, destination_sibling) || ok;
502 }
503 }
504 }
505 // Relocating the first node of a pair "after" the first node of another
506 // pair.
507 const int64_t destination_sibling =
508 GetActiveAlternativeSibling(destination);
509 if (destination_sibling == -1) return false;
510 const bool ok = MoveChain(prev, node, destination);
511 if (!destination_is_lifo) {
512 return MoveChain(Prev(sibling), sibling, destination_sibling) || ok;
513 } else {
514 if (Prev(destination_sibling) == sibling) return ok;
515 return MoveChain(Prev(sibling), sibling, Prev(destination_sibling)) || ok;
516 }
517 };
518 // TODO(user): Add support for lifo for neighbor-based move.
519 return HasNeighbors()
520 ? do_move(/*node=*/GetNeighborForBaseNode(0),
521 /*destination=*/BaseNode(0),
522 /*destination_is_lifo=*/false)
523 : do_move(/*node=*/Next(BaseNode(0)), /*destination=*/BaseNode(1),
524 force_lifo_ != nullptr && force_lifo_(StartNode(1)));
525}

542bool PairExchangeOperator::MakeNeighbor() {
543 const int64_t node1 = BaseNode(0);
544 int64_t prev1, sibling1, sibling_prev1 = -1;
545 if (!GetPreviousAndSibling(node1, &prev1, &sibling1, &sibling_prev1)) {
546 return false;
547 }
548 int64_t node2 = -1;
549 if (!HasNeighbors()) {
550 node2 = BaseNode(1);
551 } else {
552 const int64_t neighbor = GetNeighborForBaseNode(0);
553 if (IsInactive(neighbor) || IsPathStart(neighbor)) return false;
554 node2 = Prev(neighbor);
555 }
556 int64_t prev2, sibling2, sibling_prev2 = -1;
557 if (!GetPreviousAndSibling(node2, &prev2, &sibling2, &sibling_prev2)) {
558 return false;
559 }
560 bool status = true;
561 // Exchanging node1 and node2.
562 if (node1 == prev2) {
563 status = MoveChain(prev2, node2, prev1);
564 if (sibling_prev1 == node2) sibling_prev1 = node1;
565 if (sibling_prev2 == node2) sibling_prev2 = node1;
566 } else if (node2 == prev1) {
567 status = MoveChain(prev1, node1, prev2);
568 if (sibling_prev1 == node1) sibling_prev1 = node2;
569 if (sibling_prev2 == node1) sibling_prev2 = node2;
570 } else {
571 status = MoveChain(prev1, node1, node2) && MoveChain(prev2, node2, prev1);
572 if (sibling_prev1 == node1) {
573 sibling_prev1 = node2;
574 } else if (sibling_prev1 == node2) {
575 sibling_prev1 = node1;
576 }
577 if (sibling_prev2 == node1) {
578 sibling_prev2 = node2;
579 } else if (sibling_prev2 == node2) {
580 sibling_prev2 = node1;
581 }
582 }
583 if (!status) return false;
584 // Exchanging sibling1 and sibling2.
585 if (sibling1 == sibling_prev2) {
586 status = MoveChain(sibling_prev2, sibling2, sibling_prev1);
587 } else if (sibling2 == sibling_prev1) {
588 status = MoveChain(sibling_prev1, sibling1, sibling_prev2);
589 } else {
590 status = MoveChain(sibling_prev1, sibling1, sibling2) &&
591 MoveChain(sibling_prev2, sibling2, sibling_prev1);
592 }
593 // Swapping alternatives in.

624bool PairExchangeRelocateOperator::MakeNeighbor() {
625 DCHECK_EQ(StartNode(kSecondPairFirstNodeDestination),
626 StartNode(kSecondPairSecondNodeDestination));
627 DCHECK_EQ(StartNode(kSecondPairFirstNode),
628 StartNode(kFirstPairFirstNodeDestination));
629 DCHECK_EQ(StartNode(kSecondPairFirstNode),
630 StartNode(kFirstPairSecondNodeDestination));
631
632 if (StartNode(kFirstPairFirstNode) == StartNode(kSecondPairFirstNode)) {
633 SetNextBaseToIncrement(kSecondPairFirstNode);
634 return false;
635 }
636 // Through this method, <base>[X][Y] represent the <base> variable for the
637 // node Y of pair X. <base> is in node, prev, dest.
638 int64_t nodes[2][2];
639 int64_t prev[2][2];
640 int64_t dest[2][2];
641 nodes[0][0] = BaseNode(kFirstPairFirstNode);
642 nodes[1][0] = BaseNode(kSecondPairFirstNode);
643 if (nodes[1][0] <= nodes[0][0]) {
644 // Exchange is symmetric.
645 SetNextBaseToIncrement(kSecondPairFirstNode);
646 return false;
647 }
648 if (!GetPreviousAndSibling(nodes[0][0], &prev[0][0], &nodes[0][1],
649 &prev[0][1])) {
650 SetNextBaseToIncrement(kFirstPairFirstNode);
651 return false;
652 }
653 if (!GetPreviousAndSibling(nodes[1][0], &prev[1][0], &nodes[1][1],
654 &prev[1][1])) {
655 SetNextBaseToIncrement(kSecondPairFirstNode);
656 return false;
657 }
658
659 if (!LoadAndCheckDest(0, 0, kFirstPairFirstNodeDestination, nodes, dest)) {
660 SetNextBaseToIncrement(kFirstPairFirstNodeDestination);
661 return false;
662 }
663 if (!LoadAndCheckDest(0, 1, kFirstPairSecondNodeDestination, nodes, dest)) {
664 SetNextBaseToIncrement(kFirstPairSecondNodeDestination);
665 return false;
666 }
667 if (StartNode(kSecondPairFirstNodeDestination) !=
668 StartNode(kFirstPairFirstNode) ||
669 !LoadAndCheckDest(1, 0, kSecondPairFirstNodeDestination, nodes, dest)) {
670 SetNextBaseToIncrement(kSecondPairFirstNodeDestination);
671 return false;
672 }
673 if (!LoadAndCheckDest(1, 1, kSecondPairSecondNodeDestination, nodes, dest)) {
674 SetNextBaseToIncrement(kSecondPairSecondNodeDestination);
675 return false;
676 }
677
678 if (!MoveNode(0, 1, nodes, dest, prev)) {
679 SetNextBaseToIncrement(kFirstPairSecondNodeDestination);
680 return false;
681 }
682 if (!MoveNode(0, 0, nodes, dest, prev)) {
683 SetNextBaseToIncrement(kFirstPairSecondNodeDestination);
684 return false;
685 }
686 if (!MoveNode(1, 1, nodes, dest, prev)) {
687 return false;
688 }
689 if (!MoveNode(1, 0, nodes, dest, prev)) {
690 return false;
691 }
692 return true;
693}

767bool SwapIndexPairOperator::MakeNextNeighbor(Assignment* delta,
768 Assignment* deltadelta) {
769 const int64_t kNoPath = -1;
770 CHECK(delta != nullptr);
771 while (true) {
772 RevertChanges(true);
773
774 if (pair_index_ < pairs_.size()) {
775 const int64_t path =
776 ignore_path_vars_ ? 0LL : Value(first_active_ + number_of_nexts_);
777 const int64_t prev_first = prevs_[first_active_];
778 const int64_t next_first = Value(first_active_);
779 // Making current active "pickup" unperformed.
780 SetNext(first_active_, first_active_, kNoPath);
781 // Inserting "pickup" alternative at the same position.
782 const auto& [pickup_alternatives, delivery_alternatives] =
783 pairs_[pair_index_];
784 const int64_t insert_first = pickup_alternatives[first_index_];
785 SetNext(prev_first, insert_first, path);
786 SetNext(insert_first, next_first, path);
787 int64_t prev_second = prevs_[second_active_];
788 if (prev_second == first_active_) {
789 prev_second = insert_first;
790 }
791 DCHECK_EQ(path, ignore_path_vars_
792 ? int64_t{0}
793 : Value(second_active_ + number_of_nexts_));
794 const int64_t next_second = Value(second_active_);
795 // Making current active "delivery" unperformed.
796 SetNext(second_active_, second_active_, kNoPath);
797 // Inserting "delivery" alternative at the same position.
798 const int64_t insert_second = delivery_alternatives[second_index_];
799 SetNext(prev_second, insert_second, path);
800 SetNext(insert_second, next_second, path);
801 // Move to next "pickup/delivery" alternative.
802 ++second_index_;
803 if (second_index_ >= delivery_alternatives.size()) {
804 second_index_ = 0;
805 ++first_index_;
806 if (first_index_ >= pickup_alternatives.size()) {
807 first_index_ = 0;
808 ++pair_index_;
809 UpdateActiveNodes();
810 }
811 }
812 } else {
813 return false;
814 }
815
816 if (ApplyChanges(delta, deltadelta)) return true;
817 }
818 return false;
819}

1053bool RelocateSubtrip::RelocateSubTripFromPickup(const int64_t chain_first_node,
1054 const int64_t insertion_node) {
1055 if (IsPathEnd(insertion_node)) return false;
1056 if (Prev(chain_first_node) == insertion_node)
1057 return false; // Skip null move.
1058
1059 int num_opened_pairs = 0;
1060 // Split chain into subtrip and rejected nodes.
1061 rejected_nodes_ = {Prev(chain_first_node)};
1062 subtrip_nodes_ = {insertion_node};
1063 int current = chain_first_node;
1064 do {
1065 if (current == insertion_node) {
1066 // opened_pairs_set_ must be all false when we leave this function.
1067 opened_pairs_set_.assign(opened_pairs_set_.size(), false);
1068 return false;
1069 }
1070 const int pair = pd_data_.GetPairOfNode(current);
1071 if (pd_data_.IsDeliveryNode(current) && !opened_pairs_set_[pair]) {
1072 rejected_nodes_.push_back(current);
1073 } else {
1074 subtrip_nodes_.push_back(current);
1075 if (pd_data_.IsPickupNode(current)) {
1076 ++num_opened_pairs;
1077 opened_pairs_set_[pair] = true;
1078 } else if (pd_data_.IsDeliveryNode(current)) {
1079 --num_opened_pairs;
1080 opened_pairs_set_[pair] = false;
1081 }
1082 }
1083 current = Next(current);
1084 } while (num_opened_pairs != 0 && !IsPathEnd(current));
1085 DCHECK_EQ(num_opened_pairs, 0);
1086 rejected_nodes_.push_back(current);
1087 subtrip_nodes_.push_back(Next(insertion_node));
1088
1089 // Set new paths.
1090 SetPath(rejected_nodes_, Path(chain_first_node));
1091 SetPath(subtrip_nodes_, Path(insertion_node));
1092 return true;
1093}

1104 rejected_nodes_ = {Next(chain_last_node)};

1203 return false;
1204 }
1205 } else {
1206 node0 = BaseNode(0);
1207 node1 = BaseNode(1);
1208 }
1209
1210 if (pd_data_.GetPairOfNode(node0) == -1) return false;
1211 if (pd_data_.GetPairOfNode(node1) == -1) return false;
1212 // Break symmetry: a move generated from (node0, node1) is the
1213 // same as from (node1, node0): no need to do it twice.
1214 if (node0 >= node1) return false;
1215 rejects0_.clear();
1216 subtrip0_.clear();
1217 if (!ExtractChainsAndCheckCanonical(node0, &rejects0_, &subtrip0_)) {
1218 return false;
1219 }
1220 rejects1_.clear();
1221 subtrip1_.clear();
1222 if (!ExtractChainsAndCheckCanonical(node1, &rejects1_, &subtrip1_)) {
1223 return false;
1224 }
1225
1226 // If paths intersect, skip the move.
1227 if (HasNeighbors() || StartNode(0) == StartNode(1)) {
1228 if (VectorContains(rejects0_, subtrip1_.front())) return false;
1229 if (VectorContains(rejects1_, subtrip0_.front())) return false;
1230 if (VectorContains(subtrip0_, subtrip1_.front())) return false;
1231 if (VectorContains(subtrip1_, subtrip0_.front())) return false;
1232 }
1233
1234 // Assemble the new paths.
1235 path0_ = {Prev(subtrip0_.front())};
1236 path1_ = {Prev(subtrip1_.front())};
1237 const int64_t last0 = Next(subtrip0_.back());
1238 const int64_t last1 = Next(subtrip1_.back());
1239 const bool concatenated01 = last0 == subtrip1_.front();
1240 const bool concatenated10 = last1 == subtrip0_.front();
1241
1242 if (pd_data_.IsDeliveryNode(node0)) std::swap(subtrip1_, rejects0_);
1243 path0_.insert(path0_.end(), subtrip1_.begin(), subtrip1_.end());
1244 path0_.insert(path0_.end(), rejects0_.begin(), rejects0_.end());
1245 path0_.push_back(last0);
1246
1247 if (pd_data_.IsDeliveryNode(node1)) std::swap(subtrip0_, rejects1_);
1248 path1_.insert(path1_.end(), subtrip0_.begin(), subtrip0_.end());
1249 path1_.insert(path1_.end(), rejects1_.begin(), rejects1_.end());
1250 path1_.push_back(last1);
1251
1252 // When the trips are concatenated, bypass the regular extremities.
1253 if (concatenated01) {
1254 path0_.pop_back();
1255 path1_.front() = path0_.back();
1256 } else if (concatenated10) {
1257 path1_.pop_back();
1258 path0_.front() = path1_.back();
1259 }
1260
1261 // Change the paths. Since SetNext() modifies Path() values,
1262 // record path_id0 and path_id11 before calling SetPath();
1263 const int64_t path0_id = Path(node0);
1264 const int64_t path1_id = Path(node1);
1265 SetPath(path0_, path0_id);
1266 SetPath(path1_, path1_id);
1267 return true;
1268}
