77 // Returns the index of a node. A node can correspond to multiple indices if
78 // it's a start or end node. As of 03/2020, kUnassigned will be returned for
79 // all end nodes. If a node appears more than once as a start node, the index
80 // of the first node in the list of start nodes is returned.
81 int64_t NodeToIndex(NodeIndex node) const {
82 DCHECK_GE(node.value(), 0);
83 DCHECK_LT(node.value(), node_to_index_.size());
84 return node_to_index_[node];
85 }
86 // Same as NodeToIndex but for a given vector of nodes.
87 std::vector<int64_t> NodesToIndices(
88 const std::vector<NodeIndex>& nodes) const;
89 // Returns the node corresponding to an index. A node may appear more than
90 // once if it is used as the start or the end node of multiple vehicles.
91 NodeIndex IndexToNode(int64_t index) const {
92 DCHECK_GE(index, 0);
93 DCHECK_LT(index, index_to_node_.size());
94 return index_to_node_[index];
95 }
96 // Same as IndexToNode but for a given vector of indices.
97 std::vector<NodeIndex> IndicesToNodes(
98 const std::vector<int64_t>& indices) const;
