60absl::Status StringToProto(absl::string_view data,
61 google::protobuf::Message* proto,
62 bool allow_partial) {
63 // Try decompressing it.
64 std::string uncompressed;
65 if (GunzipString(data, &uncompressed)) {
66 VLOG(1) << "ReadFileToProto(): input is gzipped";
67 data = uncompressed;
68 }
69 // Try binary format first, then text format, then JSON, then proto3 JSON,
70 // then give up.
71 // For some of those, like binary format and proto3 JSON, we perform
72 // additional checks to verify that we have the right proto: it can happen
73 // to try to read a proto of type Foo as a proto of type Bar, by mistake, and
74 // we'd rather have this function fail rather than silently accept it, because
75 // the proto parser is too lenient with unknown fields.
76 // We don't require ByteSizeLong(parsed) == input.size(), because it may be
77 // the case that the proto version changed and some fields are dropped.
78 // We just fail when the difference is too large.
79 constexpr double kMaxBinaryProtoParseShrinkFactor = 2;
80 std::string binary_format_error;
81 if (proto->ParsePartialFromString(data) &&
82 (allow_partial || proto->IsInitialized())) {
83 // NOTE(user): When using ParseFromString() from a generic
84 // google::protobuf::Message, like we do here, all fields are stored, even
85 // if they are unknown in the underlying proto type. Unless we explicitly
86 // discard those 'unknown fields' here, our call to ByteSizeLong() will
87 // still count the unknown payload.
88 proto->DiscardUnknownFields();
89 if (proto->ByteSizeLong() <
90 data.size() / kMaxBinaryProtoParseShrinkFactor) {
91 binary_format_error =
92 "The input may be a binary protobuf payload, but it"
93 " probably is from a different proto class";
94 } else {
95 VLOG(1) << "StringToProto(): input seems to be a binary proto";
96 return absl::OkStatus();
97 }
98 }
99
100 struct : public google::protobuf::io::ErrorCollector {
101 void RecordError(int line, google::protobuf::io::ColumnNumber column,
102 absl::string_view error_message) override {
103 if (!message.empty()) message += ", ";
104 absl::StrAppendFormat(&message, "%d:%d: %s",
105 // We convert the 0-indexed line to 1-indexed.
106 line + 1, column, error_message);
107 }
108 std::string message;
109 } text_format_error;
110 google::protobuf::TextFormat::Parser text_parser;
111 text_parser.RecordErrorsTo(&text_format_error);
112 text_parser.AllowPartialMessage(allow_partial);
113 if (text_parser.ParseFromString(data, proto)) {
114 VLOG(1) << "StringToProto(): input is a text proto";
115 return absl::OkStatus();
116 }
117 std::string json_error;
118 absl::Status json_status =
119 JsonStringToMessage(data, proto, JsonParseOptions());
120 if (json_status.ok()) {
121 // NOTE(user): We protect against the JSON proto3 parser being very lenient
122 // and easily accepting any JSON as a valid JSON for our proto: if the
123 // parsed proto's size is too small compared to the JSON, we probably parsed
124 // a JSON that wasn't representing a valid proto.
125 constexpr int kMaxJsonToBinaryShrinkFactor = 30;
126 if (proto->ByteSizeLong() < data.size() / kMaxJsonToBinaryShrinkFactor) {
127 json_status = absl::InvalidArgumentError(
128 "The input looks like valid JSON, but probably not"
129 " of the right proto class");
130 } else {
131 VLOG(1) << "StringToProto(): input is a proto JSON";
132 return absl::OkStatus();
133 }
134 }
135 return absl::InvalidArgumentError(absl::StrFormat(
136 "binary format error: '%s', text format error: '%s', json error: '%s'",
137 binary_format_error, text_format_error.message, json_status.message()));
138}

140absl::Status WriteProtoToFile(absl::string_view filename,
141 const google::protobuf::Message& proto,
142 ProtoWriteFormat proto_write_format, bool gzipped,
143 bool append_extension_to_file_name) {
144 std::string file_type_suffix;
145 std::string output_string;
146 google::protobuf::io::StringOutputStream stream(&output_string);
147 auto make_error = [filename](absl::string_view error_message) {
148 return absl::InternalError(absl::StrFormat(
149 "WriteProtoToFile('%s') failed: %s", filename, error_message));
150 };
151 switch (proto_write_format) {
152 case ProtoWriteFormat::kProtoBinary:
153 if (!proto.SerializeToZeroCopyStream(&stream)) {
154 return make_error("SerializeToZeroCopyStream()");
155 }
156 file_type_suffix = ".bin";
157 break;
158 case ProtoWriteFormat::kProtoText:
159 if (!google::protobuf::TextFormat::PrintToString(proto, &output_string)) {
160 return make_error("TextFormat::PrintToString()");
161 }
162 break;
163 case ProtoWriteFormat::kJson: {
164 google::protobuf::util::JsonPrintOptions options;
165 options.add_whitespace = true;
166 options.always_print_primitive_fields = true;
167 options.preserve_proto_field_names = true;
168 if (!google::protobuf::util::MessageToJsonString(proto, &output_string,
169 options)
170 .ok()) {
171 LOG(WARNING) << "Printing to stream failed.";
172 return make_error("google::protobuf::util::MessageToJsonString()");
173 }
174 file_type_suffix = ".json";
175 break;
176 }
177 case ProtoWriteFormat::kCanonicalJson:
178 google::protobuf::util::JsonPrintOptions options;
179 options.add_whitespace = true;
180 if (!google::protobuf::util::MessageToJsonString(proto, &output_string,
181 options)
182 .ok()) {
183 LOG(WARNING) << "Printing to stream failed.";
184 return make_error("google::protobuf::util::MessageToJsonString()");
185 }
186 file_type_suffix = ".json";
187 break;
188 }
189 if (gzipped) {
190 std::string gzip_string;
191 GzipString(output_string, &gzip_string);
192 output_string.swap(gzip_string);
193 file_type_suffix += ".gz";
194 }
195 std::string output_filename(filename);
196 if (append_extension_to_file_name) output_filename += file_type_suffix;
197 VLOG(1) << "Writing " << output_string.size() << " bytes to '"
198 << output_filename << "'";
199 return file::SetContents(output_filename, output_string, file::Defaults());
200}
