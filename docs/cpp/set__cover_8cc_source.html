71void GreedySolutionGenerator::UpdatePriorities(
72 const std::vector<SubsetIndex>& impacted_subsets) {
73 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
74 for (const SubsetIndex subset : impacted_subsets) {
75 const ElementIndex marginal_impact(ledger_->marginal_impacts(subset));
76 if (marginal_impact != 0) {
77 const Cost marginal_cost_increase =
78 subset_costs[subset] / marginal_impact.value();
79 pq_.ChangePriority(subset, -marginal_cost_increase);
80 } else {
81 pq_.Remove(subset);
82 }
83 }
84}

90bool GreedySolutionGenerator::NextSolution(
91 const std::vector<SubsetIndex>& focus) {
92 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
93 ledger_->MakeDataConsistent();
94
95 // The priority is the minimum marginal cost increase. Since the
96 // priority queue returns the smallest value, we use the opposite.
97 for (const SubsetIndex subset : focus) {
98 if (!ledger_->is_selected(subset) &&
99 ledger_->marginal_impacts(subset) != 0) {
100 const Cost marginal_cost_increase =
101 subset_costs[subset] / ledger_->marginal_impacts(subset).value();
102 pq_.Add(subset, -marginal_cost_increase);
103 }
104 }
105 const ElementIndex num_elements(ledger_->model()->num_elements());
106 ElementIndex num_elements_covered(ledger_->num_elements_covered());
107 while (num_elements_covered < num_elements && !pq_.IsEmpty()) {
108 const SubsetIndex best_subset = pq_.TopSubset();
109 DVLOG(1) << "Best subset: " << best_subset.value()
110 << " Priority = " << pq_.Priority(best_subset)
111 << " queue size = " << pq_.Size();
112 const std::vector<SubsetIndex> impacted_subsets =
113 ledger_->Toggle(best_subset, true);
114 UpdatePriorities(impacted_subsets);
115 num_elements_covered = ledger_->num_elements_covered();
116 DVLOG(1) << "Cost = " << ledger_->cost() << " num_uncovered_elements = "
117 << num_elements - num_elements_covered;
118 }
119 DCHECK(pq_.IsEmpty());
120 DCHECK(ledger_->CheckConsistency());
121 DCHECK(ledger_->CheckSolution());
122 return true;
123}

140bool SteepestSearch::NextSolution(const std::vector<SubsetIndex>& focus,
141 int num_iterations) {
142 // Return false if ledger_ contains no solution.
143 if (!ledger_->CheckSolution()) return false;
144 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
145 // Create priority queue with cost of using a subset, by decreasing order.
146 // Do it only for removable subsets.
147 for (const SubsetIndex subset : focus) {
148 // The priority is the gain from removing the subset from the solution.
149 if (ledger_->is_selected(subset) && ledger_->is_removable(subset)) {
150 pq_.Add(subset, subset_costs[subset]);
151 }
152 }
153 for (int iteration = 0; iteration < num_iterations && !pq_.IsEmpty();
154 ++iteration) {
155 const SubsetIndex best_subset = pq_.TopSubset();
156 const Cost cost_decrease = subset_costs[best_subset];
157 DCHECK_GT(cost_decrease, 0.0);
158 DCHECK(ledger_->is_removable(best_subset));
159 DCHECK(ledger_->is_selected(best_subset));
160 const std::vector<SubsetIndex> impacted_subsets =
161 ledger_->Toggle(best_subset, false);
162 UpdatePriorities(impacted_subsets);
163 DVLOG(1) << "Cost = " << ledger_->cost();
164 }
165 DCHECK(ledger_->CheckConsistency());
166 DCHECK(ledger_->CheckSolution());
167 return true;
168}

187void GuidedTabuSearch::UpdatePenalties(const std::vector<SubsetIndex>& focus) {
188 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
189 Cost max_utility = -1.0;
190 for (const SubsetIndex subset : focus) {
191 if (ledger_->is_selected(subset)) {
192 max_utility = std::max(max_utility, utilities_[subset]);
193 }
194 }
195 const double epsilon_utility = epsilon_ * max_utility;
196 for (const SubsetIndex subset : focus) {
197 if (ledger_->is_selected(subset)) {
198 const double utility = utilities_[subset];
199 if ((max_utility - utility <= epsilon_utility) && FlipCoin()) {
200 ++times_penalized_[subset];
201 const int times_penalized = times_penalized_[subset];
202 const Cost cost =
203 subset_costs[subset]; // / columns[subset].size().value();
204 utilities_[subset] = cost / (1 + times_penalized);
205 augmented_costs_[subset] =
206 cost * (1 + penalty_factor_ * times_penalized);
207 }
208 }
209 }
210}

216bool GuidedTabuSearch::NextSolution(const std::vector<SubsetIndex>& focus,
217 int num_iterations) {
218 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
219 constexpr Cost kMaxPossibleCost = std::numeric_limits<Cost>::max();
220 Cost best_cost = ledger_->cost();
221 SubsetBoolVector best_choices = ledger_->GetSolution();
222 Cost augmented_cost =
223 std::accumulate(augmented_costs_.begin(), augmented_costs_.end(), 0.0);
224 for (int iteration = 0; iteration < num_iterations; ++iteration) {
225 Cost best_delta = kMaxPossibleCost;
226 SubsetIndex best_subset = kNotFound;
227 for (const SubsetIndex subset : focus) {
228 const Cost delta = augmented_costs_[subset];
229 DVLOG(1) << "Subset, " << subset.value() << ", at ,"
230 << ledger_->is_selected(subset) << ", is removable =, "
231 << ledger_->is_removable(subset) << ", delta =, " << delta
232 << ", best_delta =, " << best_delta;
233 if (ledger_->is_selected(subset)) {
234 // Try to remove subset from solution, if the gain from removing is
235 // worth it:
236 if (-delta < best_delta &&
237 // and it can be removed, and
238 ledger_->is_removable(subset) &&
239 // it is not Tabu OR decreases the actual cost (aspiration):
240 (!tabu_list_.Contains(subset) ||
241 ledger_->cost() - subset_costs[subset] < best_cost)) {
242 best_delta = -delta;
243 best_subset = subset;
244 }
245 } else {
246 // Try to use subset in solution, if its penalized delta is good.
247 if (delta < best_delta) {
248 // The limit kMaxPossibleCost is ill-defined,
249 // there is always a best_subset. Is it intended?
250 if (!tabu_list_.Contains(subset)) {
251 best_delta = delta;
252 best_subset = subset;
253 }
254 }
255 }
256 }
257 if (best_subset == kNotFound) { // Local minimum reached.
258 ledger_->LoadSolution(best_choices);
259 return true;
260 }
261 DVLOG(1) << "Best subset, " << best_subset.value() << ", at ,"
262 << ledger_->is_selected(best_subset) << ", is removable = ,"
263 << ledger_->is_removable(best_subset) << ", best_delta = ,"
264 << best_delta;
265
266 UpdatePenalties(focus);
267 tabu_list_.Add(best_subset);
268 const std::vector<SubsetIndex> impacted_subsets =
269 ledger_->UnsafeToggle(best_subset, !ledger_->is_selected(best_subset));
270 // TODO(user): make the cost computation incremental.
271 augmented_cost =
272 std::accumulate(augmented_costs_.begin(), augmented_costs_.end(), 0.0);
273
274 DVLOG(1) << "Iteration, " << iteration << ", current cost = ,"
275 << ledger_->cost() << ", best cost = ," << best_cost
276 << ", penalized cost = ," << augmented_cost;
277 if (ledger_->cost() < best_cost) {
278 LOG(INFO) << "Updated best cost, "
279 << "Iteration, " << iteration << ", current cost = ,"
280 << ledger_->cost() << ", best cost = ," << best_cost
281 << ", penalized cost = ," << augmented_cost;
282 best_cost = ledger_->cost();
283 best_choices = ledger_->GetSolution();
284 }
285 }
286 ledger_->LoadSolution(best_choices);
287 DCHECK(ledger_->CheckConsistency());
288 DCHECK(ledger_->CheckSolution());
289 return true;
290}

298std::vector<SubsetIndex> ClearProportionRandomly(
299 const std::vector<SubsetIndex>& focus, double proportion,
300 SetCoverLedger* ledger) {
301 CHECK_LE(proportion, 1.0);
302 CHECK_GE(proportion, 0.0);
303 std::vector<SubsetIndex> choice_indices;
304 for (const SubsetIndex subset : focus) {
305 if (ledger->is_selected(subset)) {
306 choice_indices.push_back(subset);
307 }
308 }
309 std::shuffle(choice_indices.begin(), choice_indices.end(), absl::BitGen());
310 const int num_subsets_to_clear = proportion * choice_indices.size();
311 choice_indices.resize(num_subsets_to_clear);
312 for (const SubsetIndex subset : choice_indices) {
313 // Use UnsafeToggle because we allow non-solutions.
314 ledger->UnsafeToggle(subset, false);
315 }
316 return choice_indices;
317}
