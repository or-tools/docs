41bool RandomSolutionGenerator::NextSolution() {
42 const SubsetIndex num_subsets(ledger_->model()->num_subsets());
43 std::vector<SubsetIndex> random_subset_order(num_subsets.value());
44 std::iota(random_subset_order.begin(), random_subset_order.end(),
45 SubsetIndex(0));
46 std::shuffle(random_subset_order.begin(), random_subset_order.end(),
47 absl::BitGen());
48 const ElementIndex num_elements(ledger_->model()->num_elements());
49 for (const SubsetIndex subset : random_subset_order) {
50 if (ledger_->num_elements_covered() == num_elements) {
51 break;
52 }
53 if (ledger_->marginal_impacts(subset) != 0) {
54 ledger_->Toggle(subset, true);
55 }
56 }
57 DCHECK(ledger_->CheckConsistency());
58 return true;
59}

63void GreedySolutionGenerator::UpdatePriorities(
64 const std::vector<SubsetIndex>& impacted_subsets) {
65 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
66 for (const SubsetIndex subset : impacted_subsets) {
67 const ElementIndex marginal_impact(ledger_->marginal_impacts(subset));
68 if (marginal_impact != 0) {
69 const Cost marginal_cost_increase =
70 subset_costs[subset] / marginal_impact.value();
71 pq_.ChangePriority(subset, -marginal_cost_increase);
72 } else {
73 pq_.Remove(subset);
74 }
75 }
76}

78bool GreedySolutionGenerator::NextSolution() {
79 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
80 const SubsetIndex num_subsets(ledger_->model()->num_subsets());
81 ledger_->MakeDataConsistent();
82
83 // The priority is the minimum marginal cost increase. Since the
84 // priority queue returns the smallest value, we use the opposite.
85 for (SubsetIndex subset(0); subset < num_subsets; ++subset) {
86 if (!ledger_->is_selected(subset) &&
87 ledger_->marginal_impacts(subset) != 0) {
88 const Cost marginal_cost_increase =
89 subset_costs[subset] / ledger_->marginal_impacts(subset).value();
90 pq_.Add(subset, -marginal_cost_increase);
91 }
92 }
93 const ElementIndex num_elements(ledger_->model()->num_elements());
94 ElementIndex num_elements_covered(ledger_->num_elements_covered());
95 while (num_elements_covered < num_elements && !pq_.IsEmpty()) {
96 const SubsetIndex best_subset = pq_.TopSubset();
97 DVLOG(1) << "Best subset: " << best_subset.value()
98 << " Priority = " << pq_.Priority(best_subset)
99 << " queue size = " << pq_.Size();
100 const std::vector<SubsetIndex> impacted_subsets =
101 ledger_->Toggle(best_subset, true);
102 UpdatePriorities(impacted_subsets);
103 num_elements_covered = ledger_->num_elements_covered();
104 DVLOG(1) << "Cost = " << ledger_->cost() << " num_uncovered_elements = "
105 << num_elements - num_elements_covered;
106 }
107 DCHECK(pq_.IsEmpty());
108 DCHECK(ledger_->CheckConsistency());
109 DCHECK(ledger_->CheckSolution());
110 return true;
111}

124bool SteepestSearch::NextSolution(int num_iterations) {
125 // Return false if ledger_ contains no solution.
126 if (!ledger_->CheckSolution()) return false;
127 const SparseColumnView& columns = ledger_->model()->columns();
128 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
129 // Create priority queue with cost of using a subset, by decreasing order.
130 // Do it only for removable subsets.
131 for (SubsetIndex subset(0); subset < columns.size(); ++subset) {
132 // The priority is the gain from removing the subset from the solution.
133 if (ledger_->is_selected(subset) && ledger_->is_removable(subset)) {
134 pq_.Add(subset, subset_costs[subset]);
135 }
136 }
137 for (int iteration = 0; iteration < num_iterations && !pq_.IsEmpty();
138 ++iteration) {
139 const SubsetIndex best_subset = pq_.TopSubset();
140 const Cost cost_decrease = subset_costs[best_subset];
141 DCHECK_GT(cost_decrease, 0.0);
142 DCHECK(ledger_->is_removable(best_subset));
143 DCHECK(ledger_->is_selected(best_subset));
144 const std::vector<SubsetIndex> impacted_subsets =
145 ledger_->Toggle(best_subset, false);
146 UpdatePriorities(impacted_subsets);
147 DVLOG(1) << "Cost = " << ledger_->cost();
148 }
149 DCHECK(ledger_->CheckConsistency());
150 DCHECK(ledger_->CheckSolution());
151 return true;
152}

159 times_penalized_.AssignToZero(columns.size());

163 gts_priorities_[subset] /= columns[subset].size().value();

174void GuidedTabuSearch::UpdatePenalties(
175 const std::vector<SubsetIndex>& impacted_subsets) {
176 const SparseColumnView& columns = ledger_->model()->columns();
177 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
178 const ElementIndex num_elements(ledger_->model()->num_elements());
179 Cost largest_priority = -1.0;
180 for (SubsetIndex subset(0); subset < columns.size(); ++subset) {
181 if (ledger_->is_selected(subset)) {
182 const ElementIndex num_elements_already_covered =
183 num_elements - ledger_->marginal_impacts(subset);
184 largest_priority = std::max(largest_priority, gts_priorities_[subset]) /
185 num_elements_already_covered.value();
186 }
187 }
188 const double radius = radius_factor_ * largest_priority;
189 for (SubsetIndex subset(0); subset < columns.size(); ++subset) {
190 if (ledger_->is_selected(subset)) {
191 const double subset_priority = gts_priorities_[subset];
192 if ((largest_priority - subset_priority <= radius) && FlipCoin()) {
193 ++times_penalized_[subset];
194 const int times_penalized = times_penalized_[subset];
195 const Cost cost = subset_costs[subset] / columns[subset].size().value();
196 gts_priorities_[subset] = cost / (1 + times_penalized);
197 penalized_costs_[subset] =
198 cost * (1 + penalty_factor_ * times_penalized);
199 }
200 }
201 }

204bool GuidedTabuSearch::NextSolution(int num_iterations) {
205 const SparseColumnView& columns = ledger_->model()->columns();
206 const SubsetCostVector& subset_costs = ledger_->model()->subset_costs();
207 constexpr Cost kMaxPossibleCost = std::numeric_limits<Cost>::max();
208 Cost best_cost = ledger_->cost();
209 Cost total_pen_cost = 0;
210 for (const Cost pen_cost : penalized_costs_) {
211 total_pen_cost += pen_cost;
212 }
213 SubsetBoolVector best_choices = ledger_->GetSolution();
214 for (int iteration = 0; iteration < num_iterations; ++iteration) {
215 Cost smallest_penalized_cost_increase = kMaxPossibleCost;
216 SubsetIndex best_subset = kNotFound;
217 for (SubsetIndex subset(0); subset < columns.size(); ++subset) {
218 const Cost penalized_delta = penalized_costs_[subset];
219 DVLOG(1) << "Subset: " << subset.value() << " at "
220 << ledger_->is_selected(subset)
221 << " is removable = " << ledger_->is_removable(subset)
222 << " penalized_delta = " << penalized_delta
223 << " smallest_penalized_cost_increase = "
224 << smallest_penalized_cost_increase;
225 if (!ledger_->is_selected(subset)) {
226 // Try to use subset in solution, if its penalized delta is good.
227 if (penalized_delta < smallest_penalized_cost_increase) {
228 smallest_penalized_cost_increase = penalized_delta;
229 best_subset = subset;
230 }
231 } else {
232 // Try to remove subset from solution, if the gain from removing, is
233 // OK:
234 if (-penalized_delta < smallest_penalized_cost_increase &&
235 // and it can be removed, and
236 ledger_->is_removable(subset) &&
237 // it is not Tabu OR decreases the actual cost:
238 (!tabu_list_.Contains(subset) ||
239 ledger_->cost() - subset_costs[subset] < best_cost)) {
240 smallest_penalized_cost_increase = -penalized_delta;
241 best_subset = subset;
242 }
243 }
244 }
245 if (best_subset == kNotFound) { // Local minimum reached.
246 ledger_->LoadSolution(best_choices);
247 return true;
248 }
249 total_pen_cost += smallest_penalized_cost_increase;
250 const std::vector<SubsetIndex> impacted_subsets =
251 ledger_->Toggle(best_subset, !ledger_->is_selected(best_subset));
252 UpdatePenalties(impacted_subsets);
253 tabu_list_.Add(best_subset);
254 if (ledger_->cost() < best_cost) {
255 LOG(INFO) << "Iteration:" << iteration
256 << ", current cost = " << ledger_->cost()
257 << ", best cost = " << best_cost
258 << ", penalized cost = " << total_pen_cost;
259 best_cost = ledger_->cost();
260 best_choices = ledger_->GetSolution();
261 }
262 }
263 ledger_->LoadSolution(best_choices);
264 DCHECK(ledger_->CheckConsistency());
265 DCHECK(ledger_->CheckSolution());
266 return true;
267}
