142template <typename K>
143class IdMapMatcher : public MatcherInterface<IdMap<K, double>> {
144 public:
145 IdMapMatcher(IdMap<K, double> expected, const bool all_keys,
146 const double tolerance)
147 : expected_(std::move(expected)),
148 all_keys_(all_keys),
149 tolerance_(tolerance) {
150 for (const auto [k, v] : expected_) {
151 CHECK(!std::isnan(v)) << "Illegal NaN for key: " << k;
152 }
153 }
154
155 bool MatchAndExplain(IdMap<K, double> actual,
156 MatchResultListener* const os) const override {
157 for (const auto& [key, value] : expected_) {
158 if (!actual.contains(key)) {
159 *os << "expected key " << key << " not found";
160 return false;
161 }
162 if (!(std::abs(value - actual.at(key)) <= tolerance_)) {
163 *os << "value for key " << key
164 << " not within tolerance, expected: " << value
165 << " but found: " << actual.at(key);
166 return false;
167 }
168 }
169 // Post condition: expected_ is a subset of actual.
170 if (all_keys_ && expected_.size() != actual.size()) {
171 for (const auto& [key, value] : actual) {
172 if (!expected_.contains(key)) {
173 *os << "found unexpected key " << key << " in actual";
174 return false;
175 }
176 }
177 // expected_ subset of actual && expected_.size() != actual.size() implies
178 // that there is a member A of actual not in expected. When the loop above
179 // hits A, it will return, thus this line is unreachable.
180 LOG(FATAL) << "unreachable";
181 }
182 return true;
183 }
184
185 void DescribeTo(std::ostream* const os) const override {
186 if (all_keys_) {
187 *os << "has identical keys to ";
188 } else {
189 *os << "keys are contained in ";
190 }
191 PrintTo(expected_, os);
192 *os << " and values within " << tolerance_;
193 }
194
195 void DescribeNegationTo(std::ostream* const os) const override {
196 if (all_keys_) {
197 *os << "either keys differ from ";
198 } else {
199 *os << "either has a key not in ";
200 }
201 PrintTo(expected_, os);
202 *os << " or a value differs by more than " << tolerance_;
203 }
204
205 private:
206 const IdMap<K, double> expected_;
207 const bool all_keys_;
208 const double tolerance_;
209};

287template <typename RayType>
288class RayMatcher : public MatcherInterface<RayType> {
289 public:
290 RayMatcher(RayType expected, const double tolerance)
291 : expected_(std::move(expected)), tolerance_(tolerance) {}
292 void DescribeTo(std::ostream* os) const final {
293 *os << "after L_inf normalization, is within tolerance: " << tolerance_
294 << " of expected: ";
295 PrintTo(expected_, os);
296 }
297 void DescribeNegationTo(std::ostream* const os) const final {
298 *os << "after L_inf normalization, is not within tolerance: " << tolerance_
299 << " of expected: ";
300 PrintTo(expected_, os);
301 }
302
303 protected:
304 const RayType expected_;
305 const double tolerance_;
306};

405template <typename K>
406double InfinityNorm(const IdMap<K, double>& vector) {
407 double infinity_norm = 0.0;
408 for (auto [id, value] : vector) {
409 infinity_norm = std::max(infinity_norm, std::abs(value));
410 }
411 return infinity_norm;
412}
413
414// Returns a normalized primal ray.
415//
416// The normalization is done using infinity norm:
417//
418// ray / ||ray||_inf
419//
420// If the input ray norm is zero, the ray is returned unchanged.
421PrimalRay NormalizePrimalRay(PrimalRay ray) {
422 const double norm = InfinityNorm(ray.variable_values);
423 if (norm != 0.0) {
424 for (auto entry : ray.variable_values) {
425 entry.second /= norm;
426 }
427 }
428 return ray;
429}

431class PrimalRayMatcher : public RayMatcher<PrimalRay> {
432 public:
433 PrimalRayMatcher(PrimalRay expected, const double tolerance)
434 : RayMatcher(std::move(expected), tolerance) {}
435
436 bool MatchAndExplain(PrimalRay actual,
437 MatchResultListener* const os) const override {
438 auto normalized_actual = NormalizePrimalRay(actual);
439 auto normalized_expected = NormalizePrimalRay(expected_);
440 if (os->IsInterested()) {
441 *os << "actual normalized: " << PrintToString(normalized_actual)
442 << ", expected normalized: " << PrintToString(normalized_expected);
443 }
444 return ExplainMatchResult(
445 IsNear(normalized_expected.variable_values, tolerance_),
446 normalized_actual.variable_values, os);
447 }
448};

470// Returns a normalized dual ray.
471//
472// The normalization is done using infinity norm:
473//
474// ray / ||ray||_inf
475//
476// If the input ray norm is zero, the ray is returned unchanged.
477DualRay NormalizeDualRay(DualRay ray) {
478 const double norm =
479 std::max(InfinityNorm(ray.dual_values), InfinityNorm(ray.reduced_costs));
480 if (norm != 0.0) {
481 for (auto entry : ray.dual_values) {
482 entry.second /= norm;
483 }
484 for (auto entry : ray.reduced_costs) {
485 entry.second /= norm;
486 }
487 }
488 return ray;
489}
490
491class DualRayMatcher : public RayMatcher<DualRay> {
492 public:
493 DualRayMatcher(DualRay expected, const double tolerance)
494 : RayMatcher(std::move(expected), tolerance) {}
495
496 bool MatchAndExplain(DualRay actual, MatchResultListener* os) const override {
497 auto normalized_actual = NormalizeDualRay(actual);
498 auto normalized_expected = NormalizeDualRay(expected_);
499 if (os->IsInterested()) {
500 *os << "actual normalized: " << PrintToString(normalized_actual)
501 << ", expected normalized: " << PrintToString(normalized_expected);
502 }
503 return ExplainMatchResult(
504 IsNear(normalized_expected.dual_values, tolerance_),
505 normalized_actual.dual_values, os) &&
506 ExplainMatchResult(
507 IsNear(normalized_expected.reduced_costs, tolerance_),
508 normalized_actual.reduced_costs, os);
509 }
510};
511
512} // namespace

572template <typename MatcherType>
573std::string MatcherToStringImpl(const MatcherType& matcher, const bool negate) {
574 std::ostringstream os;
575 if (negate) {
576 matcher.DescribeNegationTo(&os);
577 } else {
578 matcher.DescribeTo(&os);
579 }
580 return os.str();
581}

584std::string MatcherToString(const Matcher<T>& matcher, bool negate) {
585 return MatcherToStringImpl(matcher, negate);

688bool MightTerminateWithRays(const TerminationReason reason) {
689 switch (reason) {
690 case TerminationReason::kInfeasibleOrUnbounded:
691 case TerminationReason::kUnbounded:
692 case TerminationReason::kInfeasible:
693 return true;
694 default:
695 return false;
696 }

699std::vector<TerminationReason> CompatibleReasons(
700 const TerminationReason expected, const bool inf_or_unb_soft_match) {
701 if (!inf_or_unb_soft_match) {
702 return {expected};
703 }
704 switch (expected) {
705 case TerminationReason::kUnbounded:
706 return {TerminationReason::kUnbounded,
707 TerminationReason::kInfeasibleOrUnbounded};
708 case TerminationReason::kInfeasible:
709 return {TerminationReason::kInfeasible,
710 TerminationReason::kInfeasibleOrUnbounded};
711 case TerminationReason::kInfeasibleOrUnbounded:
712 return {TerminationReason::kUnbounded, TerminationReason::kInfeasible,
713 TerminationReason::kInfeasibleOrUnbounded};
714 default:
715 return {expected};
716 }
717}

719Matcher<std::vector<Solution>> CheckSolutions(
720 const std::vector<Solution>& expected_solutions,
721 const SolveResultMatcherOptions& options) {
722 if (options.first_solution_only && !expected_solutions.empty()) {
723 return FirstElementIs(
724 IsNear(expected_solutions[0],
725 SolutionMatcherOptions{.tolerance = options.tolerance,
726 .check_primal = true,
727 .check_dual = options.check_dual,
728 .check_basis = options.check_basis}));
729 }
730 return IsNear(expected_solutions,
731 SolutionMatcherOptions{.tolerance = options.tolerance,

772Matcher<SolveResult> IsConsistentWith(
773 const SolveResult& expected, const SolveResultMatcherOptions& options) {
774 std::vector<Matcher<SolveResult>> to_check;
775 to_check.push_back(TerminatesWithOneOf(CompatibleReasons(
776 expected.termination.reason, options.inf_or_unb_soft_match)));
777 const bool skip_solution =
778 MightTerminateWithRays(expected.termination.reason) &&
779 !options.check_solutions_if_inf_or_unbounded;
780 if (!skip_solution) {
781 to_check.push_back(Field("solutions", &SolveResult::solutions,
782 CheckSolutions(expected.solutions, options)));
783 }
784 if (options.check_rays) {
785 to_check.push_back(Field("primal_rays", &SolveResult::primal_rays,
786 CheckRays(expected.primal_rays, options.tolerance,
787 !options.first_solution_only)));
788 to_check.push_back(Field("dual_rays", &SolveResult::dual_rays,
789 CheckRays(expected.dual_rays, options.tolerance,
790 !options.first_solution_only)));
791 }
792
793 return AllOfArray(to_check);
794}
