137template <typename K>
138class MapToDoubleMatcher
139 : public MatcherInterface<absl::flat_hash_map<K, double>> {
140 public:
141 MapToDoubleMatcher(absl::flat_hash_map<K, double> expected,
142 const bool all_keys, const double tolerance)
143 : expected_(std::move(expected)),
144 all_keys_(all_keys),
145 tolerance_(tolerance) {
146 for (const auto [k, v] : expected_) {
147 CHECK(!std::isnan(v)) << "Illegal NaN for key: " << k;
148 }
149 }
150
151 bool MatchAndExplain(absl::flat_hash_map<K, double> actual,
152 MatchResultListener* const os) const override {
153 for (const auto& [key, value] : expected_) {
154 if (!actual.contains(key)) {
155 *os << "expected key " << key << " not found";
156 return false;
157 }
158 if (!(std::abs(value - actual.at(key)) <= tolerance_)) {
159 *os << "value for key " << key
160 << " not within tolerance, expected: " << value
161 << " but found: " << actual.at(key);
162 return false;
163 }
164 }
165 // Post condition: expected_ is a subset of actual.
166 if (all_keys_ && expected_.size() != actual.size()) {
167 for (const auto& [key, value] : actual) {
168 if (!expected_.contains(key)) {
169 *os << "found unexpected key " << key << " in actual";
170 return false;
171 }
172 }
173 // expected_ subset of actual && expected_.size() != actual.size() implies
174 // that there is a member A of actual not in expected. When the loop above
175 // hits A, it will return, thus this line is unreachable.
176 LOG(FATAL) << "unreachable";
177 }
178 return true;
179 }
180
181 void DescribeTo(std::ostream* const os) const override {
182 if (all_keys_) {
183 *os << "has identical keys to ";
184 } else {
185 *os << "keys are contained in ";
186 }
187 PrintTo(expected_, os);
188 *os << " and values within " << tolerance_;
189 }
190
191 void DescribeNegationTo(std::ostream* const os) const override {
192 if (all_keys_) {
193 *os << "either keys differ from ";
194 } else {
195 *os << "either has a key not in ";
196 }
197 PrintTo(expected_, os);
198 *os << " or a value differs by more than " << tolerance_;
199 }
200
201 private:
202 const absl::flat_hash_map<K, double> expected_;
203 const bool all_keys_;
204 const double tolerance_;
205};

314template <typename RayType>
315class RayMatcher : public MatcherInterface<RayType> {
316 public:
317 RayMatcher(RayType expected, const double tolerance)
318 : expected_(std::move(expected)), tolerance_(tolerance) {}
319 void DescribeTo(std::ostream* os) const final {
320 *os << "after L_inf normalization, is within tolerance: " << tolerance_
321 << " of expected: ";
322 PrintTo(expected_, os);
323 }
324 void DescribeNegationTo(std::ostream* const os) const final {
325 *os << "after L_inf normalization, is not within tolerance: " << tolerance_
326 << " of expected: ";
327 PrintTo(expected_, os);
328 }
329
330 protected:
331 const RayType expected_;
332 const double tolerance_;
333};

453template <typename K>
454double InfinityNorm(const absl::flat_hash_map<K, double>& vector) {
455 double infinity_norm = 0.0;
456 for (auto [id, value] : vector) {
457 infinity_norm = std::max(infinity_norm, std::abs(value));
458 }
459 return infinity_norm;
460}
461
462// Returns a normalized primal ray.
463//
464// The normalization is done using infinity norm:
465//
466// ray / ||ray||_inf
467//
468// If the input ray norm is zero, the ray is returned unchanged.
469PrimalRay NormalizePrimalRay(PrimalRay ray) {
470 const double norm = InfinityNorm(ray.variable_values);
471 if (norm != 0.0) {
472 for (auto& entry : ray.variable_values) {
473 entry.second /= norm;
474 }
475 }
476 return ray;
477}

518// Returns a normalized dual ray.
519//
520// The normalization is done using infinity norm:
521//
522// ray / ||ray||_inf
523//
524// If the input ray norm is zero, the ray is returned unchanged.
525DualRay NormalizeDualRay(DualRay ray) {
526 const double norm =
527 std::max(InfinityNorm(ray.dual_values), InfinityNorm(ray.reduced_costs));
528 if (norm != 0.0) {
529 for (auto& entry : ray.dual_values) {
530 entry.second /= norm;
531 }
532 for (auto& entry : ray.reduced_costs) {
533 entry.second /= norm;
534 }
535 }
536 return ray;
537}

539class DualRayMatcher : public RayMatcher<DualRay> {
540 public:
541 DualRayMatcher(DualRay expected, const double tolerance)
542 : RayMatcher(std::move(expected), tolerance) {}
543
544 bool MatchAndExplain(DualRay actual, MatchResultListener* os) const override {
545 auto normalized_actual = NormalizeDualRay(actual);
546 auto normalized_expected = NormalizeDualRay(expected_);
547 if (os->IsInterested()) {
548 *os << "actual normalized: " << PrintToString(normalized_actual)
549 << ", expected normalized: " << PrintToString(normalized_expected);
550 }
551 return ExplainMatchResult(
552 IsNear(normalized_expected.dual_values, tolerance_),
553 normalized_actual.dual_values, os) &&
554 ExplainMatchResult(
555 IsNear(normalized_expected.reduced_costs, tolerance_),
556 normalized_actual.reduced_costs, os);
557 }
558};

628template <typename MatcherType>
629std::string MatcherToStringImpl(const MatcherType& matcher, const bool negate) {
630 std::ostringstream os;
631 if (negate) {
632 matcher.DescribeNegationTo(&os);
633 } else {
634 matcher.DescribeTo(&os);
635 }
636 return os.str();
637}

639template <typename T>
640std::string MatcherToString(const Matcher<T>& matcher, bool negate) {
641 return MatcherToStringImpl(matcher, negate);
642}

794bool MightTerminateWithRays(const TerminationReason reason) {
795 switch (reason) {
796 case TerminationReason::kInfeasibleOrUnbounded:
797 case TerminationReason::kUnbounded:
798 case TerminationReason::kInfeasible:
799 return true;
800 default:
801 return false;
802 }

805std::vector<TerminationReason> CompatibleReasons(
806 const TerminationReason expected, const bool inf_or_unb_soft_match) {
807 if (!inf_or_unb_soft_match) {
808 return {expected};
809 }
810 switch (expected) {
811 case TerminationReason::kUnbounded:
812 return {TerminationReason::kUnbounded,
813 TerminationReason::kInfeasibleOrUnbounded};
814 case TerminationReason::kInfeasible:
815 return {TerminationReason::kInfeasible,
816 TerminationReason::kInfeasibleOrUnbounded};
817 case TerminationReason::kInfeasibleOrUnbounded:
818 return {TerminationReason::kUnbounded, TerminationReason::kInfeasible,
819 TerminationReason::kInfeasibleOrUnbounded};
820 default:
821 return {expected};
822 }
823}

826 const std::vector<Solution>& expected_solutions,
827 const SolveResultMatcherOptions& options) {

832 .check_primal = true,

837 SolutionMatcherOptions{.tolerance = options.tolerance,
838 .check_primal = true,
839 .check_dual = options.check_dual,
840 .check_basis = options.check_basis});

863template <typename RayType>
864Matcher<std::vector<RayType>> CheckRays(
865 const std::vector<RayType>& expected_rays, const double tolerance,
866 bool check_all) {

878Matcher<SolveResult> IsConsistentWith(
879 const SolveResult& expected, const SolveResultMatcherOptions& options) {
880 std::vector<Matcher<SolveResult>> to_check;
881 to_check.push_back(TerminatesWithOneOf(CompatibleReasons(
882 expected.termination.reason, options.inf_or_unb_soft_match)));
883 const bool skip_solution =
884 MightTerminateWithRays(expected.termination.reason) &&
885 !options.check_solutions_if_inf_or_unbounded;
886 if (!skip_solution) {
887 to_check.push_back(Field("solutions", &SolveResult::solutions,
888 CheckSolutions(expected.solutions, options)));
889 }
890 if (options.check_rays) {
891 to_check.push_back(Field("primal_rays", &SolveResult::primal_rays,
892 CheckRays(expected.primal_rays, options.tolerance,
893 !options.first_solution_only)));
894 to_check.push_back(Field("dual_rays", &SolveResult::dual_rays,
895 CheckRays(expected.dual_rays, options.tolerance,
896 !options.first_solution_only)));
897 }
898
899 return AllOfArray(to_check);
900}
