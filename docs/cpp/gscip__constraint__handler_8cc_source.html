69// Enum with supported user-implementable callback functions in the SCIP
70// constraint handler. Non-user-implementable functions are not included here
71// (e.g. CONSFREE). Same order as in type_cons.h.
72enum class ConstraintHandlerCallbackType {
73 kSepaLp, // CONSSEPALP
74 kSepaSol, // CONSSEPASOL
75 kEnfoLp, // CONSENFOLP
76 // Unsupported: kEnfoRelax, // CONSENFORELAX
77 kEnfoPs, // CONSENFOPS
78 kConsCheck, // CONSCHECK
79 // Unsupported: kConsProp, // CONSPROP
80 // Unsupported: kConsPresol, // CONSPRESOL
81 // Unsupported: kConsResProp, // CONSRESPROP
82 kConsLock // CONSLOCK
83};
84
85// Returns the "do nothing" callback result. Used to handle the edge case where
86// Enfo* callbacks need to return kFeasible instead of kDidNotRun.
87GScipCallbackResult DidNotRunCallbackResult(
88 const ConstraintHandlerCallbackType callback_type) {
89 // TODO(user): Add kEnfoRelax when we support it.
90 if (callback_type == ConstraintHandlerCallbackType::kEnfoLp ||
91 callback_type == ConstraintHandlerCallbackType::kEnfoPs) {
92 return GScipCallbackResult::kFeasible;
93 }
94 return GScipCallbackResult::kDidNotRun;
95}

97// In callbacks, SCIP requires the first SCIP_RESULT in a priority list be
98// returned when multiple results are applicable. This is a unified order of the
99// priorities extracted from type_cons.h. The higher the result, the higher
100// priority it is.
101int ConstraintHandlerResultPriority(
102 const GScipCallbackResult result,
103 const ConstraintHandlerCallbackType callback_type) {
104 // In type_cons.h, callback results are consistently ordered across all
105 // constraint handler callback methods except that SCIP_SOLVELP (kSolveLp)
106 // takes higher priority than SCIP_BRANCHED (kBranched) in CONSENFOLP, and the
107 // reverse is true for CONSENFORELAX and CONSENFOLP.
108 switch (result) {
109 case GScipCallbackResult::kUnbounded:
110 return 14;
111 case GScipCallbackResult::kCutOff:
112 return 13;
113 case GScipCallbackResult::kSuccess:
114 return 12;
115 case GScipCallbackResult::kConstraintAdded:
116 return 11;
117 case GScipCallbackResult::kReducedDomain:
118 return 10;
119 case GScipCallbackResult::kSeparated:
120 return 9;
121 case GScipCallbackResult::kBranched:
122 return callback_type == ConstraintHandlerCallbackType::kEnfoLp ? 7 : 8;
123 case GScipCallbackResult::kSolveLp:
124 return callback_type == ConstraintHandlerCallbackType::kEnfoLp ? 8 : 7;
125 case GScipCallbackResult::kInfeasible:
126 return 6;
127 case GScipCallbackResult::kFeasible:
128 return 5;
129 case GScipCallbackResult::kNewRound:
130 return 4;
131 case GScipCallbackResult::kDidNotFind:
132 return 3;
133 case GScipCallbackResult::kDidNotRun:
134 return 2;
135 case GScipCallbackResult::kDelayed:
136 return 1;
137 case GScipCallbackResult::kDelayNode:
138 return 0;
139 default:
140 // kConstraintChanged, kFoundSolution, and kSuspend are not used in
141 // constraint handlers.
142 return -1;
143 }
144}
145
146constexpr GScipCallbackResult kMinPriority = GScipCallbackResult::kDelayNode;
147
148// Calls the callback function over a span of constraints, returning the highest
149// priority callback result, along with a SCIP return code.
150absl::StatusOr<GScipCallbackResult> ApplyCallback(
151 absl::Span<SCIP_Cons*> constraints,
152 std::function<GScipCallbackResult(void*)> callback_function,
153 const ConstraintHandlerCallbackType callback_type) {
154 if (constraints.empty()) {
155 return DidNotRunCallbackResult(callback_type);
156 }
157 GScipCallbackResult callback_result = kMinPriority;
158 for (SCIP_Cons* cons : constraints) {
159 if (cons == nullptr) {
160 return absl::InternalError("Constraint handler has null constraint");
161 }
162 SCIP_CONSDATA* consdata = SCIPconsGetData(cons);
163 if (consdata == nullptr || consdata->data == nullptr) {
164 return absl::InternalError("Constraint handler has null constraint data");
165 }
166 const GScipCallbackResult cons_result = callback_function(consdata->data);
167 if (ConstraintHandlerResultPriority(cons_result, callback_type) >
168 ConstraintHandlerResultPriority(callback_result, callback_type)) {
169 callback_result = cons_result;
170 }
171 }
172 return callback_result;
173}

175// Calls the callback function over all the constraints of a constraint handler,
176// prioritizing the ones SCIP deems more useful. Returns the highest priority
177// callback result, along with a SCIP return code.
178absl::StatusOr<GScipCallbackResult> ApplyCallback(
179 SCIP_Cons** constraints, const int num_useful_constraints,
180 const int total_num_constraints,
181 std::function<GScipCallbackResult(void*)> callback_function,
182 const ConstraintHandlerCallbackType callback_type) {
183 absl::Span<SCIP_Cons*> all_constraints =
184 absl::MakeSpan(constraints, total_num_constraints);
185 absl::Span<SCIP_Cons*> useful_constraints =
186 all_constraints.subspan(0, num_useful_constraints);
187 ASSIGN_OR_RETURN(
188 const GScipCallbackResult result,
189 ApplyCallback(useful_constraints, callback_function, callback_type));
190 // The first num_useful_constraints are the ones that are more likely to be
191 // violated. If no violation was found, we consider the remaining
192 // constraints.
193 if (result == GScipCallbackResult::kDidNotFind ||
194 result == GScipCallbackResult::kDidNotRun ||
195 result == GScipCallbackResult::kFeasible) {
196 absl::Span<SCIP_Cons*> remaining_constraints =
197 all_constraints.subspan(num_useful_constraints);
198 ASSIGN_OR_RETURN(
199 const GScipCallbackResult remaining_result,
200 ApplyCallback(remaining_constraints, callback_function, callback_type));
201 if (remaining_result != GScipCallbackResult::kDidNotFind &&
202 remaining_result != GScipCallbackResult::kDidNotRun &&
203 remaining_result != GScipCallbackResult::kFeasible) {
204 return remaining_result;
205 }
206 }
207 return result;
208}

210GScipCallbackStats GetCallbackStats(SCIP* scip) {
211 GScipCallbackStats stats;
212 stats.num_processed_nodes = SCIPgetNNodes(scip);
213 stats.num_processed_nodes_total = SCIPgetNTotalNodes(scip);
214 switch (SCIPgetStage(scip)) {
215 case SCIP_STAGE_INITPRESOLVE:
216 case SCIP_STAGE_PRESOLVING:
217 case SCIP_STAGE_EXITPRESOLVE:
218 case SCIP_STAGE_SOLVING: {
219 SCIP_NODE* node = SCIPgetCurrentNode(scip);
220 stats.current_node_id = node == nullptr ? -1 : node->number;
221 break;
222 }
223 default:
224 stats.current_node_id = stats.num_processed_nodes;
225 }
226 return stats;
227}

252absl::StatusOr<GScipCallbackResult> GScipConstraintHandlerContext::AddCut(
253 const GScipLinearRange& range, const std::string& name,
254 const GScipCutOptions& options) {
255 SCIP* scip = gscip_->scip();
256 SCIP_ROW* row = nullptr;
257 RETURN_IF_SCIP_ERROR(SCIPcreateEmptyRowConshdlr(
258 scip, &row, current_handler_, name.data(), range.lower_bound,
259 range.upper_bound, options.local, options.modifiable, options.removable));
260 RETURN_IF_SCIP_ERROR(SCIPcacheRowExtensions(scip, row));
261 if (range.coefficients.size() != range.variables.size()) {
262 return absl::InternalError(
263 "GScipLinearRange variables and coefficients do not match in size");
264 }
265 for (int i = 0; i < range.variables.size(); ++i) {
266 RETURN_IF_SCIP_ERROR(SCIPaddVarToRow(scip, row, range.variables.at(i),
267 range.coefficients.at(i)));
268 }
269 RETURN_IF_SCIP_ERROR(SCIPflushRowExtensions(scip, row));
270 SCIP_Bool infeasible;
271 RETURN_IF_SCIP_ERROR(SCIPaddRow(scip, row, options.force_cut, &infeasible));
272 RETURN_IF_SCIP_ERROR(SCIPreleaseRow(scip, &row));
273 return infeasible ? GScipCallbackResult::kCutOff
274 : GScipCallbackResult::kSeparated;
275}

482static SCIP_DECL_CONSLOCK(VariableRoundingLockC) {
483 SCIP_CONSHDLRDATA* scip_handler_data = SCIPconshdlrGetData(conshdlr);
484 operations_research::GScip* gscip = scip_handler_data->gscip;
485 GScipHandler* gscip_handler = scip_handler_data->gscip_handler.get();
486 SCIP_CONSDATA* consdata = SCIPconsGetData(cons);
487 if (consdata == nullptr) {
488 return SCIP_ERROR;
489 }
490 const bool lock_type_is_model = locktype == SCIP_LOCKTYPE_MODEL;
491 for (const auto [locked_var, lock_direction] :
492 gscip_handler->RoundingLock(gscip, consdata->data, lock_type_is_model)) {
493 int lock_down;
494 int lock_up;
495 switch (lock_direction) {
496 case operations_research::RoundingLockDirection::kUp:
497 lock_down = nlocksneg;
498 lock_up = nlockspos;
499 break;
500 case operations_research::RoundingLockDirection::kDown:
501 lock_down = nlockspos;
502 lock_up = nlocksneg;
503 break;
504 case operations_research::RoundingLockDirection::kBoth:
505 lock_down = nlocksneg + nlockspos;
506 lock_up = nlocksneg + nlockspos;
507 break;
508 }
509 SCIP_CALL(
510 SCIPaddVarLocksType(scip, locked_var, locktype, lock_down, lock_up));
511 }
512 return SCIP_OKAY;
513}

520absl::Status RegisterConstraintHandler(
521 GScip* gscip,
522 std::unique_ptr<UntypedGScipConstraintHandler> constraint_handler) {
523 SCIP_CONSHDLR* c_scip_handler;
524 SCIP_CONSHDLRDATA* scip_handler_data = new SCIP_CONSHDLRDATA;
525 scip_handler_data->gscip_handler = std::move(constraint_handler);
526 scip_handler_data->gscip = gscip;
527 SCIP* scip = gscip->scip();
528 const GScipConstraintHandlerProperties& properties =
529 scip_handler_data->gscip_handler->properties();
530
531 RETURN_IF_SCIP_ERROR(SCIPincludeConshdlrBasic(
532 scip, &c_scip_handler, properties.name.c_str(),
533 properties.description.c_str(), properties.enforcement_priority,
534 properties.feasibility_check_priority, properties.eager_frequency,
535 properties.needs_constraints, EnforceLpC, EnforcePseudoSolutionC,
536 CheckFeasibilityC, VariableRoundingLockC, scip_handler_data));
537 if (c_scip_handler == nullptr) {
538 return absl::InternalError("SCIP failed to add constraint handler");
539 }
540 RETURN_IF_SCIP_ERROR(SCIPsetConshdlrSepa(
541 scip, c_scip_handler, SeparateLpC, SeparatePrimalSolutionC,
542 properties.separation_frequency, properties.separation_priority,
543 properties.delay_separation));
544 RETURN_IF_SCIP_ERROR(
545 SCIPsetConshdlrFree(scip, c_scip_handler, ConstraintHandlerFreeC));
546 RETURN_IF_SCIP_ERROR(
547 SCIPsetConshdlrDelete(scip, c_scip_handler, ConstraintDataDeleteC));
548
549 return absl::OkStatus();
550}

552absl::Status AddCallbackConstraint(GScip* gscip, absl::string_view handler_name,
553 absl::string_view constraint_name,
554 void* constraint_data,
555 const GScipConstraintOptions& options) {
556 if (constraint_data == nullptr) {
557 return absl::InvalidArgumentError(
558 "Constraint data missing when adding a constraint handler callback");
559 }
560 SCIP* scip = gscip->scip();
561 SCIP_CONSHDLR* conshdlr = SCIPfindConshdlr(scip, handler_name.data());
562 if (conshdlr == nullptr) {
563 return util::InternalErrorBuilder()
564 << "Constraint handler " << handler_name
565 << " not registered with SCIP. Check if you "
566 "registered the constraint handler before adding constraints.";
567 }
568 SCIP_CONSDATA* consdata = new SCIP_CONSDATA;
569 consdata->data = constraint_data;
570 SCIP_CONS* constraint = nullptr;
571 RETURN_IF_SCIP_ERROR(SCIPcreateCons(
572 scip, &constraint, constraint_name.data(), conshdlr, consdata,
573 options.initial, options.separate, options.enforce, options.check,
574 options.propagate, options.local, options.modifiable, options.dynamic,
575 options.removable, options.sticking_at_node));
576 if (constraint == nullptr) {
577 return absl::InternalError("SCIP failed to create constraint");
578 }
579 RETURN_IF_SCIP_ERROR(SCIPaddCons(scip, constraint));
