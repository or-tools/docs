14// This library solves 0-1 one-dimensional knapsack problems with fractional
15// profits and weights using the branch and bound algorithm. Note that
16// algorithms/knapsack_solver uses 'int64_t' for the profits and the weights.
17// TODO(user): Merge this code with algorithms/knapsack_solver.
18//
19// Given n items, each with a profit and a weight and a knapsack of
20// capacity c, the goal is to find a subset of the items which fits inside c
21// and maximizes the total profit.
22// Without loss of generality, profits and weights are assumed to be positive.
23//
24// From a mathematical point of view, the one-dimensional knapsack problem
25// can be modeled by linear constraint:
26// Sum(i:1..n)(weight_i * item_i) <= c,
27// where item_i is a 0-1 integer variable.
28// The goal is to maximize: Sum(i:1..n)(profit_i * item_i).
29//
30// Example Usage:
31// std::vector<double> profits = {0, 0.5, 0.4, 1, 1, 1.1};
32// std::vector<double> weights = {9, 6, 2, 1.5, 1.5, 1.5};
33// KnapsackSolverForCuts solver("solver");
34// solver.Init(profits, weights, capacity);
35// bool is_solution_optimal = false;
36// std::unique_ptr<TimeLimit> time_limit =
37// std::make_unique<TimeLimit>(time_limit_seconds); // Set the time limit.
38// const double profit = solver.Solve(time_limit.get(), &is_solution_optimal);
39// const int number_of_items(profits.size());
40// for (int item_id(0); item_id < number_of_items; ++item_id) {
41// solver.best_solution(item_id); // Access the solution.
42// }

67
68 int item_id;
69 bool is_in;
70};

72// ----- KnapsackItemForCuts -----
73// KnapsackItemForCuts is a small struct to pair an item weight with its
74// corresponding profit.
75// The aim of the knapsack problem is to pack as many valuable items as
76// possible. A straight forward heuristic is to take those with the greatest
77// profit-per-unit-weight. This ratio is called efficiency in this
78// implementation. So items will be grouped in vectors, and sorted by
79// decreasing efficiency.
80struct KnapsackItemForCuts {
81 KnapsackItemForCuts(int id, double weight, double profit)
82 : id(id), weight(weight), profit(profit) {}
83
84 double GetEfficiency(double profit_max) const {
85 return (weight > 0) ? profit / weight : profit_max;
86 }
87
88 // The 'id' field is used to retrieve the initial item in order to
89 // communicate with other propagators and state.
90 const int id;
91 const double weight;
92 const double profit;
93};
94using KnapsackItemForCutsPtr = std::unique_ptr<KnapsackItemForCuts>;

96// ----- KnapsackSearchNodeForCuts -----
97// KnapsackSearchNodeForCuts is a class used to describe a decision in the
98// decision search tree.
99// The node is defined by a pointer to the parent search node and an
100// assignment (see KnapsackAssignmentForCuts).
101// As the current state is not explicitly stored in a search node, one should
102// go through the search tree to incrementally build a partial solution from
103// a previous search node.
104class KnapsackSearchNodeForCuts {
105 public:
106 KnapsackSearchNodeForCuts(const KnapsackSearchNodeForCuts* parent,
107 const KnapsackAssignmentForCuts& assignment);
108
109 KnapsackSearchNodeForCuts(const KnapsackSearchNodeForCuts&) = delete;
110 KnapsackSearchNodeForCuts& operator=(const KnapsackSearchNodeForCuts&) =
111 delete;
112
113 int depth() const { return depth_; }
114 const KnapsackSearchNodeForCuts* parent() const { return parent_; }
115 const KnapsackAssignmentForCuts& assignment() const { return assignment_; }
116
117 double current_profit() const { return current_profit_; }
118 void set_current_profit(double profit) { current_profit_ = profit; }
119
120 double profit_upper_bound() const { return profit_upper_bound_; }
121 void set_profit_upper_bound(double profit) { profit_upper_bound_ = profit; }
122
123 int next_item_id() const { return next_item_id_; }
124 void set_next_item_id(int id) { next_item_id_ = id; }
125
126 private:
127 // 'depth_' is used to navigate efficiently through the search tree.
128 int depth_;
129 const KnapsackSearchNodeForCuts* const parent_;
130 KnapsackAssignmentForCuts assignment_;
131
132 // 'current_profit_' and 'profit_upper_bound_' fields are used to sort search
133 // nodes using a priority queue. That allows to pop the node with the best
134 // upper bound, and more importantly to stop the search when optimality is
135 // proved.
136 double current_profit_;
137 double profit_upper_bound_;
138
139 // 'next_item_id_' field allows to avoid an O(number_of_items) scan to find
140 // next item to select. This is done for free by the upper bound computation.
141 int next_item_id_;
142};

144// ----- KnapsackSearchPathForCuts -----
145// KnapsackSearchPathForCuts is a small class used to represent the path between
146// a node to another node in the search tree.
147// As the solution state is not stored for each search node, the state should
148// be rebuilt at each node. One simple solution is to apply all decisions
149// between the node 'to' and the root. This can be computed in
150// O(number_of_items).
151//
152// However, it is possible to achieve better average complexity. Two
153// consecutively explored nodes are usually close enough (i.e., much less than
154// number_of_items) to benefit from an incremental update from the node
155// 'from' to the node 'to'.
156//
157// The 'via' field is the common parent of 'from' field and 'to' field.
158// So the state can be built by reverting all decisions from 'from' to 'via'
159// and then applying all decisions from 'via' to 'to'.
160class KnapsackSearchPathForCuts {
161 public:
162 KnapsackSearchPathForCuts(const KnapsackSearchNodeForCuts* from,
163 const KnapsackSearchNodeForCuts* to);
164
165 KnapsackSearchPathForCuts(const KnapsackSearchPathForCuts&) = delete;
166 KnapsackSearchPathForCuts& operator=(const KnapsackSearchPathForCuts&) =
167 delete;
168
169 void Init();
170 const KnapsackSearchNodeForCuts& from() const { return *from_; }
171 const KnapsackSearchNodeForCuts& via() const { return *via_; }
172 const KnapsackSearchNodeForCuts& to() const { return *to_; }
173
174 private:
175 const KnapsackSearchNodeForCuts* from_;
176 const KnapsackSearchNodeForCuts* via_; // Computed in 'Init'.
177 const KnapsackSearchNodeForCuts* to_;
178};

215// ----- KnapsackPropagatorForCuts -----
216// KnapsackPropagatorForCuts is used to enforce a capacity constraint.
217// It is supposed to compute profit lower and upper bounds, and get the next
218// item to select, it can be seen as a 0-1 Knapsack solver. The most efficient
219// way to compute the upper bound is to iterate on items in
220// profit-per-unit-weight decreasing order. The break item is commonly defined
221// as the first item for which there is not enough remaining capacity. Selecting
222// this break item as the next-item-to-assign usually gives the best results
223// (see Greenberg & Hegerich).
224//
225// This is exactly what is implemented in this class.
226//
227// It is possible to compute a better profit lower bound almost for free. During
228// the scan to find the break element all unbound items are added just as if
229// they were part of the current solution. This is used in both
230// ComputeProfitBounds() and CopyCurrentSolution(). For incrementality reasons,
231// the ith item should be accessible in O(1). That's the reason why the item
232// vector has to be duplicated 'sorted_items_'.
233class KnapsackPropagatorForCuts {
234 public:
235 explicit KnapsackPropagatorForCuts(const KnapsackStateForCuts* state);
236 ~KnapsackPropagatorForCuts();
237
238 KnapsackPropagatorForCuts(const KnapsackPropagatorForCuts&) = delete;
239 KnapsackPropagatorForCuts& operator=(const KnapsackPropagatorForCuts&) =
240 delete;
241
242 // Initializes the data structure and then calls InitPropagator.
243 void Init(absl::Span<const double> profits, absl::Span<const double> weights,
244 double capacity);
245
246 // Updates data structure. Returns false on failure.
247 bool Update(bool revert, const KnapsackAssignmentForCuts& assignment);
248 // ComputeProfitBounds should set 'profit_lower_bound_' and
249 // 'profit_upper_bound_' which are constraint specific.
250 void ComputeProfitBounds();
251 // Returns the id of next item to assign.
252 // Returns kNoSelection when all items are bound.
253 int GetNextItemId() const { return break_item_id_; }
254
255 double current_profit() const { return current_profit_; }
256 double profit_lower_bound() const { return profit_lower_bound_; }
257 double profit_upper_bound() const { return profit_upper_bound_; }
258
259 // Copies the current state into 'solution'.
260 // All unbound items are set to false (i.e. not in the knapsack).
261 void CopyCurrentStateToSolution(std::vector<bool>* solution) const;
262
263 // Initializes the propagator. This method is called by Init() after filling
264 // the fields defined in this class.
265 void InitPropagator();
266
267 const KnapsackStateForCuts& state() const { return *state_; }
268 const std::vector<KnapsackItemForCutsPtr>& items() const { return items_; }
269
270 void set_profit_lower_bound(double profit) { profit_lower_bound_ = profit; }
271 void set_profit_upper_bound(double profit) { profit_upper_bound_ = profit; }
272
273 private:
274 // An obvious additional profit upper bound corresponds to the linear
275 // relaxation: remaining_capacity * efficiency of the break item.
276 // It is possible to do better in O(1), using Martello-Toth bound U2.
277 // The main idea is to enforce integrality constraint on the break item,
278 // i.e. either the break item is part of the solution, or it is not.
279 // So basically the linear relaxation is done on the item before the break
280 // item, or the one after the break item. This is what GetAdditionalProfit
281 // method implements.
282 double GetAdditionalProfitUpperBound(double remaining_capacity,
283 int break_item_id) const;
284
285 double capacity_;
286 double consumed_capacity_;
287 int break_item_id_;
288 std::vector<KnapsackItemForCutsPtr> sorted_items_;
289 double profit_max_;
290 std::vector<KnapsackItemForCutsPtr> items_;
291 double current_profit_;
292 double profit_lower_bound_;
293 double profit_upper_bound_;
294 const KnapsackStateForCuts* const state_;
295};

297// ----- KnapsackSolverForCuts -----

307 KnapsackSolverForCuts& operator=(const KnapsackSolverForCuts&) = delete;
308

311 double capacity);
312 int GetNumberOfItems() const { return state_.GetNumberOfItems(); }
313

319
320 // Get the best upper bound found so far.

322

350 private:
351 // Updates propagator reverting/applying all decision on the path. Returns
352 // true if the propagation fails. Note that even if it fails, propagator

359 const KnapsackAssignmentForCuts& assignment);
360 // Updates the best solution if the current solution has a better profit.
361 void UpdateBestSolution();
362
363 // Returns true if new relevant search node was added to the nodes array. That
364 // means this node should be added to the search queue too.
365 bool MakeNewNode(const KnapsackSearchNodeForCuts& node, bool is_in);
366
367 // Gets the aggregated (min) profit upper bound among all propagators.
368 double GetAggregatedProfitUpperBound();
369 double GetCurrentProfit() const { return propagator_.current_profit(); }
370 int GetNextItemId() const { return propagator_.GetNextItemId(); }
371
372 KnapsackPropagatorForCuts propagator_;
373 std::vector<std::unique_ptr<KnapsackSearchNodeForCuts>> search_nodes_;
374 KnapsackStateForCuts state_;
375 double best_solution_profit_;
376 std::vector<bool> best_solution_;
377 const std::string solver_name_;
378 double solution_lower_bound_threshold_ =
379 std::numeric_limits<double>::infinity();
380 double solution_upper_bound_threshold_ =
381 -std::numeric_limits<double>::infinity();
382 int64_t node_limit_ = std::numeric_limits<int64_t>::max();
383};
