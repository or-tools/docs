35 return false;

39
40 return true;

316std::vector<std::string> GurobiDynamicLibraryPotentialPaths() {
317 std::vector<std::string> potential_paths;
318 const std::vector<std::string> kGurobiVersions = {"1000", "952", "951", "950",
319 "911", "910", "903", "902",
320 "811", "801", "752"};
321 potential_paths.reserve(kGurobiVersions.size() * 3);
322
323 // Look for libraries pointed by GUROBI_HOME first.
324 const char* gurobi_home_from_env = getenv("GUROBI_HOME");
325 if (gurobi_home_from_env != nullptr) {
326 for (const std::string& version : kGurobiVersions) {
327 const std::string lib = version.substr(0, version.size() - 1);
328#if defined(_MSC_VER) // Windows
329 potential_paths.push_back(
330 absl::StrCat(gurobi_home_from_env, "\\bin\\gurobi", lib, ".dll"));
331#elif defined(__APPLE__) // OS X
332 potential_paths.push_back(
333 absl::StrCat(gurobi_home_from_env, "/lib/libgurobi", lib, ".dylib"));
334#elif defined(__GNUC__) // Linux
335 potential_paths.push_back(
336 absl::StrCat(gurobi_home_from_env, "/lib/libgurobi", lib, ".so"));
337 potential_paths.push_back(
338 absl::StrCat(gurobi_home_from_env, "/lib64/libgurobi", lib, ".so"));
339#else
340 LOG(ERROR) << "OS Not recognized by gurobi/environment.cc."
341 << " You won't be able to use Gurobi.";
342#endif
343 }
344 }
345
346 // Search for canonical places.
347 for (const std::string& version : kGurobiVersions) {
348 const std::string lib = version.substr(0, version.size() - 1);
349#if defined(_MSC_VER) // Windows
350 potential_paths.push_back(absl::StrCat("C:\\Program Files\\gurobi", version,
351 "\\win64\\bin\\gurobi", lib,
352 ".dll"));
353 potential_paths.push_back(absl::StrCat(
354 "C:\\gurobi", version, "\\win64\\bin\\gurobi", lib, ".dll"));
355#elif defined(__APPLE__) // OS X
356 potential_paths.push_back(absl::StrCat(
357 "/Library/gurobi", version, "/mac64/lib/libgurobi", lib, ".dylib"));
358 potential_paths.push_back(absl::StrCat("/Library/gurobi", version,
359 "/macos_universal2/lib/libgurobi",
360 lib, ".dylib"));
361#elif defined(__GNUC__) // Linux
362 potential_paths.push_back(absl::StrCat(
363 "/opt/gurobi", version, "/linux64/lib/libgurobi", lib, ".so"));
364 potential_paths.push_back(absl::StrCat(
365 "/opt/gurobi", version, "/linux64/lib64/libgurobi", lib, ".so"));
366 potential_paths.push_back(
367 absl::StrCat("/opt/gurobi/linux64/lib/libgurobi", lib, ".so"));
368 potential_paths.push_back(
369 absl::StrCat("/opt/gurobi/linux64/lib64/libgurobi", lib, ".so"));
370#else
371 LOG(ERROR) << "OS Not recognized by gurobi/environment.cc."
372 << " You won't be able to use Gurobi.";
373#endif
374 }
375
376#if defined(__GNUC__) // path in linux64 gurobi/optimizer docker image.
377 for (const std::string& version : {"10.0.0", "9.5.2", "9.5.1", "9.5.0"}) {
378 potential_paths.push_back(
379 absl::StrCat("/opt/gurobi/linux64/lib/libgurobi.so.", version));
380 }
381#endif
382 return potential_paths;
383}

385absl::Status LoadGurobiDynamicLibrary(
386 std::vector<std::string> potential_paths) {
387 static std::once_flag gurobi_loading_done;
388 static absl::Status gurobi_load_status;
389 static DynamicLibrary gurobi_library;
390 static absl::Mutex mutex;
391
392 absl::MutexLock lock(&mutex);
393
394 std::call_once(gurobi_loading_done, [&potential_paths]() {
395 const std::vector<std::string> canonical_paths =
396 GurobiDynamicLibraryPotentialPaths();
397 potential_paths.insert(potential_paths.end(), canonical_paths.begin(),
398 canonical_paths.end());
399 for (const std::string& path : potential_paths) {
400 if (gurobi_library.TryToLoad(path)) {
401 LOG(INFO) << "Found the Gurobi library in '" << path << ".";
402 break;
403 }
404 }
405
406 if (gurobi_library.LibraryIsLoaded()) {
407 LoadGurobiFunctions(&gurobi_library);
408 gurobi_load_status = absl::OkStatus();
409 } else {
410 gurobi_load_status = absl::NotFoundError(absl::StrCat(
411 "Could not find the Gurobi shared library. Looked in: [",
412 absl::StrJoin(potential_paths, "', '"),
413 "]. If you know where it"
414 " is, pass the full path to 'LoadGurobiDynamicLibrary()'."));
415 }
416 });
417 return gurobi_load_status;
418}
419
420absl::StatusOr<GRBenv*> GetGurobiEnv() {
421 RETURN_IF_ERROR(LoadGurobiDynamicLibrary({}));
422
423 GRBenv* env = nullptr;
424
425 if (GRBloadenv(&env, nullptr) != 0 || env == nullptr) {
426 return absl::FailedPreconditionError(
427 absl::StrCat("Found the Gurobi shared library, but could not create "
428 "Gurobi environment: is Gurobi licensed on this machine?",
429 GRBgeterrormsg(env)));
430 }
431 return env;
432}
