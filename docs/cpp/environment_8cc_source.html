32bool GurobiIsCorrectlyInstalled() {
33 absl::StatusOr<GRBenv*> status = GetGurobiEnv();

35 LOG(WARNING) << status.status();
36 return false;
37 }
38
39 GRBfreeenv(status.value());
40
41 return true;
42}

319std::vector<std::string> GurobiDynamicLibraryPotentialPaths() {
320 std::vector<std::string> potential_paths;
321 const std::vector<std::string> kGurobiVersions = {"1000", "952", "951", "950",
322 "911", "910", "903", "902",
323 "811", "801", "752"};
324 potential_paths.reserve(kGurobiVersions.size() * 3);
325
326 // Look for libraries pointed by GUROBI_HOME first.
327 const char* gurobi_home_from_env = getenv("GUROBI_HOME");
328 if (gurobi_home_from_env != nullptr) {
329 for (const std::string& version : kGurobiVersions) {
330 const std::string lib = version.substr(0, version.size() - 1);
331#if defined(_MSC_VER) // Windows
332 potential_paths.push_back(
333 absl::StrCat(gurobi_home_from_env, "\\bin\\gurobi", lib, ".dll"));
334#elif defined(__APPLE__) // OS X
335 potential_paths.push_back(
336 absl::StrCat(gurobi_home_from_env, "/lib/libgurobi", lib, ".dylib"));
337#elif defined(__GNUC__) // Linux
338 potential_paths.push_back(
339 absl::StrCat(gurobi_home_from_env, "/lib/libgurobi", lib, ".so"));
340 potential_paths.push_back(
341 absl::StrCat(gurobi_home_from_env, "/lib64/libgurobi", lib, ".so"));
342#else
343 LOG(ERROR) << "OS Not recognized by gurobi/environment.cc."
344 << " You won't be able to use Gurobi.";
345#endif
346 }
347 }
348
349 // Search for canonical places.
350 for (const std::string& version : kGurobiVersions) {
351 const std::string lib = version.substr(0, version.size() - 1);
352#if defined(_MSC_VER) // Windows
353 potential_paths.push_back(absl::StrCat("C:\\Program Files\\gurobi", version,
354 "\\win64\\bin\\gurobi", lib,
355 ".dll"));
356 potential_paths.push_back(absl::StrCat(
357 "C:\\gurobi", version, "\\win64\\bin\\gurobi", lib, ".dll"));
358 potential_paths.push_back(absl::StrCat("gurobi", lib, ".dll"));
359#elif defined(__APPLE__) // OS X
360 potential_paths.push_back(absl::StrCat(
361 "/Library/gurobi", version, "/mac64/lib/libgurobi", lib, ".dylib"));
362 potential_paths.push_back(absl::StrCat("/Library/gurobi", version,
363 "/macos_universal2/lib/libgurobi",
364 lib, ".dylib"));
365#elif defined(__GNUC__) // Linux
366 potential_paths.push_back(absl::StrCat(
367 "/opt/gurobi", version, "/linux64/lib/libgurobi", lib, ".so"));
368 potential_paths.push_back(absl::StrCat(
369 "/opt/gurobi", version, "/linux64/lib64/libgurobi", lib, ".so"));
370 potential_paths.push_back(
371 absl::StrCat("/opt/gurobi/linux64/lib/libgurobi", lib, ".so"));
372 potential_paths.push_back(
373 absl::StrCat("/opt/gurobi/linux64/lib64/libgurobi", lib, ".so"));
374#else
375 LOG(ERROR) << "OS Not recognized by gurobi/environment.cc."
376 << " You won't be able to use Gurobi.";
377#endif
378 }
379
380#if defined(__GNUC__) // path in linux64 gurobi/optimizer docker image.
381 for (const std::string& version : {"10.0.0", "9.5.2", "9.5.1", "9.5.0"}) {
382 potential_paths.push_back(
383 absl::StrCat("/opt/gurobi/linux64/lib/libgurobi.so.", version));
384 }
385#endif
386 return potential_paths;
387}

389absl::Status LoadGurobiDynamicLibrary(
390 std::vector<std::string> potential_paths) {
391 static std::once_flag gurobi_loading_done;
392 static absl::Status gurobi_load_status;
393 static DynamicLibrary gurobi_library;
394 static absl::Mutex mutex;
395
396 absl::MutexLock lock(&mutex);
397
398 std::call_once(gurobi_loading_done, [&potential_paths]() {
399 const std::vector<std::string> canonical_paths =
400 GurobiDynamicLibraryPotentialPaths();
401 potential_paths.insert(potential_paths.end(), canonical_paths.begin(),
402 canonical_paths.end());
403 for (const std::string& path : potential_paths) {
404 if (gurobi_library.TryToLoad(path)) {
405 LOG(INFO) << "Found the Gurobi library in '" << path << ".";
406 break;
407 }
408 }
409
410 if (gurobi_library.LibraryIsLoaded()) {
411 LoadGurobiFunctions(&gurobi_library);
412 gurobi_load_status = absl::OkStatus();
413 } else {
414 gurobi_load_status = absl::NotFoundError(absl::StrCat(
415 "Could not find the Gurobi shared library. Looked in: [",
416 absl::StrJoin(potential_paths, "', '"),
417 "]. If you know where it"
418 " is, pass the full path to 'LoadGurobiDynamicLibrary()'."));
419 }
420 });
421 return gurobi_load_status;
422}
423
424absl::StatusOr<GRBenv*> GetGurobiEnv() {
425 RETURN_IF_ERROR(LoadGurobiDynamicLibrary({}));
426
427 GRBenv* env = nullptr;
428
429 if (GRBloadenv(&env, nullptr) != 0 || env == nullptr) {
430 return absl::FailedPreconditionError(
431 absl::StrCat("Found the Gurobi shared library, but could not create "
432 "Gurobi environment: is Gurobi licensed on this machine?",
433 GRBgeterrormsg(env)));
434 }
435 return env;
436}
