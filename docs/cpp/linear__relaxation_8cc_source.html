64std::pair<IntegerValue, IntegerValue> GetMinAndMaxNotEncoded(
65 IntegerVariable var,
66 const absl::flat_hash_set<IntegerValue>& encoded_values,
67 const Model& model) {
68 CHECK(VariableIsPositive(var));
69 const PositiveOnlyIndex index = GetPositiveOnlyIndex(var);
70
71 const auto* domains = model.Get<IntegerDomains>();
72 if (domains == nullptr || index >= domains->size()) {
73 return {kMaxIntegerValue, kMinIntegerValue};
74 }
75
76 // The domain can be large, but the list of values shouldn't, so this
77 // runs in O(encoded_values.size());
78 IntegerValue min = kMaxIntegerValue;
79 for (const int64_t v : (*domains)[index].Values()) {
80 if (!encoded_values.contains(IntegerValue(v))) {
81 min = IntegerValue(v);
82 break;
83 }
84 }
85
86 IntegerValue max = kMinIntegerValue;
87 const Domain negated_domain = (*domains)[index].Negation();
88 for (const int64_t v : negated_domain.Values()) {
89 if (!encoded_values.contains(IntegerValue(-v))) {
90 max = IntegerValue(-v);
91 break;
92 }
93 }
94
95 return {min, max};
96}

98bool LinMaxContainsOnlyOneVarInExpressions(const ConstraintProto& ct) {
99 CHECK_EQ(ct.constraint_case(), ConstraintProto::ConstraintCase::kLinMax);
100 int current_var = -1;
101 for (const LinearExpressionProto& expr : ct.lin_max().exprs()) {
102 if (expr.vars().empty()) continue;
103 if (expr.vars().size() > 1) return false;
104 const int var = PositiveRef(expr.vars(0));
105 if (current_var == -1) {
106 current_var = var;
107 } else if (var != current_var) {
108 return false;
109 }
110 }
111 return true;
112}
113
114// Collect all the affines expressions in a LinMax constraint.
115// It checks that these are indeed affine expressions, and that they all share
116// the same variable.
117// It returns the shared variable, as well as a vector of pairs
118// (coefficient, offset) when each affine is coefficient * shared_var + offset.
119void CollectAffineExpressionWithSingleVariable(
120 const ConstraintProto& ct, CpModelMapping* mapping, IntegerVariable* var,
121 std::vector<std::pair<IntegerValue, IntegerValue>>* affines) {
122 DCHECK(LinMaxContainsOnlyOneVarInExpressions(ct));
123 CHECK_EQ(ct.constraint_case(), ConstraintProto::ConstraintCase::kLinMax);
124 *var = kNoIntegerVariable;
125 affines->clear();
126 for (const LinearExpressionProto& expr : ct.lin_max().exprs()) {
127 if (expr.vars().empty()) {
128 affines->push_back({IntegerValue(0), IntegerValue(expr.offset())});
129 } else {
130 CHECK_EQ(expr.vars().size(), 1);
131 const IntegerVariable affine_var = mapping->Integer(expr.vars(0));
132 if (*var == kNoIntegerVariable) {
133 *var = PositiveVariable(affine_var);
134 }
135 if (VariableIsPositive(affine_var)) {
136 CHECK_EQ(affine_var, *var);
137 affines->push_back(
138 {IntegerValue(expr.coeffs(0)), IntegerValue(expr.offset())});
139 } else {
140 CHECK_EQ(NegationOf(affine_var), *var);
141 affines->push_back(
142 {IntegerValue(-expr.coeffs(0)), IntegerValue(expr.offset())});
143 }
144 }
145 }
146}

150void AppendRelaxationForEqualityEncoding(IntegerVariable var,
151 const Model& model,
152 LinearRelaxation* relaxation,
153 int* num_tight, int* num_loose) {
154 const auto* encoder = model.Get<IntegerEncoder>();
155 const auto* integer_trail = model.Get<IntegerTrail>();
156 if (encoder == nullptr || integer_trail == nullptr) return;
157
158 std::vector<Literal> at_most_one_ct;
159 absl::flat_hash_set<IntegerValue> encoded_values;
160 std::vector<ValueLiteralPair> encoding;
161 {
162 const std::vector<ValueLiteralPair>& initial_encoding =
163 encoder->PartialDomainEncoding(var);
164 if (initial_encoding.empty()) return;
165 for (const auto value_literal : initial_encoding) {
166 const Literal literal = value_literal.literal;
167
168 // Note that we skip pairs that do not have an Integer view.
169 if (encoder->GetLiteralView(literal) == kNoIntegerVariable &&
170 encoder->GetLiteralView(literal.Negated()) == kNoIntegerVariable) {
171 continue;
172 }
173
174 encoding.push_back(value_literal);
175 at_most_one_ct.push_back(literal);
176 encoded_values.insert(value_literal.value);
177 }
178 }
179 if (encoded_values.empty()) return;
180
181 // TODO(user): PartialDomainEncoding() filter pair corresponding to literal
182 // set to false, however the initial variable Domain is not always updated. As
183 // a result, these min/max can be larger than in reality. Try to fix this even
184 // if in practice this is a rare occurrence, as the presolve should have
185 // propagated most of what we can.
186 const auto [min_not_encoded, max_not_encoded] =
187 GetMinAndMaxNotEncoded(var, encoded_values, model);
188
189 // This means that there are no non-encoded value and we have a full encoding.
190 // We substract the minimum value to reduce its size.
191 if (min_not_encoded == kMaxIntegerValue) {
192 const IntegerValue rhs = encoding[0].value;
193 LinearConstraintBuilder at_least_one(&model, IntegerValue(1),
194 kMaxIntegerValue);
195 LinearConstraintBuilder encoding_ct(&model, rhs, rhs);
196 encoding_ct.AddTerm(var, IntegerValue(1));
197 for (const auto value_literal : encoding) {
198 const Literal lit = value_literal.literal;
199 CHECK(at_least_one.AddLiteralTerm(lit, IntegerValue(1)));
200
201 const IntegerValue delta = value_literal.value - rhs;
202 if (delta != IntegerValue(0)) {
203 CHECK_GE(delta, IntegerValue(0));
204 CHECK(encoding_ct.AddLiteralTerm(lit, -delta));
205 }
206 }
207
208 // It is possible that the linear1 encoding respect our overflow
209 // precondition but not the Var = sum bool * value one. In this case, we
210 // just don't encode it this way. Hopefully, most normal model will not run
211 // into this.
212 const LinearConstraint lc = encoding_ct.Build();
213 if (!PossibleOverflow(*integer_trail, lc)) {
214 relaxation->linear_constraints.push_back(at_least_one.Build());
215 relaxation->linear_constraints.push_back(std::move(lc));
216 relaxation->at_most_ones.push_back(at_most_one_ct);
217 ++*num_tight;
218 }
219 return;
220 }
221
222 // In this special case, the two constraints below can be merged into an
223 // equality: var = rhs + sum l_i * (value_i - rhs).
224 if (min_not_encoded == max_not_encoded) {
225 const IntegerValue rhs = min_not_encoded;
226 LinearConstraintBuilder encoding_ct(&model, rhs, rhs);
227 encoding_ct.AddTerm(var, IntegerValue(1));
228 for (const auto value_literal : encoding) {
229 CHECK(encoding_ct.AddLiteralTerm(value_literal.literal,
230 rhs - value_literal.value));
231 }
232
233 const LinearConstraint lc = encoding_ct.Build();
234 if (!PossibleOverflow(*integer_trail, lc)) {
235 relaxation->at_most_ones.push_back(at_most_one_ct);
236 relaxation->linear_constraints.push_back(std::move(lc));
237 ++*num_tight;
238 }
239 return;
240 }
241
242 // min + sum l_i * (value_i - min) <= var.
243 // Note that this might overflow in corner cases, so we need to prevent that.
244 const IntegerValue d_min = min_not_encoded;
245 LinearConstraintBuilder lower_bound_ct(&model, d_min, kMaxIntegerValue);
246 lower_bound_ct.AddTerm(var, IntegerValue(1));
247 for (const auto value_literal : encoding) {
248 CHECK(lower_bound_ct.AddLiteralTerm(value_literal.literal,
249 d_min - value_literal.value));
250 }
251 const LinearConstraint built_lb_ct = lower_bound_ct.Build();
252 if (!PossibleOverflow(*integer_trail, built_lb_ct)) {
253 relaxation->linear_constraints.push_back(std::move(built_lb_ct));
254 ++*num_loose;
255 }
256
257 // var <= max + sum l_i * (value_i - max).
258 // Note that this might overflow in corner cases, so we need to prevent that.
259 const IntegerValue d_max = max_not_encoded;
260 LinearConstraintBuilder upper_bound_ct(&model, kMinIntegerValue, d_max);
261 upper_bound_ct.AddTerm(var, IntegerValue(1));
262 for (const auto value_literal : encoding) {
263 CHECK(upper_bound_ct.AddLiteralTerm(value_literal.literal,
264 d_max - value_literal.value));
265 }
266 const LinearConstraint built_ub_ct = upper_bound_ct.Build();
267 if (!PossibleOverflow(*integer_trail, built_ub_ct)) {
268 relaxation->linear_constraints.push_back(std::move(built_ub_ct));
269 ++*num_loose;
270 }
271
272 // Note that empty/trivial constraints will be filtered later.
273 relaxation->at_most_ones.push_back(at_most_one_ct);
274}

276void AppendPartialGreaterThanEncodingRelaxation(IntegerVariable var,
277 const Model& model,
278 LinearRelaxation* relaxation) {
279 const auto* integer_trail = model.Get<IntegerTrail>();
280 const auto* encoder = model.Get<IntegerEncoder>();
281 if (integer_trail == nullptr || encoder == nullptr) return;
282
283 const auto& greater_than_encoding = encoder->PartialGreaterThanEncoding(var);
284 if (greater_than_encoding.empty()) return;
285
286 // Start by the var >= side.
287 // And also add the implications between used literals.
288 {
289 IntegerValue prev_used_bound = integer_trail->LowerBound(var);
290 LinearConstraintBuilder builder(&model, prev_used_bound, kMaxIntegerValue);
291 builder.AddTerm(var, IntegerValue(1));
292 LiteralIndex prev_literal_index = kNoLiteralIndex;
293 for (const auto entry : greater_than_encoding) {
294 if (entry.value <= prev_used_bound) continue;
295
296 const LiteralIndex literal_index = entry.literal.Index();
297 const IntegerValue diff = prev_used_bound - entry.value;
298
299 // Skip the entry if the literal doesn't have a view.
300 if (!builder.AddLiteralTerm(entry.literal, diff)) continue;
301 if (prev_literal_index != kNoLiteralIndex) {
302 // Add var <= prev_var, which is the same as var + not(prev_var) <= 1
303 relaxation->at_most_ones.push_back(
304 {Literal(literal_index), Literal(prev_literal_index).Negated()});
305 }
306 prev_used_bound = entry.value;
307 prev_literal_index = literal_index;
308 }
309
310 // Note that by construction, this shouldn't be able to overflow.
311 relaxation->linear_constraints.push_back(builder.Build());
312 }
313
314 // Do the same for the var <= side by using NegationOfVar().
315 // Note that we do not need to add the implications between literals again.
316 {
317 IntegerValue prev_used_bound = integer_trail->LowerBound(NegationOf(var));
318 LinearConstraintBuilder builder(&model, prev_used_bound, kMaxIntegerValue);
319 builder.AddTerm(var, IntegerValue(-1));
320 for (const auto entry :
321 encoder->PartialGreaterThanEncoding(NegationOf(var))) {
322 if (entry.value <= prev_used_bound) continue;
323 const IntegerValue diff = prev_used_bound - entry.value;
324
325 // Skip the entry if the literal doesn't have a view.
326 if (!builder.AddLiteralTerm(entry.literal, diff)) continue;
327 prev_used_bound = entry.value;
328 }
329
330 // Note that by construction, this shouldn't be able to overflow.
331 relaxation->linear_constraints.push_back(builder.Build());
332 }
333}

347void AppendBoolOrRelaxation(const ConstraintProto& ct, Model* model,
348 LinearRelaxation* relaxation) {
349 auto* mapping = model->GetOrCreate<CpModelMapping>();
350 LinearConstraintBuilder lc(model, IntegerValue(1), kMaxIntegerValue);
351 for (const int enforcement_ref : ct.enforcement_literal()) {
352 CHECK(lc.AddLiteralTerm(mapping->Literal(NegatedRef(enforcement_ref)),
353 IntegerValue(1)));
354 }
355 for (const int ref : ct.bool_or().literals()) {
356 CHECK(lc.AddLiteralTerm(mapping->Literal(ref), IntegerValue(1)));
357 }
358 relaxation->linear_constraints.push_back(lc.Build());
359}

361void AppendBoolAndRelaxation(const ConstraintProto& ct, Model* model,
362 LinearRelaxation* relaxation,
363 ActivityBoundHelper* activity_helper) {
364 if (!HasEnforcementLiteral(ct)) return;
365
366 // TODO(user): These constraints can be many, and if they are not regrouped
367 // in big at most ones, then they should probably only added lazily as cuts.
368 // Regroup this with future clique-cut separation logic.
369 //
370 // Note that for the case with only one enforcement, what we do below is
371 // already done by the clique merging code.
372 auto* mapping = model->GetOrCreate<CpModelMapping>();
373 if (ct.enforcement_literal().size() == 1) {
374 const Literal enforcement = mapping->Literal(ct.enforcement_literal(0));
375 for (const int ref : ct.bool_and().literals()) {
376 relaxation->at_most_ones.push_back(
377 {enforcement, mapping->Literal(ref).Negated()});
378 }
379 return;
380 }
381
382 // If we have many_literals => many_fixed literal, it is important to
383 // try to use a tight big-M if we can. This is important on neos-957323.pb.gz
384 // for instance.
385 //
386 // We split the literal into disjoint AMO and we encode each with
387 // sum Not(literals) <= sum Not(enforcement)
388 //
389 // Note that what we actually do is use the decomposition into at most one
390 // and add a constraint for each part rather than just adding the sum of them.
391 //
392 // TODO(user): More generally, do not miss the same structure if the bool_and
393 // was expanded into many clauses!
394 //
395 // TODO(user): It is not 100% clear that just not adding one constraint is
396 // worse. Relaxation is worse, but then we have less constraint.
397 LinearConstraintBuilder builder(model);
398 if (activity_helper != nullptr) {
399 std::vector<int> negated_lits;
400 for (const int ref : ct.bool_and().literals()) {
401 negated_lits.push_back(NegatedRef(ref));
402 }
403 for (absl::Span<const int> part :
404 activity_helper->PartitionLiteralsIntoAmo(negated_lits)) {
405 builder.Clear();
406 for (const int negated_ref : part) {
407 CHECK(builder.AddLiteralTerm(mapping->Literal(negated_ref)));
408 }
409 for (const int enforcement_ref : ct.enforcement_literal()) {
410 CHECK(builder.AddLiteralTerm(
411 mapping->Literal(NegatedRef(enforcement_ref)), IntegerValue(-1)));
412 }
413 relaxation->linear_constraints.push_back(
414 builder.BuildConstraint(kMinIntegerValue, IntegerValue(0)));
415 }
416 } else {
417 for (const int ref : ct.bool_and().literals()) {
418 builder.Clear();
419 CHECK(builder.AddLiteralTerm(mapping->Literal(NegatedRef(ref))));
420 for (const int enforcement_ref : ct.enforcement_literal()) {
421 CHECK(builder.AddLiteralTerm(
422 mapping->Literal(NegatedRef(enforcement_ref)), IntegerValue(-1)));
423 }
424 relaxation->linear_constraints.push_back(
425 builder.BuildConstraint(kMinIntegerValue, IntegerValue(0)));
426 }
427 }

439void AppendExactlyOneRelaxation(const ConstraintProto& ct, Model* model,
440 LinearRelaxation* relaxation) {
441 if (HasEnforcementLiteral(ct)) return;
442 auto* mapping = model->GetOrCreate<CpModelMapping>();
443 auto* encoder = model->GetOrCreate<IntegerEncoder>();
444
445 const std::vector<Literal> literals =
446 mapping->Literals(ct.exactly_one().literals());
447 if (AllLiteralsHaveViews(*encoder, literals)) {
448 LinearConstraintBuilder lc(model, IntegerValue(1), IntegerValue(1));
449 for (const Literal lit : literals) {
450 CHECK(lc.AddLiteralTerm(lit, IntegerValue(1)));
451 }
452 relaxation->linear_constraints.push_back(lc.Build());
453 } else {
454 // We just encode the at most one part that might be partially linearized
455 // later.
456 relaxation->at_most_ones.push_back(literals);
457 }
458}

460std::vector<Literal> CreateAlternativeLiteralsWithView(
461 int num_literals, Model* model, LinearRelaxation* relaxation) {
462 auto* encoder = model->GetOrCreate<IntegerEncoder>();
463
464 if (num_literals == 1) {
465 // This is not supposed to happen, but it is easy enough to cover, just
466 // in case. We might however want to use encoder->GetTrueLiteral().
467 const IntegerVariable var = model->Add(NewIntegerVariable(1, 1));
468 const Literal lit =
469 encoder->GetOrCreateLiteralAssociatedToEquality(var, IntegerValue(1));
470 return {lit};
471 }
472
473 if (num_literals == 2) {
474 const IntegerVariable var = model->Add(NewIntegerVariable(0, 1));
475 const Literal lit =
476 encoder->GetOrCreateLiteralAssociatedToEquality(var, IntegerValue(1));
477
478 // TODO(user): We shouldn't need to create this view ideally. Even better,
479 // we should be able to handle Literal natively in the linear relaxation,
480 // but that is a lot of work.
481 const IntegerVariable var2 = model->Add(NewIntegerVariable(0, 1));
482 encoder->AssociateToIntegerEqualValue(lit.Negated(), var2, IntegerValue(1));
483
484 return {lit, lit.Negated()};
485 }
486
487 std::vector<Literal> literals;
488 LinearConstraintBuilder lc_builder(model, IntegerValue(1), IntegerValue(1));
489 for (int i = 0; i < num_literals; ++i) {
490 const IntegerVariable var = model->Add(NewIntegerVariable(0, 1));
491 const Literal lit =
492 encoder->GetOrCreateLiteralAssociatedToEquality(var, IntegerValue(1));
493 literals.push_back(lit);
494 CHECK(lc_builder.AddLiteralTerm(lit, IntegerValue(1)));
495 }
496 model->Add(ExactlyOneConstraint(literals));
497 relaxation->linear_constraints.push_back(lc_builder.Build());
498 return literals;
499}

501void AppendCircuitRelaxation(const ConstraintProto& ct, Model* model,
502 LinearRelaxation* relaxation) {
503 if (HasEnforcementLiteral(ct)) return;
504 auto* mapping = model->GetOrCreate<CpModelMapping>();
505 const int num_arcs = ct.circuit().literals_size();
506 CHECK_EQ(num_arcs, ct.circuit().tails_size());
507 CHECK_EQ(num_arcs, ct.circuit().heads_size());
508
509 // Each node must have exactly one incoming and one outgoing arc (note
510 // that it can be the unique self-arc of this node too).
511 absl::btree_map<int, std::vector<Literal>> incoming_arc_constraints;
512 absl::btree_map<int, std::vector<Literal>> outgoing_arc_constraints;
513 for (int i = 0; i < num_arcs; i++) {
514 const Literal arc = mapping->Literal(ct.circuit().literals(i));
515 const int tail = ct.circuit().tails(i);
516 const int head = ct.circuit().heads(i);
517
518 // Make sure this literal has a view.
519 model->Add(NewIntegerVariableFromLiteral(arc));
520 outgoing_arc_constraints[tail].push_back(arc);
521 incoming_arc_constraints[head].push_back(arc);
522 }
523 for (const auto* node_map :
524 {&outgoing_arc_constraints, &incoming_arc_constraints}) {
525 for (const auto& entry : *node_map) {
526 const std::vector<Literal>& exactly_one = entry.second;
527 if (exactly_one.size() > 1) {
528 LinearConstraintBuilder at_least_one_lc(model, IntegerValue(1),
529 kMaxIntegerValue);
530 for (const Literal l : exactly_one) {
531 CHECK(at_least_one_lc.AddLiteralTerm(l, IntegerValue(1)));
532 }
533
534 // We separate the two constraints.
535 relaxation->at_most_ones.push_back(exactly_one);
536 relaxation->linear_constraints.push_back(at_least_one_lc.Build());
537 }
538 }
539 }

542void AppendRoutesRelaxation(const ConstraintProto& ct, Model* model,
543 LinearRelaxation* relaxation) {
544 if (HasEnforcementLiteral(ct)) return;
545 auto* mapping = model->GetOrCreate<CpModelMapping>();
546 const int num_arcs = ct.routes().literals_size();
547 CHECK_EQ(num_arcs, ct.routes().tails_size());
548 CHECK_EQ(num_arcs, ct.routes().heads_size());
549
550 // Each node except node zero must have exactly one incoming and one outgoing
551 // arc (note that it can be the unique self-arc of this node too). For node
552 // zero, the number of incoming arcs should be the same as the number of
553 // outgoing arcs.
554 absl::btree_map<int, std::vector<Literal>> incoming_arc_constraints;
555 absl::btree_map<int, std::vector<Literal>> outgoing_arc_constraints;
556 for (int i = 0; i < num_arcs; i++) {
557 const Literal arc = mapping->Literal(ct.routes().literals(i));
558 const int tail = ct.routes().tails(i);
559 const int head = ct.routes().heads(i);
560
561 // Make sure this literal has a view.
562 model->Add(NewIntegerVariableFromLiteral(arc));
563 outgoing_arc_constraints[tail].push_back(arc);
564 incoming_arc_constraints[head].push_back(arc);
565 }
566 for (const auto* node_map :
567 {&outgoing_arc_constraints, &incoming_arc_constraints}) {
568 for (const auto& entry : *node_map) {
569 if (entry.first == 0) continue;
570 const std::vector<Literal>& exactly_one = entry.second;
571 if (exactly_one.size() > 1) {
572 LinearConstraintBuilder at_least_one_lc(model, IntegerValue(1),
573 kMaxIntegerValue);
574 for (const Literal l : exactly_one) {
575 CHECK(at_least_one_lc.AddLiteralTerm(l, IntegerValue(1)));
576 }
577
578 // We separate the two constraints.
579 relaxation->at_most_ones.push_back(exactly_one);
580 relaxation->linear_constraints.push_back(at_least_one_lc.Build());
581 }
582 }
583 }
584 LinearConstraintBuilder zero_node_balance_lc(model, IntegerValue(0),
585 IntegerValue(0));
586 for (const Literal& incoming_arc : incoming_arc_constraints[0]) {
587 CHECK(zero_node_balance_lc.AddLiteralTerm(incoming_arc, IntegerValue(1)));
588 }
589 for (const Literal& outgoing_arc : outgoing_arc_constraints[0]) {
590 CHECK(zero_node_balance_lc.AddLiteralTerm(outgoing_arc, IntegerValue(-1)));
591 }
592 relaxation->linear_constraints.push_back(zero_node_balance_lc.Build());
593}

635// Scan the intervals of a cumulative/no_overlap constraint, and its capacity (1
636// for the no_overlap). It returns the index of the makespan interval if found,
637// or std::nullopt otherwise.
638//
639// Currently, this requires the capacity to be fixed in order to scan for a
640// makespan interval.
641//
642// The makespan interval has the following property:
643// - its end is fixed at the horizon
644// - it is always present
645// - its demand is the capacity of the cumulative/no_overlap.
646// - its size is > 0.
647//
648// These property ensures that all other intervals ends before the start of
649// the makespan interval.
650std::optional<int> DetectMakespan(
651 const std::vector<IntervalVariable>& intervals,
652 const std::vector<AffineExpression>& demands,
653 const AffineExpression& capacity, Model* model) {
654 IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
655 IntervalsRepository* repository = model->GetOrCreate<IntervalsRepository>();
656
657 // TODO(user): Supports variable capacity.
658 if (!integer_trail->IsFixed(capacity)) {
659 return std::nullopt;
660 }
661
662 // Detect the horizon (max of all end max of all intervals).
663 IntegerValue horizon = kMinIntegerValue;
664 for (int i = 0; i < intervals.size(); ++i) {
665 if (repository->IsAbsent(intervals[i])) continue;
666 horizon = std::max(
667 horizon, integer_trail->UpperBound(repository->End(intervals[i])));
668 }
669
670 const IntegerValue capacity_value = integer_trail->FixedValue(capacity);
671 for (int i = 0; i < intervals.size(); ++i) {
672 if (!repository->IsPresent(intervals[i])) continue;
673 const AffineExpression& end = repository->End(intervals[i]);
674 if (integer_trail->IsFixed(demands[i]) &&
675 integer_trail->FixedValue(demands[i]) == capacity_value &&
676 integer_trail->IsFixed(end) &&
677 integer_trail->FixedValue(end) == horizon &&
678 integer_trail->LowerBound(repository->Size(intervals[i])) > 0) {
679 return i;
680 }
681 }
682 return std::nullopt;
683}

687std::optional<AffineExpression> DetectMakespanFromPrecedences(
688 const SchedulingConstraintHelper& helper, Model* model) {
689 if (helper.NumTasks() == 0) return {};
690
691 const absl::Span<const AffineExpression> ends = helper.Ends();
692 std::vector<IntegerVariable> end_vars;
693 for (const AffineExpression& end : ends) {
694 // TODO(user): Deal with constant end.
695 if (end.var == kNoIntegerVariable) return {};
696 if (end.coeff != 1) return {};
697 end_vars.push_back(end.var);
698 }
699
700 std::vector<FullIntegerPrecedence> output;
701 auto* precedences = model->GetOrCreate<PrecedenceRelations>();
702 precedences->ComputeFullPrecedences(end_vars, &output);
703 for (const auto& p : output) {
704 // TODO(user): What if we have more than one candidate makespan ?
705 if (p.indices.size() != ends.size()) continue;
706
707 // We have a Makespan!
708 // We want p.var - min_delta >= max(end).
709 IntegerValue min_delta = kMaxIntegerValue;
710 for (int i = 0; i < p.indices.size(); ++i) {
711 // end_vars[indices[i]] + p.offset[i] <= p.var
712 // [interval_end - end_offset][indices[i]] + p.offset[i] <= p.var
713 //
714 // TODO(user): It is possible this min_delta becomes positive but for a
715 // real makespan, we could make sure it is <= 0. This can happen if we
716 // have an optional interval because we didn't compute the offset assuming
717 // this interval was present. We could potentially fix it if we know that
718 // presence_literal => p.var >= end.
719 min_delta =
720 std::min(min_delta, p.offsets[i] - ends[p.indices[i]].constant);
721 }
722 VLOG(2) << "Makespan detected >= ends + " << min_delta;
723 return AffineExpression(p.var, IntegerValue(1), -min_delta);
724 }
725
726 return {};
727}

808// This relaxation will compute the bounding box of all tasks in the cumulative,
809// and add the constraint that the sum of energies of each task must fit in the
810// capacity * span area.
811void AddCumulativeRelaxation(const AffineExpression& capacity,
812 SchedulingConstraintHelper* helper,
813 SchedulingDemandHelper* demands_helper,
814 const std::optional<AffineExpression>& makespan,
815 Model* model, LinearRelaxation* relaxation) {
816 const int num_intervals = helper->NumTasks();
817 demands_helper->CacheAllEnergyValues();
818
819 IntegerValue min_of_starts = kMaxIntegerValue;
820 IntegerValue max_of_ends = kMinIntegerValue;
821 int num_variable_energies = 0;
822 int num_optionals = 0;
823 IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
824 for (int index = 0; index < num_intervals; ++index) {
825 if (helper->IsAbsent(index)) continue;
826 if (helper->IsOptional(index) && demands_helper->EnergyMin(index) >= 0) {
827 num_optionals++;
828 }
829 if (!helper->SizeIsFixed(index) || !demands_helper->DemandIsFixed(index)) {
830 num_variable_energies++;
831 }
832 min_of_starts = std::min(min_of_starts, helper->StartMin(index));
833 max_of_ends = std::max(max_of_ends, helper->EndMax(index));
834 }
835
836 VLOG(2) << "Span [" << min_of_starts << ".." << max_of_ends << "] with "
837 << num_optionals << " optional intervals, and "
838 << num_variable_energies << " variable energy tasks out of "
839 << num_intervals << " intervals"
840 << (makespan.has_value() ? ", and 1 makespan" : "");
841
842 // If nothing is variable, the linear relaxation will already be enforced
843 // by the scheduling propagators.
844 if (num_variable_energies + num_optionals == 0) return;
845
846 LinearConstraintBuilder lc(model, kMinIntegerValue, IntegerValue(0));
847 int num_intervals_added = 0;
848 for (int index = 0; index < num_intervals; ++index) {
849 if (helper->IsAbsent(index)) continue;
850 if (helper->IsOptional(index)) {
851 const IntegerValue energy_min = demands_helper->EnergyMin(index);
852 if (energy_min == 0) continue;
853 if (!lc.AddLiteralTerm(helper->PresenceLiteral(index), energy_min)) {
854 return;
855 }
856 } else {
857 const std::vector<LiteralValueValue>& product =
858 demands_helper->DecomposedEnergies()[index];
859 if (!product.empty()) {
860 // The energy is defined if the vector is not empty.
861 if (!lc.AddDecomposedProduct(product)) return;
862 } else {
863 // The energy is not a decomposed product, but it could still be
864 // constant or linear. If not, a McCormick relaxation will be
865 // introduced. AddQuadraticLowerBound() supports all cases.
866 lc.AddQuadraticLowerBound(helper->Sizes()[index],
867 demands_helper->Demands()[index],
868 integer_trail);
869 }
870 }
871 ++num_intervals_added;
872 }
873 if (num_intervals_added == 0) return;
874
875 // Create and link span_start and span_end to the starts and ends of the
876 // tasks.
877 //
878 // TODO(user): In some cases, we could have only one task that can be
879 // first.
880 const AffineExpression span_start = min_of_starts;
881 const AffineExpression span_end =
882 makespan.has_value() ? makespan.value() : max_of_ends;
883 lc.AddTerm(span_end, -integer_trail->UpperBound(capacity));
884 lc.AddTerm(span_start, integer_trail->UpperBound(capacity));
885 relaxation->linear_constraints.push_back(lc.Build());
886}

888// Adds the energetic relaxation sum(areas) <= bounding box area.
889void AppendNoOverlap2dRelaxation(const ConstraintProto& ct, Model* model,
890 LinearRelaxation* relaxation) {
891 CHECK(ct.has_no_overlap_2d());
892 if (HasEnforcementLiteral(ct)) return;
893
894 auto* mapping = model->GetOrCreate<CpModelMapping>();
895 std::vector<IntervalVariable> x_intervals =
896 mapping->Intervals(ct.no_overlap_2d().x_intervals());
897 std::vector<IntervalVariable> y_intervals =
898 mapping->Intervals(ct.no_overlap_2d().y_intervals());
899
900 auto* integer_trail = model->GetOrCreate<IntegerTrail>();
901 auto* intervals_repository = model->GetOrCreate<IntervalsRepository>();
902
903 IntegerValue x_min = kMaxIntegerValue;
904 IntegerValue x_max = kMinIntegerValue;
905 IntegerValue y_min = kMaxIntegerValue;
906 IntegerValue y_max = kMinIntegerValue;
907 std::vector<AffineExpression> x_sizes;
908 std::vector<AffineExpression> y_sizes;
909 for (int i = 0; i < ct.no_overlap_2d().x_intervals_size(); ++i) {
910 x_sizes.push_back(intervals_repository->Size(x_intervals[i]));
911 y_sizes.push_back(intervals_repository->Size(y_intervals[i]));
912 x_min = std::min(x_min, integer_trail->LevelZeroLowerBound(
913 intervals_repository->Start(x_intervals[i])));
914 x_max = std::max(x_max, integer_trail->LevelZeroUpperBound(
915 intervals_repository->End(x_intervals[i])));
916 y_min = std::min(y_min, integer_trail->LevelZeroLowerBound(
917 intervals_repository->Start(y_intervals[i])));
918 y_max = std::max(y_max, integer_trail->LevelZeroUpperBound(
919 intervals_repository->End(y_intervals[i])));
920 }
921
922 const IntegerValue max_area =
923 IntegerValue(CapProd(CapSub(x_max.value(), x_min.value()),
924 CapSub(y_max.value(), y_min.value())));
925 if (max_area == kMaxIntegerValue) return;
926
927 auto* product_decomposer = model->GetOrCreate<ProductDecomposer>();
928 LinearConstraintBuilder lc(model, IntegerValue(0), max_area);
929 for (int i = 0; i < ct.no_overlap_2d().x_intervals_size(); ++i) {
930 if (intervals_repository->IsPresent(x_intervals[i]) &&
931 intervals_repository->IsPresent(y_intervals[i])) {
932 const std::vector<LiteralValueValue> energy =
933 product_decomposer->TryToDecompose(x_sizes[i], y_sizes[i]);
934 if (!energy.empty()) {
935 if (!lc.AddDecomposedProduct(energy)) return;
936 } else {
937 lc.AddQuadraticLowerBound(x_sizes[i], y_sizes[i], integer_trail);
938 }
939 } else if (intervals_repository->IsPresent(x_intervals[i]) ||
940 intervals_repository->IsPresent(y_intervals[i]) ||
941 (intervals_repository->PresenceLiteral(x_intervals[i]) ==
942 intervals_repository->PresenceLiteral(y_intervals[i]))) {
943 // We have only one active literal.
944 const Literal presence_literal =
945 intervals_repository->IsPresent(x_intervals[i])
946 ? intervals_repository->PresenceLiteral(y_intervals[i])
947 : intervals_repository->PresenceLiteral(x_intervals[i]);
948 const IntegerValue area_min =
949 integer_trail->LevelZeroLowerBound(x_sizes[i]) *
950 integer_trail->LevelZeroLowerBound(y_sizes[i]);
951 if (area_min != 0) {
952 // Not including the term if we don't have a view is ok.
953 (void)lc.AddLiteralTerm(presence_literal, area_min);
954 }
955 }
956 }
957 relaxation->linear_constraints.push_back(lc.Build());
958}

1022// Part 2: Encode upper bound on X.
1023//
1024// Add linking constraint to the CP solver
1025// sum zi = 1 and for all i, zi => max = expr_i.
1026void AppendLinMaxRelaxationPart2(
1027 IntegerVariable target, const std::vector<Literal>& alternative_literals,
1028 const std::vector<LinearExpression>& exprs, Model* model,
1029 LinearRelaxation* relaxation) {
1030 const int num_exprs = exprs.size();
1031 GenericLiteralWatcher* watcher = model->GetOrCreate<GenericLiteralWatcher>();
1032
1033 // First add the CP constraints.
1034 for (int i = 0; i < num_exprs; ++i) {
1035 LinearExpression local_expr;
1036 local_expr.vars = NegationOf(exprs[i].vars);
1037 local_expr.vars.push_back(target);
1038 local_expr.coeffs = exprs[i].coeffs;
1039 local_expr.coeffs.push_back(IntegerValue(1));
1040 IntegerSumLE* upper_bound =
1041 new IntegerSumLE({alternative_literals[i]}, local_expr.vars,
1042 local_expr.coeffs, exprs[i].offset, model);
1043 upper_bound->RegisterWith(watcher);
1044 model->TakeOwnership(upper_bound);
1045 }
1046
1047 // For the relaxation, we use different constraints with a stronger linear
1048 // relaxation as explained in the .h
1049 std::vector<std::vector<IntegerValue>> sum_of_max_corner_diff(
1050 num_exprs, std::vector<IntegerValue>(num_exprs, IntegerValue(0)));
1051
1052 // Cache coefficients.
1053 // TODO(user): Remove hash_map ?
1054 absl::flat_hash_map<std::pair<int, IntegerVariable>, IntegerValue> cache;
1055 for (int i = 0; i < num_exprs; ++i) {
1056 for (int j = 0; j < exprs[i].vars.size(); ++j) {
1057 cache[std::make_pair(i, exprs[i].vars[j])] = exprs[i].coeffs[j];
1058 }
1059 }
1060 const auto get_coeff = [&cache](IntegerVariable var, int index) {
1061 const auto it = cache.find(std::make_pair(index, var));
1062 if (it == cache.end()) return IntegerValue(0);
1063 return it->second;
1064 };
1065
1066 IntegerTrail* integer_trail = model->GetOrCreate<IntegerTrail>();
1067 std::vector<IntegerVariable> active_vars;
1068 for (int i = 0; i + 1 < num_exprs; ++i) {
1069 for (int j = i + 1; j < num_exprs; ++j) {
1070 active_vars = exprs[i].vars;
1071 active_vars.insert(active_vars.end(), exprs[j].vars.begin(),
1072 exprs[j].vars.end());
1073 gtl::STLSortAndRemoveDuplicates(&active_vars);
1074 for (const IntegerVariable x_var : active_vars) {
1075 const IntegerValue diff = get_coeff(x_var, j) - get_coeff(x_var, i);
1076 if (diff == 0) continue;
1077
1078 const IntegerValue lb = integer_trail->LevelZeroLowerBound(x_var);
1079 const IntegerValue ub = integer_trail->LevelZeroUpperBound(x_var);
1080 sum_of_max_corner_diff[i][j] += std::max(diff * lb, diff * ub);
1081 sum_of_max_corner_diff[j][i] += std::max(-diff * lb, -diff * ub);
1082 }
1083 }
1084 }
1085
1086 for (int i = 0; i < num_exprs; ++i) {
1087 LinearConstraintBuilder lc(model, kMinIntegerValue, IntegerValue(0));
1088 lc.AddTerm(target, IntegerValue(1));
1089 for (int j = 0; j < exprs[i].vars.size(); ++j) {
1090 lc.AddTerm(exprs[i].vars[j], -exprs[i].coeffs[j]);
1091 }
1092 for (int j = 0; j < num_exprs; ++j) {
1093 CHECK(lc.AddLiteralTerm(alternative_literals[j],
1094 -exprs[j].offset - sum_of_max_corner_diff[i][j]));
1095 }
1096 relaxation->linear_constraints.push_back(lc.Build());
1097 }
1098}

1100void AppendLinearConstraintRelaxation(const ConstraintProto& ct,
1101 bool linearize_enforced_constraints,
1102 Model* model,
1103 LinearRelaxation* relaxation,
1104 ActivityBoundHelper* activity_helper) {
1105 auto* mapping = model->Get<CpModelMapping>();
1106
1107 // Note that we ignore the holes in the domain.
1108 //
1109 // TODO(user): In LoadLinearConstraint() we already created intermediate
1110 // Booleans for each disjoint interval, we should reuse them here if
1111 // possible.
1112 //
1113 // TODO(user): process the "at most one" part of a == 1 separately?
1114 const IntegerValue rhs_domain_min = IntegerValue(ct.linear().domain(0));
1115 const IntegerValue rhs_domain_max =
1116 IntegerValue(ct.linear().domain(ct.linear().domain_size() - 1));
1117 if (rhs_domain_min == std::numeric_limits<int64_t>::min() &&
1118 rhs_domain_max == std::numeric_limits<int64_t>::max())
1119 return;
1120
1121 if (!HasEnforcementLiteral(ct)) {
1122 LinearConstraintBuilder lc(model, rhs_domain_min, rhs_domain_max);
1123 for (int i = 0; i < ct.linear().vars_size(); i++) {
1124 const int ref = ct.linear().vars(i);
1125 const int64_t coeff = ct.linear().coeffs(i);
1126 lc.AddTerm(mapping->Integer(ref), IntegerValue(coeff));
1127 }
1128 relaxation->linear_constraints.push_back(lc.Build());
1129 return;
1130 }
1131
1132 // Reified version.
1133 if (!linearize_enforced_constraints) return;
1134
1135 // We linearize fully reified constraints of size 1 all together for a given
1136 // variable. But we need to process half-reified ones.
1137 if (!mapping->IsHalfEncodingConstraint(&ct) && ct.linear().vars_size() <= 1) {
1138 return;
1139 }
1140
1141 std::vector<Literal> enforcing_literals;
1142 enforcing_literals.reserve(ct.enforcement_literal_size());
1143 for (const int enforcement_ref : ct.enforcement_literal()) {
1144 enforcing_literals.push_back(mapping->Literal(enforcement_ref));
1145 }
1146
1147 // Compute min/max activity.
1148 std::vector<std::pair<int, int64_t>> bool_terms;
1149 IntegerValue min_activity(0);
1150 IntegerValue max_activity(0);
1151 const auto integer_trail = model->GetOrCreate<IntegerTrail>();
1152 for (int i = 0; i < ct.linear().vars_size(); i++) {
1153 const int ref = ct.linear().vars(i);
1154 const IntegerValue coeff(ct.linear().coeffs(i));
1155 const IntegerVariable int_var = mapping->Integer(ref);
1156
1157 // Everything here should have a view.
1158 CHECK_NE(int_var, kNoIntegerVariable);
1159
1160 const IntegerValue lb = integer_trail->LowerBound(int_var);
1161 const IntegerValue ub = integer_trail->UpperBound(int_var);
1162 if (lb == 0 && ub == 1 && activity_helper != nullptr) {
1163 bool_terms.push_back({ref, coeff.value()});
1164 } else {
1165 if (coeff > 0) {
1166 min_activity += coeff * lb;
1167 max_activity += coeff * ub;
1168 } else {
1169 min_activity += coeff * ub;
1170 max_activity += coeff * lb;
1171 }
1172 }
1173 }
1174 if (activity_helper != nullptr) {
1175 min_activity +=
1176 IntegerValue(activity_helper->ComputeMinActivity(bool_terms));
1177 max_activity +=
1178 IntegerValue(activity_helper->ComputeMaxActivity(bool_terms));
1179 }
1180
1181 if (rhs_domain_min > min_activity) {
1182 // And(ei) => terms >= rhs_domain_min
1183 // <=> Sum_i (~ei * (rhs_domain_min - min_activity)) + terms >=
1184 // rhs_domain_min
1185 LinearConstraintBuilder lc(model, rhs_domain_min, kMaxIntegerValue);
1186 for (const Literal& literal : enforcing_literals) {
1187 CHECK(
1188 lc.AddLiteralTerm(literal.Negated(), rhs_domain_min - min_activity));
1189 }
1190 for (int i = 0; i < ct.linear().vars_size(); i++) {
1191 const int ref = ct.linear().vars(i);
1192 const IntegerValue coeff(ct.linear().coeffs(i));
1193 const IntegerVariable int_var = mapping->Integer(ref);
1194 lc.AddTerm(int_var, coeff);
1195 }
1196 relaxation->linear_constraints.push_back(lc.Build());
1197 }
1198 if (rhs_domain_max < max_activity) {
1199 // And(ei) => terms <= rhs_domain_max
1200 // <=> Sum_i (~ei * (rhs_domain_max - max_activity)) + terms <=
1201 // rhs_domain_max
1202 LinearConstraintBuilder lc(model, kMinIntegerValue, rhs_domain_max);
1203 for (const Literal& literal : enforcing_literals) {
1204 CHECK(
1205 lc.AddLiteralTerm(literal.Negated(), rhs_domain_max - max_activity));
1206 }
1207 for (int i = 0; i < ct.linear().vars_size(); i++) {
1208 const int ref = ct.linear().vars(i);
1209 const IntegerValue coeff(ct.linear().coeffs(i));
1210 const IntegerVariable int_var = mapping->Integer(ref);
1211 lc.AddTerm(int_var, coeff);
1212 }
1213 relaxation->linear_constraints.push_back(lc.Build());
1214 }
1215}

1217// Add a static and a dynamic linear relaxation of the CP constraint to the set
1218// of linear constraints. The highest linearization_level is, the more types of
1219// constraint we encode. This method should be called only for
1220// linearization_level > 0. The static part is just called a relaxation and is
1221// called at the root node of the search. The dynamic part is implemented
1222// through a set of linear cut generators that will be called throughout the
1223// search.
1224//
1225// TODO(user): In full generality, we could encode all the constraint as an LP.
1226// TODO(user): Add unit tests for this method.
1227// TODO(user): Remove and merge with model loading.
1228void TryToLinearizeConstraint(const CpModelProto& /*model_proto*/,
1229 const ConstraintProto& ct,
1230 int linearization_level, Model* model,
1231 LinearRelaxation* relaxation,
1232 ActivityBoundHelper* activity_helper) {
1233 CHECK_EQ(model->GetOrCreate<SatSolver>()->CurrentDecisionLevel(), 0);
1234 DCHECK_GT(linearization_level, 0);
1235
1236 const int old_index = relaxation->linear_constraints.size();
1237 switch (ct.constraint_case()) {
1238 case ConstraintProto::ConstraintCase::kBoolOr: {
1239 if (linearization_level > 1) {
1240 AppendBoolOrRelaxation(ct, model, relaxation);
1241 }
1242 break;
1243 }
1244 case ConstraintProto::ConstraintCase::kBoolAnd: {
1245 if (linearization_level > 1) {
1246 AppendBoolAndRelaxation(ct, model, relaxation, activity_helper);
1247 }
1248 break;
1249 }
1250 case ConstraintProto::ConstraintCase::kAtMostOne: {
1251 AppendAtMostOneRelaxation(ct, model, relaxation);
1252 break;
1253 }
1254 case ConstraintProto::ConstraintCase::kExactlyOne: {
1255 AppendExactlyOneRelaxation(ct, model, relaxation);
1256 break;
1257 }
1258 case ConstraintProto::ConstraintCase::kIntProd: {
1259 const LinearArgumentProto& int_prod = ct.int_prod();
1260 if (int_prod.exprs_size() == 2 &&
1261 LinearExpressionProtosAreEqual(int_prod.exprs(0),
1262 int_prod.exprs(1))) {
1263 AppendSquareRelaxation(ct, model, relaxation);
1264 AddSquareCutGenerator(ct, linearization_level, model, relaxation);
1265 } else {
1266 // No relaxation, just a cut generator .
1267 AddIntProdCutGenerator(ct, linearization_level, model, relaxation);
1268 }
1269 break;
1270 }
1271 case ConstraintProto::ConstraintCase::kLinMax: {
1272 AppendLinMaxRelaxationPart1(ct, model, relaxation);
1273 const bool is_affine_max = LinMaxContainsOnlyOneVarInExpressions(ct);
1274 if (is_affine_max) {
1275 AppendMaxAffineRelaxation(ct, model, relaxation);
1276 }
1277
1278 // Add cut generators.
1279 if (linearization_level > 1) {
1280 if (is_affine_max) {
1281 AddMaxAffineCutGenerator(ct, model, relaxation);
1282 } else if (ct.lin_max().exprs().size() < 100) {
1283 AddLinMaxCutGenerator(ct, model, relaxation);
1284 }
1285 }
1286 break;
1287 }
1288 case ConstraintProto::ConstraintCase::kAllDiff: {
1289 AddAllDiffRelaxationAndCutGenerator(ct, linearization_level, model,
1290 relaxation);
1291 break;
1292 }
1293 case ConstraintProto::ConstraintCase::kLinear: {
1294 AppendLinearConstraintRelaxation(
1295 ct, /*linearize_enforced_constraints=*/linearization_level > 1, model,
1296 relaxation, activity_helper);
1297 break;
1298 }
1299 case ConstraintProto::ConstraintCase::kCircuit: {
1300 AppendCircuitRelaxation(ct, model, relaxation);
1301 if (linearization_level > 1) {
1302 AddCircuitCutGenerator(ct, model, relaxation);
1303 }
1304 break;
1305 }
1306 case ConstraintProto::ConstraintCase::kRoutes: {
1307 AppendRoutesRelaxation(ct, model, relaxation);
1308 if (linearization_level > 1) {
1309 AddRoutesCutGenerator(ct, model, relaxation);
1310 }
1311 break;
1312 }
1313 case ConstraintProto::ConstraintCase::kNoOverlap: {
1314 AppendNoOverlapRelaxationAndCutGenerator(ct, model, relaxation);
1315 break;
1316 }
1317 case ConstraintProto::ConstraintCase::kCumulative: {
1318 AppendCumulativeRelaxationAndCutGenerator(ct, model, relaxation);
1319 break;
1320 }
1321 case ConstraintProto::ConstraintCase::kNoOverlap2D: {
1322 // TODO(user): Use the same pattern as the other 2 scheduling methods:
1323 // - single function
1324 // - generate helpers once
1325 //
1326 // Adds an energetic relaxation (sum of areas fits in bounding box).
1327 AppendNoOverlap2dRelaxation(ct, model, relaxation);
1328 if (linearization_level > 1) {
1329 // Adds a completion time cut generator and an energetic cut generator.
1330 AddNoOverlap2dCutGenerator(ct, model, relaxation);
1331 }
1332 break;
1333 }
1334 default: {
1335 }
1336 }
1337
1338 if (DEBUG_MODE) {
1339 const auto& integer_trail = *model->GetOrCreate<IntegerTrail>();
1340 for (int i = old_index; i < relaxation->linear_constraints.size(); ++i) {
1341 const bool issue =
1342 PossibleOverflow(integer_trail, relaxation->linear_constraints[i]);
1343 if (issue) {
1344 LOG(INFO) << "Possible overflow in linearization of: "
1345 << ct.ShortDebugString();
1346 }
1347 }
1348 }
1349}

1389void AppendSquareRelaxation(const ConstraintProto& ct, Model* m,
1390 LinearRelaxation* relaxation) {
1391 if (HasEnforcementLiteral(ct)) return;
1392 auto* mapping = m->GetOrCreate<CpModelMapping>();
1393 IntegerTrail* const integer_trail = m->GetOrCreate<IntegerTrail>();
1394
1395 // Constraint is square == x * x.
1396 AffineExpression square = mapping->Affine(ct.int_prod().target());
1397 AffineExpression x = mapping->Affine(ct.int_prod().exprs(0));
1398 IntegerValue x_lb = integer_trail->LowerBound(x);
1399 IntegerValue x_ub = integer_trail->UpperBound(x);
1400
1401 if (x_lb == x_ub) return;
1402
1403 // We currently only support variables with non-negative domains.
1404 if (x_lb < 0 && x_ub > 0) return;
1405
1406 // Change the sigh of x if its domain is non-positive.
1407 if (x_ub <= 0) {
1408 x = x.Negated();
1409 const IntegerValue tmp = x_ub;
1410 x_ub = -x_lb;
1411 x_lb = -tmp;
1412 }
1413
1414 // Check for potential overflows.
1415 if (x_ub > (int64_t{1} << 31)) return;
1416 DCHECK_GE(x_lb, 0);
1417
1418 relaxation->linear_constraints.push_back(
1419 ComputeHyperplanAboveSquare(x, square, x_lb, x_ub, m));
1420
1421 relaxation->linear_constraints.push_back(
1422 ComputeHyperplanBelowSquare(x, square, x_lb, m));
1423 // TODO(user): We could add all or some below_hyperplans.
1424 if (x_lb + 1 < x_ub) {
1425 // The hyperplan will use x_ub - 1 and x_ub.
1426 relaxation->linear_constraints.push_back(
1427 ComputeHyperplanBelowSquare(x, square, x_ub - 1, m));
1428 }

1455void AddAllDiffRelaxationAndCutGenerator(const ConstraintProto& ct,
1456 int linearization_level, Model* m,
1457 LinearRelaxation* relaxation) {
1458 if (HasEnforcementLiteral(ct)) return;
1459 auto* mapping = m->GetOrCreate<CpModelMapping>();
1460 auto* integer_trail = m->GetOrCreate<IntegerTrail>();
1461 const int num_exprs = ct.all_diff().exprs_size();
1462
1463 const std::vector<AffineExpression> exprs =
1464 mapping->Affines(ct.all_diff().exprs());
1465
1466 // Build union of affine expressions domains to check if this is a
1467 // permutation.
1468 Domain union_of_domains;
1469 for (const AffineExpression& expr : exprs) {
1470 if (integer_trail->IsFixed(expr)) {
1471 union_of_domains = union_of_domains.UnionWith(
1472 Domain(integer_trail->FixedValue(expr).value()));
1473 } else {
1474 union_of_domains = union_of_domains.UnionWith(
1475 integer_trail->InitialVariableDomain(expr.var)
1476 .MultiplicationBy(expr.coeff.value())
1477 .AdditionWith(Domain(expr.constant.value())));
1478 }
1479 }
1480
1481 if (union_of_domains.Size() == num_exprs) {
1482 // In case of a permutation, the linear constraint is tight.
1483 int64_t sum_of_values = 0;
1484 for (const int64_t v : union_of_domains.Values()) {
1485 sum_of_values += v;
1486 }
1487 LinearConstraintBuilder relax(m, sum_of_values, sum_of_values);
1488 for (const AffineExpression& expr : exprs) {
1489 relax.AddTerm(expr, 1);
1490 }
1491 relaxation->linear_constraints.push_back(relax.Build());
1492 } else if (num_exprs <=
1493 m->GetOrCreate<SatParameters>()->max_all_diff_cut_size() &&
1494 linearization_level > 1) {
1495 relaxation->cut_generators.push_back(
1496 CreateAllDifferentCutGenerator(exprs, m));

1687// If we have an exactly one between literals l_i, and each l_i => var ==
1688// value_i, then we can add a strong linear relaxation: var = sum l_i * value_i.
1689//
1690// This codes detect this and add the corresponding linear equations.
1691//
1692// TODO(user): We can do something similar with just an at most one, however
1693// it is harder to detect that if all literal are false then none of the implied
1694// value can be taken.
1695void AppendElementEncodingRelaxation(Model* m, LinearRelaxation* relaxation) {
1696 auto* integer_trail = m->GetOrCreate<IntegerTrail>();
1697 auto* element_encodings = m->GetOrCreate<ElementEncodings>();
1698
1699 int num_exactly_one_elements = 0;
1700 for (const IntegerVariable var :
1701 element_encodings->GetElementEncodedVariables()) {
1702 for (const auto& [index, literal_value_list] :
1703 element_encodings->Get(var)) {
1704 IntegerValue min_value = kMaxIntegerValue;
1705 for (const auto& literal_value : literal_value_list) {
1706 min_value = std::min(min_value, literal_value.value);
1707 }
1708
1709 LinearConstraintBuilder builder(m, -min_value, -min_value);
1710 builder.AddTerm(var, IntegerValue(-1));
1711 for (const auto& [value, literal] : literal_value_list) {
1712 const IntegerValue delta_min = value - min_value;
1713 if (delta_min != 0) {
1714 // If the term has no view, we abort.
1715 if (!builder.AddLiteralTerm(literal, delta_min)) {
1716 return;
1717 }
1718 }
1719 }
1720 ++num_exactly_one_elements;
1721 const LinearConstraint lc = builder.Build();
1722 if (!PossibleOverflow(*integer_trail, lc)) {
1723 relaxation->linear_constraints.push_back(std::move(lc));
1724 }
1725 }
1726 }
1727
1728 if (num_exactly_one_elements != 0) {
1729 auto* logger = m->GetOrCreate<SolverLogger>();
1730 SOLVER_LOG(logger,
1731 "[ElementLinearRelaxation]"
1732 " #from_exactly_one:",
1733 num_exactly_one_elements);
1734 }

1737LinearRelaxation ComputeLinearRelaxation(const CpModelProto& model_proto,
1738 Model* m) {
1739 LinearRelaxation relaxation;
1740 const SatParameters& params = *m->GetOrCreate<SatParameters>();
1741
1742 // Collect AtMostOne to compute better Big-M.
1743 ActivityBoundHelper activity_bound_helper;
1744 if (params.linearization_level() > 1) {
1745 activity_bound_helper.AddAllAtMostOnes(model_proto);
1746 }
1747
1748 // Linearize the constraints.
1749 for (const auto& ct : model_proto.constraints()) {
1750 TryToLinearizeConstraint(model_proto, ct, params.linearization_level(), m,
1751 &relaxation, &activity_bound_helper);
1752 }
1753
1754 // Linearize the encoding of variable that are fully encoded.
1755 int num_loose_equality_encoding_relaxations = 0;
1756 int num_tight_equality_encoding_relaxations = 0;
1757 int num_inequality_encoding_relaxations = 0;
1758 auto* mapping = m->GetOrCreate<CpModelMapping>();
1759 for (int i = 0; i < model_proto.variables_size(); ++i) {
1760 if (mapping->IsBoolean(i)) continue;
1761
1762 const IntegerVariable var = mapping->Integer(i);
1763 if (m->Get(IsFixed(var))) continue;
1764
1765 // We first try to linearize the values encoding.
1766 AppendRelaxationForEqualityEncoding(
1767 var, *m, &relaxation, &num_tight_equality_encoding_relaxations,
1768 &num_loose_equality_encoding_relaxations);
1769
1770 // Then we try to linearize the inequality encoding. Note that on some
1771 // problem like pizza27i.mps.gz, adding both equality and inequality
1772 // encoding is a must.
1773 //
1774 // Even if the variable is fully encoded, sometimes not all its associated
1775 // literal have a view (if they are not part of the original model for
1776 // instance).
1777 //
1778 // TODO(user): Should we add them to the LP anyway? this isn't clear as
1779 // we can sometimes create a lot of Booleans like this.
1780 const int old = relaxation.linear_constraints.size();
1781 AppendPartialGreaterThanEncodingRelaxation(var, *m, &relaxation);
1782 if (relaxation.linear_constraints.size() > old) {
1783 ++num_inequality_encoding_relaxations;
1784 }
1785 }
1786
1787 // TODO(user): This is similar to AppendRelaxationForEqualityEncoding() above.
1788 // Investigate if we can merge the code.
1789 if (params.linearization_level() >= 2) {
1790 AppendElementEncodingRelaxation(m, &relaxation);
1791 }
1792
1793 // TODO(user): I am not sure this is still needed. Investigate and explain why
1794 // or remove.
1795 if (!m->GetOrCreate<SatSolver>()->FinishPropagation()) {
1796 return relaxation;
1797 }
1798
1799 // We display the stats before linearizing the at most ones.
1800 auto* logger = m->GetOrCreate<SolverLogger>();
1801 if (num_tight_equality_encoding_relaxations != 0 ||
1802 num_loose_equality_encoding_relaxations != 0 ||
1803 num_inequality_encoding_relaxations != 0) {
1804 SOLVER_LOG(logger,
1805 "[EncodingLinearRelaxation]"
1806 " #tight_equality:",
1807 num_tight_equality_encoding_relaxations,
1808 " #loose_equality:", num_loose_equality_encoding_relaxations,
1809 " #inequality:", num_inequality_encoding_relaxations);
1810 }
1811 if (!relaxation.linear_constraints.empty() ||
1812 !relaxation.at_most_ones.empty()) {
1813 SOLVER_LOG(logger,
1814 "[LinearRelaxationBeforeCliqueExpansion]"
1815 " #linear:",
1816 relaxation.linear_constraints.size(),
1817 " #at_most_ones:", relaxation.at_most_ones.size());
1818 }
1819
1820 // Linearize the at most one constraints. Note that we transform them
1821 // into maximum "at most one" first and we removes redundant ones.
1822 m->GetOrCreate<BinaryImplicationGraph>()->TransformIntoMaxCliques(
1823 &relaxation.at_most_ones,
1824 SafeDoubleToInt64(params.merge_at_most_one_work_limit()));
1825 for (const std::vector<Literal>& at_most_one : relaxation.at_most_ones) {
1826 if (at_most_one.empty()) continue;
1827
1828 LinearConstraintBuilder lc(m, kMinIntegerValue, IntegerValue(1));
1829 for (const Literal literal : at_most_one) {
1830 // Note that it is okay to simply ignore the literal if it has no
1831 // integer view.
1832 const bool unused ABSL_ATTRIBUTE_UNUSED =
1833 lc.AddLiteralTerm(literal, IntegerValue(1));
1834 }
1835 relaxation.linear_constraints.push_back(lc.Build());
1836 }
1837
1838 // We converted all at_most_one to LP constraints, so we need to clear them
1839 // so that we don't do extra work in the connected component computation.
1840 relaxation.at_most_ones.clear();
1841
1842 // Remove size one LP constraints, they are not useful.
1843 relaxation.linear_constraints.erase(
1844 std::remove_if(
1845 relaxation.linear_constraints.begin(),
1846 relaxation.linear_constraints.end(),
1847 [](const LinearConstraint& lc) { return lc.vars.size() <= 1; }),
1848 relaxation.linear_constraints.end());
1849
1850 // We add a clique cut generation over all Booleans of the problem.
1851 // Note that in practice this might regroup independent LP together.
1852 //
1853 // TODO(user): compute connected components of the original problem and
1854 // split these cuts accordingly.
1855 if (params.linearization_level() > 1 && params.add_clique_cuts()) {
1856 LinearConstraintBuilder builder(m);
1857 for (int i = 0; i < model_proto.variables_size(); ++i) {
1858 if (!mapping->IsBoolean(i)) continue;
1859
1860 // Note that it is okay to simply ignore the literal if it has no
1861 // integer view.
1862 const bool unused ABSL_ATTRIBUTE_UNUSED =
1863 builder.AddLiteralTerm(mapping->Literal(i), IntegerValue(1));
1864 }
1865
1866 // We add a generator touching all the variable in the builder.
1867 const LinearExpression& expr = builder.BuildExpression();
1868 if (!expr.vars.empty()) {
1869 relaxation.cut_generators.push_back(
1870 CreateCliqueCutGenerator(expr.vars, m));
1871 }
1872 }
1873
1874 if (!relaxation.linear_constraints.empty() ||
1875 !relaxation.cut_generators.empty()) {
1876 SOLVER_LOG(logger,
1877 "[FinalLinearRelaxation]"
1878 " #linear:",
1879 relaxation.linear_constraints.size(),
1880 " #cut_generators:", relaxation.cut_generators.size());
1881 }
1882
1883 return relaxation;
1884}
