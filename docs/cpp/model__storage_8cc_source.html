48template <typename K, typename V>
49std::vector<K> MapKeys(const absl::flat_hash_map<K, V>& in_map) {
50 std::vector<K> keys;
51 keys.reserve(in_map.size());
52 for (const auto& key_pair : in_map) {
53 keys.push_back(key_pair.first);
54 }
55 return keys;
56}
57
58template <typename K, typename V>
59std::vector<K> SortedMapKeys(const absl::flat_hash_map<K, V>& in_map) {
60 std::vector<K> keys = MapKeys(in_map);
61 std::sort(keys.begin(), keys.end());
62 return keys;
63}
64
65template <typename T>
66std::vector<T> SortedSetKeys(const absl::flat_hash_set<T>& in_set) {
67 std::vector<T> keys;
68 keys.reserve(in_set.size());
69 for (const auto& key : in_set) {
70 keys.push_back(key);
71 }
72 std::sort(keys.begin(), keys.end());
73 return keys;
74}

76// ids should be sorted.
77template <typename IdType>
78void AppendFromMapOrDefault(const absl::Span<const IdType> ids,
79 const absl::flat_hash_map<IdType, double>& values,
80 SparseDoubleVectorProto& sparse_vector) {
81 for (const IdType id : ids) {
82 sparse_vector.add_ids(id.value());
83 sparse_vector.add_values(gtl::FindWithDefault(values, id));
84 }
85}
86
87// ids should be sorted.
88template <typename IdType, typename IdIterable>
89void AppendFromMapIfPresent(const IdIterable& ids,
90 const absl::flat_hash_map<IdType, double>& values,
91 SparseDoubleVectorProto& sparse_vector) {
92 for (const IdType id : ids) {
93 const double* const double_value = gtl::FindOrNull(values, id);
94 if (double_value != nullptr) {
95 sparse_vector.add_ids(id.value());
96 sparse_vector.add_values(*double_value);
97 }
98 }
99}
100
101template <typename IdType, typename DataType>
102void AppendFromMap(const absl::flat_hash_set<IdType>& dirty_keys,
103 const absl::flat_hash_map<IdType, DataType>& values,
104 double DataType::*field,
105 SparseDoubleVectorProto& sparse_vector) {
106 for (const IdType id : SortedSetKeys(dirty_keys)) {
107 sparse_vector.add_ids(id.value());
108 sparse_vector.add_values(values.at(id).*field);
109 }
110}
111
112// If an element in keys is not found in coefficients, it is set to 0.0 in
113// matrix. Keys must be in lexicographic ordering (i.e. sorted).
114// NOTE: This signature can be updated to take a Span instead of a vector if
115// needed in the future, but it required specifying parameters at the callsites.
116template <typename RK, typename CK>
117SparseDoubleMatrixProto ExportMatrix(
118 const absl::flat_hash_map<std::pair<RK, CK>, double>& coefficients,
119 const std::vector<std::pair<RK, CK>>& keys) {
120 SparseDoubleMatrixProto matrix;
121 matrix.mutable_row_ids()->Reserve(keys.size());
122 matrix.mutable_column_ids()->Reserve(keys.size());
123 matrix.mutable_coefficients()->Reserve(keys.size());
124 for (const auto [row_id, column_id] : keys) {
125 matrix.add_row_ids(row_id.value());
126 matrix.add_column_ids(column_id.value());
127 matrix.add_coefficients(
128 gtl::FindWithDefault(coefficients, {row_id, column_id}));
129 }
130 return matrix;
131}

135absl::StatusOr<std::unique_ptr<ModelStorage>> ModelStorage::FromModelProto(
136 const ModelProto& model_proto) {
137 // We don't check names since ModelStorage does not do so before exporting
138 // models. Thus a model built by ModelStorage can contain duplicated
139 // names. And since we use FromModelProto() to implement Clone(), we must make
140 // sure duplicated names don't fail.
141 RETURN_IF_ERROR(ValidateModel(model_proto, /*check_names=*/false).status());
142
143 auto storage = std::make_unique<ModelStorage>(model_proto.name());
144
145 // Add variables.
146 storage->AddVariables(model_proto.variables());
147
148 // Set the objective.
149 storage->set_is_maximize(model_proto.objective().maximize());
150 storage->set_objective_offset(model_proto.objective().offset());
151 storage->UpdateLinearObjectiveCoefficients(
152 model_proto.objective().linear_coefficients());
153 storage->UpdateQuadraticObjectiveCoefficients(
154 model_proto.objective().quadratic_coefficients());
155
156 // Add linear constraints.
157 storage->AddLinearConstraints(model_proto.linear_constraints());
158
159 // Set the linear constraints coefficients.
160 storage->UpdateLinearConstraintCoefficients(
161 model_proto.linear_constraint_matrix());
162
163 return storage;
164}

166void ModelStorage::UpdateLinearObjectiveCoefficients(
167 const SparseDoubleVectorProto& coefficients) {
168 for (const auto [var_id, value] : MakeView(coefficients)) {
169 set_linear_objective_coefficient(VariableId(var_id), value);
170 }
171}
172
173void ModelStorage::UpdateQuadraticObjectiveCoefficients(
174 const SparseDoubleMatrixProto& coefficients) {
175 for (int i = 0; i < coefficients.row_ids_size(); ++i) {
176 // This call is valid since this is an upper triangular matrix; there is no
177 // duplicated terms.
178 set_quadratic_objective_coefficient(VariableId(coefficients.row_ids(i)),
179 VariableId(coefficients.column_ids(i)),
180 coefficients.coefficients(i));
181 }
182}
183
184void ModelStorage::UpdateLinearConstraintCoefficients(
185 const SparseDoubleMatrixProto& coefficients) {
186 for (int i = 0; i < coefficients.row_ids_size(); ++i) {
187 // This call is valid since there are no duplicated pairs.
188 set_linear_constraint_coefficient(
189 LinearConstraintId(coefficients.row_ids(i)),
190 VariableId(coefficients.column_ids(i)), coefficients.coefficients(i));
191 }
192}

194std::unique_ptr<ModelStorage> ModelStorage::Clone(
195 const std::optional<absl::string_view> new_name) const {
196 ModelProto model_proto = ExportModel();
197 if (new_name.has_value()) {
198 model_proto.set_name(std::string(*new_name));
199 }
200 absl::StatusOr<std::unique_ptr<ModelStorage>> clone =
201 ModelStorage::FromModelProto(model_proto);
202 // Unless there is a very serious bug, a model exported by ExportModel()
203 // should always be valid.
204 CHECK_OK(clone.status());
205
206 // Update the next ids so that the clone does not reused any deleted id from
207 // the original.
208 CHECK_LE(clone.value()->next_variable_id_, next_variable_id_);
209 clone.value()->next_variable_id_ = next_variable_id_;
210 CHECK_LE(clone.value()->next_linear_constraint_id_,
211 next_linear_constraint_id_);
212 clone.value()->next_linear_constraint_id_ = next_linear_constraint_id_;
213
214 return std::move(clone).value();
215}

230void ModelStorage::AddVariableInternal(const VariableId id,
231 const double lower_bound,
232 const double upper_bound,
233 const bool is_integer,
234 const absl::string_view name) {
235 CHECK_GE(id, next_variable_id_);
236 next_variable_id_ = id + VariableId(1);
237
238 VariableData& var_data = variables_[id];
239 var_data.lower_bound = lower_bound;
240 var_data.upper_bound = upper_bound;
241 var_data.is_integer = is_integer;
242 var_data.name = std::string(name);
243 gtl::InsertOrDie(&matrix_columns_, id, {});
244}

246void ModelStorage::AddVariables(const VariablesProto& variables) {
247 const bool has_names = !variables.names().empty();
248 for (int v = 0; v < variables.ids_size(); ++v) {
249 // This call is valid since ids are unique and increasing.
250 AddVariableInternal(VariableId(variables.ids(v)),
251 /*lower_bound=*/variables.lower_bounds(v),
252 /*upper_bound=*/variables.upper_bounds(v),
253 /*is_integer=*/variables.integers(v),
254 has_names ? variables.names(v) : absl::string_view());
255 }
256}

258void ModelStorage::DeleteVariable(const VariableId id) {
259 CHECK(variables_.contains(id));
260 linear_objective_.erase(id);
261 for (const auto& [_, update_tracker] :
262 update_trackers_.GetUpdatedTrackers()) {
263 if (id < update_tracker->variables_checkpoint) {
264 update_tracker->dirty_variable_deletes.insert(id);
265 update_tracker->dirty_variable_lower_bounds.erase(id);
266 update_tracker->dirty_variable_upper_bounds.erase(id);
267 update_tracker->dirty_variable_is_integer.erase(id);
268 update_tracker->dirty_linear_objective_coefficients.erase(id);
269 for (const VariableId related :
270 quadratic_objective_.RelatedVariables(id)) {
271 // We can only have a dirty update to wipe clean if both variables are
272 // old
273 if (related < update_tracker->variables_checkpoint) {
274 update_tracker->dirty_quadratic_objective_coefficients.erase(
275 internal::MakeOrderedPair(id, related));
276 }
277 }
278 }
279 }
280
281 quadratic_objective_.Delete(id);
282 for (const LinearConstraintId related_constraint : matrix_columns_.at(id)) {
283 CHECK_GT(matrix_rows_.at(related_constraint).erase(id), 0);
284 CHECK_GT(linear_constraint_matrix_.erase({related_constraint, id}), 0);
285 }
286 for (const auto& [_, update_tracker] :
287 update_trackers_.GetUpdatedTrackers()) {
288 for (const LinearConstraintId related_constraint : matrix_columns_.at(id)) {
289 if (id < update_tracker->variables_checkpoint &&
290 related_constraint < update_tracker->linear_constraints_checkpoint) {
291 update_tracker->dirty_linear_constraint_matrix_keys.erase(
292 {related_constraint, id});
293 }
294 }
295 }
296 CHECK_GT(matrix_columns_.erase(id), 0);
297 variables_.erase(id);
298}

318void ModelStorage::AddLinearConstraintInternal(const LinearConstraintId id,
319 const double lower_bound,
320 const double upper_bound,
321 const absl::string_view name) {
322 CHECK_GE(id, next_linear_constraint_id_);
323 next_linear_constraint_id_ = id + LinearConstraintId(1);
324
325 LinearConstraintData& lin_con_data = linear_constraints_[id];
326 lin_con_data.lower_bound = lower_bound;
327 lin_con_data.upper_bound = upper_bound;
328 lin_con_data.name = std::string(name);
329 gtl::InsertOrDie(&matrix_rows_, id, {});
330}

332void ModelStorage::AddLinearConstraints(
333 const LinearConstraintsProto& linear_constraints) {
334 const bool has_names = !linear_constraints.names().empty();
335 for (int c = 0; c < linear_constraints.ids_size(); ++c) {
336 // This call is valid since ids are unique and increasing.
337 AddLinearConstraintInternal(
338 LinearConstraintId(linear_constraints.ids(c)),
339 /*lower_bound=*/linear_constraints.lower_bounds(c),
340 /*upper_bound=*/linear_constraints.upper_bounds(c),
341 has_names ? linear_constraints.names(c) : absl::string_view());
342 }
343}
344
345void ModelStorage::DeleteLinearConstraint(const LinearConstraintId id) {
346 CHECK(linear_constraints_.contains(id));
347 linear_constraints_.erase(id);
348 for (const auto& [_, update_tracker] :
349 update_trackers_.GetUpdatedTrackers()) {
350 if (id < update_tracker->linear_constraints_checkpoint) {
351 update_tracker->dirty_linear_constraint_deletes.insert(id);
352 update_tracker->dirty_linear_constraint_lower_bounds.erase(id);
353 update_tracker->dirty_linear_constraint_upper_bounds.erase(id);
354 }
355 }
356 for (const VariableId related_variable : matrix_rows_.at(id)) {
357 CHECK_GT(matrix_columns_.at(related_variable).erase(id), 0);
358 CHECK_GT(linear_constraint_matrix_.erase({id, related_variable}), 0);
359 }
360 for (const auto& [_, update_tracker] :
361 update_trackers_.GetUpdatedTrackers()) {
362 for (const VariableId related_variable : matrix_rows_.at(id)) {
363 if (id < update_tracker->linear_constraints_checkpoint &&
364 related_variable < update_tracker->variables_checkpoint) {
365 update_tracker->dirty_linear_constraint_matrix_keys.erase(
366 {id, related_variable});
367 }
368 }
369 }
370 CHECK_GT(matrix_rows_.erase(id), 0);
371}

406ModelProto ModelStorage::ExportModel() const {
407 ModelProto result;
408 result.set_name(name_);
409 // Export the variables.
410 for (const VariableId variable : SortedMapKeys(variables_)) {
411 AppendVariable(variable, *result.mutable_variables());
412 }
413
414 // Pull out the objective.
415 result.mutable_objective()->set_maximize(is_maximize_);
416 result.mutable_objective()->set_offset(objective_offset_);
417 AppendFromMapOrDefault<VariableId>(
418 SortedMapKeys(linear_objective_), linear_objective_,
419 *result.mutable_objective()->mutable_linear_coefficients());
420 *result.mutable_objective()->mutable_quadratic_coefficients() =
421 quadratic_objective_.Proto();
422
423 // Pull out the linear constraints.
424 for (const LinearConstraintId con : SortedMapKeys(linear_constraints_)) {
425 AppendLinearConstraint(con, *result.mutable_linear_constraints());
426 }
427
428 // Pull out the constraint matrix.
429 *result.mutable_linear_constraint_matrix() =
430 ExportMatrix<LinearConstraintId, VariableId>(
431 linear_constraint_matrix_, SortedMapKeys(linear_constraint_matrix_));
432 return result;
433}

435std::optional<ModelUpdateProto>
436ModelStorage::UpdateTrackerData::ExportModelUpdate(
437 const ModelStorage& storage) const {
438 // We must detect the empty case to prevent unneeded copies and merging in
439 // ExportModelUpdate().
440 if (variables_checkpoint == storage.next_variable_id_ &&
441 linear_constraints_checkpoint == storage.next_linear_constraint_id_ &&
442 !dirty_objective_direction && !dirty_objective_offset &&
443 dirty_variable_deletes.empty() && dirty_variable_lower_bounds.empty() &&
444 dirty_variable_upper_bounds.empty() &&
445 dirty_variable_is_integer.empty() &&
446 dirty_linear_objective_coefficients.empty() &&
447 dirty_quadratic_objective_coefficients.empty() &&
448 dirty_linear_constraint_deletes.empty() &&
449 dirty_linear_constraint_lower_bounds.empty() &&
450 dirty_linear_constraint_upper_bounds.empty() &&
451 dirty_linear_constraint_matrix_keys.empty()) {
452 return std::nullopt;
453 }
454
455 ModelUpdateProto result;
456
457 // Variable/constraint deletions.
458 for (const VariableId del_var : SortedSetKeys(dirty_variable_deletes)) {
459 result.add_deleted_variable_ids(del_var.value());
460 }
461 for (const LinearConstraintId del_lin_con :
462 SortedSetKeys(dirty_linear_constraint_deletes)) {
463 result.add_deleted_linear_constraint_ids(del_lin_con.value());
464 }
465
466 // Update the variables.
467 auto var_updates = result.mutable_variable_updates();
468 AppendFromMap(dirty_variable_lower_bounds, storage.variables_,
469 &VariableData::lower_bound,
470 *var_updates->mutable_lower_bounds());
471 AppendFromMap(dirty_variable_upper_bounds, storage.variables_,
472 &VariableData::upper_bound,
473 *var_updates->mutable_upper_bounds());
474
475 for (const VariableId integer_var :
476 SortedSetKeys(dirty_variable_is_integer)) {
477 var_updates->mutable_integers()->add_ids(integer_var.value());
478 var_updates->mutable_integers()->add_values(
479 storage.variables_.at(integer_var).is_integer);
480 }
481 for (VariableId new_id = variables_checkpoint;
482 new_id < storage.next_variable_id_; ++new_id) {
483 if (storage.variables_.contains(new_id)) {
484 storage.AppendVariable(new_id, *result.mutable_new_variables());
485 }
486 }
487
488 // Update the objective
489 auto obj_updates = result.mutable_objective_updates();
490 if (dirty_objective_direction) {
491 obj_updates->set_direction_update(storage.is_maximize_);
492 }
493 if (dirty_objective_offset) {
494 obj_updates->set_offset_update(storage.objective_offset_);
495 }
496 AppendFromMapOrDefault<VariableId>(
497 SortedSetKeys(dirty_linear_objective_coefficients),
498 storage.linear_objective_, *obj_updates->mutable_linear_coefficients());
499 // TODO(b/182567749): Once StrongInt is in absl, use
500 // AppendFromMapIfPresent<VariableId>(
501 // MakeStrongIntRange(variables_checkpoint, next_variable_id_),
502 // linear_objective_, *obj_updates->mutable_linear_coefficients());
503 for (VariableId var_id = variables_checkpoint;
504 var_id < storage.next_variable_id_; ++var_id) {
505 const double* const double_value =
506 gtl::FindOrNull(storage.linear_objective_, var_id);
507 if (double_value != nullptr) {
508 obj_updates->mutable_linear_coefficients()->add_ids(var_id.value());
509 obj_updates->mutable_linear_coefficients()->add_values(*double_value);
510 }
511 }
512
513 *result.mutable_objective_updates()->mutable_quadratic_coefficients() =
514 storage.quadratic_objective_.Update(
515 storage.variables_, variables_checkpoint, storage.next_variable_id_,
516 dirty_quadratic_objective_coefficients);
517
518 // Update the linear constraints
519 auto lin_con_updates = result.mutable_linear_constraint_updates();
520 AppendFromMap(dirty_linear_constraint_lower_bounds,
521 storage.linear_constraints_, &LinearConstraintData::lower_bound,
522 *lin_con_updates->mutable_lower_bounds());
523 AppendFromMap(dirty_linear_constraint_upper_bounds,
524 storage.linear_constraints_, &LinearConstraintData::upper_bound,
525 *lin_con_updates->mutable_upper_bounds());
526
527 for (LinearConstraintId new_id = linear_constraints_checkpoint;
528 new_id < storage.next_linear_constraint_id_; ++new_id) {
529 if (storage.linear_constraints_.contains(new_id)) {
530 storage.AppendLinearConstraint(new_id,
531 *result.mutable_new_linear_constraints());
532 }
533 }
534
535 // Extract changes to the matrix of linear constraint coefficients
536 std::vector<std::pair<LinearConstraintId, VariableId>>
537 constraint_matrix_updates(dirty_linear_constraint_matrix_keys.begin(),
538 dirty_linear_constraint_matrix_keys.end());
539 for (VariableId new_var = variables_checkpoint;
540 new_var < storage.next_variable_id_; ++new_var) {
541 if (storage.variables_.contains(new_var)) {
542 for (const LinearConstraintId lin_con :
543 storage.matrix_columns_.at(new_var)) {
544 constraint_matrix_updates.emplace_back(lin_con, new_var);
545 }
546 }
547 }
548 for (LinearConstraintId new_lin_con = linear_constraints_checkpoint;
549 new_lin_con < storage.next_linear_constraint_id_; ++new_lin_con) {
550 if (storage.linear_constraints_.contains(new_lin_con)) {
551 for (const VariableId var : storage.matrix_rows_.at(new_lin_con)) {
552 // NOTE(user): we will do at most twice as much as needed here.
553 if (var < variables_checkpoint) {
554 constraint_matrix_updates.emplace_back(new_lin_con, var);
555 }
556 }
557 }
558 }
559 std::sort(constraint_matrix_updates.begin(), constraint_matrix_updates.end());
560 *result.mutable_linear_constraint_matrix_updates() = ExportMatrix(
561 storage.linear_constraint_matrix_, constraint_matrix_updates);
562
563 // Named returned value optimization (NRVO) does not apply here since the
564 // return type if not the same type as `result`. To make things clear, we
565 // explicitly call the constructor here.
566 return {std::move(result)};
567}

569void ModelStorage::UpdateTrackerData::Checkpoint(const ModelStorage& storage) {
570 variables_checkpoint = storage.next_variable_id_;
571 linear_constraints_checkpoint = storage.next_linear_constraint_id_;
572 dirty_objective_direction = false;
573 dirty_objective_offset = false;
574
575 dirty_variable_deletes.clear();
576 dirty_variable_lower_bounds.clear();
577 dirty_variable_upper_bounds.clear();
578 dirty_variable_is_integer.clear();
579
580 dirty_linear_objective_coefficients.clear();
581 dirty_quadratic_objective_coefficients.clear();
582
583 dirty_linear_constraint_deletes.clear();
584 dirty_linear_constraint_lower_bounds.clear();
585 dirty_linear_constraint_upper_bounds.clear();
586 dirty_linear_constraint_matrix_keys.clear();
587}

608absl::Status ModelStorage::ApplyUpdateProto(
609 const ModelUpdateProto& update_proto) {
610 // Check the update first.
611 {
612 // Do not check for duplicate names, as with FromModelProto();
613 ModelSummary summary(/*check_names=*/false);
614 // IdNameBiMap requires Insert() calls to be in sorted id order.
615 for (const auto id : SortedVariables()) {
616 RETURN_IF_ERROR(summary.variables.Insert(id.value(), variable_name(id)))
617 << "invalid variable id in model";
618 }
619 RETURN_IF_ERROR(summary.variables.SetNextFreeId(next_variable_id_.value()));
620 for (const auto id : SortedLinearConstraints()) {
621 RETURN_IF_ERROR(summary.linear_constraints.Insert(
622 id.value(), linear_constraint_name(id)))
623 << "invalid linear constraint id in model";
624 }
625 RETURN_IF_ERROR(summary.linear_constraints.SetNextFreeId(
626 next_linear_constraint_id_.value()));
627 RETURN_IF_ERROR(ValidateModelUpdate(update_proto, summary))
628 << "update not valid";
629 // TODO(b/227217735): Remove if block when ModelStorage supports quadratic
630 // constraints.
631 if (update_proto.has_quadratic_constraint_updates()) {
632 return absl::UnimplementedError(
633 "quadratic constraint updates are not currently supported");
634 }
635 }
636
637 // Remove deleted variables and constraints.
638 for (const int64_t v_id : update_proto.deleted_variable_ids()) {
639 DeleteVariable(VariableId(v_id));
640 }
641 for (const int64_t c_id : update_proto.deleted_linear_constraint_ids()) {
642 DeleteLinearConstraint(LinearConstraintId(c_id));
643 }
644
645 // Update existing variables' properties.
646 for (const auto [v_id, lb] :
647 MakeView(update_proto.variable_updates().lower_bounds())) {
648 set_variable_lower_bound(VariableId(v_id), lb);
649 }
650 for (const auto [v_id, ub] :
651 MakeView(update_proto.variable_updates().upper_bounds())) {
652 set_variable_upper_bound(VariableId(v_id), ub);
653 }
654 for (const auto [v_id, is_integer] :
655 MakeView(update_proto.variable_updates().integers())) {
656 set_variable_is_integer(VariableId(v_id), is_integer);
657 }
658
659 // Update existing constraints' properties.
660 for (const auto [c_id, lb] :
661 MakeView(update_proto.linear_constraint_updates().lower_bounds())) {
662 set_linear_constraint_lower_bound(LinearConstraintId(c_id), lb);
663 }
664 for (const auto [c_id, ub] :
665 MakeView(update_proto.linear_constraint_updates().upper_bounds())) {
666 set_linear_constraint_upper_bound(LinearConstraintId(c_id), ub);
667 }
668
669 // Add the new variables and constraints.
670 AddVariables(update_proto.new_variables());
671 AddLinearConstraints(update_proto.new_linear_constraints());
672
673 // Update the objective.
674 if (update_proto.objective_updates().has_direction_update()) {
675 set_is_maximize(update_proto.objective_updates().direction_update());
676 }
677 if (update_proto.objective_updates().has_offset_update()) {
678 set_objective_offset(update_proto.objective_updates().offset_update());
679 }
680 UpdateLinearObjectiveCoefficients(
681 update_proto.objective_updates().linear_coefficients());
682 UpdateQuadraticObjectiveCoefficients(
683 update_proto.objective_updates().quadratic_coefficients());
684
685 // Update the linear constraints' coefficients.
686 UpdateLinearConstraintCoefficients(
687 update_proto.linear_constraint_matrix_updates());
688
689 return absl::OkStatus();
690}
