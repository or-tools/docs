63// Returns true if the equation a * X + b * Y = cte has some integer solutions.
64// For now, we check that a and b are different from 0 and from int64_t min.
65//
66// There is actually always a solution if cte % gcd(|a|, |b|) == 0. And because
67// a, b and cte fit on an int64_t, if there is a solution, there is one with X
68// and Y fitting on an int64_t.
69//
70// We will divide everything by gcd(a, b) first, so it is why we take reference
71// and the equation can change.
72//
73// If there are solutions, we return one of them (x0, y0).
74// From any such solution, the set of all solutions is given for Z integer by:
75// X = x0 + b * Z;
76// Y = y0 - a * Z;
77//
78// Given a domain for X and Y, it is possible to compute the "exact" domain of Z
79// with our Domain functions. Note however that this will only compute solution
80// where both x-x0 and y-y0 do fit on an int64_t:
81// DomainOf(x).SubtractionWith(x0).InverseMultiplicationBy(b).IntersectionWith(
82// DomainOf(y).SubtractionWith(y0).InverseMultiplicationBy(-a))
83bool SolveDiophantineEquationOfSizeTwo(int64_t& a, int64_t& b, int64_t& cte,
84 int64_t& x0, int64_t& y0);

102// Given a linear equation "sum coeff_i * X_i <= rhs. We can rewrite it using
103// ClosestMultiple() as "base * new_terms + error <= rhs" where error can be
104// bounded using the provided bounds on each variables. This will return true if
105// the error can be ignored and this equation is completely equivalent to
106// new_terms <= new_rhs.
107//
108// This is useful for cases like 9'999 X + 10'0001 Y <= 155'000 where we have
109// weird coefficient (maybe due to scaling). With a base of 10K, this is
110// equivalent to X + Y <= 15.
111//
112// Preconditions: All coeffs are assumed to be positive. You can easily negate
113// all the negative coeffs and corresponding bounds before calling this.
114bool LinearInequalityCanBeReducedWithClosestMultiple(
115 int64_t base, const std::vector<int64_t>& coeffs,
116 const std::vector<int64_t>& lbs, const std::vector<int64_t>& ubs,
117 int64_t rhs, int64_t* new_rhs);

119// The model "singleton" random engine used in the solver.
120//
121// In test, we usually set use_absl_random() so that the sequence is changed at
122// each invocation. This way, clients do not relly on the wrong assumption that
123// a particular optimal solution will be returned if they are many equivalent
124// ones.
125class ModelRandomGenerator : public absl::BitGenRef {
126 public:
127 // We seed the strategy at creation only. This should be enough for our use
128 // case since the SatParameters is set first before the solver is created. We
129 // also never really need to change the seed afterwards, it is just used to
130 // diversify solves with identical parameters on different Model objects.
131 explicit ModelRandomGenerator(Model* model)
132 : absl::BitGenRef(deterministic_random_) {
133 const auto& params = *model->GetOrCreate<SatParameters>();
134 deterministic_random_.seed(params.random_seed());
135 if (params.use_absl_random()) {
136 absl_random_ = absl::BitGen(absl::SeedSeq({params.random_seed()}));
137 absl::BitGenRef::operator=(absl::BitGenRef(absl_random_));
138 }
139 }

164// Context: this function is not really generic, but required to be unit-tested.
165// It is used in a clause minimization algorithm when we try to detect if any of
166// the clause literals can be propagated by a subset of the other literal being
167// false. For that, we want to enqueue in the solver all the subset of size n-1.
168//
169// This moves one of the unprocessed literal from literals to the last position.
170// The function tries to do that while preserving the longest possible prefix of
171// literals "amortized" through the calls assuming that we want to move each
172// literal to the last position once.
173//
174// For a vector of size n, if we want to call this n times so that each literal
175// is last at least once, the sum of the size of the changed suffixes will be
176// O(n log n). If we were to use a simpler algorithm (like moving the last
177// unprocessed literal to the last position), this sum would be O(n^2).
178//
179// Returns the size of the common prefix of literals before and after the move,
180// or -1 if all the literals are already processed. The argument
181// relevant_prefix_size is used as a hint when keeping more that this prefix
182// size do not matter. The returned value will always be lower or equal to
183// relevant_prefix_size.
184int MoveOneUnprocessedLiteralLast(
185 const absl::btree_set<LiteralIndex>& processed, int relevant_prefix_size,
186 std::vector<Literal>* literals);

372inline int64_t SafeDoubleToInt64(double value) {
373 if (std::isnan(value)) return 0;
374 if (value >= static_cast<double>(std::numeric_limits<int64_t>::max())) {
375 return std::numeric_limits<int64_t>::max();
376 }
377 if (value <= static_cast<double>(std::numeric_limits<int64_t>::min())) {
378 return std::numeric_limits<int64_t>::min();
379 }
380 return static_cast<int64_t>(value);
381}
