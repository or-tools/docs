58// Small utility class to store a vector<vector<>> where one can only append new
59// vector and never change previously added ones. This allows to store a static
60// key -> value(s) mapping.
61//
62// This is a lot more compact memorywise and thus faster than vector<vector<>>.
63// Note that we implement a really small subset of the vector<vector<>> API.
64//
65// We support int and StrongType for key K and any copyable type for value V.
66template <typename K = int, typename V = int>
67class CompactVectorVector {
68 public:
69 // Size of the "key" space, always in [0, size()).
70 size_t size() const;
71 bool empty() const;
72
73 // Getters, either via [] or via a wrapping to be compatible with older api.
74 //
75 // Warning: Spans are only valid until the next modification!
76 absl::Span<V> operator[](K key);
77 absl::Span<const V> operator[](K key) const;
78 std::vector<absl::Span<const V>> AsVectorOfSpan() const;
79
80 // Restore to empty vector<vector<>>.
81 void clear();
82
83 // Reserve memory if it is already known or tightly estimated.
84 void reserve(int size) {
85 starts_.reserve(size);
86 sizes_.reserve(size);
87 }
88 void reserve(int size, int num_terms) {
89 reserve(size);
90 buffer_.reserve(num_terms);
91 }

93 // Given a flat mapping (keys[i] -> values[i]) with two parallel vectors, not
94 // necessarily sorted by key, regroup the same key so that
95 // CompactVectorVector[key] list all values in the order in which they appear.
96 //
97 // We only check keys.size(), so this can be used with IdentityMap() as
98 // second argument.
99 template <typename Keys, typename Values>
100 void ResetFromFlatMapping(Keys keys, Values values);
101
102 // Initialize this vector from the transpose of another.
103 // IMPORTANT: This cannot be called with the vector itself.
104 //
105 // If min_transpose_size is given, then the transpose will have at least this
106 // size even if some of the last keys do not appear in other.
107 //
108 // If this is called twice in a row, then it has the side effect of sorting
109 // all inner vectors by values !
110 void ResetFromTranspose(const CompactVectorVector<V, K>& other,
111 int min_transpose_size = 0);
112
113 // Append a new entry.
114 // Returns the previous size() as this is convenient for how we use it.
115 int Add(absl::Span<const V> values);
116
117 // Hacky: same as Add() but for sat::Literal or any type from which we can get
118 // a value type V via L.Index().value().
119 template <typename L>
120 int AddLiterals(const std::vector<L>& wrapped_values);
121
122 // We lied when we said this is a pure read-only class :)
123 // It is possible to shrink inner vectors with not much cost.
124 //
125 // Removes the element at index from this[key] by swapping it with
126 // this[key].back() and then decreasing this key size. It is an error to
127 // call this on an empty inner vector.
128 void RemoveBySwap(K key, int index) {
129 DCHECK_GE(index, 0);
130 DCHECK_LT(index, sizes_[key]);
131 const int start = starts_[key];
132 std::swap(buffer_[start + index], buffer_[start + sizes_[key] - 1]);
133 sizes_[key]--;
134 }
135
136 // Replace the values at the given key.
137 // This will crash if there are more values than before.
138 void ReplaceValuesBySmallerSet(K key, absl::Span<const V> values);
139
140 // Interface so this can be used as an output of
141 // FindStronglyConnectedComponents().
142 void emplace_back(V const* begin, V const* end) {
143 Add(absl::MakeSpan(begin, end - begin));
144 }
145
146 private:
147 // Convert int and StrongInt to normal int.
148 static int InternalKey(K key);
149
150 std::vector<int> starts_;
151 std::vector<int> sizes_;
152 std::vector<V> buffer_;
153};

191// Returns true if the equation a * X + b * Y = cte has some integer solutions.
192// For now, we check that a and b are different from 0 and from int64_t min.
193//
194// There is actually always a solution if cte % gcd(|a|, |b|) == 0. And because
195// a, b and cte fit on an int64_t, if there is a solution, there is one with X
196// and Y fitting on an int64_t.
197//
198// We will divide everything by gcd(a, b) first, so it is why we take reference
199// and the equation can change.
200//
201// If there are solutions, we return one of them (x0, y0).
202// From any such solution, the set of all solutions is given for Z integer by:
203// X = x0 + b * Z;
204// Y = y0 - a * Z;
205//
206// Given a domain for X and Y, it is possible to compute the "exact" domain of Z
207// with our Domain functions. Note however that this will only compute solution
208// where both x-x0 and y-y0 do fit on an int64_t:
209// DomainOf(x).SubtractionWith(x0).InverseMultiplicationBy(b).IntersectionWith(
210// DomainOf(y).SubtractionWith(y0).InverseMultiplicationBy(-a))
211bool SolveDiophantineEquationOfSizeTwo(int64_t& a, int64_t& b, int64_t& cte,
212 int64_t& x0, int64_t& y0);

241// Given a linear equation "sum coeff_i * X_i <= rhs. We can rewrite it using
242// ClosestMultiple() as "base * new_terms + error <= rhs" where error can be

245// new_terms <= new_rhs.

255 absl::Span<const int64_t> lbs, absl::Span<const int64_t> ubs, int64_t rhs,

310// Context: this function is not really generic, but required to be unit-tested.
311// It is used in a clause minimization algorithm when we try to detect if any of
312// the clause literals can be propagated by a subset of the other literal being
313// false. For that, we want to enqueue in the solver all the subset of size n-1.
314//
315// This moves one of the unprocessed literal from literals to the last position.
316// The function tries to do that while preserving the longest possible prefix of
317// literals "amortized" through the calls assuming that we want to move each
318// literal to the last position once.
319//
320// For a vector of size n, if we want to call this n times so that each literal
321// is last at least once, the sum of the size of the changed suffixes will be
322// O(n log n). If we were to use a simpler algorithm (like moving the last
323// unprocessed literal to the last position), this sum would be O(n^2).
324//
325// Returns the size of the common prefix of literals before and after the move,
326// or -1 if all the literals are already processed. The argument
327// relevant_prefix_size is used as a hint when keeping more that this prefix
328// size do not matter. The returned value will always be lower or equal to
329// relevant_prefix_size.
330int MoveOneUnprocessedLiteralLast(
331 const absl::btree_set<LiteralIndex>& processed, int relevant_prefix_size,
332 std::vector<Literal>* literals);

385// Simple DP to keep the set of the first n reachable value (n > 1).
386//
387// TODO(user): Maybe modulo some prime number we can keep more info.
388// TODO(user): Another common case is a bunch of really small values and larger
389// ones, so we could bound the sum of the small values and keep the first few
390// reachable by the big ones. This is similar to some presolve transformations.
391template <int n>
392class FirstFewValues {
393 public:
394 FirstFewValues() { Reset(); }
395
396 void Reset() {
397 reachable_.fill(std::numeric_limits<int64_t>::max());
398 reachable_[0] = 0;
399 new_reachable_[0] = 0;
400 }
401
402 // We assume the given positive value can be added as many time as wanted.
403 //
404 // TODO(user): Implement Add() with an upper bound on the multiplicity.
405 void Add(const int64_t positive_value) {
406 DCHECK_GT(positive_value, 0);
407 if (positive_value >= reachable_.back()) return;
408
409 // We copy from reachable_[i] to new_reachable_[j].
410 // The position zero is already copied.
411 int i = 1;
412 int j = 1;
413 for (int base = 0; j < n && base < n; ++base) {
414 const int64_t candidate = CapAdd(new_reachable_[base], positive_value);
415 while (j < n && i < n && reachable_[i] < candidate) {
416 new_reachable_[j++] = reachable_[i++];
417 }
418 if (j < n) {
419 // Eliminate duplicates.
420 while (i < n && reachable_[i] == candidate) i++;
421
422 // insert candidate in its final place.
423 new_reachable_[j++] = candidate;
424 }
425 }
426 std::swap(reachable_, new_reachable_);
427 }
428
429 // Returns true iff sum might be expressible as a weighted sum of the added
430 // value. Any sum >= LastValue() is always considered potentially reachable.
431 bool MightBeReachable(int64_t sum) const {
432 if (sum >= reachable_.back()) return true;
433 return std::binary_search(reachable_.begin(), reachable_.end(), sum);
434 }
435
436 const std::array<int64_t, n>& reachable() const { return reachable_; }
437 int64_t LastValue() const { return reachable_.back(); }
438
439 private:
440 std::array<int64_t, n> reachable_;
441 std::array<int64_t, n> new_reachable_;
442};

589// Keep the top n elements from a stream of elements.
590//
591// TODO(user): We could use gtl::TopN when/if it gets open sourced. Note that
592// we might be slighlty faster here since we use an indirection and don't move
593// the Element class around as much.
594template <typename Element, typename Score>
595class TopN {
596 public:
597 explicit TopN(int n) : n_(n) {}
598
599 void Clear() {
600 heap_.clear();
601 elements_.clear();
602 }
603
604 void Add(Element e, Score score) {
605 if (heap_.size() < n_) {
606 const int index = elements_.size();
607 heap_.push_back({index, score});
608 elements_.push_back(std::move(e));
609 if (heap_.size() == n_) {
610 // TODO(user): We could delay that on the n + 1 push.
611 std::make_heap(heap_.begin(), heap_.end());
612 }
613 } else {
614 if (score <= heap_.front().score) return;
615 const int index_to_replace = heap_.front().index;
616 elements_[index_to_replace] = std::move(e);
617
618 // If needed, we could be faster here with an update operation.
619 std::pop_heap(heap_.begin(), heap_.end());

666// These functions are copied from MathUtils. However, the original ones are
667// incompatible with absl::int128 as MathLimits<absl::int128>::kIsInteger ==
668// false.
669template <typename IntType, bool ceil>
670IntType CeilOrFloorOfRatio(IntType numerator, IntType denominator) {
671 static_assert(std::numeric_limits<IntType>::is_integer,
672 "CeilOfRatio is only defined for integral types");
673 DCHECK_NE(0, denominator) << "Division by zero is not supported.";
674 DCHECK(numerator != std::numeric_limits<IntType>::min() || denominator != -1)
675 << "Dividing " << numerator << "by -1 is not supported: it would SIGFPE";
676
677 const IntType rounded_toward_zero = numerator / denominator;
678 const bool needs_round = (numerator % denominator) != 0;
679 const bool same_sign = (numerator >= 0) == (denominator >= 0);
680
681 if (ceil) { // Compile-time condition: not an actual branching
682 return rounded_toward_zero + static_cast<IntType>(same_sign && needs_round);
683 } else {
684 return rounded_toward_zero -
685 static_cast<IntType>(!same_sign && needs_round);
686 }
687}

690IntType CeilOfRatio(IntType numerator, IntType denominator) {
691 return CeilOrFloorOfRatio<IntType, true>(numerator, denominator);
692}

695IntType FloorOfRatio(IntType numerator, IntType denominator) {
696 return CeilOrFloorOfRatio<IntType, false>(numerator, denominator);
697}

789template <typename K, typename V>
790template <typename Keys, typename Values>
791inline void CompactVectorVector<K, V>::ResetFromFlatMapping(Keys keys,
792 Values values) {
793 if (keys.empty()) return clear();
794
795 // Compute maximum index.
796 int max_key = 0;
797 for (const K key : keys) {
798 max_key = std::max(max_key, InternalKey(key) + 1);
799 }
800
801 // Compute sizes_;
802 sizes_.assign(max_key, 0);
803 for (const K key : keys) {
804 sizes_[InternalKey(key)]++;
805 }
806
807 // Compute starts_;
808 starts_.assign(max_key, 0);
809 for (int k = 1; k < max_key; ++k) {
810 starts_[k] = starts_[k - 1] + sizes_[k - 1];
811 }
812
813 // Copy data and uses starts as temporary indices.
814 buffer_.resize(keys.size());
815 for (int i = 0; i < keys.size(); ++i) {
816 buffer_[starts_[InternalKey(keys[i])]++] = values[i];
817 }
818
819 // Restore starts_.
820 for (int k = max_key - 1; k > 0; --k) {
821 starts_[k] = starts_[k - 1];
822 }
823 starts_[0] = 0;
824}
