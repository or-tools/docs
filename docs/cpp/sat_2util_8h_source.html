56// Small utility class to store a vector<vector<>> where one can only append new

76 // Restore to empty vector<vector<>>.

145// Returns true if the equation a * X + b * Y = cte has some integer solutions.
146// For now, we check that a and b are different from 0 and from int64_t min.
147//
148// There is actually always a solution if cte % gcd(|a|, |b|) == 0. And because
149// a, b and cte fit on an int64_t, if there is a solution, there is one with X
150// and Y fitting on an int64_t.
151//
152// We will divide everything by gcd(a, b) first, so it is why we take reference
153// and the equation can change.
154//
155// If there are solutions, we return one of them (x0, y0).
156// From any such solution, the set of all solutions is given for Z integer by:
157// X = x0 + b * Z;
158// Y = y0 - a * Z;
159//
160// Given a domain for X and Y, it is possible to compute the "exact" domain of Z
161// with our Domain functions. Note however that this will only compute solution
162// where both x-x0 and y-y0 do fit on an int64_t:
163// DomainOf(x).SubtractionWith(x0).InverseMultiplicationBy(b).IntersectionWith(
164// DomainOf(y).SubtractionWith(y0).InverseMultiplicationBy(-a))
165bool SolveDiophantineEquationOfSizeTwo(int64_t& a, int64_t& b, int64_t& cte,
166 int64_t& x0, int64_t& y0);

195// Given a linear equation "sum coeff_i * X_i <= rhs. We can rewrite it using
196// ClosestMultiple() as "base * new_terms + error <= rhs" where error can be
197// bounded using the provided bounds on each variables. This will return true if
198// the error can be ignored and this equation is completely equivalent to
199// new_terms <= new_rhs.
200//
201// This is useful for cases like 9'999 X + 10'0001 Y <= 155'000 where we have

206// all the negative coeffs and corresponding bounds before calling this.
207bool LinearInequalityCanBeReducedWithClosestMultiple(
208 int64_t base, const std::vector<int64_t>& coeffs,
209 const std::vector<int64_t>& lbs, const std::vector<int64_t>& ubs,
210 int64_t rhs, int64_t* new_rhs);

258// Context: this function is not really generic, but required to be unit-tested.
259// It is used in a clause minimization algorithm when we try to detect if any of
260// the clause literals can be propagated by a subset of the other literal being
261// false. For that, we want to enqueue in the solver all the subset of size n-1.
262//
263// This moves one of the unprocessed literal from literals to the last position.
264// The function tries to do that while preserving the longest possible prefix of
265// literals "amortized" through the calls assuming that we want to move each
266// literal to the last position once.
267//
268// For a vector of size n, if we want to call this n times so that each literal
269// is last at least once, the sum of the size of the changed suffixes will be
270// O(n log n). If we were to use a simpler algorithm (like moving the last
271// unprocessed literal to the last position), this sum would be O(n^2).
272//
273// Returns the size of the common prefix of literals before and after the move,
274// or -1 if all the literals are already processed. The argument
275// relevant_prefix_size is used as a hint when keeping more that this prefix
276// size do not matter. The returned value will always be lower or equal to
277// relevant_prefix_size.
278int MoveOneUnprocessedLiteralLast(
279 const absl::btree_set<LiteralIndex>& processed, int relevant_prefix_size,
280 std::vector<Literal>* literals);

333// Simple DP to keep the set of the first n reachable value (n > 1).
334//
335// TODO(user): Maybe modulo some prime number we can keep more info.
336// TODO(user): Another common case is a bunch of really small values and larger
337// ones, so we could bound the sum of the small values and keep the first few
338// reachable by the big ones. This is similar to some presolve transformations.
339template <int n>
340class FirstFewValues {
341 public:
342 FirstFewValues() { Reset(); }
343
344 void Reset() {
345 reachable_.fill(std::numeric_limits<int64_t>::max());
346 reachable_[0] = 0;
347 new_reachable_[0] = 0;
348 }
349
350 // We assume the given positive value can be added as many time as wanted.
351 //
352 // TODO(user): Implement Add() with an upper bound on the multiplicity.
353 void Add(const int64_t positive_value) {
354 DCHECK_GT(positive_value, 0);
355 if (positive_value >= reachable_.back()) return;
356
357 // We copy from reachable_[i] to new_reachable_[j].
358 // The position zero is already copied.
359 int i = 1;
360 int j = 1;
361 for (int base = 0; j < n && base < n; ++base) {
362 const int64_t candidate = CapAdd(new_reachable_[base], positive_value);
363 while (j < n && i < n && reachable_[i] < candidate) {
364 new_reachable_[j++] = reachable_[i++];
365 }
366 if (j < n) {
367 // Eliminate duplicates.
368 while (i < n && reachable_[i] == candidate) i++;
369
370 // insert candidate in its final place.
371 new_reachable_[j++] = candidate;
372 }
373 }
374 std::swap(reachable_, new_reachable_);
375 }
376
377 // Returns true iff sum might be expressible as a weighted sum of the added
378 // value. Any sum >= LastValue() is always considered potentially reachable.
379 bool MightBeReachable(int64_t sum) const {
380 if (sum >= reachable_.back()) return true;
381 return std::binary_search(reachable_.begin(), reachable_.end(), sum);
382 }
383
384 const std::array<int64_t, n>& reachable() const { return reachable_; }
385 int64_t LastValue() const { return reachable_.back(); }
386
387 private:
388 std::array<int64_t, n> reachable_;
389 std::array<int64_t, n> new_reachable_;
390};

537// Keep the top n elements from a stream of elements.
538//
539// TODO(user): We could use gtl::TopN when/if it gets open sourced. Note that
540// we might be slighlty faster here since we use an indirection and don't move
541// the Element class around as much.
542template <typename Element, typename Score>
543class TopN {
544 public:
545 explicit TopN(int n) : n_(n) {}
546
547 void Clear() {
548 heap_.clear();
549 elements_.clear();
550 }
551
552 void Add(Element e, Score score) {
553 if (heap_.size() < n_) {
554 const int index = elements_.size();
555 heap_.push_back({index, score});
556 elements_.push_back(std::move(e));
557 if (heap_.size() == n_) {
558 // TODO(user): We could delay that on the n + 1 push.
559 std::make_heap(heap_.begin(), heap_.end());
560 }
561 } else {
562 if (score <= heap_.front().score) return;
563 const int index_to_replace = heap_.front().index;
564 elements_[index_to_replace] = std::move(e);
565
566 // If needed, we could be faster here with an update operation.
567 std::pop_heap(heap_.begin(), heap_.end());
568 heap_.back() = {index_to_replace, score};
569 std::push_heap(heap_.begin(), heap_.end());
570 }
571 }
572
573 bool empty() const { return elements_.empty(); }
574
575 const std::vector<Element>& UnorderedElements() const { return elements_; }
576
577 private:
578 const int n_;
579
580 // We keep a heap of the n highest score.
581 struct HeapElement {
582 int index; // in elements_;
583 Score score;
584 bool operator<(const HeapElement& other) const {
585 return score > other.score;
586 }
587 };
588 std::vector<HeapElement> heap_;
589 std::vector<Element> elements_;
590};

597 if (std::isnan(value)) return 0;
598 if (value >= static_cast<double>(std::numeric_limits<int64_t>::max())) {

603 }
604 return static_cast<int64_t>(value);
605}

620 DCHECK_NE(0, denominator) << "Division by zero is not supported.";
621 DCHECK(numerator != std::numeric_limits<IntType>::min() || denominator != -1)
622 << "Dividing " << numerator << "by -1 is not supported: it would SIGFPE";
623
624 const IntType rounded_toward_zero = numerator / denominator;
625 const bool needs_round = (numerator % denominator) != 0;
626 const bool same_sign = (numerator >= 0) == (denominator >= 0);
627
628 if (ceil) { // Compile-time condition: not an actual branching
629 return rounded_toward_zero + static_cast<IntType>(same_sign && needs_round);
630 } else {
631 return rounded_toward_zero -
632 static_cast<IntType>(!same_sign && needs_round);
633 }
634}

711template <typename K, typename V>
712template <typename Keys, typename Values>
713inline void CompactVectorVector<K, V>::ResetFromFlatMapping(Keys keys,
714 Values values) {
715 if (keys.empty()) return clear();
716
717 // Compute maximum index.
718 int max_key = 0;
719 for (const K key : keys) {
720 max_key = std::max(max_key, InternalKey(key) + 1);
721 }
722
723 // Compute sizes_;
724 sizes_.assign(max_key, 0);
725 for (const K key : keys) {
726 sizes_[InternalKey(key)]++;
727 }
728
729 // Compute starts_;
730 starts_.assign(max_key, 0);
731 for (int k = 1; k < max_key; ++k) {
732 starts_[k] = starts_[k - 1] + sizes_[k - 1];
733 }
734
735 // Copy data and uses starts as temporary indices.
736 buffer_.resize(keys.size());
737 for (int i = 0; i < keys.size(); ++i) {
738 buffer_[starts_[InternalKey(keys[i])]++] = values[i];
739 }
740
741 // Restore starts_.
742 for (int k = max_key - 1; k > 0; --k) {
743 starts_[k] = starts_[k - 1];
744 }
745 starts_[0] = 0;
746}
