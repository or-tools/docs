14// Runs multiple Dijkstra searches simultaneously (single-threaded, but growing
15// their search radii at the same time) on the same graph.
16// Supports custom arc length functors, and custom stopping criterion and
17// tracking.
18//
19// EXAMPLE:
20// With two sources and a custom stopping criterion that stops the first
21// Dijkstra when it has settled 1000 nodes and the second when it has reached
22// the search radius 123.45:
23//
24// ListGraph<> graph; // From ortools/graph/graph.h
25// ... build the graph ...
26// int source1 = ... , source2 = ...;
27// vector<double> arc_lengths(graph.num_arcs(), 0);
28// ... set the arc lengths ...
29// int num_nodes_to_settle_in_first_search = 1000;
30// std::vector<absl::flat_hash_map<int, DistanceAndParentArc<double>>>
31// reached_nodes; // One map per source: node -> DistanceAndParentArc.
32// reached_nodes = MultiDijkstra<double>(
33// graph,
34// [&arc_lengths](int a) { return arc_lengths[a]; },
35// {{source1}, {source2}},
36// [&num_nodes_to_settle_in_first_search](
37// int settled_node, int source_index, double settled_distance) {
38// if (source_index == 0) {
39// return --num_nodes_to_settle_in_first_search == 0;
40// } else {
41// return settled_distance >= 123.45;
42// }
43// });

73// Runs multiple Dijkstra searches simultaneously on the same graph, in a single
74// thread. All the Dijkstras share the same priority queue: their search radius
75// will grow "simulatenously".
76//
77// Moreover, each individual Dijkstra search can have several nodes as its
78// "source", and the stopping criterion for each Dijkstra search is highly
79// customizable: the user controls it via a "settled node callback",
80// called every time a node is settled. See the API below.
81//
82// The Dijkstra are sparse: the global space complexity will be linear in the
83// number of search states explored. Ditto for the time complexity, with an
84// additional logarithmic factor caused by the priority queue.
85//
86// This class has many similarities with BoundedDijkstraWrapper from
87// ./bounded_dijkstra.h but adds the overhead of tracking the source index for
88// every search state, and of the sparse (but slower) node hash maps.
89//
90// Arguments:
91// - "graph": the graph. The Graph class must support the interface of
92// ortools/graph/graph.h: OutgoingArcs(), Head(), Tail().
93// - "arc_length_functor": we will call arc_length_functor(a) on every
94// arc "a" explored, where "a" is cast to an int. It should return the arc's
95// length, which is statically cast to DistanceType.
96// - "source_sets" contains the sources. Each source is itself a set of nodes.
97// - "settled_node_callback" will be called every time we settled a node, with 3
98// arguments (node, source index, distance of node from that source).
99// If it returns true, the Dijkstra search from that source will stop.
100//
101// Returns the list of Dijkstra search: for each source #s, the element #s of
102// the returned vector will map every node explored in the Dijkstra from source
103// #s to its distance and parent arc.
104template <class DistanceType, class Graph, class ArcLengthFunctor,
105 class SettledNodeCallbackType>
106std::vector<absl::flat_hash_map<int, DistanceAndParentArc<DistanceType>>>
107MultiDijkstra(const Graph& graph, ArcLengthFunctor arc_length_functor,
108 const std::vector<std::vector<int>>& source_sets,
109 SettledNodeCallbackType settled_node_callback);

122template <class DistanceType, class Graph, class ArcLengthFunctor,
123 class SettledNodeCallbackType>
124std::vector<absl::flat_hash_map<int, DistanceAndParentArc<DistanceType>>>
125MultiDijkstra(const Graph& graph, ArcLengthFunctor arc_length_functor,
126 const std::vector<std::vector<int>>& source_sets,
127 SettledNodeCallbackType settled_node_callback) {
128 // Initialize the return data structure and the priority queue.
129 const int num_sources = source_sets.size();
130 std::vector<absl::flat_hash_map<int, DistanceAndParentArc<DistanceType>>>
131 reached_from(num_sources); // Also the returned output!
132 struct State {
133 DistanceType distance;
134 int node;
135 int source_index;
136 };
137 struct StateComparator {
138 bool operator()(const State& s0, const State& s1) const {
139 if (s0.distance != s1.distance) return s0.distance > s1.distance;
140 if (s0.node != s1.node) return s0.node > s1.node;
141 return s0.source_index > s1.source_index;
142 }
143 };
144 StateComparator pq_state_comparator;
145 std::priority_queue<State, std::vector<State>, StateComparator> pq(
146 pq_state_comparator);
147 std::vector<bool> dijkstra_is_active(num_sources, false);
148 int num_active_dijkstras = 0;
149 for (int source_index = 0; source_index < num_sources; ++source_index) {
150 if (!source_sets[source_index].empty()) {
151 dijkstra_is_active[source_index] = true;
152 ++num_active_dijkstras;
153 }
154 for (const int node : source_sets[source_index]) {
155 if (gtl::InsertIfNotPresent(
156 &reached_from[source_index],
157 {node, {/*distance=*/0, /*parent_arc=*/-1}})) {
158 pq.push({/*distance=*/0, /*node=*/node, /*source_index=*/source_index});
159 }
160 }
161 }
162
163 // Main Dijkstra loop.
164 while (!pq.empty() && num_active_dijkstras > 0) {
165 const State state = pq.top();
166 pq.pop();
167
168 if (!dijkstra_is_active[state.source_index]) continue;
169
170 // Dijkstra optimization: ignore states that don't correspond to the optimal
171 // (such states have been preceded by better states in the queue order,
172 // without being deleted since priority_queue doesn't have erase()).
173 if (gtl::FindOrDie(reached_from[state.source_index], state.node).distance <
174 state.distance) {
175 continue;
176 }
177
178 if (settled_node_callback(state.node, state.source_index, state.distance)) {
179 dijkstra_is_active[state.source_index] = false;
180 --num_active_dijkstras;
181 continue;
182 }
183
184 for (const int arc : graph.OutgoingArcs(state.node)) {
185 const DistanceType distance = arc_length_functor(arc) + state.distance;
186 const int head_node = graph.Head(arc);
187 auto insertion_result = reached_from[state.source_index].insert(
188 {head_node, {/*distance=*/distance, /*parent_arc=*/arc}});
189 if (!insertion_result.second) { // Already reached...
190 if (insertion_result.first->second.distance <= distance) continue;
191 insertion_result.first->second = {/*distance=*/distance,
192 /*parent_arc=*/arc};
193 }
194 pq.push(State{/*distance=*/distance, /*node=*/head_node,
195 /*source_index=*/state.source_index});
196 }
197 }
198 return reached_from;
199}
