31absl::Status ValidateSolutionHint(const SolutionHintProto& solution_hint,
32 const ModelSummary& model_summary) {
33 RETURN_IF_ERROR(CheckIdsAndValues(
34 MakeView(solution_hint.variable_values()),
35 {.allow_positive_infinity = false, .allow_negative_infinity = false}))
36 << "Invalid solution_hint.variable_values";
37 RETURN_IF_ERROR(CheckIdsSubset(
38 solution_hint.variable_values().ids(), model_summary.variables,
39 "solution_hint.variable_values ids", "model variable ids"));
40
41 RETURN_IF_ERROR(CheckIdsAndValues(
42 MakeView(solution_hint.dual_values()),
43 {.allow_positive_infinity = false, .allow_negative_infinity = false}))
44 << "Invalid solution_hint.dual_values";
45 RETURN_IF_ERROR(CheckIdsSubset(
46 solution_hint.dual_values().ids(), model_summary.linear_constraints,
47 "solution_hint.dual_values ids", "model linear constraint ids"));
48
49 return absl::OkStatus();
50}

67absl::Status ValidateSparseVectorFilter(const SparseVectorFilterProto& v,
68 const IdNameBiMap& valid_ids) {
69 RETURN_IF_ERROR(CheckIdsRangeAndStrictlyIncreasing(v.filtered_ids()));
70 RETURN_IF_ERROR(
71 CheckIdsSubset(v.filtered_ids(), valid_ids, "filtered_ids", "model IDs"));
72 if (!v.filter_by_ids() && !v.filtered_ids().empty()) {
73 return absl::InvalidArgumentError(
74 "Invalid SparseVectorFilterProto.filter_by_id* specification. To "
75 "filter by "
76 "IDs you must set SparseVectorFilterProto.filter_by_ids to 'true'.");
77 }
78
79 return absl::OkStatus();
80}

82absl::Status ValidateModelSolveParameters(
83 const ModelSolveParametersProto& parameters,
84 const ModelSummary& model_summary) {
85 RETURN_IF_ERROR(ValidateSparseVectorFilter(
86 parameters.variable_values_filter(), model_summary.variables))
87 << "invalid variable_values_filter";
88 RETURN_IF_ERROR(ValidateSparseVectorFilter(parameters.reduced_costs_filter(),
89 model_summary.variables))
90 << "invalid reduced_costs_filter";
91 RETURN_IF_ERROR(ValidateSparseVectorFilter(parameters.dual_values_filter(),
92 model_summary.linear_constraints))
93 << "invalid dual_values_filter";
94 if (parameters.has_initial_basis()) {
95 RETURN_IF_ERROR(ValidateBasis(parameters.initial_basis(), model_summary,
96 /*check_dual_feasibility=*/false));
97 }
98 for (const SolutionHintProto& solution_hint : parameters.solution_hints()) {
99 RETURN_IF_ERROR(ValidateSolutionHint(solution_hint, model_summary));
100 }
101 RETURN_IF_ERROR(ValidateBranchingPriorities(parameters.branching_priorities(),
102 model_summary));
103 return absl::OkStatus();
104}
