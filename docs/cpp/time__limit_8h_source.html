310// Wrapper around TimeLimit to make it thread safe and add Stop() support.
311class SharedTimeLimit {
312 public:
313 explicit SharedTimeLimit(TimeLimit* time_limit)
314 : time_limit_(time_limit), stopped_boolean_(false) {
315 // We use the one already registered if present or ours otherwise.
316 stopped_ = time_limit->ExternalBooleanAsLimit();
317 if (stopped_ == nullptr) {
318 stopped_ = &stopped_boolean_;
319 time_limit->RegisterExternalBooleanAsLimit(stopped_);
320 }
321 }
322
323 ~SharedTimeLimit() {
324 if (stopped_ == &stopped_boolean_) {
325 time_limit_->RegisterExternalBooleanAsLimit(nullptr);
326 }
327 }
328
329 bool LimitReached() const {
330 // Note, time_limit_->LimitReached() is not const, and changes internal
331 // state of time_limit_, hence we need a writer's lock.
332 absl::MutexLock lock(&mutex_);
333 return time_limit_->LimitReached();
334 }
335
336 void Stop() {
337 absl::MutexLock lock(&mutex_);
338 *stopped_ = true;
339 }
340
341 void UpdateLocalLimit(TimeLimit* local_limit) {
342 absl::MutexLock lock(&mutex_);
343 local_limit->MergeWithGlobalTimeLimit(time_limit_);
344 }
345
346 void AdvanceDeterministicTime(double deterministic_duration) {
347 absl::MutexLock lock(&mutex_);
348 time_limit_->AdvanceDeterministicTime(deterministic_duration);
349 }
350
351 double GetTimeLeft() const {
352 absl::ReaderMutexLock lock(&mutex_);
353 return time_limit_->GetTimeLeft();
354 }
355
356 double GetElapsedDeterministicTime() const {
357 absl::ReaderMutexLock lock(&mutex_);
358 return time_limit_->GetElapsedDeterministicTime();
359 }
360
361 std::atomic<bool>* ExternalBooleanAsLimit() const {
362 absl::ReaderMutexLock lock(&mutex_);
363 // We can simply return the "external bool" and remain thread-safe because
364 // it's wrapped in std::atomic.
365 return time_limit_->ExternalBooleanAsLimit();
366 }
367
368 private:
369 mutable absl::Mutex mutex_;
370 TimeLimit* time_limit_ ABSL_GUARDED_BY(mutex_);
371 std::atomic<bool> stopped_boolean_ ABSL_GUARDED_BY(mutex_);
372 std::atomic<bool>* stopped_ ABSL_GUARDED_BY(mutex_);
373};

453inline TimeLimit::TimeLimit(double limit_in_seconds, double deterministic_limit)
454 : safety_buffer_ns_(static_cast<int64_t>(kSafetyBufferSeconds * 1e9)),
455 running_max_(kHistorySize),
456 external_boolean_as_limit_(nullptr) {
457 ResetTimers(limit_in_seconds, deterministic_limit);
458}

460inline void TimeLimit::ResetTimers(double limit_in_seconds,
461 double deterministic_limit) {
462 elapsed_deterministic_time_ = 0.0;
463 deterministic_limit_ = deterministic_limit;
464
465 if (absl::GetFlag(FLAGS_time_limit_use_usertime)) {
466 user_timer_.Start();
467 limit_in_seconds_ = limit_in_seconds;
468 }
469 start_ns_ = absl::GetCurrentTimeNanos();
470 last_ns_ = start_ns_;
471 // Note that duration arithmetic is properly saturated.
472 limit_ns_ = (absl::Seconds(limit_in_seconds) + absl::Nanoseconds(start_ns_)) /
473 absl::Nanoseconds(1);
474}
475
476template <typename Parameters>
477inline void TimeLimit::ResetLimitFromParameters(const Parameters& parameters) {
478 ResetTimers(parameters.max_time_in_seconds(),
479 parameters.max_deterministic_time());
480}

482inline void TimeLimit::MergeWithGlobalTimeLimit(TimeLimit* other) {
483 if (other == nullptr) return;
484 ResetTimers(
485 std::min(GetTimeLeft(), other->GetTimeLeft()),
486 std::min(GetDeterministicTimeLeft(), other->GetDeterministicTimeLeft()));
487 if (other->ExternalBooleanAsLimit() != nullptr) {
488 RegisterExternalBooleanAsLimit(other->ExternalBooleanAsLimit());
489 }
490}

492inline bool TimeLimit::LimitReached() {
493 if (external_boolean_as_limit_ != nullptr &&
494 external_boolean_as_limit_->load()) {
495 return true;
496 }
497 if (secondary_external_boolean_as_limit_ != nullptr &&
498 secondary_external_boolean_as_limit_->load()) {
499 return true;
500 }
501
502 if (GetDeterministicTimeLeft() <= 0.0) {
503 return true;
504 }
505
506 const int64_t current_ns = absl::GetCurrentTimeNanos();
507 running_max_.Add(std::max(safety_buffer_ns_, current_ns - last_ns_));
508 last_ns_ = current_ns;
509 if (current_ns + running_max_.GetCurrentMax() >= limit_ns_) {
510 if (absl::GetFlag(FLAGS_time_limit_use_usertime)) {
511 // To avoid making many system calls, we only check the user time when
512 // the "absolute" time limit has been reached. Note that the user time
513 // should advance more slowly, so this is correct.
514 const double time_left_s = limit_in_seconds_ - user_timer_.Get();
515 if (time_left_s > kSafetyBufferSeconds) {
516 limit_ns_ = static_cast<int64_t>(time_left_s * 1e9) + last_ns_;
517 return false;
518 }
519 }
520
521 // To ensure that future calls to LimitReached() will return true.
522 limit_ns_ = 0;
523 return true;
524 }
525 return false;
526}
527
528inline double TimeLimit::GetTimeLeft() const {
529 if (limit_ns_ == kint64max) return std::numeric_limits<double>::infinity();
530 const int64_t delta_ns = limit_ns_ - absl::GetCurrentTimeNanos();
531 if (delta_ns < 0) return 0.0;
532 if (absl::GetFlag(FLAGS_time_limit_use_usertime)) {
533 return std::max(limit_in_seconds_ - user_timer_.Get(), 0.0);
534 } else {
535 return delta_ns * 1e-9;
536 }
537}
