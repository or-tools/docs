59// To simplify cut generation code, we use a more complex data structure than
60// just a LinearConstraint to represent a cut with shifted/complemented variable
61// and implied bound substitution.
62struct CutTerm {
63 bool IsBoolean() const { return bound_diff == 1; }
64 bool IsSimple() const { return expr_coeffs[1] == 0; }
65 bool HasRelevantLpValue() const { return lp_value > 1e-2; }
66 double LpDistToMaxValue() const {
67 return static_cast<double>(bound_diff.value()) - lp_value;
68 }
69
70 std::string DebugString() const;
71
72 // Do the subtitution X -> (1 - X') and update the rhs.
73 //
74 // Our precondition on the sum of variable domains fitting an int64_t should
75 // ensure that this can never overflow.
76 void Complement(absl::int128* rhs);
77
78 // This assumes bound_diff == 1. It replaces the inner expression by either
79 // var or 1 - var depending on the positiveness of var.
80 void ReplaceExpressionByLiteral(IntegerVariable var);
81
82 // If the term correspond to literal_view or (1 - literal_view) return the
83 // integer variable representation of that literal. Returns kNoIntegerVariable
84 // if this is not the case.
85 IntegerVariable GetUnderlyingLiteralOrNone() const;
86
87 // Each term is of the form coeff * X where X is a variable with given
88 // lp_value and with a domain in [0, bound_diff]. Note X is always >= 0.
89 double lp_value = 0.0;
90 IntegerValue coeff = IntegerValue(0);
91 IntegerValue bound_diff = IntegerValue(0);
92
93 // X = the given LinearExpression.
94 // We only support size 1 or 2 here which allow to inline the memory.
95 // When a coefficient is zero, we don't care about the variable.
96 IntegerValue expr_offset = IntegerValue(0);
97 std::array<IntegerVariable, 2> expr_vars;
98 std::array<IntegerValue, 2> expr_coeffs;
99};

166// Given an upper-bounded linear relation (sum terms <= ub), this algorithm
167// inspects the integer variable appearing in the sum and try to replace each of
168// them by a tight lower bound (>= coeff * binary + lb) using the implied bound
169// repository. By tight, we mean that it will take the same value under the
170// current LP solution.
171//
172// We use a class to reuse memory of the tmp terms.
173class ImpliedBoundsProcessor {
174 public:
175 // We will only replace IntegerVariable appearing in lp_vars_.
176 ImpliedBoundsProcessor(absl::Span<const IntegerVariable> lp_vars_,
177 IntegerTrail* integer_trail,
178 ImpliedBounds* implied_bounds)
179 : lp_vars_(lp_vars_.begin(), lp_vars_.end()),
180 integer_trail_(integer_trail),
181 implied_bounds_(implied_bounds) {}
182
183 // See if some of the implied bounds equation are violated and add them to
184 // the IB cut pool if it is the case.
185 //
186 // Important: This must be called before we process any constraints with a
187 // different lp_values or level zero bounds.
188 void RecomputeCacheAndSeparateSomeImpliedBoundCuts(
189 const absl::StrongVector<IntegerVariable, double>& lp_values);
190
191 // This assumes the term is simple: expr[0] = var - LB / UB - var. We use an
192 // implied lower bound on this expr, independently of the term.coeff sign.
193 //
194 // If possible, returns true and express X = bool_term + slack_term.
195 // If coeff of X is positive, then all coeff will be positive here.
196 bool DecomposeWithImpliedLowerBound(const CutTerm& term,
197 IntegerValue factor_t, CutTerm& bool_term,
198 CutTerm& slack_term);
199
200 // This assumes the term is simple: expr[0] = var - LB / UB - var. We use
201 // an implied upper bound on this expr, independently of term.coeff sign.
202 //
203 // If possible, returns true and express X = bool_term + slack_term.
204 // If coeff of X is positive, then bool_term will have a positive coeff but
205 // slack_term will have a negative one.
206 bool DecomposeWithImpliedUpperBound(const CutTerm& term,
207 IntegerValue factor_t, CutTerm& bool_term,
208 CutTerm& slack_term);
209
210 // We are about to apply the super-additive function f() to the CutData. Use
211 // implied bound information to eventually substitute and make the cut
212 // stronger. Returns the number of {lb_ib, ub_ib} applied.
213 //
214 // This should lead to stronger cuts even if the norms migth be worse.
215 std::pair<int, int> PostprocessWithImpliedBound(
216 const std::function<IntegerValue(IntegerValue)>& f, IntegerValue factor_t,
217 CutData* cut, CutDataBuilder* builder);
218
219 bool TryToExpandWithLowerImpliedbound(IntegerValue factor_t, int i,
220 bool complement, CutData* cut,
221 CutDataBuilder* builder);
222
223 // Add a new variable that could be used in the new cuts.
224 // Note that the cache must be computed to take this into account.
225 void AddLpVariable(IntegerVariable var) { lp_vars_.insert(var); }
226
227 // Once RecomputeCacheAndSeparateSomeImpliedBoundCuts() has been called,
228 // we can get the best implied bound for each variables.
229 //
230 // Note that because the variable level zero lower bound might change since
231 // the time this was cached, we just store the implied bound here.
232 struct BestImpliedBoundInfo {
233 double var_lp_value = 0.0;
234 double bool_lp_value = 0.0;
235 bool is_positive;
236 IntegerValue implied_bound;
237 IntegerVariable bool_var = kNoIntegerVariable;
238
239 double SlackLpValue(IntegerValue lb) const {
240 const double bool_term = ToDouble(implied_bound - lb) * bool_lp_value;
241 return var_lp_value - ToDouble(lb) - bool_term;
242 }
243
244 std::string DebugString() const {
245 return absl::StrCat("var - lb == (", implied_bound.value(),
246 " - lb) * bool(", bool_lp_value, ") + slack.");
247 }
248 };
249 BestImpliedBoundInfo GetCachedImpliedBoundInfo(IntegerVariable var) const;
250
251 // As we compute the best implied bounds for each variable, we add violated
252 // cuts here.
253 TopNCuts& IbCutPool() { return ib_cut_pool_; }
254
255 private:
256 BestImpliedBoundInfo ComputeBestImpliedBound(
257 IntegerVariable var,
258 const absl::StrongVector<IntegerVariable, double>& lp_values);
259
260 absl::flat_hash_set<IntegerVariable> lp_vars_;
261 mutable absl::flat_hash_map<IntegerVariable, BestImpliedBoundInfo> cache_;
262
263 TopNCuts ib_cut_pool_ = TopNCuts(50);
264
265 // Data from the constructor.
266 IntegerTrail* integer_trail_;
267 ImpliedBounds* implied_bounds_;
268};

270// A single node flow relaxation is a constraint of the form
271// Sum in_flow - Sum out_flow <= demand
272// where each flow variable F_i is in [0, capacity_i] and satisfy
273// F_i <= capacity_i B_i
274// with B_i a Boolean representing the arc usage.
275//
276// From a generic constraint sum coeff_i X_i <= b, we try to put it in this
277// format. We can first transform all variables to be in [0, max_value].
278//
279// Then we cover different cases:
280// 1/ A coeff * Boolean, can be easily transformed.
281// 2/ A coeff * Integer in [0, capacity] with Bool => integer == 0 too.
282// 3/ For a general integer, we can always use a Bool == 1 for the arc usage.
283//
284// TODO(user): cover case 3/. We loose a lot of relaxation here, except if
285// the variable is at is upper/lower bound.
286//
287// TODO(user): Altough the cut should still be correct, we might use the same
288// Boolean more than once in the implied bound. Or this Boolean might already
289// appear in the constraint. Not sure if we can do something smarter here.
290struct FlowInfo {
291 // Flow is always in [0, capacity] with the given current value in the
292 // lp relaxation. Now that we usually only consider tight constraint were
293 // flow_lp_value = capacity * bool_lp_value.
294 IntegerValue capacity;
295 double bool_lp_value;
296
297 // TODO(user): We don't use this in the heuristic currently.
298 double flow_lp_value;
299
300 // The definition of the flow variable and the arc usage variable in term
301 // of original problem variables. After we compute a cut on the flow and
302 // usage variable, we can just directly substitute these variable by the
303 // expression here to have a cut in term of the original problem variables.
304 AffineExpression flow_expr;
305 AffineExpression bool_expr;
306};

308struct SingleNodeFlow {
309 bool empty() const { return in_flow.empty() && out_flow.empty(); }
310 void clear() {
311 demand = 0;
312 in_flow.clear();
313 out_flow.clear();
314 num_bool = 0;
315 num_to_lb = 0;
316 num_to_ub = 0;
317 }
318 std::string DebugString() const;
319
320 absl::int128 demand;
321 std::vector<FlowInfo> in_flow;
322 std::vector<FlowInfo> out_flow;
323
324 // Stats filled during extraction.
325 int num_bool = 0;
326 int num_to_lb = 0;
327 int num_to_ub = 0;
328};

390// Visible for testing. Returns a function f on integers such that:
391// - f is non-decreasing.
392// - f is super-additive: f(a) + f(b) <= f(a + b)
393// - 1 <= f(divisor) <= max_scaling
394// - For all x, f(x * divisor) = x * f(divisor)
395// - For all x, f(x * divisor + remainder) = x * f(divisor)
396//
397// Preconditions:
398// - 0 <= remainder < divisor.
399// - 1 <= max_scaling.
400//
401// This is used in IntegerRoundingCut() and is responsible for "strengthening"
402// the cut. Just taking f(x) = x / divisor result in the non-strengthened cut
403// and using any function that stricly dominate this one is better.
404//
405// Algorithm:
406// - We first scale by a factor t so that rhs_remainder >= divisor / 2.
407// - Then, if max_scaling == 2, we use the function described
408// in "Strenghtening Chvatal-Gomory cuts and Gomory fractional cuts", Adam N.
409// Letchfrod, Andrea Lodi.
410// - Otherwise, we use a generalization of this which is a discretized version
411// of the classical MIR rounding function that only take the value of the
412// form "an_integer / max_scaling". As max_scaling goes to infinity, this
413// converge to the real-valued MIR function.
414//
415// Note that for each value of max_scaling we will get a different function.
416// And that there is no dominance relation between any of these functions. So
417// it could be nice to try to generate a cut using different values of
418// max_scaling.
419IntegerValue GetFactorT(IntegerValue rhs_remainder, IntegerValue divisor,
420 IntegerValue max_magnitude);
421std::function<IntegerValue(IntegerValue)> GetSuperAdditiveRoundingFunction(
422 IntegerValue rhs_remainder, IntegerValue divisor, IntegerValue t,
423 IntegerValue max_scaling);
424
425// If we have an equation sum ci.Xi >= rhs with everything positive, and all
426// ci are >= min_magnitude then any ci >= rhs can be set to rhs. Also if
427// some ci are in [rhs - min, rhs) then they can be strenghtened to rhs - min.
428//
429// If we apply this to the negated equation (sum -ci.Xi + sum cj.Xj <= -rhs)
430// with potentially positive terms, this reduce to apply a super-additive
431// function:
432//
433// Plot look like:
434// x=-rhs x=0
435// | |
436// y=0 : | ---------------------------------
437// | ---
438// | /
439// |---
440// y=-rhs -------
441//
442// TODO(user): Extend it for ci >= max_magnitude, we can probaly "lift" such
443// coefficient.
444std::function<IntegerValue(IntegerValue)> GetSuperAdditiveStrengtheningFunction(
445 IntegerValue positive_rhs, IntegerValue min_magnitude);
446
447// Similar to above but with scaling of the linear part to just have at most
448// scaling values.
449std::function<IntegerValue(IntegerValue)>
450GetSuperAdditiveStrengtheningMirFunction(IntegerValue positive_rhs,
451 IntegerValue scaling);
452
453// Given a super-additive non-decreasing function f(), we periodically extend
454// its restriction from [-period, 0] to Z. Such extension is not always
455// super-additive and it is up to the caller to know when this is true or not.
456inline std::function<IntegerValue(IntegerValue)> ExtendNegativeFunction(
457 std::function<IntegerValue(IntegerValue)> base_f, IntegerValue period) {
458 const IntegerValue m = -base_f(-period);
459 return [m, period, base_f](IntegerValue v) {
460 const IntegerValue r = PositiveRemainder(v, period);
461 const IntegerValue output_r = m + base_f(r - period);
462 return FloorRatio(v, period) * m + output_r;
463 };
464}

466// Given an upper bounded linear constraint, this function tries to transform it
467// to a valid cut that violate the given LP solution using integer rounding.
468// Note that the returned cut might not always violate the LP solution, in which
469// case it can be discarded.
470//
471// What this does is basically take the integer division of the constraint by an
472// integer. If the coefficients where doubles, this would be the same as scaling
473// the constraint and then rounding. We choose the coefficient of the most
474// fractional variable (rescaled by its coefficient) as the divisor, but there
475// are other possible alternatives.
476//
477// Note that if the constraint is tight under the given lp solution, and if
478// there is a unique variable not at one of its bounds and fractional, then we
479// are guaranteed to generate a cut that violate the current LP solution. This
480// should be the case for Chvatal-Gomory base constraints modulo our loss of
481// precision while doing exact integer computations.
482//
483// Precondition:
484// - We assumes that the given initial constraint is tight using the given lp
485// values. This could be relaxed, but for now it should always be the case, so
486// we log a message and abort if not, to ease debugging.
487// - The IntegerVariable of the cuts are not used here. We assumes that the
488// first three vectors are in one to one correspondence with the initial order
489// of the variable in the cut.
490//
491// TODO(user): There is a bunch of heuristic involved here, and we could spend
492// more effort tuning them. In particular, one can try many heuristics and keep
493// the best looking cut (or more than one). This is not on the critical code
494// path, so we can spend more effort in finding good cuts.
495struct RoundingOptions {
496 IntegerValue max_scaling = IntegerValue(60);
497 bool use_ib_before_heuristic = true;
498 bool prefer_positive_ib = true;
499};

555 // Try to find a cut with a knapsack heuristic. This assumes an input with all
556 // coefficients positive. If this returns true, you can get the cut via cut().
557 //
558 // This uses a lifting procedure similar to what is described in "Lifting the
559 // Knapsack Cover Inequalities for the Knapsack Polytope", Adam N. Letchfod,
560 // Georgia Souli. In particular the section "Lifting via mixed-integer
561 // rounding".
562 bool TrySimpleKnapsack(const CutData& input_ct,

565 // Applies the lifting procedure described in "On Lifted Cover Inequalities: A
566 // New Lifting Procedure with Unusual Properties", Adam N. Letchford, Georgia
567 // Souli. This assumes an input with all coefficients positive.
568 //
569 // The algo is pretty simple, given a cover C for a given rhs. We compute
570 // a rational weight p/q so that sum_C min(w_i, p/q) = rhs. Note that q is
571 // pretty small (lower or equal to the size of C). The generated cut is then
572 // of the form
573 // sum X_i in C for which w_i <= p / q
574 // + sum gamma_i X_i for the other variable <= |C| - 1.
575 //
576 // gamma_i being the smallest k such that w_i <= sum of the k + 1 largest
577 // min(w_i, p/q) for i in C. In particular, it is zero if w_i <= p/q.
578 //
579 // Note that this accept a general constraint that has been canonicalized to
580 // sum coeff_i * X_i <= base_rhs. Each coeff_i >= 0 and each X_i >= 0.
581 //
582 // TODO(user): Generalize to non-Boolean, or use a different cover heuristic
583 // for this:
584 // - We want a Boolean only cover currently.
585 // - We can always use implied bound for this, since there is more chance
586 // for a Bool only cover.
587 // - Also, f() should be super additive on the value <= rhs, i.e. f(a + b) >=
588 // f(a) + f(b), so it is always good to use implied bounds of the form X =
589 // bound * B + Slack.
590 bool TryWithLetchfordSouliLifting(
591 const CutData& input_ct, ImpliedBoundsProcessor* ib_processor = nullptr);
592

726// A cut generator for all_diff(xi). Let the united domain of all xi be D. Sum
727// of any k-sized subset of xi need to be greater or equal to the sum of
728// smallest k values in D and lesser or equal to the sum of largest k values in
729// D. The cut generator first sorts the variables based on LP values and adds
730// cuts of the form described above if they are violated by lp solution. Note
731// that all the fixed variables are ignored while generating cuts.
732CutGenerator CreateAllDifferentCutGenerator(
733 const std::vector<AffineExpression>& exprs, Model* model);
734
735// Consider the Lin Max constraint with d expressions and n variables in the
736// form: target = max {exprs[k] = Sum (wki * xi + bk)}. k in {1,..,d}.
737// Li = lower bound of xi
738// Ui = upper bound of xi.
739// Let zk be in {0,1} for all k in {1,..,d}.
740// The target = exprs[k] when zk = 1.
741//
742// The following is a valid linearization for Lin Max.
743// target >= exprs[k], for all k in {1,..,d}
744// target <= Sum (wli * xi) + Sum((Nlk + bk) * zk), for all l in {1,..,d}
745// Where Nlk is a large number defined as:
746// Nlk = Sum (max((wki - wli)*Li, (wki - wli)*Ui))
747// = Sum (max corner difference for variable i, target expr l, max expr k)
748//
749// Consider a partition of variables xi into set {1,..,d} as I.
750// i.e. I(i) = j means xi is mapped to jth index.
751// The following inequality is valid and sharp cut for the lin max constraint
752// described above.
753//
754// target <= Sum(i=1..n)(wI(i)i * xi + Sum(k=1..d)(MPlusCoefficient_ki * zk))
755// + Sum(k=1..d)(bk * zk) ,
756// Where MPlusCoefficient_ki = max((wki - wI(i)i) * Li,
757// (wki - wI(i)i) * Ui)
758// = max corner difference for variable i,
759// target expr I(i), max expr k.
760//
761// For detailed proof of validity, refer
762// Reference: "Strong mixed-integer programming formulations for trained neural
763// networks" by Ross Anderson et. (https://arxiv.org/pdf/1811.01988.pdf).
764//
765// In the cut generator, we compute the most violated partition I by computing
766// the rhs value (wI(i)i * lp_value(xi) + Sum(k=1..d)(MPlusCoefficient_ki * zk))
767// for each variable for each partition index. We choose the partition index
768// that gives lowest rhs value for a given variable.
769//
770// Note: This cut generator requires all expressions to contain only positive
771// vars.
772CutGenerator CreateLinMaxCutGenerator(
773 IntegerVariable target, const std::vector<LinearExpression>& exprs,
774 const std::vector<IntegerVariable>& z_vars, Model* model);
