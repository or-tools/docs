39void LinearConstraintBuilder::AddTerm(IntegerVariable var, IntegerValue coeff) {
40 if (coeff == 0) return;
41 // We can either add var or NegationOf(var), and we always choose the
42 // positive one.
43 if (VariableIsPositive(var)) {
44 terms_.push_back({var, coeff});
45 } else {
46 terms_.push_back({NegationOf(var), -coeff});
47 }
48}

50void LinearConstraintBuilder::AddTerm(AffineExpression expr,
51 IntegerValue coeff) {
52 if (coeff == 0) return;
53 // We can either add var or NegationOf(var), and we always choose the
54 // positive one.
55 if (expr.var != kNoIntegerVariable) {
56 if (VariableIsPositive(expr.var)) {
57 terms_.push_back({expr.var, coeff * expr.coeff});
58 } else {
59 terms_.push_back({NegationOf(expr.var), -coeff * expr.coeff});
60 }
61 }
62 offset_ += coeff * expr.constant;
63}

70void LinearConstraintBuilder::AddLinearExpression(const LinearExpression& expr,
71 IntegerValue coeff) {
72 for (int i = 0; i < expr.vars.size(); ++i) {
73 // We must use positive variables.
74 if (VariableIsPositive(expr.vars[i])) {
75 terms_.push_back({expr.vars[i], expr.coeffs[i] * coeff});
76 } else {
77 terms_.push_back({NegationOf(expr.vars[i]), -expr.coeffs[i] * coeff});
78 }
79 }
80 offset_ += expr.offset * coeff;
81}

83ABSL_MUST_USE_RESULT bool LinearConstraintBuilder::AddDecomposedProduct(
84 const std::vector<LiteralValueValue>& product) {
85 if (product.empty()) return true;
86
87 IntegerValue product_min = kMaxIntegerValue;
88 // TODO(user): Checks the value of literals.
89 for (const LiteralValueValue& term : product) {
90 product_min = std::min(product_min, term.left_value * term.right_value);
91 }
92
93 for (const LiteralValueValue& term : product) {
94 IntegerValue coeff = term.left_value * term.right_value - product_min;
95 if (coeff == 0) continue;
96 if (!AddLiteralTerm(term.literal, coeff)) {
97 return false;
98 }
99 }
100 AddConstant(product_min);
101 return true;
102}

104void LinearConstraintBuilder::AddQuadraticLowerBound(
105 AffineExpression left, AffineExpression right, IntegerTrail* integer_trail,
106 bool* is_quadratic) {
107 if (integer_trail->IsFixed(left)) {
108 AddTerm(right, integer_trail->FixedValue(left));
109 } else if (integer_trail->IsFixed(right)) {
110 AddTerm(left, integer_trail->FixedValue(right));
111 } else {
112 const IntegerValue left_min = integer_trail->LowerBound(left);
113 const IntegerValue right_min = integer_trail->LowerBound(right);
114 AddTerm(left, right_min);
115 AddTerm(right, left_min);
116 // Substract the energy counted twice.
117 AddConstant(-left_min * right_min);
118 if (is_quadratic != nullptr) *is_quadratic = true;
119 }
120}

126ABSL_MUST_USE_RESULT bool LinearConstraintBuilder::AddLiteralTerm(
127 Literal lit, IntegerValue coeff) {
128 DCHECK(encoder_ != nullptr);
129 IntegerVariable var = kNoIntegerVariable;
130 bool view_is_direct = true;
131 if (!encoder_->LiteralOrNegationHasView(lit, &var, &view_is_direct)) {
132 return false;
133 }
134
135 if (view_is_direct) {
136 AddTerm(var, coeff);
137 } else {
138 AddTerm(var, -coeff);
139 offset_ += coeff;
140 }
141 return true;
142}

164double ComputeActivity(
165 const LinearConstraint& constraint,
166 const absl::StrongVector<IntegerVariable, double>& values) {
167 double activity = 0;
168 for (int i = 0; i < constraint.vars.size(); ++i) {
169 const IntegerVariable var = constraint.vars[i];
170 const IntegerValue coeff = constraint.coeffs[i];
171 activity += coeff.value() * values[var];
172 }
173 return activity;
174}

176double ComputeL2Norm(const LinearConstraint& constraint) {
177 double sum = 0.0;
178 for (const IntegerValue coeff : constraint.coeffs) {
179 sum += ToDouble(coeff) * ToDouble(coeff);
180 }
181 return std::sqrt(sum);
182}

186 for (const IntegerValue coeff : constraint.coeffs) {
187 result = std::max(result, IntTypeAbs(coeff));

192double ScalarProduct(const LinearConstraint& constraint1,
193 const LinearConstraint& constraint2) {
194 DCHECK(std::is_sorted(constraint1.vars.begin(), constraint1.vars.end()));
195 DCHECK(std::is_sorted(constraint2.vars.begin(), constraint2.vars.end()));
196 double scalar_product = 0.0;
197 int index_1 = 0;
198 int index_2 = 0;
199 while (index_1 < constraint1.vars.size() &&
200 index_2 < constraint2.vars.size()) {
201 if (constraint1.vars[index_1] == constraint2.vars[index_2]) {
202 scalar_product += ToDouble(constraint1.coeffs[index_1]) *
203 ToDouble(constraint2.coeffs[index_2]);
204 index_1++;
205 index_2++;
206 } else if (constraint1.vars[index_1] > constraint2.vars[index_2]) {
207 index_2++;
208 } else {
209 index_1++;
210 }
211 }
212 return scalar_product;
213}

218IntegerValue ComputeGcd(const std::vector<IntegerValue>& values) {
219 if (values.empty()) return IntegerValue(1);
220 int64_t gcd = 0;
221 for (const IntegerValue value : values) {
222 gcd = MathUtil::GCD64(gcd, std::abs(value.value()));
223 if (gcd == 1) break;
224 }
225 if (gcd < 0) return IntegerValue(1); // Can happen with kint64min.
226 return IntegerValue(gcd);
227}

231void DivideByGCD(LinearConstraint* constraint) {
232 if (constraint->coeffs.empty()) return;
233 const IntegerValue gcd = ComputeGcd(constraint->coeffs);
234 if (gcd == 1) return;
235
236 if (constraint->lb > kMinIntegerValue) {
237 constraint->lb = CeilRatio(constraint->lb, gcd);
238 }
239 if (constraint->ub < kMaxIntegerValue) {
240 constraint->ub = FloorRatio(constraint->ub, gcd);
241 }
242 for (IntegerValue& coeff : constraint->coeffs) coeff /= gcd;
243}

245void RemoveZeroTerms(LinearConstraint* constraint) {
246 int new_size = 0;
247 const int size = constraint->vars.size();
248 for (int i = 0; i < size; ++i) {
249 if (constraint->coeffs[i] == 0) continue;
250 constraint->vars[new_size] = constraint->vars[i];
251 constraint->coeffs[new_size] = constraint->coeffs[i];
252 ++new_size;
253 }
254 constraint->vars.resize(new_size);
255 constraint->coeffs.resize(new_size);
256}

259 const int size = constraint->vars.size();
260 for (int i = 0; i < size; ++i) {
261 const IntegerValue coeff = constraint->coeffs[i];
262 if (coeff < 0) {
263 constraint->coeffs[i] = -coeff;

298IntegerValue LinearExpression::Min(const IntegerTrail& integer_trail) const {
299 IntegerValue result = offset;
300 for (int i = 0; i < vars.size(); ++i) {
301 if (coeffs[i] > 0) {
302 result += coeffs[i] * integer_trail.LowerBound(vars[i]);
303 } else {
304 result += coeffs[i] * integer_trail.UpperBound(vars[i]);
305 }
306 }
307 return result;
308}

341void CanonicalizeConstraint(LinearConstraint* ct) {
342 std::vector<std::pair<IntegerVariable, IntegerValue>> terms;
343
344 const int size = ct->vars.size();
345 for (int i = 0; i < size; ++i) {
346 if (VariableIsPositive(ct->vars[i])) {
347 terms.push_back({ct->vars[i], ct->coeffs[i]});
348 } else {
349 terms.push_back({NegationOf(ct->vars[i]), -ct->coeffs[i]});
350 }
351 }
352 std::sort(terms.begin(), terms.end());
353
354 ct->vars.clear();
355 ct->coeffs.clear();
356 for (const auto& term : terms) {
357 ct->vars.push_back(term.first);
358 ct->coeffs.push_back(term.second);
359 }
360}

362bool NoDuplicateVariable(const LinearConstraint& ct) {
363 absl::flat_hash_set<IntegerVariable> seen_variables;
364 const int size = ct.vars.size();
365 for (int i = 0; i < size; ++i) {
366 if (VariableIsPositive(ct.vars[i])) {
367 if (!seen_variables.insert(ct.vars[i]).second) return false;
368 } else {
369 if (!seen_variables.insert(NegationOf(ct.vars[i])).second) return false;
370 }
371 }
372 return true;
373}

375LinearExpression CanonicalizeExpr(const LinearExpression& expr) {
376 LinearExpression canonical_expr;
377 canonical_expr.offset = expr.offset;
378 for (int i = 0; i < expr.vars.size(); ++i) {
379 if (expr.coeffs[i] < 0) {
380 canonical_expr.vars.push_back(NegationOf(expr.vars[i]));
381 canonical_expr.coeffs.push_back(-expr.coeffs[i]);
382 } else {
383 canonical_expr.vars.push_back(expr.vars[i]);
384 canonical_expr.coeffs.push_back(expr.coeffs[i]);
385 }
386 }
387 return canonical_expr;
388}

390// TODO(user): Avoid duplication with PossibleIntegerOverflow() in the checker?
391// At least make sure the code is the same.
392bool ValidateLinearConstraintForOverflow(const LinearConstraint& constraint,
393 const IntegerTrail& integer_trail) {
394 int64_t positive_sum(0);
395 int64_t negative_sum(0);
396 for (int i = 0; i < constraint.vars.size(); ++i) {
397 const IntegerVariable var = constraint.vars[i];
398 const IntegerValue coeff = constraint.coeffs[i];
399 const IntegerValue lb = integer_trail.LevelZeroLowerBound(var);
400 const IntegerValue ub = integer_trail.LevelZeroUpperBound(var);
401
402 int64_t min_prod = CapProd(coeff.value(), lb.value());
403 int64_t max_prod = CapProd(coeff.value(), ub.value());
404 if (min_prod > max_prod) std::swap(min_prod, max_prod);
405
406 positive_sum = CapAdd(positive_sum, std::max(int64_t{0}, max_prod));
407 negative_sum = CapAdd(negative_sum, std::min(int64_t{0}, min_prod));
408 }
409
410 const int64_t limit = std::numeric_limits<int64_t>::max();
411 if (positive_sum >= limit) return false;
412 if (negative_sum <= -limit) return false;
413 if (CapSub(positive_sum, negative_sum) >= limit) return false;
414
415 return true;
416}
