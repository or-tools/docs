40void LinearConstraintBuilder::AddTerm(IntegerVariable var, IntegerValue coeff) {
41 if (coeff == 0) return;
42 // We can either add var or NegationOf(var), and we always choose the
43 // positive one.
44 if (VariableIsPositive(var)) {
45 terms_.push_back({var, coeff});
46 } else {
47 terms_.push_back({NegationOf(var), -coeff});
48 }
49}

51void LinearConstraintBuilder::AddTerm(AffineExpression expr,
52 IntegerValue coeff) {
53 if (coeff == 0) return;
54 // We can either add var or NegationOf(var), and we always choose the
55 // positive one.
56 if (expr.var != kNoIntegerVariable) {
57 if (VariableIsPositive(expr.var)) {
58 terms_.push_back({expr.var, coeff * expr.coeff});
59 } else {
60 terms_.push_back({NegationOf(expr.var), -coeff * expr.coeff});
61 }
62 }
63 offset_ += coeff * expr.constant;
64}

71void LinearConstraintBuilder::AddLinearExpression(const LinearExpression& expr,
72 IntegerValue coeff) {
73 for (int i = 0; i < expr.vars.size(); ++i) {
74 // We must use positive variables.
75 if (VariableIsPositive(expr.vars[i])) {
76 terms_.push_back({expr.vars[i], expr.coeffs[i] * coeff});
77 } else {
78 terms_.push_back({NegationOf(expr.vars[i]), -expr.coeffs[i] * coeff});
79 }
80 }
81 offset_ += expr.offset * coeff;
82}

84ABSL_MUST_USE_RESULT bool LinearConstraintBuilder::AddDecomposedProduct(
85 absl::Span<const LiteralValueValue> product) {
86 if (product.empty()) return true;
87
88 IntegerValue product_min = kMaxIntegerValue;
89 // TODO(user): Checks the value of literals.
90 for (const LiteralValueValue& term : product) {
91 product_min = std::min(product_min, term.left_value * term.right_value);
92 }
93
94 for (const LiteralValueValue& term : product) {
95 IntegerValue coeff = term.left_value * term.right_value - product_min;
96 if (coeff == 0) continue;
97 if (!AddLiteralTerm(term.literal, coeff)) {
98 return false;
99 }
100 }
101 AddConstant(product_min);
102 return true;
103}

105void LinearConstraintBuilder::AddQuadraticLowerBound(
106 AffineExpression left, AffineExpression right, IntegerTrail* integer_trail,
107 bool* is_quadratic) {
108 if (integer_trail->IsFixed(left)) {
109 AddTerm(right, integer_trail->FixedValue(left));
110 } else if (integer_trail->IsFixed(right)) {
111 AddTerm(left, integer_trail->FixedValue(right));
112 } else {
113 const IntegerValue left_min = integer_trail->LowerBound(left);
114 const IntegerValue right_min = integer_trail->LowerBound(right);
115 AddTerm(left, right_min);
116 AddTerm(right, left_min);
117 // Substract the energy counted twice.
118 AddConstant(-left_min * right_min);
119 if (is_quadratic != nullptr) *is_quadratic = true;
120 }
121}

127ABSL_MUST_USE_RESULT bool LinearConstraintBuilder::AddLiteralTerm(
128 Literal lit, IntegerValue coeff) {
129 DCHECK(encoder_ != nullptr);
130 IntegerVariable var = kNoIntegerVariable;
131 bool view_is_direct = true;
132 if (!encoder_->LiteralOrNegationHasView(lit, &var, &view_is_direct)) {
133 return false;
134 }
135
136 if (view_is_direct) {
137 AddTerm(var, coeff);
138 } else {
139 AddTerm(var, -coeff);
140 offset_ += coeff;
141 }
142 return true;
143}

165double ComputeActivity(
166 const LinearConstraint& constraint,
167 const absl::StrongVector<IntegerVariable, double>& values) {
168 int i = 0;
169 const int size = constraint.num_terms;
170 const int shifted_size = size - 3;
171 double a0 = 0.0;
172 double a1 = 0.0;
173 double a2 = 0.0;
174 double a3 = 0.0;
175 for (; i < shifted_size; i += 4) {
176 a0 += static_cast<double>(constraint.coeffs[i].value()) *
177 values[constraint.vars[i]];
178 a1 += static_cast<double>(constraint.coeffs[i + 1].value()) *
179 values[constraint.vars[i + 1]];
180 a2 += static_cast<double>(constraint.coeffs[i + 2].value()) *
181 values[constraint.vars[i + 2]];
182 a3 += static_cast<double>(constraint.coeffs[i + 3].value()) *
183 values[constraint.vars[i + 3]];
184 }
185 double activity = a0 + a1 + a2 + a3;
186 if (i < size) {
187 activity += static_cast<double>(constraint.coeffs[i].value()) *
188 values[constraint.vars[i]];
189 if (i + 1 < size) {
190 activity += static_cast<double>(constraint.coeffs[i + 1].value()) *
191 values[constraint.vars[i + 1]];
192 if (i + 2 < size) {
193 activity += static_cast<double>(constraint.coeffs[i + 2].value()) *
194 values[constraint.vars[i + 2]];
195 }
196 }
197 }
198 return activity;
199}

201double ComputeL2Norm(const LinearConstraint& ct) {
202 double sum = 0.0;
203 for (int i = 0; i < ct.num_terms; ++i) {
204 sum += ToDouble(ct.coeffs[i]) * ToDouble(ct.coeffs[i]);
205 }
206 return std::sqrt(sum);
207}

212 result = std::max(result, IntTypeAbs(ct.coeffs[i]));

217double ScalarProduct(const LinearConstraint& ct1, const LinearConstraint& ct2) {
218 DCHECK(std::is_sorted(ct1.vars.get(), ct1.vars.get() + ct1.num_terms));
219 DCHECK(std::is_sorted(ct2.vars.get(), ct2.vars.get() + ct2.num_terms));
220 double scalar_product = 0.0;
221 int index_1 = 0;
222 int index_2 = 0;
223 while (index_1 < ct1.num_terms && index_2 < ct2.num_terms) {
224 if (ct1.vars[index_1] == ct2.vars[index_2]) {
225 scalar_product +=
226 ToDouble(ct1.coeffs[index_1]) * ToDouble(ct2.coeffs[index_2]);
227 index_1++;
228 index_2++;
229 } else if (ct1.vars[index_1] > ct2.vars[index_2]) {
230 index_2++;
231 } else {
232 index_1++;
233 }
234 }
235 return scalar_product;
236}

241IntegerValue ComputeGcd(absl::Span<const IntegerValue> values) {
242 if (values.empty()) return IntegerValue(1);
243 int64_t gcd = 0;
244 for (const IntegerValue value : values) {
245 gcd = MathUtil::GCD64(gcd, std::abs(value.value()));
246 if (gcd == 1) break;
247 }
248 if (gcd < 0) return IntegerValue(1); // Can happen with kint64min.
249 return IntegerValue(gcd);
250}

254void DivideByGCD(LinearConstraint* constraint) {
255 if (constraint->num_terms == 0) return;
256 const IntegerValue gcd = ComputeGcd(
257 {constraint->coeffs.get(), static_cast<size_t>(constraint->num_terms)});
258 if (gcd == 1) return;
259
260 if (constraint->lb > kMinIntegerValue) {
261 constraint->lb = CeilRatio(constraint->lb, gcd);
262 }
263 if (constraint->ub < kMaxIntegerValue) {
264 constraint->ub = FloorRatio(constraint->ub, gcd);
265 }
266 for (int i = 0; i < constraint->num_terms; ++i) {
267 constraint->coeffs[i] /= gcd;
268 }
269}

271void RemoveZeroTerms(LinearConstraint* constraint) {
272 int new_size = 0;
273 const int size = constraint->num_terms;
274 for (int i = 0; i < size; ++i) {
275 if (constraint->coeffs[i] == 0) continue;
276 constraint->vars[new_size] = constraint->vars[i];
277 constraint->coeffs[new_size] = constraint->coeffs[i];
278 ++new_size;
279 }
280 constraint->resize(new_size);

360bool NoDuplicateVariable(const LinearConstraint& ct) {
361 absl::flat_hash_set<IntegerVariable> seen_variables;
362 const int size = ct.num_terms;
363 for (int i = 0; i < size; ++i) {
364 if (VariableIsPositive(ct.vars[i])) {
365 if (!seen_variables.insert(ct.vars[i]).second) return false;
366 } else {
367 if (!seen_variables.insert(NegationOf(ct.vars[i])).second) return false;
368 }
369 }
370 return true;
371}

373LinearExpression CanonicalizeExpr(const LinearExpression& expr) {
374 LinearExpression canonical_expr;
375 canonical_expr.offset = expr.offset;
376 for (int i = 0; i < expr.vars.size(); ++i) {
377 if (expr.coeffs[i] < 0) {
378 canonical_expr.vars.push_back(NegationOf(expr.vars[i]));
379 canonical_expr.coeffs.push_back(-expr.coeffs[i]);
380 } else {
381 canonical_expr.vars.push_back(expr.vars[i]);
382 canonical_expr.coeffs.push_back(expr.coeffs[i]);
383 }
384 }
385 return canonical_expr;
386}

388// TODO(user): Avoid duplication with PossibleIntegerOverflow() in the checker?
389// At least make sure the code is the same.
390bool ValidateLinearConstraintForOverflow(const LinearConstraint& constraint,
391 const IntegerTrail& integer_trail) {
392 int64_t positive_sum(0);
393 int64_t negative_sum(0);
394 for (int i = 0; i < constraint.num_terms; ++i) {
395 const IntegerVariable var = constraint.vars[i];
396 const IntegerValue coeff = constraint.coeffs[i];
397 const IntegerValue lb = integer_trail.LevelZeroLowerBound(var);
398 const IntegerValue ub = integer_trail.LevelZeroUpperBound(var);
399
400 int64_t min_prod = CapProd(coeff.value(), lb.value());
401 int64_t max_prod = CapProd(coeff.value(), ub.value());
402 if (min_prod > max_prod) std::swap(min_prod, max_prod);
403
404 positive_sum = CapAdd(positive_sum, std::max(int64_t{0}, max_prod));
405 negative_sum = CapAdd(negative_sum, std::min(int64_t{0}, min_prod));
406 }
407
408 const int64_t limit = std::numeric_limits<int64_t>::max();
409 if (positive_sum >= limit) return false;
410 if (negative_sum <= -limit) return false;
411 if (CapSub(positive_sum, negative_sum) >= limit) return false;
412
413 return true;
414}

462bool PossibleOverflow(const IntegerTrail& integer_trail,
463 const LinearConstraint& constraint) {
464 IntegerValue min_activity(0);
465 IntegerValue max_activity(0);
466 const int size = constraint.num_terms;
467 for (int i = 0; i < size; ++i) {
468 const IntegerVariable var = constraint.vars[i];
469 const IntegerValue coeff = constraint.coeffs[i];
470 CHECK_NE(coeff, 0);
471 const IntegerValue lb = integer_trail.LevelZeroLowerBound(var);
472 const IntegerValue ub = integer_trail.LevelZeroUpperBound(var);
473 if (coeff > 0) {
474 if (!AddProductTo(lb, coeff, &min_activity)) return true;
475 if (!AddProductTo(ub, coeff, &max_activity)) return true;
476 } else {
477 if (!AddProductTo(ub, coeff, &min_activity)) return true;
478 if (!AddProductTo(lb, coeff, &max_activity)) return true;
479 }
480 }
481 return AtMinOrMaxInt64(CapSub(max_activity.value(), min_activity.value()));
482}
