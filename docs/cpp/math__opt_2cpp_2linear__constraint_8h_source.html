31// A value type that references a linear constraint from ModelStorage. Usually
32// this type is passed by copy.
33class LinearConstraint {
34 public:
35 // The typed integer used for ids.
36 using IdType = LinearConstraintId;
37
38 inline LinearConstraint(const ModelStorage* storage, LinearConstraintId id);
39
40 inline int64_t id() const;
41
42 inline LinearConstraintId typed_id() const;
43 inline const ModelStorage* storage() const;
44
45 inline double lower_bound() const;
46 inline double upper_bound() const;
47 inline const std::string& name() const;
48
49 inline bool is_coefficient_nonzero(Variable variable) const;
50
51 // Returns 0.0 if the variable is not used in the constraint.
52 inline double coefficient(Variable variable) const;
53
54 friend inline bool operator==(const LinearConstraint& lhs,
55 const LinearConstraint& rhs);
56 friend inline bool operator!=(const LinearConstraint& lhs,
57 const LinearConstraint& rhs);
58 template <typename H>
59 friend H AbslHashValue(H h, const LinearConstraint& linear_constraint);
60 friend std::ostream& operator<<(std::ostream& ostr,
61 const LinearConstraint& linear_constraint);
62
63 private:
64 const ModelStorage* storage_;
65 LinearConstraintId id_;
66};

68// Implements the API of std::unordered_map<LinearConstraint, V>, but forbids
69// LinearConstraints from different models in the same map.

98bool LinearConstraint::is_coefficient_nonzero(const Variable variable) const {
99 CHECK_EQ(variable.storage(), storage_)
100 << internal::kObjectsFromOtherModelStorage;
101 return storage_->is_linear_constraint_coefficient_nonzero(
102 id_, variable.typed_id());
103}
104
105double LinearConstraint::coefficient(const Variable variable) const {
106 CHECK_EQ(variable.storage(), storage_)
107 << internal::kObjectsFromOtherModelStorage;
108 return storage_->linear_constraint_coefficient(id_, variable.typed_id());
109}
110
111bool operator==(const LinearConstraint& lhs, const LinearConstraint& rhs) {
112 return lhs.id_ == rhs.id_ && lhs.storage_ == rhs.storage_;
113}
114
115bool operator!=(const LinearConstraint& lhs, const LinearConstraint& rhs) {
116 return !(lhs == rhs);
117}
118
119template <typename H>
120H AbslHashValue(H h, const LinearConstraint& linear_constraint) {
121 return H::combine(std::move(h), linear_constraint.id_.value(),
122 linear_constraint.storage_);
123}
124
125std::ostream& operator<<(std::ostream& ostr,
126 const LinearConstraint& linear_constraint) {
127 ostr << linear_constraint.name();
128 return ostr;
129}
