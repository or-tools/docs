40// this type is passed by copy.

130BoundedLinearExpression LinearConstraint::AsBoundedLinearExpression() const {
131 LinearExpression terms;
132 for (const VariableId var :
133 storage()->variables_in_linear_constraint(typed_id())) {
134 terms += Variable(storage(), var) *
135 storage()->linear_constraint_coefficient(typed_id(), var);
136 }
137 return storage()->linear_constraint_lower_bound(typed_id()) <=
138 std::move(terms) <=
139 storage()->linear_constraint_upper_bound(typed_id());
140}

142std::string LinearConstraint::ToString() const {
143 if (!storage()->has_linear_constraint(id_)) {
144 return std::string(kDeletedConstraintDefaultDescription);
145 }
146 std::stringstream str;
147 str << AsBoundedLinearExpression();
148 return str.str();
149}

151bool operator==(const LinearConstraint& lhs, const LinearConstraint& rhs) {
152 return lhs.id_ == rhs.id_ && lhs.storage_ == rhs.storage_;
153}
154
155bool operator!=(const LinearConstraint& lhs, const LinearConstraint& rhs) {
156 return !(lhs == rhs);
157}

159template <typename H>
160H AbslHashValue(H h, const LinearConstraint& linear_constraint) {
161 return H::combine(std::move(h), linear_constraint.id_.value(),
162 linear_constraint.storage_);
163}
