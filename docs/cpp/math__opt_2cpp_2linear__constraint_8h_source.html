32// A value type that references a linear constraint from ModelStorage. Usually
33// this type is passed by copy.
34class LinearConstraint {
35 public:
36 // The typed integer used for ids.
37 using IdType = LinearConstraintId;
38
39 inline LinearConstraint(const ModelStorage* storage, LinearConstraintId id);
40
41 inline int64_t id() const;
42
43 inline LinearConstraintId typed_id() const;
44 inline const ModelStorage* storage() const;
45
46 inline double lower_bound() const;
47 inline double upper_bound() const;
48 inline const std::string& name() const;
49
50 inline bool is_coefficient_nonzero(Variable variable) const;
51
52 // Returns 0.0 if the variable is not used in the constraint.
53 inline double coefficient(Variable variable) const;
54
55 friend inline bool operator==(const LinearConstraint& lhs,
56 const LinearConstraint& rhs);
57 friend inline bool operator!=(const LinearConstraint& lhs,
58 const LinearConstraint& rhs);
59 template <typename H>
60 friend H AbslHashValue(H h, const LinearConstraint& linear_constraint);
61 friend std::ostream& operator<<(std::ostream& ostr,
62 const LinearConstraint& linear_constraint);
63
64 private:
65 const ModelStorage* storage_;
66 LinearConstraintId id_;
67};

99bool LinearConstraint::is_coefficient_nonzero(const Variable variable) const {
100 CHECK_EQ(variable.storage(), storage_)
101 << internal::kObjectsFromOtherModelStorage;
102 return storage_->is_linear_constraint_coefficient_nonzero(
103 id_, variable.typed_id());
104}
105
106double LinearConstraint::coefficient(const Variable variable) const {
107 CHECK_EQ(variable.storage(), storage_)
108 << internal::kObjectsFromOtherModelStorage;
109 return storage_->linear_constraint_coefficient(id_, variable.typed_id());
110}
111
112bool operator==(const LinearConstraint& lhs, const LinearConstraint& rhs) {
113 return lhs.id_ == rhs.id_ && lhs.storage_ == rhs.storage_;
114}
115
116bool operator!=(const LinearConstraint& lhs, const LinearConstraint& rhs) {
117 return !(lhs == rhs);
118}
119
120template <typename H>
121H AbslHashValue(H h, const LinearConstraint& linear_constraint) {
122 return H::combine(std::move(h), linear_constraint.id_.value(),
123 linear_constraint.storage_);
124}

126std::ostream& operator<<(std::ostream& ostr,
127 const LinearConstraint& linear_constraint) {
128 // TODO(b/170992529): handle quoting of invalid characters in the name.
129 const std::string& name = linear_constraint.name();
130 if (name.empty()) {
131 ostr << "__lin_con#" << linear_constraint.id() << "__";
132 } else {
133 ostr << name;
134 }
135 return ostr;
136}
