26public class IntegerProgramming {
27 private static void runIntegerProgrammingExample(String solverType) {
28 MPSolver solver = MPSolver.createSolver(solverType);
29 if (solver == null) {
30 System.out.println("Could not create solver " + solverType);
31 return;
32 }
33 double infinity = java.lang.Double.POSITIVE_INFINITY;
34 // x1 and x2 are integer non-negative variables.
35 MPVariable x1 = solver.makeIntVar(0.0, infinity, "x1");
36 MPVariable x2 = solver.makeIntVar(0.0, infinity, "x2");
37
38 // Minimize x1 + 2 * x2.
39 MPObjective objective = solver.objective();
40 objective.setCoefficient(x1, 1);
41 objective.setCoefficient(x2, 2);
42
43 // 2 * x2 + 3 * x1 >= 17.
44 MPConstraint ct = solver.makeConstraint(17, infinity);
45 ct.setCoefficient(x1, 3);
46 ct.setCoefficient(x2, 2);
47
48 final MPSolver.ResultStatus resultStatus = solver.solve();
49
50 // Check that the problem has an optimal solution.
51 if (resultStatus != MPSolver.ResultStatus.OPTIMAL) {
52 System.err.println("The problem does not have an optimal solution!");
53 return;
54 }
55
56 // Verify that the solution satisfies all constraints (when using solvers
57 // others than GLOP_LINEAR_PROGRAMMING, this is highly recommended!).
58 if (!solver.verifySolution(/*tolerance=*/1e-7, /* log_errors= */ true)) {
59 System.err.println("The solution returned by the solver violated the"
60 + " problem constraints by at least 1e-7");
61 return;
62 }
63
64 System.out.println("Problem solved in " + solver.wallTime() + " milliseconds");
65
66 // The objective value of the solution.
67 System.out.println("Optimal objective value = " + solver.objective().value());
68
69 // The value of each variable in the solution.
70 System.out.println("x1 = " + x1.solutionValue());
71 System.out.println("x2 = " + x2.solutionValue());
72
73 System.out.println("Advanced usage:");
74 System.out.println("Problem solved in " + solver.nodes() + " branch-and-bound nodes");
75 }
