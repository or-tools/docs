24public class Xkcd {
26 private static void solve() {
27 Solver solver = new Solver("Xkcd");
28
29 int n = 6;
30 // for price and total: multiplied by 100 to be able to use integers
31 int[] price = {215, 275, 335, 355, 420, 580};
32 int total = 1505;
33
34 //
35 // Variables
36 //
37 IntVar[] x = solver.makeIntVarArray(n, 0, 10, "x");
38
39 //
40 // Constraints
41 //
42 solver.addConstraint(solver.makeEquality(solver.makeScalProd(x, price).var(), total));
43
44 //
45 // Search
46 //
47 DecisionBuilder db = solver.makePhase(x, solver.CHOOSE_FIRST_UNBOUND, solver.ASSIGN_MIN_VALUE);
48 solver.newSearch(db);
49
50 while (solver.nextSolution()) {
51 System.out.print("x: ");
52 for (int i = 0; i < n; i++) {
53 System.out.print(x[i].value() + " ");
54 }
55 System.out.println();
56 }
57 solver.endSearch();
58
59 // Statistics
60 System.out.println();
61 System.out.println("Solutions: " + solver.solutions());
62 System.out.println("Failures: " + solver.failures());
63 System.out.println("Branches: " + solver.branches());
64 System.out.println("Wall time: " + solver.wallTime() + "ms");
65 }
