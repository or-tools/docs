39 private static void solve(String solverType) {
40 System.out.println("---- CoinsGridMIP with " + solverType);
41
42 MPSolver solver = MPSolver.createSolver(solverType);
43 if (solver == null)
44 return;
45
46 double infinity = MPSolver.infinity();
47
49 int noCols = 5; // variables number
50 int noNodes = 11; // constraints number
51
52 Edge[] edges = {new Edge(1, 2), new Edge(1, 4), new Edge(1, 7), new Edge(1, 9), new Edge(2, 3),
53 new Edge(2, 6), new Edge(2, 8), new Edge(3, 5), new Edge(3, 7), new Edge(3, 10),
54 new Edge(4, 5), new Edge(4, 6), new Edge(4, 10), new Edge(5, 8), new Edge(5, 9),
55 new Edge(6, 11), new Edge(7, 11), new Edge(8, 11), new Edge(9, 11), new Edge(10, 11)};
56
58 MPVariable[][] x = new MPVariable[noNodes][noCols];
59 for (Integer i = 0; i < noNodes; i++) {
60 x[i] = solver.makeBoolVarArray(noCols);
61 }
62
63 MPVariable[] colUsed = solver.makeBoolVarArray(noCols);
64
65 MPObjective obj = solver.objective();
66 for (MPVariable objVar : colUsed) {
67 obj.setCoefficient(objVar, 1);
68 }
69
71 MPConstraint[] constraints = new MPConstraint[noNodes];
72 for (int i = 0; i < noNodes; i++) {
73 constraints[i] = solver.makeConstraint(1, 1);
74 for (int j = 0; j < noCols; j++) {
75 constraints[i].setCoefficient(x[i][j], 1);
76 }
77 }
78
80 MPConstraint[][] adjacencies = new MPConstraint[edges.length][noCols];
81 for (int i = 0; i < edges.length; i++) {
82 for (int j = 0; j < noCols; j++) {
83 adjacencies[i][j] = solver.makeConstraint(-infinity, 0);
84 adjacencies[i][j].setCoefficient(x[edges[i].a - 1][j], 1);
85 adjacencies[i][j].setCoefficient(x[edges[i].b - 1][j], 1);
86 adjacencies[i][j].setCoefficient(colUsed[j], -1);
87 }
88 }
89
91 final MPSolver.ResultStatus resultStatus = solver.solve();
92
94 if (resultStatus != MPSolver.ResultStatus.OPTIMAL) {
95 System.err.println("The problem does not have an optimal solution!");
96 return;
97 } else {
98 System.out.println("Problem solved in " + solver.wallTime() + "ms");
99
100 System.out.print("Colors used: ");
101 for (MPVariable var : colUsed) {
102 System.out.print((int) var.solutionValue() + " ");
103 }
104 System.out.println("\n");
105
106 for (int i = 0; i < noNodes; i++) {
107 System.out.print("Col of vertex " + i + " : ");
108 for (int j = 0; j < noCols; j++) {
109 if (x[i][j].solutionValue() > 0) {
110 System.out.println(j);
111 }
112 }
113 }
114 }
115 }
