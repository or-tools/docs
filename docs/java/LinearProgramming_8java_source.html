26public class LinearProgramming {
27 private static void runLinearProgrammingExample(String solverType, boolean printModel) {
28 MPSolver solver = MPSolver.createSolver(solverType);
29 if (solver == null) {
30 System.out.println("Could not create solver " + solverType);
31 return;
32 }
33 double infinity = java.lang.Double.POSITIVE_INFINITY;
34 // x1, x2 and x3 are continuous non-negative variables.
35 MPVariable x1 = solver.makeNumVar(0.0, infinity, "x1");
36 MPVariable x2 = solver.makeNumVar(0.0, infinity, "x2");
37 MPVariable x3 = solver.makeNumVar(0.0, infinity, "x3");
38
39 // Maximize 10 * x1 + 6 * x2 + 4 * x3.
40 MPObjective objective = solver.objective();
41 objective.setCoefficient(x1, 10);
42 objective.setCoefficient(x2, 6);
43 objective.setCoefficient(x3, 4);
44 objective.setMaximization();
45
46 // x1 + x2 + x3 <= 100.
47 MPConstraint c0 = solver.makeConstraint(-infinity, 100.0);
48 c0.setCoefficient(x1, 1);
49 c0.setCoefficient(x2, 1);
50 c0.setCoefficient(x3, 1);
51
52 // 10 * x1 + 4 * x2 + 5 * x3 <= 600.
53 MPConstraint c1 = solver.makeConstraint(-infinity, 600.0);
54 c1.setCoefficient(x1, 10);
55 c1.setCoefficient(x2, 4);
56 c1.setCoefficient(x3, 5);
57
58 // 2 * x1 + 2 * x2 + 6 * x3 <= 300.
59 MPConstraint c2 = solver.makeConstraint(-infinity, 300.0);
60 c2.setCoefficient(x1, 2);
61 c2.setCoefficient(x2, 2);
62 c2.setCoefficient(x3, 6);
63
64 System.out.println("Number of variables = " + solver.numVariables());
65 System.out.println("Number of constraints = " + solver.numConstraints());
66
67 if (printModel) {
68 String model = solver.exportModelAsLpFormat();
69 System.out.println(model);
70 }
71
72 final MPSolver.ResultStatus resultStatus = solver.solve();
73
74 // Check that the problem has an optimal solution.
75 if (resultStatus != MPSolver.ResultStatus.OPTIMAL) {
76 System.err.println("The problem does not have an optimal solution!");
77 return;
78 }
79
80 // Verify that the solution satisfies all constraints (when using solvers
81 // others than GLOP_LINEAR_PROGRAMMING, this is highly recommended!).
82 if (!solver.verifySolution(/*tolerance=*/1e-7, /* log_errors= */ true)) {
83 System.err.println("The solution returned by the solver violated the"
84 + " problem constraints by at least 1e-7");
85 return;
86 }
87
88 System.out.println("Problem solved in " + solver.wallTime() + " milliseconds");
89
90 // The objective value of the solution.
91 System.out.println("Optimal objective value = " + solver.objective().value());
92
93 // The value of each variable in the solution.
94 System.out.println("x1 = " + x1.solutionValue());
95 System.out.println("x2 = " + x2.solutionValue());
96 System.out.println("x3 = " + x3.solutionValue());
97
98 final double[] activities = solver.computeConstraintActivities();
99
100 System.out.println("Advanced usage:");
101 System.out.println("Problem solved in " + solver.iterations() + " iterations");
102 System.out.println("x1: reduced cost = " + x1.reducedCost());
103 System.out.println("x2: reduced cost = " + x2.reducedCost());
104 System.out.println("x3: reduced cost = " + x3.reducedCost());
105 System.out.println("c0: dual value = " + c0.dualValue());
106 System.out.println(" activity = " + activities[c0.index()]);
107 System.out.println("c1: dual value = " + c1.dualValue());
108 System.out.println(" activity = " + activities[c1.index()]);
109 System.out.println("c2: dual value = " + c2.dualValue());
110 System.out.println(" activity = " + activities[c2.index()]);
111 }

119 }
