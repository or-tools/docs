38public class GolombRuler {
40 private static void solve(int m) {
41 Solver solver = new Solver("GR " + m);
42
43 IntVar[] ticks = solver.makeIntVarArray(m, 0, ((m < 31) ? (1 << (m + 1)) - 1 : 9999), "ticks");
44
45 solver.addConstraint(solver.makeEquality(ticks[0], 0));
46
47 for (int i = 0; i < ticks.length - 1; i++) {
48 solver.addConstraint(solver.makeLess(ticks[i], ticks[i + 1]));
49 }
50
51 IntVar[] diff = new IntVar[(m * m - m) / 2];
52
53 for (int k = 0, i = 0; i < m - 1; i++) {
54 for (int j = i + 1; j < m; j++, k++) {
55 diff[k] = solver.makeDifference(ticks[j], ticks[i]).var();
56 solver.addConstraint(solver.makeGreaterOrEqual(diff[k], (j - i) * (j - i + 1) / 2));
57 }
58 }
59
60 solver.addConstraint(solver.makeAllDifferent(diff));
61
62 // break symetries
63 if (m > 2) {
64 solver.addConstraint(solver.makeLess(diff[0], diff[diff.length - 1]));
65 }
66
67 OptimizeVar opt = solver.makeMinimize(ticks[m - 1], 1);
68 DecisionBuilder db =
69 solver.makePhase(ticks, solver.CHOOSE_MIN_SIZE_LOWEST_MIN, solver.ASSIGN_MIN_VALUE);
70 SolutionCollector collector = solver.makeLastSolutionCollector();
71 collector.add(ticks);
72 collector.addObjective(ticks[m - 1]);
73 SearchMonitor log = solver.makeSearchLog(10000, opt);
74 solver.solve(db, opt, log, collector);
75 System.out.println("Optimal solution = " + collector.objectiveValue(0));
76 for (int i = 0; i < m; ++i) {
77 System.out.print("[" + collector.value(0, ticks[i]) + "] ");
78 }
79 System.out.println();
80 }
