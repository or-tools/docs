16 *
17 * Coins grid problem in Google CP Solver.
18 *
19 * Problem from
20 * Tony Hurlimann: "A coin puzzle - SVOR-contest 2007"
21 * http://www.svor.ch/competitions/competition2007/AsroContestSolution.pdf
22 * "
23 * In a quadratic grid (or a larger chessboard) with 31x31 cells, one should
24 * place coins in such a way that the following conditions are fulfilled:
25 * 1. In each row exactly 14 coins must be placed.
26 * 2. In each column exactly 14 coins must be placed.
27 * 3. The sum of the quadratic horizontal distance from the main diagonal
28 * of all cells containing a coin must be as small as possible.
29 * 4. In each cell at most one coin can be placed.
30 * The description says to place 14x31 = 434 coins on the chessboard each row
31 * containing 14 coins and each column also containing 14 coins.
32 * "
33 *
34 * This is a Java MIP version of
35 * http://www.hakank.org/google_or_tools/coins_grid_mip.py
36 *
37 * which is the MIP version of
38 * http://www.hakank.org/google_or_tools/coins_grid.py
39 *
40 * by Hakan Kjellerstrand (hakank@gmail.com).
41 *
42 * Java version by Darian Sastre (darian.sastre@minimaxlabs.com)
43 */

49public class CoinsGridMIP {
50 private static void solve(String solverType) {
51 System.out.println("---- CoinsGridMIP with " + solverType);
52
53 MPSolver solver = MPSolver.createSolver(solverType);
54 if (solver == null)
55 return;
56
58 int n = 31;
59 int c = 14;
60
62 MPVariable[][] x = new MPVariable[n][n];
63 for (int i = 0; i < n; i++) {
64 x[i] = solver.makeBoolVarArray(n);
65 }
66
68 MPConstraint[] constraints = new MPConstraint[2 * n];
69 MPObjective obj = solver.objective();
70
71 for (int i = 0; i < n; i++) {
72 constraints[2 * i] = solver.makeConstraint(c, c);
73 constraints[2 * i + 1] = solver.makeConstraint(c, c);
74
75 for (int j = 0; j < n; j++) {
76 constraints[2 * i].setCoefficient(x[i][j], 1);
77 constraints[2 * i + 1].setCoefficient(x[j][i], 1);
78
79 obj.setCoefficient(x[i][j], (i - j) * (j - i));
80 }
81 }
82
83 solver.solve();
84
85 System.out.println("Problem solved in " + solver.wallTime() + "ms");
86
87 for (int i = 0; i < n; i++) {
88 for (int j = 0; j < n; j++) {
89 System.out.print((int) x[i][j].solutionValue() + " ");
90 }
91 System.out.println();
92 }
93 }
94
95 public static void main(String[] args) {
96 Loader.loadNativeLibraries();
97 solve("SCIP");
98 solve("CBC");
99 solve("GLPK");
100 solve("SAT");
101 }
