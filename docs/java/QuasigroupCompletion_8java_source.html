24public class QuasigroupCompletion {
25 static int X = 0;
26
27 /*
28 * default problem
29 *
30 * Example from Ruben Martins and Ines Lynce
31 * Breaking Local Symmetries in Quasigroup Completion Problems, page 3
32 * The solution is unique:
33 *
34 * 1 3 2 5 4
35 * 2 5 4 1 3
36 * 4 1 3 2 5
37 * 5 4 1 3 2
38 * 3 2 5 4 1
39 */
40 static int default_n = 5;
41 static int[][] default_problem = {
42 {1, X, X, X, 4}, {X, 5, X, X, X}, {4, X, X, 2, X}, {X, 4, X, X, X}, {X, X, 5, X, 1}};
43
44 // for the actual problem
45 static int n;
46 static int[][] problem;

52 private static void solve() {
53 Solver solver = new Solver("QuasigroupCompletion");
54
55 //
56 // data
57 //
58 System.out.println("Problem:");
59 for (int i = 0; i < n; i++) {
60 for (int j = 0; j < n; j++) {
61 System.out.print(problem[i][j] + " ");
62 }
63 System.out.println();
64 }
65 System.out.println();
66
67 //
68 // Variables
69 //
70 IntVar[][] x = new IntVar[n][n];
71 IntVar[] x_flat = new IntVar[n * n]; // for branching
72 for (int i = 0; i < n; i++) {
73 for (int j = 0; j < n; j++) {
74 x[i][j] = solver.makeIntVar(1, n, "x[" + i + "," + j + "]");
75 x_flat[i * n + j] = x[i][j];
76 }
77 }
78
79 //
80 // Constraints
81 //
82 for (int i = 0; i < n; i++) {
83 for (int j = 0; j < n; j++) {
84 if (problem[i][j] > X) {
85 solver.addConstraint(solver.makeEquality(x[i][j], problem[i][j]));
86 }
87 }
88 }
89
90 //
91 // rows and columns must be different
92 //
93
94 // rows
95 for (int i = 0; i < n; i++) {
96 IntVar[] row = new IntVar[n];
97 for (int j = 0; j < n; j++) {
98 row[j] = x[i][j];
99 }
100 solver.addConstraint(solver.makeAllDifferent(row));
101 }
102
103 // columns
104 for (int j = 0; j < n; j++) {
105 IntVar[] col = new IntVar[n];
106 for (int i = 0; i < n; i++) {
107 col[i] = x[i][j];
108 }
109 solver.addConstraint(solver.makeAllDifferent(col));
110 }
111
112 //
113 // Search
114 //
115 DecisionBuilder db = solver.makePhase(x_flat, solver.INT_VAR_SIMPLE, solver.ASSIGN_MIN_VALUE);
116 solver.newSearch(db);
117
118 int sol = 0;
119 while (solver.nextSolution()) {
120 sol++;
121 System.out.println("Solution #" + sol + ":");
122 for (int i = 0; i < n; i++) {
123 for (int j = 0; j < n; j++) {
124 System.out.print(x[i][j].value() + " ");
125 }
126 System.out.println();
127 }
128 System.out.println();
129 }
130 solver.endSearch();
131
132 // Statistics
133 System.out.println();
134 System.out.println("Solutions: " + solver.solutions());
135 System.out.println("Failures: " + solver.failures());
136 System.out.println("Branches: " + solver.branches());
137 System.out.println("Wall time: " + solver.wallTime() + "ms");
138 }

148 private static void readFile(String file) {
149 System.out.println("readFile(" + file + ")");
150 int lineCount = 0;
151
152 try {
153 BufferedReader inr = new BufferedReader(new FileReader(file));
154 String str;
155 while ((str = inr.readLine()) != null && str.length() > 0) {
156 str = str.trim();
157
158 // ignore comments
159 if (str.startsWith("#") || str.startsWith("%")) {
160 continue;
161 }
162
163 System.out.println(str);
164 if (lineCount == 0) {
165 n = Integer.parseInt(str); // number of rows
166 problem = new int[n][n];
167 } else {
168 // the problem matrix
169 String row[] = str.split(" ");
170 for (int i = 0; i < n; i++) {
171 String s = row[i];
172 if (s.equals(".")) {
173 problem[lineCount - 1][i] = 0;
174 } else {
175 problem[lineCount - 1][i] = Integer.parseInt(s);
176 }
177 }
178 }
179
180 lineCount++;
181 } // end while
182
183 inr.close();
184
185 } catch (IOException e) {
186 System.out.println(e);
187 }
188 } // end readFile
