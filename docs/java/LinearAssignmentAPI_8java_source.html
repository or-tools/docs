25public class LinearAssignmentAPI {
26 private static void runAssignmentOn4x4Matrix() {
27 final int numSources = 4;
28 final int numTargets = 4;
29 final int[][] cost = {
30 {90, 76, 75, 80}, {35, 85, 55, 65}, {125, 95, 90, 105}, {45, 110, 95, 115}};
31 final int expectedCost = cost[0][3] + cost[1][2] + cost[2][1] + cost[3][0];
32
33 LinearSumAssignment assignment = new LinearSumAssignment();
34 for (int source = 0; source < numSources; ++source) {
35 for (int target = 0; target < numTargets; ++target) {
36 assignment.addArcWithCost(source, target, cost[source][target]);
37 }
38 }
39
40 if (assignment.solve() == LinearSumAssignment.Status.OPTIMAL) {
41 System.out.println("Total cost = " + assignment.getOptimalCost() + "/" + expectedCost);
42 for (int node = 0; node < assignment.getNumNodes(); ++node) {
43 System.out.println("Left node " + node + " assigned to right node "
44 + assignment.getRightMate(node) + " with cost " + assignment.getAssignmentCost(node));
45 }
46 } else {
47 System.out.println("No solution found.");
48 }
49 }
