23public class MagicSquare {
25 private static void solve(int n, int num) {
26 Solver solver = new Solver("MagicSquare");
27
28 System.out.println("n: " + n);
29
30 //
31 // variables
32 //
33 IntVar[][] x = new IntVar[n][n];
34
35 // for the branching
36 IntVar[] x_flat = new IntVar[n * n];
37
38 //
39 // constraints
40 //
41 final long s = (n * (n * n + 1)) / 2;
42 System.out.println("s: " + s);
43 // IntVar s = solver.makeIntVar(0, n*n*n, "s");
44
45 IntVar[] diag1 = new IntVar[n];
46 IntVar[] diag2 = new IntVar[n];
47 for (int i = 0; i < n; i++) {
48 IntVar[] row = new IntVar[n];
49 for (int j = 0; j < n; j++) {
50 x[i][j] = solver.makeIntVar(1, n * n, "x[" + i + "," + j + "]");
51 x_flat[i * n + j] = x[i][j];
52 row[j] = x[i][j];
53 }
54 // sum row to s
55 solver.addConstraint(solver.makeSumEquality(row, s));
56
57 diag1[i] = x[i][i];
58 diag2[i] = x[i][n - i - 1];
59 }
60 // sum diagonals to s
61 solver.addConstraint(solver.makeSumEquality(diag1, s));
62 solver.addConstraint(solver.makeSumEquality(diag2, s));
63
64 // sum columns to s
65 for (int j = 0; j < n; j++) {
66 IntVar[] col = new IntVar[n];
67 for (int i = 0; i < n; i++) {
68 col[i] = x[i][j];
69 }
70 solver.addConstraint(solver.makeSumEquality(col, s));
71 }
72
73 // all are different
74 solver.addConstraint(solver.makeAllDifferent(x_flat));
75
76 // symmetry breaking: upper left is 1
77 // solver.addConstraint(solver.makeEquality(x[0][0], 1));
78
79 //
80 // Solve
81 //
82 DecisionBuilder db =
83 solver.makePhase(x_flat, solver.CHOOSE_FIRST_UNBOUND, solver.ASSIGN_CENTER_VALUE);
84 solver.newSearch(db);
85 int c = 0;
86 while (solver.nextSolution()) {
87 for (int i = 0; i < n; i++) {
88 for (int j = 0; j < n; j++) {
89 System.out.print(x[i][j].value() + " ");
90 }
91 System.out.println();
92 }
93 System.out.println();
94 c++;
95 if (num > 0 && c >= num) {
96 break;
97 }
98 }
99 solver.endSearch();
100
101 // Statistics
102 System.out.println();
103 System.out.println("Solutions: " + solver.solutions());
104 System.out.println("Failures: " + solver.failures());
105 System.out.println("Branches: " + solver.branches());
106 System.out.println("Wall time: " + solver.wallTime() + "ms");
107 }
