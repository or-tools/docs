23public class Sudoku {
25 private static void solve() {
26 Solver solver = new Solver("Sudoku");
27
28 int cell_size = 3;
29 int n = cell_size * cell_size;
30
31 // 0 marks an unknown value
32 int[][] initial_grid = new int[][] {{0, 6, 0, 0, 5, 0, 0, 2, 0}, {0, 0, 0, 3, 0, 0, 0, 9, 0},
33 {7, 0, 0, 6, 0, 0, 0, 1, 0}, {0, 0, 6, 0, 3, 0, 4, 0, 0}, {0, 0, 4, 0, 7, 0, 1, 0, 0},
34 {0, 0, 5, 0, 9, 0, 8, 0, 0}, {0, 4, 0, 0, 0, 1, 0, 0, 6}, {0, 3, 0, 0, 0, 8, 0, 0, 0},
35 {0, 2, 0, 0, 4, 0, 0, 5, 0}};
36
37 //
38 // variables
39 //
40 IntVar[][] grid = new IntVar[n][n];
41 IntVar[] grid_flat = new IntVar[n * n];
42
43 for (int i = 0; i < n; i++) {
44 for (int j = 0; j < n; j++) {
45 grid[i][j] = solver.makeIntVar(1, 9, "grid[" + i + "," + j + "]");
46 grid_flat[i * n + j] = grid[i][j];
47 }
48 }
49
50 //
51 // constraints
52 //
53
54 // init and rows
55 for (int i = 0; i < n; i++) {
56 IntVar[] row = new IntVar[n];
57 for (int j = 0; j < n; j++) {
58 if (initial_grid[i][j] > 0) {
59 solver.addConstraint(solver.makeEquality(grid[i][j], initial_grid[i][j]));
60 }
61 row[j] = grid[i][j];
62 }
63 solver.addConstraint(solver.makeAllDifferent(row));
64 }
65
66 // columns
67 for (int j = 0; j < n; j++) {
68 IntVar[] col = new IntVar[n];
69 for (int i = 0; i < n; i++) {
70 col[i] = grid[i][j];
71 }
72 solver.addConstraint(solver.makeAllDifferent(col));
73 }
74
75 // cells
76 for (int i = 0; i < cell_size; i++) {
77 for (int j = 0; j < cell_size; j++) {
78 IntVar[] cell = new IntVar[n];
79 for (int di = 0; di < cell_size; di++) {
80 for (int dj = 0; dj < cell_size; dj++) {
81 cell[di * cell_size + dj] = grid[i * cell_size + di][j * cell_size + dj];
82 }
83 }
84 solver.addConstraint(solver.makeAllDifferent(cell));
85 }
86 }
87
88 //
89 // Search
90 //
91 DecisionBuilder db =
92 solver.makePhase(grid_flat, solver.INT_VAR_SIMPLE, solver.INT_VALUE_SIMPLE);
93
94 solver.newSearch(db);
95
96 while (solver.nextSolution()) {
97 for (int i = 0; i < n; i++) {
98 for (int j = 0; j < n; j++) {
99 System.out.print(grid[i][j].value() + " ");
100 }
101 System.out.println();
102 }
103 System.out.println();
104 }
105 solver.endSearch();
106
107 // Statistics
108 System.out.println();
109 System.out.println("Solutions: " + solver.solutions());
110 System.out.println("Failures: " + solver.failures());
111 System.out.println("Branches: " + solver.branches());
112 System.out.println("Wall time: " + solver.wallTime() + "ms");
113 }
