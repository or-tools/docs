52public class CapacitatedVehicleRoutingProblemWithTimeWindows {
53 private static Logger logger =
54 Logger.getLogger(CapacitatedVehicleRoutingProblemWithTimeWindows.class.getName());
55
56 // Locations representing either an order location or a vehicle route
57 // start/end.
58 private List<Pair<Integer, Integer>> locations = new ArrayList();
59
60 // Quantity to be picked up for each order.
61 private List<Integer> orderDemands = new ArrayList();
62 // Time window in which each order must be performed.
63 private List<Pair<Integer, Integer>> orderTimeWindows = new ArrayList();
64 // Penalty cost "paid" for dropping an order.
65 private List<Integer> orderPenalties = new ArrayList();
66
67 // Capacity of the vehicles.
68 private int vehicleCapacity = 0;
69 // Latest time at which each vehicle must end its tour.
70 private List<Integer> vehicleEndTime = new ArrayList();
71 // Cost per unit of distance of each vehicle.
72 private List<Integer> vehicleCostCoefficients = new ArrayList();
73 // Vehicle start and end indices. They have to be implemented as int[] due
74 // to the available SWIG-ed interface.
75 private int vehicleStarts[];
76 private int vehicleEnds[];
77
78 // Random number generator to produce data.
79 private final Random randomGenerator = new Random(0xBEEF);
80
87 private LongBinaryOperator buildManhattanCallback(
88 RoutingIndexManager manager, int costCoefficient) {
89 return new LongBinaryOperator() {
90 public long applyAsLong(long firstIndex, long secondIndex) {
91 try {
92 int firstNode = manager.indexToNode(firstIndex);
93 int secondNode = manager.indexToNode(secondIndex);
94 Pair<Integer, Integer> firstLocation = locations.get(firstNode);
95 Pair<Integer, Integer> secondLocation = locations.get(secondNode);
96 return (long) costCoefficient
97 * (Math.abs(firstLocation.first - secondLocation.first)
98 + Math.abs(firstLocation.second - secondLocation.second));
99 } catch (Throwable throwed) {
100 logger.warning(throwed.getMessage());
101 return 0;
102 }
103 }
104 };
105 }
106
120 private void buildOrders(int numberOfOrders, int xMax, int yMax, int demandMax, int timeWindowMax,
121 int timeWindowWidth, int penaltyMin, int penaltyMax) {
122 logger.info("Building orders.");
123 for (int order = 0; order < numberOfOrders; ++order) {
124 locations.add(Pair.of(randomGenerator.nextInt(xMax + 1), randomGenerator.nextInt(yMax + 1)));
125 orderDemands.add(randomGenerator.nextInt(demandMax + 1));
126 int timeWindowStart = randomGenerator.nextInt(timeWindowMax + 1);
127 orderTimeWindows.add(Pair.of(timeWindowStart, timeWindowStart + timeWindowWidth));
128 orderPenalties.add(randomGenerator.nextInt(penaltyMax - penaltyMin + 1) + penaltyMin);
129 }
130 }
131
143 private void buildFleet(
144 int numberOfVehicles, int xMax, int yMax, int endTime, int capacity, int costCoefficientMax) {
145 logger.info("Building fleet.");
146 vehicleCapacity = capacity;
147 vehicleStarts = new int[numberOfVehicles];
148 vehicleEnds = new int[numberOfVehicles];
149 for (int vehicle = 0; vehicle < numberOfVehicles; ++vehicle) {
150 vehicleStarts[vehicle] = locations.size();
151 locations.add(Pair.of(randomGenerator.nextInt(xMax + 1), randomGenerator.nextInt(yMax + 1)));
152 vehicleEnds[vehicle] = locations.size();
153 locations.add(Pair.of(randomGenerator.nextInt(xMax + 1), randomGenerator.nextInt(yMax + 1)));
154 vehicleEndTime.add(endTime);
155 vehicleCostCoefficients.add(randomGenerator.nextInt(costCoefficientMax) + 1);
156 }
157 }
158
160 private void solve(final int numberOfOrders, final int numberOfVehicles) {
161 logger.info(
162 "Creating model with " + numberOfOrders + " orders and " + numberOfVehicles + " vehicles.");
163 // Finalizing model
164 final int numberOfLocations = locations.size();
165
166 RoutingIndexManager manager =
167 new RoutingIndexManager(numberOfLocations, numberOfVehicles, vehicleStarts, vehicleEnds);
168 RoutingModel model = new RoutingModel(manager);
169
170 // Setting up dimensions
171 final int bigNumber = 100000;
172 final LongBinaryOperator callback = buildManhattanCallback(manager, 1);
173 final String timeStr = "time";
174 model.addDimension(
175 model.registerTransitCallback(callback), bigNumber, bigNumber, false, timeStr);
176 RoutingDimension timeDimension = model.getMutableDimension(timeStr);
177
178 LongUnaryOperator demandCallback = new LongUnaryOperator() {
179 public long applyAsLong(long index) {
180 try {
181 int node = manager.indexToNode(index);
182 if (node < numberOfOrders) {
183 return orderDemands.get(node);
184 }
185 return 0;
186 } catch (Throwable throwed) {
187 logger.warning(throwed.getMessage());
188 return 0;
189 }
190 }
191 };
192 final String capacityStr = "capacity";
193 model.addDimension(
194 model.registerUnaryTransitCallback(demandCallback), 0, vehicleCapacity, true, capacityStr);
195 RoutingDimension capacityDimension = model.getMutableDimension(capacityStr);
196
197 // Setting up vehicles
198 LongBinaryOperator[] callbacks = new LongBinaryOperator[numberOfVehicles];
199 for (int vehicle = 0; vehicle < numberOfVehicles; ++vehicle) {
200 final int costCoefficient = vehicleCostCoefficients.get(vehicle);
201 callbacks[vehicle] = buildManhattanCallback(manager, costCoefficient);
202 final int vehicleCost = model.registerTransitCallback(callbacks[vehicle]);
203 model.setArcCostEvaluatorOfVehicle(vehicleCost, vehicle);
204 timeDimension.cumulVar(model.end(vehicle)).setMax(vehicleEndTime.get(vehicle));
205 }
206
207 // Setting up orders
208 for (int order = 0; order < numberOfOrders; ++order) {
209 timeDimension.cumulVar(order).setRange(
210 orderTimeWindows.get(order).first, orderTimeWindows.get(order).second);
211 long[] orderIndices = {manager.nodeToIndex(order)};
212 model.addDisjunction(orderIndices, orderPenalties.get(order));
213 }
214
215 // Solving
216 RoutingSearchParameters parameters =
217 main.defaultRoutingSearchParameters()
218 .toBuilder()
219 .setFirstSolutionStrategy(FirstSolutionStrategy.Value.ALL_UNPERFORMED)
220 .build();
221
222 logger.info("Search");
223 Assignment solution = model.solveWithParameters(parameters);
224
225 if (solution != null) {
226 String output = "Total cost: " + solution.objectiveValue() + "\n";
227 // Dropped orders
228 String dropped = "";
229 for (int order = 0; order < numberOfOrders; ++order) {
230 if (solution.value(model.nextVar(order)) == order) {
231 dropped += " " + order;
232 }
233 }
234 if (dropped.length() > 0) {
235 output += "Dropped orders:" + dropped + "\n";
236 }
237 // Routes
238 for (int vehicle = 0; vehicle < numberOfVehicles; ++vehicle) {
239 String route = "Vehicle " + vehicle + ": ";
240 long order = model.start(vehicle);
241 // Empty route has a minimum of two nodes: Start => End
242 if (model.isEnd(solution.value(model.nextVar(order)))) {
243 route += "Empty";
244 } else {
245 for (; !model.isEnd(order); order = solution.value(model.nextVar(order))) {
246 IntVar load = capacityDimension.cumulVar(order);
247 IntVar time = timeDimension.cumulVar(order);
248 route += order + " Load(" + solution.value(load) + ") "
249 + "Time(" + solution.min(time) + ", " + solution.max(time) + ") -> ";
250 }
251 IntVar load = capacityDimension.cumulVar(order);
252 IntVar time = timeDimension.cumulVar(order);
253 route += order + " Load(" + solution.value(load) + ") "
254 + "Time(" + solution.min(time) + ", " + solution.max(time) + ")";
255 }
256 output += route + "\n";
257 }
258 logger.info(output);
259 }
260 }

270 final int timeWindowWidth = 4 * 60;

284 }
