24public class SetCovering {
26 private static void solve() {
27 Solver solver = new Solver("SetCovering");
28
29 //
30 // data
31 //
32 // Placing of firestations, from Winston 'Operations Research',
33 // page 486.
34 int min_distance = 15;
35 int num_cities = 6;
36
37 int[][] distance = {{0, 10, 20, 30, 30, 20}, {10, 0, 25, 35, 20, 10}, {20, 25, 0, 15, 30, 20},
38 {30, 35, 15, 0, 15, 25}, {30, 20, 30, 15, 0, 14}, {20, 10, 20, 25, 14, 0}};
39
40 //
41 // variables
42 //
43 IntVar[] x = solver.makeIntVarArray(num_cities, 0, 1, "x");
44 IntVar z = solver.makeSum(x).var();
45
46 //
47 // constraints
48 //
49
50 // ensure that all cities are covered
51 for (int i = 0; i < num_cities; i++) {
52 ArrayList<IntVar> b = new ArrayList<IntVar>();
53 for (int j = 0; j < num_cities; j++) {
54 if (distance[i][j] <= min_distance) {
55 b.add(x[j]);
56 }
57 }
58 solver.addConstraint(solver.makeSumGreaterOrEqual(b.toArray(new IntVar[1]), 1));
59 }
60
61 //
62 // objective
63 //
64 OptimizeVar objective = solver.makeMinimize(z, 1);
65
66 //
67 // search
68 //
69 DecisionBuilder db = solver.makePhase(x, solver.INT_VAR_DEFAULT, solver.INT_VALUE_DEFAULT);
70 solver.newSearch(db, objective);
71
72 //
73 // output
74 //
75 while (solver.nextSolution()) {
76 System.out.println("z: " + z.value());
77 System.out.print("x: ");
78 for (int i = 0; i < num_cities; i++) {
79 System.out.print(x[i].value() + " ");
80 }
81 System.out.println();
82 }
83 solver.endSearch();
84
85 // Statistics
86 System.out.println();
87 System.out.println("Solutions: " + solver.solutions());
88 System.out.println("Failures: " + solver.failures());
89 System.out.println("Branches: " + solver.branches());
90 System.out.println("Wall time: " + solver.wallTime() + "ms");
91 }
