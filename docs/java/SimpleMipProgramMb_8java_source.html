14// Minimal example to call the MIP solver.
15// [START program]
16package com.google.ortools.modelbuilder.samples;
17// [START import]
18import com.google.ortools.Loader;
19import com.google.ortools.modelbuilder.LinearExpr;
20import com.google.ortools.modelbuilder.ModelBuilder;
21import com.google.ortools.modelbuilder.ModelSolver;
22import com.google.ortools.modelbuilder.SolveStatus;
23import com.google.ortools.modelbuilder.Variable;
24// [END import]
25
27public final class SimpleMipProgramMb {
28 public static void main(String[] args) {
29 Loader.loadNativeLibraries();
30 // [START model]
31 // Create the linear model.
32 ModelBuilder model = new ModelBuilder();
33 // [END model]
34
35 // [START variables]
36 double infinity = java.lang.Double.POSITIVE_INFINITY;
37 // x and y are integer non-negative variables.
38 Variable x = model.newIntVar(0.0, infinity, "x");
39 Variable y = model.newIntVar(0.0, infinity, "y");
40
41 System.out.println("Number of variables = " + model.numVariables());
42 // [END variables]
43
44 // [START constraints]
45 // x + 7 * y <= 17.5.
46 model.addLessOrEqual(LinearExpr.newBuilder().add(x).addTerm(y, 7), 17.5).withName("c0");
47
48 // x <= 3.5.
49 model.addLessOrEqual(x, 3.5).withName("c1");
50
51 System.out.println("Number of constraints = " + model.numConstraints());
52 // [END constraints]
53
54 // [START objective]
55 // Maximize x + 10 * y.
56 model.maximize(LinearExpr.newBuilder().add(x).addTerm(y, 10.0));
57 // [END objective]
58
59 // [START solve]
60 // Solve with the SCIP MIP solver.
61 ModelSolver solver = new ModelSolver("scip");
62 final SolveStatus status = solver.solve(model);
63 // [END solve]
64
65 // [START print_solution]
66 if (status == SolveStatus.OPTIMAL) {
67 System.out.println("Solution:");
68 System.out.println("Objective value = " + solver.getObjectiveValue());
69 System.out.println("x = " + solver.getValue(x));
70 System.out.println("y = " + solver.getValue(y));
71 } else {
72 System.err.println("The problem does not have an optimal solution!");
73 }
74 // [END print_solution]
75
76 // [START advanced]
77 System.out.println("\nAdvanced usage:");
78 System.out.println("Problem solved in " + solver.getWallTime() + " seconds");
79 // [END advanced]
80 }
81
82 private SimpleMipProgramMb() {}
83}
