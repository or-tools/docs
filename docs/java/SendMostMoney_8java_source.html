24public class SendMostMoney {
29 private static long solve(long MONEY) {
30 Solver solver = new Solver("SendMostMoney");
31
32 //
33 // data
34 //
35 final int base = 10;
36
37 //
38 // variables
39 //
40 IntVar s = solver.makeIntVar(0, base - 1, "s");
41 IntVar e = solver.makeIntVar(0, base - 1, "e");
42 IntVar n = solver.makeIntVar(0, base - 1, "n");
43 IntVar d = solver.makeIntVar(0, base - 1, "d");
44 IntVar m = solver.makeIntVar(0, base - 1, "m");
45 IntVar o = solver.makeIntVar(0, base - 1, "o");
46 IntVar t = solver.makeIntVar(0, base - 1, "t");
47 IntVar y = solver.makeIntVar(0, base - 1, "y");
48
49 IntVar[] x = {s, e, n, d, m, o, t, y};
50
51 IntVar[] eq = {s, e, n, d, m, o, s, t, m, o, n, e, y};
52 int[] coeffs = {
53 1000, 100, 10,
54 1, // S E N D +
55 1000, 100, 10,
56 1, // M O S T
57 -10000, -1000, -100, -10,
58 -1 // == M O N E Y
59 };
60 solver.addConstraint(solver.makeScalProdEquality(eq, coeffs, 0));
61
62 IntVar money =
63 solver.makeScalProd(new IntVar[] {m, o, n, e, y}, new int[] {10000, 1000, 100, 10, 1})
64 .var();
65
66 //
67 // constraints
68 //
69
70 // s > 0
71 solver.addConstraint(solver.makeGreater(s, 0));
72 // m > 0
73 solver.addConstraint(solver.makeGreater(m, 0));
74
75 solver.addConstraint(solver.makeAllDifferent(x));
76
77 if (MONEY > 0) {
78 // Search for all solutions.
79 solver.addConstraint(solver.makeEquality(money, MONEY));
80 }
81
82 //
83 // search
84 //
85 DecisionBuilder db = solver.makePhase(x, solver.CHOOSE_FIRST_UNBOUND, solver.ASSIGN_MAX_VALUE);
86
87 if (MONEY == 0) {
88 // first round: get the optimal value
89 OptimizeVar obj = solver.makeMaximize(money, 1);
90 solver.newSearch(db, obj);
91 } else {
92 // search for all solutions
93 solver.newSearch(db);
94 }
95
96 long money_ret = 0;
97 while (solver.nextSolution()) {
98 System.out.println("money: " + money.value());
99 money_ret = money.value();
100 for (int i = 0; i < x.length; i++) {
101 System.out.print(x[i].value() + " ");
102 }
103 System.out.println();
104 }
105 solver.endSearch();
106
107 // Statistics
108 System.out.println();
109 System.out.println("Solutions: " + solver.solutions());
110 System.out.println("Failures: " + solver.failures());
111 System.out.println("Branches: " + solver.branches());
112 System.out.println("Wall time: " + solver.wallTime() + "ms");
113
114 return money_ret;
115 }
