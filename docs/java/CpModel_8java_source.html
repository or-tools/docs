217 public Constraint addLinearExpressionInDomain(LinearArgument expr, Domain domain) {
218 Constraint ct = new Constraint(modelBuilder);
219 LinearConstraintProto.Builder lin = ct.getBuilder().getLinearBuilder();
220 final LinearExpr e = expr.build();
221 for (int i = 0; i < e.numElements(); ++i) {
222 lin.addVars(e.getVariableIndex(i)).addCoeffs(e.getCoefficient(i));
223 }
224 long offset = e.getOffset();
225 for (long b : domain.flattenedIntervals()) {
226 if (b == Long.MIN_VALUE || b == Long.MAX_VALUE) {
227 lin.addDomain(b);
228 } else {
229 lin.addDomain(b - offset);
230 }
231 }
232 return ct;
233 }

1004 LinearExpressionProto.Builder builder = LinearExpressionProto.newBuilder();
1005 final LinearExpr expr = arg.build();

1012 builder.setOffset(expr.getOffset() * mult);
