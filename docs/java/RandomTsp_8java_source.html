35 static class RandomManhattan implements LongBinaryOperator {
36 public RandomManhattan(RoutingIndexManager manager, int size, int seed) {
37 this.xs = new int[size];
38 this.ys = new int[size];
39 this.indexManager = manager;
40 Random generator = new Random(seed);
41 for (int i = 0; i < size; ++i) {
42 xs[i] = generator.nextInt(1000);
43 ys[i] = generator.nextInt(1000);
44 }
45 }
46
47 public long applyAsLong(long firstIndex, long secondIndex) {
48 int firstNode = indexManager.indexToNode(firstIndex);
49 int secondNode = indexManager.indexToNode(secondIndex);
50 return Math.abs(xs[firstNode] - xs[secondNode]) + Math.abs(ys[firstNode] - ys[secondNode]);
51 }
52
53 private int[] xs;
54 private int[] ys;
55 private RoutingIndexManager indexManager;
56 }

58 static class ConstantCallback implements LongUnaryOperator {
59 public long applyAsLong(long index) {
60 return 1;
61 }
62 }

64 static void solve(int size, int forbidden, int seed) {
65 RoutingIndexManager manager = new RoutingIndexManager(size, 1, 0);
66 RoutingModel routing = new RoutingModel(manager);
67
68 // Setting the cost function.
69 // Put a permanent callback to the distance accessor here. The callback
70 // has the following signature: ResultCallback2<int64_t, int64_t, int64_t>.
71 // The two arguments are the from and to node inidices.
72 LongBinaryOperator distances = new RandomManhattan(manager, size, seed);
73 routing.setArcCostEvaluatorOfAllVehicles(routing.registerTransitCallback(distances));
74
75 // Forbid node connections (randomly).
76 Random randomizer = new Random();
77 long forbidden_connections = 0;
78 while (forbidden_connections < forbidden) {
79 long from = randomizer.nextInt(size - 1);
80 long to = randomizer.nextInt(size - 1) + 1;
81 if (routing.nextVar(from).contains(to)) {
82 logger.info("Forbidding connection " + from + " -> " + to);
83 routing.nextVar(from).removeValue(to);
84 ++forbidden_connections;
85 }
86 }
87
88 // Add dummy dimension to test API.
89 routing.addDimension(routing.registerUnaryTransitCallback(new ConstantCallback()), size + 1,
90 size + 1, true, "dummy");
91
92 // Solve, returns a solution if any (owned by RoutingModel).
93 RoutingSearchParameters search_parameters =
94 RoutingSearchParameters.newBuilder()
95 .mergeFrom(main.defaultRoutingSearchParameters())
96 .setFirstSolutionStrategy(FirstSolutionStrategy.Value.PATH_CHEAPEST_ARC)
97 .build();
98
99 Assignment solution = routing.solveWithParameters(search_parameters);
100 if (solution != null) {
101 // Solution cost.
102 logger.info("Objective : " + solution.objectiveValue());
103 // Inspect solution.
104 // Only one route here; otherwise iterate from 0 to routing.vehicles() - 1
105 int route_number = 0;
106 String route = "";
107 for (long node = routing.start(route_number); !routing.isEnd(node);
108 node = solution.value(routing.nextVar(node))) {
109 route += "" + node + " -> ";
110 }
111 logger.info(route + "0");
112 }
113 }
