16 *
17 * Each knapsack perceives a different weight for each item. Item values are
18 * the same across knapsacks. Optimizing constrains the count of each item such
19 * that all knapsack capacities are respected, and their values are maximized.
20 *
21 * This model was created by Hakan Kjellerstrand (hakank@gmail.com)
22 */

25import com.google.ortools.Loader;
26import com.google.ortools.linearsolver.*;

28public class KnapsackMIP {
29 private static void solve(String solverType) {
30 MPSolver solver = MPSolver.createSolver(solverType);
31 if (solver == null) {
32 System.out.println("Could not create solver");
33 return;
34 }
35
37 int itemCount = 12;
38 int capacityCount = 7;
39
40 int[] capacity = {18209, 7692, 1333, 924, 26638, 61188, 13360};
41 int[] value = {96, 76, 56, 11, 86, 10, 66, 86, 83, 12, 9, 81};
42 int[][] weights = {{19, 1, 10, 1, 1, 14, 152, 11, 1, 1, 1, 1},
43 {0, 4, 53, 0, 0, 80, 0, 4, 5, 0, 0, 0}, {4, 660, 3, 0, 30, 0, 3, 0, 4, 90, 0, 0},
44 {7, 0, 18, 6, 770, 330, 7, 0, 0, 6, 0, 0}, {0, 20, 0, 4, 52, 3, 0, 0, 0, 5, 4, 0},
45 {0, 0, 40, 70, 4, 63, 0, 0, 60, 0, 4, 0}, {0, 32, 0, 0, 0, 5, 0, 3, 0, 660, 0, 9}};
46
47 int maxCapacity = -1;
48 for (int c : capacity) {
49 if (c > maxCapacity) {
50 maxCapacity = c;
51 }
52 }
53
54 MPVariable[] taken = solver.makeIntVarArray(itemCount, 0, maxCapacity);
55
57 MPConstraint constraints[] = new MPConstraint[capacityCount];
58 for (int i = 0; i < capacityCount; i++) {
59 constraints[i] = solver.makeConstraint(0, capacity[i]);
60 for (int j = 0; j < itemCount; j++) {
61 constraints[i].setCoefficient(taken[j], weights[i][j]);
62 }
63 }
64
66 MPObjective obj = solver.objective();
67 obj.setMaximization();
68 for (int i = 0; i < itemCount; i++) {
69 obj.setCoefficient(taken[i], value[i]);
70 }
71
72 solver.solve();
73
75 System.out.println("Max cost: " + obj.value());
76 System.out.print("Item quantities: ");
77 for (MPVariable var : taken) {
78 System.out.print((int) var.solutionValue() + " ");
79 }
80 }
81
82 public static void main(String[] args) {
83 Loader.loadNativeLibraries();
84 solve("CBC_MIXED_INTEGER_PROGRAMMING");
85 }
