24public class Map2 {
28 private static void solve() {
29 Solver solver = new Solver("Map2");
30
31 //
32 // data
33 //
34 int Belgium = 0;
35 int Denmark = 1;
36 int France = 2;
37 int Germany = 3;
38 int Netherlands = 4;
39 int Luxembourg = 5;
40
41 int n = 6;
42 int max_num_colors = 4;
43
44 int[][] neighbours = {{France, Belgium}, {France, Luxembourg}, {France, Germany},
45 {Luxembourg, Germany}, {Luxembourg, Belgium}, {Belgium, Netherlands}, {Belgium, Germany},
46 {Germany, Netherlands}, {Germany, Denmark}};
47
48 //
49 // Variables
50 //
51 IntVar[] color = solver.makeIntVarArray(n, 1, max_num_colors, "x");
52
53 //
54 // Constraints
55 //
56 for (int i = 0; i < neighbours.length; i++) {
57 solver.addConstraint(
58 solver.makeNonEquality(color[neighbours[i][0]], color[neighbours[i][1]]));
59 }
60
61 // Symmetry breaking
62 solver.addConstraint(solver.makeEquality(color[Belgium], 1));
63
64 //
65 // Search
66 //
67 DecisionBuilder db =
68 solver.makePhase(color, solver.CHOOSE_FIRST_UNBOUND, solver.ASSIGN_MIN_VALUE);
69 solver.newSearch(db);
70
71 while (solver.nextSolution()) {
72 System.out.print("Colors: ");
73 for (int i = 0; i < n; i++) {
74 System.out.print(color[i].value() + " ");
75 }
76 System.out.println();
77 }
78 solver.endSearch();
79
80 // Statistics
81 System.out.println();
82 System.out.println("Solutions: " + solver.solutions());
83 System.out.println("Failures: " + solver.failures());
84 System.out.println("Branches: " + solver.branches());
85 System.out.println("Wall time: " + solver.wallTime() + "ms");
86 }
