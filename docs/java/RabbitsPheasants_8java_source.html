35 private static void solve(boolean traceSearch) {
36 ConstraintSolverParameters parameters =
37 Solver.defaultSolverParameters().toBuilder().setTraceSearch(traceSearch).build();
38 Solver solver = new Solver("RabbitsPheasants", parameters);
39 IntVar rabbits = solver.makeIntVar(0, 100, "rabbits");
40 IntVar pheasants = solver.makeIntVar(0, 100, "pheasants");
41 solver.addConstraint(solver.makeEquality(solver.makeSum(rabbits, pheasants), 20));
42 solver.addConstraint(solver.makeEquality(
43 solver.makeSum(solver.makeProd(rabbits, 4), solver.makeProd(pheasants, 2)), 56));
44 DecisionBuilder db =
45 solver.makePhase(rabbits, pheasants, Solver.CHOOSE_FIRST_UNBOUND, Solver.ASSIGN_MIN_VALUE);
46 solver.newSearch(db);
47 solver.nextSolution();
48 logger.info(rabbits.toString());
49 logger.info(pheasants.toString());
50 solver.endSearch();
51 }
