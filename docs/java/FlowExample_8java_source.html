24public class FlowExample {
25 private static void solveMinCostFlow() {
26 System.out.println("Min Cost Flow Problem - Simple interface");
27 final int numSources = 4;
28 final int numTargets = 4;
29 final int[][] costs = {
30 {90, 75, 75, 80}, {35, 85, 55, 65}, {125, 95, 90, 105}, {45, 110, 95, 115}};
31 final int expectedCost = 275;
32 MinCostFlow minCostFlow = new MinCostFlow();
33 for (int source = 0; source < numSources; ++source) {
34 for (int target = 0; target < numTargets; ++target) {
35 minCostFlow.addArcWithCapacityAndUnitCost(
36 source, numSources + target, 1, costs[source][target]);
37 }
38 }
39 for (int node = 0; node < numSources; ++node) {
40 minCostFlow.setNodeSupply(node, 1);
41 minCostFlow.setNodeSupply(numSources + node, -1);
42 }
43 if (minCostFlow.solve() == MinCostFlow.Status.OPTIMAL) {
44 final long totalFlowCost = minCostFlow.getOptimalCost();
45 System.out.println("total flow = " + totalFlowCost + "/" + expectedCost);
46 for (int i = 0; i < minCostFlow.getNumArcs(); ++i) {
47 if (minCostFlow.getFlow(i) > 0) {
48 System.out.println("From source " + minCostFlow.getTail(i) + " to target "
49 + minCostFlow.getHead(i) + ": cost " + minCostFlow.getUnitCost(i));
50 }
51 }
52 } else {
53 System.out.println("No solution found");
54 }
55 }

57 private static void solveMaxFlow() {
58 System.out.println("Max Flow Problem - Simple interface");
59 final int[] tails = {0, 0, 0, 0, 1, 2, 3, 3, 4};
60 final int[] heads = {1, 2, 3, 4, 3, 4, 4, 5, 5};
61 final int[] capacities = {5, 8, 5, 3, 4, 5, 6, 6, 4};
62 final int expectedTotalFlow = 10;
63 MaxFlow maxFlow = new MaxFlow();
64 for (int i = 0; i < tails.length; ++i) {
65 maxFlow.addArcWithCapacity(tails[i], heads[i], capacities[i]);
66 }
67 if (maxFlow.solve(0, 5) == MaxFlow.Status.OPTIMAL) {
68 System.out.println("Total flow " + maxFlow.getOptimalFlow() + "/" + expectedTotalFlow);
69 for (int i = 0; i < maxFlow.getNumArcs(); ++i) {
70 System.out.println("From source " + maxFlow.getTail(i) + " to target " + maxFlow.getHead(i)
71 + ": " + maxFlow.getFlow(i) + " / " + maxFlow.getCapacity(i));
72 }
73 // TODO(user): Our SWIG configuration does not currently handle these
74 // functions correctly in Java:
75 // maxFlow.getSourceSideMinCut(...);
76 // maxFlow.getSinkSideMinCut(...);
77 } else {
78 System.out.println("There was an issue with the input.");
79 }
80 }
