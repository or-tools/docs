28public class DietMIP {
29 private static void solve(String solverType) {
30 System.out.println("---- DietMIP with " + solverType);
31
32 MPSolver solver = MPSolver.createSolver(solverType);
33 if (solver == null)
34 return;
35
36 double infinity = MPSolver.infinity();
37
38 int n = 4; // variables number
39 int m = 4; // constraints number
40
41 int[] price = {50, 20, 30, 80};
42
43 int[] limits = {500, 6, 10, 8};
44
45 int[] calories = {400, 200, 150, 500};
46 int[] chocolate = {3, 2, 0, 0};
47 int[] sugar = {2, 2, 4, 4};
48 int[] fat = {2, 4, 1, 5};
49
50 int[][] values = {calories, chocolate, sugar, fat};
51
52 MPVariable[] x = solver.makeIntVarArray(n, 0, 100, "x");
53 MPObjective objective = solver.objective();
54 MPConstraint[] targets = new MPConstraint[4];
55
56 for (int i = 0; i < n; i++) {
57 objective.setCoefficient(x[i], price[i]);
58
59 // constraints
60 targets[i] = solver.makeConstraint(limits[i], infinity);
61 for (int j = 0; j < m; j++) {
62 targets[i].setCoefficient(x[j], values[i][j]);
63 }
64 }
65
66 final MPSolver.ResultStatus resultStatus = solver.solve();
67
69 if (resultStatus != MPSolver.ResultStatus.OPTIMAL) {
70 System.err.println("The problem does not have an optimal solution!");
71 return;
72 } else {
73 System.out.println("Optimal objective value = " + solver.objective().value());
74
75 System.out.print("Item quantities: ");
76 System.out.print((int) x[0].solutionValue() + " ");
77 System.out.print((int) x[1].solutionValue() + " ");
78 System.out.print((int) x[2].solutionValue() + " ");
79 System.out.print((int) x[3].solutionValue() + " ");
80 }
81 }

87 }
