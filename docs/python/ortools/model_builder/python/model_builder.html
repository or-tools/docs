ortools.model_builder.python.model_builder

Methods for building and solving model_builder models.

The following two sections describe the main methods for building and solving those models.

  • ModelBuilder: Methods for creating models, including variables and constraints.
  • ModelSolver: Methods for solving a model and evaluating solutions.

Additional methods for solving ModelBuilder models:

  • Constraint: A few utility methods for modifying constraints created by ModelBuilder.
  • LinearExpr: Methods for creating constraints and the objective from large arrays of coefficients.

Other methods and functions listed are primarily used for developing OR-Tools, rather than for solving specific optimization problems.

 43class LinearExpr(object):
 44    """Holds an linear expression.
 45
 46  A linear expression is built from constants and variables.
 47  For example, `x + 2.0 * (y - z + 1.0)`.
 48
 49  Linear expressions are used in ModelBuilder models in constraints and in the
 50  objective:
 51
 52  * You can define linear constraints as in:
 53
 54  ```
 55  model.add(x + 2 * y <= 5.0)
 56  model.add(sum(array_of_vars) == 5.0)
 57  ```
 58
 59  * In ModelBuilder, the objective is a linear expression:
 60
 61  ```
 62  model.minimize(x + 2.0 * y + z)
 63  ```
 64
 65  * For large arrays, using the LinearExpr class is faster that using the python
 66  `sum()` function. You can create constraints and the objective from lists of
 67  linear expressions or coefficients as follows:
 68
 69  ```
 70  model.minimize(model_builder.LinearExpr.sum(expressions))
 71  model.add(model_builder.LinearExpr.weighted_sum(expressions, coeffs) >= 0)
 72  ```
 73  """
 74
 75    @classmethod
 76    def sum(cls, expressions):
 77        """Creates the expression sum(expressions)."""
 78        if len(expressions) == 1:
 79            return expressions[0]
 80        return _SumArray(expressions)
 81
 82    @classmethod
 83    def weighted_sum(cls, expressions, coefficients):
 84        """Creates the expression sum(expressions[i] * coefficients[i])."""
 85        if LinearExpr.is_empty_or_null(coefficients):
 86            return 0
 87        elif len(expressions) == 1:
 88            return expressions[0] * coefficients[0]
 89        else:
 90            return _WeightedSum(expressions, coefficients)
 91
 92    @classmethod
 93    def term(cls, expression, coefficient):
 94        """Creates `expression * coefficient`."""
 95        if mbh.is_zero(coefficient):
 96            return 0
 97        else:
 98            return expression * coefficient
 99
100    @classmethod
101    def is_empty_or_null(cls, coefficients):
102        for c in coefficients:
103            if not mbh.is_zero(c):
104                return False
105        return True
106
107    def get_var_value_map(self):
108        """Scans the expression. Returns (var_coef_map, constant)."""
109        coeffs = {}
110        constant = 0.0
111        to_process = [(self, 1.0)]
112        while to_process:  # Flatten to avoid recursion.
113            expr, coeff = to_process.pop()
114            if mbh.is_a_number(expr):
115                constant += coeff * mbh.assert_is_a_number(expr)
116            elif isinstance(expr, _ProductCst):
117                to_process.append((expr.expression, coeff * expr.coefficient))
118            elif isinstance(expr, _Sum):
119                to_process.append((expr.left, coeff))
120                to_process.append((expr.right, coeff))
121            elif isinstance(expr, _SumArray):
122                for e in expr.expressions:
123                    to_process.append((e, coeff))
124                constant += expr.constant * coeff
125            elif isinstance(expr, _WeightedSum):
126                for e, c in zip(expr.expressions, expr.coefficients):
127                    to_process.append((e, coeff * c))
128                constant += expr.constant * coeff
129            elif isinstance(expr, Variable):
130                if expr in coeffs:
131                    coeffs[expr] += coeff
132                else:
133                    coeffs[expr] = coeff
134            else:
135                raise TypeError('Unrecognized linear expression: ' + str(expr))
136        return coeffs, constant
137

139        return object.__hash__(self)

298class _SumArray(LinearExpr):
299    """Represents the sum of a list of LinearExpr and a constant."""
300
301    def __init__(self, expressions, constant=0):
302        self.__expressions = []
303        self.__constant = constant
304        for x in expressions:
305            if mbh.is_a_number(x):
306                if mbh.is_zero(x):
307                    continue
308                x = mbh.assert_is_a_number(x)
309                self.__constant += x
310            elif isinstance(x, LinearExpr):
311                self.__expressions.append(x)
312            else:
313                raise TypeError('Not an linear expression: ' + str(x))
314
315    def __str__(self):
316        if self.__constant == 0:
317            return '({})'.format(' + '.join(map(str, self.__expressions)))
318        else:
319            return '({} + {})'.format(' + '.join(map(str, self.__expressions)),
320                                      self.__constant)
321
322    def __repr__(self):
323        return 'SumArray({}, {})'.format(
324            ', '.join(map(repr, self.__expressions)), self.__constant)
325
326    @property
327    def expressions(self):
328        return self.__expressions
329
330    @property
331    def constant(self):
332        return self.__constant

335class _WeightedSum(LinearExpr):
336    """Represents sum(ai * xi) + b."""
337
338    def __init__(self, expressions, coefficients, constant=0.0):
339        self.__expressions = []
340        self.__coefficients = []
341        self.__constant = constant
342        if len(expressions) != len(coefficients):
343            raise TypeError(
344                'In the LinearExpr.weighted_sum method, the expression array and the '
345                ' coefficient array must have the same length.')
346        for e, c in zip(expressions, coefficients):
347            c = mbh.assert_is_a_number(c)
348            if mbh.is_zero(c):
349                continue
350            if mbh.is_a_number(e):
351                e = mbh.assert_is_a_number(e)
352                self.__constant += e * c
353            elif isinstance(e, LinearExpr):
354                self.__expressions.append(e)
355                self.__coefficients.append(c)
356            else:
357                raise TypeError('Not an linear expression: ' + str(e))
358
359    def __str__(self):
360        output = None
361        for expr, coeff in zip(self.__expressions, self.__coefficients):
362            if not output and mbh.is_one(coeff):
363                output = str(expr)
364            elif not output and mbh.is_minus_one(coeff):
365                output = '-' + str(expr)
366            elif not output:
367                output = '{} * {}'.format(coeff, str(expr))
368            elif mbh.is_one(coeff):
369                output += ' + {}'.format(str(expr))
370            elif mbh.is_minus_one(coeff):
371                output += ' - {}'.format(str(expr))
372            elif coeff > 1:
373                output += ' + {} * {}'.format(coeff, str(expr))
374            elif coeff < -1:
375                output += ' - {} * {}'.format(-coeff, str(expr))
376        if self.__constant > 0:
377            output += ' + {}'.format(self.__constant)
378        elif self.__constant < 0:
379            output += ' - {}'.format(-self.__constant)
380        if output is None:
381            output = '0'
382        return output
383
384    def __repr__(self):
385        return 'WeightedSum([{}], [{}], {})'.format(
386            ', '.join(map(repr, self.__expressions)),
387            ', '.join(map(repr, self.__coefficients)), self.__constant)
388
389    @property
390    def expressions(self):
391        return self.__expressions
392
393    @property
394    def coefficients(self):
395        return self.__coefficients
396
397    @property
398    def constant(self):
399        return self.__constant

402class Variable(LinearExpr):
403    """An variable (continuous or integral).
404
405  An Variable is an object that can take on any integer value within defined
406  ranges. Variables appear in constraint like:
407
408      x + y >= 5
409
410  Solving a model is equivalent to finding, for each variable, a single value
411  from the set of initial values (called the initial domain), such that the
412  model is feasible, or optimal if you provided an objective function.
413  """
414
415    def __init__(self, helper, lb, ub, is_integral, name):
416        """See ModelBuilder.new_var below."""
417        self.__helper = helper
418        # Python do not support multiple __init__ methods.
419        # This method is only called from the ModelBuilder class.
420        # We hack the parameter to support the two cases:
421        # case 1:
422        #     helper is a ModelBuilderHelper, lb is a double value, ub is a double
423        #     value, is_integral is a Boolean value, and name is a string.
424        # case 2:
425        #     helper is a ModelBuilderHelper, lb is an index (int), ub is None,
426        #     is_integral is None, and name is None.
427        if mbh.is_integral(lb) and ub is None and is_integral is None:
428            self.__index = int(lb)
429            self.__helper = helper
430        else:
431            index = helper.add_var()
432            self.__index = index
433            self.__helper = helper
434            helper.set_var_lower_bound(index, lb)
435            helper.set_var_upper_bound(index, ub)
436            helper.set_var_integrality(index, is_integral)
437            if name:
438                helper.set_var_name(index, name)
439
440    @property
441    def index(self):
442        """Returns the index of the variable in the helper."""
443        return self.__index
444
445    @property
446    def helper(self):
447        """Returns the underlying ModelBuilderHelper."""
448        return self.__helper
449
450    def is_equal_to(self, other):
451        """Returns true if self == other in the python sense."""
452        if not isinstance(other, Variable):
453            return False
454        return self.index == other.index
455
456    def __str__(self):
457        name = self.__helper.var_name(self.__index)
458        if not name:
459            if self.__helper.VarIsInteger(self.__index):
460                return 'unnamed_int_var_%i' % self.__index
461            else:
462                return 'unnamed_num_var_%i' % self.__index
463        return name
464
465    def __repr__(self):
466        index = self.__index
467        name = self.__helper.VarName(index)
468        lb = self.__helper.VarLowerBound(index)
469        ub = self.__helper.VarUpperBound(index)
470        is_integer = self.__helper.VarIsInteger(index)
471        if name:
472            if is_integer:
473                return f'{name}(index={index}, lb={lb}, ub={ub}, integer)'
474            else:
475                return f'{name}(index={index}, lb={lb}, ub={ub})'
476        else:
477            if is_integer:
478                return f'unnamed_var(index={index}, lb={lb}, ub={ub}, integer)'
479            else:
480                return f'unnamed_var(index={index}, lb={lb}, ub={ub})'
481
482    @property
483    def name(self):
484        return self.__helper.var_name(self.__index)
485
486    @name.setter
487    def name(self, name):
488        """Sets the name of the variable."""
489        self.__helper.set_var_name(self.__index, name)
490
491    @property
492    def lower_bound(self):
493        """Returns the lower bound of the variable."""
494        return self.__helper.var_lower_bound(self.__index)
495
496    @lower_bound.setter
497    def lower_bound(self, bound):
498        """Sets the lower bound of the variable."""
499        self.__helper.set_var_lower_bound(self.__index, bound)
500
501    @property
502    def upper_bound(self):
503        """Returns the upper bound of the variable."""
504        return self.__helper.var_upper_bound(self.__index)
505
506    @upper_bound.setter
507    def upper_bound(self, bound):
508        """Sets the upper bound of the variable."""
509        self.__helper.set_var_upper_bound(self.__index, bound)
510
511    @property
512    def is_integral(self):
513        """Returns whether the variable is integral."""
514        return self.__helper.var_is_integral(self.__index)
515
516    @is_integral.setter
517    def integrality(self, is_integral):
518        """Sets the integrality of the variable."""
519        self.__helper.set_var_integrality(self.__index, is_integral)
520
521    @property
522    def objective_coefficient(self):
523        return self.__helper.var_objective_coefficient(self.__index)
524
525    @objective_coefficient.setter
526    def objective_coefficient(self, coeff):

529    def __eq__(self, arg):
530        if arg is None:
531            return False
532        if isinstance(arg, Variable):
533            return VarCompVar(self, arg, True)
534        else:
535            if mbh.is_a_number(arg):

684    """Methods for building a linear model.
685
686  Methods beginning with:
