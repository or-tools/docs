32def solve(
33 opt_model: model.Model,
34 solver_type: parameters.SolverType,
35 *,
36 params: Optional[parameters.SolveParameters] = None,
37 model_params: Optional[model_parameters.ModelSolveParameters] = None,
38 msg_cb: Optional[message_callback.SolveMessageCallback] = None,
39 callback_reg: Optional[callback.CallbackRegistration] = None,
40 cb: Optional[SolveCallback] = None,
41) -> result.SolveResult:
42 """Solves an optimization model.
43
44 Thread-safety: this function must not be called while modifying the Model
45 (adding variables...). Some solvers may add more restriction regarding
46 threading. Please see SolverType::XXX documentation for details.
47
48 Args:
49 opt_model: The optimization model.
50 solver_type: The underlying solver to use.
51 params: Configuration of the underlying solver.
52 model_params: Configuration of the solver that is model specific.
53 msg_cb: A callback that gives back the underlying solver's logs by the line.
54 callback_reg: Configures when the callback will be invoked (if provided) and
55 what data will be collected to access in the callback.
56 cb: A callback that will be called periodically as the solver runs.
57
58 Returns:
59 A SolveResult containing the termination reason, solution(s) and stats.
60
61 Raises:
62 RuntimeError: On a solve error.
63 """
64 # First, initialize optional arguments that were not set to default values.
65 # Note that in python, default arguments must be immutable, and these are not.
66 params = params or parameters.SolveParameters()
67 model_params = model_params or model_parameters.ModelSolveParameters()
68 callback_reg = callback_reg or callback.CallbackRegistration()
69 model_proto = opt_model.export_model()
70 proto_cb = None
71 if cb is not None:
72 proto_cb = lambda x: cb( # pylint: disable=g-long-lambda
73 callback.parse_callback_data(x, opt_model)
74 ).to_proto()
75 # Solve
76 try:
77 proto_result = solver.solve(
78 model_proto,
79 solver_type.value,
80 parameters_pb2.SolverInitializerProto(),
81 params.to_proto(),
82 model_params.to_proto(),
83 msg_cb,
84 callback_reg.to_proto(),
85 proto_cb,
86 None,
87 )
88 except StatusNotOk as e:
89 raise RuntimeError(str(e)) from None
90 return result.parse_solve_result(proto_result, opt_model)

137class IncrementalSolver:
138 """Solve an optimization multiple times, with modifications between solves.
139
140 Prefer calling simply solve() above in most cases when incrementalism is not
141 needed.
142
143 Thread-safety: The __init__(), solve() methods must not be called while
144 modifying the Model (adding variables...). The user is expected to use proper
145 synchronization primitives to serialize changes to the model and the use of
146 this object. Note though that it is safe to call methods from different
147 IncrementalSolver instances on the same Model concurrently. The solve() method
148 must not be called concurrently on different threads for the same
149 IncrementalSolver. Some solvers may add more restriction regarding
150 threading. Please see to SolverType::XXX documentation for details.
151
152 This class references some resources that are freed when it is garbage
153 collected (which should usually happen when the last reference is lost). In
154 particular, it references some C++ objects. Although it is not mandatory, it
155 is recommended to free those as soon as possible. To do so it is possible to
156 use this class in the `with` statement:
157
158 with IncrementalSolver(model, SolverType.GLOP) as solver:
159 ...
160
161 When it is not possible to use `with`, the close() method can be called.
162 """
163
164 def __init__(self, opt_model: model.Model, solver_type: parameters.SolverType):
165 self._model = opt_model
166 self._solver_type = solver_type
167 self._update_tracker = self._model.add_update_tracker()
168 try:
169 self._proto_solver = solver.new(
170 solver_type.value,
171 self._model.export_model(),
172 parameters_pb2.SolverInitializerProto(),
173 )
174 except StatusNotOk as e:
175 raise RuntimeError(str(e)) from None
176 self._closed = False

178 def solve(
179 self,
180 *,
181 params: Optional[parameters.SolveParameters] = None,
182 model_params: Optional[model_parameters.ModelSolveParameters] = None,
183 msg_cb: Optional[message_callback.SolveMessageCallback] = None,
184 callback_reg: Optional[callback.CallbackRegistration] = None,
185 cb: Optional[SolveCallback] = None,
186 ) -> result.SolveResult:
187 """Solves the current optimization model.
188
189 Args:
190 params: The non-model specific solve parameters.
191 model_params: The model specific solve parameters.
192 msg_cb: An optional callback for solver messages.
193 callback_reg: The parameters controlling when cb is called.
194 cb: An optional callback for LP/MIP events.
195
196 Returns:
197 The result of the solve.
198
199 Raises:
200 RuntimeError: If called after being closed, or on a solve error.
201 """
202 if self._closed:
203 raise RuntimeError("the solver is closed")
204
205 update = self._update_tracker.export_update()
206 if update is not None:
207 try:
208 if not self._proto_solver.update(update):
209 self._proto_solver = solver.new(
210 self._solver_type.value,
211 self._model.export_model(),
212 parameters_pb2.SolverInitializerProto(),
213 )
214 except StatusNotOk as e:
215 raise RuntimeError(str(e)) from None
216 self._update_tracker.advance_checkpoint()
217 params = params or parameters.SolveParameters()
218 model_params = model_params or model_parameters.ModelSolveParameters()
219 callback_reg = callback_reg or callback.CallbackRegistration()
220 proto_cb = None
221 if cb is not None:
222 proto_cb = lambda x: cb( # pylint: disable=g-long-lambda
223 callback.parse_callback_data(x, self._model)
224 ).to_proto()
225 try:
226 result_proto = self._proto_solver.solve(
227 params.to_proto(),
228 model_params.to_proto(),
229 msg_cb,
230 callback_reg.to_proto(),
231 proto_cb,
232 None,
233 )
234 except StatusNotOk as e:
235 raise RuntimeError(str(e)) from None
236 return result.parse_solve_result(result_proto, self._model)

238 def close(self) -> None:
239 """Closes this solver, freeing all its resources.
240
241 This is optional, the code is correct without calling this function. See the
242 class documentation for details.
243
244 After a solver has been closed, it can't be used anymore. Prefer using the
245 context manager API when possible instead of calling close() directly:
246
247 with IncrementalSolver(model, SolverType.GLOP) as solver:
248 ...
249 """
250 if self._closed:
251 return
252 self._closed = True
253
254 del self._model
255 del self._solver_type
256 del self._update_tracker
257 del self._proto_solver
