66class LinearExpr(metaclass=abc.ABCMeta):
67 """Holds an linear expression.
68
69 A linear expression is built from constants and variables.
70 For example, `x + 2.0 * (y - z + 1.0)`.
71
72 Linear expressions are used in ModelBuilder models in constraints and in the
73 objective:
74
75 * You can define linear constraints as in:
76
77 ```
78 model.add(x + 2 * y <= 5.0)
79 model.add(sum(array_of_vars) == 5.0)
80 ```
81
82 * In ModelBuilder, the objective is a linear expression:
83
84 ```
85 model.minimize(x + 2.0 * y + z)
86 ```
87
88 * For large arrays, using the LinearExpr class is faster that using the python
89 `sum()` function. You can create constraints and the objective from lists of
90 linear expressions or coefficients as follows:
91
92 ```
93 model.minimize(model_builder.LinearExpr.sum(expressions))
94 model.add(model_builder.LinearExpr.weighted_sum(expressions, coeffs) >= 0)
95 ```
96 """
97
98 @classmethod
99 def sum( # pytype: disable=annotation-type-mismatch # numpy-scalars
100 cls, expressions: Sequence[LinearExprT], *, constant: NumberT = 0.0
101 ) -> LinearExprT:
102 """Creates `sum(expressions) + constant`.
103
104 It can perform simple simplifications and returns different objects,
105 including the input.
106
107 Args:
108 expressions: a sequence of linear expressions or constants.
109 constant: a numerical constant.
110
111 Returns:
112 a LinearExpr instance or a numerical constant.
113 """
114 checked_constant: np.double = mbn.assert_is_a_number(constant)
115 if not expressions:
116 return checked_constant
117 if len(expressions) == 1 and mbn.is_zero(checked_constant):
118 return expressions[0]
119
120 return LinearExpr.weighted_sum(
121 expressions, np.ones(len(expressions)), constant=checked_constant
122 )
123
124 @classmethod
125 def weighted_sum( # pytype: disable=annotation-type-mismatch # numpy-scalars
126 cls,
127 expressions: Sequence[LinearExprT],
128 coefficients: Sequence[NumberT],
129 *,
130 constant: NumberT = 0.0,
131 ) -> Union[NumberT, "_LinearExpression"]:
132 """Creates `sum(expressions[i] * coefficients[i]) + constant`.
133
134 It can perform simple simplifications and returns different object,
135 including the input.
136
137 Args:
138 expressions: a sequence of linear expressions or constants.
139 coefficients: a sequence of numerical constants.
140 constant: a numerical constant.
141
142 Returns:
143 a _LinearExpression instance or a numerical constant.
144 """
145 if len(expressions) != len(coefficients):
146 raise ValueError(
147 "LinearExpr.weighted_sum: expressions and coefficients have"
148 " different lengths"
149 )
150 checked_constant: np.double = mbn.assert_is_a_number(constant)
151 if not expressions:
152 return checked_constant
153 return _sum_as_flat_linear_expression(
154 to_process=list(zip(expressions, coefficients)), offset=checked_constant
155 )

157 @classmethod
158 def term( # pytype: disable=annotation-type-mismatch # numpy-scalars
159 cls,
160 expression: LinearExprT,
161 coefficient: NumberT,
162 *,
163 constant: NumberT = 0.0,
164 ) -> LinearExprT:
165 """Creates `expression * coefficient + constant`.
166
167 It can perform simple simplifications and returns different object,
168 including the input.
169 Args:
170 expression: a linear expression or a constant.
171 coefficient: a numerical constant.
172 constant: a numerical constant.
173
174 Returns:
175 a LinearExpr instance or a numerical constant.
176 """
177 checked_coefficient: np.double = mbn.assert_is_a_number(coefficient)
178 checked_constant: np.double = mbn.assert_is_a_number(constant)
179
180 if mbn.is_zero(checked_coefficient):
181 return checked_constant
182 if mbn.is_one(checked_coefficient) and mbn.is_zero(checked_constant):
183 return expression
184 if mbn.is_a_number(expression):
185 return np.double(expression) * checked_coefficient + checked_constant
186 if isinstance(expression, LinearExpr):
187 return _as_flat_linear_expression(
188 expression * checked_coefficient + checked_constant
189 )
190 raise TypeError(f"Unknown expression {expression!r} of type {type(expression)}")

393def _add_linear_constraint_to_helper(
394 bounded_expr: Union[bool, _BoundedLinearExpr],
395 helper: mbh.ModelBuilderHelper,
396 name: Optional[str],
397):
398 """Creates a new linear constraint in the helper.
399
400 It handles boolean values (which might arise in the construction of
401 BoundedLinearExpressions).
402
403 Args:
404 bounded_expr: The bounded expression used to create the constraint.
405 helper: The helper to create the constraint.
406 name: The name of the constraint to be created.
407
408 Returns:
409 LinearConstraint: a constraint in the helper corresponding to the input.
410
411 Raises:
412 TypeError: If constraint is an invalid type.
413 """
414 if isinstance(bounded_expr, bool):
415 c = LinearConstraint(helper)
416 if name is not None:
417 helper.set_constraint_name(c.index, name)
418 if bounded_expr:
419 # constraint that is always feasible: 0.0 <= nothing <= 0.0
420 helper.set_constraint_lower_bound(c.index, 0.0)
421 helper.set_constraint_upper_bound(c.index, 0.0)
422 else:
423 # constraint that is always infeasible: +oo <= nothing <= -oo
424 helper.set_constraint_lower_bound(c.index, 1)
425 helper.set_constraint_upper_bound(c.index, -1)
426 return c
427 if isinstance(bounded_expr, _BoundedLinearExpr):
428 # pylint: disable=protected-access
429 return bounded_expr._add_linear_constraint(helper, name)
430 raise TypeError("invalid type={}".format(type(bounded_expr)))

465class BoundedLinearExpression(_BoundedLinearExpr):
466 """Represents a linear constraint: `lb <= linear expression <= ub`.
467
468 The only use of this class is to be added to the ModelBuilder through
469 `ModelBuilder.add(bounded expression)`, as in:
470
471 model.Add(x + 2 * y -1 >= z)
472 """
473
474 def __init__(self, expr: LinearExprT, lb: NumberT, ub: NumberT):
475 self.__expr: LinearExprT = expr
476 self.__lb: np.double = mbn.assert_is_a_number(lb)
477 self.__ub: np.double = mbn.assert_is_a_number(ub)
478
479 def __str__(self) -> str:
480 if self.__lb > -math.inf and self.__ub < math.inf:
481 if self.__lb == self.__ub:
482 return str(self.__expr) + " == " + str(self.__lb)
483 else:
484 return (
485 str(self.__lb) + " <= " + str(self.__expr) + " <= " + str(self.__ub)
486 )
487 elif self.__lb > -math.inf:
488 return str(self.__expr) + " >= " + str(self.__lb)
489 elif self.__ub < math.inf:
490 return str(self.__expr) + " <= " + str(self.__ub)
491 else:
492 return str(self.__expr) + " free"
493
494 def __repr__(self):
495 return self.__str__()
496
497 @property
498 def expression(self) -> LinearExprT:
499 return self.__expr
500
501 @property
502 def lower_bound(self) -> np.double:
503 return self.__lb
504
505 @property
506 def upper_bound(self) -> np.double:
507 return self.__ub
508
509 def __bool__(self) -> bool:
510 raise NotImplementedError(
511 f"Cannot use a BoundedLinearExpression {self} as a Boolean value"
512 )

514 def _add_linear_constraint(
515 self, helper: mbh.ModelBuilderHelper, name: Optional[str]
516 ) -> "LinearConstraint":

518 flat_expr = _as_flat_linear_expression(self.__expr)

525 # pylint: enable=protected-access
526 if name is not None:
527 helper.set_constraint_name(c.index, name)
528 return c

1447def _sum_as_flat_linear_expression(
1448 to_process: List[Tuple[LinearExprT, float]], offset: float = 0.0
1449) -> _LinearExpression:
1450 """Creates a _LinearExpression as the sum of terms."""
1451 indices = []
1452 coeffs = []
1453 helper = None
1454 while to_process: # Flatten AST of LinearTypes.
1455 expr, coeff = to_process.pop()
1456 if isinstance(expr, _Sum):
1457 to_process.append((expr._left, coeff))
1458 to_process.append((expr._right, coeff))
1459 elif isinstance(expr, Variable):
1460 indices.append([expr.index])
1461 coeffs.append([coeff])
1462 if helper is None:
1463 helper = expr.helper
1464 elif mbn.is_a_number(expr):
1465 offset += coeff * cast(NumberT, expr)
1466 elif isinstance(expr, _Product):
1467 to_process.append((expr._expression, coeff * expr._coefficient))
1468 elif isinstance(expr, _LinearExpression):
1469 offset += coeff * expr._offset
1470 if expr._helper is not None:
1471 indices.append(expr.variable_indices)
1472 coeffs.append(np.multiply(expr.coefficients, coeff))
1473 if helper is None:
1474 helper = expr._helper
1475 else:
1476 raise TypeError(
1477 "Unrecognized linear expression: " + str(expr) + f" {type(expr)}"
1478 )
1479
1480 if helper is not None:
1481 all_indices: npt.NDArray[np.int32] = np.concatenate(indices, axis=0)
1482 all_coeffs: npt.NDArray[np.double] = np.concatenate(coeffs, axis=0)
1483 sorted_indices, sorted_coefficients = helper.sort_and_regroup_terms(
1484 all_indices, all_coeffs
1485 )
1486 return _LinearExpression(sorted_indices, sorted_coefficients, offset, helper)
1487 else:
1488 assert not indices
1489 assert not coeffs
1490 return _LinearExpression(
1491 _variable_indices=np.zeros(dtype=np.int32, shape=[0]),
1492 _coefficients=np.zeros(dtype=np.double, shape=[0]),
1493 _offset=offset,
1494 _helper=None,
