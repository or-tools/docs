63def assert_is_int64(x: Any) -> int:
64 """Asserts that x is integer and x is in [min_int_64, max_int_64] and returns it casted to an int."""

106def capped_subtraction(x: int, y: int) -> int:
107 """Saturated arithmetics. Returns x - y truncated to the int64_t range."""
108 assert_is_int64(x)
109 assert_is_int64(y)
110 if y == 0:
111 return x
112 if x == y:
113 if x == INT_MAX or x == INT_MIN:
114 raise OverflowError("Integer NaN: subtracting INT_MAX or INT_MIN to itself")
115 return 0
116 if x == INT_MAX or x == INT_MIN:
117 return x
118 if y == INT_MAX:
119 return INT_MIN
120 if y == INT_MIN:
121 return INT_MAX
122 return to_capped_int64(x - y)
