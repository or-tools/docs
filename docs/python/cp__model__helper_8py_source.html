54def assert_is_int64(x):
55 """Asserts that x is integer and x is in [min_int_64, max_int_64]."""
56 if not is_integral(x):
57 raise TypeError("Not an integer: %s" % x)
58 if x < INT_MIN or x > INT_MAX:
59 raise OverflowError("Does not fit in an int64_t: %s" % x)
60 return int(x)

88def to_capped_int64(v):
89 """Restrict v within [INT_MIN..INT_MAX] range."""
90 if v > INT_MAX:
91 return INT_MAX
92 if v < INT_MIN:
93 return INT_MIN
94 return v

97def capped_subtraction(x, y):
98 """Saturated arithmetics. Returns x - y truncated to the int64_t range."""
99 assert_is_int64(x)
100 assert_is_int64(y)
101 if y == 0:
102 return x
103 if x == y:
104 if x == INT_MAX or x == INT_MIN:
105 raise OverflowError("Integer NaN: subtracting INT_MAX or INT_MIN to itself")
106 return 0
107 if x == INT_MAX or x == INT_MIN:
108 return x
109 if y == INT_MAX:
110 return INT_MIN
111 if y == INT_MIN:
112 return INT_MAX
113 return to_capped_int64(x - y)
