14"""Methods for building and solving CP-SAT models.
15
16The following two sections describe the main
17methods for building and solving CP-SAT models.
18
19* [`CpModel`](#cp_model.CpModel): Methods for creating
20models, including variables and constraints.
21* [`CPSolver`](#cp_model.CpSolver): Methods for solving
22a model and evaluating solutions.
23
24The following methods implement callbacks that the
25solver calls each time it finds a new solution.
26

38

40 constraints created by `CpModel`.
41* [`LinearExpr`](#lineacp_model.LinearExpr): Methods for creating constraints

43

219class LinearExpr:
220 """Holds an integer linear expression.
221
222 A linear expression is built from integer constants and variables.
223 For example, `x + 2 * (y - z + 1)`.
224
225 Linear expressions are used in CP-SAT models in constraints and in the
226 objective:
227
228 * You can define linear constraints as in:
229
230 ```
231 model.add(x + 2 * y <= 5)
232 model.add(sum(array_of_vars) == 5)
233 ```
234
235 * In CP-SAT, the objective is a linear expression:
236
237 ```
238 model.minimize(x + 2 * y + z)
239 ```
240
241 * For large arrays, using the LinearExpr class is faster that using the python
242 `sum()` function. You can create constraints and the objective from lists of
243 linear expressions or coefficients as follows:
244
245 ```
246 model.minimize(cp_model.LinearExpr.sum(expressions))
247 model.add(cp_model.LinearExpr.weighted_sum(expressions, coefficients) >= 0)
248 ```
249 """

344 def get_integer_var_value_map(self) -> Tuple[Dict["IntVar", int], int]:
345 """Scans the expression, and returns (var_coef_map, constant)."""
346 coeffs: Dict["IntVar", int] = collections.defaultdict(int)
347 constant = 0
348 to_process: List[Tuple[LinearExprT, int]] = [(self, 1)]
349 while to_process: # Flatten to avoid recursion.
350 expr: LinearExprT
351 coeff: int
352 expr, coeff = to_process.pop()
353 if isinstance(expr, IntegralTypes):
354 constant += coeff * int(expr)
355 elif isinstance(expr, _ProductCst):
356 to_process.append((expr.expression(), coeff * expr.coefficient()))
357 elif isinstance(expr, _Sum):
358 to_process.append((expr.left(), coeff))
359 to_process.append((expr.right(), coeff))
360 elif isinstance(expr, _SumArray):
361 for e in expr.expressions():
362 to_process.append((e, coeff))
363 constant += expr.constant() * coeff
364 elif isinstance(expr, _WeightedSum):
365 for e, c in zip(expr.expressions(), expr.coefficients()):
366 to_process.append((e, coeff * c))
367 constant += expr.constant() * coeff
368 elif isinstance(expr, IntVar):
369 coeffs[expr] += coeff
370 elif isinstance(expr, _NotBooleanVariable):
371 constant += coeff
372 coeffs[expr.negated()] -= coeff
373 elif isinstance(expr, NumberTypes):
374 raise TypeError(
375 f"Floating point constants are not supported in constraints: {expr}"
376 )
377 else:
378 raise TypeError("Unrecognized linear expression: " + str(expr))
379
380 return coeffs, constant

382 def get_float_var_value_map(
383 self,
384 ) -> Tuple[Dict["IntVar", float], float, bool]:
385 """Scans the expression. Returns (var_coef_map, constant, is_integer)."""
386 coeffs: Dict["IntVar", Union[int, float]] = {}
387 constant: Union[int, float] = 0
388 to_process: List[Tuple[LinearExprT, Union[int, float]]] = [(self, 1)]
389 while to_process: # Flatten to avoid recursion.
390 expr, coeff = to_process.pop()
391 if isinstance(expr, IntegralTypes): # Keep integrality.
392 constant += coeff * int(expr)
393 elif isinstance(expr, NumberTypes):
394 constant += coeff * float(expr)
395 elif isinstance(expr, _ProductCst):
396 to_process.append((expr.expression(), coeff * expr.coefficient()))
397 elif isinstance(expr, _Sum):
398 to_process.append((expr.left(), coeff))
399 to_process.append((expr.right(), coeff))
400 elif isinstance(expr, _SumArray):
401 for e in expr.expressions():
402 to_process.append((e, coeff))
403 constant += expr.constant() * coeff
404 elif isinstance(expr, _WeightedSum):
405 for e, c in zip(expr.expressions(), expr.coefficients()):
406 to_process.append((e, coeff * c))
407 constant += expr.constant() * coeff
408 elif isinstance(expr, IntVar):
409 if expr in coeffs:
410 coeffs[expr] += coeff
411 else:
412 coeffs[expr] = coeff
413 elif isinstance(expr, _NotBooleanVariable):
414 constant += coeff
415 if expr.negated() in coeffs:
416 coeffs[expr.negated()] -= coeff
417 else:
418 coeffs[expr.negated()] = -coeff
419 else:
420 raise TypeError("Unrecognized linear expression: " + str(expr))
421 is_integer = isinstance(constant, IntegralTypes)
422 if is_integer:
423 for coeff in coeffs.values():
424 if not isinstance(coeff, IntegralTypes):
425 is_integer = False
426 break
427 return coeffs, constant, is_integer

1767 def add_automaton(
1768 self,
1769 transition_variables: Sequence[VariableT],
1770 starting_state: IntegralT,
1771 final_states: Sequence[IntegralT],
1772 transition_triples: Sequence[Tuple[IntegralT, IntegralT, IntegralT]],
1773 ) -> Constraint:
1774 """Adds an automaton constraint.
1775
1776 An automaton constraint takes a list of variables (of size *n*), an initial
1777 state, a set of final states, and a set of transitions. A transition is a
1778 triplet (*tail*, *transition*, *head*), where *tail* and *head* are states,
1779 and *transition* is the label of an arc from *head* to *tail*,
1780 corresponding to the value of one variable in the list of variables.
1781
1782 This automaton will be unrolled into a flow with *n* + 1 phases. Each phase
1783 contains the possible states of the automaton. The first state contains the
1784 initial state. The last phase contains the final states.
1785
1786 Between two consecutive phases *i* and *i* + 1, the automaton creates a set
1787 of arcs. For each transition (*tail*, *transition*, *head*), it will add
1788 an arc from the state *tail* of phase *i* and the state *head* of phase
1789 *i* + 1. This arc is labeled by the value *transition* of the variables
1790 `variables[i]`. That is, this arc can only be selected if `variables[i]`
1791 is assigned the value *transition*.
1792
1793 A feasible solution of this constraint is an assignment of variables such
1794 that, starting from the initial state in phase 0, there is a path labeled by
1795 the values of the variables that ends in one of the final states in the
1796 final phase.
1797
1798 Args:
1799 transition_variables: A non-empty list of variables whose values
1800 correspond to the labels of the arcs traversed by the automaton.
1801 starting_state: The initial state of the automaton.
1802 final_states: A non-empty list of admissible final states.
1803 transition_triples: A list of transitions for the automaton, in the
1804 following format (current_state, variable_value, next_state).
1805
1806 Returns:
1807 An instance of the `Constraint` class.
1808
1809 Raises:
1810 ValueError: if `transition_variables`, `final_states`, or
1811 `transition_triples` are empty.
1812 """
1813
1814 if not transition_variables:
1815 raise ValueError(
1816 "add_automaton expects a non-empty transition_variables array"
1817 )
1818 if not final_states:
1819 raise ValueError("add_automaton expects some final states")
1820
1821 if not transition_triples:
1822 raise ValueError("add_automaton expects some transition triples")
1823
1824 ct = Constraint(self)
1825 model_ct = self.__model.constraints[ct.index]
1826 model_ct.automaton.vars.extend(
1827 [self.get_or_make_index(x) for x in transition_variables]
1828 )
1829 starting_state = cmh.assert_is_int64(starting_state)
1830 model_ct.automaton.starting_state = starting_state
1831 for v in final_states:
1832 v = cmh.assert_is_int64(v)
1833 model_ct.automaton.final_states.append(v)
1834 for t in transition_triples:
1835 if len(t) != 3:
1836 raise TypeError("Tuple " + str(t) + " has the wrong arity (!= 3)")
1837 tail = cmh.assert_is_int64(t[0])
1838 label = cmh.assert_is_int64(t[1])
1839 head = cmh.assert_is_int64(t[2])
1840 model_ct.automaton.transition_tail.append(tail)
1841 model_ct.automaton.transition_label.append(label)
1842 model_ct.automaton.transition_head.append(head)
1843 return ct

3064def evaluate_linear_expr(
3065 expression: LinearExprT, solution: cp_model_pb2.CpSolverResponse
3066) -> int:
3067 """Evaluate a linear expression against a solution."""
3068 if isinstance(expression, IntegralTypes):
3069 return int(expression)
3070 if not isinstance(expression, LinearExpr):
3071 raise TypeError("Cannot interpret %s as a linear expression." % expression)
3072
3073 value = 0
3074 to_process = [(expression, 1)]
3075 while to_process:
3076 expr, coeff = to_process.pop()
3077 if isinstance(expr, IntegralTypes):
3078 value += int(expr) * coeff
3079 elif isinstance(expr, _ProductCst):
3080 to_process.append((expr.expression(), coeff * expr.coefficient()))
3081 elif isinstance(expr, _Sum):
3082 to_process.append((expr.left(), coeff))
3083 to_process.append((expr.right(), coeff))
3084 elif isinstance(expr, _SumArray):
3085 for e in expr.expressions():
3086 to_process.append((e, coeff))
3087 value += expr.constant() * coeff
3088 elif isinstance(expr, _WeightedSum):
3089 for e, c in zip(expr.expressions(), expr.coefficients()):
3090 to_process.append((e, coeff * c))
3091 value += expr.constant() * coeff
3092 elif isinstance(expr, IntVar):
3093 value += coeff * solution.solution[expr.index]
3094 elif isinstance(expr, _NotBooleanVariable):
3095 value += coeff * (1 - solution.solution[expr.negated().index])
3096 else:
3097 raise TypeError(f"Cannot interpret {expr} as a linear expression.")
3098

3462 def value(self, expression: LinearExprT) -> int:
3463 """Evaluates an linear expression in the current solution.
3464
3465 Args:
3466 expression: a linear expression of the model.
3467
3468 Returns:
3469 An integer value equal to the evaluation of the linear expression
3470 against the current solution.
3471
3472 Raises:
3473 RuntimeError: if 'expression' is not a LinearExpr.
3474 """
3475 if not self.has_response():
3476 raise RuntimeError("solve() has not been called.")
3477
3478 value = 0
3479 to_process = [(expression, 1)]
3480 while to_process:
3481 expr, coeff = to_process.pop()
3482 if isinstance(expr, IntegralTypes):
3483 value += int(expr) * coeff
3484 elif isinstance(expr, _ProductCst):
3485 to_process.append((expr.expression(), coeff * expr.coefficient()))
3486 elif isinstance(expr, _Sum):
3487 to_process.append((expr.left(), coeff))
3488 to_process.append((expr.right(), coeff))
3489 elif isinstance(expr, _SumArray):
3490 for e in expr.expressions():
3491 to_process.append((e, coeff))
3492 value += expr.constant() * coeff
3493 elif isinstance(expr, _WeightedSum):
3494 for e, c in zip(expr.expressions(), expr.coefficients()):
3495 to_process.append((e, coeff * c))
3496 value += expr.constant() * coeff
3497 elif isinstance(expr, IntVar):
3498 value += coeff * self.SolutionIntegerValue(expr.index)
3499 elif isinstance(expr, _NotBooleanVariable):
3500 value += coeff * (1 - self.SolutionIntegerValue(expr.negated().index))
3501 else:
3502 raise TypeError(
3503 f"cannot interpret {expression} as a linear expression."
3504 )
