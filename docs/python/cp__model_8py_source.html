175class LinearExpr:
176 """Holds an integer linear expression.
177
178 A linear expression is built from integer constants and variables.
179 For example, `x + 2 * (y - z + 1)`.
180
181 Linear expressions are used in CP-SAT models in constraints and in the
182 objective:
183
184 * You can define linear constraints as in:
185
186 ```
187 model.Add(x + 2 * y <= 5)
188 model.Add(sum(array_of_vars) == 5)
189 ```
190
191 * In CP-SAT, the objective is a linear expression:
192
193 ```
194 model.Minimize(x + 2 * y + z)
195 ```
196
197 * For large arrays, using the LinearExpr class is faster that using the python
198 `sum()` function. You can create constraints and the objective from lists of
199 linear expressions or coefficients as follows:
200
201 ```
202 model.Minimize(cp_model.LinearExpr.Sum(expressions))
203 model.Add(cp_model.LinearExpr.WeightedSum(expressions, coefficients) >= 0)
204 ```
205 """

296 all_ones = False

302 def GetIntegerVarValueMap(self) -> Tuple[Dict[VariableT, IntegralT], int]:
303 """Scans the expression, and returns (var_coef_map, constant)."""
304 coeffs = collections.defaultdict(int)
305 constant = 0
306 to_process: List[Tuple[LinearExprT, IntegralT]] = [(self, 1)]
307 while to_process: # Flatten to avoid recursion.
308 expr, coeff = to_process.pop()
309 if cmh.is_integral(expr):
310 constant += coeff * int(expr)
311 elif isinstance(expr, _ProductCst):
312 to_process.append((expr.Expression(), coeff * expr.Coefficient()))
313 elif isinstance(expr, _Sum):
314 to_process.append((expr.Left(), coeff))
315 to_process.append((expr.Right(), coeff))
316 elif isinstance(expr, _SumArray):
317 for e in expr.Expressions():
318 to_process.append((e, coeff))
319 constant += expr.Constant() * coeff
320 elif isinstance(expr, _WeightedSum):
321 for e, c in zip(expr.Expressions(), expr.Coefficients()):
322 to_process.append((e, coeff * c))
323 constant += expr.Constant() * coeff
324 elif isinstance(expr, IntVar):
325 coeffs[expr] += coeff
326 elif isinstance(expr, _NotBooleanVariable):
327 constant += coeff
328 coeffs[expr.Not()] -= coeff
329 else:
330 raise TypeError("Unrecognized linear expression: " + str(expr))
331
332 return coeffs, constant

334 def GetFloatVarValueMap(self) -> Tuple[Dict[VariableT, float], float, bool]:
335 """Scans the expression. Returns (var_coef_map, constant, is_integer)."""
336 coeffs = {}
337 constant = 0
338 to_process: List[Tuple[LinearExprT, Union[IntegralT, float]]] = [(self, 1)]
339 while to_process: # Flatten to avoid recursion.
340 expr, coeff = to_process.pop()
341 if cmh.is_integral(expr): # Keep integrality.
342 constant += coeff * int(expr)
343 elif cmh.is_a_number(expr):
344 constant += coeff * float(expr)
345 elif isinstance(expr, _ProductCst):
346 to_process.append((expr.Expression(), coeff * expr.Coefficient()))
347 elif isinstance(expr, _Sum):
348 to_process.append((expr.Left(), coeff))
349 to_process.append((expr.Right(), coeff))
350 elif isinstance(expr, _SumArray):
351 for e in expr.Expressions():
352 to_process.append((e, coeff))
353 constant += expr.Constant() * coeff
354 elif isinstance(expr, _WeightedSum):
355 for e, c in zip(expr.Expressions(), expr.Coefficients()):
356 to_process.append((e, coeff * c))
357 constant += expr.Constant() * coeff
358 elif isinstance(expr, IntVar):
359 if expr in coeffs:
360 coeffs[expr] += coeff
361 else:
362 coeffs[expr] = coeff
363 elif isinstance(expr, _NotBooleanVariable):
364 constant += coeff
365 if expr.Not() in coeffs:
366 coeffs[expr.Not()] -= coeff
367 else:
368 coeffs[expr.Not()] = -coeff
369 else:
370 raise TypeError("Unrecognized linear expression: " + str(expr))
371 is_integer = cmh.is_integral(constant)
372 if is_integer:
373 for coeff in coeffs.values():
374 if not cmh.is_integral(coeff):
375 is_integer = False
376 break
377 return coeffs, constant, is_integer

628class _WeightedSum(LinearExpr):
629 """Represents sum(ai * xi) + b."""
630
631 def __init__(self, expressions, coefficients, constant=0):
632 self.__expressions = []
633 self.__coefficients = []
634 self.__constant = constant
635 if len(expressions) != len(coefficients):
636 raise TypeError(
637 "In the LinearExpr.WeightedSum method, the expression array and the "
638 " coefficient array must have the same length."
639 )
640 for e, c in zip(expressions, coefficients):
641 c = cmh.assert_is_a_number(c)
642 if cmh.is_zero(c):
643 continue
644 if cmh.is_a_number(e):
645 e = cmh.assert_is_a_number(e)
646 self.__constant += e * c
647 elif isinstance(e, LinearExpr):
648 self.__expressions.append(e)
649 self.__coefficients.append(c)
650 else:
651 raise TypeError("Not an linear expression: " + str(e))
652
653 def __str__(self):
654 output = None
655 for expr, coeff in zip(self.__expressions, self.__coefficients):
656 if not output and cmh.is_one(coeff):
657 output = str(expr)
658 elif not output and cmh.is_minus_one(coeff):
659 output = "-" + str(expr)
660 elif not output:
661 output = f"{coeff} * {expr}"
662 elif cmh.is_one(coeff):
663 output += f" + {expr}"
664 elif cmh.is_minus_one(coeff):
665 output += f" - {expr}"
666 elif coeff > 1:
667 output += f" + {coeff} * {expr}"
668 elif coeff < -1:
669 output += f" - {-coeff} * {expr}"
670 if self.__constant > 0:
671 output += f" + {self.__constant}"
672 elif self.__constant < 0:
673 output += f" - {-self.__constant}"
674 if output is None:
675 output = "0"
676 return output
677
678 def __repr__(self):
679 exprs_str = ", ".join(map(repr, self.__expressions))
680 coeffs_str = ", ".join(map(repr, self.__coefficients))
681 return f"WeightedSum([{exprs_str}], [{coeffs_str}], {self.__constant})"

693class IntVar(LinearExpr):
694 """An integer variable.
695
696 An IntVar is an object that can take on any integer value within defined
697 ranges. Variables appear in constraint like:
698
699 x + y >= 5
700 AllDifferent([x, y, z])
701
702 Solving a model is equivalent to finding, for each variable, a single value
703 from the set of initial values (called the initial domain), such that the
704 model is feasible, or optimal if you provided an objective function.
705 """
706
707 def __init__(
708 self,
709 model: cp_model_pb2.CpModelProto,
710 domain: Union[int, Domain],
711 name: Optional[str],
712 ):
713 """See CpModel.NewIntVar below."""
714 self.__negation: Optional[_NotBooleanVariable] = None
715 # Python do not support multiple __init__ methods.
716 # This method is only called from the CpModel class.
717 # We hack the parameter to support the two cases:
718 # case 1:
719 # model is a CpModelProto, domain is a Domain, and name is a string.
720 # case 2:
721 # model is a CpModelProto, domain is an index (int), and name is None.
722 if cmh.is_integral(domain) and name is None:
723 self.__index: int = int(domain)
724 self.__var: cp_model_pb2.IntegerVariableProto = model.variables[domain]
725 else:
726 self.__index: int = len(model.variables)
727 self.__var: cp_model_pb2.IntegerVariableProto = model.variables.add()
728 self.__var.domain.extend(cast(Domain, domain).FlattenedIntervals())
729 self.__var.name = name

805

1505 def AddAutomaton(
1506 self,
1507 transition_variables: Sequence[VariableT],
1508 starting_state: IntegralT,
1509 final_states: Sequence[IntegralT],
1510 transition_triples: Sequence[Tuple[IntegralT, IntegralT, IntegralT]],
1511 ) -> Constraint:
1512 """Adds an automaton constraint.
1513
1514 An automaton constraint takes a list of variables (of size *n*), an initial
1515 state, a set of final states, and a set of transitions. A transition is a
1516 triplet (*tail*, *transition*, *head*), where *tail* and *head* are states,
1517 and *transition* is the label of an arc from *head* to *tail*,
1518 corresponding to the value of one variable in the list of variables.
1519
1520 This automaton will be unrolled into a flow with *n* + 1 phases. Each phase
1521 contains the possible states of the automaton. The first state contains the
1522 initial state. The last phase contains the final states.
1523
1524 Between two consecutive phases *i* and *i* + 1, the automaton creates a set
1525 of arcs. For each transition (*tail*, *transition*, *head*), it will add
1526 an arc from the state *tail* of phase *i* and the state *head* of phase
1527 *i* + 1. This arc is labeled by the value *transition* of the variables
1528 `variables[i]`. That is, this arc can only be selected if `variables[i]`
1529 is assigned the value *transition*.
1530
1531 A feasible solution of this constraint is an assignment of variables such
1532 that, starting from the initial state in phase 0, there is a path labeled by
1533 the values of the variables that ends in one of the final states in the
1534 final phase.
1535
1536 Args:
1537 transition_variables: A non-empty list of variables whose values
1538 correspond to the labels of the arcs traversed by the automaton.
1539 starting_state: The initial state of the automaton.
1540 final_states: A non-empty list of admissible final states.
1541 transition_triples: A list of transitions for the automaton, in the
1542 following format (current_state, variable_value, next_state).
1543
1544 Returns:
1545 An instance of the `Constraint` class.
1546
1547 Raises:
1548 ValueError: if `transition_variables`, `final_states`, or
1549 `transition_triples` are empty.
1550 """
1551
1552 if not transition_variables:
1553 raise ValueError(
1554 "AddAutomaton expects a non-empty transition_variables array"
1555 )
1556 if not final_states:
1557 raise ValueError("AddAutomaton expects some final states")
1558
1559 if not transition_triples:
1560 raise ValueError("AddAutomaton expects some transition triples")
1561
1562 ct = Constraint(self)
1563 model_ct = self.__model.constraints[ct.Index()]
1564 model_ct.automaton.vars.extend(
1565 [self.GetOrMakeIndex(x) for x in transition_variables]
1566 )
1567 starting_state = cmh.assert_is_int64(starting_state)
1568 model_ct.automaton.starting_state = starting_state
1569 for v in final_states:
1570 v = cmh.assert_is_int64(v)
1571 model_ct.automaton.final_states.append(v)
1572 for t in transition_triples:
1573 if len(t) != 3:
1574 raise TypeError("Tuple " + str(t) + " has the wrong arity (!= 3)")
1575 tail = cmh.assert_is_int64(t[0])
1576 label = cmh.assert_is_int64(t[1])
1577 head = cmh.assert_is_int64(t[2])
1578 model_ct.automaton.transition_tail.append(tail)
1579 model_ct.automaton.transition_label.append(label)
1580 model_ct.automaton.transition_head.append(head)
1581 return ct

2817 def Value(self, expression: LinearExprT) -> int:
2818 """Evaluates an linear expression in the current solution.
2819
2820 Args:
2821 expression: a linear expression of the model.
2822
2823 Returns:
2824 An integer value equal to the evaluation of the linear expression
2825 against the current solution.
2826
2827 Raises:
2828 RuntimeError: if 'expression' is not a LinearExpr.
2829 """
2830 if not self.HasResponse():
2831 raise RuntimeError("Solve() has not been called.")
2832
2833 value = 0
2834 to_process = [(expression, 1)]
2835 while to_process:
2836 expr, coeff = to_process.pop()
2837 if cmh.is_integral(expr):
2838 value += int(expr) * coeff
2839 elif isinstance(expr, _ProductCst):
2840 to_process.append((expr.Expression(), coeff * expr.Coefficient()))
2841 elif isinstance(expr, _Sum):
2842 to_process.append((expr.Left(), coeff))
2843 to_process.append((expr.Right(), coeff))
2844 elif isinstance(expr, _SumArray):
2845 for e in expr.Expressions():
2846 to_process.append((e, coeff))
2847 value += expr.Constant() * coeff
2848 elif isinstance(expr, _WeightedSum):
2849 for e, c in zip(expr.Expressions(), expr.Coefficients()):
2850 to_process.append((e, coeff * c))
2851 value += expr.Constant() * coeff
2852 elif isinstance(expr, IntVar):
2853 value += coeff * self.SolutionIntegerValue(expr.Index())
2854 elif isinstance(expr, _NotBooleanVariable):
2855 value += coeff * (1 - self.SolutionIntegerValue(expr.Not().Index()))
2856 else:

2859 )
